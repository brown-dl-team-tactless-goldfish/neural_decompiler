int binary_search(int arr[], int size, int num) { 
    int left = 0;
    int right = size - 1; 

    while (left <= right) { 
        int mid = left + (right - left) / 2; 
        
        if (arr[mid] == num) { 
            return mid;
        } 
        else if (arr[mid] < num) {
            left = mid + 1; 
        }
        else {
            right = mid - 1; 
        } 
    }

    return -1; 
}
