#include<stdlib.h>
#include<stdio.h>
#include<string.h>
#include<stdbool.h>
#include<stdint.h>
#include<math.h>
void reverse_array(int arr[], int size) { 
int left = 0;
int right = size - 1; 
while (left < right) { 
int temp = arr[left]; 
arr[left] = arr[right]; 
arr[right] = temp; 
left++; 
right--; 
} 
}
