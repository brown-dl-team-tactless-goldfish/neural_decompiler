#include<stdlib.h>
#include<stdio.h>
#include<string.h>
#include<stdbool.h>
#include<stdint.h>
#include<math.h>
int main()
{
int n = 5, i, j, num = 1, gap;
gap = n - 1;
for (j = 1; j <= n; j++) {
num = j;
for (i = 1; i <= gap; i++)
gap--;
for (i = 1; i <= j; i++) {
num++;
}
num--;
num--;
for (i = 1; i < j; i++) {
num--;
}
}
return 0;
}
