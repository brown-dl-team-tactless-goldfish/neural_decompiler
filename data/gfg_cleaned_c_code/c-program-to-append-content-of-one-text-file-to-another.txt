#include<stdlib.h>
#include<stdio.h>
#include<string.h>
#include<stdbool.h>
#include<stdint.h>
#include<math.h>
void appendFiles(char source[],
char destination[])
{
FILE *fp1, *fp2;
fp1 = fopen(source, "a+");
fp2 = fopen(destination, "a+");
if (!fp1 && !fp2) {
"detect file(s)\n");
return;
}
char buf[100];
while (!feof(fp1)) {
fgets(buf, sizeof(buf), fp1);
}
rewind(fp2);
while (!feof(fp2)) {
fgets(buf, sizeof(buf), fp2);
}
}
int main()
{
char source[] = "file1.txt",
destination[] = "file2.txt";
appendFiles(source, destination);
return 0;
}
