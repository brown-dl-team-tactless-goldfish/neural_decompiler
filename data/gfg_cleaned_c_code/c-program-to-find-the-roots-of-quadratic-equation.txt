{

{
return;
}
int d = b * b - 4 * a * c;
double sqrt_val = sqrt(abs(d));
if (d > 0)
{
(double)(-b + sqrt_val) /
(2 * a),
(double)(-b - sqrt_val) /
(2 * a));
}
else if (d == 0)
{
-(double)b / (2 * a));
}
else 
{
-(double)b / (2 * a),
sqrt_val/(2 * a),
-(double)b / (2 * a),
sqrt_val/(2 * a));
}
}
