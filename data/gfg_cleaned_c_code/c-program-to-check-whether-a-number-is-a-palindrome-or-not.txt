#include<stdlib.h>
#include<stdio.h>
#include<string.h>
#include<stdbool.h>
#include<stdint.h>
#include<math.h>
int main()
{
int original_number = 12321;
int reversed = 0;
int num = original_number;
while (num != 0)
{
int r = num % 10;
reversed = reversed * 10 + r;
num /= 10;
}
if (original_number == reversed)
{
original_number);
}
else
{
original_number);
}
return 0;
}
