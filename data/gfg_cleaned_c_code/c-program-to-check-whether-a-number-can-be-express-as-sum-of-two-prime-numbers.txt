int isPrime(int n)
{
int i, isPrime = 1;
if (n == 0 || n == 1)
{
isPrime = 0;
}
else
{
for (i = 2; i <= n / 2; ++i)
{
if (n % i == 0)
{
isPrime = 0;
break;
}
}
}
return isPrime;
}
