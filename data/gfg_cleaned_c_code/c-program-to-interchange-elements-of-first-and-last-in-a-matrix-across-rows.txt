#include<stdlib.h>
#include<stdio.h>
#include<string.h>
#include<stdbool.h>
#include<stdint.h>
#include<math.h>
#define n 3
void interchangeFirstLast(int m[][n])
{
int rows = n;
for (int i = 0; i < n; i++)
{
int t = m[0][i];
m[0][i] = m[rows - 1][i];
m[rows - 1][i] = t;
}
}
int main()
{
int m[n][n] = {{6, 3, 1},
{4, 5, 2},
{2, 4, 9}};
for (int i = 0; i < n; i++)
{
for (int j = 0; j < n; j++)
{
}
}
interchangeFirstLast(m);
for (int i = 0; i < n; i++)
{
for (int j = 0; j < n; j++)
{
}
}
}
