struct Student {
char* name;

};
int comparator(const void* p, const void* q)
{
return strcmp(((struct Student*)p)->name,
((struct Student*)q)->name);
}

for (i = 0; i < n; i++) {
arr[i].id, arr[i].name, arr[i].age);

arr[i].id, arr[i].name, arr[i].age);

}
