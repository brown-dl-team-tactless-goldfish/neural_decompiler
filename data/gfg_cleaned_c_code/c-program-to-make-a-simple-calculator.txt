#include<stdlib.h>
#include<stdio.h>
#include<string.h>
#include<stdbool.h>
#include<stdint.h>
#include<math.h>
int main()
{
char ch;
double a, b;
while (1) {
scanf(" %c", &ch);
if (ch == 'x')
exit(0);
scanf("%lf %lf",&a,&b);
switch (ch) {
case '+':
break;
case '-':
break;
case '*':
break;
case '/':
break;
default:
}
}
}
