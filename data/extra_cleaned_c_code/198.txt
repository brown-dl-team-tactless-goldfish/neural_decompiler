#include<stdlib.h>
#include<stdio.h>
#include<string.h>
#include<stdbool.h>
#include<stdint.h>
#include<math.h>
int rob(int num[], int n) {
if (n == 0) return 0;
int *d = (int *)calloc(n + 2, sizeof(int));
int i;
d[0] = d[1] = 0; 
for (i = 2; i < n + 2; i++) {
if (d[i - 2] + num[i - 2] > d[i - 1]) { 
d[i] = d[i - 2] + num[i - 2];
}
else { 
d[i] = d[i - 1];
}
}
return d[n + 1];
}

int num[] = { 1, 2, 3, 4, 5 };
return 0;
}
