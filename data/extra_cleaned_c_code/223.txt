#define max(a,b)(((a) > (b)) ? (a) : (b))
#define min(a,b)(((a) < (b)) ? (a) : (b))
int computeArea(int A, int B, int C, int D, int E, int F, int G, int H) {
int s1 = (C - A) * (D - B);
int s2 = (G - E) * (H - F);
int s0 = 0;
int x1 = max(A, E);
int y1 = max(B, F);
int x2 = min(C, G);
int y2 = min(D, H);
if ((x2 - x1 > 0) && (y2 - y1) > 0)
{
s0 = (x2 - x1) * (y2 - y1);
}
return s1 + s2 - s0;
}
