#include<stdlib.h>
#include<stdio.h>
#include<string.h>
#include<stdbool.h>
#include<stdint.h>
#include<math.h>
int countPrimes(int n) {
bool *flag = (bool *)calloc(n + 1, sizeof(bool));
int *prime = (int *)calloc(n / 2, sizeof(int));
flag[0] = 1; 
flag[1] = 1; 
int i, j;
int count = 0;
for (i = 2; i < n; i++) {
if (flag[i] == 0) { 
prime[count ++] = i;
}
for (j = 0; j < count && i * prime[j] < n; j++) {
flag[i * prime[j]] = 1;
if (i % prime[j] == 0) { 
break;
}
}
}
return count;
}
