struct ListNode {

struct ListNode *next;
};
struct ListNode* removeElements(struct ListNode* head, int val) {
struct ListNode *ret;
struct ListNode *prev, *p;
ret = p = head;
prev = head;
while (p) {
if (p->val == val) {
if (p == ret) { 
ret = p->next;
prev = ret; 
}
else { 
prev->next = p->next;
}
}
else {
prev = p;
}
p = p->next;
}
return ret;
}
