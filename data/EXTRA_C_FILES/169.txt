int majorityElement(int num[], int n) {
int major;
int i;
major = num[0];
int count = 1;
for (i = 1; i < n; i++) {
if (num[i] == major) count ++;
else count --;
if (count == 0) {
major = num[i];
count = 1;
}
}
return major;
}
