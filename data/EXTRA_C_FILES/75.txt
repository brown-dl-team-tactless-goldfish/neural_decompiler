void swap(int *a, int *b) {
int c = *a;
*a = *b;
*b = c;
}
void sortColors(int* nums, int numsSize) {
int i, zero, two;
i = 0;
zero = 0;
two = numsSize - 1;
while (i <= two) {
if (nums[i] == 0) {
swap(&nums[i], &nums[zero++]);
i++; 
}
else if (nums[i] == 2) {
swap(&nums[i], &nums[two--]);
}
else {
i++;
}
}
}
