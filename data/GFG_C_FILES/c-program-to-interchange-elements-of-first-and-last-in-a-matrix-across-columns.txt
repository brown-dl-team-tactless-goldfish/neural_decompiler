#include<stdlib.h>
#include<stdio.h>
#include<string.h>
#include<stdbool.h>
#include<stdint.h>
#include<math.h>
#define n 3 
void interchangeFirstLast(int mat[][n])
{
for (int i = 0; i < n; i++) {
int t = mat[i][0];
mat[i][0] = mat[i][n - 1];
mat[i][n - 1] = t;
}
}
int main()
{
int mat[n][n]
= { { 2, 4, 6 }, { 8, 2, 3 }, { 1, 9, 4 } };
for (int i = 0; i < n; i++) {
for (int j = 0; j < n; j++) {
}
}
interchangeFirstLast(mat);
for (int i = 0; i < n; i++) {
for (int j = 0; j < n; j++) {
}
}
}
