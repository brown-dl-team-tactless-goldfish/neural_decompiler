int numerical(char *s) {
    int value = 0;
    
    while (*s) {
        value = value * 10 + (*s) - 'a';
        s++;
    }

    return value;
}

bool isSumEqual(char * firstWord, char * secondWord, char * targetWord) {
    int val1, val2, val3;
    val1 = val2 = val3 = 0;
    
    val1 = numerical(firstWord);
    val2 = numerical(secondWord);
    val3 = numerical(targetWord);
    
    if (val3 == (val1 + val2)) {
        return true;
    }
    
    return false;
}
