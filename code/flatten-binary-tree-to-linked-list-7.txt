void preorder(struct TreeNode * root, struct TreeNode ** new);
    
void flatten(struct TreeNode* root){
    if (root == NULL)
        return;
    struct TreeNode * new = malloc(sizeof(struct TreeNode));
    preorder(root, &new);
}

void preorder(struct TreeNode * root, struct TreeNode ** new){
    if (root == NULL)
        return;
    struct TreeNode * left = root->left;
    struct TreeNode * right = root->right;
    (*new)->right = root;
    (*new)->left = NULL;
    *new = (*new)->right;
    preorder(left, new);
    preorder(right, new);
    return;
}
