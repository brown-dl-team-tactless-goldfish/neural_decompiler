int longestPalindrome(char* s) {
    int table[128] = {0};
    int size = strlen(s);
    int sum = 0;
    int odd = 0;
    for (int i = 0; i < size; i++)
    {
        table[s[i]]++;
    }
    
    for (int i = 0; i < 128; i++)
    {
        if((table[i] % 2) == 0)
        {
            sum += table[i];
        }
        else
        {
            sum += table[i]-1;
            odd = 1;
        }
    }
    return sum+odd;
}
