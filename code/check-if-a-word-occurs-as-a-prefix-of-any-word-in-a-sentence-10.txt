int getword(char *s, char *w){
    for(;*w;++w,s++){
        if(*w!=*s)return 0;
    }
    return 1;
}

int isPrefixOfWord(char * sentence, char * searchWord){
    int count=1;
    
    if(getword(sentence,searchWord))return count;
    else{
        while((sentence = strchr(sentence,' '))){
            count++;
            if(getword(++sentence,searchWord))return count;
        }
    }
    return -1;
}
