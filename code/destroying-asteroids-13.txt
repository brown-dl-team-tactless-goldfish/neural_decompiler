public class Solution {
    public bool AsteroidsDestroyed(int mass, int[] asteroids) {
        Array.Sort(asteroids);
        long val = mass;
        for(int i=0; i<asteroids.Length; i++){
            if(asteroids[i] <= val){
                val += asteroids[i];
            }else{
                return false;
            }
        }
        return true;
    }
}
