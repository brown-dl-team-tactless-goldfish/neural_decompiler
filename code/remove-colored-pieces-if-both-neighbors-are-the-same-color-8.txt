int countA = 0;
        int countB = 0;
        int totalCountA = 0;
        int totalCountB = 0;        
        for (int i = 0; i < colors.Length; i++) {
            
            char c = colors[i];
            if (c == 'A') {
                if (++countA >= 3) totalCountA++;
                countB = 0;
            } else {
                if (++countB >= 3) totalCountB++;
                countA = 0;
            }
        }
        
        return totalCountA > totalCountB;
