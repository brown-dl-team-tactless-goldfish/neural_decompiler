int cmp(const void *x, const void *y)
{
    return *(int*)x - *(int*)y;
}
int distinctAverages(int* nums, int numsSize)
{
    int sum = 1;
    qsort(nums, numsSize, sizeof(int), cmp);
    for(int i = 0; i < numsSize/2; i++)
    {
        nums[i] = nums[i] + nums[(numsSize - 1) - i];
    }
    qsort(nums, numsSize/2, sizeof(int), cmp);
    for(int i = 1; i < numsSize/2; i++)
    {
        if(nums[i-1] != nums[i])
        {
            sum = 1 + sum;
        }
    }
    return sum;
}
