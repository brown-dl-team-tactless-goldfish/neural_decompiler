
class Solution {
public:
    
    int countStudents(vector<int>& students, vector<int>& sandwiches) 
    {
        int initialCount = students.size();
        int count = 0;
        
        while(students.size() > 0 )
        {
            if(students[0] == sandwiches[0])
            {
                count++;
                students.erase (begin(students),begin(students)+1);
                sandwiches.erase (begin(sandwiches),begin(sandwiches)+1);
            }
            else
            {
                auto it = find(students.begin(),students.end(),sandwiches[0]);
                if(it == end(students))
                    break;
                rotate(students.begin(),it,students.end());
            }
        }
        
        return (initialCount - count);
    }
};
