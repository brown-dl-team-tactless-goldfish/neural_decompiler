constexpr char pattern[] = "abc";

class Solution {
public:
    int addMinimum(string word) {
        int res = 0, j = 0;
        for (int i = 0; i < word.size(); ) {
            if (word[i] != pattern[j]) {
                ++res;       
            } else {
                ++i;
            }
            j = (j + 1) % 3;
        }
        return res + (j == 0 ? 0 : (3 - j));
    }
};
