typedef long long ll;
class Solution {
public:
    ll M = 1e9+7;
    int firstDayBeenInAllRooms(vector<int>& nextVisit) {
        int n = nextVisit.size();
        ll days[100001];
        days[0] = 2;
        for(int i=1; i<n-1; i++){
            ll visit = nextVisit[i];
            days[i] = days[i-1] + 2;
            if(visit != i) days[i] += (M + days[i-1] - (visit > 0?days[visit-1]:0))%M;
            days[i] %= M;
        }
        return days[n-2];
    }
};
