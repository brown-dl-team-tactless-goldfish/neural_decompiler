class Solution {
public:
    vector<int> div(int n){
        vector<int> out;
        int i=sqrt(n);
         while(i>=1){
             if(n%i==0)
                 return {n/i,i};
             i--;
         }
        return {1,n};
    }
    
    vector<int> closestDivisors(int num) {
        vector<int> p1= div(num+1);
        vector<int> p2= div(num+2);
        return (abs(p1[0]-p1[1])<abs(p2[0]-p2[1]))?p1:p2;
    }
};
