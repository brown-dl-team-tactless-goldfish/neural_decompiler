class Solution {
public:
    vector<int> sortJumbled(vector<int>& mapping, vector<int>& nums) {
        map<int, vector<int>> m;
        for (auto& n : nums){
            string s = to_string(n);
            for (auto& c : s)
                c = mapping[c - '0'] + '0';
            int f = stoi(s);
            m[f].push_back(n);
        }
        vector<int> ans;
        for (auto& [f,s] : m)
            for (auto& n : s)
                ans.push_back(n);
        return ans;
    }
};
