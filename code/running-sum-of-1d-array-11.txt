/**
 * Note: The returned array must be malloced, assume caller calls free().
 */
int* runningSum(int* nums, int numsSize, int* returnSize){
int i;
int* sum = (int*)malloc(numsSize*sizeof(int));
sum[i]=0;
sum[i] = sum[i]+nums[i];
for(i=1; i<numsSize; i++){
    sum[i] = sum[i-1]+nums[i];
}
*returnSize=numsSize;
return sum;

}
