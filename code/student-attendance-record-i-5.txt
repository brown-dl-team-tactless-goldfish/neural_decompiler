bool checkRecord(char * s){
int i=0;
int j=0;
int countA=0;//counts the number of A's
int countL=0;//counts the number of **consecutive** L's
while(i<strlen(s))
{
    if(s[i]=='A')
    {

        countA++;
        if(countA>=2)
            return false;
    }
        
    i++;
}


while(j<strlen(s))
{
    if(s[j]=='L')
    {
        countL++;//counting for consecutive L's
        if(countL==3)
            return false;
        j++;
    }
    else
    {
        countL=0;    //reinitializing the countL, this is done to make sure that the L's that we are counting are consecutive.
        j++;
    }
}
return true;


}
