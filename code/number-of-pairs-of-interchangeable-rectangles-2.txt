class Solution {
public:
	long long interchangeableRectangles(vector<vector<int>>& rectangles) {
		unordered_map<double, long long> um;
		double t;
		for(int i = 0; i < rectangles.size(); i++){
			t = (double)rectangles[i][1]/rectangles[i][0];
			um[t]++;
		}
		long long ans = 0;
		for(auto i : um){
			ans += (i.second)*(i.second - 1)/2;
		}
		return ans;
	}
};
