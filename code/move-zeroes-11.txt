void moveZeroes(int* nums, int numsSize){
    register unsigned int i, j, p = 0, arrSize = 0;
	int arr[10000] = {0};
	for(i = 0; i < numsSize; i++) {
		if(nums[i] != 0) {
			arr[p] = nums[i];
			p++;
			arrSize++;
		}
	}
	for(j = 0; j < numsSize; j++) {
		if(j <= arrSize)
			nums[j] = arr[j];
		else
			nums[j] = 0;	
	}
}
