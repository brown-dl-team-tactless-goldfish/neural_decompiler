    public int BestClosingTime(string customers)
    {
        int n = customers.Length;
        int[] penalty = new int[n];


        int countRemainingCustomerVisit = customers.Count(c => c == 'Y');
        int countPastNotVisited = 0;

        int bestTime = 0;
        int minimumPenalty = countRemainingCustomerVisit;


        for (int i = 0; i < n; i++)
        {
            if (customers[i] == 'Y')
            {
                countRemainingCustomerVisit--;
            }
            else
            {
                countPastNotVisited++;
            }

            int currPenalty = countPastNotVisited + countRemainingCustomerVisit;
            if (currPenalty < minimumPenalty)
            {
                minimumPenalty = currPenalty;
                bestTime = i + 1;
            }
        }
        return bestTime;
    }
