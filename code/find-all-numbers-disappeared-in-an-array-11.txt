int* findDisappearedNumbers(int* nums, int numsSize, int* returnSize){
    int val, j = 0, c = 0;
    *returnSize = numsSize;
    for(int i = 0; i < numsSize; i++)
    {   //val will turn element at nums[i] into an index
        val = abs(nums[i]) - 1;        
        //if element at index val is positive,
        //turn it negative and decrement returnSize
        //negative elements will indicate the index
        //can be used as a sign that the array has that value
        if(nums[val] > 0){
            nums[val] *= -1;     
            (*returnSize)--;
        }
    }
    //create response array using returnSize
    int *res = malloc((*returnSize) * sizeof(int));
    for(int i = 0; i < numsSize; i++)
        //if element is positive,
        //add it to response array
        if(nums[i] > 0)
            res[j++] = i + 1;
    
    return res;
}
