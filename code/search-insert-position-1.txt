/**
 * Method: Use binary search to find the index n where nums[n] >= target.
 * Target can be bigger than last element in nums array. So, having the 
 * search space set to one element bigger for right index in necessary. 
 */

int searchInsert(int* nums, int numsSize, int target){
    int left = 0, right = numsSize, mid;
    
    while(left < right) {
        mid = left + (right - left) / 2;
        
        if (nums[mid] >= target) 
            right = mid;            
        else
            left = mid + 1;
    }
    return left;
}
