long long sum(int n){
    long long t = 0;
    for(int i = 1; i <=n; i++){
        t += i;
    }
    return t;
}

long long getDescentPeriods(int* prices, int pricesSize){
    long long ans = 0;
    int cn = 1;    
    for(int i = 1; i < pricesSize; i++){
        if((prices[i-1] - prices[i]) == 1)
            cn++;
        else{
            ans += sum(cn);
            cn = 1;
        }
    }
    ans += sum(cn);
    return ans;
}
