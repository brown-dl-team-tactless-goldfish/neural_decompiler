    long long getDescentPeriods(vector<int>& prices) {
        
        int n=prices.size();
        vector<int> dp(n,0);
        
        for(int i=1;i<n;i++){
            
            if(prices[i-1] - prices[i]==1){
                dp[i]+=dp[i-1] + 1;
            }
            
        }
        
        long long ans=n;
        
        
        for(auto x: dp){
            ans+=x;
        }
        
        return ans;
    }
