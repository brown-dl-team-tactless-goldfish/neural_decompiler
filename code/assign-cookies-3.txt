int sortf (const void * s, const void * g) {
   return *(int*)s - *(int*)g;
}

int findContentChildren(int* g, int gSize, int* s, int sSize){
    int n = 0, i = 0, j = 0;
    
    qsort(g, gSize, sizeof(int), sortf);
    qsort(s, sSize, sizeof(int), sortf);
    
    while(i < sSize && j < gSize)
    {
        if (s[i] >= g[j]) 
        {
            n++; i++; j++;
        }
        else i++;
    }
    
    return n;
}
