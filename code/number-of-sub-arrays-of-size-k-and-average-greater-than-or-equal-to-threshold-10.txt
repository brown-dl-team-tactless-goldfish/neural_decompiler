class Solution {
public:
    int numOfSubarrays(vector<int>& a, int k, int threshold) {
        int n = a.size(), c = 0;
        int sum = 0;
        for(int i=0;i<k;i++) sum += a[i];
        if(sum>=(k*threshold)) c++;
        for(int i=k;i<n;i++)
        {
            sum = sum - a[i-k] + a[i];
            if(sum>=(k*threshold)) c++;
        }
        return c;
    }
};
