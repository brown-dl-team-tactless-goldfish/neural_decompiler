class Solution {
public:
    int numOfSubarrays(vector<int>& arr) {
        int e=0,o=0,s=0,res=0,mod=1e9+7;
        for(auto j:arr)
        {
            s=s+j;
            if(s%2==0)
            {res=(res+o)%mod;
             e++;
            }
            else
            {
                res++;
                res=(res+e)%mod;
                o++;
            }
        }
        return(res);
    }
};
