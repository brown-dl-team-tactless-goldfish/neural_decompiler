    int longestSubsequence(vector<int>& arr, int d) {
        map<int,int>mp;
        int i,m=0;
        for(i=0;i<arr.size();i++)
        {
            if(mp.find(arr[i]-d)==mp.end())
            {
                mp[arr[i]]=1;
            }
            else
            {
                mp[arr[i]]=mp[arr[i]-d]+1;
            }
            m=max(m,mp[arr[i]]);
        }
        return m;
    }
};
