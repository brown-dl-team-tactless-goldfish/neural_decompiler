int numSpecial(int** a, int n, int* m)
{
    int res=0,row[100][2]={0},col[100]={0};
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<m[0];j++)
        {
            if(a[i][j])
            {
                row[i][0]++;
                col[j]++;
                row[i][1]=j;
            }
        }
    }
    for(int i=0;i<n;i++)
        if(row[i][0]==1&&col[row[i][1]]==1)res++;
    return res;
}
