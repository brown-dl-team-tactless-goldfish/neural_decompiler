int cmp(const void* a, const void* b)
{
	return *(int *)a - *(int *)b;
}


int findContentChildren(int* g, int gSize, int* s, int sSize){
	if(!sSize) return 0;
	int ans = 0;
	qsort(s, sSize, sizeof(int), cmp);
	for (int i = 0; i < gSize; i++)
	{
		for(int j = 0; j < sSize; j++){
			if(s[j] >= g[i]){
				ans++;
				s[j] = -1;
				break;
			}
		}
	}

	return ans;
}
