int numPairsDivisibleBy60(int* time, int timeSize) {
    int* map=(int*)calloc(60,sizeof(int));
    for(int i=0;i<timeSize;i++){
        map[time[i]%60]++;
    }
    int ret=0;
    ret+=(map[0]-1)*map[0]/2;
    ret+=(map[30]-1)*map[30]/2;
    for(int i=1;i<30;i++){
        ret+=map[i]*map[60-i];   
    }
    return ret;
}
