    priority_queue<int> pq;
    
    for(auto &x: nums)
    {
        x = x %2 == 1?x*2 :x;
        
        pq.push(x);
        
        mini = min(mini,x);
    }
    
    while(pq.top() % 2 == 0)
    {
        res = min(res,pq.top() - mini);
        
        mini = min(mini,pq.top()/2);
        
        pq.push(pq.top()/2);
        
        pq.pop();
    }
    
    return min(res,pq.top()-mini);
}
