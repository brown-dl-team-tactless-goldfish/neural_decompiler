public class Solution
{
    Dictionary<int, int> memo = new Dictionary<int, int>();
    public int MinDays(int n)
    {
        if(n < 3) return n;
        if(memo.ContainsKey(n)) return memo[n];
        return memo[n] = Math.Min(MinDays(n/2) + MinDays(n%2), MinDays(n/3) + MinDays(n%3)) + 1;
    }
}
