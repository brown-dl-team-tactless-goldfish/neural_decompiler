int* getRow(int rowIndex, int* returnSize){
    *returnSize = rowIndex + 1;
    
    int* ans = malloc(*returnSize * sizeof(int));

    ans[0]        = 1;
    ans[rowIndex] = 1;
    
    for (int i = 1; i < rowIndex/2 + 1; ++i) {
        ans[i] = (long) ans[i - 1] * (rowIndex - i + 1) / i;       
        ans[rowIndex - i] = ans[i];
    }
    
    return ans;
}
