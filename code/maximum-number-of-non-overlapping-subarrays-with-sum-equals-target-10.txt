class Solution {
public:
    int maxNonOverlapping(vector<int>& nums, int target) {
        unordered_map<int, int> m;
        m[0]++;
        int s = 0, ans = 0;
        for(int i : nums) {
            s += i;
            if(m.find(s - target) != m.end()) ans++, m.clear();
            m[s]++;
        }
        return ans;
    }
};
