string reformatNumber(string number) {

	int num = 0, cnt = 0;
	for (char& c : number) num += (int)((c <= '9') && (c >= '0'));

	string res = "";

	for (char& c : number) {
		if (c < '0' || c > '9') continue;
		++cnt;
		res.push_back(c);

		if (num - cnt == 2 && num % 3 != 0) res.push_back('-');
		else if (num - cnt >= 3 && cnt % 3 == 0) res.push_back('-');
	}

	return res;
}
