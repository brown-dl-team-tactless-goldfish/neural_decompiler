bool canConstruct(char * ransomNote, char * magazine){
    int nRansomNoteStrLen = strlen(ransomNote);
    int nMagazineStrLen = strlen(magazine);

    if(nRansomNoteStrLen > nMagazineStrLen) return false;

    int nMagazineStart = 0;
    for(int i=0; i<nRansomNoteStrLen; i++)
    {
        bool bFlag = false;
        
        for(int j=nMagazineStart; j<nMagazineStrLen; j++)
        {
            if(ransomNote[i] == magazine[j])
            {
                // swap the start index to current position
                if(j > nMagazineStart) magazine[j] = magazine[nMagazineStart];

                nMagazineStart++;
                bFlag = true;
                break;
            }
        }

        if(bFlag == false) return false;
    }
    
    return true;
}
