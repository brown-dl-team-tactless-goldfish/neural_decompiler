    int longestSubsequence(vector<int>& arr, int difference) {
        unordered_map<int, int> m; //last element, length
        int res = 0;
        for(auto x:arr){
            int pre = x - difference;
            m[x] = m[pre] + 1;
            res = max(res, m[x]);
        }
        return res;
    }
