public class Solution {
    public IList<int> LexicalOrder(int n) {
        var result = new List<int>();
        for (int i = 1; i <= Math.Min(n, 9); i++) {
            result.Add(i);
            DFS(n, i * 10, result);
        }
        return result;
    }

    private void DFS(int n, int baseNumber, IList<int> result) {
        if (baseNumber > n) return;

        for (int i = 0; i <= 9; i++) {
            var oneResult = baseNumber + i;
            if (oneResult > n) return;
            result.Add(oneResult);

            DFS(n, oneResult * 10, result);
        }
    }
}
