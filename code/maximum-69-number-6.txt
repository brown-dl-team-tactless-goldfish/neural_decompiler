int maximum69Number (int num){
    
    int * ptr = malloc(5 * sizeof(int));
    
    int i = 0;
    while (num){
        ptr[i] = num % 10;
        num /= 10;
        i++;
    }
    
    int n = i;
    for (int j=n; j>-1; j--){
        if (ptr[j] == 6){
            ptr[j] = 9;
            break;
        }
    }
    
    int ans = 0;
    for (int i=0; i<n; i++){
        ans += ptr[i]*pow(10, i);
    }
        
    return ans;
}
