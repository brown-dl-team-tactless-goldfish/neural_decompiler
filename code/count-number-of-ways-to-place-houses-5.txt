class Solution {
public:
    int countHousePlacements(int n) {
         int mod=1000000007;
         vector<long long> v(n+1,0);
         v[0]=1;
         v[1]=2;
         for(int i=2;i<=n;i++)
         {
             v[i]=(v[i-1]+v[i-2])%mod;
         }
        long long res=v[n];
        return (res*res)%mod;
    }
};
