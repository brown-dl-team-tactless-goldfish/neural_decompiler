char * longestCommonPrefix(char ** strs, int strsSize)
{ 
    if (strs[0][0] == '\0')
    {
        return strs[0];
    }

    for (size_t c = 0; c < strlen(strs[0]); c++)
    {   
        for(int k = 0; k < strsSize; k++)
        {
            if(strs[0][c] != strs[k][c])
            {
                    strs[0][c] = '\0';
                    return strs[0];
            } 
        }
    }        
    
    return strs[0];          
}
