int countBinarySubstrings(char * s){
    int * temp = (int *)malloc(strlen(s)*sizeof(int));
    temp[0] = 1;
    int idx = 0, sum = 0;
    for(int i=1; i<strlen(s); i++)
        if(s[i] == s[i-1])          
            temp[idx]++;
        else temp[++idx] = 1;
    for(int i=0; i<idx; i++)
        sum += temp[i] < temp[i+1] ? temp[i] : temp[i+1];
    free(temp);
    return sum;
}
