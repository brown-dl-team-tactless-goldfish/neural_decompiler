    vector<int> getAverages(vector<int>& nums, int k) {
        if(k==0) return nums;
        int n=nums.size();
        vector<int> average(n, -1);
        if(k>(n/2-(n%2==0)))
            return vector<int>(n, -1);
        long long sum=accumulate(nums.begin(), nums.begin()+((2*k)+1), (long long)0);
        for(int i=k; i+k<n; i++){
            average[i]=sum/((k*2)+1);
            sum-=nums[i-k];
            if(i+k+1<n) sum+=nums[i+k+1];
        }
        return average;
    }
