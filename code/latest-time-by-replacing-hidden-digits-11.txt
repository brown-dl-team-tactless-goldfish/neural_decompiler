class Solution {
public:
    string maximumTime(string time) {
        string hour = time.substr(0, 2);
        string minute = time.substr(3, 2);
        if (hour[0] == '?' && hour[1] == '?') hour = "23";
        else if (hour[0] == '?') hour[0] = hour[1] > '3' ? '1' : '2';
        else if (hour[1] == '?') hour[1] = hour[0] == '2' ? '3' : '9';
        if (minute[0] == '?') minute[0] = '5';
        if (minute[1] == '?') minute[1] = '9';
        return hour + ":" + minute;
    }
};
