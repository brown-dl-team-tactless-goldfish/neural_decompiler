class Solution {
public:
    bool isCovered(vector<vector<int>>& ranges, int left, int right) {
        std::vector<int> acc(52, 0);
        for(const auto &r: ranges) ++acc[r[0]], --acc[r[1]+1];
        
        for(int i{0}, c{0}; i <= right; ++i) {
            c += acc[i];
            if(i >= left && i <= right && !c) return false;
        }
        
        return true;
    }
};
