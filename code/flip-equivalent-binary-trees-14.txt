    bool flipEquiv(TreeNode* p1, TreeNode* p2) {
        if(!p1 && !p2) return true;
        if(!p1 && p2 || p1 && !p2 || p1->val != p2->val) return false;      
        return flipEquiv(p1->left, p2->left) && flipEquiv(p1->right, p2->right) || flipEquiv(p1->right, p2->left) && flipEquiv(p1->left, p2->right);
    }
