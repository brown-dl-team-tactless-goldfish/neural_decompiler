int cmp(const void*a, const void* b){
    return *(int*)a - *(int*)b ;
}
int minimizeSum(int* nums, int numsSize){
    int n = numsSize ;
    if(n < 4)
        return 0 ;
    qsort(nums, n, sizeof(int), cmp) ;
    int a = nums[n-3] - nums[0] ;
    int b = nums[n-1] - nums[2] ;
    int c = nums[n-2] - nums[1] ;

    if(a <= b && a <= c)
        return a ;
    
    else if(b <= c && b <= c)
        return b ;
    
    else 
        return c ;
}
