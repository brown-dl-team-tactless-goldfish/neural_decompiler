class Solution {
public:
    bool canReach(string s, int minJump, int maxJump) 
    {
        int n=s.size(),prev=0;
        queue<int>q;
        q.push(0);
        while(!q.empty())
        {
            int i=q.front();q.pop();
            if(i==n-1)
                return true;
            for(int k=max(i+minJump,prev+1);k<=min(i+maxJump,n-1);k++)
                if(s[k]=='0')
                    q.push(k);
            prev=max(prev,i+maxJump);
        }
        return false;
    }
};
