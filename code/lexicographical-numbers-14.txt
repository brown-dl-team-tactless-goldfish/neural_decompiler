class Solution {
public:
    vector<int> lexicalOrder(int n) {
        vector<int> res;
        if(n<1) return res;
        res.reserve(n);
        build(res, n, 1);
        return res;
    }
    void build(vector<int>& res, int n, int i){
        for(int k=0; k<10; ++k){
            int v=i+k;
            if(v>n || res.size()>=n) break;
            res.push_back(v);
            build(res, n, v*10);
        }
    }
};
