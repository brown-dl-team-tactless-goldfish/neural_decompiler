class Solution {
public:
    int maxDivScore(vector<int>& nums, vector<int>& divisors) {
        int cnt=0,ans=INT_MAX;
        for(int i=0;i<divisors.size();i++){
            int c=0;
            for(int j=0;j<nums.size();j++){
                if(nums[j]%divisors[i] == 0) c+=1;
            }
            if(c>cnt){
                cnt = c;
                ans = divisors[i];
            }
            if(c==cnt && divisors[i]<ans) ans = divisors[i];
        }
        return ans;
    }
};
