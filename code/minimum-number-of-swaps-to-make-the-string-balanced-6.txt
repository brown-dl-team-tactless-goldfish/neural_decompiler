class Solution {
public:
    int minSwaps(string s) {
        stack<char> stk;
        
        for (auto c : s) {
            if (c == ']' && !stk.empty() && stk.top() == '[') {
                stk.pop();
            } else {
                stk.push(c);
            }
        }
        
        return (stk.size() / 2 + 1) / 2;
    }
};
