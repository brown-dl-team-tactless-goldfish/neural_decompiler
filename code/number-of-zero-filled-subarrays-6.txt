long long zeroFilledSubarray(int* a, int n)
{
    if(n==1&&a[0]==0)return 1;
    if(n==1&&a[0]!=0)return 0;
    long long k=0,s=0;
    for(int i=0;i<n;i++)
    {
        k=0;
        if(a[i]==0)
        {
            while(a[i]==0)
            {
                if(a[i++]==0)
                {
                    k++;
                    s+=k;
                }
                if(i>n-1)break;
            }
        }
    }
    return s;
}
