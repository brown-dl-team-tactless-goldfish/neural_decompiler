class Solution {
public:
    vector<int> minSubsequence(vector<int>& nums) {
        sort(begin(nums),end(nums),greater<>());
        vector<int>ans;
        int s= accumulate(begin(nums),end(nums),0),fs=0;
        for(auto&i:nums){
            ans.push_back(i);
            fs+=i;
            s-=i;
            if(fs>s) return ans;
        }
        return ans;
    }
};
