void searchTree(struct TreeNode *root, int val, bool *result) {
    if (!root) {
        return;
    } else {
        *result = (root->val == val) && *result;
        if (!*result) {
            return;
        }
        searchTree(root->left, val, result);
        searchTree(root->right, val, result);
    }
}

bool isUnivalTree(struct TreeNode *root) {
    bool result = true;
    searchTree(root, root->val, &result);
    return result;
}
