int prefixCount(char ** words, int wordsSize, char * pref)
{
  int count=0,i,j,ind=0,flag=1;
    
  for(i=0;i<wordsSize;i++)
  {
      for(flag=1,j=0,ind=0;j<strlen(words[i]);j++)
      {
          if(pref[ind]==words[i][j]&&flag)
          {
              ind++;
          }
          else if(ind<strlen(pref))
          {
              flag=0;
          }
      }
      count=flag==1&&ind==strlen(pref)?count+1:count;
  }
    
    return count; 
}
