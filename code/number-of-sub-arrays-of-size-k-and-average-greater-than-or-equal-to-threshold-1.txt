int numOfSubarrays(int* arr, int arrSize, int k, int threshold){
    int sum = 0, right = 0, left = 0, target = k * threshold, ans = 0;
    while(right < arrSize)
    {
        sum += arr[right++];
        if((right - left) == k)
        {
            if(sum >= target)
            {
                ans++;
            }
            sum -= arr[left++];
        }
    }
    return ans;
}
