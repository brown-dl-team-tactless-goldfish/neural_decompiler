class Solution:
    def countPrefixes(self, words: List[str], s: str) -> int:
        num_words = len(words)
        n = len(s)
        
        res = 0
        
        for i in range(1, n+1):
            subS = s[:i]
            print ("subS = ", subS)
            res += words.count(subS)
            
        return res
