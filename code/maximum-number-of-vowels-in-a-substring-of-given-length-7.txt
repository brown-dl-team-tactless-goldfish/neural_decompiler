class Solution:
    def maxVowels(self, s: str, k: int) -> int:
        
    
        self.res = 0
        dq = collections.deque()
        n = len(s)
        curRes = 0
        
        target = "aeiou"
        
        for i in range(n):
            while dq and len(dq) >= k:
                tmpC = dq.popleft()
                if (tmpC in target):
                    curRes -= 1
            
            curC = s[i]     
            dq.append(s[i])
            
            if (curC in target):
                curRes += 1
                
            self.res = max(self.res, curRes)

        print ("end self.res = ", self.res)
        return self.res
