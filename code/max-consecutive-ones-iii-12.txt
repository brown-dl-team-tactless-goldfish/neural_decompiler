int longestOnes(vector<int>& nums, int k) {
        int l=0, r=0, zeroes=0, ans=0; //l : iterator for left corner of window, r : iterator for right corner of the window
        while(r < nums.size()){ //start traversing the array
            if(nums[r]==0)
                zeroes++;
            while(zeroes>k){  //if zeroes are more than k we decrease thewindow from right until zeroes are less than k
                if(nums[l]==0)  //increase the left pointer one by one until we reach a zero and it decreases the # of  zeroes to less than k.
                    zeroes--;
                l++;
            }
            ans=max(ans,r-l+1); // consider the subarray length for the answer
            r++; 
        }
        return ans;
    }
