class Solution {
public:
    int maxScore(string s) {
		// Time Complexity: O(n)
		// Space Complexity: O(1)
		
        int zeros(0), ones(0);
        
        for (const char& c: s)
            if (c == '1') ones++;
        
        s.pop_back();
        int ans(0);
        
        for (const char& c: s) {
            c == '0' ? zeros++ : ones--;
            ans = max(ans, zeros + ones);
        }
        
        return ans;
    }
};
