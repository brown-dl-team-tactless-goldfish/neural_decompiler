public class Solution {
    public int MinimumDifference(int[] nums, int k) {
        
        if(nums == null || nums.Length < k)
            return 0;

        Array.Sort(nums);
        int left = 0, right = 0, min = Int32.MaxValue;
        
        while(right < nums.Length)
        {
            if(right - left + 1 == k)
            {      
                min = Math.Min(min, nums[right] - nums[left]);
                left++;
            }
            
            right++;   
        }
        
        return min;
    }
}
