class Solution {
public:
    int numOfSubarrays(vector<int>& arr) {
        int mod=(int)1e9+7;
        int ans=0;
        int sum=0;
        int even=1,odd=0;
        for(auto &it:arr){
            sum+=it;
            if(sum&1){
                ans=(ans+even)%mod;
                odd++;
            }else{
                ans=(ans+odd)%mod;
                even++;
            }
        }
        return ans;
    }
};
