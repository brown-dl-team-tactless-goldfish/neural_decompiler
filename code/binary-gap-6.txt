int binaryGap(int N) {
    int ans = 0;
    int cur = 0;        // Cur idx
    int old = INT_MAX;  // Old idx
    
    while (N) {
        if (N & 0x01) {
            int d = cur - old;          // Compute distance
            
            ans = ans > d ? ans : d;    // Update ans
            old = cur;                  // Update old
        }
        
        ++cur;
        N >>= 1;
    }
    
    return ans;
}
