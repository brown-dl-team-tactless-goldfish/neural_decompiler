int guessNumber(int n) 
    {
        // check special cases
        if(n <= 0) return 0;
        if(n == 1) return 1;
        if(guess(n) == 0) return n;

        unsigned int max = n;   // store new max range

        n /= 2;     // divide n by 2 to get middle number in range 

        // calculate the number
        while(1)
        {
            // if lower
            if(guess(n) == -1)
            {
                max = n;    // set new range max
                n /= 2;     // divide n again by 2 to get new middle number in new range
                continue;
            }

            // if higher
            if(guess(n) == 1)
            {
                n = (n + max) / 2;   // get middle number within range 
                continue;
            }
            
            // if number
            if(guess(n) == 0)
                break;
        }

        return n;
    }
