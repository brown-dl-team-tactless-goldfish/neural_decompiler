char * interpret(char * command){
    char* result = malloc(sizeof(char)*strlen(command)+1); 
    int x = 0;
    for (int i = 0; i<strlen(command);i++){
        if (command[i] == 'G'){
            result[x++]= 'G';
        } else if(command[i] == ')' && command[i-1] == '('){
            result[x++]= 'o';
        }else if (command[i] == 'a'){
            result[x++]='a';
        }else if(command[i] == 'l'){
            result[x++]='l';
        }
    }
    result[x]='\0';
    return result;
}
