int getMaximumGenerated(int n){
    if(n<2)
        return n;
    int* q = malloc((n+1)*sizeof(int));
    *q = 0; *(q+1) = 1;
    for(int i = 2; i <= n; i++)
    {
        q[i] = q[i/2] + (i % 2)*q[i/2 +1];
    }
    int m = 1;
    for(int i = 2; i <= n; i++)
        m = fmax(m, q[i]);
    return m;
}
