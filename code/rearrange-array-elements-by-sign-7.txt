int* rearrangeArray(int* nums, int n, int* returnSize)

{

    int *arr,i,neg=1,pos=0,temp;
    
   
   arr=(int*)malloc(sizeof(int)*n);
    
    
    
    for(i=0;i<n;i++)
    {
        if(nums[i]>0)
        {
            arr[pos]=nums[i];
            pos=pos+2;
        }
        else
        {
            arr[neg]=nums[i];
            neg=neg+2;
        }
    }
    
    *returnSize=n;
    return arr;
}
