 int findNumbers(int* nums, int numsSize)
 {
     
     int counter = 0 , even = 0;
     
     for ( int i = 0 ; i < numsSize ; i++)
     {
         while ( nums[i] > 0 )
         {
             nums[i] = nums[i] / 10;
             even++;
         }
         
         if ( even % 2 == 0)
         {
             counter++;
         }
         
         even = 0;
     }
     
     
     return counter;
 
 }
