bool isSubsequence(char * s, char * t){

    //i holds the current position in t
    //j holds the current position in s
    
    int i = 0;
    int j = 0;
    
    //if we have a common character, we advance the position of s
    //at the end, we know we have a subsequence if j equals the length of s
    
    
    for (; t[i] != 0; i++)
        if(s[j] == t[i])
            j++;
    
    return j == strlen(s);
}
