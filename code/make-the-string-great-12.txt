char * makeGood(char * s){
    char* ans = malloc(sizeof(char)* (strlen(s)+1));
    int idx = -1;
    for(int i = 0; i < strlen(s); i++){
        if(idx > -1 && abs(ans[idx] - s[i]) == 32){ // bad 
            idx--;
        }
        else{ // empty ans or good
            idx++;
            ans[idx] = s[i];
        }
    }
    ans[idx+1] = '\0';
    return ans;
}
