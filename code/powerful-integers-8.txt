    vector<int> powerfulIntegers(int x, int y, int bound) {
        vector<int> powx{1}, powy{1};
        while(powx.back()<bound && x!=1) powx.push_back(powx.back()*x);
        while(powy.back()<bound && y!=1) powy.push_back(powy.back()*y);
        unordered_set<int> s;
        for(auto px : powx){
            for(auto py:powy)
                if(px+py<=bound) s.insert(px+py);            
        }
        return vector<int>(s.begin(), s.end());
    }
