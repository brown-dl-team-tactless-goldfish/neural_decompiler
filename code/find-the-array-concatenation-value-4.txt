#include<bits/stdc++.h>
class Solution {
public:
    long long findTheArrayConcVal(vector<int>& nums) {
        
        long long int n=nums.size();
        long long int i=0,j=n-1;
        long long int ans=0;
        while(i<j)
        {
            string s1 = to_string(nums[i]);
            string s2 = to_string(nums[j]);
            string s = s1 + s2;
            long long int c = stoi(s);
            ans+=c;
            i++;
            j--;
        }
        if(n%2)
            ans+=nums[n/2];
        return ans;  
    }
};
