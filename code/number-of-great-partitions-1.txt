class Solution:
    def countPartitions(self, nums: List[int], k: int) -> int:
        mod = 1_000_000_007
        if sum(nums) < 2*k: return 0 
        dp = [0]*k
        dp[0] = 1
        for x in nums: 
            for i in range(k-1-x, -1, -1): 
                dp[i+x] += dp[i]
        return (pow(2, len(nums), mod) - 2*sum(dp)) % mod
