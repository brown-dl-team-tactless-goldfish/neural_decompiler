SELECT 
  E.employee_id, 
  E.department_id 
FROM 
  Employee AS E JOIN (
    SELECT 
      employee_id, 
      COUNT(department_id) AS cnt
    FROM 
      Employee
    GROUP BY 
      employee_id
  ) AS C 
  ON E.employee_id = C.employee_id
WHERE 
  cnt = 1 OR primary_flag = 'Y'
