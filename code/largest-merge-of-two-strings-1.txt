char * largestMerge(char * word1, char * word2){
	static char answer[6005];
	int index_answer = 0;
	memset(answer, 0, sizeof(answer));
	
	while(1)
	{
		if(strcmp(word1,word2) == 1)
		{
			answer[index_answer++] = *word1;
			word1++;
		}
		else if(strcmp(word1,word2) == 0)
		{
			answer[index_answer++] = *word1;
			word1++;
		}
		else if(strcmp(word1,word2) == -1)
		{
			answer[index_answer++] = *word2;
			word2++;
		}
		if(strlen(word1) == 0 && strlen(word2) == 0)
			break;
	}
	return answer;
}
