int countCollisions(string directions) {
    int ans = 0, rc = 0, sc = 0;
    for (char c : directions) {
        if (c == 'R')
            rc++, sc = 0;
        else if (c == 'S')
            ans += rc, rc = 0, sc = 1;
        else if (rc)
            ans += rc + 1, sc = 1, rc = 0;
        else if (sc)
            ans++;
    }
    return ans;
}
