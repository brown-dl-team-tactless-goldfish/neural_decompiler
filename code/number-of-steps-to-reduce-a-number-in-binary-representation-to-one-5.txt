int numSteps(char * s){
    int step = 0, CARRY = 0;
    for(int i = strlen(s) - 1; i > 0; i--)
    {
        if(((s[i] == '1') && (CARRY == 0)) || ((s[i] == '0') && (CARRY == 1)))
        {
            step++;/* for add 1 */
            CARRY = 1;/*  for next char */
        }
        else if((s[i] == '1') && (CARRY == 1))
        {
            CARRY = 1;
        }
        step++;/* for divide it by 2 */
    }
    if(s[0] == '1' && CARRY == 1)
        step++;
    return step;
}
