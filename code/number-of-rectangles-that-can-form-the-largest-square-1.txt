int countGoodRectangles(int **rectangles, int rectanglesSize, int *rectanglesColSize) {
    int squares[rectanglesSize];
    int largest_square = 0, count = 0;
    
    for (int i = 0; i < rectanglesSize; i++) {
        if (rectangles[i][0] < rectangles[i][1])
            squares[i] = rectangles[i][0];
        else
            squares[i] = rectangles[i][1];
			
        if (squares[i] >= largest_square) {
            if (squares[i] > largest_square) {
                largest_square = squares[i];
                count = 1;
            } else
                count++;
        }
    }
    
    return count;
}
