class Solution {
public:
    vector<int> beautifulArray(int n) {
      if(n==1)
          return {1};
        vector<int> left=beautifulArray(n/2);
        vector<int> right=beautifulArray((n+1)/2);
        
        vector<int> ans;
        
        for(auto n:left)
            ans.push_back(n*2);
        for(auto n:right)
            ans.push_back(n*2-1);
        return ans;
    }
};
