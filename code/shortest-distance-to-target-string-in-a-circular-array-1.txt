#define MIN(A, B) (A < B ? A : B)
#define MINIMUM(a,b,c) MIN(MIN(a,b),c)

int closetTarget(char ** words, int wordsSize, char * target, int startIndex){
    int i, n = wordsSize, mn = n + 1;
    
    for (i = 0; i < n; i++) 
        if (strcmp(words[i], target) == 0) 
            mn = MINIMUM(mn, abs((n - i + startIndex) % n), abs((n + i - startIndex) % n));

    return (mn == n + 1) ? -1 : mn; 
}
