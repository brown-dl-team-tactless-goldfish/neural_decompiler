void DFS(int* alpha, int* sum){
    for(int i = 0; i < 26 ;i++){
        if(alpha[i] == 0)
            continue ;
        (*sum)++;
        alpha[i]--;
        DFS(alpha, sum) ;
        alpha[i]++;
    }
}
int numTilePossibilities(char * tiles){
    int* alpha = calloc(26, sizeof(int) ) ;
    for(int i = 0; i < strlen(tiles) ; i++){
        alpha[ tiles[i] - 'A' ]++ ;
    }
    int sum = 0;
    DFS( alpha, &sum ) ;
    free(alpha) ;
    return sum ;
}
