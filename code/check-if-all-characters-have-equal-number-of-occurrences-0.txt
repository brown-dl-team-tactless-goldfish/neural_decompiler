bool areOccurrencesEqual(char * s){
    int counts[26];
    int i;
    int max;
    for (i=0; i<26; i++)
        { counts[i]=0; }
    for (i=0; i<strlen(s); i++)
        {
         counts[s[i]-'a']+=1;
         max = counts[s[i]-'a'];
        }
    for (i=0; i<26; i++)
        {
         if (counts[i]!=0 && counts[i]!=max) return false; 
        }
    return true;
      
}
