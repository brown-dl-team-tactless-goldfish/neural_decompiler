class Solution:
    def numSmallerByFrequency(self, queries: List[str], words: List[str]) -> List[int]:
        # print("queries = ", queries)

        def f(word):
            # print("word = ", word)
            word = sorted(word)
            # print("after sort word = ", word)
            c = word[0]
            res_num = 0
            for character in word:
                if character == c:
                    res_num += 1

            return res_num

        queries_data = []
        for q in queries:
            tmp = f(q)
            queries_data.append(tmp)

        word_data = []
        for q in words:
            tmp = f(q)
            word_data.append(tmp)

        # print("end queries_data = ", queries_data)
        # print("end word_data = ", word_data)

        res = []
        for q in queries_data:
            tmp = 0
            for w in word_data:
                if (q < w):
                    tmp += 1
            res.append(tmp)

        # print("end res = ", res)
        return res
