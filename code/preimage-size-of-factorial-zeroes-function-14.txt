#define ll long long int
    bool bs(ll l,ll r,int K){
        if(l>r) return false;
        ll mid=l+(r-l)/2;
        ll ans=5,cnt=0;
        
        while(ans<=mid){
            cnt+=mid/ans;
            ans=ans*5;
        }
        if(cnt==K) return true;
        if(cnt>K) return bs(l,mid-1,K);
        return bs(mid+1,r,K);
            
    }
    int preimageSizeFZF(int K) {
        bool f=bs(0,1e10,K);
        if(f) return 5;
        return 0;
    }
