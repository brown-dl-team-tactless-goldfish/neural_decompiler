class Solution {
public:
    long long taskSchedulerII(vector<int>& t, int s) {
        long long curDay = 0;
        unordered_map<long long, long long> m;
        for (int i = 0; i < t.size(); ++i){
            int n = t[i];
            if (m.find(n) == m.end() || m[n] + s < curDay + 1)
                m[n] = ++curDay;
            else m[n] = curDay = m[n] + s + 1;
        }
        return curDay;
    }
};
