public class Solution {
    public int MinSwaps(string s) {
        if(string.IsNullOrEmpty(s)) return 0;
        
        int leftB = 0, invalid = 0;
        foreach(var c in s)
        {
            if(c=='[') leftB++;
            else
            {
                if(leftB == 0) invalid++;
                else leftB--;
            }
        }
        
        return (invalid+1)/2;
    }
}
