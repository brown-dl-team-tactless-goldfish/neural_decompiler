char ** findOcurrences(char * text, char * first, char * second, int* returnSize){
    *returnSize = 0;
    char **out = (char *) malloc(sizeof(out) * 333), *p = strtok(text, " "), *f  = NULL, *s = NULL;       
    while (p) {
        if (f && strcmp(first, f) == 0 && s && strcmp(second, s) == 0)
           out[(*returnSize)++] = p; 
        f = s, s = p, p = strtok(NULL, " ");
	}
    return out; 
}
