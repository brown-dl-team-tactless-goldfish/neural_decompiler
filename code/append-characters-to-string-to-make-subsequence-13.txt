class Solution {
public:
    int appendCharacters(string s, string t) {
        int i = 0, j = 0;
        int ans = 0;
        while(j < s.size())
        {
            if(t[i] == s[j]) {
                i++;
                j++;
            }
            else {
                j++;
            } 
        }
        if(i == t.size()) return 0;
        return t.size() - i;
    }
};
