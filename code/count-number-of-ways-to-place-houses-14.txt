typedef long long ll;
class Solution {
public:
    ll M = 1e9+7;
    int countHousePlacements(int n) {
        vector<ll> vec(n+1,0);
        vec[0] = 1;
        vec[1] = 2;
        for(int i=2; i<=n; i++){
            vec[i] = vec[i-1] + vec[i-2];
            vec[i] %= M;
        }
        return (vec[n]*vec[n])%M;
    }
};
