public class Solution {
    public int FindMaxK(int[] nums) {
        Array.Sort(nums);
        HashSet<int> set = new(nums);
        
        for(int i = nums.Length - 1; i >= 0; i--)
        {
            if(nums[i] < 0)
                return -1;
            if(set.Contains(-nums[i]))
                return nums[i];
        }
        
        return -1;
    }
}
