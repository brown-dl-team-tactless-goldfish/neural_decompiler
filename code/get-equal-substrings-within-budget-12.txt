class Solution {
public:
    int equalSubstring(string s, string t, int maxCost) {
        int res = 0, cost = 0, i = 0, j = 0;
        while (j < s.size())
        {
            cost += abs(s[j] - t[j]);
            while (cost > maxCost)
            {
                cost -= abs(s[i] - t[i]);
                ++i;
            }
            
            res = max(j - i + 1, res);
            ++j;
        }
        
        return i < s.size() ? max(res, j - i) : res;
    }
};
