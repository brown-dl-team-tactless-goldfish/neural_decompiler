class Solution {
public:
    string removeOccurrences(string s, string part) {
        int pLen = part.length();
        string res;
        for (char c : s)
        {
            res.push_back(c);
            if (c == part.back() && res.length() >= pLen && res.substr(res.length() - pLen) == part)
                res.resize(res.length() - pLen);
        }
        
        return res;
    }
};
