int abc(int * arr, int n, int o){

    for(int i=0; i<n; i++){
        if(arr[i] == o){
            o = o*2;
            abc(arr, n, o);
            i=-1;
        }
        
    }

    return o;
}





int findFinalValue(int* nums, int numsSize, int original){

    original = abc(nums, numsSize, original);

    return original;
}
