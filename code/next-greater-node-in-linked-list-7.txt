typedef struct _ListNode {
    int val;
    struct _ListNode *next;
}ListNode;

int* nextLargerNodes(ListNode* head, int* returnSize){
    int max, count = 0, pos = 0;
    ListNode *iterator = head;
    
    while (iterator != NULL) {
        count++;
        iterator = iterator->next;
    }
    int support[count];
    iterator = head;
    while (iterator != NULL) {
        support[pos++] = iterator->val;
        iterator = iterator->next;
    }
    int *answer = (int *) malloc (count * sizeof(int));
    for (int i = 0; i < pos; i++) {
        max = 0;
        for (int j = (i+1); j < pos; j++)
            if (support[i] < support[j]) {
                max = support[j];
                j = pos;
            }
        answer[i] = max;
    }
    *returnSize = pos;
    return answer;
}
