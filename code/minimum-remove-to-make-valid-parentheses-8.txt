char * minRemoveToMakeValid(char * s){
    int S[100005] = {0};
    int n = 0;
    int i, j, l;
    j = 0;
    for (i = 0; s[i]; i++) {
        if (s[i] == '(') {
            S[n++] = i;
        } else if (s[i] == ')') {
            if (n == 1) {
                memmove(&s[j], &s[S[n-1]], i-S[n-1]+1);
                j += i-S[n-1]+1;
                n--;
            } else if (n > 1) {
                // later
                n--;
            } else { // n == 0
                // skip
            }
        } else {
            if (n > 0) {
                // later
            } else { // n == 0
                s[j++] = s[i];
            }
        }
    }
    l = i;
    
    if (n > 0) {
        for (i = 0; i < n; i++) {
            if (i == n-1) {
                memmove(&s[j], &s[S[i]+1], l-S[i]-1);
                j += l-S[i]-1;
            } else {
                memmove(&s[j], &s[S[i]+1], S[i+1]-S[i]-1);
                j += S[i+1]-S[i]-1;
            }
        }
    }
    s[j] = 0;
    return s;
}
