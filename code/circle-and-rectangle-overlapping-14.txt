class Solution {
 public:
  bool checkOverlap(int radius,
                    int x_center,
                    int y_center,
                    int x1,
                    int y1,
                    int x2,
                    int y2) {
    for (int i = x1; i <= x2; ++i) {
      for (int j = y2; j >= y1; --j) {
        if (pow(abs(i - x_center), 2) + pow(abs(j - y_center), 2) <=
            radius * radius) {
          return true;
        }
      }
    }

    return false;
  }
};
