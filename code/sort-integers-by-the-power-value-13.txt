	int countpower(int n) {
    int c=0;
    
    while(n!=1){
        if(n%2==0){
            n=n/2;
        } else {
            n=n*3+1;
        }
        c++;
    }
    
    return c;
}

int getKth(int lo, int hi, int k) {
    vector<pair<int,int>> powerArr;
    
    for(int i=lo; i<=hi; i++) {
        powerArr.push_back({countpower(i), i});
    }
    
    sort(powerArr.begin(), powerArr.end());
    
    return powerArr[k-1].second;
}
