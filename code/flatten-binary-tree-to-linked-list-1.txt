void flatten(struct TreeNode* root)
{
        if(root)
        {
            struct TreeNode* flat_tree = root->right;
            
            flatten(root->left);
            flatten(root->right);
            
            root->right = root->left;
            root->left = NULL;
            while(root->right)
            {
                root = root->right;
            }
            
            root->right = flat_tree;
        }
   
}
