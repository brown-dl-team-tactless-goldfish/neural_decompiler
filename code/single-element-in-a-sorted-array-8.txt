class Solution {
    public int singleNonDuplicate(int[] nums) {
        int hash[]=new int[100000];
        int i;
        for(i=0;i<nums.length;i++)
        {
            hash[nums[i]]++;
        }
        for(i=0;i<100000;i++)
        {
            if(hash[i]==1)
            {
                return i;
            }
        }
        return i;
    }
}
