class Solution(object):
    def mostFrequentEven(self, nums):
        maxfreq = 0
        ele = -1
        for i in range(0, len(nums)):
            count = 0
            if (nums[i] % 2 == 0):
                count = nums.count(nums[i])
                if count>maxfreq:
                    maxfreq = count
                    ele = nums[i]

                if count == maxfreq:
                    if nums[i]<ele:
                        ele = nums[i]

        return ele
