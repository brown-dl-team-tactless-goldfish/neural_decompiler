class Solution {
public:
    double averageWaitingTime(vector<vector<int>>& customers) {
        double ans = 0;
        long long sum = 0;
        int prev = 0;
        for(int i = 0;i<customers.size();i++)
        {
            prev = max(customers[i][0], prev) + customers[i][1];
            int temp = prev - customers[i][0];
            sum += temp;
        }
        ans = (double)sum / (double)customers.size();
        return ans;
    }
};
