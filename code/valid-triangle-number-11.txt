class Solution {
public:
    int triangleNumber(vector<int>& nums) {
        
        sort(nums.begin(), nums.end());
        int n = nums.size();
        int res = 0; 
        
        for(int i = n-1; i >= 2; i--)
        {
            int lo = 0;
            int hi = i-1;
            
            while(lo  < hi)
            {
                if(nums[lo] + nums[hi] > nums[i])
                {
                    res += (hi-lo);
                    hi--;
                }
                else
                {
                    lo++;
                }
            }
        }
        
        return res;
    }
};
