bool search(int* arr, int arrSize, int idx, int* tmp) {
    if(idx >= arrSize || idx < 0 || tmp[idx] == 1) return false;
    if(arr[idx] == 0) return true;
    tmp[idx] = 1;
    return search(arr, arrSize, idx+arr[idx], tmp) | search(arr, arrSize, idx-arr[idx], tmp);
}

bool canReach(int* arr, int arrSize, int start) {
    int* tmp = (int*) calloc(arrSize, sizeof(int));
    bool ans = search(arr, arrSize, start, tmp);
    free(tmp);
    return ans;
}
