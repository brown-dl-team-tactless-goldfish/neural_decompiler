char stack[10001];
int top=-1;
void push(char c){
    top++;
    stack[top]=c;
}
void pop(){
    top--;
}

int minAddToMakeValid(char * s){
    
    top=0;
       for(int i=0;i<strlen(s);i++){
        if (s[i]==')'&& stack[top]=='(')
            pop();
        else
            push(s[i]);}
    return top;
}
