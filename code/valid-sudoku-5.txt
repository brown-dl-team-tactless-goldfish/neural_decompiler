bool checkrow(vector<vector<char>>& board, int row, int num)
{
    int cnt = 0;
    for(int i = 0; i < 9; i++)
        if(board[row][i] == num)
            cnt++;
    if(cnt > 1)
        return false;
    return true;
}
bool checkcol(vector<vector<char>>& board, int col, int num)
{
    int cnt = 0;
    for(int i = 0; i < 9; i++)
        if(board[i][col] == num)
            cnt++;
    if(cnt > 1)
        return false;
    return true;
}
bool checkgrid(vector<vector<char>>& board, int row, int col, int num)
{
    int cnt = 0;
    for(int i = 0; i <  3; i++)
        for(int j = 0; j <  3; j++)
            if(board[i + row][j + col] == num)
                cnt++;
    if(cnt > 1)
        return false;
    return true;
}
