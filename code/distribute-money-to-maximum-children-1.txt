int distMoney(int money, int children){
    if (money == children*8) return children;
    if (children > money) return -1;
    
    if (children *8  < money) return children - 1;
    else if (money  < 8+children-1 || (children==2 && money==12) ) return 0;
    return (money - ((money -children)/7  * 8)==4 && children-(money-children)/7==1 )? (money-children)/7-1 : (money-children)/7  ;
}
