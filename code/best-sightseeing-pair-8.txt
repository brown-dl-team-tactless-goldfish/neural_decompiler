class Solution {
public:
    int maxScoreSightseeingPair(vector<int>& nums)
    {
        int n=nums.size();
        int max_sum=nums[0];
        int ans=0;
        for(int i=1;i<n;i++)
        {
            ans=max(ans,max_sum+nums[i]-i);
            max_sum=max(max_sum,nums[i]+i);
        }
        return ans;
    }
};
