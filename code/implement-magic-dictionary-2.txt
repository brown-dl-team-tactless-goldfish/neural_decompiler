def __init__(self):
    """
    Initialize your data structure here.
    """
    self.dic = []

def buildDict(self, dict):
    """
    Build a dictionary through a list of words
    :type dict: List[str]
    :rtype: void
    """
    self.dic = dict
    #print "self.dic = ", self.dic
def search(self, word):
    """
    Returns if there is any word in the trie that equals to the given word after modifying exactly one character
    :type word: str
    :rtype: bool
    """

    if len(self.dic) < 1:
        return False

    len_word = len(word)
    for word_in_dic in self.dic:
        #print "word_in_dic = ", word_in_dic
        len_word_in_dic = len(word_in_dic)
        #print "len_word_in_dic = ", len_word_in_dic

        if len_word == len_word_in_dic:
            num_diff = 0
            for idx in range(len_word):
                c_1, c_2 = word[idx], word_in_dic[idx]
                #print "c_1 = ", c_1, " c_2 = ", c_2

                if c_1 != c_2:
                    #print "found one difference"
                    num_diff += 1

                if num_diff > 1:
                    break

            #print "1 num_diff = ", num_diff
            if num_diff == 1:
                return True

    return False
