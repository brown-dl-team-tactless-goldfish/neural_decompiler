public class Solution {
    public long MinimumRemoval(int[] beans) {
        long min = long.MaxValue, sum = 0;
        Array.Sort(beans);
        foreach (var item in beans) sum += item;
        for (int i = 0; i < beans.Length; i++) {
            long cnt = beans.Length - i;
            min = Math.Min(min, sum - (beans[i] * cnt));
        }
        return min;
    }
}
