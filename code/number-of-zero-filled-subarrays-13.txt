
long long zeroFilledSubarray(int* nums, int numsSize){
        
        int count =0; 
        long long sum = 0;
        
        for(int i =0; i<numsSize; i++) {
            
            if(nums[i] == 0) {
                count++;
                sum += count;
            }
            else {
                count = 0;
            }
        }
        
        return sum;
}
