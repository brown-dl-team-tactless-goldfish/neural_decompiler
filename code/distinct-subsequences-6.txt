int dp[1000];
int numDistinct(char* s, char* t) {
    int m = strlen(t);
    memset(dp, 0, (m + 1) << 2);
    
    dp[0] = 1;
    for(int i = 0; s[i]; i++)
    {
        for(int j = m; j >= 1; j--)
        {
            if(s[i] == t[j - 1])
            {
                dp[j] += dp[j - 1];
            }
        }
    }

    return dp[m];
}
