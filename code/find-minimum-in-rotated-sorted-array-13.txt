
int bs(int *nums, int l, int r)
{
        if (l >= r)
                return nums[l];
        
        int mid = l + (r - l) / 2;
        if (nums[mid] > nums[r])
                return bs(nums, mid + 1, r);
        else
                return bs(nums, l, mid);
}

int findMin(int* nums, int numsSize)
{
        return bs(nums, 0, numsSize - 1);
}
