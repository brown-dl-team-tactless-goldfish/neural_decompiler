class Solution:
    def removePalindromeSub(self, s: str) -> int:
        if (s == ""):
            return 0
        
        def isPalindrome(s):
            # print("isPalindrome s = ", s)
            len_s = len(s)
            if (len_s < 2):
                return True
            elif (s[0] != s[-1]):
                return False
            else:
                return isPalindrome(s[1:len_s-1])

        if (isPalindrome(s) == True):
            return 1
        return 2
