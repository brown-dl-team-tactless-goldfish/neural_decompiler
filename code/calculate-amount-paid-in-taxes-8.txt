class Solution {
public:
    double calculateTax(vector<vector<int>>& b, int inc) {
        double ans=double(min(b[0][0],inc)*b[0][1]) / 100;
        if(b[0][0]>=inc)    return ans;
        for(int i=1; i < b.size(); i++){
            ans += double((min(b[i][0], inc)-b[i-1][0]) * b[i][1]) / 100;
            if(b[i][0] > inc)   break;
        }
        return ans;
    }
};
