int maxProfit(int* prices, int pricesSize){
    
    int restPtr = 0;
    int soldPtr = 0;
    int holdPtr = INT_MIN;
    int hold, sold, rest;
    for(int i  = 1; i <= pricesSize; i++){
        hold = fmax(holdPtr, (restPtr-prices[i-1]));
        sold = holdPtr + prices[i-1];
        rest = fmax(restPtr, soldPtr);
        
        holdPtr = hold;
        soldPtr = sold;
        restPtr = rest;
    }
    
    int ans = fmax(restPtr, soldPtr);

    return ans;
}
