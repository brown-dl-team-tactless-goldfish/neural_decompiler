public class Solution {
    public bool CarPooling(int[][] trips, int capacity) {
        
        int[] arr = new int[1001];
        
        foreach(var trip in trips)
        {
            arr[trip[1]] -= trip[0];
            arr[trip[2]] += trip[0];
        }
        
        foreach(int c in arr)
        {
            capacity += c;
            
            if(capacity < 0)
                return false;
        }
             
        return true;
    }
}
