while(low < high) { 
	mid = low + (high - low) / 2;
	flowers = 0;
	boquets = 0;
	for (i = 0; i < bloomDaySize; i++) {
		if (bloomDay[i] > mid)
			flowers=0;
		else {
			flowers++;
			if (flowers == k) {
				boquets++;
				flowers=0;
			}    
		}
        
		if (boquets >= m)
			break;
	}
	
    if (boquets < m)
		low = mid + 1;
	else
		high = mid;
}
return low;
