    int high =n-1,low=0;
    while(high>=low){
        int mid = low+(high-low)/2;
        int maxno = 0;int c =-1;
      for(int i=0;i<mat[mid].size();i++){
          if(mat[mid][i]>maxno){
              c=i;maxno=mat[mid][i];
          }
      }
        if(mid>0 && mat[mid][c]<mat[mid-1][c]){
            high=mid-1;
        }
       else  if(mid<n-1 && mat[mid][c]<mat[mid+1][c]){
            low=mid+1;
        }
       else{  return {mid,c};}
         
    }
    return{-1,-1};
}
