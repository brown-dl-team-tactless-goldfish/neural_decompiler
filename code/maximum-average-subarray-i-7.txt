double findMaxAverage(int* nums, int numsSize, int k){
	double tmp = 0;
	double ans = 0;
	for(int i = 0; i < k; i++){
		tmp += nums[i];
	}
	tmp /= k;
	ans = tmp;
	int i = 0;
	int j = k - 1;
	while(j < numsSize - 1){
		tmp = tmp + (double)nums[++j]/k - (double)nums[i++]/k;
		ans = ans < tmp ? tmp : ans;
	}
	return ans;
}
