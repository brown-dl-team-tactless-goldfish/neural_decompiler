int chalkReplacer(int* chalk, int chalkSize, int k){
    long long s = 0; 
    for(int i=0;i<chalkSize;i++) s+=chalk[i];
    if(k>=s) k=k%s;
    int ans = -1;
    for(int i=0;i<chalkSize;i++){
        if(k-chalk[i]==0) return i+1;
        if(k-chalk[i]<0) return i;
        k-=chalk[i];
    }
    return ans;
}
