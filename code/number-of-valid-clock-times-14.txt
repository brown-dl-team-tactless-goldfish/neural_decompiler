    int countTime(string time) {
        auto h1 = time[0], h2 = time[1], m1 = time[3], m2 = time[4];
        int res = 1;
        if(m2 == '?') res *= 10;
        if(m1 == '?') res *=6;
        if(h1 == '?' && h2 == '?') res *=24;
        else if(h1 == '?' && h2 != '?')
            res *= h2 > '3' ?  2 : 3;
        else if(h1 != '?' && h2 == '?')
            res *= h1 == '2' ? 4 : 10;
        return res;
    }
