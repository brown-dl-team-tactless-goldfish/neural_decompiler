select
    Request_at as Day,
    round(SUM(CASE WHEN Status like 'cancelled%' THEN 1 ELSE 0 END)
          /count(Status),2) as 'Cancellation Rate'
from
    Trips
where
    Request_at between '2013-10-01' and '2013-10-03'
    and
    Client_id not in (select Users_Id from Users where Banned = 'Yes')
group by
    Request_at;
