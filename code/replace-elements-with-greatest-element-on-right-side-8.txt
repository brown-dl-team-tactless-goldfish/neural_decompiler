int* replaceElements(int* arr, int arrSize, int* returnSize){
    *returnSize = arrSize;
    int j=-1,i=-2;
    for(arrSize--;arrSize>=0;arrSize--)
        if(j>i){
            i = arr[arrSize];
            arr[arrSize]=j;
        }else{
            j = arr[arrSize];
            arr[arrSize]=i;
        }
    return arr;
}
