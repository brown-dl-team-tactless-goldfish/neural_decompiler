/**
 * @param {number} target
 * @param {number} maxDoubles
 * @return {number}
 */
var minMoves = function(target, d) 
{
  let count=0;
  let n=target;
  while(d>0&&n>0)
    {
      if(n%2==0)
        {
          n=n/2;
          d--;
        }
      else
        {
          n=n-1;
        }
      count++;
    }
  return n-1+count;
};
