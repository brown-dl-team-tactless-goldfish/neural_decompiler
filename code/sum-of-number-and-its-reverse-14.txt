class Solution {
public:
    int reverse(int num){
        int result=0,rem;
        while(num>0){
            rem=num%10;
            result=(result*10)+rem;
            num/=10;
        }
        return result;

    }
    bool sumOfNumberAndReverse(int num) {
       for(int i=num/2;i<=num;i++)
       {
           int rev=reverse(i);
           if(i+rev==num)return true;
       }
       return false;    
       
    }
};
