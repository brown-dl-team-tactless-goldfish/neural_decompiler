class Solution {
public:
    bool isPrime(int n) {
      if(n == 1) return false; 
      for(int i=2;i<=sqrt(n);i++) if(n%i == 0) return  false; 
      return true;
    }
    int diagonalPrime(vector<vector<int>>& nums) {
        int j=nums[0].size()-1;
        int ans = 0;
        for(int i=0;i<nums.size();i++){
            if(isPrime(nums[i][i])) ans = max(ans,nums[i][i]);
            if(isPrime(nums[i][j])) ans = max(ans,nums[i][j]); 
            j--;
        }
        return ans;
    }
};
