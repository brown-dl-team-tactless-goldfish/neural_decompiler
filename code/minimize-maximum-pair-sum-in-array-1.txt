int cmp(const void* a, const void* b) {
    return *(const int*) a - *(const int*) b;
}

int minPairSum(int* nums, int numsSize) {
    qsort(nums, numsSize, sizeof(int), cmp);
int max=0;
    for(int i=0;i<numsSize/2;i++)
    {
        int sum=0;
        sum=nums[i]+nums[numsSize-1-i];
        if(sum>max)
        {
            max=sum;
        }
    }
    return max;
}
