   int len=strlen(s);

   int maxpow=1,count;

    for(int i=0;i<len-1;i++){
        count=1;
        for(int j=i+1;j<len;j++){
            
            if(s[i]==s[j]){

            }

        }
        
         if(maxpow<count){maxpow=count;}
    }

return maxpow;
