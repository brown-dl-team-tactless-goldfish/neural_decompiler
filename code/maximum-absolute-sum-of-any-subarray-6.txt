class Solution {
public:
    int maxAbsoluteSum(vector<int>& nums) {
        int max_{0}, min_{0}, itmax{0}, itmin{0};
        for(const auto &n: nums) {
            itmax += n, itmin += n;
            itmax = std::max(itmax, 0);
            itmin = std::min(itmin, 0);
            max_ = std::max(max_, itmax);
            min_ = std::min(min_, itmin);
        }
        return std::max(max_, -min_);
    }
};
