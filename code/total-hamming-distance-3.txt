int totalHammingDistance(int* nums, int numsSize){
    int ct[30] = {0};
    for (int i = 0; i < numsSize; i++) {
        int tmp = nums[i];
        int count = 0;
        while (tmp) {
            ct[count] += tmp & 0b1;
            tmp=tmp>>1;
            count++;
        }
    }
    int res = 0;
    for (int i = 0; i < 30; i++) {
        res += ct[i]*(numsSize-ct[i]);
    }
    return res;
}
