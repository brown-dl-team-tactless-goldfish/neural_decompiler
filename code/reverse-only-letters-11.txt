class Solution {
public:
    string reverseOnlyLetters(string s) {
        string k="";
        for(int i=s.size()-1;i>=0;i--){
            if(s[i]>='a' && s[i]<='z' || s[i]>='A' && s[i]<='Z'){
                    k+=s[i];
        }
        }
        int t=0;
        for(int i=0;i<s.size();i++){
             if(s[i]>='a' && s[i]<='z' || s[i]>='A' && s[i]<='Z'){
                 s[i]=k[t];
                 t++;
             }
        }
        return s;


        
        
    }
};
