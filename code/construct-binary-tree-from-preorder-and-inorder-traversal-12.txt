struct TreeNode* buildTree(int* preorder, int preorderSize, int* inorder, int inorderSize){
    
    if(inorderSize<1||preorderSize<1) return NULL; 
    
    struct TreeNode*t1 = (struct TreeNode*)malloc(sizeof(struct TreeNode)); 
    t1->val = preorder[0]; 
    
    int i; 
    for( i=0;i<inorderSize;i++)
    {
        if(inorder[i]==t1->val)
        {
            break; 
        }
    }
    
    t1->left = buildTree((preorder+1),preorderSize-1,inorder,i);
    t1->right = buildTree((preorder+i+1),preorderSize-i-1,(inorder+i+1),inorderSize-i-1);
        
    return t1; 
    
}
