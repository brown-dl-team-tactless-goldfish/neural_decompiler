int passThePillow(int n, int time){
    int i=1,count=0;
    while(i<=n&&count<time)
    {
        if(i<n)
        {
            count++;
            i++;
        }
        else if(i=n)
        {
            while(count<time)
            {
             i--;
             count++;
             if(i==1)
             break;
            }   
        }
    }
    return i;
}
