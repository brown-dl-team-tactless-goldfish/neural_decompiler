class Solution {
public:
    vector<int> rearrangeArray(vector<int>& nums) 
    {
        int n=nums.size();
        sort(nums.begin(),nums.end());
        vector<int>ans(n,0);
        int l=0;
        int r=n-1;
        int c=0;
        while(l<r)
        {
            ans[c++]=nums[l];
            ans[c++]=nums[r];
            l++;
            r--;
        }
        if(n%2==1)
        {
            ans[c++]=nums[l];
        }
        return ans;
    }
};
