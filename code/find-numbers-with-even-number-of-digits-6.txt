int result(int n)
{
    int counter=0;
    
    if (n<0)
        n*(-1);
    
    while (n>0)
    {
        n=n/10;
        counter++;
    }
    
    return counter;
}

int findNumbers(int* nums, int numsSize){
    int c=0;
    
    for (int i=0; i<numsSize; i++)
    {
        if ((result(nums[i]))%2==0)
            c++;
    }
    
    return c;
}
