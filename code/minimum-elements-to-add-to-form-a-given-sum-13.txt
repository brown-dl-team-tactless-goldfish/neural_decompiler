class Solution {
public:
    int minElements(vector<int>& nums, int limit, int goal) {
         long long ans=0;
         long long x=0;
         for(int i=0; i<nums.size(); i++) x+=nums[i];
         long long val=abs(goal-x);
         ans+=val%limit?1+(val/limit):val/limit;
        return ans;
    }
};
