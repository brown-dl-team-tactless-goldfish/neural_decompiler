char * mergeAlternately(char * word1, char * word2){
    int size1 = strlen(word1);
    int size2 = strlen(word2);
    char *res = (char*)malloc(sizeof(char) * (size1 + size2 + 2));
    
    int ind = 0;
    int i = 0;
    int j = 0;
    
    while((i<size1) && (j<size2)){
		res[ind++] = word1[i++];
		res[ind++] = word2[j++];}
    
    while(i<size1){res[ind++] = word1[i++];}
    while(j<size2){res[ind++] = word2[j++];}
    
    res[ind] = '\0';
    return res;
}
