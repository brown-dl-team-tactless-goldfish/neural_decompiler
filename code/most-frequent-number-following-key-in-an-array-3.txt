class Solution {
public:
    int mostFrequent(vector<int>& nums, int key) {
        map<int,int> target_count;
        
		//count for each target
        for (int i = 0; i < nums.size() - 1; i++)
            if (nums[i] == key)
                target_count[nums[i + 1]]++;
        
        int max_target_count = 0;
        int target;
	//find the target with highest count and return it
        for (auto c : target_count)
            if (c.second > max_target_count){
                target = c.first;
                max_target_count = c.second;
            }
        
        return target;
    }
};
