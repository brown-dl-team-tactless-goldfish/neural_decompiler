class Solution {
public:
    bool isIdealPermutation(vector<int>& nums) {
        int n = nums.size();
        int curMax = INT_MIN;
        
        for (int i = 2; i < n; i++) {
            curMax = max(curMax, nums[i-2]);
            if (curMax > nums[i]) {
                return false;
            }
        }
        
        return true;
    }
};
