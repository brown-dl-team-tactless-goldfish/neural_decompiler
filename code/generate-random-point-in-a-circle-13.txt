  Solution(double radius, double x_center, double y_center)
    : radius_(radius), centerX_(x_center), centerY_(y_center) {
    }
    
  vector<double> randPoint() {
    while (true) {
      double x = (double(rand())/RAND_MAX-0.5)*2*radius_;
      double y = (double(rand())/RAND_MAX-0.5)*2*radius_;
      if (x * x + y * y < radius_*radius_)
        return vector<double>{x+centerX_, y+centerY_};
    }
  }
private:
  double radius_;
  double centerX_;
  double centerY_;
