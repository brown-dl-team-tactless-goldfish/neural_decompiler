class Solution {
public:
    int longestContinuousSubstring(string s) {
        vector<int> dp(s.size(), 0);
        int res = 0;
        for(int i=0;i<s.size();i++) {
            dp[i] = 1;
            if(i>0 && s[i] == s[i-1]+1) {
                dp[i] = dp[i-1] + 1;
            }
            res = max(res, dp[i]);
        }

        return res;
    }
};
