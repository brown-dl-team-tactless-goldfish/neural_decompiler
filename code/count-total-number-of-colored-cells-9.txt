class Solution {
    long long recur(long long n){
        if(n == 0)
            return 0;
        if(n == 1)
            return 2;
        return 2*n + recur(n-2);
    }
public:
    long long coloredCells(int n) {
        long long tmp = 2*n-1;
        if(tmp > 1)
           return tmp + recur(tmp-2);
        return tmp;
    }
};
