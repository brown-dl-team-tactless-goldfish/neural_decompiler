TreeNode* getTargetCopy(TreeNode* original, TreeNode* cloned, TreeNode* target) {
  std::queue<TreeNode*> orig_queue;
  std::queue<TreeNode*> clone_queue;
  orig_queue.emplace(original);
  clone_queue.emplace(cloned);

  while (!orig_queue.empty()) {
    TreeNode* const orig_node = orig_queue.front();
    TreeNode* const clone_node = clone_queue.front();
    orig_queue.pop();
    clone_queue.pop();
    if (orig_node == target) {
      return clone_node;
    }
    if (orig_node->left) {
      orig_queue.emplace(orig_node->left);
      clone_queue.emplace(clone_node->left);
    }
    if (orig_node->right) {
      orig_queue.emplace(orig_node->right);
      clone_queue.emplace(clone_node->right);
    }
  }
  return NULL;
}
