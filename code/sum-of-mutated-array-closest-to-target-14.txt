class Solution {
public:
    int findBestValue(vector<int>& arr, int target) {
        std::sort(arr.begin(), arr.end());
        for (std::size_t i = 0; i < arr.size(); ++i) {
            int candidate = std::round((target - 0.0001) * 1.0 / (arr.size() - i));
            if (candidate <= arr[i]) {
                return candidate;
            }
            target -= arr[i];
        }
        return *arr.end();
    }
};
