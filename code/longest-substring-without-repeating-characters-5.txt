int lengthOfLongestSubstring(char * s){

    int arr[256] = {0};
    int start = 0;
    int len = 0;

    for (int i = 0; s[i]; i++)
    {
        char c = s[i];
        arr[c] != 0 ? (arr[c] > start ? start = arr[c] : start) : arr[c];
        (i - start + 1) > len ? len = (i - start + 1) : len;
        arr[c] = i + 1;
    }
    return (len);
  return(len);
}
