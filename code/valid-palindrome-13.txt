#include<string.h>
#include<stdio.h>
bool isChar(char a) {
    if ((a>='a' && a<='z') ||
        (a>='A' && a<='Z') ||
        (a>='0' && a<='9'))
        return true;
    return false;
}
char lowCase(char a) {
    if (a>='A' && a<='Z')
        return a-'A'+'a';
    return a;
}
bool isPalindrome(char* s) {
    
    unsigned int slen = strlen(s);
    if (slen==0 || slen==1) return true;
    
    int i,j;
    
    i = 0;
    j = slen-1;
    
    while(1) {
        while(!isChar(s[i])) {
            i++;
            if (i==j) return true;
        }
        while(!isChar(s[j])) {
            j--;
            if (j==i) return true;
        }
        if (lowCase(s[i])!=lowCase(s[j]))
            return false;
        i++;
        j--;
        if (i>=j) return true;
    }
    
}
