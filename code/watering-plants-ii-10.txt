class Solution {
public:
  int minimumRefill(vector<int>& p, int A, int B) {
    int a = A, b = B, ia = 0, ib = p.size()-1, tot = 0;
    while(ia < ib){
      if(p[ia] <= a) a -= p[ia];
      else a = A - p[ia], tot++;
      
      if(p[ib] <= b) b -= p[ib];
      else b = B - p[ib], tot++;
      
      ia++, ib--;
    }
    
    if(ia == ib){
      if(p[ia] <= a || p[ib] <= b);
      else tot++;
    }
    
    return tot;
  }
};
