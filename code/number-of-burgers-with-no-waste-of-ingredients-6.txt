vector<int> numOfBurgers(int t, int c) {
        if(t==0 and c==0) return {0,0};
        if(t%2!=0 or c>=t or c>t/2) return {};
        
        if( (4*c - t) < 0 or (4*c - t)%2 != 0 ) return {};
        else {
            int s  = (4*c - t)/2;
            int l = c - s;
            return {l,s};
        }
    }
