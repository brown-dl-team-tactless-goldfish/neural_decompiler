// Program to check whether a given number is an ugly number

bool isUgly(int num){
    if(num <= 0) {
        return false;
    }

    // Check and divide by 3 till it is possible
    while(num % 3 == 0) {
        num /= 3;
    }
    
    // Check and divide by 5 till it is possible
    while(num % 5 == 0) {
        num /= 5;
    }

    // Check for divisibility by 2/power of 2
    return ((num & num-1) == 0);
}
