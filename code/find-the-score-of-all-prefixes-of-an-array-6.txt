class Solution {
public:
    vector<long long> findPrefixScore(vector<int>& nums) {
        int maxi=nums[0];
        vector<long long int>ans(nums.size(), 0);
        for(int i=0; i<nums.size(); i++){
            maxi=max(maxi, nums[i]);
            ans[i]=nums[i];
            ans[i]+=maxi;
        }
        for(int i=1; i<nums.size(); i++){
            ans[i]+=ans[i-1];
        }
        return ans;
    }
};
