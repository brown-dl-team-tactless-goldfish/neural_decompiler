int canBeTypedWords(char * text, char * brokenLetters){
    int dic[26]={0};
    int x=0;
    while(brokenLetters[x]!='\0'){
        dic[brokenLetters[x]-'a']++;
        x++;
    }
    int count=0;
    char tok[2]=" ";
    char* token=strtok(text,tok);
    while(token){
        while(*token!='\0'&&dic[(*token)-'a']==0)*token++;
        if(*token=='\0')count++;
        token=strtok(NULL,tok);
    }
    return count;
}
