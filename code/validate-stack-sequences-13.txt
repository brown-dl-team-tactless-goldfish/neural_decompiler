class Solution:
    def validateStackSequences(self, pushed: List[int], popped: List[int]) -> bool:
        stack = list()
        count = 0

        for i, val in enumerate(pushed):
            stack.append(val)
            while count < len(pushed) and len(stack) > 0 and stack[-1] == popped[count]:
                stack.pop()
                count += 1

        return count == len(pushed)
