class Solution {
public:
    int destroyTargets(vector<int>& nums, int space) {

        unordered_map<int, int> freq;
        
        for (auto &i: nums) {
            int mod = i % space;
            freq[mod]++;   
        }
        
        int maxFreq = INT_MIN;
        int ans = INT_MAX;
        
        for (auto &i: nums) {
            int currFreq = freq[i % space];
            
            if (currFreq > maxFreq) {
                maxFreq = currFreq;
                ans = i;
            }
            
            if (currFreq == maxFreq) {
                ans = min(ans, i);
            }
        }

        return ans;
    }
};
