int countLargestGroup(int n){
    int record[37] = {0};
    for(int i = 1; i <= n; i++){
        int val = i;
        int d = 0;
        while(val){
            d += val % 10;
            val /= 10;
        }
        record[d]++;
    }
    int max = -1;
    int cn = 0;
    for(int i = 1; i < 37; i++){
        if(record[i] == max){
            cn++;    
        }
        else if(record[i] > max){
            max = record[i];
            cn = 1;
        }
    }
    return cn;
}
