bool canReach(string s, int mn, int mx) {
    int n = s.size();
    vector<bool>dp(n,false);
    dp[0] = true; //s[0] is always 0.
    int c = 0;
    for(int i=1;i<n;i++){
	//if we can take minimum jump
        if(i>=mn) c+=dp[i-mn];
		//exclude the value which is out of range
        if(i>mx) c-=dp[i-mx-1];
        
        dp[i] = (s[i]=='0') && (c>0); // if we can jump and s[i] must be 0.
    }
    return dp[n-1];
}
