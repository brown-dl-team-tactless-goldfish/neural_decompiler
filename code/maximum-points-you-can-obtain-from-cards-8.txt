int maxScore(int* cardPoints, int cardPointsSize, int k){
    if (k > cardPoints) {return;}
    int headSum = 0, tailSum = 0, maxSum = 0;
    int i;

    for (i = 0; i < k; ++i) {
        headSum += cardPoints[i];
    }
    i--; maxSum = headSum;

    for (int j = cardPointsSize - 1; j > cardPointsSize - 1 -k; --j) {
        headSum = headSum + cardPoints[j] - cardPoints[i]; i--;
        maxSum = (headSum > maxSum)? headSum : maxSum;
    }
    return maxSum;
}
