bool validMountainArray(int* arr, int arrSize){
    int flag_inc=0,flag_dec=0,start_index_inc,flag=1;
    if(arrSize<3)
        flag=0;
    for(int i=1;i<arrSize && flag==1;i++){
        if(arr[i]==arr[i-1]) flag=0;
        if(flag_dec==1 && arr[i]>arr[i-1])
            flag = 0;
        if(arr[i]>arr[i-1] && flag_inc==0){
            flag_inc = 1;
            start_index_inc = i;
        }
        else if(arr[i]<arr[i-1])
            flag_dec = 1;
    }

    if(start_index_inc>1 || flag_dec==0 || flag_inc==0) flag = 0;
    return flag;
}
