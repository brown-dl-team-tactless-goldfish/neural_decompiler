bool* camelMatch(char ** queries, int queriesSize, char * pattern, int* returnSize){
    bool *b = calloc(1, sizeof (bool *[queriesSize])); // Question needs this
    for (char *p, *q, i = 0 ; i < queriesSize ; b[i++] = !*p && !*q) {
        for (p = pattern, q = queries[i] ; *p && *q && ((*p == *q && p++) || !isupper(*q)) ; q++);
        while (!*p && *q && !isupper(*q++));
    }
    return *returnSize = queriesSize, b;
}
