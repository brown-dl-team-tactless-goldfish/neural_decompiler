public class Solution {
    public int CountPrefixes(string[] words, string s) {
        
        HashSet<string> set = new HashSet<string>();
        string prefix = "";
        foreach(var c in s)
        {
            prefix += c;
            set.Add(prefix);
        }
        
        int cnt = 0;
        foreach(var word in words)
        {
            if(set.Contains(word))
                cnt++;
        }
        
        return cnt;
    }
}
