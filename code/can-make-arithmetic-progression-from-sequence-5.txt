int cmp(const void* _a, const void* _b)
{
    int* a = (int*)_a;
    int* b = (int*)_b;
    return *a-*b;
}
bool canMakeArithmeticProgression(int* arr, int arrSize){
    qsort(arr, arrSize, sizeof(arr[0]), cmp);
    int DIS = arr[1]-arr[0];
    bool flag= true;
    for(int i=1;i<arrSize;i++)
    {
        if (arr[i]-arr[i-1]!=DIS)
        {
            flag = false;
            break;
        }
    }
    return flag;
}
