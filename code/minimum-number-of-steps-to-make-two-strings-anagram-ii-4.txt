class Solution {
public:
    int minSteps(string s, string t) {
    
       int mps[26] = {0}; 
        int mpt[26] = {0}; 
        for(int i = 0 ; i< s.length(); i++)
        {
            mps[s[i] - 'a']++; 
        }
        for(int i =0; i<t.length(); i++)
        {
            mpt[t[i] -'a']++; 
        }
        
        int count = 0; 
        for(int i = 0; i<26; i++)
        {
            count += (abs(mps[i] - mpt[i])); 
        }
        return count; 
    }
};
