class Solution {
public:
    int minimumDifference(vector<int>& nums, int k) {
        int diff = INT_MAX;
        sort(nums.begin() , nums.end());
        for(int i = 0;i + k <= nums.size();i++){
            int min1 = nums[i];
            int max1 = nums[i + k - 1];
            diff = min(diff , max1 - min1);
        }
        return diff;
    }
};
