class Solution:
    def combinationSum2(self, candidates: List[int], target: int) -> List[List[int]]:
        candidates.sort()
        # print("candidates = ", candidates)
        # print("target = ", target)

        res = []

        def helper(curSum, remainCandidates, curCandidates):
            if (curSum == target):
                curCandidates.sort()
                if (curCandidates not in res):
                    res.append(curCandidates)
                return

            if (remainCandidates == [] or curSum > target):
                return

            for num in remainCandidates:
                if (curSum+num <= target):
                    tmpArray = remainCandidates[:]
                    tmpArray.remove(num)

                    tmpArray2 = curCandidates[:]
                    tmpArray2.append(num)
                    helper(curSum+num, tmpArray, tmpArray2)

        helper(0, candidates, [])
        # print("end res = ", res)
        return res
