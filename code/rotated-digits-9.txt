bool is_valid_and_changed (int N) {
    bool res = false;
    while (N) {
        int d = N % 10;
        if (d == 2 || d == 5 || d == 6 || d == 9) res = true;
        if (d == 3 || d == 4 || d == 7) return false;
        N /= 10;
    }
    return res;
}

int rotatedDigits(int N) {
    int c = 0;
    for (int i = 1; i <= N; i++)  if (is_valid_and_changed (i)) c++;
    return c;
}
