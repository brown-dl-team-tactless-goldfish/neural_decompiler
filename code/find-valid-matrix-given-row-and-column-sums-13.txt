class Solution {
public:
    vector<vector<int>> restoreMatrix(vector<int>& rowSum, vector<int>& colSum) 
    {
        const int r = rowSum.size();
        const int c = colSum.size();
        
        vector<vector<int>> a(r, vector<int>(c, 0));
        
        int i = 0, j = 0;
        while(i < r && j < c) 
        {
            a[i][j] = min(rowSum[i], colSum[j]);
            rowSum[i] -= a[i][j];
            colSum[j] -= a[i][j];
            
            if(rowSum[i] == 0) i += 1;
            if(colSum[j] == 0) j += 1;
        }
        return a;
    }
};
