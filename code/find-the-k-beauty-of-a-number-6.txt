class Solution {
public:
    int divisorSubstrings(int num, int k) {
        int count = 0;
        
        string s = to_string(num);
        
        for (int i = 0; i <= s.size() - k; i++){
            int divisor = 0;
            
            for (int j = 0; j < k; j++)
                divisor = divisor * 10 + (s[j + i] - '0');
            
            if (divisor != 0 && num % divisor == 0)
                count++;
        }
        
        return count;
    }
};
