char * removeStars(char * s){
    
    int len = strlen(s);
    char ans[len];
    int j = 0;
    int i;
    // i is the counter for s string, while j is counter for ans string
    for(i = 0; i < len; i++){
        if(s[i]=='*'){          //this is the equivalent of string pop() function
            j-=1;
            ans[j] = '\0';
        }
         else{
             ans[j] = s[i];     //append characters to ans string
             j+=1;
         }
    }
    memset(s,0,strlen(s));      //clear s string
    strncpy(s, ans, j);         // and then copy ans string to s before returning s
    return s;
}
