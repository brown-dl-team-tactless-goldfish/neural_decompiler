int mostWordsFound(char ** sentences, int sentencesSize)
{
    int i, j, res=0, c ;
    for(i=0;i<sentencesSize;i++)
    {
        c=0;
        for(j=0;sentences[i][j]!='\0';j++)
        {
            if(sentences[i][j]==' ')
            {
                c++;
            }
        }
        c=c+1;
        if(c>res)
        {
            res=c;
        }
    }
    return res;
}
