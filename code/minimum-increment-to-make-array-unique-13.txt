 public class Solution {
    public int MinIncrementForUnique(int[] A) {
			 Array.Sort(A);

            var result = 0;

            for(var i = 1; i < A.Length; i++)
            {
                if(A[i] <= A[i - 1])
                {
                    result += A[i - 1] - A[i] + 1;
                    A[i] = A[i - 1] + 1;
                }
            }

            return result;
    }
}
