vector<int> findColumnWidth(vector<vector<int>>& grid) {
	vector<int> res(grid[0].size(), 1);
	for (int j = 0; j < grid[0].size(); j++) {
		for (int i = 0; i < grid.size(); i++) {
			int curLen = 0, num = grid[i][j];
			if (num < 0) {
				num *= -1;
				curLen++;
			}
			while (num) {
				num /= 10;
				curLen++;
			}
			res[j] = max(res[j], curLen);
		}
	}
	return res;
}
