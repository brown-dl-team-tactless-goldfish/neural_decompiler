    sort(arr.begin(),arr.end());
    
    int n = arr.size(), min_dif = INT_MAX;
    
    vector<vector<int>> ans;
    
    for(int i=1; i<n; i++)
    {
        min_dif = min(min_dif, arr[i] - arr[i-1]);
    }
    
    for(int i=1; i<n; i++)
    {
        if(min_dif == arr[i] - arr[i-1])
        {
            ans.push_back({arr[i-1], arr[i]});
        }
    }
    
    return ans;
    
}
