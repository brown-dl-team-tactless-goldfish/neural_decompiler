    vector<bool> camelMatch(vector<string>& queries, string pattern) {
        vector<bool> res;
        string p = "[a-z]*";
        for (char &c: pattern)
            p += string(1, c) + "[a-z]*";
        regex r(p);
        for (string &query : queries)
            res.push_back(regex_match(query, r) );
        return res;
    }
