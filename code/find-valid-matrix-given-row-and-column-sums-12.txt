class Solution {
public:
    vector<vector<int>> restoreMatrix(vector<int>& rowSum, vector<int>& colSum) {
     int r = rowSum.size();
     int c = colSum.size();
        
     vector<vector<int>> result(r, vector<int>(c, 0));
        
     for(int i = 0; i < r; i++) 
     {
         for(int j = 0; j < c; j++)
         {
             int k = min(rowSum[i], colSum[j]);
             rowSum[i] -= k;
             colSum[j] -= k;
             result[i][j] = k;
         }
     }
     return result;
    }
};
