int longestSubarray(int* nums, int numsSize){
    int n = numsSize;
    int zeroCn = 0;
    int max = 0;
    int l = 0;
    for(int r = 0; r < n; r++){
        if(nums[r] == 0)
            zeroCn++;
        while(zeroCn > 1){
            if(nums[l] == 0)
                zeroCn--;
            l++;
        }
        
        max = fmax(max, r -l);
    }
    return max;
}
