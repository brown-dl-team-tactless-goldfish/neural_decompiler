class Solution {
public:
    using VI = vector<int>;
    bool isGoodArray(VI& A, bool good = false) {
        auto x = A.back(); A.pop_back();
        for_each(A.begin(), A.end(), [&](auto y) { x = gcd(x, y); });
        return x == 1;
    }
private:
    int gcd(int a, int b) { return !b ? a : gcd(b, a % b); }
};
