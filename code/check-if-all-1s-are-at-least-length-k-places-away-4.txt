bool kLengthApart(int* nums, int numsSize, int k){
    
    int i,j=-1000000,sum=0;
    
    for(i=0;i<numsSize;i++)
    {
        if(nums[i]==1)
        {
            if(i-j<=k)
            {
                sum=1;
                break;
            }
            j=i;
            
        }
    }
    
    if(sum==0)
    {
        return true;
    }
    else
    {
        return false;
    }

}
