class NestedIterator {
public:
    vector<int> res;
    int n,c=0;
    NestedIterator(vector<NestedInteger> &nl) {
        helper(nl);
        n = res.size();
    }
    void helper(vector<NestedInteger>& nl){
        for(auto x: nl)
            if(x.isInteger()) res.push_back(x.getInteger());
            else helper(x.getList());
        return;
    }
    int next(){
        return res[c++];
    }
    bool hasNext() {
        return c < n;
    }
};
