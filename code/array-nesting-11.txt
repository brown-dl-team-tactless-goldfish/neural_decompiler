int arrayNesting(vector<int>& a) {
        int n = a.size(), ans = INT_MIN;
        vector<int> v(n,0);
        for(int i: a) {
            int count = 0;
            while(!v[i]) {
               v[i] = true;
               i = a[i];
               count++;
            }
            ans = max(ans, count);
        }
        return ans;
}
