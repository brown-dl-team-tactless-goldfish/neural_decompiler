    /* Factorial */
    long factorial(long n) { return (n == 1 || n == 0) ? 1 : (n * factorial(n - 1)); }
    
    long combination1(int n, int r) {
        if (l[n][r]) return l[n][r];
        /* Formula for nCr = n! / r! * (n - r)! */
        l[n][r] = factorial(n) / (factorial(r) * factorial(n - r));    /* memoize */
        return l[n][r];
    }
