
int chk(int num) {
    for (int i = 0; num != 0; i++, num = num / 10) {
        if (!(num % 10)) {
            return 1;
        }
    }
    return 0;
}

int* getNoZeroIntegers(int n, int* returnSize) {

    int* ret = malloc(sizeof(int) * 2);
	ret[0] = 0;
    ret[1] = 0;
    *returnSize = 2;

    for (int i = 1; i <= n / 2; i++) {
        ret[0] = i;
        ret[1] = n - i;
        if (!chk(ret[0]) && !chk(ret[1])) {
            break;
        }
    }
    return ret;
}
