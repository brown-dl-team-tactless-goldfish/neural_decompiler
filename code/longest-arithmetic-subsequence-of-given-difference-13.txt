class Solution {
public:
    int longestSubsequence(vector<int>& arr, int difference) {
        int res = 0;
        unordered_map<int, int> seqs;
        for (int i = 0; i < arr.size(); ++i)
        {
            seqs[arr[i]] = seqs[arr[i] - difference] + 1;
            res = max(res, seqs[arr[i]]);
        }
        
        return res;
    }
};
