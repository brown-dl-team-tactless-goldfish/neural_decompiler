Dictionary<int, int> map = new Dictionary<int, int>();
	int n = nums.Length;
	for (int i = 0; i < n; i++)
	{
		if (nums[i] % 2 == 0)
		{
			if (map.ContainsKey(nums[i]))
				map[nums[i]]++; // increment value depend on key
			else
				map.Add(nums[i], 1); // add by one 
		}
	}
	var result = 0;
	var comparator = 0;
	foreach (var entry in map)
	{
		if (entry.Value > comparator)
		{
			comparator = entry.Value;
			result = entry.Key;
		}
		if (entry.Key < result && entry.Value == comparator)
			result = entry.Key;
	}
	if (comparator == 0)
		return -1;
	return result;
