class Solution {
public:
    int minOperations(vector<int>& nums) {
        int ret = 0;
        int m = 0;
        for(int x : nums) {
            int t = 0;
            while(x){
                if(x&1){
                    x ^= 1; 
                    ret++;
                } else {
                    x >>= 1;
                    t++;
                }
            }
            m = max(m,t);
        }
        ret += m; 
        return ret; 
    }
};
