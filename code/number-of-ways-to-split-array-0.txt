int waysToSplitArray(int* nums, int numsSize){
    int n = numsSize;
    long long* presum = malloc(n * sizeof( long long));
    
    presum[0] = nums[0];
    for(int i = 1; i < n; i++){
        presum[i] = presum[i-1] + nums[i];
    }
    long long half = presum[n-1]/2;
    if(presum[n-1] %2){
        if(presum[n-1] > 0)
            half++;
    }
    int ans = 0;
    for(int i = 0; i <n-1;i++){
        if(presum[i] >= half)
            ans++;
    }
    return ans;
}
