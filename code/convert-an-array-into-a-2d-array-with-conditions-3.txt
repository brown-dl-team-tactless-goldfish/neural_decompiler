vector<vector<int>> findMatrix(vector<int>& nums) {
	vector<int> items(201, 0);
	for (const int& n : nums)
		items[n]++;
	vector<vector<int>> res;
	while (true) {
		vector<int> row;
		for (int i = 0; i < items.size(); i++)
			if (items[i]-- > 0)
				row.push_back(i);
		if (row.size() > 0)
			res.push_back(row);
		else
			break;
	}
	return res;
}
