class Solution:
    def getNoZeroIntegers(self, n: int) -> List[int]:
        # print("n = ", n)

        for i in range(n):
            tmp1 = i
            tmp2 = n-i

            s1 = str(tmp1)

            # print("s1 = ", s1)
            if ("0" in s1):
                continue

            s2 = str(tmp2)
            # print("s2 = ", s2)
            if ("0" in s2):
                continue
            else:
                return [tmp1, tmp2]
