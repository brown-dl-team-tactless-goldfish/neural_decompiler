int maximumUniqueSubarray(int* nums, int numsSize){
    
    int max = 0;
    int a [10001];
    memset(a, 0 , sizeof(int)*10001);
    int sum = 0;
    int j = 0;
    for (int i = 0; i < numsSize; i++) {
        sum += nums[i];
        a[nums[i]]++;
        while (a[nums[i]] > 1) {
            a[nums[j]]--;
            sum -= nums[j];
            j ++;
        }
        max = max < sum ? sum : max;
    }
    
    return max;
    
}
