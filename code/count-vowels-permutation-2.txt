int countVowelPermutation(int n) {
    int m = 1000000007, a = 1, e = 1, i = 1, o = 1, u = 1, tempO = 0, tempU = 0, tempA = 0;
    for (int j = 1; j < n; j++) {
        tempO = o;
        o = i;
        i = (tempO + e) % m;
        tempU = u;
        u = (o + tempO) % m;
        tempA = a;
        a = ((e + o) % m + tempU) % m;
        e = (tempA + o) % m;
    }
    return ((((a + e) % m + i) % m + o) % m + u) % m;
}
