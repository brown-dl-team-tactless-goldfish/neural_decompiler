public class Solution
{
    public long WaysToBuyPensPencils(int total, int costPen, int costPensil) 
    {
        long res = 0;
        var pens = 0;
        while(costPen * pens <= total)
        {
            res++;
            res += (total - costPen * pens) / costPensil;
            pens++;
        }
        return res;
    }
}
