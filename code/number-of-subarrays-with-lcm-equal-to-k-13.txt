class Solution {
public:
    int subarrayLCM(vector<int>& nums, int k) {
        int n = nums.size();

        int count = 0;
        for(int i=0;i<n;i++){
            int lcm = 1;
            for(int j=i;j<n;j++){
                lcm = (lcm * nums[j])/(__gcd(lcm, nums[j]));
                if(lcm == k)
                    count++;

                //  Once lcm becomes greater than k, we know that any larger          .
                //  subarrays containing all the current elements will not work
                // bcoz lcm will more greater as subarray size increases
                if(lcm > k)
                    break;
            }
        }
        return count;
    }
};
