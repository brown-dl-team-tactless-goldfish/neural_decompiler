int numOfSubarrays(int* arr, int arrSize, int k, int t)
{  
 int i=0,j=0,count=0,sum=0;
    
 while(i<k)
 {
     sum+=arr[i++];
 }  
 count=sum/k>=t?count+1:count;  
        
 while(i<arrSize)
 {
     sum-=arr[j++];
     sum+=arr[i++];
     count=sum/k>=t?count+1:count;   
 }      
  return count;          
 }
