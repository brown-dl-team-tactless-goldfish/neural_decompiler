class Solution {
public:
    int numEquivDominoPairs(vector<vector<int>>& dominoes) {
        
        vector<int> m (100, 0);
        
        int ans = 0;
        
        for (const auto& d : dominoes)
        {
            int k1 = d[0] * 10 + d[1];
            int k2 = d[1] * 10 + d[0];
            
            ans += m[k1];
            
            if (k1 != k2) ans += m[k2];
            
            ++m[k1];
        }
        
        return ans;
        
    }
};
