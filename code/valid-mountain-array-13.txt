bool validMountainArray(int* arr, int arrSize){
    bool inc=true;
    int i;
    if (arrSize == 1) return false;
    for(i=1; i<arrSize; i++) {
        
        if (arr[i] == arr[i-1]) return false;
        else {
            if (arr[i] > arr[i-1]) {
                if (!inc) return false;
            }
            else {
                if (i==1) return false;
                inc = false;
            }
        }
    }
    if (inc) return false;
    return true;
}
