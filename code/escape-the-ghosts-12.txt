class Solution {
public:
    bool escapeGhosts(vector<vector<int>>& ghost, vector<int>& target) {
        int x=target[0],y=target[1];
        int dist=abs(x)+abs(y);
        for(auto it:ghost){
            int temp=abs(it[0]-x)+abs(it[1]-y);
            if(temp<=dist)return false;
        }
        return true;
    }
};
