bool checkInclusion(char * s1, char * s2){
    int c1set[26] = {0}, c2set[26] = {0};

    char *j, *i;
    for (i = s1, j = s2; *i && *j; ++i, ++j) 
    {
        ++c1set[*i - 'a'];
        ++c2set[*j - 'a'];
    }

    if (*i && !(*j)) return false;
    if (!memcmp(c1set, c2set, sizeof(int) * 26)) return true;

    for (; *j; ++s2, ++j)
    {
        --c2set[*s2 - 'a'];
        ++c2set[*j - 'a'];
        if (!memcmp(c1set, c2set, sizeof(int) * 26)) return true;
    }

    return false;
}
