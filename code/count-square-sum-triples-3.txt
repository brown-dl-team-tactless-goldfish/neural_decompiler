class Solution:
    def countTriples(self, n: int) -> int:
        data = {}
        
        for i in range(1, n+1):
            data[i] = i * i
            
        print ("data = ", data)
        
        values = data.values()
        print ("values = ", values)
        
        res = 0
        
        for target in values:
            for n1 in values:
                if (target < n1):
                    break
                if (target - n1 in values):
                    res += 1
                    
        return res
