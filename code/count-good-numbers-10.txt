#define mod 1000000007
class Solution {
    long long help(long long x,long long y)//return x^y
    {
        if(y==0)return 1;
        long long ans=help(x,y/2);
        ans*=ans;
        ans%=mod;
        if(y%2)ans*=x;//for odd case 
        ans%=mod;
        return ans;
    }
public:
    int countGoodNumbers(long long n) {
       long long odd=n/2;
        long long even=n/2+n%2;
        return (help(5,even) *help(4,odd))%mod;
        
    }
};
