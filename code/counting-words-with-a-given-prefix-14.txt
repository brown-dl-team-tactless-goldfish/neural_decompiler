class Solution {
public:
    int prefixCount(vector<string>& words, string pref) {
        int ans = 0 , n = pref.length();        
        for(auto word : words){
            if(word.length()>=n){
                if(word.substr(0,n) == pref)
                    ans++;
            }
        }
        return ans; 
    }
};
