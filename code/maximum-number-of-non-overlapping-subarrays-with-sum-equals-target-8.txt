class Solution {
public:
  int maxNonOverlapping(vector<int>& v, int t) {
        unordered_map<long long int,int>mp;
        mp[0]=1;
        long long int sum=0; int ans=0;
        for(int i=0;i<v.size();i++){
            sum+=v[i];
            int req=sum-t;            
            if(mp.count(req)){
                ans++;
                sum=0;
                mp.clear();
            }
            mp[sum]=1;
        }
        return ans;
    }
};
