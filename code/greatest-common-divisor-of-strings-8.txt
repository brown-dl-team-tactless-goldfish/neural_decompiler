class Solution {
public:
    string gcdOfStrings(string a, string b) {
        while(a + b == b + a)
        {
            if(a==b)
                return a;
            if(a.size()>b.size())
                a=a.substr(b.size());
            if(a.size()<b.size())
                b=b.substr(a.size());
        }
        return "";
    }
};
