int arithmeticTriplets(int* nums, int numsSize, int diff){
   
    int* numPtr=nums;
    int cnt=1, total=0;
    for(int i=0; i<numsSize-2; i++){
        numPtr=nums+i+1;
        cnt=1;
            while(numPtr!=nums+numsSize){                
                if(*numPtr-nums[i]==diff*cnt) cnt++;
                if(cnt==3){
                    total++;
                    break;
                }
                *numPtr++;
            }   
    }
    return total;
}
