class Solution {
public:
    stack<string>st;
    int minOperations(vector<string>& logs) {
        int count=0;
        for(auto &i:logs){
            if(i=="../" && !st.empty()){
                st.pop();
            }
            else if(i=="./"){
                
            }
            else{
                if(i!="../")
                st.push(i);
            }
        }
        while(!st.empty()){
            count++;
            st.pop();
        }
        return count;
    }
};
