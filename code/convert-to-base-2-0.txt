char * baseNeg2(int n){
    if(n == 0)
        return "0";
    char* ans = malloc(64 * sizeof(char));
    int idx = 0;
    int base = 1;
    while (n != 0) {
        int rem = n % -2;
        n /= -2;
        if (rem < 0) rem += 2, n += 1;
        ans[idx] = rem + '0';
        idx++;
    }
    for(int i = 0; i < idx/2; i++){
        char tmp = ans[i];
        ans[i] = ans[idx-1-i] ;
        ans[idx-1-i] = tmp;
    }
    ans[idx] ='\0';
    return ans;
}
