    int numTilePossibilities(string tiles) {
		unordered_map<char, int> hmap;
    
		for (auto& e : tiles)
			hmap[e]++;
    
		helper(tiles, 0, hmap);
    
		return ans - 1;
	}

void helper(string tiles, int size, unordered_map<char, int>& hmap) {
    if (size > tiles.size())
        return;
    
    ans++;
    
    for (auto& c : hmap) {
        if (c.second > 0 ) {
            c.second--; 
            helper(tiles, size + 1, hmap); 
            c.second++;
        }
    }
}
