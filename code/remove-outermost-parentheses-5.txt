char * removeOuterParentheses(char * S){
    char* slow = S;
    char* fast = S;
    int   stck = 0;
    
    while (*fast) {
        *slow = *fast;
        
        if (*fast == '(') {
            slow += (stck != 0);
            stck += 1;
        }
        else {
            stck -= 1;
            slow += (stck != 0);
        }
        
        ++fast;
    }
    *slow = NULL;
    return S;
}
