int maximumDifference(int* nums, int nums_len) {
  int max_diff = INT_MIN;
  --nums_len;
  for (short i = 0; i < nums_len; ++i) {
    for (short j = i + 1; j <= nums_len; ++j) {
      int diff = nums[j] - nums[i];
      if (diff > max_diff) {
        max_diff = diff;
      }
    }
  }
  if (max_diff <= 0) {
    return -1;
  }
  return max_diff;
}
