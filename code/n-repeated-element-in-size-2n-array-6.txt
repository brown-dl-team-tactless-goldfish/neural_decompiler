#define MAX_NUM 10001
int repeatedNTimes(int* A, int ASize){
    
    int hashMap[MAX_NUM] = {0};
    int index, maxValue = 0;
    
    for(index = 0; index < ASize; index++)
    {
        hashMap[A[index]]++;
        if(hashMap[A[index]] == ((ASize)/2))
           break;
    }
    return A[index];
}
