bool canFormArray(vector<int>& arr, vector<vector<int>>& pieces) {
	for (auto pIt : pieces) {
		auto arrIt = find(arr.begin(), arr.end(), pIt[0]);
		if (arrIt != arr.end()) {
			for (auto p : pIt) {
				if (p != *arrIt)
					return false;
				arrIt++;
			}
		}
		else
			return false;
	}
	return true;
}
