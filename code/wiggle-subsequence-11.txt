
#define UND -1
#define NEG 0
#define POS 1

int wiggleMaxLength(int* nums, int numsSize) {
    if(numsSize < 2) return 1;
    int dnums;
    int count = 1;
    char sign_fg = UND;

    for(int i=1;i<numsSize;i++) {
        dnums = nums[i] - nums[i-1];
        if(dnums > 0 && sign_fg != POS) {
            sign_fg = POS;
            count++;
        }
        else if(dnums < 0 && sign_fg != NEG) {
            sign_fg = NEG;
            count++;
        }

    }

    return count;
}
