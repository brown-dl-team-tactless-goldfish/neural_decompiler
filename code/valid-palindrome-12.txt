bool isPalindrome(char* s) {
    int len=strlen(s)-1;
    if(len<1){return true;}
    int i=0;
    while(i<len){
        while((s[i]<'A'&&s[i]>'9')||(s[i]>'Z'&&s[i]<'a')||s[i]>'z'||s[i]<'0'){
            i++;
            if(i>=len){return true;}
        }
        while((s[len]<'A'&&s[len]>'9')||(s[len]>'Z'&&s[len]<'a')||s[len]>'z'||s[len]<'0'){
            len--;
            if(i>=len){return true;}
        }
        if(s[i]<='9'&&s[i]!=s[len]){
            return false;
        }
        else if(s[i]!=s[len]&&s[i]!=s[len]+32&&s[i]!=s[len]-32){
            return false;
        }
        i++;
        len--;
    }
    return true;
}
