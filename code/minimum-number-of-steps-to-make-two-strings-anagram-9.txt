

int minSteps(char * s, char * t){
    int map_s[26] = { 0 };
    int map_t[26] = { 0 };
    int i = 0;
    int max_map_index = 0;
    while (s[i]) {
        map_s[s[i] - 'a']++;
        map_t[t[i] - 'a']++;  
        i++;
    }
    int num_steps = 0;
    i = 0;
    while (i < 26) {
        if (map_s[i] < map_t[i]) {
            num_steps += map_t[i] - map_s[i];
        }
        i++;
    }
    return num_steps;
}
