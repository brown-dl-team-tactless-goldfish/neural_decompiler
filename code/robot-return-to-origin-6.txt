bool judgeCircle(char* moves) {
    int x = 0;
    int y = 0;
    int len = strlen(moves);
    int i;
    for(i = 0;i<len;i++){
        if(moves[i]=='L'){
            x--;
        }
        if(moves[i]=='R'){
            x++;
        }
        if(moves[i]=='U'){
            y++;
        }
        if(moves[i]=='D'){
            y--;
        }
    }
    if(x==0&&y==0){
        return true;
    }
    return false;
}
