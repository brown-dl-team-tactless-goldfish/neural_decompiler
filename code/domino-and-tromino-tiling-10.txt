class Solution {
public:
    int numTilings(int n) {
        int mod = 1000000007;

       //https://www.youtube.com/watch?v=7cijrfUkQzc

        int a =1 , b =2 , c =5 , currAns ;
        if(n<3) return n;
        if(n==3) return 5;

        for(int i=4;i<=n;i++){
           currAns = ((2*c)%mod + a % mod) % mod;
           a = b;
           b = c;
           c = currAns;   
        }
        return currAns;
    }
};
