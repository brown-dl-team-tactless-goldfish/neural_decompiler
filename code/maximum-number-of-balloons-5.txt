int maxNumberOfBalloons(char * text){
    int txtcnt[26]={0},count=0;
    int len=strlen(text);
    for(int i=0;i<len;i++)
        txtcnt[text[i]-'a']++;
    
    while(1)
    {
        txtcnt[0]-=1;   //a
        txtcnt[1]-=1;   //b
        txtcnt[11]-=2;  //l
        txtcnt[13]-=1;  //n
        txtcnt[14]-=2;  //o
        if(txtcnt[0]>=0&&txtcnt[1]>=0&&txtcnt[11]>=0&&txtcnt[13]>=0&&txtcnt[14]>=0)
            count++;
        else 
            break;
    }
    return count;
}
