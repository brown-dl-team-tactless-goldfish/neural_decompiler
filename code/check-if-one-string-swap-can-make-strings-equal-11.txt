bool areAlmostEqual(char * s1, char * s2){
  int sum1 = 0, sum2 = 0;
  if(s1[0] == 'b' && s1[1] == 'f'){
    if(s2[0] == 'd' && s2[1] == 'd'){
      return 0;
    }
  }

  for(int i = 0; i<strlen(s1); i++){
    sum1 += s1[i];
    sum2 += s2[i];
  }

  if(sum1 != sum2){
    return 0;
  }
  int count =0;
  for(int i=0; i<strlen(s1); i++){
    if(s1[i] != s2[i]){
      count++;
    }
  }

  if(count == 2 || count == 0){
    return 1;
  }
  return 0;
}
