int candy(const int* ratings, const int N){
   int ls[N];
   int rs[N];
     ls[0] = 1;
     rs[N - 1] = 1;
    for (int i = 1; i < N; i++) 
        ls[i] = (ratings[i] > ratings[i-1]) ? ls[i-1] + 1 : 1;
    
    for (int i = N-2; i >= 0; i--) 
        rs[i] = (ratings[i] > ratings[i+1]) ? rs[i+1] + 1 : 1;
    
    
    int ans = 0;
    for (int i = 0; i < N; i++) 
        ans += (ls[i] > rs[i]) ? ls[i] : rs[i];
    
       
   return ans;
}
