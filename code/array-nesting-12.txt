class Solution {
public:
    int arrayNesting(vector<int>& nums) {
        int m = 0;
        
        for (int i = 0; i < nums.size(); ++i) {
            int c = 0;
            
            for (int j = i; nums[j] != -1; ++c) {
                int t = j;
                j = nums[j];
                nums[t] = -1;
            }
            
            m = max(m, c);
        }
        
        return m;
    }
};
