int differenceOfSum(int* nums, int numsSize){
int elementSum = 0, digitSum = 0;
for(int i = 0; i < numsSize; i++){
elementSum += nums[i]; // add the current element to the element sum
int currentNum = nums[i]; // store the current element in a variable for easier manipulation
while(currentNum > 0){
digitSum += currentNum % 10; // add the last digit of the current element to the digit sum
currentNum /= 10; // remove the last digit of the current element
}
}
return abs(elementSum - digitSum); // return the absolute difference between the element sum and digit sum
}
