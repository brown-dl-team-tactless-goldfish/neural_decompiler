class Solution {
public:
    int maximumTop(vector<int>& nums, int k) {
        if(nums.size() == 1) return k % 2 ? -1 : nums[0];
        if(k == 0) return nums[0];
        int mx = 0, j = 0;
        while(k > 1 && j < nums.size()) {
            mx = max(nums[j], mx);
            j++;
            k--;
        }
        if(j + 1 < nums.size() && nums[j + 1] > mx) return nums[j + 1];
        return mx;
    }
};
