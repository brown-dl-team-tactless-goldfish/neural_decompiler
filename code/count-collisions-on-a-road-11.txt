class Solution {
public:
    int countCollisions(string d) {
        int n = d.size(), res = n;
        for(int i = 0; i < n; i++){
            if(d[i] == 'L') res--;
            else break;
        }
        for(int i = n-1; i >= 0; i--){
            if(d[i] == 'R') res--;
            else break;
        }
        for(auto c : d) if(c == 'S') res--;
        return res;
    }
};
