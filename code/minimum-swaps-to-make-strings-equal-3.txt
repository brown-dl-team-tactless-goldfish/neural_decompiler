class Solution {
public:
    int minimumSwap(string A, string B, int a=0, int b=0) {
        for (auto i=0; i < A.size(); ++i)
            if (A[i] != B[i])
                a += A[i] == 'x',
                b += B[i] == 'x';
        return ((a + b) % 2) ? -1 : (a / 2) + (b / 2) + ((a % 2) + (b % 2));
    }
};
