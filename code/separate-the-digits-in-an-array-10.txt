class Solution {
public:
    vector<int> separateDigits(vector<int>& nums) {
        vector<int> res;
        string num = "";
        for(int i = 0; i < nums.size(); i++){
            num = to_string(nums[i]);
            for(int j = 0; j < num.size(); j++){
                res.push_back(num[j] - '0');
            }
        }
        return res;
    }
};
