class Solution {
public:
    bool checkZeroOnes(string s) {
        int z = 0, fz = 0, c = 0, fc = 0;
        for(int i=0;i<s.size();i++){
            if(s[i]=='1'){
                c++;
                z = 0;
            }
            else{
                z++;
                c = 0;
            }
            fz = max(fz,z);
            fc = max(fc,c);
        }
        return fc>fz;
    }
};
