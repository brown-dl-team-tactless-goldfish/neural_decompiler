bool isValid(char * s){
    int n = strlen(s) ;
    if(n % 3)
        return false ;
    char* stk = malloc(n * sizeof(char)) ;
    int s_idx = 0 ;
    for(int i = 0; i < n; i++){
        if(s[i] == 'c'){
            if(s_idx >= 2 && stk[s_idx-1] == 'b' && stk[s_idx-2] == 'a')
                s_idx -= 2 ;
            else
                return false ;
        }
        else{
            stk[s_idx] = s[i] ;
            s_idx++ ;
        }
    }

    if(s_idx > 0)
        return false ;
    return true ;

}
