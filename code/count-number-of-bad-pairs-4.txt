    long long countBadPairs(vector<int>& nums) {
        long long count=0;
        unordered_map<int,int>mp;
        for(int i=nums.size()-1;i>=0;i--)
        {
            auto it=mp.find(nums[i]-i);
            if(it!=mp.end())
                count+=nums.size()-i-1-it->second;
            else
                count+=nums.size()-i-1;
            mp[nums[i]-i]++;
        }
        return count;
    }

Time Complexity: O(n)
Space Complexity: O(n)
