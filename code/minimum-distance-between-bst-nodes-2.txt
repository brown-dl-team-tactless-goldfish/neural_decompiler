int arr[100]={0};
void travel(struct TreeNode* root,int* idx)
{
    if(!root)
        return;
    travel(root->left,idx);
    arr[(*idx)++]=root->val;
    travel(root->right,idx);
    return;
}

int minDiffInBST(struct TreeNode* root){
    int min=INT_MAX;

    int idx=0;
    travel(root,&idx);

    for(int i=1;i<idx;i++)
    {
        if(arr[i]-arr[i-1]<min)
            min=arr[i]-arr[i-1];
    }
    return min;
}
