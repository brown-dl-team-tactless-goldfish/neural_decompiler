int maxDistance(int* colors, int colorsSize)
{
int curdistance = 0;
int finaldistance = 0;

   for(int i=0;i<colorsSize;i++)
   {
       for(int j=colorsSize-1;j>0;j++)
       {
           if(*(colors+i)!=*(colors+j))
           {
               curdistance=abs(i-j);
               if ( curdistance>finaldistance )
               {
                   finaldistance=curdistance;
               }
               break;
           }
       }
   }
    return( finaldistance );
}
