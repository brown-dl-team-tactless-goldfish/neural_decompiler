class Solution {
public:
    int partitionDisjoint(vector<int>& nums) {
        int globalMax = nums.at(0), localMax  = nums.at(0);
        int index = 1;
        for(int i = 1; i < nums.size(); i++){
            localMax = max(localMax, nums.at(i));
            if(nums.at(i) < globalMax){
                globalMax = localMax;
                index = i + 1;
            }
        }
        
        return index;
    }
};
