int findMin(int* A, int N){
    if(N == 1) return A[0];
    int start = 0, end = N-1, mid;
    while(start < end){
        mid = (start+end) / 2;
        if(mid > 0 && A[mid] < A[mid-1]) return A[mid];
        if(A[start] <= A[mid] && A[mid] > A[end]) start = mid + 1;
        else end = mid - 1;
    }
    return A[start];
}
