class Solution {
public:
    int sumEvenGrandparent(TreeNode* root) {
        if(!root) return 0;
        int sum = 0;
        calculate(NULL,NULL,root,sum);
        return sum;
        
    }
     void calculate(TreeNode* parent,TreeNode* grand,TreeNode* root,int &sum)
    {
        if(!root) return ;
        if(grand && grand->val % 2 ==0)
            sum+=root->val;
        calculate(root,parent,root->left,sum);
        calculate(root,parent,root->right,sum);
       
        
        
    }
};
