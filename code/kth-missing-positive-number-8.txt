

int findKthPositive(int* arr, int arrSize, int k){
    /*Data*/
    int* sss = (int*)malloc(sizeof(int)*1000);
    int y=0,i=0;
    int x=1;
    
    /*Start*/
    for(x=1;x<=1000;x++){
        sss[y]=x;
        if(arr[i]==sss[y]){
            i++;
            if(i==arrSize) break;
        }else{
            y++;
            if(y==k) break;
        }
    }
    for(;y<k;y++){
        sss[y]=++x;
    }
    return sss[y-1];
    
}
