class Solution {
public:
    int maximumGroups(vector<int>& grades) {
        int ans = 0;
        sort(grades.begin(),grades.end());
        int prev = -1, sz = 0, sum = 0;
        for(int i=0,j=0;j<grades.size();j++){
            sum += grades[j];
            if(sum > prev and j-i+1 > sz){
                ans++;
                prev = sum;
                sum = 0;
                sz = j-i+1;
                i = j+1;
            }
        }
        return ans;
    }
};
