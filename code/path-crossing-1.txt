int count = 0;
int i, x, y;
x = y = 0;
_coord[count].x = x;
_coord[count].y = y;
count++;

while(*path) {
    if(*path == 'N') { y++;}
    if(*path == 'S') { y--;}
    if(*path == 'W') { x--;}
    if(*path == 'E') { x++;}
    for(i = 0; i < count; i++) {
        if (_coord[i].x == x && _coord[i].y == y) {
            return true;
        }
    }
    _coord[count].x = x;
    _coord[count].y = y;
    count++;
    path++;
}
return false;
