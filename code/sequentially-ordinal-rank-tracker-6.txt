public class SORTracker {

    SortedList<int, SortedList<string, int>> sl;
    int qi = 1;
    public SORTracker() {
        sl = new SortedList<int, SortedList<string, int>>();
    }
    
    public void Add(string name, int score) {
        if (!sl.ContainsKey(score)) {
            sl.Add(score, new SortedList<string, int>());
            sl[score].Add(name, 1);
        } else {
            sl[score].Add(name, 1);
        }
    }

    public string Get() {
        int ith = qi++;
        string res = null;
        
        foreach (KeyValuePair<int, SortedList<string, int>> e in sl.OrderByDescending(x => x.Key)) {
            if (sl[e.Key].Count < ith) {
                ith -= e.Value.Count;
            } else {
                res = e.Value.Keys[ith - 1];
                break;
            }
        }
        
        return res;
    }
}
