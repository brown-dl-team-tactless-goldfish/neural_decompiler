class Solution {
public:
    int reductionOperations(vector<int>& a) {
        int c = 0;  //counts total operations
        int n = a.size();
        sort(a.begin(), a.end());
        for (int j = n - 1; j > 0; --j)
            if (a[j - 1] != a[j])
                c = c + n - j;
        return c;
    }
};
