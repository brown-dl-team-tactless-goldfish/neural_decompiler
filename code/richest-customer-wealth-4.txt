int maximumWealth(int** accounts, int accountsSize, int* accountsColSize){
    register unsigned char i, j, k;
    register unsigned int sum = 0, res = 0;
    for(i = 0; i < accountsSize; i++) {
        for(j = 0; j < *accountsColSize; j++)
            sum += accounts[i][j];
        if(sum > res)
            res = sum;
        sum = 0;
    }
    return res;
}
