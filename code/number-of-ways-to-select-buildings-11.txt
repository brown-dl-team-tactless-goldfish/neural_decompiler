public class Solution {
    public long NumberOfWays(string s) {
        long r = 0;
        var dp = new long[4];//0, 1, 01, 10
        foreach(char c in s){
            if (c == '0'){
                dp[0]++;
                dp[3] += dp[1];
                r += dp[2];
            } else {
                dp[1]++;
                dp[2] += dp[0];
                r += dp[3];
            }
        }
        return r;
    }
}
