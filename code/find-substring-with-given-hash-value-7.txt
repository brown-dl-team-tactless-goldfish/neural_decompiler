    string subStrHash(string s, int p, int m, int k, int h) {
        long n = s.size(), res = -1, sum = 0, sub = 1;
        for(int i = n-1; i >= 0; i--) {
            if(i + k > n) sub = (sub*p)%m;
            if(i < n-k) sum = ((sum-(s[i+k]-'a'+1)*sub)%m+m)%m;
            sum = (sum*p+s[i]-'a'+1)%m;
            if(i <= n-k && sum == h) res = i;
        }
        return s.substr(res, k);
    }
