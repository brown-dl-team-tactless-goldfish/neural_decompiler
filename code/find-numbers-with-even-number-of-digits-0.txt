int findNumbers(int* nums, int numsSize){
    int intSizeTable[] = {9, 99, 999, 9999, 99999, 999999, 9999999};
    int result = 0;
    for(int i = 0;i < numsSize; i++){
        int k = 0;
        for (k; ; k++) {
            if(nums[i]<=intSizeTable[k]){
                break;
            }
        }
        if((k+1)%2==0){
            result++;
        }
    }
    return result;
}
