    public string EntityParser(string text)
    {
        Dictionary<string, char> map = new Dictionary<string, char>()
            {
                {"&quot;", '\"'},
                {"&apos;", '\'' },
                {"&gt;", '>' },
                {"&lt;", '<' },
                {"&frasl;", '/' }
            };

        foreach (var p in map)
            text = text.Replace(p.Key, p.Value.ToString());

        return text.Replace("&amp;", "&"); ;
    }
