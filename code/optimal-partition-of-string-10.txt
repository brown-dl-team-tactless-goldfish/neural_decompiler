int partitionString(char * s)
{
    unsigned int bitmask = 0, num_substring = 0, new_bit_pos_mask = 0;

    for (int i = 0; i < strlen(s); i++)
    {
        new_bit_pos_mask = (1 << (s[i] - 'a'));

        if (bitmask & new_bit_pos_mask)
        {
            num_substring++;
            bitmask = 0;
        }
        bitmask |= new_bit_pos_mask;
    }

    if (bitmask)
        num_substring++;

    return num_substring;
}
