int firstMissingPositive(int* nums, int numsSize) {
    int i;
    for (i=0; i<numsSize; i++) {
        if (nums[i] <= 0)
            nums[i] = numsSize+1;
    }
    
    for (i=0; i<numsSize; i++) {
        int val = abs(nums[i]) -1;
        if (val < numsSize)  {
            if (nums[val] > 0) nums[val] *= -1;
        }

    }
    for (i=0; i<numsSize; i++) {
        if (nums[i] >= 0) return i+1;
    }
    return numsSize+1;
}
