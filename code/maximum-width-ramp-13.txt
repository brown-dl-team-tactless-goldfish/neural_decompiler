    int maxWidthRamp(vector<int>& b) {
        vector<pair<int,int>>v;
        int i,m,res=0;
        for(i=0;i<b.size();i++)
        {
            v.push_back({b[i],i});
        }
        sort(v.begin(),v.end());
        m=v[0].second;
        for(i=0;i<v.size();i++)
        {
            m=min(m,v[i].second);
            res=max(res,v[i].second-m);
        }
        return res;
    }
};
