class Solution {
public:
    int numOfStrings(vector<string>& patterns, string word) {
        int ans = 0;
        
        unordered_set<string>us;
        //here we have generated all the substring of the word
        for(int i = 0; i<word.size(); i++){
            string s = "";
            for(int j = i; j<word.size(); j++){
                s.push_back(word[j]);
                us.insert(s);
            }
        }
        
        //now we have just checked which are the patterns that are substrings in the word
        for(string x: patterns){
            if(us.find(x) != us.end()){
                ans++;
            }
        }
        
        return ans;
    }
};
