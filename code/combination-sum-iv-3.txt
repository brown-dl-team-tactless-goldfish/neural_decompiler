int combinationSum4(int* nums, int numsSize, int target) {
    int* dp=(int*)calloc(target+1,sizeof(int));
    dp[0]=1;
    for(int i=0;i<=target;i++){
        for(int j=0;j<numsSize;j++){
            if(i>=nums[j]){
               dp[i]+=dp[i-nums[j]];
            }
        }
    }
    return dp[target];
}
