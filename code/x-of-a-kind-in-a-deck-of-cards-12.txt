class Solution {
    int go( const int a, const int b ) const noexcept {  return ( b == 0 ) ? a : go( b, a % b );  }
public:
    bool hasGroupsSizeX( vector<int>& deck, unordered_map<int,int> m = {} ){
        for( const auto value: deck ) ++m[ value ];
        auto gcd = m.begin()->second; for( const auto& x: m ) gcd = go( gcd, x.second );
        return 1 < gcd;
    }
};
