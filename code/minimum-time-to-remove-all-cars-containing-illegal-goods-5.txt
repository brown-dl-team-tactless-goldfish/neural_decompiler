class Solution {
public:
    int minimumTime(string s) {
//         n-minsubarray sum
        int n=s.size();
        vector<int> res;
        for(int i=0;i<n;i++){
            if(s[i]=='1') res.push_back(1);
            else res.push_back(-1);
        }
        int mini=0;
        int sum=0;
        for(int i=0;i<n;i++){
            sum+=res[i];
            mini=min(mini,sum);
            if(sum>0) sum=0;
        }
        return n+mini;
    }
};
