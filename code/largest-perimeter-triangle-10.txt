int* comp(const void* a, const void* b){
    return (*(int*)a - *(int*)b);
}

int largestPerimeter(int* nums, int numsSize){
    qsort(nums, numsSize, sizeof(int), comp);
    int cur=0;
    for (int i=2; i<numsSize; i++)
        if (nums[i] < nums[i-1] + nums[i-2])
            cur = nums[i] + nums[i-1] + nums[i-2];
    
    return cur;
}
