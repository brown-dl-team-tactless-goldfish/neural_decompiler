// P = A + B + C
// A > B + C
// B > C + A
// C > A + B

// Time: O(n log n)
// Space: O(1)
var largestPerimeter = function (nums) {
    // O (n log n)
    nums.sort((a, b) => b - a); // largest to smallest

    // O(n)
    for (let i = 0; i <= nums.length - 3; i++) {
        if (nums[i] < nums[i + 1] + nums[i + 2]) {
            return nums[i] + nums[i + 1] + nums[i + 2];
        }
    }

    return 0;
};
