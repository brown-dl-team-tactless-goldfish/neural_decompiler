public class Solution {
    public int DistributeCoins(TreeNode root) {
        int[] numMoves = new int[1];
        
        TraverseTree(root, numMoves);
        
        return numMoves[0];
    }
    
    int TraverseTree(TreeNode node, int[] numMoves) {
        if (node == null) {
            return 0; 
        }
        
        int left = TraverseTree(node.left, numMoves);
        int right = TraverseTree(node.right, numMoves);
        
        numMoves[0] += Math.Abs(left) + Math.Abs(right);
        
        return node.val + left + right - 1;
        
    }
}
