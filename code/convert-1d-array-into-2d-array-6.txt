public class Solution {
    public int[][] Construct2DArray(int[] original, int m, int n) {
        
        if(m * n != original.Length)
            return new int[0][];
        
        int[][] res = new int[m][];
        for(int i = 0; i < m; i++)
            res[i] = new int[n];
        
        for(int i = 0; i < original.Length; i++)
        {
            res[i / n][i % n] = original[i];
        }
        
        return res;
    }
}
