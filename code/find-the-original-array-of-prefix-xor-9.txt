class Solution {
public:
    vector<int> findArray(vector<int>& pref) {
        
        int n = pref.size();
        vector<int> ans(n),cum(n);
        
        ans[0] = pref[0];
        cum[0] = pref[0];
        for(int i=1;i<n;i++)
        {
            ans[i] = cum[i-1]^pref[i];
           cum[i] = ans[i]^cum[i-1];
        }
        
        
        return ans;
    }
};
