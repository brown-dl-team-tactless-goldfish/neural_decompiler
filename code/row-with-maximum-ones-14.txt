class Solution {
public:
    vector<int> rowAndMaximumOnes(vector<vector<int>>& mat) {
        int n=mat.size(),m=mat[0].size();
        vector<int> ans(2,0);
        for(int i=0;i<n;i++){
            for(int j=1;j<m;j++){
                mat[i][j]+=mat[i][j-1];
            }
            if(mat[i][m-1]>ans[1]){
                ans[0]=i;
                ans[1]=mat[i][m-1];
            }
        }
        return ans;
    }
};
