//Approach-2 (Using C++ STL std::count_if)
class Solution {
public:
    int numOfStrings(vector<string>& patterns, string word) {
        
        auto lambda = [&word](string &s) {
            return word.find(s) != string::npos;
        };
        
        return count_if(begin(patterns), end(patterns), lambda);
    }
};
