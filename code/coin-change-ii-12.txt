public class Solution {        
    public int Change(int amount, int[] coins) {
        int[] cached = new int[amount + 1];
        cached[0] = 1;
        Array.ForEach(coins, c => { for (int j=1; j <= amount; j++) {
                                        if (j >= c) cached[j] += cached[j-c];
                                    }});
        return cached[amount];
    }
}
