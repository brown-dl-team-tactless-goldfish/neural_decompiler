int countKDifference(int* nums, int numsSize, int k){

    int count = 0;

    for (int index = 0; index < numsSize; index++)
    {
        for (int jIndex = 0; jIndex < numsSize; jIndex++)
        {
            if (((nums[index]-nums[jIndex])) == k)
            {
                count++;
            }
        }
    }
    return count;
}
