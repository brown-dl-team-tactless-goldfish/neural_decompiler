int maxPower(char * s){
    
    int i;
    char c = s[0];
    int count = 1;
    int max = 0;
    int len = strlen(s);
    for (i=1; i < len; i++)
    {
        if (c == s[i])
        {
            count++;
        }
        else
        {
            c = s[i];
            if (count > max)
                max = count;
             
            count=1;
        }
    }
    
    if (count > max)
        return count;
    else
        return max;

}
