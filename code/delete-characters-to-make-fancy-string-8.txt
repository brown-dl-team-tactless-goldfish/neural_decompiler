class Solution {
public:
    string makeFancyString(string s) {
        string ret;
        
        for(auto c : s){
            if(ret.size() >= 2 && ret.back() == ret[ret.size()-2] && ret.back() == c)
                continue;
            ret.push_back(c);
        }
        
        return ret;
    }
};
