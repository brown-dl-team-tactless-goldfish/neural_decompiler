int partitionString(char * s){
    int ans = 0, len = strlen(s);
    int bitmask = 0, offset;
    for(int i = 0; i < len; i++)
    {
        offset = s[i] - 'a';
        if(bitmask & (1 << offset)){
            ans++;
            bitmask = 0;
        }
        bitmask |= (1 << offset);
    }
    if(bitmask)
        ans++;
    return ans;
}
