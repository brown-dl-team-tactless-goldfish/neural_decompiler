int ordfunc(const void * a, const void * b)
{
    return (*(int*)a - *(int*)b);
}

int maximumBags(int* capacity, int capacitySize, int* rocks, int rocksSize, int additionalRocks){
    
    int cnt = 0;
    for(cnt = 0; cnt < capacitySize; ++cnt)
    {
        *(rocks+cnt) = *(capacity+cnt) - *(rocks+cnt);
    }
    
    qsort(rocks, rocksSize, sizeof(int), ordfunc);
    
    cnt = 0;
    
    while((cnt < capacitySize) && (*(rocks+cnt) - additionalRocks <= 0))
    {
        additionalRocks = additionalRocks - *(rocks+cnt);
        ++cnt;
    }
    
    return cnt;
}
