int numberOfArithmeticSlices(int* A, int ASize){
    int cnt=2,total=0;
    if(ASize<3) return 0;
    for(int i=0;i<ASize-2;i++)
    {
        if((A[i+1]-A[i])==(A[i+2]-A[i+1]))
            cnt++;
        else if(cnt>2)
        {
            total+=(1+cnt-2)*(cnt-2)/2;
            cnt=2;
        }
        else 
            cnt=2;
    
    }

    if(cnt>2&&(A[ASize-1]-A[ASize-2])==(A[ASize-2]-A[ASize-3]))
    {
        total+=(1+cnt-2)*(cnt-2)/2;
    }
    return total;
}
