    int miceAndCheese(vector<int>& A, vector<int>& B, int k) {
        int res = 0, N = A.size();
        vector<array<int, 2>> C;
        for(int i = 0; i < N; i++) C.push_back({B[i]-A[i], i});
        sort(begin(C), end(C));
        for(int i = 0; i < N; i++) res += i < k ? A[C[i][1]] : B[C[i][1]];
        return res;
    }
