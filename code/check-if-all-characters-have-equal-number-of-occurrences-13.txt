bool areOccurrencesEqual(char * s)
{

   int dic[26]={0},i,con;
    
    
    while(i<strlen(s))
    {
        dic[s[i]-'a']++;
        i++;
    }
    
    for(i=0;i<26;i++){
   
        if(dic[i]!=0){
        con=dic[i];
    }
    }
    for(i=0;i<26;i++){
        
        if(dic[i]!=con&&dic[i]!=0)
            return false;
    }
    return true;
}
