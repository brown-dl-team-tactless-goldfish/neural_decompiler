int maxAscendingSum(int* nums, int numsSize){
int i,msf=-9999,meh=nums[0],max=nums[0];
   for(i=1;i<numsSize;i++){  // using something similar to Kadane's algorithm 
       meh=meh+nums[i];
       if(nums[i-1]>=nums[i]){ // meh= max ending here - at the ith position 
           meh=nums[i];
       }  
       if(msf<meh){   
           msf=meh;   // msf= max so far - the largest value of the sum
       }
       if(max<nums[i]){ // extracting the max from the array in case of not existing any sorted subarray 
           max=nums[i];
       }
   }
  if(max<msf){return msf;}
  else{return max;} 
}
