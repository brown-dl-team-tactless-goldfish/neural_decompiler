class Solution {
public:
    string fractionAddition(string expression) {
        istringstream iss(expression) ;
        int A = 0, a = 1 ;
        int B, b ;
        char slash ;
        //A/a + B/b
        while(iss >> B >> slash >> b ){
            A = A*b + a*B ;
            a = a*b ;
        }
        int g = gcd(A, a) ;
        A /= g ;
        a /= g ;
        return to_string(A) + "/" + to_string(a) ;
    }
};
