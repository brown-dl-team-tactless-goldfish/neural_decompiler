bool isvowel(char ch)
{

    if(ch=='a'||ch=='e'||ch=='i'||ch=='o'||ch=='u')
        return true;
    return false;
}

int maxVowels(string s, int k) {
    int n=s.length();
    int a[n+1];
    a[0]=0;
    for(int i=1;i<=n;i++)
    {
        a[i]=(isvowel(s[i-1]))?1:0;
    }
    for(int i=1;i<=n;i++)
        a[i]+=a[i-1];
    
    int start=0,ans=0;
    for(int end=k;end<=n;end++)
    {
        ans=max(ans,a[end]-a[start]);
        start++;
    }
    return ans;
}
