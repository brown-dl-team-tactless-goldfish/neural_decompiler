class Solution {
public:
    vector<int> separateDigits(vector<int>& nums) {
        vector<int>v;
        for(int i=0; i<nums.size(); i++){
            int l=nums[i];
            vector<int>v1;
            while(l){
                v1.push_back(l%10);
                l/=10;
            }
            reverse(v1.begin(), v1.end());
            for(int j=0; j<v1.size(); j++){
                v.push_back(v1[j]);
            }
        }
        return v;
    }
};
