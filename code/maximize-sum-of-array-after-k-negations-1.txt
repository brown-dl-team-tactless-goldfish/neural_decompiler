int cmpfunc(const void* a, const void* b) {
    return *(int*)a - *(int*)b;
}

int largestSumAfterKNegations(int* nums, int numsSize, int k){
    qsort(nums, numsSize, sizeof(int), cmpfunc);

    int i = 0, sum = 0;

    while (k--) {
        nums[i] = -nums[i];

        if (i == numsSize - 1)
            continue;

        if (nums[i] > nums[i + 1])
            i++;
    }

    for (i = 0; i < numsSize; i++)
        sum += nums[i];

    return sum;
}
