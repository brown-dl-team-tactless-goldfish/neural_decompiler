

int minOperations(char ** logs, int logsSize)
{
    int count = 0;
    
    int i=0;
    for(i=0; i<logsSize; i++)
    {
        if(strcmp(logs[i], "../") == 0)
        {
            if(count != 0)
                count--;
        }
        else if(strcmp(logs[i], "./") == 0)
        {
            count = count;
        }
        else
        {
            count += 1;
        }
    }
    return count;
}
