class Solution {
public:
    vector<int> evenOddBit(int n) {
        int pos1=0, pos2=0;
        bool chk=1;
        while (n>0){
            if (chk){
                pos1+= (n&1);
            }
            else{
                pos2+= (n&1);
            }
            chk= !chk;
            n= n>>1;
        }
        return {pos1,pos2};
    }
};
