class Solution {
public:
    int triangleNumber(vector<int>& nums) {
	
        vector<int> &a = nums;
        int n = a.size();
        sort(a.begin(), a.end());
        int ans = 0;
        for(int i =0;i<n;i++) {
            for(int j = i+1;j<n;j++) {
                int t = a[i]+a[j];//t = a+b
                int L = j, R = n-1;
                //find maximum c=a[M] s.t. t=a+b and  a[M] < t <=a [M+1] by binary search
                while(L<=R) { 
                    int M = L+(R-L)/2;
                    if( (M==j||t>a[M]) &&  (M+1>=n||t<=a[M+1])) {
                        ans+=M-j;
                        break;
                    }  
                    if( (M==j||t>a[M]) ) L=M+1; else R=M-1;
                }
            }
        }
        return ans;
    }
};
