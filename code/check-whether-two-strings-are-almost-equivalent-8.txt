class Solution {
public:
    bool checkAlmostEquivalent(string a, string b) {
        vector<int> res(26,0);
        for (auto& c : a){
            int x = c - 97;
            res[x]++;
        }
        for (auto& d : b){
            int y = d - 97;
            res[y]--;
        }
        for (auto &e:res) {
            if (abs(e) > 3) return false;
        }
        return true;
    }
};
