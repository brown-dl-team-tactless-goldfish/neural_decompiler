class Solution {
public:
    int maximumGroups(vector<int>& grades) {
        int l = 1, r = grades.size()+1;
        int n = grades.size();
        while (l < r) {
            long mid = l + (r - l) / 2;
            if ((1+mid)*mid/2 > n) {
                r = mid;
            } else {
                l = mid + 1;
            }
        }
        return l - 1;
    }
};
