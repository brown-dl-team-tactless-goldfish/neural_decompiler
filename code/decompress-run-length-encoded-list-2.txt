
int* decompressRLElist(int* nums, int numsSize, int* returnSize){
    int i=0;	
    int count=0;// For result array size.
	
    // To get the size of result array.
    while(i<numsSize){
        count=count+nums[i];
        i=i+2;
    }
    
    int* result=(int*) malloc(sizeof(int)*count);
    
    i=0; // For how many times to be repeated.
    int j=1; // Which number to be repeated.
    int k=0; // To access result array.
    while(i<numsSize){
        int c=0; // Used so that number is repeated nums[i] number of times.
        while(c<nums[i]){
            result[k]=nums[j];
            k++;
            c++;
        }
        j=j+2;
        i=i+2;
    }
    *returnSize=count;
    return result;
}
