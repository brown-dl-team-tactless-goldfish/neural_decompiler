bool digitCount(char *num){
    int tmp[10];
    int n = strlen(num);
    for(int i=0; i<10; i++){
        tmp[i] = 0;
    }
    for (int i = 0; i < n; i++){
        tmp[num[i] - '0']++;
    }
    for (int i = 0; i < n; i++){
        if(num[i] - '0' != tmp[i]){
            return false;
        }

    }
    return true;
}
