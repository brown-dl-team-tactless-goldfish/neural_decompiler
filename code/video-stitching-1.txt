 public int VideoStitching(int[][] clips, int T) 
 {        
	var d = new int[T + 1];
	Array.Fill(d, int.MaxValue);
	d[0] = 0;

	for (int i = 1; i <= T; i++)
	{
		foreach (var c in clips)
		{
			int start = Math.Min(c[0], T);
			int end = Math.Min(c[1], T);    

			if (d[start] != int.MaxValue && start <= i && end >= i)
			{
				d[i] = Math.Min(d[i], d[start] + 1);
			}
		}
	}

	return d[T] == int.MaxValue ? -1 : d[T];
}
