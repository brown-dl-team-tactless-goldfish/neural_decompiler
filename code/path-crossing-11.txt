bool isPathCrossing(string path) {
	pair<int,int> cur { 0, 0 };
	map<pair<int,int>, int> mp;
	mp[cur]++;
	for (const char& c : path) {
		if (c == 'N')
			cur.second++;
		else if (c == 'S')
			cur.second--;
		else if (c == 'W')
			cur.first++;
		else
			cur.first--;
		if (++mp[cur] == 2)
			return true;
	}
	return false;
}
