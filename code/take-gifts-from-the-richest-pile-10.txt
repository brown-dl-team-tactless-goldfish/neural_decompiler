class Solution {
public:
    #define ll long long 
    long long pickGifts(vector<int>& gifts, int k) {
        ll res=0;
        priority_queue<int> pq;
        for(auto i:gifts)
            pq.push(i);
        for(int i=0;i<k;i++)
        {
            //remove max and push the floor of sqrt of number
            int top=pq.top();
            pq.pop();
            top=(int)sqrt(top);
            pq.push(top);
        }
        //find the sum
        while(!pq.empty())
        {
            res+=pq.top();
            pq.pop();
        }
        return res;
    }
};
