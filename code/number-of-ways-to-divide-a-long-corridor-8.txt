#define MOD 1000000007
#define ll long long
class Solution {
public:
    int numberOfWays(string corridor) {
        vector<int> index;
        for(int i=0; i<corridor.size(); ++i) if(corridor[i]=='S') index.push_back(i);
        
        if(index.size()&1 || index.size()==0) return 0; 
        ll cnt = 1;
        for(int i=2; i<index.size(); i+=2){
            cnt = (cnt * (index[i]-index[i-1])) % MOD;
        }
        return cnt;
    }
};
