int cmp(const void *x, const void *y)
{
    return (*(int*)x-*(int*)y);
}
int maxConsecutive(int b, int t, int* a, int n)
{
    if(t-b+1==n)return 0;
    int res=0;
    qsort(a, n, sizeof(int), cmp);
    if(a[0]-b>res)res=a[0]-b;
    for(int i=1;i<n;i++)
        if(a[i]-a[i-1]-1>res)res=a[i]-a[i-1]-1;
    if(t-a[n-1]>res)res=t-a[n-1];
    return res;
}
