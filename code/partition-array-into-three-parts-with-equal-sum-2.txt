bool canThreePartsEqualSum(int* arr, int arrSize){
  for (int i = 0; i < arrSize; i++)
  {
    s = s + arr[i];
  }

  d = s / 3;
  int se = 0, c = 0;
  if (s % 3 == 0)
  {
    for (int i = 0; i < arrSize; i++)
    {
      se = se + arr[i];
      if (se == d)
      {
        c++;
        se = 0;
      }
    }
  }

  return c >= 3;
}
