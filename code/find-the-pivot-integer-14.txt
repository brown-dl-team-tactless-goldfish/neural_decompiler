public class Solution 
{
    public int PivotInteger(int n)
    {
        var ls = Enumerable.Range(1,n).Select(i=>i).ToArray();
        var sum = ls.Sum();
        var c = 0;
        for(var  i=0; i<ls.Length; i++)
        {
            c += ls[i];
            if (2*c == sum + ls[i]) return i + 1;
        }
        Console.WriteLine(sum);
        return -1;
    }
}
