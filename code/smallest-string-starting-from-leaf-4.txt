class Solution {
public:
    string smallestFromLeaf(TreeNode* root) {
        if (!root) {
            return "";
        }
        char c = 'a' + root->val;
        std::string left = smallestFromLeaf(root->left);
        std::string right = smallestFromLeaf(root->right);
        if (left.empty() && !right.empty()) {
            return right + c;
        }
        if (right.empty() && !left.empty()) {
            return left + c;
        }
        return std::min(left, right) + c;
    }
};
