void deeplayercount(struct TreeNode* root,int* sum,int nowlayer, int* maxlayer)
{
	if(root==NULL)
	{
		return;
	}
	if(nowlayer>*maxlayer)
	{
		*sum=0;
		*maxlayer=nowlayer;
	}
	if(nowlayer== *maxlayer &&root->left==NULL && root->right==NULL)
	{
		*sum=*sum+root->val;
	}
	deeplayercount(root->left,sum,nowlayer+1,maxlayer);
	deeplayercount(root->right,sum,nowlayer+1,maxlayer);

}


int deepestLeavesSum(struct TreeNode* root){

int sum=0;
int maxlayer=0;
deeplayercount(root,&sum,0,&maxlayer);

return sum;
}
