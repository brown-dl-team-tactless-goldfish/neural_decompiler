class Solution 
{
public:
    int minTimeToType(string word) 
    {
        auto cur = 'a';
        auto res = 0;
        for (auto c : word)
        {
            auto distance = abs(cur - c);
            if (distance > 13)              // more than halfway around ? 
                distance = 26 - distance;   // go in the other direction
            res += distance + 1;
            cur = c;
        }
        return res;
    }
};
