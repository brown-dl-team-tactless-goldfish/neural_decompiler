    int countGoodStrings(int low, int high, int zero, int one) {
        const long long mod = 1e9+7;
        vector<long long> dp(high*2+1, 0);
        dp[0] = 1;
        for(int i=0;i<=high;i++){
            dp[i] %= mod;
            dp[i+one] += dp[i];
            dp[i+zero] += dp[i];
        }
        return accumulate(dp.begin()+low, dp.begin()+high+1, 0ll) % mod; 
    }
