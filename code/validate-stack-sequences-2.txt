bool validateStackSequences(int* pushed, int pushedSize, int* popped, int poppedSize)
{
    int i=0,j=0;
    int top=0;
    int k=0;
    int *arr=(int*)malloc(pushedSize*(sizeof(int)));
    for(i=0;i<pushedSize;i++)
    {
        arr[top]=pushed[i];
        while(top>=0&&j<pushedSize && popped[j]==arr[top])
        {
            top--;
            j++;
        }
        top++;
    }
    if(top==0)
    return 1;
    return 0;
}
