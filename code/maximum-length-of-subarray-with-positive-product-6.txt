int getMaxLen(vector<int>& n, int start = 0) {
    int cnt[] = {start - 1, INT_MAX}, max_len = 0, is_odd = false;
    for (int i = start; i < n.size(); ++i) {
        if (n[i] == 0)
            return max(max_len, getMaxLen(n, i + 1));
      is_odd = n[i] < 0 ? !is_odd : is_odd;
      cnt[is_odd] = min(cnt[is_odd], i);
      max_len = max(max_len, i - cnt[is_odd]);
    }
    return max_len;
}
