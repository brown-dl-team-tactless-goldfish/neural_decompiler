class Solution {
    int index(char c) {
        return c == 'B' ? 0 : 1;
    }
public:
    int minimumRecolors(string blocks, int k) {
        array<int, 2U> mp{0};
        int N = blocks.length();
        for(int i = 0; i < min(N, k); ++i) {
            ++mp[index(blocks[i])];
        }
        int ans = mp[1];
        for(int i = k; i < N; ++i) {
            ++mp[index(blocks[i])];
            --mp[index(blocks[i-k])];
            ans = min(ans, mp[1]);
        }
        return ans;
    }
};
