class Solution {
public:
    long long wonderfulSubstrings(string word) {
        unordered_map<int, int> seen;
        seen[0] = 1;
        int mask = 0;
        long long ans = 0;
        for (int i = 0; i < word.length(); ++i){
            mask ^= (1 << (word[i] - 'a'));
            ans += seen[mask];
            for (int j = 0; j < 10; ++j){
                ans += seen[mask ^ (1<<j)];
            }
            seen[mask]++;
        }
        return ans;
    }
};
