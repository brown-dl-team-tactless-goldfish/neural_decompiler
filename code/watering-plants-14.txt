class Solution {
public:
    int wateringPlants(vector<int>& A, int C) {
        int ans = 0  , r = C;
        
        for (int i =0 ; i < A.size() ; i++){
            if (C < A[i]){
                ans += i + 1;
                C = r;
                ans += i ;
            }
            else ans++;
            C-= A[i];
            
        }
        return ans;
    }
};
