bool divideArray(int* nums, int numsSize){
    int freq[501]={0};
    int i,res=0;
    for (i=0; i < numsSize; i++)
    {
        if (freq[nums[i]]%2 == 0)
            freq[nums[i]]++;
        else
            freq[nums[i]]--;
    }
    for (i=0; i < numsSize; i++)
    {
        if (freq[nums[i]] > 0)
            return 0;
    }
    return 1;
}
