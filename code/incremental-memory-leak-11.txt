/**
 * @param {number} memory1
 * @param {number} memory2
 * @return {number[]}
 */
var memLeak = function(memory1, memory2) {
    let counter = 0;
    
    while(true){
      counter=counter+1
      if(counter > memory1 && counter > memory2) {
        break
      }
      if(memory1 >= memory2){
        memory1-=counter;
        continue
      }
      if(memory1 < memory2){
        memory2 -=counter;
        continue
      }
      
    }
    return [ counter,memory1, memory2]
};
