    public int AppendCharacters(string s, string t) 
    {
        if (s.Contains(t))
            return 0;

        int m = s.Length;
        int n = t.Length;

        int sIndex = 0;
        int tIndex = 0;

        for (int i = 0; i < m; i++)
        {
            if (s[i] == t[tIndex])
                tIndex++;
            if (tIndex >= n)
                return 0;
        }

        return n - tIndex;
    }
