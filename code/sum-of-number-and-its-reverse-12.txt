class Solution {
private:
    int reverseInt(int n){
        int cat = 0;
        while(n){
            cat = cat * 10 + n%10;
            n/=10;
        }
        return cat;
    }
public:
    bool sumOfNumberAndReverse(int num) {
        for(int i=0; i<=num; ++i)
            if(i+reverseInt(i)==num) return true;
        return false;
    }
};
