int countStudents(int* students, int studentsSize, int* sandwiches, int sandwichesSize){
    int count = studentsSize, aux, i, block_code = 0;
    
    for (i = 0; i < studentsSize && block_code <= studentsSize;) {
        if (students[i] != sandwiches[i]) {
            aux = students[i];
            for (int j = i; j < studentsSize-1; j++) {
                students[j] = students[j+1];
            }
            students[studentsSize-1] = aux;
            block_code++;
        } else {
            count--;
            i++;
            block_code = 0;
        }
    }
    return count;
}
