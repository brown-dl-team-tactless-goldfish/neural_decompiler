int* postorderTraversal(struct TreeNode* root, int* returnSize) { 
	int *results=NULL;
    *returnSize=0;
    
    int *leftarr=NULL,*rightarr=NULL,leftarrSize=0,rightarrSize=0;
    if(!root) return results;
    else{
        if(root->left)  leftarr=postorderTraversal(root->left,&leftarrSize);
        if(root->right) rightarr=postorderTraversal(root->right,&rightarrSize);
    }
    *returnSize=1+leftarrSize+rightarrSize;
    results=(int*)malloc(sizeof(int)*(*returnSize));
    int i,j;
    for(i=0;i<leftarrSize;i++)
        results[i]=leftarr[i];
    for(j=0;j<rightarrSize;j++)
        results[i+j]=rightarr[j];
        results[*returnSize-1]=root->val;
    
    free(leftarr);
    free(rightarr);
    
    return results;
 }
