class Solution {
public:
    int minFlips(int a, int b, int c) {
        int result = 0;
        //int is of 32 bit, so check each bit
        for (int i = 0; i < 32; i++) {
            //Determine if ith bit of a, and c is set or not
            bool aIthSet = a & 1<<i, bIthSet = b & 1<<i, cIthSet = c & 1<<i;

            if (cIthSet) { // ith bit of c is set (ie 1)
                //aIthSet | bIthSet must be one
                //Otherwise alter anyone
                if (!(aIthSet | bIthSet)) {
                    result++;
                } 
            } else { // ith bit of c is not set (ie 0)
                if (aIthSet & bIthSet) {
                    //both (a, b) set alter both
                    result += 2;
                } else if (aIthSet | bIthSet) {
                    //One (a or b) set so alter 1
                    result++;
                }
                //if both unset do nothing
            }
        }
        return result;
        
    }
};
