class Solution {
public:
    bool makeEqual(vector<string>& words) {
        int n = words.size();
        unordered_map<char, int> m;
        for(auto word: words){
            for(auto c : word)
                m[c]++;
        }
        for(auto p : m){
            if(p.second%n != 0)
                return false;
        }
        
        return true;
    }
};
