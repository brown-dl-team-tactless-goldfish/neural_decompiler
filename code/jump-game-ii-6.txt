int jump(int* nums, int numsSize){
if(numsSize < 2) return 0;
    if(nums[0] >=numsSize - 1) return 1;
    int jumps = 0, max = 0, end = 0;
    for(int i=0; i<numsSize-1; i++)
    {
        max = nums[i] + i > max?nums[i]+i:max;
        if(i == end)
        {
            end = max;
            jumps++;
        }
    }
    return jumps;
	}
