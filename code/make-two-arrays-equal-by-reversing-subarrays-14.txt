int cmp(const void * a,const void * b)
{
    return ( *(int*)a - *(int*)b );
    
}

bool canBeEqual(int* target, int tlen, int* arr, int alen){
    if(tlen!=alen) return false;
    else
    {
        qsort(arr,alen,sizeof(int),cmp);
        qsort(target,tlen,sizeof(int),cmp);
        int k,flg=0;
        for(k=0;k<alen;k++)
        {
            if(arr[k]!=target[k]) flg++;
        }
        
        if(flg>0) return false;
        else return true;
    }
}
