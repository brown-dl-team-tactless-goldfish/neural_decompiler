class Solution {
public:
    int minNumber(vector<int>& A, vector<int>& B) {
        set<int> S(begin(A),end(A)),T(begin(B),end(B)), U;
        set_intersection(begin(S),end(S),begin(T),end(T),inserter(U,begin(U)));
        if(!U.empty()) return *begin(U);
        else{
            int first = ((*begin(S)) * 10) + (*begin(T));
            int second = ((*begin(T)) * 10) + (*begin(S));
            cout<<first<<' '<<second;
            return min(first,second);
        }
    }
};
