var simplifiedFractions = function(n) {
    const gcd = (a, b) => a == 0 ? b : gcd(b % a, a);
    const result = [];
    for(let i = 1; i < n; ++i) {
        for(let j = i + 1; j <= n; ++j) {
            if (gcd(i, j) == 1)
                result.push(`${i}/${j}`);
        }
    }
    return result;
};
