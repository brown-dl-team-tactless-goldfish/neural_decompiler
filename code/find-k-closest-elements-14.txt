public class Solution 
{
    public IList<int> FindClosestElements(int[] arr, int k, int x) 
    {
        int start = 0, end = arr.Length - 1;

	    while (end - start >= k)
        {
		    if (Math.Abs(x - arr[end]) < Math.Abs(x - arr[start]))
            {
			    start++;
            }
		    else
            {
			    end--;
            }
        }
        
        int[] ret = new int[k];
        Array.Copy(arr, start, ret, 0, k);
            
        return ret.ToList();
    }
}
