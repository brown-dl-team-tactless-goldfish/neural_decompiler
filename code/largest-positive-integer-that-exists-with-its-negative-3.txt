#include<stdlib.h>
int cmp(const void *x, const void *y)
{
    int a = *(int*)x;
    int b = *(int*)y;
    return a-b;
}
int findMaxK(int* nums, int numsSize)
{
    qsort(nums, numsSize, sizeof(int), cmp);
    for(int i = 0; i < numsSize; i++)
    {
        for(int j = numsSize - 1; j >= 0; j--)
        {
            if(nums[i] == (0-nums[j]))
            {
                return nums[j];
            }
        }
    }
    return -1;
}
