class Solution {
public:
    int minSwaps(string s) {
        int zeroes = 0, ones = 0, d1 = 0, d2 = 0;
        for(char c : s) c == '1' ? ones++ : zeroes++;
        if(abs(ones - zeroes) > 1) return -1;
        
        for(int i=0; i<s.size(); i += 2) d1 += s[i] != '1', d2 += s[i] != '0';
        return ones > zeroes ? d1 : ones < zeroes ? d2 : min(d1, d2);
    }
};
