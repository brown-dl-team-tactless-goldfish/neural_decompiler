class Solution {
public:
    void f(int n, int cur, vector<int> &ans) {
        if(cur > n ) return;
            
        ans.push_back(cur);
        for(int i = 0; i < 10; i++) {
            f(n, cur*10 + i, ans);
        }
    }
    vector<int> lexicalOrder(int n) {
        
        vector<int> ans;
        for(int cur = 1; cur < 10; cur++) {
            f(n, cur, ans);
        }
        return ans;
    }
};
