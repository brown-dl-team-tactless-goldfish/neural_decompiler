 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     struct TreeNode *left;
 *     struct TreeNode *right;
 * };
 */

void traverse(struct TreeNode* root, struct TreeNode **ptr) {
    
    if (root != NULL) {
        traverse(root->left, ptr);
        struct TreeNode *newnode = malloc(sizeof(struct TreeNode));

        (*ptr) = (*ptr)->right;
        traverse(root->right, ptr);
    }
}
