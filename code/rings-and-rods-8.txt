int countPoints(char * a)
{
   int map[10][3]={0},res=0;
   for(int i=0;i<strlen(a);i=i+2)
   {
       if(a[i]=='R')map[a[i+1]-'0'][0]++;
       if(a[i]=='G')map[a[i+1]-'0'][1]++;
       if(a[i]=='B')map[a[i+1]-'0'][2]++;
   }
    for(int i=0;i<10;i++)if(map[i][0]&&map[i][1]&&map[i][2])res++;
    return res;
}
