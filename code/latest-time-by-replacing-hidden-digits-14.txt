class Solution {
public:
    bool valid(string_view s){
        return (s[0]<='2' || s[0]=='?') && (s[1]<= (s[0]=='2'?'3':'9') || s[1]=='?') && (s[3]<='5' || s[3]=='?')&&(s[4]<='9' || s[4]=='?');
    }
    string maximumTime(string time) {
        for(int i = 0;i<time.size();i++)
            if(time[i]=='?')
                for(int j = '9';j>='0';j--){
                    time[i]=j;
                    if(valid(time)) break;
                }
        return time;
    }
};
