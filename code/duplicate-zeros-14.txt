int a=arrSize;
int count=0;
for(int i=0;i<a;i++)
{
    if(arr[i]==0)count++;
}
if(count<1)return;
int *cpy=calloc(a+count,sizeof(int));
int temp=0;
for(int i=0;i<a;i++)
{
    if(arr[i]==0)
    {
        cpy[temp++]=arr[i];
        cpy[temp++]=0;
        continue;
    }
    cpy[temp++]=arr[i];
}


for(int i=0;i<arrSize;i++)
    arr[i]=cpy[i];
return ;
