class Solution {
public:
    using VI = vector<int>;
    int distanceBetweenBusStops(VI& A, int i, int j) {
        if (j < i) {
            swap(i, j);
        }
        auto route = accumulate(A.begin() + i, A.begin() + j, 0),
             total = accumulate(A.begin(), A.end(), 0);
        return min(route, total - route);
    }
};
