int diagonalSum(int** mat, int matSize, int* matColSize){
    int result = 0;
    int end = matSize - 1;
    for (int i = 0; i < matSize; i++)
    {
        result += (mat[i][i] + mat[i][end - i]);
    }
    
    if (matSize % 2)
    {
        result -= mat[matSize / 2][matSize / 2];
    }
    
    return result;
}
