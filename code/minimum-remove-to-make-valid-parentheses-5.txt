char * minRemoveToMakeValid(char * s){
    char * result = (char*)calloc(strlen(s) + 1, sizeof(char));
    int * stack = (int *)malloc(strlen(s)*sizeof(int));
    int idx = -1;
    for(int i=0; i<strlen(s); i++)
        if(s[i] == '(')
            stack[++idx] = i;
        else if(s[i] == ')')
        {
            if(idx >= 0)
                idx--;
            else
                s[i] = ' ';
        }
    for(int i=0; i<=idx; i++)
        s[stack[i]] = ' ';
    int id = 0;
    for(int i=0; i<strlen(s); i++)
        if(s[i] != ' ')
            result[id++] = s[i];
    return result;
}
