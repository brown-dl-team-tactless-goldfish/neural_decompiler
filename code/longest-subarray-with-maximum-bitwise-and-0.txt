int longestSubarray(int* nums, int numsSize){
    int max = 0;
    int cn = 0;
    int MaxCn = 0;
    for(int i = 0; i < numsSize; i++){
        if(nums[i] > max){
            max = nums[i];
            MaxCn = 1;
            cn = 1;
        }
        else if(nums[i] == max)
            cn++;
        else{
            MaxCn = fmax(MaxCn, cn);
            cn = 0;
        }
    }
    MaxCn = fmax(MaxCn, cn);
    return MaxCn;
}
