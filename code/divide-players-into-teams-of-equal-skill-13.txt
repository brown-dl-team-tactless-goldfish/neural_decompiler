class Solution {
public:
    long long dividePlayers(vector<int>& vt) {
        int n = vt.size();
        sort(vt.begin(),vt.end());
        int l = 0,r = n-1;
        long long prevSum = vt[l]+vt[r];
        long long sum = 0;
        while(l<=r)
        {
            if(vt[l]+vt[r] != prevSum)
            {
                return -1;
            }
            sum += (vt[l]*vt[r]);
            l++;
            r--;
        }
        return sum;
    }
};
