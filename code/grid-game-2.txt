class Solution{
public:
    long long gridGame(vector<vector<int>>&mat)
    {
        int n=2;
        int m=mat[0].size();
        long long top=0;
        for(int i=0;i<m;i++)
        {
            top+=mat[0][i];
        }
        long long ans=1e17;
        long long bottom=0;
        for(int i=0;i<m;i++)
        {
            top-=mat[0][i];
            ans=min(ans,max(top,bottom));
            bottom+=mat[1][i];
        }
        return ans;
    }
};
