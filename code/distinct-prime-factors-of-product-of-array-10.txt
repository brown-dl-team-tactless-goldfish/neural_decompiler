class Solution {
public:
    int distinctPrimeFactors(vector<int>& nums) {
         unordered_set<int> st;
        for(int i=0; i<nums.size(); i++){
            
            for(int j=2; j<=nums[i]; j++){
                while(nums[i]%j==0){
                    nums[i] = nums[i]/j;
                    st.insert(j);
                }
            }
            
        }
        return st.size();
    }
};
