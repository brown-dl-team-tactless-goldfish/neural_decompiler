/*
Simple take the max diff of index with diff 2
*/

class Solution {
public:
    int maxJump(vector<int>& stones) {
        int  ans = stones[1] - stones[0];
        for(int i = 0; i < stones.size(); ++i){
            if(i+2 < stones.size()) ans = max(ans, stones[i+2] - stones[i]);
        }
        return ans;
    }
};
