class Solution {
public:
    int halveArray(vector<int>& nums) {
        priority_queue<long double>pq;
        long double tot=0,runningSum;
        for(auto & i:nums){
            long double temp=i;
            tot+=temp;
            pq.push(temp);
        }
        runningSum=tot;
        int ans=0;
        while(runningSum*2>tot){
            long double t=pq.top();
            pq.pop();
            runningSum-=t/2;
            pq.push(t/2);
            ans++;
        }
            return ans;   
    }
};
