		#define mod 1000000007;
		int countVowelPermutation(int n) 
			{
				long long int dp[n+1][5];
				for(int i=0;i<5;i++)
					dp[1][i]=1;
				for(int i=2;i<n+1;i++)
				{
					dp[i][0]=(dp[i-1][1]+dp[i-1][2]+dp[i-1][4])%mod;
					dp[i][1]=(dp[i-1][0]+dp[i-1][2])%mod;
					dp[i][2]=(dp[i-1][1]+dp[i-1][3])%mod;
					dp[i][3]=dp[i-1][2];
					dp[i][4]=(dp[i-1][2]+dp[i-1][3])%mod;
				}
				int res=0;
				for(int i=0;i<5;i++)
					res=(res+dp[n][i])%mod;
				return res;
			}
