int getMinimumDifference(struct TreeNode* root){
    
    int res=INT_MAX, prev=-1;
    Inorder(root,&res,&prev);
    return res;

}
void Inorder(struct TreeNode* root, int *res, int *prev)
{
    if (root == NULL)
        return;
    Inorder(root->left,res,prev);
    if (*prev != -1)
    {
        if ((root->val - *prev) < *res)
            *res = root->val - *prev;
    }
    *prev = root->val;
    Inorder(root->right,res,prev);
    return;
}
