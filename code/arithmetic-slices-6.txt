
int numberOfArithmeticSlices(int* nums, int numsSize){
    
    int k,l,sum,s_1,j;
    s_1=numsSize-1;
    l=1;
    sum=0;
    for (k=1;k<numsSize;k++){
        if ((k!=s_1)&& nums[k]-nums[k-1]==nums[k+1]-nums[k]){
            l+=1;

        }else{
            l+=1;
            if(l>2){
                for(j=3;j<(l+1);j++){
                    sum+=l-j+1;
                }
            }
            l=1;
        }
    }
    return sum;
}
