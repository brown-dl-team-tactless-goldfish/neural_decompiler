typedef struct {
    double radius;
    double x_center;
    double y_center;
} Solution;


Solution* solutionCreate(double radius, double x_center, double y_center) {
    Solution *obj = (Solution *)malloc(sizeof(Solution));
    obj->radius = radius;
    obj->x_center = x_center;
    obj->y_center = y_center;
    srand(time(NULL));
    return obj;
}

static double get_random(void) { 
    return (double)rand() / (double)RAND_MAX; 
}

double* solutionRandPoint(Solution* obj, int* retSize) {
    *retSize = 2;
    double *result = calloc(*retSize, sizeof(double));
    double theta = get_random() * 2 * M_PI;
    double r = obj->radius * sqrt(get_random());
    result[0] = obj->x_center + r * cos(theta);
    result[1] = obj->y_center + r * sin(theta);
    return result;
}

void solutionFree(Solution* obj) {
    free(obj);
}
