int lc = 0, rc = 0;

int count(TreeNode* root, int val)
{
    if(!root)
    {
        return 0;
    }
    
    int left = count(root->left,val);
    int right = count(root->right, val);
    
    if(root->val == val)
    {
        lc = left;
        rc = right;
    }
    
    return left + right + 1;
}

bool btreeGameWinningMove(TreeNode* root, int n, int x)
{
    int tmp = count(root,x);
    
    int var = n - (lc+rc+1);
    
    int maxi = max({var,lc,rc});
    
    return maxi > n/2;
}
