bool isPalindrome(char *s,int *fwd,int *rev) {
    while(*fwd < *rev) {
        if(s[*fwd] != s[*rev])
            return false;

        (*fwd)++;
        (*rev)--;
    }

    return true;
}

bool validPalindrome(char * s){
    int i = 0, j = strlen(s) - 1;

     if (!isPalindrome(s, &i, &j)) {
        int fwd = i + 1;
        int rev = j - 1;

        return isPalindrome(s, &fwd, &j) || isPalindrome(s, &i, &rev);
    }
    
    return true;
}
