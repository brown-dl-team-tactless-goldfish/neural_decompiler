int maxAscendingSum(int* nums, int numsSize){
    int i, sum = nums[0], max = nums[0];
    
    for(i = 1; i < numsSize; i++){
        if(nums[i] > nums[i - 1])
            sum += nums[i];
        else
            sum = nums[i];
        
        max = sum > max? sum: max;
    }
    
    return max;
}
