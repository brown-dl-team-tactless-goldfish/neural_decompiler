bool uniqueOccurrences(int* arr, int arrSize){
    int n [ 2000 ] = {0};
    int m [ 2000 ] = {0};
    
    int i;
    for(i=0;i<arrSize;i++)
    {
        n[arr[i]+1000]++;
    }
    
    for(i=0;i<2000;i++)
    {
        
        if(n[i]!=0)
        {
            m[n[i]]++;
            if(m[n[i]]>1)
            {
                return(0);
            }
        }
    }
    return(1);
}
