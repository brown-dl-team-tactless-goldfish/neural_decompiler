char * minRemoveToMakeValid(char * s)
{
    int i, j, len;
    int left = 0, right = 0;

    if (!s) {
        return NULL;
    }

    len = strlen(s);
    for (i = 0; i < len; i++) {
        if (s[i] == ')') {
            right++;
        }
    }

    for (j = 0, i = 0; i < len; i++) {
        if (s[i] == '(') {
            if (left == right) continue;
            left++;
        } else if (s[i] == ')') {
            right--;
            if (left == 0) continue;
            left--;
        }
        s[j++] = s[i];
    }
    s[j] = '\0';
    return s;
}
