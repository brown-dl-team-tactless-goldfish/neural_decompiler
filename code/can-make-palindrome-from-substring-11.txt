class Solution {
public:
    vector<bool> canMakePaliQueries(string s, vector<vector<int>>& queries) {
        vector<int> pref; pref.push_back(0);
        for(char c : s) {
            int i = c-'a';
            pref.push_back(pref.back() ^ (1<<i));
        }
        vector<bool> ans;
        for(auto &q : queries) {
            int mask = pref[q[0]]^pref[q[1]+1];
            int mnc = __builtin_popcount(mask)/2;
            ans.push_back(mnc<=q[2]);
        }
        return ans;
    }
};
