int getMaximumGenerated(int n)
{
    int *nums = malloc(sizeof(int) * (n + 1)), max_generated = 0;
	
    if (n == 0 || n == 1)
        return n;
		
    nums[0] = 0;
    nums[1] = 1;
    
    for (size_t i = 2; i <= n; i++)
		nums[i] = i % 2 == 0 ?  nums[i / 2] : nums[i / 2] + nums[i / 2 + 1];
    
    for (size_t i = 0; i <= n; i++)
		max_generated = nums[i] > max_generated ? nums[i] : max_generated;
    
    return max_generated;
}
