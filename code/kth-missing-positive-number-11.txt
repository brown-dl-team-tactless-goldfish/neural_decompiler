int findKthPositive(int* arr, int arrSize, int k){
    int i = 0;
    int num = 1;
    
    while (k != 0 && i < arrSize) {
        if (arr[i] == num) {
            ++i;
        } else {
            --k;
        }
        ++num;
    }
    
    while (k != 0) {
        --k;
        ++num;
    }
    
    return num - 1;
}
