    int minimumSwap(string s1, string s2) {
        int xy = 0, yx = 0, res = 0;
        for(int i = 0; i < s1.length(); i++) {
            if(s1[i] == s2[i]) continue;
            if(s1[i] == 'x') xy++;
            else yx++;
        }
        
        res += xy/2 + yx/2;
        xy %= 2;
        yx %= 2;
        
        return xy != yx ? -1 : res + xy*2;
    }
