class Solution {
public:
int helper(int n, int len, int idx, vector<char> &vovels)
{
    if (len == n)
    {
        return 1;
    }

    int ans = 0;

    for (int x = idx; x < vovels.size(); x++)
    {
        ans += helper(n, len + 1, x, vovels);
    }

    return ans;
}

int countVowelStrings(int n)
{
    vector<char> vovels = {'a', 'e', 'i', 'o', 'u'};

    return helper(n, 0, 0, vovels);
}
};
