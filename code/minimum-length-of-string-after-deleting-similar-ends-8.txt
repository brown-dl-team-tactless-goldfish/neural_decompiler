class Solution {
public:
    int minimumLength(string s) {
        int n = s.length();
        int start = 0 , end = n-1;
        while(start<end && s[start]==s[end]){
           auto c = s[start];
           while(s[start]==c && start<=end)
               start++;
           while(s[end]==c && start<end)
               end--;
        }
        return end-start+1 ;
    }
};
