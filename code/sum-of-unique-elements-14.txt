class Solution {
public:
    int sumOfUnique(vector<int>& nums) {
        if(nums.size()==NULL)
            return 0;
        unordered_map<int,int>Occurance;
        for(int i:nums)
            Occurance[i]++;
        int sum=0;
        for(auto itr=Occurance.begin();itr!=Occurance.end();itr++)
        {
            if(itr->second==1)
                sum+=itr->first;
        }
        return sum;
    }
};
