int maxAscendingSum(int* nums, const int numsSize){
        int max_sum = 0;
        int sum = nums[0];
        for (size_t idx = 1; idx < numsSize; ++idx) {
                if (nums[idx] <= nums[idx - 1]) {
                            if (sum > max_sum)
                                    max_sum = sum;
                            sum = 0;
                    }
                    sum += nums[idx];
        }
         if (sum > max_sum)
                 max_sum = sum;
         return max_sum;
}
