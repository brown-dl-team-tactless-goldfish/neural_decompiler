class Solution {
public:
    bool isPrefixString(string s, vector<string>& words) {
        int i=0,n=s.length();
        for(auto &w:words) {
            for(auto &c:w) 
                if(c==s[i]) 
                    i++;
                else 
                    return false;
            if(i>=n) return true;
        }
        return false;
    }
};
