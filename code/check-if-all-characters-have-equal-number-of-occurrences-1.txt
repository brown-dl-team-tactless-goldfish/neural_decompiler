	int[] occurence = new int[26];
	HashSet<int> occ_map = new HashSet<int>();
	char[] chars = s.ToCharArray();
	foreach (char c in chars)
		occurence[c - 'a'] += 1;

	for (int i = 0; i < 26; i++)
	{
		if (occurence[i] != 0)
		{
			occ_map.Add(occurence[i]);
		}
		if (occ_map.Count > 1) return false;
	}
	return true;
