    unordered_map<int,int> mp;
    
    for(auto &x: nums)
    {
        mp[x]++;
    }
    
    int sum = 0, rem = 0;
    
    for(auto &x: mp)
    {
          sum += x.second / 2;  
            
          rem += x.second % 2;
        
    }
    
    vector<int> ans;
    
    ans.push_back(sum);
    ans.push_back(rem);
    
    return ans;
    
}
