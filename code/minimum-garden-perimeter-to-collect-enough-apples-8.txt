class Solution {
public:
  #define ll long long 
    long long minimumPerimeter(long long neededApples) {
        ll  i=1;
        for(;neededApples >0;i++)
        {
            neededApples-=3*i*4; // i*4+2*i*4
            neededApples-=(((2*i)*(2*i-1))/2-(i*(i+1))/2)*8; // summation using (n*(n+1))/2
        }
        return (i-1)*8;
      
      
//       Length of side = i * 2
// Also, Perimeter = side * 4 = (i * 2) * 4 = i * 8
        
    }
};
