
```public:
    void solve(int idx,int n,vector<int>&v)
    {
        if(idx>n)
        {
            return ;
        }
        v.push_back(idx);
        for(int i=0;i<=9;i++)
        {
            solve(idx*10+i,n,v);
        }
    }
    vector<int> lexicalOrder(int n) {
        vector<int>v;
        for(int i=1;i<=9;i++)
        {
            solve(i,n,v);
        }
        return v;
    }
};
