class Solution {
public:
    int reachNumber(int target) {
        double N = abs(target);
        double x = (-1 + sqrt(1 + 8 * N)) / 2;
        if (x == (int)x) return (int)x;
        int b = (int)x + 1;
        while (true) {
            int c = (1 + b) * b / 2 - N;
            if ((c & 1) == 0) break;
            b++;
        }
        return b;
    }
};
