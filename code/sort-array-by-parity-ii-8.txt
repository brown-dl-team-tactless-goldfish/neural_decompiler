int f_next, temp; 
bool target = 0;

for(int i = 0; i < numsSize; i++) {
    if(nums[i] % 2 != target) {
        f_next = i + 1;
        while(f_next < numsSize) {
            if(nums[f_next] % 2 == target) {
                temp = nums[f_next];
                nums[f_next] = nums[i];
                nums[i] = temp;
                break;
            } else {
                f_next += 2;
            }
        }
    }
    target ^= 1;
}

*returnSize = numsSize;
return nums;
