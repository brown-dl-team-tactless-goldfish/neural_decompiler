// while performing given conditions on any integer
// you may not observed that but, every integer got into loop for last three values: 4, 2, 1
class Solution {
public:
    int getKth(int lo, int hi, int k) {
        vector<pair<int, int>> temp;
        for(int i = lo; i <= hi; i++) {
            int steps = 0;
            int x = i;
            while(x != 1) {
                if(x & 1) x = 3 * x + 1;
                else x /= 2;
                steps++;
            }
            
            temp.push_back(make_pair(steps, i));
        }
        
        sort(temp.begin(), temp.end());
        return temp.at(k-1).second;
    }
};
