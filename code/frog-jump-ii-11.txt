class Solution {
public:
    int maxJump(vector<int>& vt) {
        int n=vt.size();
        int ans=-1e9;
        if(n<=2){
            ans=max(ans,vt[1]-vt[0]);
            return ans;
        }
        for(int i=2;i<n;++i){
            ans=max(ans,max(vt[i]-vt[i-1],vt[i]-vt[i-2]));
        }
        return ans;
        
    }
};
