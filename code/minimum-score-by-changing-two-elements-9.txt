class Solution {
public:
    int minimizeSum(vector<int>& nums) {
        sort(nums.begin(),nums.end());
        int n=nums.size();
        // either increase nums[0] and nums[1] to nums[2] to minimize difference 
        // or decrease nums[n-1] and nums[n-2] to nums[n-3] 
        // or increase nums[0] to nums[1] and decrease nums[n-1] to nums[n-2] 
        return min({nums[n-1]-nums[2],nums[n-3]-nums[0],nums[n-2]-nums[1]});
    }
};
