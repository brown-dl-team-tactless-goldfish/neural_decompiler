class Solution {
public:

	int gcd(int a, int b){
	if (a == 0)
		return b;
	return gcd(b % a, a);
}

	int minOperations(vector<int>& nums, vector<int>& numsDivide) {
		int res = 0,ans = 0;
		for(auto x:numsDivide) res = gcd(res,x);
		sort(begin(nums),end(nums));
		for(auto x:nums){
			if(x==res or res%x==0) return ans;
			ans++;
		}
		return -1;
	}
};
