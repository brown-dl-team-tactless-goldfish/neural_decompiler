/**
 * Return an array of arrays of size *returnSize.
 * The sizes of the arrays are returned as *returnColumnSizes array.
 * Note: Both returned array and *columnSizes array must be malloced, assume caller calls free().
 */
int** matrixReshape(int** mat, int matSize, int* matColSize, int r, int c, int* returnSize, int** returnColumnSizes){
    int matRow = matSize, matCol = *matColSize;
    if (r * c != matRow * matCol){
		*returnColumnSizes = calloc(matRow, sizeof(int));
        *returnSize = matRow;
        for (int i = 0; i < matRow; i++){
            (*returnColumnSizes)[i] = matCol;
        }
		return mat;
    }

    int **result = calloc(r, sizeof(int *));
    *returnColumnSizes = calloc(r, sizeof(int));
    *returnSize = r;
    for (int i = 0; i < r; i++){
        result[i] = calloc(c, sizeof(int));
        (*returnColumnSizes)[i] = c;
    }

    for (int ind = 0; ind < r * c; ind++){
        result[ind / c][ind % c] = mat[ind / matCol][ind % matCol];
    }
 
    return result;
}
