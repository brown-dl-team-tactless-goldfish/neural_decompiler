

bool canBeEqual(int* target, int targetSize, int* arr, int arrSize){
    int ht[1001]= {0};
    int hr[1001] = {0};
    int i;
    for(i=0;i<targetSize;i++) {
        ht[target[i]]++;
        hr[arr[i]]++;
    } 
    for(i=1;i<1001;i++) {
        if(ht[i]!=hr[i]) return false;
    }
    return true;
}
