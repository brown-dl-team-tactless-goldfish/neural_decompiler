int mostWordsFound(char ** sentences, int sentencesSize){
    int maxWord = 0, cnt;
    for (int i = 0; i < sentencesSize; i++)
    {
        /* code */
        cnt = 0;
        char *token = strtok(sentences[i], " ");
        while (token != NULL)
        {
            /* code */
            cnt++;
            token = strtok(NULL, " ");
        }
        maxWord = fmax(cnt, maxWord);
    }
    return maxWord;
}
