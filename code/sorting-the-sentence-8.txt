	string [] words = s.Split(' ');
	string[] word = new string[words.Length + 1];
	for (int i = 0; i < words.Length;i++)
	{
		// get last char of each word
		char LastElement = words[i][words[i].Length - 1];
		// convert char to int 
		int index = (int)Char.GetNumericValue(LastElement);
		// add word to array without the number
		word[index] = words[i].Remove(words[i].Length - 1, 1);
	}
	// convert array to string then return
	return string.Join(" ", word).TrimStart();
