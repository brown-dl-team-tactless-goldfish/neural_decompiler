int* decompressRLElist(int* nums, int numsSize, int* returnSize) {
    int size = 10001;
    int* res = (int*) malloc(size * sizeof(int));
    *returnSize = 0;
    int pair[2] = {0};
    for (int i = 0; i < numsSize; i++) {
        int tmp = 2 * i;
        if (tmp + 1 >= numsSize) break;
        pair[0] = nums[tmp];
        pair[1] = nums[tmp + 1];
        while (pair[0] > 0) {
            res[(*returnSize)++] = pair[1];
            pair[0]--;
        }
    }
    return res;
}
