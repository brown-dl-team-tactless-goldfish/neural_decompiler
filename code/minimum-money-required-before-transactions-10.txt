class Solution {
public:
    long long minimumMoney(vector<vector<int>>& t) {
        
        
        int mn=INT_MAX;
        int  mx=INT_MIN;
        long long ans=0;
        for(auto it:t)
        {
            mn = min(it[0],it[1]);
            mx=max(mx,mn);
        }
            
        for(auto it:t)
        {
            ans += max(it[0]-it[1],0);
        }
        ans+=mx;   
        return ans;
    }
};
