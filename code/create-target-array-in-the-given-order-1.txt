int* createTargetArray(int* nums, int numsSize, int* index, int indexSize, int* returnSize){
    *returnSize=numsSize;
    int i, *target,j;
    target = (int*)calloc(numsSize, sizeof(int));
    for(i=0;i<numsSize;i++){
        if(target[index[i]]!=0)
            for(j=numsSize-1;j>index[i];j--)
                target[j]=target[j-1];
        target[index[i]] = nums[i];
    }
    return target;
}
