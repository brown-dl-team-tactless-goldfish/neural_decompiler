// C++
class Solution {
public:
    int xorBeauty(vector<int>& nums) {
        int ans = 0;
        for(int num: nums) {
            ans ^= num;
        }
        return ans;
    }
};


// Golang
func xorBeauty(nums []int) int {
    var ans int = 0
    for _, num := range nums {
        ans = ans ^ num
    }
    return ans
}
