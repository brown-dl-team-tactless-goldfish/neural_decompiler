class Solution {
public:
    string maximumBinaryString(string binary) {
        int p = -1;
        const int n = binary.length();
        for (int i = 0; i < n; ++i) {
            if (binary[i] == '0') {
                if (p < 0) {
                    p = i;
                } else {
                    ++p;
                }
            }
        }
        string r(n, '1');
        if (p >= 0) {
            r[p] = '0';
        }
        return r;
     
    }
};
