void rotate(int* nums, int numsSize, int k){

    int i, j, tmp;
    int N = numsSize;
    int K = k;
    // write your code in C99 (gcc 6.2.0)
    if (N==K) return;
    
    for(i=N-1; i>=N-K; i--) {
        tmp = nums[N-1];
        for (j=N-1;j>0;j--)
            nums[j] = nums[j-1];
        nums[0] = tmp;
        
    }
    return;    
}
