class Solution {
private:
    vector<int> v;
public:
    vector<int> postorder(Node* root) {
        postorder(root, v);
        return v;
    }

    void postorder(Node* node, vector<int>& v) {
        if (NULL == node) return;
        
        vector<Node*> c = node->children;
        vector<Node*>::iterator iter = c.begin();
        for (; iter != c.end(); ++iter) {
            postorder(*iter, v);
        }
        v.push_back(node->val);
    }
};
