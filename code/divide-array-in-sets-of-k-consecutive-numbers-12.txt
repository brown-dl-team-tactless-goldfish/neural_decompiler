    for(auto &x: nums)
    {
        count[x]++;
    }
    
    sort(nums.begin(),nums.end());
    
    for(auto &a:nums)
    {
        if(count[a] == 0)
        {
            continue;
        }
        
        for(int i=0; i<k; i++)
        {
            if(--count[i+a] < 0)
            {
                return false;
            }
        }
    }
    
    return true;
