    def rotateTheBox(self, box: List[List[str]]) -> List[List[str]]:
        for row in box:
            i = len(box[0]) - 1
            for c in range(len(row) - 1, -1, -1):
                if row[c] == '*':
                    i = c - 1
                elif row[c] == '#':
                    row[c], row[i] = row[i], row[c]
                    i -= 1
        return zip(*box[:: -1])
