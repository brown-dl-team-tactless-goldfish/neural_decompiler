long long sum(long long n){
    return n*(n+1)/2;
}

long long zeroFilledSubarray(int* nums, int numsSize){
    long long res =0;
    int n =0;
    for (int i=0;i<numsSize;i++){
        if (nums[i] == 0)
            n++;
        else{
            if (n){
                res+= sum(n);
                n = 0; }
        }
    }
    return res + sum(n);
}
