int matrixScore (int** A, int ASize, int* AColSize) {
    int i, j, m = ASize, n = AColSize[0];
    int cnt, ret = m * (1 << (n - 1));
    
    for (j = 1; j < n; j++) {
        for (i = cnt = 0; i < m; i++) {
            if (A[i][j] == A[i][0]) cnt++;
        }
        ret += (cnt > m - cnt ? cnt : m - cnt) * (1 << (n - j - 1));
    }
    
    return ret;
}
