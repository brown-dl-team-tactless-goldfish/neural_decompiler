int len1=strlen(word1);
int len2=strlen(word2);

char* ans=malloc(sizeof(char)*(len1+len2+1));


ans[len1+len2]='\0';

int k=0;


for(int i=0,j=0;i<len1||j<len2;i++,j++)
{
    
    if(i<len1)
    {
    ans[k] = word1[i];
        k++;
    }
    
    if(j<len2)
    {
    ans[k] = word2[j];
        k++;
    }
}


return ans;
