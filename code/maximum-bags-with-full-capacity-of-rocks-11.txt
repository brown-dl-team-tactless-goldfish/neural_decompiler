int cmpFunc(const void* a, const void* b){ return *(int*)a - *(int*)b; }

int maximumBags(int* caps, int capsSize, int* rocks, int rocksSize, int adds){
    int result=0, rems[capsSize];

    for(int i=0; i<capsSize; i++){ rems[i] = caps[i]-rocks[i]; }
    qsort(rems, capsSize, sizeof(int), cmpFunc);
    for(int i=0; i<capsSize; i++){
        if(rems[i]){
            if(adds<rems[i]){ break; }
            else{ adds -= rems[i]; }
        } result++;
    }

    return result;
}
