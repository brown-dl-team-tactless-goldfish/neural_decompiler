class Solution {
public:
    int subarrayGCD(vector<int>& nums, int k) {
        int res = 0;
        if (nums.size() == 1) return nums[0] == k;
        for (int i = 0; i < nums.size(); i++) {
            int gcdVal = nums[i];
            for (int j = i; j < nums.size(); j++) {
                gcdVal = gcd(gcdVal, nums[j]);
                if (gcdVal == k) res++;
                else if (gcdVal % k == 0) continue;
                else break;
            }
        }
        return res;
    }
};
