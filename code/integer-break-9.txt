int i=0,j=0,m=0,max_pro=1,max_proNew=1;;

if(n<2)return n;

else
{
    for(i=2;i<=n;i++)
    {
        m = n;
		max_proNew=1;
        for(j=i;j>=1;j--)
        {
         max_proNew *= (m/j);
         m-=(m/j);
        }
        if(max_pro<=max_proNew)
        {
            max_pro = max_proNew;
        }
    }
}
    
return max_pro;
