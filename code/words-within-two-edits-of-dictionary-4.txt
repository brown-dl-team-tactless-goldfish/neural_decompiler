class Solution:
    def twoEditWords(self, queries: List[str], dictionary: List[str]) -> List[str]:
        v = []
        for i in queries:
            for j in dictionary:
                c,p,q = 0,0,0
                while(p<len(i) and q<len(j)):
                    if(i[p] != j[q]): c += 1
                    if(c>2): break
                    p += 1
                    q += 1
                if(c<=2):
                    v.append(i)
                    break
        return v
