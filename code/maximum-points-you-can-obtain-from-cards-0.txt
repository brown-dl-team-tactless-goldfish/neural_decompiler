int maxScore(int* cardPoints, int cardPointsSize, int k){
    int n = cardPointsSize;
    int* prefix = malloc((k+1) * sizeof(int));
    int* suffix = malloc((k+1) * sizeof(int));
    prefix[0] = 0;
    suffix[0] = 0;
    for(int i = 1; i <= k; i++){
        prefix[i] = prefix[i-1] + cardPoints[i-1];
        suffix[i] = suffix[i-1] + cardPoints[n-i];
    }
    int max = INT_MIN;
    for(int i = 0; i <= k; i++){
        max = fmax(max , prefix[i] + suffix[k-i]);
    }
    return max;
}
