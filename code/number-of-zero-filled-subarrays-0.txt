long long sum(int n){
    long long t = 0; 
    for(int i = 1; i <=n; i++){
        t += i;
    }
    return t;
}

long long zeroFilledSubarray(int* nums, int numsSize){
    long long ans = 0;
    int cn = 0;

    for(int i = 0; i < numsSize; i++){
        if(nums[i] == 0){
            cn++;
        }else{
            if(cn == 0)
                continue;
            ans += sum(cn);
            cn = 0;
        }    
    }
    ans += sum(cn);
    return ans;
}
