vector<vector<int>> rangeAddQueries(int n, vector<vector<int>>& queries) 
{
	vector<vector<int>> result(n,vector<int>(n));
	int s[n+1][n+1],t=queries.size();
	memset(s,0,sizeof(s));
	for(int i=0;i<t;i++)
		for(int k=queries[i][0];k<=queries[i][2];k++)
			s[k][queries[i][1]]++,s[k][queries[i][3]+1]--;

	for(int i=0;i<n;i++)
		for(int j=0;j<n;j++)
			result[i][j]=(j>0?result[i][j-1]:0)+s[i][j];

	return result;
}
