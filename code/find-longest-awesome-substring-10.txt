class Solution {
public:
    int longestAwesome(string s) {
        int n = s.size(), res = 0, seen[1024]{[0]=-1,[1 ... 1023]=n};
        for(int i = 0, cur = 0; i < n; i++) {
            cur ^= (1 << (s[i]-'0'));
            for(int j = 0; j < 10; j++) res = max(res, i-seen[cur^(1 << j)]);
            res = max(res, i-seen[cur]);
            seen[cur] = min(i, seen[cur]);
        }
        return res;
    }
};
