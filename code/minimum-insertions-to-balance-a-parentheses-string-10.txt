class Solution {
public:
    int minInsertions(string s) {
        int count=0,ans=0;
        for(int i=0;s[i];++i){
            if(s[i]=='(')
                count++;
            else{
                if(s[i+1]==')')
                    i++;
                else
                    ans++;
                if(count==0)
                    ans++;
                else
                    count--;
            }
        }
        return ans+2*count;
    }
};
