class Solution {
public:
    int mostFrequent(vector<int>& nums, int key) {
        map<int, int> mp;
        for(int i = 0; i <= nums.size()-2; i++) {
            if(nums[i] == key)
                mp[nums[i+1]] += 1;
        }
        
        int mx = -1;
        for(auto it: mp) 
            mx = max(mx, it.second);
        
        int ans = -1;
        for(auto it: mp) {
            if(it.second == mx)
                return it.first;
        }
        
        return ans;
    }
};
