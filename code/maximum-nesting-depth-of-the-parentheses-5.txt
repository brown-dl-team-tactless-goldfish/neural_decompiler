class Solution:
    def maxDepth(self, s: str) -> int:
        # print("s = ", s)

        stack = []
        res = 0
        for c in s:
            if (c == "("):
                stack.append(c)
            elif (c == ")"):
                # print("stack = ", stack)
                res = max(res, len(stack))
                stack.pop()

        return res
