// convert string to int
unsigned int hash33(char *key)
{
	unsigned int hash_value = 0;
	while(*key != '\0')
		hash_value = (hash_value << 5) + hash_value + *key++;
	return hash_value % STATION_SIZE;
}

int hash_search_insert(UndergroundSystem *obj, char *key)
{
	int hash_index = hash33(key);
	while(obj->station_name[hash_index] != 0)
	{
		if(strcmp(key, obj->station_name[hash_index]) == 0)
			return hash_index;
		hash_index = (hash_index + 1) % STATION_SIZE;
	}
	obj->station_name[hash_index] = strdup(key); // insert key if not found
	return hash_index;
}
