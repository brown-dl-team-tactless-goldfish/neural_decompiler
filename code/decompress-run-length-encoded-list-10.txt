/**
 * Note: The returned array must be malloced, assume caller calls free().
 */
int* decompressRLElist(int* nums, int numsSize, int* returnSize){
    register unsigned short i = 0, j, k = 0;
    register unsigned int returnArrayLength = 0;
	while(i < numsSize) {
		returnArrayLength += nums[i];
		i += 2;
	}

	*returnSize = returnArrayLength;
	int * res = (int*)malloc(returnArrayLength * sizeof(int));

	i = 0;
	while(i < numsSize) {
		for(j = 0; j < nums[i]; j++) {
			res[k] = nums[i+1];
			k++;
		}
		i += 2;
	}

	return res;
}
