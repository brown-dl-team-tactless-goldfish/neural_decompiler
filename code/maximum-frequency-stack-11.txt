multimap<int,int,greater_equal<int>> mp;

unordered_map<int, int> fre;

FreqStack() {
    
}

void push(int val) {
    
    mp.insert({fre[val]++, val});
    
}

int pop() {
    
    auto it = mp.begin();
    
    int val = it->second;
    
    mp.erase(it);
    
    fre[val]--;
    
    return val;
    
}
