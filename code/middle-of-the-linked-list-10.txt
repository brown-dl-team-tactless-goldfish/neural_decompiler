struct ListNode* middleNode(struct ListNode* head){
    if (!head->next) return head;   // only one node
    if (!head->next->next) return head->next;   //only two node

    struct ListNode *fast = head->next->next;
    while(head){
        if (!fast->next){
            return head->next;
        }
        else if (!fast->next->next){
            return head->next->next;
        }
        head = head->next;
        fast = fast->next->next;
    }
    return 0;
}
