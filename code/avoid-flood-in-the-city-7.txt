                  int last_seen = mp[rains[i]];
                  auto it =s.upper_bound(last_seen);
                  
                  if(it==s.end())
                      return {};
                  
                  int ind = *it;
                  ans[ind]=rains[i];
                  s.erase(ind);
              }
              mp[rains[i]]=i;
              ans.push_back(-1);
            }
    }   
    return ans;
}
