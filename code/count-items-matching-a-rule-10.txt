int countMatches(char *** items, int itemsSize, int* itemsColSize, char * ruleKey, char * ruleValue){
	int count = 0;

	for(int i = 0; i < itemsSize; i++){

		if ((!strcmp("color", ruleKey) && (!strcmp(items[i][1], ruleValue)))){
			count++;
		}

		else if ((!strcmp("name", ruleKey) && (!strcmp(items[i][2], ruleValue)))){
			count++;
		}

		else if ((!strcmp("type", ruleKey) && (!strcmp(items[i][0], ruleValue)))){
			count++;
		}
	}
	return count;
}
