int countPrefixes(char ** words, int wordsSize, char * s){
    int ans=0;
    for(int i=0;i<wordsSize;i++){
        int k=0,flag=0;
        for(int j=0;j<strlen(words[i]);j++){
            if(strlen(words[i])>strlen(s)){
                flag=1;
                continue;
            }
            if(words[i][j]==s[k]){
                k++;
            }
            else{
                flag++;
            }
        }
        if(flag==0){
            ans++;
        }
    }
    return ans;
}
