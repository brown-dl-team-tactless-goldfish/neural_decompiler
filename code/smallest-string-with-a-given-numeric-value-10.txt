char * getSmallestString(int n, int k){
    char *str= malloc ( sizeof(char) * (n+1));
    int i=0;
    for(i=0;i<n;i++){
        str[i]='a';
    }

 
    k-=n;
    if(k==0){
        str[n]='\0';
        return str;
    }
    i=n-1;
       
    while(k>25){
        str[i]='z';
        i--;
        k-=25;
    }
    str[i]=k+'a';
    str[n]='\0';
    return str;
}
