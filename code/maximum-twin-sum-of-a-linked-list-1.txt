void help(struct ListNode *head1, struct ListNode **head2, int *ans) {
    
    
    if(head1 == NULL)
        return;
    
    help(head1->next, head2, ans);
    
    int temp;
    
    temp = (head1->val + ((*head2)->val));
    if ((*ans) < temp)
        *ans = temp;
    
    (*head2) = (*head2)->next;
    
}

int pairSum(struct ListNode* head){
    
    if(head == NULL)
        return 0;
    
    int ans = INT_MIN;
    
    help(head, &head, &ans);
    return ans;
}
