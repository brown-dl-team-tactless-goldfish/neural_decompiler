int minSteps(char * s, char * t){
    int i = 0, k = 0;
    int a[123];
    int b[123];
    for (; i<123; i++){
        a[i] = 0;
        b[i] = 0;
    }
    for (i=0; i<strlen(s); i++){
        a[s[i]]++;
        b[t[i]]++;
    }
    for (i=97; i<123; i++){
        if (a[i] > b[i]){
            k += a[i]-b[i];
        }
    }
    return k;
}
