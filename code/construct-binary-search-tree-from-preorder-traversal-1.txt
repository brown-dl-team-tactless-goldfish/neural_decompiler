struct TreeNode* bstFromPreorder(int* preorder, int preorderSize){
    if(!preorderSize) return NULL;
    struct TreeNode* root = (struct TreeNode*)calloc(1, sizeof(struct TreeNode));
 
    root->val = preorder[0];
    int ridx = 0, lidx = 0;
	/* find right node first for left node's arr range */
    for(ridx = 1; ridx < preorderSize; ridx++){
        if(preorder[ridx] > preorder[0]){
            root->right = bstFromPreorder(preorder+ridx, preorderSize-ridx);
            break;
        }
    }
    for(lidx = 1; lidx < preorderSize; lidx++){
        if(preorder[lidx] < preorder[0]){
            root->left = bstFromPreorder(preorder+lidx, ridx-lidx);
            break;
        }
    }
       
    return root;
}
