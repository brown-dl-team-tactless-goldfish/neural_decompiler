


int cmpfunc (const void * a, const void * b) {
   return ( *(int*)a - *(int*)b );
}
int maximumBags(int* capacity, int capacitySize, int* rocks, int rocksSize, int additionalRocks){
    int counter=0;
    int *arr=malloc(capacitySize*(sizeof(int)));
    for (int i=0;i<capacitySize;i++)
    {
        *(arr+i)=capacity[i]-rocks[i];
        
    }
    qsort(arr,capacitySize,sizeof(int),cmpfunc);
    while(additionalRocks>0&& counter<capacitySize)
    {
        additionalRocks-=arr[counter];
        if(additionalRocks>=0)
        {
            counter++;
        }
    }
    free(arr);
    return counter;
}
