int findLHS(vector<int>& nums) {
    std::unordered_map<int, int> scan;
    for(auto& c : nums){
        ++scan[c];
    }
    
    int res = 0;
    for(auto& it : scan) {
        int anchor = it.first;
        if (scan.find(anchor + 1) != scan.end()) {
            res = std::max(res, it.second + scan[anchor + 1]);
        }
        
        if (scan.find(anchor - 1) != scan.end()) {
            res = std::max(res, it.second + scan[anchor - 1]);
        }
    }
    
    return res;
}
