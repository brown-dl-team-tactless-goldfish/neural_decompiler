bool containsPattern(vector<int>& arr, int m, int k) {
	if (m * k > arr.size())
		return false;
	for (int s = 0; s < arr.size() - m * k + 1; s++) {
		int curK = 0;
		for (int j = s; j < arr.size(); j += m) {
			int l = 0;
			bool bEqual = true;
			while (l < m) {
				if (arr[s + l] != arr[j + l]) {
					bEqual = false;
					break;
				}
				l++;
			}
			curK = bEqual == true ? ++curK : 0;
			if (curK == k)
				return true;
		}
	}
	return false;
}
