int countNumbersWithUniqueDigits(int n){
    if(n == 0)
        return 1;
    int* cn = malloc( n * sizeof(int));
    cn[0] = 10;
    if(n > 1)
        cn[1] = 81;
    int base = 8;
    for(int i = 2; i < n; i++){
        cn[i] = cn[i-1] * base;
        base--;
    }

    int ans = 0;
    for(int i = 0; i < n; i++){
        ans += cn[i];
    }

    return ans;
}
