int countElements(int* nums, int numsSize){
    int cnt=0;
 for(int i=0;i<numsSize;i++)
 {
    int a=0;
    int p=0;
     for(int j=0;j<numsSize;j++)
     {
         if(nums[i]>nums[j])
         {
             a++;
         }
         if(nums[j]>nums[i])
         {
             p++;
         }
         if(a>0 && p>0)
         {
             cnt++;
             break;
         }
     }
 }
 return cnt;
}
