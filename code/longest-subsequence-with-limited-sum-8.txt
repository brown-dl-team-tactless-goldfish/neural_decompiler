class Solution {
public:
    vector<int> answerQueries(vector<int>& nums, vector<int>& queries) {
        sort(nums.begin(),nums.end());
        vector<int> ans;
        for(auto i:queries){
            int msum = i;
            int c = 0;
            for(int j=0;j<nums.size();j++){
                if(msum-nums[j]>=0){
                    msum-=nums[j];
                    c++;
                }
            }
            ans.push_back(c);
        }
        return ans;
    }
};
