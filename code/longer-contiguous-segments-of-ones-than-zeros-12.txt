class Solution {
public:
    bool checkZeroOnes(string s)
    {
        iterator_traits<string::iterator>::difference_type len[2] = {0, 0};
        for (auto left = s.begin(); left != s.end(); )
        {
            auto right = find_if(left, s.end(), [=](char c) { return c != *left; });
            const auto idx = *left - '0';
            len[idx] = max(len[idx], right - left);
            left = right;
        }
        return len[1] > len[0];
    }
};
