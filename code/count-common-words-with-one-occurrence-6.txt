class Solution {
public:
    int countWords(vector<string>& w1, vector<string>& w2)
    {
       unordered_map<string,int>mp1,mp2;
        for(auto &it:w1)
        {
            mp1[it]++;
        }
        for(auto &it:w2)
        {
            mp2[it]++;
        }
        int count=0;
        for(int i=0;i<w1.size();i++)
        {
            if(mp1[w1[i]]==1&&mp2[w1[i]]==1)
            {
                count++;
            }
        }
        return count;
    }
};
