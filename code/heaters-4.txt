int findRadius(vector<int>& houses, vector<int>& heaters) 
{
    auto curHeater = 0, maxDistance = 0;
    auto heatersSize = heaters.size();
    
    if (heatersSize > 0)
    {
        sort(houses.begin(), houses.end());
        sort(heaters.begin(), heaters.end());
        for (auto house : houses)
        {
            while (curHeater + 1 < heatersSize && abs(house - heaters[curHeater + 1]) <= abs(house - heaters[curHeater]))
            {
                ++curHeater; // cycle to the closest heater.
            }
            
            maxDistance = max (maxDistance, abs(house - heaters[curHeater]));
        }
    }

    return maxDistance;
}
