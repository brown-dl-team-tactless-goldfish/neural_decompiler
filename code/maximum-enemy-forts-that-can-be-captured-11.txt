class Solution {
public:
    int captureForts(vector<int>& forts) {
        int l = 0, res = 0;
        for (int i = l + 1; i < forts.size(); ++i) {
            if (forts[i]) {
                if (forts[l] * forts[i] < 0) {
                    res = max(res, i - l - 1);
                }
                l = i;
            }
        }
        return res;
    }
};
