int numOfSubarrays(vector<int>& arr) {
    long long temp[2] = { 1, 0 }; 
    int n = arr.size();
    int result = 0, val = 0; 


    for (int i = 0; i <= n - 1; i++) { 
        val = ((val + arr[i]) % 2 + 2) % 2; 

        temp[val] = (temp[val]+1)%1000000007; 
    } 

    result = (int)((temp[0] * temp[1])%1000000007); 

    return (result); 
    
}
