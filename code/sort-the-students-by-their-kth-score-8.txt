class Solution {
public:
    vector<vector<int>> sortTheStudents(vector<vector<int>>& score, int k) {
        map<int,int,greater<int>>m;
        for(int i=0;i<score.size();i++) {
            m[score[i][k]]=i;
        }
        int row=0;
        for(auto it:m) {
            int tmp=score[row][k];
            swap(score[row],score[it.second]);
            m[tmp]=it.second;
            m[it.first]=row;
            row++;
        }
        return score;
    }
};
