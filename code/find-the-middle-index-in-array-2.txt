int findMiddleIndex(int* nums, int numsSize){
    int tsum = 0,csum=0; 
    for(int i = 0; i<numsSize; i++) //loop to find total sum of array and store in tsum variable
        tsum+=nums[i];
    for(int i = 0; i<numsSize; i++)
    {
        csum+=nums[i]; //summation of array elements upto current i
        if(csum-nums[i] == tsum-csum) //csum-nums[i] gives left side sum of the array and tsum-csum gives right side sum of the array/
            return i;
    }
    return -1;
}
