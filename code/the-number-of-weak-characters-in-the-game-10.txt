class Solution {
    static bool cmp(vector<int> &a, vector<int> &b) {
        return a[0] == b[0] ? a[1] > b[1] : a[0] < b[0]; 
    }
    
public:
    int numberOfWeakCharacters(vector<vector<int>>& p) {
        sort(p.begin(), p.end(), cmp);
        
        int ans = 0, mx = INT_MIN;
        for(int i=p.size() - 1; i>=0; i--) {
            mx = max(mx, p[i][1]);
            if(p[i][1] < mx) ans++;
        }
        
        return ans;
    }
};
