int numWaterBottles(int numBottles, int numExchange)
{
    int count = 0, result = 0;
    
    while(numBottles!=0)
    {
        count++;
        result++;
        if(count==numExchange)
        {
            result = result+1;
            count = 1;
        }
         numBottles--;
    }
    
    return result;
}
