public class Solution {
    
    public int MaximumBags(int[] capacity, int[] rocks, int additionalRocks) {
        var deltas = capacity
            .Zip(rocks, (c, r) => c - r)
            .OrderBy(item => item);

        int result = 0;    
            
        foreach (int delta in deltas) {
            if (additionalRocks < delta)
                break;
            
            additionalRocks -= delta;
            
            result += 1;
        }
        
        return result;
    }
    
}
