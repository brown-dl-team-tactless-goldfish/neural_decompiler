class Solution {
public:
    long long div(long long a, long long b) {
        return (a + b - 1) / b;
    }
    int minElements(vector<int>& nums, int limit, int goal) {
        int n = nums.size();
        long long sum = 0;
        for(int i : nums)
            sum += i;
        int ans = 0;
        if(goal > sum) {
            long long diff = goal - sum;
            ans += div(diff, limit);
        }
        else if (sum > goal) {
            long long diff = sum - goal;
            cout << diff << endl;
            ans += div(diff, limit);
        }
        return ans;
    }
};
