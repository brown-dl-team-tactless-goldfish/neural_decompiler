#define MAX(A,B) ((A > B) ? A : B)

int maxSum(int** grid, int gridSize, int* gridColSize){
    int maxSum = 0;
    for (int row = 0; row < gridSize-2; row++){
        for (int col = 0; col < *gridColSize-2; col++){
            // Creating a variable with the value of half the hourglass assigned
            int auxSum = grid[row+1][col+1];
            
            // Calculating the sum of the top and bottom of the hourglass
            for (int i = col; i <= col+2; i++){
                auxSum += grid[row][i] + grid[row+2][i];
            }
            maxSum = MAX(maxSum, auxSum);
        }
    }
    return maxSum;
}
