char * convertToTitle(int n){
    if(!n) return NULL;
    int r,count=0;
    char *c = malloc(8);
    char *dst = malloc(8);
    while(n){
        n = n-1;
        r = n%26;
        c[count++]=r +'A';
        n /=26;
    }
    
    for(int i=count-1;i>=0;i--){
        dst[count-1-i] = c[i];
    }
   
    dst[count]='\0';
    return dst;
}
