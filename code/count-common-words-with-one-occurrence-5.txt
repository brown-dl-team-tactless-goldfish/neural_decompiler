int countWords(vector<string>& words1, vector<string>& words2) {
	map<string, int> freq1;
	for (const string& w : words1)
		freq1[w]++;
	map<string, int> freq2;
	for (const string& w : words2)
		freq2[w]++;
	int res = 0;
	for (const auto& it : freq1)
		if (it.second == 1 && freq2[it.first] == 1)
			res++;
	return res;
}
