
void *cmpfunc(void *a, void *b){
    return (*(int*)a - *(int*)b);
}

int* mostVisited(int n, int* rounds, int roundsSize, int* returnSize){

    int count =1, pos = rounds[0];
    int *output = (int*)malloc(sizeof(int)*(n+1));
    output[0] = pos;
    while(pos != rounds[roundsSize-1]) {
        pos %=n;
        output[count++] = ++pos;
    }
    qsort(output, count, sizeof(int), cmpfunc);
    *returnSize = count;
    return output;
}
