class Solution {
public:
    bool checkValidString(string s) {
        stack<char> stk;
        int stars = 0;
        int val = 0;
        for (char& c : s) {
            if (c == '(') {
                val++;
            } else if (c == ')') {
                val--;
            } else if (c == '*') {
                stars++;
            }
            if (val + stars < 0) return false;
            stk.push(c);
        }
        stars = 0;
        val = 0;
        while (!stk.empty()) {
            if (stk.top() == '(') {
                val--;
            } else if (stk.top() == ')') {
                val++;
            } else if (stk.top() == '*') {
                stars++;
            }
            if (val + stars < 0) return false;
            stk.pop();
        }
        return true;
    }
};
