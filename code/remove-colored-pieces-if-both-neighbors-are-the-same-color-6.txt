class Solution {
public:
    bool winnerOfGame(string colors) {
        int a = 0 ,b = 0, cntA = 0, cntB = 0;
        for(char c : colors){
            if(c == 'A')
                a++ , b = 0;
            else
                b++, a = 0;
            if(a >= 3) cntA++;
            if(b >= 3) cntB++;
        }
        return cntA > cntB;
    }
};
