int oddCells(int n, int m, int** indices, int indicesSize, int* indicesColSize){
    int matrix[n][m], i, j, odds=0;
    for(i=0;i<n;i++)
        for(j=0;j<m;j++)
            matrix[i][j]=0;
    for(i=0;i<indicesSize;i++){
        for(j=0;j<m;j++)    
            matrix[indices[i][0]][j]^=1;
        for(j=0;j<n;j++)
            matrix[j][indices[i][1]]^=1;
    }
    for(i=0;i<n;i++)
        for(j=0;j<m;j++)
            odds+=matrix[i][j];

    return odds;
}
