#define xor ^
#define max(a,b) (a > b ? a : b)

int findMaximumXOR(int* nums, int numsSize) {
    int i, j, answer = 0;
    for(i = 0; i < numsSize-1; ++i)
    {
        for(j = i+1; j < numsSize; ++j)
        {
            answer = max(answer, (nums[i] xor nums[j]));
        }
    }
    
    return answer;
}
