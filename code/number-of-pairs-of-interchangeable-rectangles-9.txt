class Solution {
public:
    long long interchangeableRectangles(vector<vector<int>>& r) {
        unordered_map<double, int> m;
        for(auto &e : r){
            double ratio = (double) e[0]/ e[1];
            m[ratio]++;
        }
        long long ans = 0;
        
        for(auto &e : m){
         ans += e.second * 1LL*(e.second - 1) >> 1;    
        }
        return ans;
    }
};
