class Solution {
public:
    int minTime(int n, vector<vector<int>>& edges, vector<bool>& hasApple) {
    
        int count = 0;
        for (int i = edges.size()-1; i >= 0; i--) {
            if (hasApple[edges[i][1]]) {
                count++;
                hasApple[edges[i][0]] = true;
            }
        }
        
        return count*2;
    }
};
