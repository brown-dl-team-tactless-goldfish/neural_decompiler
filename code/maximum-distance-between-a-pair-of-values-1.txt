int maxDistance(int* nums1, int nums1Size, int* nums2, int nums2Size){
    int first=0,second=0,d=0;
    while(first<nums1Size && second<nums2Size)
    {
        if(first<=second && nums1[first]<=nums2[second])
        {
            if(d<(second-first))
            {
                d=second-first;
            }
        second++;    
        }
        else if(nums1[first]>nums2[second])
        {
        first++;
        }
        
        else
        second++;

    }
    return d;
}
