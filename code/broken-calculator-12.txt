public:
    int brokenCalc(int x,int y){
        double p = log2 ( double( y ) / x);
        int a = p;
        if( a < 0 || y < x ) a = 0;
        else if(a != p) a++;
        int b = pow(2 , a)*x - y;
        int c = a ,d = 0;
        while(c > 0){
            d += (b & 1);
            b = b >> 1;
            c--;
        }
        return b + d + a;
    }
};
