int cmpfunc (const int * a, const int * b) {
   return ( *b - *a );
}

int minDeletions(char * s){
    int len=strlen(s);
    int res[26]={};
    int count = 0;

    for(int i=0;s[i]!='\0'; i++) res[s[i]-'a']++;

    qsort(res,26, sizeof(int), cmpfunc);

    for(int i=1;i<26;i++){
        if(res[i]==0) continue;
        while(res[i]>=res[i-1]){
            count++;
            res[i]--;
            if(res[i]==0) break;
        }
    }
    return count;
}
