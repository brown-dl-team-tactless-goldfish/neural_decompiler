struct ListNode* middleNode(struct ListNode* head){
    int i = 0;
    struct ListNode *new = head;
    while (new->next != NULL)
    {
        new = new->next;
        i++;
    }
    i = (i % 2 == 0) ? i / 2 : i / 2 + 1;
    new = head;
    while (i != 0)
    {
        new = new->next;
        i--;
    }
    return new;
}
