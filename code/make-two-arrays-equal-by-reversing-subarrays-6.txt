bool canBeEqual(int* target, int targetSize, int* arr, int arrSize){
    if (targetSize != arrSize)
        return false;
    
    int i, *hash = calloc(sizeof(int), 1001);
    
    for (i = 0; i < targetSize; i++) {
        hash[target[i]]++;
        hash[arr[i]]--;
    }
     
    for (i = 0; i < 1001; i++) {
        if (hash[i])
            return false;
    }
    
    return true;
}
