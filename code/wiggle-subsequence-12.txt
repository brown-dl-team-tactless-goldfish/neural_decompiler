int wiggleMaxLength(int* nums, int numsSize){
    int big=1, small=1, ret=1;
    
    for(int i=0;i<numsSize-1;i++){
        if(nums[i]<nums[i+1]&&big) {big=0; small=1; ret++;}
        else if(nums[i]>nums[i+1]&&small) {big=1; small=0; ret++;}
    }
    
    if(numsSize==0) return 0;
    else return ret;
}
