class Solution {
public:
    vector<vector<int>> findMatrix(vector<int>& nums) {
        int row=0;
        unordered_map<int,int>m;
        for(auto it:nums) {
            m[it]++;
            row=max(row,m[it]);
        }
        vector<vector<int>>v(row);
        for(auto it:m) {
            for(int i=0;i<it.second;i++) {
                v[i].push_back(it.first);
            }
        }
        return v;
    }
};
