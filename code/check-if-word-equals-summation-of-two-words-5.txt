

bool isSumEqual(char * firstWord, char * secondWord, char * targetWord){
    int sumFirstWord = 0;
    int sumSecondWord = 0;
    int sumTargetWord = 0;
    int lenFirst = strlen(firstWord);
    int lenSecond = strlen(secondWord);
    int lenTarget = strlen(targetWord);
    int i;
    for (i = 0; i < lenFirst; i++){
        sumFirstWord += pow(10, lenFirst-1-i)*(firstWord[i]-'a');
    }
    for (i = 0; i < lenSecond; i++){
        sumSecondWord += pow(10, lenSecond-1-i)*(secondWord[i]-'a');
    }
    for (i = 0; i < lenTarget; i++){
        sumTargetWord += pow(10, lenTarget-1-i)*(targetWord[i]-'a');
    }
   
    return (sumFirstWord+sumSecondWord)==sumTargetWord;
}
