char * reverseStr(char * s, int k){
int ptr=0;
char swap;
int i;
    
while (ptr < strlen(s)) {
    if ((ptr + k) > strlen(s))
        k=strlen(s)-ptr;
    for (i=0; i<(k/2); i++) {
        swap=s[ptr+i];
        s[ptr+i]=s[ptr+k-i-1];
        s[ptr+k-i-1]=swap;  
    }
    ptr += 2*k;
}

return s;
}
