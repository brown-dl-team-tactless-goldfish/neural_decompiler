

/**
 * Note: The returned array must be malloced, assume caller calls free().
 */
int* distributeCandies(int candies, int n, int* r){
    *r=n;
    int i = 1;
    int * ret = (int *)calloc((*r),sizeof(int));
    while(candies>0){
        if(candies>i)
            ret[(i-1)%n]+=i;
        else
            ret[(i-1)%n]+=candies;
        candies-=i;
        i++;
    }
    return ret;
}
