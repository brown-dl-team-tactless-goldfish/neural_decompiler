class Solution {
public:
    long long getDescentPeriods(vector<int>& prices) {
        long long res = 0;
        int count = 0;
        for(int i=0;i<prices.size();i++) {
            if(i==0 || prices[i] == prices[i-1]-1) {
                count++;
            }else {
                count = 1;
            }

            res += count;
        }
        return res;
    }
};
