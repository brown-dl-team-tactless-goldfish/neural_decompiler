class Solution {
public:
    vector<int> constructArray(int n, int k) {
        vector<int> ans;
        int x = 1;
        int c = k+1;
        int t = k;
        while(t > 0)
        {
            ans.push_back(x++);
            ans.push_back(c--);  
            t -= 2;

        }
        if(x != ans.back())
            ans.push_back(x);
        
        x = k+2;
        
        while(x <= n)
            ans.push_back(x++);
        
        return ans;
    }
};
