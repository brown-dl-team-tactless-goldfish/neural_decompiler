int maximumSum(vector<int>& nums) 
{
	int result=-1,n=nums.size();
	sort(nums.rbegin(),nums.rend());
	vector<int> total[100];
	for(int i=0;i<n;i++)
	{
		int c=0,d=nums[i];
		while(d) c+=d%10,d/=10;
		total[c].push_back(nums[i]);
	}
	for(int i=0;i<100;i++)
		if(total[i].size()>1) result=max(result,total[i][0]+total[i][1]);

	return result;
}
