class Solution:
    def sumSubarrayMins(self, arr: List[int]) -> int:
        stack,s = [],[0]*len(arr)
        for i in range(len(arr)):
            while stack and arr[stack[-1]] > arr[i]:
                stack.pop()
            j = stack[-1] if stack else -1
            s[i] = s[j] + arr[i]*(i-j)
            stack.append(i)
        return sum(s) % (10**9+7)
