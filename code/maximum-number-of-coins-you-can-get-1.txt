int cmp(const void* a, const void* b) {
    return *(const int*) b - *(const int*) a;
}

int maxCoins(int* piles, int pilesSize) {
    int size = pilesSize - pilesSize / 3;
    qsort(piles, pilesSize, sizeof(int), cmp);
    int ans = 0;
    for (int i = 1; i < size; i+=2) {
        ans += piles[i];
    }
    return ans;
}
