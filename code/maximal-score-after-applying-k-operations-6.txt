class Solution {
public:
    long long maxKelements(vector<int>& nums, int k) 
    {
        priority_queue<int> p;
        for(auto i : nums)
        {
            p.push(i);
        }
        long max=0;
        for(int i=k; i>0; i--)
        {
            int c = p.top();
            p.pop();
            max+=c;
            int f=(int)ceil(c/3.0);
            p.push(f);
        }
        return max;    
    }
};
