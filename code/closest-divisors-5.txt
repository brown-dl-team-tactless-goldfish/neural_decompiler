public int[] ClosestDivisors(int num) {
	var divs1 = GetClosestDivisors(num + 1);
	var divs2 = GetClosestDivisors(num + 2);
	var diff1 = divs1[1] - divs1[0];
	var diff2 = divs2[1] - divs2[0];

	return diff1 < diff2 ? divs1 : divs2;

	static int[] GetClosestDivisors(int num) {
		for (var i = (int)Math.Sqrt(num); i > 1; i--)
			if (num % i == 0)
				return new[] { i, num / i };
		return new[] { 1, num };
	}
}
