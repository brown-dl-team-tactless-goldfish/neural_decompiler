bool checkString(char * s){
    int n = strlen(s);
    // 0: reads As
    // 1: reads Bs
    int state = 0;
    for (int i = 0; i < n; i++) {
        char c = s[i];
        if (state == 0 && c == 'b') {
            state = 1;
        }
        if (state == 1 && c == 'a') {
            // invalid string
            return false;
        }
    }
    return true;
}
