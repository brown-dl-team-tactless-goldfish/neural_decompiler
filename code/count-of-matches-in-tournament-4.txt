int numberOfMatches(int n){
    int played=0;
    while(n > 1) {
        // if n is an odd number..
        if (n & 1) {
            played += ((n-1) >> 1);
            n = ((n-1) >> 1) + 1;
        } else {
            played += (n >> 1);
            n >>= 1;
        }
    }
    
    return played;
}
