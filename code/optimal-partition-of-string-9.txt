class Solution {
public:
    int partitionString(string s) {
        int arr[26]={0};
        int cnt=0;
        for(int i=0;i<s.size();i++){
            if(arr[s[i]-'a']>=1){
                cnt++;
                memset(arr, 0, sizeof(arr));
            }
            arr[s[i]-'a']++;
        }
        return cnt+1;
    }
};
