bool checkPerfectNumber(int num) {
    if(num<6){return false;}
    int sum=num-1;
    for(int i=2;i<=num/i;i++){
        if(num%i==0){
            if(i!=num/i){sum=sum-i-num/i;}
            else {sum=sum-i;}
        }
        if(sum<0){return false;}
    }
    if(sum!=0){return false;}
    return true;
}
