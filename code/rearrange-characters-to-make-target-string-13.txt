class Solution {
public:
    int rearrangeCharacters(string s, string target) {
        unordered_map<char, int> m, m2;
        for(auto c : s)
            m[c]++;
        for(auto c : target)
            m2[c]++;
        int copies = INT_MAX;
        for(auto k : m2){
            char c = k.first;
            copies = min(copies, m[c]/k.second);
        }
        return copies;
    }
};
