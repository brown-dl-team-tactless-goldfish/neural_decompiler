 vector<int> findArray(vector<int>& pref) {
        vector<int> ans;
        ans.push_back(pref[0]);
        
        int prev_xor=pref[0];
        for(int i=1;i<pref.size();i++){
            int curr=pref[i];
            curr=prev_xor^curr;
            ans.push_back(curr);
            prev_xor=prev_xor^curr;
        }
        return ans;
    }
