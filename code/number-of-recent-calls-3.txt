typedef struct {
    int data[10000];
    int now;
} RecentCounter;


RecentCounter* recentCounterCreate() {
    RecentCounter * obj = calloc(1, sizeof(RecentCounter));
    return obj;
}

int recentCounterPing(RecentCounter* obj, int t) {
    int top, count=0;
    obj->data[obj->now++] = t;
    top= obj->now;
    while (top>0 && obj->data[--top] >= t-3000)  count++;
    return count;
  
}

void recentCounterFree(RecentCounter* obj) {
    free(obj);
}
