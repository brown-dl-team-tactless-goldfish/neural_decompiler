class Solution {
public:
    using fun = function<string(int)>;
    char findKthBit(int n, int k) {
        fun go = [&](int i) -> string {
            if (!i)
                return "0";
            auto pre = go(i - 1),
                 rev = pre;
            transform(rev.begin(), rev.end(), rev.begin(), [](auto c) { return c == '0' ? '1' : '0'; });
            return pre.append("1").append({ rev.rbegin(), rev.rend() });
        };
        return go(n - 1)[k - 1];
    }
};
