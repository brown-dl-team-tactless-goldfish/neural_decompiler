class Solution {
public:
    int longestZigZag(TreeNode* root) {
        dfs(root);
        return maxDepth;
    }
    
    //return left and right zig depth 
    pair<int,int> dfs(TreeNode* root){
        if(!root) return {-1,-1};
        auto [x1, x2] = dfs(root->left);
        auto [y1, y2] = dfs(root->right);
        int z1 = 1+x2, z2 = 1+y1;
        maxDepth = max({maxDepth, z1, z2});
        return {z1, z2};
    }
    
    int maxDepth = 0;
};
