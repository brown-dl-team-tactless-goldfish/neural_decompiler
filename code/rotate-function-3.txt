int maxRotateFunction(int* A, int ASize){
    if(ASize<2){
        return 0;
    }
    long int sum=0;
    for(int i=0;i<ASize;i++){
        sum+=A[i]*i;
    }
    long int max=sum;
    long int del=0;
    for(int i=1;i<ASize;i++){
        del+=A[i];
    }
    for(int i=0;i<ASize-1;i++){
        sum+=A[i]*(ASize-1);
        sum-=del;
        del+=A[i]-A[i+1];
        if(max<sum){
            max=sum;
        }
    }
    return (int)max;
}
