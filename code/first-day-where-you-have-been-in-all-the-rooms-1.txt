class Solution {
public:
    int firstDayBeenInAllRooms(vector<int>& nextVisit) {
        int n = nextVisit.size() ;
        int M = 1e9 + 7 ;
        vector<long long>dp(n) ;
        dp[0] = 0 ;
        for(int i = 0; i < n - 1; i++){
            int j = nextVisit[i] ;
            //dp[i+1] = 第一次到達i , 跳回 j (步數+1) + dp[i] - dp[j]
            //          i --> i+1, 再加 1   
            dp[i+1] = ((dp[i] + 1 + dp[i] - dp[j] + M) + 1 ) % M ;
        }
        return dp[n-1] ;
    }
};
