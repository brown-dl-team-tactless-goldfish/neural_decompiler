class Solution {
public:
    int maxCoins(vector<int>& piles) {
        int n = piles.size();
        //sorting the piles
        sort(piles.begin(),piles.end());
        //initializing the pointers
        int i = 0;
        int j = n-1;
        int sol = 0;
        
        while(i<j)
        { 
            j--;
            i++;
            //you will only choose the middle element of the triplet.
            sol += piles[j];
            j--;
        }
        return sol;
        
    }
};
