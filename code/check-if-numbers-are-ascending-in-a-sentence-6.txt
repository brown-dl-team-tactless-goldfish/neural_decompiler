class Solution:
    def areNumbersAscending(self, s: str) -> bool:
        data = []
        for element in s.split(" "):
            print ("element = ", element)
            flag = True
            for c in element:
                if (c not in "0123456789"):
                    flag = False
                    break
                    
            if (flag == True):
                tmpInt = int(element)
                print ("tmpInt = ", tmpInt)
                if (data == []):
                    data.append(tmpInt)
                else:
                    if (tmpInt <= data[-1]):
                        return False
                    else:
                        data.append(tmpInt)
              
        print ("data = ", data)
        if (data == []):
            return False
        else:
            return True
