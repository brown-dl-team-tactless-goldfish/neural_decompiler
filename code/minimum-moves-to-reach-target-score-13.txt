class Solution {
public:
    int minMoves(int target, int maxDoubles){
        
        int ans = 0;
        
        while(target > 1){
            while(target > 1 and target%2 == 0 and maxDoubles > 0){target /= 2; ans++; maxDoubles--;}
            if(maxDoubles == 0){ ans += target-1 ; target = 1;}
            if(target > 1) {target--; ans++;}   
        }
        return ans;
    }
};
