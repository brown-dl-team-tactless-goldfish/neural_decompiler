public class Solution {
    public int MatchPlayersAndTrainers(int[] players, int[] trainers) {
        Array.Sort(players);
        Array.Sort(trainers);
        
        int result = 0;
        
        for (int p = 0, t = 0; p < players.Length && t < trainers.Length; ++t) 
            if (players[p] <= trainers[t]) {
                p += 1;
                
                result += 1;
            }
        
        return result;
    }
}
