int leaf(struct TreeNode* root,int sum){ 
    if(root==NULL)
        return 0;
    if(root->left!=NULL && root->left->left==NULL && root->left->right==NULL){
        return root->left->val+leaf(root->left,sum)+leaf(root->right,sum);
    } 
     return leaf(root->left,sum)+leaf(root->right,sum);

}
int sumOfLeftLeaves(struct TreeNode* root){
    int sum=0;
    if(root==NULL)
        return 0;
     sum=leaf(root,sum);
return sum;
}
