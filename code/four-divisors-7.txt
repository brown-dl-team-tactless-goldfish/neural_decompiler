class Solution {
public:
    int sumFourDivisors(vector<int>& nums) {
        int sum = 0;
        for (int i = 0; i < nums.size(); i++) {
            vector<int> vDivisors = calculateDivisors(nums[i]);
            if (vDivisors[0] == 4)
                sum += vDivisors[1];
        }
        return sum;
    }
private:
    vector<int> calculateDivisors(int num) {
        int count = 0, sum = 0;
        for (int i = 1; i <= sqrt(num); i++) {
            if (num % i == 0) {
                count++;
                sum += i;
                if (i * i != num) {
                    count++;
                    sum += (num / i);
                }
            }
        }
        return { count, sum };
    }
};
