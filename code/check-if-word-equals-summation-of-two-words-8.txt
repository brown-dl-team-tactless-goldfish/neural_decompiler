bool isSumEqual(char * a, char * b, char * c)
{
    int k=0,la=strlen(a)-1,lb=strlen(b)-1,lc=strlen(c)-1;
    while(la>=0||lb>=0)
    {
        if(lc<0)break;
        int s1=0,s2=0;
        if(la>=0)s1=a[la]-'a';
        if(lb>=0)s2=b[lb]-'a';
        int s=s1+s2+k;
        if(s>9)
        {
            s=s%10;
            k=1;
        }
        else
            k=0;
        lc--;
        la--;
        lb--;
    }
    if(k&&la<0&&lb<0&&lc<0)return false;
    return true;
}
