class Solution {
public:
    void solve(long long n,long long least,vector <long long> &res){
        if (n<=2*(least+2)){
            if (n!=0) res.push_back(n);
            return;
        }
        solve(n-(least+2),least+2,res);
        res.push_back(least+2);
    }
    vector<long long> maximumEvenSplit(long long finalSum) {
        ios_base::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);
        vector <long long> res;
        if (finalSum%2) return res;
        solve (finalSum,0,res);
        return res;
    }
};
