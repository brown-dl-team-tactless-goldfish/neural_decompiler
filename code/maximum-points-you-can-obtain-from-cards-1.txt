int maxScore(int* cardPoints, int cardPointsSize, int k){
    
    int sum = 0;
    int a = 0;
    int l_w = 0;
    int h_w = cardPointsSize - k-1;
    for (int i = h_w+1; i < cardPointsSize; i++) {
        a += cardPoints[i];
    }
    sum = a;
    h_w++;
    l_w ++;
    while (h_w < cardPointsSize) {
        sum -= cardPoints[h_w];
        sum += cardPoints[l_w-1];
        h_w++;
        l_w++;
        if (sum > a)
            a = sum;
    }
    
    return a;
}
