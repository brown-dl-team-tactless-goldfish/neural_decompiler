public class Solution 
{
    public int SumSubseqWidths(int[] arr) 
    {
        Array.Sort(arr);
        long res = 0, mod = 1_000_000_007, pow = 1, n = arr.Length;
        for(int i = 0; i < arr.Length; i++, pow = (2 * pow) % mod)
            res =  (res + (long) arr[i] * pow - (long) arr[n - i - 1] * pow) % mod;
        return  (int) res;
    }
}
