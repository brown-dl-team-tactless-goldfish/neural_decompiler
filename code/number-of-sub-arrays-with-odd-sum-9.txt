#define MOD 1000000007

class Solution {
public:
    int numOfSubarrays(vector<int>& arr) {
        std::vector<int> m{0, 0};
        int res{0}, acc{0};
        for(const auto &n: arr) {
            acc += n;
            ++m[acc&1];
            res = (res + m[!(acc&1)]+(acc&1)) % MOD;
        }
        return res;
    }
};
