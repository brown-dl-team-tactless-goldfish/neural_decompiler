double calculateTax(int** brackets, int bracketsSize, int* bracketsColSize, int income){
    if(income == 0) return 0;
    
    int idx = 0, preUpper = 0, ret = 0;
    double ans = 0, currDollar;
    
    while(idx < bracketsSize && ret == 0)
    {
        if(brackets[idx][0] > income)
        {
            ret = 1;
            currDollar = income - preUpper;
        }
        else
        {
            currDollar = brackets[idx][0] - preUpper;
        }
        ans += currDollar * brackets[idx][1] / 100;
        preUpper = brackets[idx++][0];
    }
    return ans;
}
