SELECT s.id, s.visit_date, s.people
FROM (SELECT a.id as start, b.id as end
      FROM (SELECT id, people FROM stadium WHERE id-1 not in (SELECT id FROM stadium WHERE people >= 100)) a,
            (SELECT id FROM stadium WHERE id+1 not in (SELECT id FROM stadium WHERE people >= 100)) b
    WHERE a.id < b.id AND a.people >= 100 GROUP BY a.id) c JOIN stadium s ON s.id BETWEEN c.start AND c.end
WHERE c.end - c.start >= 2
