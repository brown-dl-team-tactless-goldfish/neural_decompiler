if(num<0)
{
   num = num + UINT_MAX;
}

int length=0;
char word[]="0123456789abcdef";

for(int i=0;num>=pow(16,i);i++)
{
    length++;
}

if(num==0)
{
    length=1;
}

char* ans=malloc(sizeof(char)*(length+1));
ans[length]='\0';

for(int j=0;j<length;j++)
{
    unsigned int a=(num/pow(16,length-1-j));
    ans[j]=word[a];
    num=num%(int)pow(16,length-1-j);
}

return ans;
