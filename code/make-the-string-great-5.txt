char * makeGood(char * s){
    int len = strlen(s);
    bool rem = true;
    while (rem) {
        rem = false;
        for (int i=0; i<len-1; i++) {
            if (tolower(s[i]) == tolower(s[i+1]) && s[i] != s[i+1]) {
                memmove(s+i, s+i+2, len-i-2);
                s[len-1] = '\0';
                s[len-2] = '\0';
                len -= 2;
                rem = true;
            }
        }
    }
    return s;
}
