bool checkPerfectNumber(int num)
{
    int sq;
    int i;
    int sum;
    sq = sqrt(num) + 1;
   
    if( num == 1 )
    {
        return false;
    }

    sum = 1;
    for(i=2; i<sq; i++)
    {
        if( num%i == 0 )
        {
            sum += ( i + (num/i) );  
        }
    }

    if( sum == num)
    {
        return true;
    }
    else
    {
        return false;
    }
}
