    public string MaximumNumber(string num, int[] change) {
        var ret=new StringBuilder();
        bool mutated=false;
        foreach(var ch in num)
        {
            int id=ch-'0';
            int val=change[id];
            if(mutated==true&&id>val)
            {
                ret.Append(num.Substring(ret.Length));
                return ret.ToString();
            }
            if(val>id)
            {
                ret.Append(val.ToString());
                mutated=true;
            }
            else
            {
                ret.Append(id.ToString());
            }
        }
        return ret.ToString();
    }
