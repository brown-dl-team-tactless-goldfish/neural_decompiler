int fastpow(int base,int pow,int k){
    long long ans=1;
    while(pow>0){
        if(pow%2==0){
            base=(base*base)%k;
            pow/=2;
        }
        else{
            pow--;
            ans=(ans*base)%k;
        }
    }
    return ans%k;
}
int superPow(int a, int* b, int size){
    const int k=1337;
    a=a%k;
    int i,ans=1,sub=a;
    for(i=size-1;i>-1;i--){
        ans=(ans*fastpow(sub,b[i],k))%k;
        sub=fastpow(sub,10,k);
            
        
    }
    return ans;
}
