int* diStringMatch(char * S, int* returnSize){
    int tmp=0;
    int n = strlen(S);
    int Nmax = n, Nmin = 0;
    int *dst = (int*)malloc((n+1)*sizeof(int));
    
    while(*S){
        if(*S == 'D'){
            dst[tmp]=Nmax;
            Nmax = Nmax -1;
        }else{
            dst[tmp]=Nmin;
            Nmin = Nmin+1;
        }
        S++;
        tmp++;
    }
    
    dst[tmp]=Nmax;
    *returnSize = tmp+1;
    return dst;
}
