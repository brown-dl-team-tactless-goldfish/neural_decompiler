class Solution {
public:
    bool canConvertString(string s, string t, int k) {
        if(s.size() != t.size() ) return false;
        vector<int> arr;
        map<int, int> m;
        int mx = 0;
        for(int i = 0; i < s.size(); ++i) {
            int c1 = s[i], c2 = t[i];
            if(s[i] == t[i]) continue;
            int step = (c2 - c1 + 26) % 26;
            ++m[step];
            mx = max(mx, step + (m[step] - 1) * 26 );
        }
        return k >= mx;
    }
};
