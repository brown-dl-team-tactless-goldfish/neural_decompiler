int findClosestNumber(int* nums, int numsSize){
    int closestValue=1000000,closestIndex,i;
    for(i=0;i<numsSize;i++){
		if(abs(nums[i])==closestValue){
			if(nums[i]>nums[closestIndex]){
				closestValue = nums[i];
				closestIndex = i;
			}
		}
		if(abs(nums[i])<closestValue){
			closestValue = abs(nums[i]);
			closestIndex = i;
		}
	}
    return nums[closestIndex];
}
