class Solution {
public:
    int distinctAverages(vector<int>& nums) {
        set<int> avgs;
        sort(nums.begin(), nums.end());
        int l = 0;
        int r = nums.size()-1;
        while(l < r){
            int a = nums[l];
            int b = nums[r];
            int avg = (a + b); //can avoid problem of fractional division by skipping the division entirely
            avgs.insert(avg);
            l++;
            r--;
        }
        return avgs.size();
    }
};
