class Solution {
public:
    vector<long long> sumOfThree(long long num) {
        vector<long long> x;
        if(num%3!=0)
        {
            return x;
        }
        else
        {
            long long r = (num/3)-1;
            x.push_back(r);
            x.push_back(r+1);
            x.push_back(r+2);
            return x;
        }
        
        
    }
};
