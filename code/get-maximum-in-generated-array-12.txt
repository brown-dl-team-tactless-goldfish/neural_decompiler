class Solution:
    def getMaximumGenerated(self, n: int) -> int:
        if (n == 0):
            return 0
        elif (n == 1):
            return 1
        
        dp = [0,1]
        for i in range(2, n+1):
            if (i % 2 == 0):
                idx = i // 2
                dp.append(dp[idx])
            else:
                idx = i // 2
                dp.append(dp[idx] + dp[idx+1])
                
        print ("dp = ", dp)
        return max(dp)
