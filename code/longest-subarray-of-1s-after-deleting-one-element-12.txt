    int n = nums.size();
    
    int start = 0;
    int ans = 0 ;
    int c = 0 ;
    
    for(int i=0 ; i<n ; i++){
        
        if(nums[i]!=1){
            c++ ;
        }
        while(c>1){
            ans = max(ans , i - start - 1 );
            if(nums[start]==0){
                c-- ;
            }
            start++;
            
        }
    }
    
    
    
    ans = max(ans , n - start - 1) ;
    
    
    return ans ;
}
