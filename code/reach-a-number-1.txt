int reachNumber(int target){
    target = abs(target) ;
    int step = sqrt(2*target) ;
    int k = (1+step)*step/2 ;
    if( k == target )
        return step ; 
    if(k < target)
        step++ ;
    while(1){            
        k = (1+step)*step/2 ;
        if((k - target)%2 == 0)
            return step ;
        step++ ;
    }
    return -1 ;
}
