int smallestRepunitDivByK(int k){
    int len = 1;
    int rem = len % k;
    
    bool* visited = (bool*) malloc(k * sizeof(bool));
    
    for (int i = 0; i < k; i++) {
        *(visited + i) = false;
    }
    
    while (rem != 0) {
        if (*(visited + rem)) {
            return -1;
        }
        *(visited + rem) = true;
        len++;
        rem = (10 * rem + 1) % k;
    }
    
    return len;
}
