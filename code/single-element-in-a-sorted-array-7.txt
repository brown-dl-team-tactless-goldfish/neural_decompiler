

int singleNonDuplicate(int* nums, int numsSize){
    int* l = nums;
    int* r = l + numsSize - 1;
    
    while (l < r) {
        int* pvt = l + (r - l) / 2; // avoid overflow
        if ((pvt-l) % 2) pvt--;
        if (*pvt == *(pvt+1)) {
            l = pvt + 2;
        } else {
            r = pvt - 1;
        }
    }
    
    return *l;
}
