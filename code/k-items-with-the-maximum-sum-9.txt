public class Solution 
{
    public int KItemsWithMaximumSum(int numOnes, int numZeros, int numNegOnes, int k) 
    {
        int result = Math.Min(numOnes, k);
      
        if (k <= numOnes)
            return result;
        k -= numOnes;

        if (k <= numZeros)
            return result;
        k -= numZeros;

        result -= Math.Min(numNegOnes, k);

        return result;
    }
}
