char * removeOuterParentheses(char * S){
    char *result = malloc(10000 * sizeof(char));
    int parentheses_flag = 0, idx = 0;
    for (int i = 1; i < strlen(S); i++) {
        if (S[i] == '(') {
            parentheses_flag++;
        } else {
            parentheses_flag--;
        }
        if (parentheses_flag >= 0) {
            result[idx++] = S[i];
        } else {
            parentheses_flag = 0;
            i++;
        }
    }
    result[idx] = '\0';
    return result;
}
