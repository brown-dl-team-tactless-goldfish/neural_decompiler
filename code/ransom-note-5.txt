bool canConstruct(char * ransomNote, char * magazine){
	int temp[26];
	for(int i = 0; i < 26; i++){
		temp[i] = 0;
	}
	while(*magazine){
		temp[*magazine - 'a']++;
		magazine++;
	}
	while(*ransomNote){
	   temp[*ransomNote - 'a']--;
		ransomNote++; 
	}
	for(int i = 0; i < 26; i++){
		// printf("%d ", temp[i]);
		if(temp[i] < 0){
			return 0;
		}
	}
	return 1;
}
