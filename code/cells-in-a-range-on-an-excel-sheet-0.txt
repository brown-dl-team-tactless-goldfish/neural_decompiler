char ** cellsInRange(char * s, int* returnSize){
    int a = s[3] - s[0];
    int b = s[4] - s[1];
    int n = (a + 1) * (b + 1);
    char** ans = malloc(n * sizeof(char*) );
    int idx_a = 0;
    for(int i = 0; i <= a; i++){
        for(int j = 0; j <= b; j++){
            ans[idx_a] = malloc(3* sizeof(char));
            ans[idx_a][2] = '\0';
            ans[idx_a][0] = s[0] + i;
            ans[idx_a][1] = s[1] + j;
            idx_a++;
        }
    }
    *returnSize = n;
    return ans;
}
