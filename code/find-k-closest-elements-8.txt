int* findClosestElements(int* arr, int arrSize, int k, int x, int* returnSize){
    int left = 0, right = arrSize - 1, count = 0;
    while(left <= right && arrSize - k > count)
    {
        if(abs(arr[left] - x) < abs(arr[right] - x) || abs(arr[left] - x) == abs(arr[right] - x))
            right--;
        else left++;
        count++;
    }
    *returnSize = k;
    return arr + left;
}
