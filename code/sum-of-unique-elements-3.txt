int sumOfUnique(int* nums, int numsSize){
  int *unique = malloc(sizeof(int) * 101); // nums[i] value range + 1
  int res = 0;
  for (size_t i = 0; i < 101; i++)
  {
    unique[i] = 0;
  }
  for (size_t i = 0; i < numsSize; i++)
  {
    unique[nums[i]] += 1;
  }
  for (size_t i = 0; i < 101; i++)
  {
    if (unique[i] == 1)
    {
      res += i;
    }
  }
  return res;
}
