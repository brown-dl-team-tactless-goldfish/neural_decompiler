int x_l = 0, x_r = 0;
int count(TreeNode* root, int x) {
  if (root == nullptr) return 0;
  auto l = count(root->left, x), r = count(root->right, x);
  if (root->val == x) x_l = l, x_r = r;
  return 1 + l + r;
}
bool btreeGameWinningMove(TreeNode* r, int n, int x) {
  count(r, x);
  return x_r + x_l < n / 2 || x_l > n / 2 || x_r > n / 2;
}
