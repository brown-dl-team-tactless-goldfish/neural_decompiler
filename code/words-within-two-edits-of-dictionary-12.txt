class Solution {
public:
    vector<string> twoEditWords(vector<string>& q, vector<string>& d) {
        vector<string> ans;
        for(string s : q) {
            for(string str : d) {
                int d = 0;
                for(int i=0; i<s.size(); i++) {
                    if(str[i] != s[i]) d++;
                }
                if(d <= 2) {ans.push_back(s); break;}
            }
        }
        return ans;
    }
};
