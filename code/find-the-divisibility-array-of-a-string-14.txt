class Solution {
public:
    vector<int> divisibilityArray(string v, int m) {
       long long k=0;
        int n=v.size();
        vector<int>ans(n);
        for(int i=0;i<v.size();i++){
            k = k*10 + (v[i]-'0');
            k%=m;
            if(k)ans[i]=0;
            else ans[i]=1;
        }
        return ans;
    }
};
