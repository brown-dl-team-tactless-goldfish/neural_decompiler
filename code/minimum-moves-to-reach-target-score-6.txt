class Solution {
public:
	int minMoves(int target, int maxDoubles) {
		int c = 0;
		while(maxDoubles > 0){
			if(target == 1){
				return c;
			}
			else{
				if(target % 2 == 0){
					target /= 2;
					maxDoubles--;
				}
				else{
					target--;
				}
				c++;
			}
		}
		return c + (target - 1);
	}
};
