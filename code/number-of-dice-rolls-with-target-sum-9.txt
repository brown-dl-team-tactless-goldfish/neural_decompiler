int numRollsToTarget(int d, int f, int target) {
	vector<int> prev(target + 1);
	prev[0] = 1;
	for(int i = 0; i < d; i++) {
		vector<int> cur(target + 1);
		for(int j = 1; j <= target; j++) 
			for(int k = 1; k <= min(f, j); k++) 
				cur[j] = (cur[j] + prev[j - k]) % (int)(1e9 + 7);   
		prev = cur;
	}
	return prev.back();
}
