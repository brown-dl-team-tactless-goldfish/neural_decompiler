    public int MostFrequentEven(int[] nums) 
    {
        //Number - Frequency
        Dictionary<int,int> freq = new Dictionary<int, int>();
        foreach (int n in nums)
        {
            if(n %2 ==0)
            {
                freq.TryAdd(n, 0);
                freq[n]++;
            }
        }
        if (freq.Count == 0)
            return -1;
        return freq.OrderByDescending(a => a.Value).ThenBy(a => a.Key).First().Key;
    }
