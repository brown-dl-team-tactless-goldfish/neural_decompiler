int maxScoreSightseeingPair (int* values, int valuesSize) {
    int i, j, score, max = INT_MIN;
    
    for (i = 0, j = 1; j < valuesSize; j++) {
        score = values[i] + values[j] + i - j;
        if (max < score) {
            max = score;
        }
        if (values[j] + j > values[i] + i) {
            i = j;
        }
    }
    
    return max;
}
