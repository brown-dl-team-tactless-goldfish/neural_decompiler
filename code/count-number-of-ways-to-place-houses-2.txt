class Solution {
public:
    int mod = 1e9 + 7;
    int countHousePlacements(int n) {
        int num = findNum(n);
        return ((long long)num * num) % mod;
    }

    int findNum(int n) {
        if(n == 1 || n == 2)
            return n+1;
            
        int a = 2, b = 3, c;

        for(int i=2; i<n; i++) {
            c = (a+b) % mod;
            a = b;
            b = c;
        }
        return c;
    }
};
