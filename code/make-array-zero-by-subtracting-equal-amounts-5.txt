    // No ops are applied on earlier 0's/numbers which have been
    // reduce to 0; and identical numbers get reduced to 0 in the
    // same operation => total number of operation required will
    // same as the number of unique positive numbers; as each will
    // of the unique numbers will require a final single operation
    // to reduce them to 0
    
    // Build a hash table of the unique non zero numbers
    unordered_set<int> numsTbl;
    for (auto num : nums) {
        if (num != 0) {
            numsTbl.insert(num);
        }
    }
    
    return numsTbl.size();
}
