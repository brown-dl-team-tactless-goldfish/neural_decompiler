class Solution {
public:
    int minMoves(int target, int maxDoubles) {
        int res = 0;
        while(maxDoubles-- > 0 && target > 1){
            if(target % 2 == 1) {
                target--;
                res++;
            }
            target /= 2;
            res++;
        }
        return res + target - 1;
    }
};
