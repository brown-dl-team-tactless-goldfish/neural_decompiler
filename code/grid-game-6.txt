class Solution {
public:
    long long gridGame(vector<vector<int>>& grid) {
        long long ans = LONG_MAX, prefix = 0, suffix = accumulate(grid[0].begin(), grid[0].end(), 0l); 
        for (int i = 0; i < grid[0].size(); ++i) {
            suffix -= grid[0][i]; 
            ans = min(ans, max(prefix, suffix)); 
            prefix += grid[1][i]; 
        }
        return ans; 
    }
};
