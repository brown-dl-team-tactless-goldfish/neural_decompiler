    long ans = 0, maxi = 0, i = 0;
    
    while(maxi < n)
    {
        
        if(i < nums.size( ) && nums[i] <= maxi+1)
        {
            maxi = maxi + nums[i++];
        }
        else
        {
            ans++;
            
            maxi = maxi+ maxi+1;
        }
    }
    
    return ans;
