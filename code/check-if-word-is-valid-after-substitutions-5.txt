bool isValid(char * s)
{

	char* stack_ptr[13334];
	int i;
	int position = 0;
	while (*s) {
		switch (*s) {
		case 'a':
			s++;
			if (*s == 'b') {
				stack_ptr[position++] = "ab";
				s++;
			}
			else
				stack_ptr[position++] = "a";
			break;
		case 'b':
			s++;
			if (*s == 'c') {
				if (position && strcmp(stack_ptr[--position], "a") == 0)
					s++;
				else
					return false;
			}
			else {
				if (position && strcmp(stack_ptr[--position], "a") == 0) {
					stack_ptr[position++] = "ab";
				}
				else {
					stack_ptr[position++] = "b";
				}
			}
			break;
		case 'c':
			if (position && strcmp(stack_ptr[--position], "ab") == 0)
				s++;
			else
				return false;
			break;
		}
		if (position >= 13334)
			return false;
	}
	if (position)
		return false;
	else
		return true;
}
