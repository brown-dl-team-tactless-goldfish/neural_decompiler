int countPrimeSetBits(int left, int right){
                    
    bool prim[] = {false, false, true, true, false, true, false, true, false, false,\
                 false, true, false, true, false, false, false, true, false, true, \
                 false};
    int ans = 0;
    for(int i = left; i <= right; i++){
        int val = i;
        int cn = 0;
        while(val){
            cn += val % 2;
            val /= 2;
        }
        ans += prim[cn];
    }
    return ans;
}
