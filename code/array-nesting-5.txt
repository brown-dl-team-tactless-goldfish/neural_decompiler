

int arrayNesting(int* nums, int numsSize){
    int res=0;
    for(int j=0;j<numsSize;j++){
        int i=j, count=0;
        while(nums[i]>=0){
            int index = nums[i];
            nums[i] = -1;
            count++;
            i = index;
        }
        res = count > res ? count : res;
    }
    return res;
}
