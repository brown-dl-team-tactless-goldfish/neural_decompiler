class Solution {
public:
    int countDistinctIntegers(vector<int>& nums) {
        unordered_set<int>st;
        for(int i=0;i<nums.size();i++)
        {
            int n=nums[i];
            int num=0;
            while(n)
            {
                int r=n%10;
                num=num*10+r;
                n=n/10;
            }
            st.insert(num);
            st.insert(nums[i]);
        }
        return st.size();
    }
};
