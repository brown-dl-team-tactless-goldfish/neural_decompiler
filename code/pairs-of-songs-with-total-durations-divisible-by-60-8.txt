int numPairsDivisibleBy60(int *time, int size)
{
        int remainders[60] = { 0 }, count = 0, mod60 = 0;

        for (int i = 0; i < size; i++) {
            mod60 = time[i] % 60;
            count += mod60 ? remainders[60 - mod60] : remainders[0];
            remainders[mod60]++;
        }

        return count;
}
