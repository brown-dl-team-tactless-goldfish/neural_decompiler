// i is used to loop from 0 to last element
// j is used to loop from (i+1) to last element
// count is use to count how many nums[j] are there ahead of the nums[i] where nums[j] == nums[i]. 
//Eg:[1,2,3,1,1,3] ==> first nums[i]=1 and there are 2 1's so count will become 2 then nums[i]=2 and there are 0 2 so count remains 2 only,
//then nums[i]=3 and count becomes 3 because there is one match. Then nums[i]=1 and count becomes 4. Then nums[i]=1 no matcch so count=4. 
//Then nums[i]=3 last element so count remains 4 only. 
int numIdenticalPairs(int* nums, int numsSize){
    int i=0;   
    int count=0;
    while(i<numsSize){
        int j=i+1;
        while(j<numsSize){
            if(nums[i] == nums[j]){
                count++;
            }
            j++;
        }
        i++;
    }
    return count;
}
