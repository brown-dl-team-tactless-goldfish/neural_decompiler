class Solution {
public:
    int minIncrementForUnique(vector<int>& A) {
        
        int n = A.size();
        sort(A.begin(), A.end());
        
        int res = 0;
        for (int i = 0; i < n-1; i++) {
            if (A[i] == A[i+1]) {
                A[i+1]++;
                res++;
            } else if (A[i] > A[i+1]) {
                res += A[i]-A[i+1]+1;
                A[i+1] = A[i]+1;
            }
        }
        
        return res;
    }
};
