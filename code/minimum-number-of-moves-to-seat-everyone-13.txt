class Solution {
public:
    int minMovesToSeat(vector<int>& v, vector<int>& vv) {
        
        sort(v.begin(), v.end());
        sort(vv.begin(), vv.end());
        int ans = 0;
        int temp = 0;
        for(int i=0; i<v.size(); i++)
        {
            temp = abs(v[i] - vv[i]);
            ans+=temp;
        }
        return ans;
    }
};
