int numberOfBeams(char ** bank, int bankSize){
    int r = bankSize, ans = 0, prev = 0;
    for(int i = 0; i < r; i++)
    {
        int currCnt = 0;
        int len = strlen(bank[i]);
        for(int j = 0; j < len; j++)
            currCnt += bank[i][j]=='1'? 1 : 0;
        ans += (prev * currCnt);
        if(currCnt) /* update prev if get beams for next connection */
            prev = currCnt;
    }
    return ans;
}
