double findMaxAverage(int* nums, int numsSize, int k)
{
    int window = 0;
    int max;
    int i;
    for(i = 0;i < k; i++ )
    {
        window += nums[i];
    }
    max = window;
    for(; i < numsSize; i++)
    {
        window -= nums[i - k] - nums[i];
        max = window  > max ? window : max;
    }
    return (double) max / k;
}
