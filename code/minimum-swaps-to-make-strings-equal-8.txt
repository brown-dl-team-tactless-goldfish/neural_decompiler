int minimumSwap(string s1, string s2) {
	if (s1.size() != s2.size())
		return -1;
	int numXY = 0, numYX = 0;
	for (int i = 0; i < s1.size(); i++) {
		if (s1[i] == 'x' && s2[i] == 'y')
			numXY++;
		else if (s1[i] == 'y' && s2[i] == 'x')
			numYX++;
	}
	if (numXY % 2 != numYX % 2)
		return -1;
	return ceil((float)numXY / 2) + ceil((float)numYX / 2);
}
