

int distributeCandies(int* candyType, int candyTypeSize){

    int candy_types = 0;
    int hash_table[200001] = {0};
    
    
    for(size_t i = 0; i < candyTypeSize ; i++)
    {
		//check if candy type is unique
        if(1 == (hash_table[candyType[i]+100000] += 1))
        {
            candy_types++;
        }
    }
    
    return candy_types > (candyTypeSize / 2) ? candyTypeSize / 2 : candy_types;
}
