int max(int a, int b){
    return a>=b? a : b;
}

int maxScoreSightseeingPair(int* values, int valuesSize){
    int i;
    int a=values[0];     // for storing values[i]+i
    int b;              //for storing values[j]-j
    int maxVal=0;
    for(i=1;i<valuesSize;i++){
        a=max(a,values[i-1]+i-1);
        b=values[i]-i;
        maxVal=max(maxVal,a+b);
    }
    return maxVal;
}
