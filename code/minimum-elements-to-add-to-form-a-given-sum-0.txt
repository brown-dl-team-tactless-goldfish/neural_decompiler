int minElements(int* nums, int numsSize, int limit, int goal){
    int n = numsSize ;
    long long sum = 0 ;
    
    for(int i = 0; i < n; i++)
        sum += nums[i] ;
    
    long long diff = fabs(sum - goal) ;  //must use fabs(), abs is not fit for long long
    
    if(diff == 0)
        return 0 ;

    return diff/limit + ((diff%limit > 0) ? 1 : 0);
    
}
