   int* diStringMatch(char* S, int* returnSize) {
   *returnSize=strlen(S)+1;
    int n=*returnSize;
    int *ans = (int *)malloc((*returnSize)*sizeof(int));
    int a=0,b=n-1;
    for(int i=0;i<n-1;i++)
    {
        if(S[i]=='I') ans[i]=a++;
        else ans[i]=b--;
    }
    ans[n-1]=b; 
    return ans;
}
