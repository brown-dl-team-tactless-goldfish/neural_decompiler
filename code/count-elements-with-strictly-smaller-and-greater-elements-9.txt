class Solution {
public:
    int countElements(vector<int>& nums) {
        int n = nums.size();
        sort(nums.begin(),nums.end());
        int i=0,j=n-1;
        while(i<n-1 && nums[i]==nums[i+1])
            i++;
        if(i==n) return 0;
        int count = 0;
        while(j>i && nums[j]==nums[j-1])
        {
            j--;
            count++;
        }
        if(j<=i) return 0;
        return j-i-1;
    }
};
