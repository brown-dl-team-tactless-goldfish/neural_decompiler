    bool reachingPoints(int sx, int sy, int tx, int ty) {
        // Check edge case
        if (tx<sx || ty<sy) return false;
        
        // Reverse Thinking & Logical Deduction : Proof-1
        while (sx < tx && sy < ty) {
            if (tx > ty) tx %= ty;  
            else ty %= tx;
        }
        // Return : Proof-2
        return ((tx == sx && ty == sy) || 
            (tx == sx && ty > sy && (ty-sy)%sx==0) || 
            (ty == sy && tx > sx && (tx-sx)%sy==0));
    }
