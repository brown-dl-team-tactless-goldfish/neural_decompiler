int wateringPlants(int* plants, int plantsSize, int capacity) {
    int cnt = 1, tmp = capacity;
    for (int i = 0; i+1 < plantsSize; i++) {
        tmp -= plants[i];
        if (tmp < plants[i+1]) {
            cnt += 2 * (i + 1);
            tmp = capacity;
        }
        cnt++;
    }
    return cnt;
}
