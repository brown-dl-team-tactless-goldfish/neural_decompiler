class Solution {
public:
    int maxWidthOfVerticalArea(vector<vector<int>>& p) {
        //sort the x points
        vector<int>v;
        for(int i=0;i<p.size();i++){
            v.push_back(p[i][0]);
        }
        sort(v.begin(),v.end());
        int ans=0;//points are +ve;
        for(int i=0;i<v.size()-1;i++){
            ans=max(ans,v[i+1]-v[i]);
        }
        return ans;
    }
};
