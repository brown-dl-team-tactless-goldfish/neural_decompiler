void dfs(int cur, int n, int* ret){
    if(cur > n)
        return ;
    *ret = *ret + 1 ;
    if((cur & 1) == 1)
        dfs((cur << 1) + 0, n, ret) ;
    else{
        dfs((cur << 1) + 0, n, ret) ;
        dfs((cur << 1) + 1, n, ret) ;
    }       
}
int findIntegers(int n){
    int ret = 1 ; // 0 is the ans
    dfs(1, n, &ret) ;
    return ret ;
}
