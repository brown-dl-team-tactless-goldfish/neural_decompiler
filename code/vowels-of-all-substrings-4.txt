class Solution {
public:
    long long countVowels(string word) {
        auto isvowel = [] (char c) { 
            return c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u';
        };
        long long ans = 0, cnt = 0;
        for (int i = 0; i < word.size(); i++) {
            if (isvowel(word[i]))
                cnt += i + 1;
            ans += cnt;
        }
        return ans;
    }
};
