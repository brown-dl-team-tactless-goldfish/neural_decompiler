    def countPartitions(self, A: List[int], k: int) -> int:
        if sum(A) < k * 2: return 0
        mod = 10**9 + 7
        dp = [1] + [0] * (k - 1)
        for a in A:
            for i in range(k - 1 - a, -1, -1):
                dp[i + a] += dp[i]
        return (pow(2, len(A), mod) - sum(dp) * 2) % mod
