

int findDuplicate(int* nums, int numsSize){
    //DATA
    int* arr=(int*)malloc(numsSize*sizeof(int)); 
    
    //INIT
    for(int x=0;x<numsSize;x++) arr[x]=0;
    
    //Start
    for(int x=0;x<numsSize;x++){
        if(arr[nums[x]]==1){
            return nums[x];
        }else{
            arr[nums[x]]=1;
        }
    }
    return 0;
}
