int strToInt(char* str){
    int ans = 0;
    int n = strlen(str);
    for(int i = 0; i < n; i++){
        ans = ans*10 + str[i]-'a';
    }
    return ans;
}

bool isSumEqual(char * firstWord, char * secondWord, char * targetWord){

    return (strToInt(firstWord) + strToInt(secondWord))== strToInt(targetWord);
}
