typedef struct ListNode* node_t;

node_t mergeNodes(node_t head){
    node_t newhead = head->next, curr = newhead, temp = NULL;
    free(head);
    while (curr) {
        temp = curr->next;
        curr->next = temp->next;    // since list length is at least 3 nodes
        if (!temp) {
            break;
        }
        else if (temp->val != 0) {
            curr->val += temp->val; // temp is a non-zero node, incrament curr value
        }
        else {
            curr = curr->next;      // temp is a 0-valued node, move curr forward
        }
        free(temp);                 // free temp
    }
    return newhead;
}
