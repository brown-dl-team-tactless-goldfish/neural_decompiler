int minDeletionSize(char ** strs, int strsSize){

    int l = strlen(strs[0]);
    int c=0;

    for(int j=0;j<l;j++)
    {
        for(int i=1;i<strsSize;i++)
        {
            if(strs[i-1][j]>strs[i][j])
            {
                c+=1;
                break;
            }
        }
    }
    return c;

}
