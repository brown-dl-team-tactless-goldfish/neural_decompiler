class Solution {
public:
    int maxCount(vector<int>& banned, int n, int maxSum) {
        int cnt=0, sum=0;
        unordered_map<int, int>m;
        for(int i=0; i<banned.size(); i++){
            m[banned[i]]++;
        }
        for(int i=1; i<=n; i++){
            if(m.find(i)==m.end()){
                sum+=i;
                cout<<i<<endl;
                if(sum<=maxSum){
                cnt++;}
                else{break;}
            }
        }
        return cnt;
    }
};
