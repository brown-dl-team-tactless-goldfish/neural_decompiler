class Solution {
public:
    using VI = vector<int>;
    using VVI = vector<VI>;
    VVI findSolution(CustomFunction& customFuction, int z, VVI ans={}) {
        for (auto i=1; i <= 1000; ++i)
            for (auto j=1; j <= 1000; ++j)
                if (customFuction.f(i, j) == z)
                    ans.push_back({i, j});
        return ans;
    }
};
