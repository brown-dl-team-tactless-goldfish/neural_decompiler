int tupleSameProduct(vector<int>& nums) {
    int n = nums.size();
    unordered_map<int,int>m;
    for(int i=0;i<n;i++){
        for(int j=i+1;j<n;j++){
            int p = (nums[i]*nums[j]);
            m[p]++;
        }
    }
    int c = 0;
    for(auto x:m){
        int y = x.second;
		//((nC2)/2)*4 = (n*(n-1)*4)
        c+=(y*(y-1)*4);
    }
    return c;
}
