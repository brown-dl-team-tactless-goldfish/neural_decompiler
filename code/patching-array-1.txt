int minPatches(int* nums, int numsSize, int n)
{
    int ret = 0,ptr = 0;
    unsigned int range = 0,limit = n;
    while(range < limit)
    {
        if(ptr == numsSize || range + 1 < nums[ptr])
            ret ++,range = (range << 1) | 1;
        else 
            range += nums[ptr++];
    }
    return ret;
}
