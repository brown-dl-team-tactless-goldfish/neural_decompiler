#include <stdlib.h>
char * sortSentence(char * s){
    int i=0,j=0;
    char a[200][200];
    char res[200];
    int g=0;
    int l=strlen(s);
    while(i<l)
    {
          if(s[i]=='1'||s[i]=='2'||s[i]=='3'||s[i]=='4'||s[i]=='5'||s[i]=='6'||s[i]=='7'||s[i]=='8'||s[i]=='9')
          {
             
              int h=s[i]-'0';
              
              res[j]='\0';
              strcpy(a[h-1],res);
               g++;
             
              j=0;
              i++;
          }
          else
          {
             res[j]=s[i];
              j++;

          }
          i++;

    }
   
     
    

    strcpy(s,a[0]);
    for(int i=1;i<g;i++)
    {
        strcat(s," ");
        strcat(s,a[i]);
    }
    
    
return s;
}
