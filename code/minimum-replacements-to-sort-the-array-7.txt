    long long minimumReplacement(vector<int>& nums) {
        if(nums.size() <= 1) return 0;
        long long res = 0, pre = nums.back();
        for(int i = nums.size()-2; i>=0; i--){
            if(nums[i] > pre){
                int b = nums[i] / pre + (nums[i] % pre != 0);
                res += b-1;
                pre = nums[i] / b;
            }
            else pre = nums[i];
        }
        return res;
    }
