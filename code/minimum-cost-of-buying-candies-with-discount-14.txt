class Solution {
public:
    int minimumCost(vector<int>& cost) 
    {
        int sum=0;
        int n=cost.size();
        int count=0;
        sort(cost.rbegin(),cost.rend());
        for(int i=0;i<n;i++)
        {
            if(count==2)
            {
                count=0;
                continue;
            }
            sum+=cost[i];
            count++;
        }
        return sum;
    }
};
