class Solution {
public:
    int numberOfUniqueGoodSubsequences(string binary) {
        const int MOD = 1'000'000'007; 
        long f0 = 0, f1 = 0; 
        for (auto& ch : binary) {
            if (ch == '0') f0 = (f0 + f1) % MOD; 
            else f1 = (f0 + f1 + 1) % MOD; 
        }
        return (f0 + f1 + (f0 || binary[0] == '0')) % MOD; 
    }
};
