int* runningSum(int* nums, int numsSize, int* returnSize){
    int i, prev=0;
    *returnSize = numsSize;
    int* returnSum;
    returnSum = (int*)malloc(sizeof(int)*numsSize);
    returnSum[0] = nums[0];
    for(i=1;i<numsSize;i++)
        returnSum[i]=returnSum[i-1] + nums[i];
    return returnSum;

}
