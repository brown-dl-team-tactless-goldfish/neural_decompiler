 public int LongestSubarray(int[] nums)
    {
        var max=nums.Max();
        var length = 0;
        var result = 0;
        for (int i = 0; i < nums.Length; i++)
        {
            if (nums[i] == max)
                length++;
            else
                length = 0;
            result = Math.Max(result, length);
        }
        return result;
    }
