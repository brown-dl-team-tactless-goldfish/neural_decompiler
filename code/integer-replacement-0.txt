        long long t = n;
        return 2 + fmin(integerReplacement((t + 1) / 2), integerReplacement((t - 1) / 2));       
    }

    if (n % 2 == 0) 
        return 1 + integerReplacement(n / 2);
    else{
        return 2 + fmin(integerReplacement((n + 1) / 2), integerReplacement((n - 1) / 2));
