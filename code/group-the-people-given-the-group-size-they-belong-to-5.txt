int** groupThePeople(int* groupSizes, int groupSizesSize, int* returnSize, int** returnColumnSizes){
    int **result = (int**)malloc(sizeof(int*)*groupSizesSize);
    int flag;
    int count = 0;
    
    *returnSize = 0; // the number of result's row
    *returnColumnSizes = (int*)malloc(sizeof(int)*groupSizesSize); // the number of result's column
    
    for(size_t i=1; i<=groupSizesSize; i++){
        flag = 0;
        for(size_t j=0; j<groupSizesSize; j++)
            if(groupSizes[j]==i){
                count++;
                if(flag%i==0){
                    (*returnColumnSizes)[*returnSize] = i;
                    result[*returnSize] = (int*)malloc(sizeof(int)*i);
                    (*returnSize)++;  
                }
                result[(*returnSize)-1][flag%i] = j;
                flag++;
            }
        if(count>=groupSizesSize)
            break;
    }
    
    return result;
}
