class Solution {
public:
    int minimumSwap(string s1, string s2) {
        int len = s1.length();
        int xy = 0,yx = 0;
        for(int i = 0;i<len;i++){
            if(s1[i]=='x'&&s2[i]=='y')xy ++;
            else if(s1[i]=='y'&&s2[i]=='x') yx ++;
        }
        if((xy+yx)&1)return -1;
        int ans = (xy+yx)/2;
        if(xy&1)ans ++;
        return ans;
    }
};
