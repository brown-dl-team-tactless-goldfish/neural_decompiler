

int longestValidParentheses(char * s){

    int f =0;
  int max = 0;
    int leng = strlen(s);
    if(leng == 0) return 0; 
    
    for(int i=0; i<strlen(s);i++){
        if(*(s+i) == ')'){
            
            for(int j = i-1 ; j>=0 ;j--){
                 if(*(s+j) == '('){
                     *(s+i) = '0';
                     *(s+j)='0';
                    break;
              } 
            }
         }
       }
    
    for(int i=0;i<leng;i++){
        if(*(s+i)=='0'){
            f++;
        }
        else{
            
            if(f > max){
            max = f;
        } 
          f =0;
        }
        
    }
    
    if(f > max)
        max =f ;
    
     return max;
}
