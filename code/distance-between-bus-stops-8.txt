int distanceBetweenBusStops(int* distance, int distanceSize, int start, int destination){
    int dist_clock = 0;
    int dist_count = distance[destination];

    int i;
    for(i=start; i != destination; i = (i+1)%distanceSize) dist_clock += distance[i]; 
    for(i=((start-1)<0?distanceSize-1:start-1); i != destination; i = ((i-1)<0?distanceSize-1:i-1)) dist_count += distance[i];
    
    return (dist_clock<dist_count)? dist_clock : dist_count;
}
