int diagonalSum(int** mat, int matSize, int* matColSize){   
    int i=0;
    int j=matSize-1;
    int sum=0;
    
    while(i<matSize){
        sum = sum+mat[i][i];
        i++;
    }
    i=0;
    while(i<matSize){
        if(i!=j){
            sum = sum+mat[i][j];
        }
        i++;
        j--;
    }
    return sum;
}
