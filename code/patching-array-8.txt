public class Solution 
{
    public int MinPatches(int[] nums, int n) 
    {
        long miss = 1;
        int result = 0;
        int i = 0;

        while(miss <= n)
        {
            if (i < nums.Length && nums[i] <= miss)
            {
                miss += nums[i];
                i++;
            }
            else
            {
                miss += miss;
                result++;
            }
        }

        return result;
    }
}
