class Solution {
public:
    bool containsPattern(vector<int>& arr, int m, int k) {
        // support variables + main loop
        for (int i = 0, t = m * --k, c, len = arr.size() - m; i < len; i++) {
            if (arr[i] == arr[i + m]) {
                c++;
                if (c == t) return true;
            }
            else c = 0;
        }
        return false;
    }
};
