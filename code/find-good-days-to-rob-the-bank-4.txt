class Solution {
public:
    vector<int> goodDaysToRobBank(vector<int>& s, int t) {
        int N = s.size();
        vector<int> non_inc(N, 0), non_dec(N, 0), res;

        for(int i = 1; i < N; i++) {
            if(s[i] <= s[i-1]) non_inc[i] = non_inc[i-1]+1;
            int j = N-i-1;
            if(s[j] <= s[j+1]) non_dec[j] = non_dec[j+1]+1;
        }

        for(int i = t; i+t < N; i++) if(non_inc[i] >= t && non_dec[i] >= t) res.push_back(i);
        return res;
    }
};
