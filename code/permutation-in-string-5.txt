    vector < int > freq1(26);
    for(int i = 0 ; i < size(s1) ;i++)
        freq1[int(s1[i]) - 'a']++;  
    
    while(l <= r && r < size(s2))
    {   
        vector < int > freq2(26);
        for(int i = l ; i <= r ; i++)
            freq2[int(s2[i]) - 'a']++;

        if(freq1==freq2)        return 1;
        l++,r++;
    }
