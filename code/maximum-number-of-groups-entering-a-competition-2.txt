class Solution {
public:
    int maximumGroups(vector<int>& nums) {
        sort(begin(nums),end(nums)) ;
        
        int groups = 0 , sz = 1 ;
        
        for(int i = 0 ; i < nums.size(); ){
            if(size(nums) - i >= sz) i += sz , ++groups ;
            else break ;
            ++sz ;
        }
        return groups ;
        
    }
};
