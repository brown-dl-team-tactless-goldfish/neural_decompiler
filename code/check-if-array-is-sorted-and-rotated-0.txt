bool check(int* nums, int numsSize){
    int gap = 0 ; 
    for(int i = 1 ; i < numsSize ; i++){
        if(nums[i] < nums[i-1])
            gap++ ;
    }
    if(gap == 0)
        return true;
    else if(gap == 1){
        if(nums[0] >= nums[numsSize - 1] )
            return true ;
        else
            return false ;
    }
    else
        return false ;
    
}
