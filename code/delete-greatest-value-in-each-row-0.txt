int cmp(const void* a, const void* b){
    return *(int*)b - *(int*)a;
}
int deleteGreatestValue(int** grid, int gridSize, int* gridColSize){
    int m = gridSize;
    for(int i = 0; i < m; i++){
        qsort(grid[i], *gridColSize, sizeof(int), cmp);
    }
    int ans = 0;
    for(int i = 0; i < (*gridColSize); i++ ){
        int max = grid[0][i];
        for(int j = 1; j < m; j++){
            max = fmax(max, grid[j][i]);
        }
        ans += max;
    }
    return ans;
}
