class Solution {
public:

    int mostFrequent(vector<int>& nums, int key) {
        unordered_map<int,int>mpp;
        for(int i=0;i<nums.size()-1;i++)
        {
            if(nums[i]==key)
                mpp[nums[i+1]]++;
        }
        int m=0;
        for(auto it:mpp)
        {
            m=max(m,it.second);
        }
        int res=0;
        for(auto it:mpp)
        {
            if(it.second==m)
                res=it.first;
        }
        return res;
    }
};
