bool isMonotonic(int* A, int ASize){
    int i, isIncreasing, isDecreasing;
    isIncreasing=isDecreasing=0;
    for(i=0;i<ASize-1;i++){
        if(A[i]<A[i+1]){
            isIncreasing = 1;
            if(isDecreasing)
                return false;
        }
        else if(A[i]>A[i+1]){
            isDecreasing = 1;
            if(isIncreasing)
                return false;
        }
    }
    return true;
}
