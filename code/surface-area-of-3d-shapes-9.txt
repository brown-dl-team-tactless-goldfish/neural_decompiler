class Solution {
public:
    int surfaceArea(vector<vector<int>>& grid) {
        int ans = 0;
        for (int r = 0; r < grid.size(); r++) {
            for (int c = 0; c < grid[0].size(); c++) {
                int cubes = grid[r][c];
                if (cubes) ans += 4 * cubes + 2;
                if (r) ans -= 2 * min(cubes, grid[r - 1][c]); // top
                if (c) ans -= 2 * min(cubes, grid[r][c - 1]); // left
            }
        }
        return ans;
    }
};
