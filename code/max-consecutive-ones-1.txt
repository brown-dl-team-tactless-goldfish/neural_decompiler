int findMaxConsecutiveOnes(int* nums, int numsSize){
    int counter = 0;
    int max = 0;
    for(int i = 0; i<numsSize; i++)
    {
        if(nums[i] == 1)
        {
            counter++;
            max = max < counter ? counter : max;
        }
        else
            counter = 0;
    }
    return max;
}
