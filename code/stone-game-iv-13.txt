class Solution {
public:
    int dp[(int)1e5] ;
    bool winnerSquareGame(int n) {
        dp[0] = 0 ;
        dp[1] = 1 ;
        for( int i = 1 ; i <= n ; i ++ ){
            for(int j = 1 ; j * j <= i ; j ++ ){
                if(dp[ i - ( j * j ) ] == 0 ){
                    dp[i] = 1 ;
                    break ;
                }
            }
        }
        return dp[n] ;
    }
};
