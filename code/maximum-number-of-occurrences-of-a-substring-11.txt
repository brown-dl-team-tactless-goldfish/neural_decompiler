int maxFreq(string s, int maxl, int mins, int) {
    unordered_map<string, int> m;
    for (int i = 0; i <= s.size() - mins; ++i) {
        string str(s.begin() + i, s.begin() + i + mins);
        m[str]++;
    }
    int res = 0;
    for (auto const& [tmp, c] : m) {
        auto str = tmp;
        sort(str.begin(), str.end());
        auto i = unique(str.begin(), str.end());
        if (distance(str.begin(), i) <= maxl)
            res = max(res, c);
    }
    return res;
}
