    string removeOccurrences(string s, string part) {
        string result;
        
        for (auto c : s) {
            result.push_back(c);
            
            if (result.size() >= part.size()) {
                size_t found = result.find(part);
                if (found != std::string::npos) {
                    result.resize(result.size() - part.size());
                }
            }
        }
        
        return result;
    }
