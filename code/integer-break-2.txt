

int integerBreak(int n)
{
        int *dp = malloc(sizeof(int) * (n + 1));
        memset(dp, 0, sizeof(int) * (n + 1));
        dp[2] = 1;

        int i, j;
        for (i = 2; i <= n; i++) {
                for (j = 1; j < i; j++) {
                        int tmp = j * (dp[i - j] >= (i - j) ? dp[i - j] : i - j);
                        dp[i] = tmp > dp[i] ? tmp : dp[i];
                }
        }
        
        return dp[n];
        //free dp[]
}
