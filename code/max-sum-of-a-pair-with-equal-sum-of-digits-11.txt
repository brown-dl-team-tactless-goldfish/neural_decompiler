class Solution {
public:
    int maximumSum(vector<int>& nums) {
                int n=nums.size();
        int m=-1;
        unordered_map <int,int> f; 
        for(int i=0;i<n;i++){
            int s=0, k=nums[i];
            while(k>0){
                s+=k%10;
                k/=10;
            }
            if(f.find(s)==f.end()) f[s]=i;
            else m=max(nums[i]+ nums[f[s]],m);
                if(nums[i]>nums[f[s]]) f[s]=i;
        }
        return m;
    }
    
};
