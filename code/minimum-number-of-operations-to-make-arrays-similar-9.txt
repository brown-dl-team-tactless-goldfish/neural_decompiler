class Solution {
public:
    long long makeSimilar(vector<int>& nums, vector<int>& target) {
        sort(begin(nums), end(nums)), sort(begin(target), end(target));
        stable_partition(begin(nums), end(nums), [] (int x) { return x&1; });
        stable_partition(begin(target), end(target), [] (int x) { return x&1; });
        long long ans = 0;
        for (int i = 0; i < nums.size(); i++)
            if (nums[i] > target[i]) ans += (nums[i] - target[i])/2;
        return ans;
    }
};
