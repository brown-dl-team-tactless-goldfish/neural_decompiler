*returnSize=numsSize;
int *ans=(int*)malloc(numsSize*sizeof(int));
int *posi=(int*)malloc((numsSize/2)*sizeof(int));
int *negi=(int*)malloc((numsSize/2)*sizeof(int));
int i,p=0,n=0;
for(i=0;i<numsSize;i++)
{
    if(nums[i]>0) 
        posi[p++]=nums[i];
    else if(nums[i]<0)  
        negi[n++]=nums[i];
}
for(i=0;i<numsSize/2;i++)
{
    ans[2*i]=posi[i];     
    ans[2*i+1]=negi[i];
}
free(posi);
free(negi);
return ans;
