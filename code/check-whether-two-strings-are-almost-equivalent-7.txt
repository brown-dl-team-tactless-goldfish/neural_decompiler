public class Solution {
    public bool CheckAlmostEquivalent(string word1, string word2) {
        
        int[] freq = new int[26];
        foreach(var c in word1)
            freq[c - 'a']++;
        
        foreach(var c in word2)
            freq[c - 'a']--;
        
        foreach(var f in freq)
        {
            if(Math.Abs(f) > 3)
                return false;
        }
        
        return true;
    }
}
