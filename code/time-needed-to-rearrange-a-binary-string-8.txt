class Solution {
public:
    int secondsToRemoveOccurrences(string s) 
    {
        int zero=0,second=0;
        for(int i=0;i<s.size();i++)
        {
            zero+=(s[i]=='0');
            if(s[i]=='1'&&zero)
            {
                second=max(second+1,zero);
            }
        }
        return second;
        
    }
};
