struct TreeNode* insertIntoMaxTree(struct TreeNode* root, int val){
    
    struct TreeNode* newNode;
        
    if(root==NULL)
    {
        newNode = (struct TreeNode*)malloc(sizeof(struct TreeNode));
        newNode->val = val;
        newNode->left = NULL;
        newNode->right = NULL;
        return newNode;
    }
    
    if(root->val > val)
    {
        root->right = insertIntoMaxTree(root->right, val);
    }
    else
    {
        newNode = (struct TreeNode*)malloc(sizeof(struct TreeNode));
        newNode->val = val;
        newNode->left = root;
        newNode->right = NULL;
        root = newNode;
    }
    
    return root;
}
