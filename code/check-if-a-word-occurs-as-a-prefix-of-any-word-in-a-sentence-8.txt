int isPrefixOfWord(char * sentence, char * searchWord){
    char *w=strstr(sentence,searchWord);
    if(w==sentence)return 1; 
    int count=1; 
    char word[strlen(searchWord)+2];
    sprintf(word," %s",searchWord);
    w=strstr(sentence,word);
    if(w==NULL)return -1;
    *w++;
    while(sentence!=w){
        if(*sentence==' ')count++;
        *sentence++;
    }
    return count;
}
