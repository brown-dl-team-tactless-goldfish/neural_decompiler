int findFinalValue(int* nums, int numsSize, int original){
    // note that any variables declared within a do while loop is not global so you can't reference a variable from the body of the do portion within the test expression of the while loop.
    int changed = 1;
    while (changed == 1){
        changed = 0;
        for (int i = 0; i < numsSize; i++){
            if (*(nums+i) == original){
                original = original * 2;
                changed = 1;
            }
        }
    }
    return original;
}
