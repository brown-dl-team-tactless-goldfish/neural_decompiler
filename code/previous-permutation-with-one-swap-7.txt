    int n = arr.size();
    
    if(n==1) return arr;
    
    int j=n-2;
    int i=n-1;
    
    while(j>=0){
        
        if(arr[j] <= arr[i]){
            i--;
            j--;
        }
        else{
            break;
        }
    }
    
    if(j<0) return arr;
    
    i = n-1;
    
    while(arr[j] <= arr[i])i--;
    
    int val = arr[i];
    
    while(arr[i] == val) i--;
    
    i++;
    
    swap(arr[j], arr[i]);
    
    return arr;
    
}
