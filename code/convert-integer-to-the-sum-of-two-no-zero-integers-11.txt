vector<int> getNoZeroIntegers(int n) {
    for (int i = 1; i < n; ++i) {
        auto a = to_string(i);
        auto b = to_string(n-i);
        if (none_of(a.begin(), a.end(), [](auto c) { return c == '0'; }) && 
            none_of(b.begin(), b.end(), [](auto c) { return c == '0'; }))
            return {i, n - i};
    }
    return {};
}
