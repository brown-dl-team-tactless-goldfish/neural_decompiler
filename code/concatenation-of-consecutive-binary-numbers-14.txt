class Solution {
public:
    int concatenatedBinary(int n) {
        long long M = 1000000007, cur = 1, p = 1;
        
        for(int i=2; i<=n; i++) {
            if(!(i & (i-1))) p++; // is i is a power of 2, number of bits in representation will increase
            cur = ((cur << p) + i) % M;
        }
        
        return cur;
    }
};
