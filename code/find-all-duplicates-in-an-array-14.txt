/**
 * Note: The returned array must be malloced, assume caller calls free().
 */
int* findDuplicates(int* nums, int numsSize, int* returnSize){
*returnSize = 0;
int* return_nums = (int*)calloc(numsSize,sizeof(int));
int* hash_table = (int*)calloc(100001,sizeof(int));
for(int ptr =0 ;ptr < numsSize; ptr++){
    hash_table[nums[ptr]]++;
    if(hash_table[nums[ptr]] > 1){
        return_nums[(*returnSize)++] = nums[ptr];
    }
}
return return_nums;
}
