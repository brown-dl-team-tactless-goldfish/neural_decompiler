class Solution {
public:
    string strWithout3a3b(int a, int b) {
        string res(a+b,0);
        char cs[] = {'a','b'};
        int iab[] = {a, b};
        int i=0, c=1;
        while(i < a+b)
        {
            c = iab[0]>iab[1]?0:1;
            if (i>=2 && (res[i-1] == cs[c] && res[i-2] == cs[c]))
                c = (c+1) & 1;
            res[i++] = cs[c];
            iab[c] -= 1;
        }
        return res;
    }
};
