class SORTracker {
    set<pair<int, string>> st; 
    set<pair<int, string>>::iterator it = st.end(); 
public:
    SORTracker() {}
    
    void add(string name, int score) {
        auto cand = st.insert({-score, name}).first; 
        if (it == st.end() || *cand < *it) --it; 
    }
    
    string get() {
        return (it++)->second; 
    }
};
