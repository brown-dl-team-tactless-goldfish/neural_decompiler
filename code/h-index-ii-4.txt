def hIndex(self, citations):
    """
    :type citations: List[int]
    :rtype: int
    """
    if not citations:
        return 0




    # newCitation = []
    # for cit in citations:
    #     if cit != 0:
    #         newCitation.append(cit)
    #
    # citations = newCitation
    print "citatinos = ", citations

    numCitations = len(citations)
    print "numCitations = ", numCitations


    tmpRes = 0
    l = 0
    r = numCitations-1
    #print "0 r  = ", r
    m = 0
    while l <= r:
        m = l + (r-l)/2
        #print "1 l = ", l, " m =  ",m, " r = ", r

        curCitation = citations[m]
        biggerPapers = numCitations - m

        #print "curCitation = ", curCitation, " biggerPapers = ", biggerPapers

        if biggerPapers >= curCitation:
            if m > l:
                l = m
            else:
                l += 1
        else:
            tmpRes = biggerPapers
            if r > m:
                r = m
            else:
                r -= 1

    #print "2 l = ", l, " m =  ", m, " r = ", r
    #print "tmpRes = ", tmpRes

    if r >= 0:
        return max(citations[r], tmpRes)
    else:
        return numCitations
