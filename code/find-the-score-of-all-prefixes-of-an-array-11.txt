class Solution {
public:
    vector<long long> findPrefixScore(vector<int>& nums) {
        long long x = nums[0];
        for(int i=0;i<nums.size();i++){
            if(x<nums[i]) x = nums[i];
            nums[i] += x;
        }
        vector<long long> ans(nums.size());
        ans[0] = nums[0];
        for(int i=1;i<nums.size();i++){
            ans[i] += (nums[i]+ans[i-1]);
        }
        return ans;
    }
};
