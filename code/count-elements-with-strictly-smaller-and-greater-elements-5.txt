int cmp(const void *x, const void *y)
{
    int a = *(int*)x;
    int b = *(int*)y;
    return a-b;
}
int countElements(int* nums, int numsSize)
{
    int count = 0;
    qsort(nums, numsSize, sizeof(int), cmp);
    for(int i = 1; i < numsSize-1; i++)
    {
        if((nums[0] != nums[i]) && (nums[i] < nums[numsSize-1]))
        {
            count++;
        }
    }
    return count;
}
