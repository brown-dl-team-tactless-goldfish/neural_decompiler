class Solution(object):
    def rearrangeArray(self, nums):
        ans = [0] * len(nums)
        p = 0
        
        nums.sort()
        for i in range(1, len(nums), 2):
            ans[i] = nums[p]
            p += 1
        
        for i in range(0, len(nums), 2):
            ans[i] = nums[p]
            p += 1
        
        return ans
