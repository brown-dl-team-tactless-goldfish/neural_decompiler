bool isLongPressedName(char * name, char * typed){
    if (*name != *typed) return false;
    else if (!*name) return true;           // condition could also be (*name == '\0' || *typed == '\0')
    else if (*(name+1) == *(typed+1)) return isLongPressedName(++name, ++typed);
    return isLongPressedName(name, ++typed);
}
