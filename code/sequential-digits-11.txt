    vector<int> res;
    
    for(int i=1; i<=9; i++)
    {
        q.push(i);
    }
    
    while(!q.empty())
    {
        int cur = q.front();
        q.pop();
        
        if(cur>= low && cur <= high)
        {
            res.push_back(cur);
        }
        
        if(cur > high)
        {
            break;
        }
        
        int rem = cur % 10;
        
        if(rem< 9)
        {
            q.push(cur*10 + rem+1);
        }
    }
    
    return res;
}
