public class Solution 
{
    public int BulbSwitch(int n)
    {
        if(n == 0)
            return 0;
        
        if(n < 4)
            return 1;
        
        int bulbs = 2;
        int a = 2;
        int b = 4;
        int c = a*b;
        
        for(int i = 4; i < n; i++)
        {
            if(i < c)
                continue;
            
            bulbs++;
            a++;
            b++;
            c = a * b;
        }
        
        return bulbs;
    }
}
