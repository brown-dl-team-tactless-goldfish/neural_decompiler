bool isPossibleDivide(vector<int>& nums, int k) {
	map<int, int> mp;
	for (int n : nums)
		mp[n]++;

	int cardNum = 0;
	while (mp.size() >= 1) {
		int curCard = mp.begin()->first;
		while (cardNum++ < k) {
			if (mp.find(curCard) == mp.end())
				return false;

			mp[curCard]--;
			if (mp[curCard] == 0)
				mp.erase(curCard);

			curCard++;
		}
		cardNum = 0;
	}
	return mp.size() == 0;
}
