public class Solution {
    public string MaximumTime(string time) {
        string res = "";
        for(int i = 0; i < time.Length;i++){
            char c = time[i];
            if(time[i] == '?' && i > 2) c = (i == 3)?'5':'9';
            else if(time[i] == '?' && i == 0 ) c = '2';
            else if(time[i] == '?') c = (res[i - 1] == '2')?'3':'9';
            res += c;
        }
        if(Int32.Parse(res.Substring(0,2)) > 23) res = "1"+res.Substring(1);
        return res;
    }
}
