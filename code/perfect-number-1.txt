bool checkPerfectNumber(int num){
    if (num == 1)
        return false;
    int sum = 1;
    int x = sqrt(num);
    for (int i = 2; i <= x; i++){
        if (num % i == 0){
            sum += i;
            sum += num / i;
        }
    }
    if (sum == num)
        return true;
    else
        return false;
}
