class Solution {
public:
    bool escapeGhosts(vector<vector<int>>& g, vector<int>& t){
        int s=abs(t[0])+abs(t[1]);
        for(int i=0;i<g.size();i++){
            int x=g[i][0];
            int y=g[i][1];
            int h = abs(x-t[0]) + abs(y-t[1]);
            if(h<=s) return false;
        }
        return true;
    }
};
