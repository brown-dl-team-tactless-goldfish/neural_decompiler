class Solution {
public:
    vector<vector<int>> shiftGrid(vector<vector<int>>& mat, int k) {
        int r = mat.size(),c = mat[0].size(); 
        int sz = r*c; vector<int> a(r*c,0);
       
        for(int i=0;i<r*c;i++){
            a[(i+k)%sz] = mat[i / c][i % c];
        }
        
        for(int i = 0;i<r*c;i++){
            mat[i / c][i % c]  = a[i];
        }
        
        return mat;
    }
};
