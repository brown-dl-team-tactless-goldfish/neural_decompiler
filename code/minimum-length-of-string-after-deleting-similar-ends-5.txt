class Solution {
public:
    int minimumLength(const string& s) {
        int i{0}, size{(int)s.size()}, j{size-1};
        char c;
        while(i < j) {
            if(s[i] != s[j]) break;
            c = s[i];
            ++i, --j;
            while(i < size && s[i] == s[i-1]) ++i;
            while(j >= 0 && s[j] == s[j+1]) --j;
        }
        
        return j < i ? 0 : j-i+1;
    }
};
