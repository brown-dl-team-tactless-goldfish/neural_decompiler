// The rand7() API is already defined for you.
// int rand7();
// @return a random integer in the range 1 to 7

int rand10() {
    int x=0,y=0;
    while(1){
        x=rand7();
        if(x>5){
            if(x==7){
                x=5;
            }else{
                x=0;
            }
            break;
        }
    }
    while(1){
        y=rand7();
        if(y<=5){
            break;
        }
    }
    return x+y;
}
