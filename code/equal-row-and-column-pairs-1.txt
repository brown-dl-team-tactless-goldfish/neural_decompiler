class Solution:
    def equalPairs(self, grid: List[List[int]]) -> int:
        num_rows = len(grid)
        num_cols = num_rows
        
        rows = []
        for r in range(num_rows):
            rows.append(grid[r])
            
        print ("rows = ", rows)
        
        cols = []
        for c in range(num_cols):
            tmp = []
            for r in range(num_rows):
                tmp.append(grid[r][c])
                
            cols.append(tmp)

            
        print ("cols = ", cols)
        
        res = 0
        for col in cols:
            cnt = rows.count(col)
            res += cnt
            
        return res
