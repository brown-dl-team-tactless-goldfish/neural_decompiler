bool validUtf8(int* data, int dataSize){
    int i = 0;
    int count = 0;
    int start = 0;
    while(count<dataSize)
    {
        for(i=0;i<6;i++)
        {
            if(i==5)
            {
                return false;
            }
            if(((data[start]>>(7-i))&0x01) == 0)
            {
                if(i==0)
                {
                    count++;
                }
                else if((count > dataSize) || (((count-start) == 1)))
                {
                    return false;
                }
                break;
            }
            else
            {
                count++;
            }

        }
        for(i=start+1;i<count;i++)
        {
            if((data[i]&0xC0) != 0x80)
            {
                return false;
            }
        }
        start = count;
    }
    
    
    return true;
    
}
