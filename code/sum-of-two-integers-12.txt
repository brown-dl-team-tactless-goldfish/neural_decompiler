int getSum(int a, int b) {
    uint32_t mask = 1;
    int result = 0;
    uint32_t carry = 0;
    
    for (int i=0; i<32; i++) {
        uint32_t c = carry << i;
        result |= ((a ^ b ^ c) & mask);
        if ((a & b & mask) | (a & c & mask) | (b & c & mask)) {
            carry = 1;
        } else {
            carry = 0;
        }
        mask <<= 1;
    }
    return result;
}
