#include<string.h>
int first[27];
int second[27][27],third[27][27];

int countPalindromicSubsequence(char * s){
    int n = strlen(s);
    memset(first,0,sizeof(first));
    memset(second,0,sizeof(second));
    memset(third,0,sizeof(third));
    for(int i = 0;i<n;i++){
        for(int j=0;j<26;j++){
            third[j][s[i]-'a'] = second[s[i]-'a'][j];
        }
        for(int j=0;j<26;j++){
            second[j][s[i]-'a'] = first[j];
        }
        first[s[i]-'a']=1;
    }
    int ans=0;
    for(int i=0;i<26;i++){
        for(int j=0;j<26;j++){
        ans+=third[i][j];
    }}
    return ans;
}
