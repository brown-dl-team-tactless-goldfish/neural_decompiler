        { 
            
             m =  m>abs(matrix[i][j]) ? abs(matrix[i][j]) : m;
            if(matrix[i][j] < 0)

                c++;
                sum += abs(matrix[i][j]);
            }
            else
             sum += matrix[i][j];
                
        
         }
