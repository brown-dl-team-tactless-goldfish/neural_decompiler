int vowelStrings(char ** words, int wordsSize, int left, int right){
    char vowel[5]={'a','e','i','o','u'};
    int ans = 0;
    for(int i=left;i<=right;i++){
        int s=0,e=0;
        for(int j=0;j<5;j++){
            if(vowel[j]==words[i][0]){
                s=1;
            }
            if(vowel[j]==words[i][strlen(words[i])-1]){
                e=1;
            }
        }
        if(s==1 && e==1){
            ans++;
        }
    }
    return ans;
}
