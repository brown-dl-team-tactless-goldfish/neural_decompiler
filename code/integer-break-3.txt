int integerBreak(int n) {
    int temp=1;
    int x=0;
    int y=0;
    int j=0;
    int ret=0;
    if(n==2){
        return 1;
    }
    for(int i=2;i<n;i++){
        temp=1;
        x=n/i;
        y=n%i;
        for(j=0;j<i-y;j++){
            temp=temp*x;
        }
        for(;j<i;j++){
            temp=temp*(x+1);
        }
        if(temp>ret){
            ret=temp;
        }
        else{
            return ret;
        }
    }
    return ret;
}
