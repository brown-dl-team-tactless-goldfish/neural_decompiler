bool canThreePartsEqualSum(int* arr, int arrSize){
    int n = arrSize;
    int total  = 0;
    for(int i = 0; i < n; i++){
        total += arr[i];
    }
    
    if(total %3 )
        return false;
    int avg = total/3;
    int sum = 0;
    int gp = 0;
    for(int i = 0; i < n; i++){
        sum += arr[i];
        if(sum == avg){
            sum = 0;
            gp++;
        }
    }
    if(gp >= 3)
        return true;
    else
        return false;
}
