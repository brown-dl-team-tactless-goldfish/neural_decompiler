class Solution 
{
public:
    string makeFancyString(string s)
    {
        int i=0;
        while(i<s.length())
        {
            char c=s[i];
            int count=1;
            while(i<s.length() && s[i]==c)
            {
                // if there are more than 2 consecutive same characters then mark them with '#' so that they can be discarded later
                if(count>2)
                {
                    s[i]='#';
                }
                count++;
                i++;
            }
        }
        
        // take an empty string that will the result
        string ans="";
        
        // iterate over the actual string and discard the marked characters
        for(int i=0;i<s.length();i++)
        {
            if(s[i]!='#')
            {
                ans.push_back(s[i]);
            }
        }
        return ans;
    }
};
