int countNegatives(int** grid, int gridSize, int* gridColSize){
    int i, j, count=0, jMaxTmp, jMax=*gridColSize-1;
    for(i=0;i<gridSize;i++){
        for(j=0;j<=jMax;j++)
            if(grid[i][j]>=0){
                count++;
                jMaxTmp=j;
            }
        jMax = jMaxTmp;
    }
            
    return gridSize*(*gridColSize) - count;
                
}
