    int numRabbits(vector<int>& answers) {
        vector<int> occ(1000, 0);
        for (auto& a : answers)
            occ[a]++;
        
        int total = 0;
        for (auto i = 0; i < occ.size(); ++i) {
            int c = ceil(float(occ[i]) / float((i+1)));
            total += (i+1)*c;
        }
        return total;
    }
