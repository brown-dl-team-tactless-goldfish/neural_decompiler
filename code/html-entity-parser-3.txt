char * entityParser(char * text){
    char *t = text, *b = text;
    struct { const char *e, c, l } m[] = {
        "&quot;", '"', 6, "&apos;", '\'', 6, "&amp;", '&', 5,
        "&gt;", '>', 4, "&lt;", '<', 4, "&frasl;", '/', 7
    };
    for (int i = -1, j = 0 ; *t ; t += i < 6 ? *b++ = m[i].c, m[i].l : (*b++ = *t, 1), i = -1)
        while (++i < 6 && strncmp(t, m[i].e, m[i].l));
    return *b = 0, text;
}
