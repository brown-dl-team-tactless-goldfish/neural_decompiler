class Solution {
public:
    bool canArrange(vector<int>& arr, int k) {
        unordered_map<int, int> counter;
        
        for (const auto& num : arr) {
            counter[(num % k + k) % k]++; 
        }
        
        for (int i = 1; i < k - i; i++) {
            if ((counter[i] != counter[k - i])) return false;
        }
        
        if (k % 2 == 0 && (counter[k / 2] % 2)) return false;
        
        return true;
    }
};
