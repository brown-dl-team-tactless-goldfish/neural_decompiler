int findKthPositive(int* arr, int arrSize, int k){
    int max = arr[arrSize - 1] + 1;
    bool hash_map[max];
    memset(hash_map, 0, sizeof(bool) * max);
    
    for (int i = 0; i < arrSize; i++)
    {
        hash_map[arr[i]] = true;
    }
    
    int count = 0;
    for (int i = 1; i < max; i++)
    {
        count += !hash_map[i];
        if (count == k)
        {
            return i;
        }
    }

    return (k - count + max - 1);
}
