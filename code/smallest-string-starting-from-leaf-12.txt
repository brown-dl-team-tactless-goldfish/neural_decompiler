class Solution {
public:
    string smallestFromLeaf(TreeNode* root) {
        string res, cur;
        smallestFromLeaf(root, cur, res);
        return res;
    }
    
private:
    void smallestFromLeaf(TreeNode* root, string& cur, string& res)
    {
        if (!root)
        {
            return;
        }
        
        cur = (char)(root->val + 'a') + cur;
        if (!root->left && !root->right)
        {
            res = res.empty() ? cur : min(res, cur);
        }
        
        smallestFromLeaf(root->left, cur, res);
        smallestFromLeaf(root->right, cur, res);
        cur = cur.substr(1);
    }
};
