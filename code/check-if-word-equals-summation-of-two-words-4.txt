bool isSumEqual(char * firstWord, char * secondWord, char * targetWord){
    int f=0,s=0,t=0;
    int k=1;
    for(int i=strlen(firstWord)-1;i>=0;i--){
        f = f+((int)firstWord[i]-97)*k;
        k=k*10;
    }
    k=1;
    for(int i=strlen(secondWord)-1;i>=0;i--){
        s = s+((int)secondWord[i]-97)*k;
        k=k*10;
    }
    k=1;
    for(int i=strlen(targetWord)-1;i>=0;i--){
        t = t+((int)targetWord[i]-97)*k;
        k=k*10;
    }
    return s+f==t;
}
