/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     struct ListNode *next;
 * };
 */


int getDecimalValue(struct ListNode* head){
int accumulator=0;
while (head!=NULL) {
    accumulator = (2 * accumulator) + head->val;
    head = head->next;
}
return accumulator;
}
