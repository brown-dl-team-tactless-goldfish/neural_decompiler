if count of -ve sign is even they will eliminate themselves  so just print sum
if count of -ve sign is odd then one of them cannot be replaced so optimal element is minimal of matrix

class Solution {
public:
    long long maxMatrixSum(vector<vector<int>>& matrix) {
        int n=matrix.size();
        
        int mina=INT_MAX;
        int cnt=0;
        long long sum=0;
        for(int i=0; i<n; i++)
        {
            for(int j=0; j<n; j++)
            {
                mina=min(abs(matrix[i][j]),mina);
                sum+=abs(matrix[i][j]);
                if(matrix[i][j]<0)
                    cnt++;
            }
        }
        
        if(cnt%2==0)
            return sum;
        else
            return sum-2*mina;
        
    }
};
