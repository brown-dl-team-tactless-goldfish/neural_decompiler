class LUPrefix {
public:
    unordered_set<int> st;
    int cnt;
    LUPrefix(int n) {
        cnt=0;
    }
    
    void upload(int video) {
        st.insert(video);
        while(st.count(cnt+1))
            cnt++;
    }
    
    int longest() {
        return cnt;
    }
};

/**
 * Your LUPrefix object will be instantiated and called as such:
 * LUPrefix* obj = new LUPrefix(n);
 * obj->upload(video);
 * int param_2 = obj->longest();
 */
