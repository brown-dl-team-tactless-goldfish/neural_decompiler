class Solution {
public:
    int partitionArray(vector<int>& nums, int k) {
        int n = nums.size();
        sort(nums.begin(), nums.end());
        int c = 0; 
        int j = 0;
        for(int i=1;i<n;i++){
            if(nums[i] - nums[j] > k){
                c++;
                j = i;
            }
        }
        c++;
        return c;
    }
};
