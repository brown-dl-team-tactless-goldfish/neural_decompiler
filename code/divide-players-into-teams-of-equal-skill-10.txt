    public long DividePlayers(int[] skill) 
    {
        long result = 0;
        Array.Sort(skill);
        int n = skill.Length;
        int twoSum = skill[0] + skill[n - 1];

        for (int i = 0; i < n/2; i++)
        {
            if (skill[i] + skill[n - 1 - i] != twoSum)
                return -1;
            result += skill[i] * skill[n - 1 - i];
        }
        return result;
    }
