char * removeDuplicates(char * s){
    int i = 1, index = 0, len = strlen(s);
    char *st = malloc(len+1);
    
    st[0] = s[0];
    
    while (i < len) {
        if (index >= 0 && st[index] == s[i]) 
            index--;
        else 
            st[++index] = s[i];
        
        i++;
    }
    
    st[index+1] = '\0';
    return st;
}
