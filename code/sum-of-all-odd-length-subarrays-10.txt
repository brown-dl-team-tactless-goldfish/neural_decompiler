int sumOddLengthSubarrays(int *arr, int arrSize)
{
	int sum = 0;
	int window_sum = 0;
	int left = 0, right = 0;

	for(int i=0; i<arrSize; i+=2)
	{
		left = 0;
		right = i;
		window_sum = 0;
		// first window sum
		for(int j=left; j<=right; ++j)
			window_sum += arr[j];
		sum += window_sum;
		while(right+1 < arrSize) // check the right boundary
		{
			window_sum = window_sum + arr[++right] - arr[left++]; // slide the window to right
			sum += window_sum;
		}
	}
	return sum;
}
