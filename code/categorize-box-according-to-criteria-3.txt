string categorizeBox(int l, int w, int h, int m) {
	long v = (long)l * w * h;
	bool Bulky = v >= pow(10, 9) ||
				(l >= pow(10, 4) || w >= pow(10, 4) || h >= pow(10, 4));
	bool Heavy = m >= 100;
	return Bulky && Heavy ? "Both" :
		   (!Bulky && !Heavy) ? "Neither" :
		   (Bulky && !Heavy) ? "Bulky" : "Heavy";
}
