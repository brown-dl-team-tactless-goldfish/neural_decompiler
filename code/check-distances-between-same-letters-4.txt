bool checkDistances(char * s, int* distance, int distanceSize){
    char ch;
    int d,pos1,pos2;
    for(int i=0;i<distanceSize;i++){
        d = distance[i];
        ch = i + 'a';
        pos1 = -1;
        for(int j=0;j<strlen(s);j++){
            if(s[j]==ch && pos1==-1)
                pos1 = j;
            if(s[j]==ch && pos1!=-1)
                pos2 = j;
        }
        if(pos2-pos1-1!=d && pos1!=-1)
            return false;
    }
    return true;
}
