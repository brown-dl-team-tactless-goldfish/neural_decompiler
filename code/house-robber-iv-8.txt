int minCapability(vector<int>& nums, int k) {
int s = 1, e = 1e9;
while (s < e) {
    int mid = (s + e) / 2;
    int take = 0;
    for (int i = 0; i < nums.size() && take < k; i++) {
        take += nums[i] <= mid;
        i += nums[i] <= mid;
    }
    if (take < k)
        s = mid + 1;
    else
        e = mid;
}
return s;
}
