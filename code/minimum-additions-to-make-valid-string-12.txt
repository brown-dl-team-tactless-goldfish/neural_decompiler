class Solution {
public:
    void helper(string s, char compa, int i, int& cnt){
        if (i>=s.size())return;
        if (s[i]!=compa){cnt++;i=i-1;}
        helper(s,('a'+(compa-'a'+1)%3),i+1,cnt);
    }
    int addMinimum(string s) {
        int cnt=0;
        int n= s.size();
        char compa='a';
        helper(s,compa,0,cnt);
        if (s.back()=='a')cnt+=2;
        if (s.back()=='b')cnt++;
        return cnt;
    }
};
