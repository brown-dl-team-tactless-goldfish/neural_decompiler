class Solution {
public:
    bool im(char *s, char *p) {        
        int adv = 1; 
        if (*p == 0) return *s == 0;
        if (*p && *(p+1) && *(p+1) == '*') { if (im(s, p + 2)) return 1; adv = 0; }
        if (*s && *p == '.' || *s == *p) return im(s + 1, p + adv);
        return 0;
    }
    bool isMatch(string s, string p) {
        return im((char *)s.c_str(), (char *)p.c_str());
    }
};
