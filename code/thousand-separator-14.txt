class Solution {
public:
    string thousandSeparator(int n) {
        string ans="";
        if(n==0) {
            ans = "0";
            return ans;
        }
        int c=0;
        while(n>0) {
            if(c==3) {
                ans = "." + ans;
                c =0;
            }
            ans = to_string(n%10) + ans;
            n = n/10;
            c++;   
        }
    return ans;
    }
};
