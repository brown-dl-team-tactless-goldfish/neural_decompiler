/**
 * Note: The returned array must be malloced, assume caller calls free().
 */
int* minOperations(char * boxes, int* returnSize){
    int *ans = malloc(sizeof(int)*strlen(boxes));;
    int i=0;
    for(i=0;i<strlen(boxes);i++){
        int res = 0;
        for(int j=0;j<strlen(boxes);j++){
            if(i!=j && boxes[j]=='1'){
                res = res + abs(i-j);
            }
        }
        ans[i]=res;
    }
    *returnSize = i;
    return ans;
}
