class Solution {
public:
    vector<int> missingRolls(vector<int>& rolls, int mean, int n) 
    {
        int val = (rolls.size()+n)*mean - accumulate(rolls.begin(), rolls.end(), 0);
        if(val>n*6 || val<0 || val<n)
            return {};
        int rem = val%n;
        vector<int> res(n, val/n); 
        for(int i=0; i<res.size(); i++)
        {
            if(rem != 0)
                res[i]+=1, rem--;
            else
                break;
        }
        return res;
    }
};
