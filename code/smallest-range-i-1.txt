int smallestRangeI(int *A, int ASize, int K) {
    int max = -1, min = 10001;
    for (int i = 0; i < ASize; ++i) {
        if (A[i] > max) {
            max = A[i];
        }
        if (A[i] < min) {
            min = A[i];
        }
    }
    if (max - min > 2 * K) {
        return max - min - 2 * K;
    }
    return 0;
}
