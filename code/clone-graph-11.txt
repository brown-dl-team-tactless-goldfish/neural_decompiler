    if not node:
        return
    
    copyRoot = UndirectedGraphNode(node.label)
    dic[node] = copyRoot
    
    
    
    queue.append(node)
    
    while queue:
        curNode = queue.pop(0)

        
        for nb in curNode.neighbors:
            #print "nb = ", nb.label
            if nb in dic:
                tmpNb = (dic[nb])
            else:
                tmpNb = UndirectedGraphNode(nb.label)
                dic[nb] = tmpNb
                queue.append(nb)
                
            dic[curNode].neighbors.append(tmpNb)    
    #print "dic = ", dic
            
    return dic[node]
