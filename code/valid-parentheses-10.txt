bool isValid(char * s){
  if (strlen(s) % 2 != 0) {
    return false;
  }
  
  char *stack = malloc(strlen(s));
  int top = 0;
  
  for (int i = 0; i < strlen(s); i++) {
    char c = s[i];
    
    if (c == ')') {
      if (top == 0 || stack[top - 1] != '(') {
        top = -1;
        break;
      }
      
      top--;
    } else if (c == ']') {
      if (top == 0 || stack[top - 1] != '[') {
        top = -1;
        break;
      }
      
      top--;
    } else if (c == '}') {
      if (top == 0 || stack[top - 1] != '{') {
        top = -1;
        break;
      }
      
      top--;
    } else {
      stack[top] = c;
      top++;
    }
  }
  
  free(stack);
  return top == 0;
}
