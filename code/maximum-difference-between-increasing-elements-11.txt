public class Solution {
    public int MaximumDifference(int[] nums) {
        
        if(nums == null || nums.Length == 0)
            return -1;
        
        int max = -1, minNum = nums[0];
        for(int i = 1; i < nums.Length; i++)
        {
            if(nums[i] != minNum)
                max = Math.Max(max, nums[i] - minNum);

            minNum = Math.Min(minNum, nums[i]);     
        }
        
        return max;
    }
}
