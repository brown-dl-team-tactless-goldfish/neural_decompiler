int score(char* S,int len){
    if(len==2){
        return 1;
    }
    return 2*split(&S[1],len-2);
}
int split(char* S,int len){
    int sum=0;
    int start=0;
    int count=0;
    for(int i=0;i<len;i++){
        if(S[i]=='('){
            count++;
        }else{
            count--;
        }
        if(count==0){
            sum+=score(&S[start],i-start+1);
            start=i+1;;
        }
    }
    return sum;
}
int scoreOfParentheses(char* S) {
    return split(S,strlen(S));
}
