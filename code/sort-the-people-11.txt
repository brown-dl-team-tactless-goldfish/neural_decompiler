class Solution {
public:
    vector<string> sortPeople(vector<string>& names, vector<int>& heights) {
        vector<pair<int,string>> sortednames;  //create vector of pair
        for(int i=0;i<names.size();i++){
            sortednames.push_back({heights[i],names[i]});   //storing values in pair
        }
        sort(sortednames.rbegin(),sortednames.rend());      //sorting
        vector<string> ans;
        for(int i=0;i<sortednames.size();i++)               //storing names a/c to desc. order height
            ans.push_back(sortednames[i].second);
        return ans;
    }
};
