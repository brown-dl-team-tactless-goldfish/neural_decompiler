    if(length%primes[j]==0)
    {
        half = length/primes[j];
        for(i=0;i<half;i++)
        {
            k = primes[j]-1;
            while(k>0)
            {
                if(s[i]!=s[i+(half*k)])
                    goto next;
                k--;
            }
        }
        return 1;
    }
    next:
        continue;
}   
return 0;
