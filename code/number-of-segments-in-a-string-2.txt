int countSegments(char * s){
    int ret = 0;
    bool bIn = false;
    int i;
    
    for (i = 0; s[i] != '\0'; i++) {
        if (!bIn && s[i] != ' ') {
            bIn = true;
            ret++;
        }
        else if (s[i] == ' ') {
            bIn = false;
        }
    }
    
    return ret;
}
