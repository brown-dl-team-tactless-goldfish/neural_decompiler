	string decodeMessage(string key, string message) {
        int i = 97;
        map<char, int> mp;
        map<char, char> decode;
        for(char c : key){
            mp[c]++;
            if(mp[c]==1){
                if(c != ' ') decode[c] = char(i++);
            }
        }
        string s = "";
        for(char c : message){
            if(c != ' ') s += decode[c];
            else s += ' ';
        }
        return s;
    }
