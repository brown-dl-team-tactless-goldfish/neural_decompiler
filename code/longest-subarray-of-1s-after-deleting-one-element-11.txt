public class Solution 
{
    public int LongestSubarray(int[] nums) 
    {
        int start = 0, end = 0, zeros = 0, max = 0;
        while(end < nums.Length)
        {
            if(nums[end++] == 0)
                zeros++;
            while(zeros > 1)
                if(nums[start++] == 0)
                    zeros--;
            max = Math.Max(max, end - start - 1);
        }
        
        return max;
    }
}
