class Solution {
public:
    int minDifference(vector<int>& nums) {
        sort(nums.begin(), nums.end());
        int count = 0;
        if (nums.size() <= 4)   return 0;
        
        int n = nums.size();
        int a = nums[n - 1] - nums[3], b = nums[n - 2] - nums[2], 
            c = nums[n - 3] - nums[1], d = nums[n - 4]- nums[0];
        return min(min(a, b), min(c, d));
    }
};
