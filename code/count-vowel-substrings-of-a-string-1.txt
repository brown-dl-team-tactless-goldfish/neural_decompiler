#define vowel  0x104111

int countVowelSubstrings(char * word){
    int idx = 0, len = strlen(word);
    int ansCnt = 0, mask;
    for(int i = 0; i < len-4; i++)
    {
        idx = i; 
        mask = (1 << (word[idx]-'a'));
        while(idx < len && mask & vowel)
        {
            mask |= (1 << (word[idx++]-'a'));
            if(mask == vowel)
                ansCnt++;
            else if(mask & ~vowel)
                break;
        }
    }
    return ansCnt;
}
