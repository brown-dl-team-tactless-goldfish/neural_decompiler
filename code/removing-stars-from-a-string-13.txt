class Solution {
    public String removeStars(String s) {
        char[] sArr = s.toCharArray();
        int writeP = 1;
        int readP = 1;
        while(readP < s.length()){
            sArr[writeP] = s.charAt(readP);
            if(s.charAt(readP) == '*'){
                writeP -= 2;
            }
            writeP++;
            readP++;
        }
        return new String(sArr, 0, writeP);
    }
}
