class Solution {
public:
    int m = 1000000007;
    int countHousePlacements(int n) {
        long long aa = 1, bb = 1, c;
        
        while(n--){
            c = (aa + bb) % m;
            aa = bb;
            bb = c;
        }
        
        c = (c * c) % m;
            
        return c;
    }
};
