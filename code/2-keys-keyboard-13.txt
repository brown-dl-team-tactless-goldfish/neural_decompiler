int minSteps(int n) {
    int c = 0;
	//copy and paste untill n is even and it takes 2 cost (1 for copy and 1 for paste).
    while(n%2==0){
        n/=2;
        c+=2; // 2 cost for copy and paste
     }
    int x = 3; // taking 3 because if n remains 1 so we're not proceed operations.
    while(x<=n){
        while(n%x==0){
            n/=x;
            c+=x;
        }
        x++;
    }
    return c;
}
