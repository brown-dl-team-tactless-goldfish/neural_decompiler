int cmpfunc (const void * a, const void * b)
{
   return ( *(int*)a - *(int*)b );
}



int check(int* nums, int numsSize){
        int a=numsSize-1;//æœ€å¤§å€¼
        int i=0;
        int ans=0;
        do{
            if(nums[a-2-i]+nums[a-1-i]>nums[a-i]){
                ans=nums[a-2-i]+nums[a-1-i]+nums[a-i];
