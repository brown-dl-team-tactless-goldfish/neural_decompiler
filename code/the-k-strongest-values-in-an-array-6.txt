class Solution {
public:
    using VI = vector<int>;
    VI getStrongest(VI& A, int K) {
        sort(A.begin(), A.end());
        auto m = A[(A.size() - 1) / 2];
        sort(A.begin(), A.end(), [&](auto a, auto b) { return abs(b - m) == abs(a - m) ? b < a : abs(b - m) < abs(a - m); });
        A.resize(K);
        return A;
    }
};
