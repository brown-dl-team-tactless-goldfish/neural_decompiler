class Solution {
public:
    long long solve(vector<vector<int>>&A,vector<long long >&DP,int i,int n)
    {
        if(i >= n)
            return 0;
        if(DP[i] != LLONG_MIN)
           return DP[i];
        DP[i] = max(solve(A,DP,i + 1,n), A[i][0] + solve(A,DP,i + A[i][1] + 1,n));
        return DP[i];
        
    }
    long long mostPoints(vector<vector<int>>&A) {
        int n = A.size();
        vector<long long >DP(n,LLONG_MIN);
        return solve(A,DP,0,n);
        
    }
};
