class Solution {
public:
    int passThePillow(int n, int time) {
        int c=0,t=0;
        for(int i=0;i<=time;i++)
        {
            if(c==1)
               t=0;                
            if(c==n)
               t=-1;   
            if(t==0)
               c++;    
            else if(t==-1)
               c--;    
        }
        return c;
    }
};
