class Solution {
public:
    vector<long long> findPrefixScore(vector<int>& nums) {
        vector<long long> pre((int)nums.size(),0);
        pre[0] = nums[0]*2;
        int mx = nums[0];
        for(int i= 1;i<size(nums);i++){
            mx = max(mx,nums[i]);
            pre[i] += pre[i-1] + nums[i] + mx;
        }
        return pre;
    }
};
