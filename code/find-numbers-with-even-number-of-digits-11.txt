

int findNumbers(int* nums, int numsSize){
    int even_digs_count = 0;
    
    for (int i = 0; i < numsSize; ++i){
        int curr_digits = 0;
        int n = nums[i];
        while (n >= 10){
            ++curr_digits;
            n /= 10;
        }
        ++curr_digits;
        if (curr_digits % 2 == 0){
            ++even_digs_count;
        }
    }
    
    return even_digs_count;
}
