int countBattleships(char** board, int boardRowSize, int boardColSize) {
    int count=0,i,j;
    for(i=0;i<boardRowSize;++i)
        for(j=0;j<boardColSize;++j){
            if((board[i][j]=='X')&&(i==0||board[i-1][j]=='.')&&(j==0||board[i][j-1]=='.'))
                ++count;
        }
    return count;
}
