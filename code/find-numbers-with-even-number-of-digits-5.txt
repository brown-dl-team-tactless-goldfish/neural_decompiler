int findNumbers(int* nums, int numsSize){
int even_count = 0;
int i;
int digit_counter;
int accumulator;
for (i=0; i<numsSize; i++) {
    digit_counter = 1;
    accumulator = nums[i] / 10;
    while (accumulator > 0) {
        digit_counter++;
        accumulator = accumulator / 10;
    }
    if ((digit_counter % 2)==0) even_count++;
}
return even_count;
}
