int* singleNumber(int* nums, int numsSize, int* returnSize) {
    int a=0,b=0,c=0,temp=1;
    int i=0;
    int *ret=(int*)calloc(2,sizeof(int));
    for(i=0;i<numsSize;i++) {
        a^=nums[i];
    }
    while((a&temp)==0) {
        temp<<=1;
    }
    for(i=0;i<numsSize;i++) {
        if(temp&nums[i]) {
            ret[0]^=nums[i];
        }
        else ret[1]^=nums[i];
    }
    *returnSize=2;
    return ret;
}
