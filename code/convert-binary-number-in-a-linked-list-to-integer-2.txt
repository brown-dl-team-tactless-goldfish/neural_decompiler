int getDecimalValue(struct ListNode* head){
    
    
    int count = 0;
    
    struct ListNode *temp = head;
    
    while(temp != NULL) {
        count++;
        temp = temp->next;
        
    }

    
    int sum = 0;
    
    count = count - 1;
    
    temp = head;
    
    while(temp != NULL && count >=0) {
        
        if(temp->val == 1)
        sum += pow(2,count);

        count--;
        temp = temp->next;
    }
    
    return sum;
}
