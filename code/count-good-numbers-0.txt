long long myPow(long long a, long long b){
    int M = 1e9 + 7 ;
    if(b == 0)
        return 1 ;
    
    if(b%2)
        return a * myPow(a , b-1) % M ;
    else
        return myPow((a*a)%M, b/2) ;
}
int countGoodNumbers(long long n){
    long long ret = 1 ;
    int M = 1e9 + 7 ;
    long long odd = n/2 ;
    long long even = n - odd ;
    
    ret = myPow(20 , odd) * ((n%2) ? 5 : 1 ) % M;
    return ret ;
}
