int minDeletionSize(char ** A, int ASize){
    int i,j, res=0;
    int n = strlen(*A);
    bool tmp = true;
    
    for(j = 0; j < n; j++){
        tmp = true;
        for(i = 1; i < ASize; i++){
            if(A[i][j]<A[i-1][j]) {tmp = false; break;}
        }
        if(!tmp) res++;
    }
    return res;
}
