public class Solution {
    public string MakeFancyString(string s) {
        List<char> list = new();
        
        foreach(var c in s)
        {
            if(list.Count < 2)
            {
                list.Add(c);
            }
            else
            {
                if(!(list.Last() == c && list[list.Count - 2] == c))
                    list.Add(c);
            }
        }
        
        return string.Join("", list);
    }
}
