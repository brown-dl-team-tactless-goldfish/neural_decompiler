bool
lemonadeChange(const int *bills, int billsSize)
{
	int bill5, bill10;

	bill5 = bill10 = 0;
	while (billsSize-- > 0) {
		switch (*bills++) {
			case 5:
				++bill5;
				break;
			case 10:
				++bill10;
				--bill5;
				break;
			case 20:
				if (bill10 == 0) {
					bill5 -= 3;
				} else {
					--bill10;
					--bill5;
				}
				break;
		}
		if (bill5 < 0)
			return false;
	}

	return true;
}
