int removeElement(int* nums, int numsSize, int val) {
int pivot = numsSize-1;
int result = 0;
for (int i=0; i<numsSize, i<=pivot; i++) {
    if (nums[i] == val) {
        for (int j=pivot; j>i; j--)      // when find an element is val, swap it with the last value that is not val
            if (nums[j] != val) {
                nums[i] = nums[j];
                //nums[j] = val;           // thanks to animageofmine, this line is no longer needed
                pivot = j-1;
                result++;
                break;
            }
    }
    else
        result++;
}
return result;
