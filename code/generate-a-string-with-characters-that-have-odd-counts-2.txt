char * generateTheString(int n){
char* ret_string;
int ptr=0;
ret_string = malloc((n+1) * sizeof(char));
if (n==0) return NULL;
if (n==1) return "x";
ret_string[n]='\0';
ret_string[ptr++]='x';
if ((n % 2)==1) while (ptr<n) ret_string[ptr++]='x';
    else while (ptr<n) ret_string[ptr++]='y';
return ret_string;
}
