class Solution {
public:
    static bool equalFrequency(const string& word) {
        array<int, 26> cnt{};
        for (auto c : word)
            ++cnt[c - 'a'];
        sort(begin(cnt), end(cnt));
        auto it = upper_bound(begin(cnt), end(cnt), 0);
        return cnt[24] == 0 || *it == 1 && it[1] == cnt[25] || *it == cnt[24] && *it == --cnt[25];
    }
};
