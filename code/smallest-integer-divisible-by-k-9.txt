class Solution {
public:
    int smallestRepunitDivByK(int K) {
        
        int N = 1;
        int len = 1;
        
        if (K % 2 == 0) return -1;
        
        while (len <= K  && N % K != 0) {
            N = (N*10+1)%K;
            len++;
        }
        
        return (len <= K) ? len : -1;
    }
};
