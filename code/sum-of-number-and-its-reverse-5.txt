public class Solution {
    public bool SumOfNumberAndReverse(int num) {
        for(int i = num / 2; i <= num; i++)
        {
            if(i + Reverse(i) == num)
                return true;
        }
        
        return false;
    }
    
    private int Reverse(int num)
    {
        char[] arr = num.ToString().ToArray();
        Array.Reverse(arr);
        return Convert.ToInt32(new string(arr));
    }
}
