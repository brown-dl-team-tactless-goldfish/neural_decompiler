int findGCD(int* nums, int numsSize){
    int temp;
   for(int i=0; i<numsSize; i++){
       for(int j=i+1; j<numsSize; j++){
           if(nums[i]>nums[j]){
               temp=nums[i];
               nums[i]=nums[j];
               nums[j]=temp;
           }
       }
   }
   int smallest=nums[0];
   int greatest=nums[numsSize-1];
   while(greatest!=0){
       temp=greatest;
       greatest=smallest%greatest;
       smallest=temp;
   }
   return smallest;
   return 0;
}
