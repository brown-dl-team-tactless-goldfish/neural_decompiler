class Solution {
public:
    string predictPartyVictory(string senate) {
        queue<char> Q;
        for (char c : senate) Q.push(c);
        int vote = 0;
        while (abs(vote) < Q.size()) {
            if (Q.front() == 'R' && vote >= 0) Q.push('R');
            if (Q.front() == 'D' && vote <= 0) Q.push('D');
            vote += Q.front() == 'R' ? 1 : -1;
            Q.pop();
        }
        return Q.front() == 'R' ? "Radiant" : "Dire";
    }
};
