int cal(struct TreeNode* root){
    if(root == NULL) return 0;
    int left = cal(root->left);
    int right = cal(root->right);
    root->val = left + right;
    return (left > right ? left : right) + 1;
}

int max;
void findmax(struct TreeNode* root){
    if (root == NULL) return;
    if(root->val > max) max = root->val;
    findmax(root->right);
    findmax(root->left);
}

int diameterOfBinaryTree(struct TreeNode* root){
    max = -1;
    cal(root);
    findmax(root);
    return max;
}
