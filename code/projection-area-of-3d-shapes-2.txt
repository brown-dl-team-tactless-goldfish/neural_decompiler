int projectionArea(int **grid, int gridRowSize, int *gridColSizes) {
    int ret = 0, row_max = 0, col_max = 0;
    for (int i = 0; i < gridRowSize; ++i) {
        col_max = row_max = 0;
        for (int j = 0; j < gridColSizes[i]; ++j) {
            ret += grid[i][j] > 0 ? 1 : 0;
            col_max = grid[i][j] > col_max ? grid[i][j] : col_max;
            row_max = grid[j][i] > row_max ? grid[j][i] : row_max;
        }
        ret += col_max + row_max;
    }
    return ret;
}
