class Solution
{
public:
    int reverse(int n)
    {
        int ans = 0;
        while (n > 0)
        {
            ans = ans * 10 + n % 10;
            n /= 10;
        }
        return ans;
    }
    bool sumOfNumberAndReverse(int n)
    {
        for (int i = n / 2; i <= n; i++)
            if (i + reverse(i) == n)
                return true;
        return false;
    }
};
