class Solution {
public:
         int n;
    int longestSubarray(vector<int>& nums) {
         n = nums.size();
        int val=*max_element(nums.begin(),nums.end());
        int len=0;
        int count=0;
        for(int i=0;i<n;i++)
        {
            if(nums[i]==val)
            {
                count++;
            }
            else
            {
                len=max(count,len);
                count=0;
            }
        }
        return max(len,count);
    }
};
