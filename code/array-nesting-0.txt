int arrayNesting(int* nums, int numsSize){
    int n = numsSize;
    bool* visit = calloc(n , sizeof(int));
    int longest = 1;
    for(int i = 0; i < n; i++){
        int pos = i;
        int cn = 0;
        while(visit[pos] == false){
            visit[pos] = true;
            pos = nums[pos];
            cn++;
        }
        longest = fmax(longest, cn);
    }
    return longest;
}
