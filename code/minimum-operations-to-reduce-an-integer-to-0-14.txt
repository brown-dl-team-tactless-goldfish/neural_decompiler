int minOperations(int n)
{
    int i = 0, c = 0;
    while (n > 0)
    {
        if (n == pow(2, i))
        {
            n = n - pow(2, i);
            c++;
            break;
        }
        if (pow(2, i) > n)
        {
            cout << n << endl;
            n = min((pow(2, i) - n), (n - pow(2, i - 1)));
            c++;
            i = 0;
        }
        i++;
    }
    return c;
}
