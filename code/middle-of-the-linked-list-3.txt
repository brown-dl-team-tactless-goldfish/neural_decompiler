struct ListNode *middleNode(struct ListNode *head) {
    if (head->next == NULL) {
        return head;
    }
    struct ListNode *one = head, *two = head->next;
    while (two->next!=NULL){
        if (two->next->next == NULL){
            break;
        }
        two = two->next->next;
        one = one->next;
    }
    return one->next;
}
