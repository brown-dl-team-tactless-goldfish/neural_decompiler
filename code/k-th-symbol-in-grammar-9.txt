int count;
int helper(int N, int K) {
    if(K==1){return 0^count;}
    if(K==2||K==0){return 1^count;}
    int len=pow(2,N-2);
    if(K-len<=0){
        return helper(N-1,K);
    }
    count^=1;
    return helper(N-1,K-len);   
}
int kthGrammar(int N, int K) {
    count=0;
    return helper(N,K);
}
