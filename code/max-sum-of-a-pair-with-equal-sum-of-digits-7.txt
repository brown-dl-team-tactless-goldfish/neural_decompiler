    public int MaximumSum(int[] nums)
    {
        int n = nums.Length;
        Dictionary<int, int> maxSum = new Dictionary<int, int>();
        int result = -1;
        foreach (int num in nums)
        {
            int numSum = FindNumSum(num);
            if (!maxSum.ContainsKey(numSum))
                maxSum.Add(numSum, num);
            else
            {
                result = Math.Max(result, maxSum[numSum] + num);
                maxSum[numSum] = Math.Max(maxSum[numSum], num);
            }
        }
        return result;
    }

    private int FindNumSum(int num)
    {
        int result = 0;
        while (num > 0)
        {
            result += num % 10;
            num /= 10;
        }
        return result;
    }
