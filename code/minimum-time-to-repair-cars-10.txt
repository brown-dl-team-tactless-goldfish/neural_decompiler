class Solution {
public:
    long long repairCars(vector<int>& v, int cars) {
        long long i=1,j=100000000000000;
        long long ans=j;
        while(i<=j){
            long long m=(i+j)/2;
            long long tt=0;
            for(int i=0;i<v.size();i++){
                long long c=sqrt(m/v[i]);
                tt += c;
            }
            if(tt>=cars){
                ans=m;
                j=m-1;
            }
            else i=m+1;
        }
        return ans;
    }
};
