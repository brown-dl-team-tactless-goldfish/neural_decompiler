    int cnt=1;
    while(!st.empty()){
        int x=st.front();
        st.pop();
        visited[x]=true;
        for(auto i:v[x]){
            if(!visited[i]){
                    cnt++;
                    visited[i]=1;
                    st.push(i);
                }
            }
    }
    return cnt;
