int min(int a, int b){ return (a>b)?b:a; }

int timeRequiredToBuy(int* tickets, int ticketsSize, int k){
    int ticketCnt = tickets[k];
    int seconds = 0;
    for(int i = 0; i < ticketsSize; i++)
    {
        seconds += min(tickets[i], ticketCnt);
        if(k == i) ticketCnt--;
    }
    return seconds;
}
