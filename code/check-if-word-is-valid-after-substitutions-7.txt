class Solution {
public:
    bool isValid(string s) {

        //push the a and b into the stack
        //when ever you get c  make sure you have b at the top and next a
        //if not return false;
        stack<char> stk;
        for(int i=0;i<s.length();i++){
            if(s[i]=='a' || s[i]=='b')
                stk.push(s[i]);
            else 
            {
                //if it is a 'c'
                if(stk.size()<2)
                    return false;
                if(stk.top()!='b')
                    return false;
                stk.pop();
                if(stk.top()!='a')
                    return false;
                stk.pop();
                
            }
        }
        return (stk.empty());
    }
};
