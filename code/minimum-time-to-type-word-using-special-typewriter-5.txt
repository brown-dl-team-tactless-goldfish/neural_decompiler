public class Solution {
    public int MinTimeToType(string word) {
        
        int cnt = word.Length;
        char prev = 'a';
        foreach(var c in word)
        {
			//clockwise distance
            int distance = Math.Abs(c - prev);
			//choose the min distance between clockwise and counterclockwise
            cnt += Math.Min(distance, 26 - distance);
            prev = c;
        }
        
        return cnt;
    }
}
