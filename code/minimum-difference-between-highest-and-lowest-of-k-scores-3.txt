class Solution:
    def minimumDifference(self, nums: List[int], k: int) -> int:
        n = len(nums)
        nums.sort()
        
        print ("nums = ", nums)
        
        dp = []
        print ("dp = ", dp)
        
        for i in range(n - k + 1 ):
            dp.append(nums[i+k - 1] - nums[i])
            
        # print ("dp = ", dp)
        
        if (dp == []):
            return 0
        
        return min(dp)
