class Solution {
public:
    string removeDigit(string number, char digit) {
        string t, r = "";
        int l = number.length();
        for(int i=0; i<l; ++i){
            t = number;
            if(t[i] == digit ){
                t.erase(i,1);
                if(t > r){
                    r = t;
                }
            }
        }
        return r;
    }
};
