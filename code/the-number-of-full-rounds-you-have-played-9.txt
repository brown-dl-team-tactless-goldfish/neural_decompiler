class Solution {
public:
    int numberOfRounds(string l1, string l2) {
        auto fun = [](string t)->int {
            return 60*((t[0]-'0')*10+(t[1]-'0'))+(t[3]-'0')*10+(t[4]-'0');
        };
        int t1 = fun(l1), t2 = fun(l2);
        if(t2 < t1) t2 += 24*60;
        return max(0, t2/15-t1/15-(t1%15 != 0));
    }
};
