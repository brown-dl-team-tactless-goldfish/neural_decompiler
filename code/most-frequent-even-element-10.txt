class Solution {
public:    
    int mostFrequentEven(vector<int>& nums) {
            map<int, int> um;
            for(int i = 0; i < nums.size(); i++){
                if(nums[i] % 2 == 0){
                   um[nums[i]]++;    
                }
            }
            int res = -1 , mx = 0; 
            for(auto i: um){
                if(i.second > mx){
                    res = i.first, mx = i.second;
                }
            }
            return res;
    }
};
