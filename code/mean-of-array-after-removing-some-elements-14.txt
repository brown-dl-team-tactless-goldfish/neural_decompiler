class Solution {
public:
    double trimMean(vector<int>& arr) {
        sort(arr.begin(),arr.end());
        int skip = arr.size() * 0.05; 
        long long total = 0; 
        for(int i = skip;i<arr.size()-skip;i++) {
            total += arr[i];
        }
        return (double)total / (arr.size() - skip * 2);
    }
};
