class Solution {
public:
    int recu(int val, int n, int maxValue) {
        int t = 1;
        for (int i=1; i<n+1; i++) {
            for (int j=2; j<maxValue/val+1; j++) {
                t += recu(val*j, n-i, maxValue);
            }
        }
        return t;
    }
    int idealArrays(int n, int maxValue) {
        return recu(1, n, maxValue);
    }
};
