int findLongestChain(vector<vector<int>>& pairs) {
    
    int c = 1;
    
    vector<pair<int,int>> v;
    
    for(auto x : pairs)
    {
        v.push_back({x[0], x[1]});
    }
    
    sort(v.begin( ), v.end( ), comp);
    
    auto t = v[0];
    
    for(int i=1; i < v.size( ); i++)
    {
        if(v[i].first > t.second)
        {
            c++;
            
            t  = v[i];
        }
        
        
    }
    
    return c;
    
}
