/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     struct TreeNode *left;
 *     struct TreeNode *right;
 * };
 */

int treetraversal(struct TreeNode* root,int sum_val){
    //condtion for termination
    if(!root){return 0;}

    //condtion to ceck we have reached the end of the tree ie leaves
    if(!root->left && !root->right){return sum_val*10 + root->val;}
    //condtion to travese the tree where we are in middle of the tree
    else{
        //traversing mechanism: reccursion 
        return treetraversal(root->left,sum_val*10 + root->val) + treetraversal(root->right,sum_val*10 + root->val);
    }
} 
int sumNumbers(struct TreeNode* root){
//mechanism to implement the tree traversal to get hold of the value read over the sumNumber
return treetraversal(root,0);
}
