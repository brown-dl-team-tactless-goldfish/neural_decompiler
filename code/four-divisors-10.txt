class Solution {
    vector<int> cntDiv(int n) {
        int cnt = 0, sum = 0;
        for(int i=1; i<=sqrt(n); i++) {
            if(n % i == 0) {
                cnt++; sum += i;
                if(i != n / i) cnt++, sum += n / i;
            }
        }
        return {cnt, sum};
    }
    
public:
    int sumFourDivisors(vector<int>& nums) {
        int ans = 0;
        for(int i : nums) {
            if(cntDiv(i)[0] == 4) ans += cntDiv(i)[1];
        }
        return ans;
    }
};
