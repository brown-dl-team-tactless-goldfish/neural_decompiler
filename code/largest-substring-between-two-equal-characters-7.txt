#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int max(int a, int b){
    return (a>b)?a:b;
}

int maxLengthBetweenEqualCharacters(char * s){
    int maxLen=-1;
    int first[26];
    int dis[26];
    int cnt=0;
    int n=strlen(s);
    for(int i=0;i<26;i++){
        first[i]=-1; dis[i]=-1;
    }
    for(int i=0;i<n;i++){
        if(first[s[i]-'a']==-1){
            first[s[i]-'a']=i;
            cnt++;
        }
    }
    if(cnt==n) return -1;
    for(int i=0;i<n;i++){
        dis[s[i]-'a']=i-first[s[i]-'a']-1;
        maxLen=max(maxLen,dis[s[i]-'a']);
    }

    return maxLen;
}
