int findUnsortedSubarray(int* nums, int numsSize){
int low = -1, high = -1;
int max = INT_MIN, min = INT_MAX;
for (int i = 0; i < numsSize; i++)
{
    if (nums[i] >= max)
    {
        max = nums[i];
    }
    else
    {
        high = i;
    }
    if (nums[numsSize - i - 1] <= min)
    {
        min = nums[numsSize - i - 1];
    }
    else
    {
        low = numsSize - i - 1;
    }
}
//printf("%d %d", low, high);
if (low < 0 || high < 0)
    return 0;

return high - low + 1;
