class Solution {
public:
    vector<int> corpFlightBookings(vector<vector<int>>& bookings, int n) {
        std::vector<int> res(n, 0);
        for(const auto &b: bookings) {
            res[b[0]-1] += b[2];
            if(b[1] < n) res[b[1]] -= b[2];
        }
        
        for(int i{0}, cur{0}; i < n; ++i) res[i] = cur += res[i];
        return res;
    }
};
