int maxNumberOfBalloons(char * text){
    int temp[26] = {0}, result = 0;
    for(int i=0; i<strlen(text); i++)
        temp[text[i]-'a']++;
    result = temp['a'-'a'];
    result = result < temp['b'-'a'] ? result : temp['b'-'a'];
    result = result < temp['n'-'a'] ? result : temp['n'-'a'];
    result = result < (temp['l'-'a']/2) ? result : (temp['l'-'a']/2);
    result = result < (temp['o'-'a']/2) ? result : (temp['o'-'a']/2);
    return result;
}
