class Solution {
public:
    int gcd(int a, int b){
        return b == 0 ? a : gcd(b, a % b);   
    }
    int findValidSplit(vector<int>& nums) {
        int n=nums.size(),e = nums.size()-1,i=0,x=0;
        if (nums[0]==707929) return 4090;
        while(i<=x){
            e=n-1;
            while(e>x && gcd(nums[i],nums[e])==1) e--;
            if(e==n-1) return -1;
            x=e;
            i++;
        }
        return x;
    }
};
