class Solution {
public:
    bool canChange(string s, string t) {
        for(int i=0,j=0,n=s.size();i<n&&j<n;i++,j++){
            while(i<n&&s[i]=='_') i++;
            while(j<n&&t[j]=='_') j++;
            if(s[i]!=t[j]) return 0;
            if(i<n&&j<n&&(s[i]=='L'&&i<j||t[j]=='R'&&i>j))
                return 0;
        }
        return 1;
    }
};
