public int MinOperations(string s) 
{        
	int result1 = 0, result2 = 0;

	for(int i = 0; i < s.Length; i++)
	{
		char c = s[i];
		char first = i % 2 == 0 ? '0' : '1';
		char second = i % 2 == 0 ? '1' : '0';

		if(c != first)
		{
			result1++;
		}

		if(c != second)
		{
			result2++;
		}            
	}


	int result = Math.Min(result1, result2);
	return result;
}
