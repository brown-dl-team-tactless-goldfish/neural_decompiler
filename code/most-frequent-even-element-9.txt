int mostFrequentEven(vector<int>& nums) 
{
    
    map<int,int> mp;
    
    for(auto &x: nums)
    {
        mp[x]++;
    }
    
    int ans = -1, res = -1;
    
    for(auto &x: mp)
    {
        if(x.first % 2 == 0 && x.second > ans)
        {
            ans = x.second;
            
            res = x.first;
        }
    }
    
    return res;
    
}
