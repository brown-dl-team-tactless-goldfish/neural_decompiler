public class Solution 
{
    public int FindKthNumber(int n, int k) 
    {
        return (int) FindKthNumber(1, n, k);
    }
    
    public long FindKthNumber(long start, long end, long k) 
    {
        if(k == 1) return start;
        var count = NumberOfChildren(start, end);
        return  count < k ? FindKthNumber(start + 1, end, k - count) : FindKthNumber(start * 10, end, k - 1);
    }
    
    public long NumberOfChildren(long start, long end) 
    {
        long childrenCount = 0, rightSibling = start + 1;
        while (start <= end) 
        {
            childrenCount += Math.Min(end + 1, rightSibling) - start;
            start *= 10;
            rightSibling *= 10;
        }
        
        return childrenCount;
    }
}
