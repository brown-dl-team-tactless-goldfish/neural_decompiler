int integerBreak(int n) {
    if (n == 2) return 1;
    if (n == 3) return 2;
    int res = n % 3;
    int num = (n - res) / 3;
    int ans = 1;
    if (res == 0) ans = pow(3, num);
    if (res == 1) {
        num--;
        ans = pow(3, num);
        ans *= (res + 3);
    }
    if (res == 2) {
        ans = pow(3, num);
        ans *= res;
    }
    return ans;
}
