public int MinSteps(string s, string t) {
	var dict = new Dictionary<char, int>();
	foreach (var c in s) {
		if (dict.ContainsKey(c)) {
			dict[c]++;
		} else {
			dict[c] = 1;
		}
	}

	var result = 0;
	foreach (var c in t) {
		if (dict.ContainsKey(c)) {
			dict[c]--;
			if (dict[c] == 0) {
				dict.Remove(c);
			}
		} else {
			result++;
		}
	}

	return result;
}
