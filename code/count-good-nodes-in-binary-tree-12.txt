 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     struct TreeNode *left;
 *     struct TreeNode *right;
 * };
 */
int count(struct TreeNode* root,int n) {
  int x=0;
    int y=n;
    if (root==NULL) {
        return 0;
    }
    if (root->val >= n) {
        x++;
        y=root->val;
    }
    return x+count(root->left,y) +count(root->right,y);
}

int goodNodes(struct TreeNode* root){
    int x=1 + count(root->left,root->val) + count(root->right,root->val);
return x;
