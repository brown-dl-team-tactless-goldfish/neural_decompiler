int bfs(struct TreeNode* root, int level) {
    if (root->left == NULL && root->right == NULL) {
        return level;
    }
    int a = root->left == NULL ? INT_MAX : bfs(root->left, level + 1);
    int b = root->right == NULL ? INT_MAX : bfs(root->right, level + 1);
    return a < b ? a : b;
}

int minDepth(struct TreeNode* root) {
    if (root == NULL) return 0;
    return bfs(root, 1);
}
