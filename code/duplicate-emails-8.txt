# row number
select email 
from (
    select email, row_number() over (partition by email) as row_num
    from person
) t
where row_num = 2

# count
select email 
from (
    select email, count(id) as c
    from person
    group by email
) t
where c > 1

# self join
select distinct p1.email
from person p1
join person p2
on p1.id < p2.id and p1.email = p2.email
