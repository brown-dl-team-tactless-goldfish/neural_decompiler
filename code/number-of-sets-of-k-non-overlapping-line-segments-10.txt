class Solution {
public:
    int numberOfSets(int n, int k) {
        vector<vector<int>> r(n, vector<int>(2,0));
        r[0][0] = 1;
        for(int i=1; i<n; i++)
        {
            for(int j= min(i-1,k); j>=0; j--)
            {
                r[j+1][1] += (r[j][0] + r[j][1]) % 1000000007;
                r[j+1][1] %= 1000000007;
                r[j][0] += r[j][1];
                r[j][0] %= 1000000007;
            }
        }
        return (r[k][0]+r[k][1])%1000000007;
    }
};
