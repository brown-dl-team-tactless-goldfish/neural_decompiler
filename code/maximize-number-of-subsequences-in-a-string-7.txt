long long maximumSubsequenceCount(string text, string pattern) 
{
	long long result=0,n=text.length(),c1=0,c2=0;
	for(int i=0;i<n;i++)
	{
		if(text[i]==pattern[0])
		{
			if(pattern[0]==pattern[1]) result+=c1++; // same characters, then just sum c up (eg. the result is just c*(c+1)/2)
			else c1++; // c and c1 is the count of pattern[0]
		}
		else if(text[i]==pattern[1]) result+=c1,c2++; // result is the counts before inserting, c2 is the count of pattern[1]
	}
	return result+max(c1,c2);
}
