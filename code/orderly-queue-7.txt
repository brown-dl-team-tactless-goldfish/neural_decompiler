class Solution {
public:
    string orderlyQueue(string s, int k) {
        if(k!=1){
            sort(s.begin(), s.end());
            return s;
        }
        string mn = s;
        int i=0; 
        while(i<s.size()){
            s.push_back(s[0]);
            s.erase(0,1);
            if(mn>s){
                mn = s;
            }
            i++;
        }
        return mn;
    }
};
