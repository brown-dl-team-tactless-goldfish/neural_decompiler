class Solution {
public:
    int res = 0;
    int m = 0;
    void dfs(vector<int>& nums, int start, int curr){
        if(curr == m)
            res++;
        
        for(int i = start;i<nums.size();i++)
            dfs(nums, i+1, curr|nums[i]);
    }
    
    int countMaxOrSubsets(vector<int>& nums) {
        for(const auto& n : nums)
            m |= n;
        
        dfs(nums, 0, 0);
        return res;
    }
};
