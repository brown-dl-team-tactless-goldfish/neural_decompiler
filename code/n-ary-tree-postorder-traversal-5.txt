int idx;

void dfs(struct Node* root, int *A){
    if(!root) return;
    for(int i=0; i<root->numChildren; i++)
        dfs(root->children[i], A);
    A[idx++]=root->val;
    return;
}
int* postorder(struct Node* root, int* returnSize) {
    int *A = (int*)malloc(10000*sizeof(int));
    idx=0;
    dfs(root, A);
    *returnSize = idx;
    return A;
}
