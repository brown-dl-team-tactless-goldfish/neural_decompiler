/**
 * Note: The returned array must be malloced, assume caller calls free().
 */
long long* findPrefixScore(int* nums, int numsSize, int* returnSize){
    *returnSize=numsSize;
    long long max_=-1;
    long long temp=0;
    long long *res=malloc(sizeof(long long)*numsSize);
    for(int i=0;i<numsSize;i++){
        if(max_<nums[i]) max_=nums[i];
        temp=nums[i]+max_+temp;
        res[i]=temp;
    }
    return res;
}
