bool isPower(long power, long product, long n) {

	if (power > n)
		return false;

	if (product == n)
		return true;

	if (product > n)
		return false;

	return isPower(power, power * product, n);
}

bool isPowerOfFour(int n) {

	if (n <= 0) return false;
	if (n == 1) return true;
	return isPower(4, 1, n);
}
