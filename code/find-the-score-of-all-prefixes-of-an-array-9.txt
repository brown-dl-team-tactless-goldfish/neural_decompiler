class Solution {
public:
    vector<long long> findPrefixScore(vector<int>& arr) {
        int n=arr.size();
        vector<long long>ans(n);
        int max_till=0;
        for(int i=0;i<n;++i){
            max_till=max(arr[i],max_till);
            long long curr=arr[i]+max_till;
            if(i==0){
                ans[i]=curr;
            }
            else ans[i]=ans[i-1]+curr;
        }
        return ans;
    }
};
