char * defangIPaddr(char * address){
    char *result = (char *)malloc(sizeof(char) * (strlen(address) + 10));
    memset(result,0,strlen(address) + 10);
    int k = 0;
    for (int i = 0;address[i] != '\0';i++) {
        if (address[i]=='.') {
            result[k] = '[';
            result[k+1] = '.';
            result[k+2] = ']';
            k+=3;
        }else {
            result[k++] = address[i];
        }
    }
    result[k] = 0;
    return result;
}
