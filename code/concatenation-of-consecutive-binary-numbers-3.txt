int concatenatedBinary(int n){
    long long ans = 0;
    int count=0;
    //int j, temp;
    for(int i=1;i<=n;i++){
        
        //Get bit-size. (method 1)
        //temp = i;        
        //count=0;
        //while(temp!=0){
        //    temp = temp>>1;
        //    count++;
        //}
        
        //Get bit-size.(method 2) faster than method 1.
        if((i & (i-1)) == 0)
            count++;
        
        ans = ((ans<<count)+i)%1000000007;
    }
    
    return ans;
}
