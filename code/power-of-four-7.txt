bool isPowerOfFour(int n){
    if (n <= 0) {
        return false;
    }
    if (n == 1) {
        return true;
    } else {
        long a = 1;
        for (int i=1; a<=n; i++) {
            a *= 4;
            if (a == n) {
                return true;
            }
        }
    }
    
    return false;
}
