bool checkprefix(char* w, char* s)
{
    for(char* t = w; *t != '\0'; t++, s++)
    {
        if(*t != *s)
            return false;
    }
    return true;
}

int countPrefixes(char ** words, int wordsSize, char * s){

    int ct = 0;
    for(int i = 0; i<wordsSize; i++)
    {
        if(checkprefix(words[i], s))
            ct++;
    }
    return ct;
    
}
