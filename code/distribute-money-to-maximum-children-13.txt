public class Solution 
{
    public int DistMoney(int money, int children) 
    {
        if (money < children)
            return -1;
        money -= children;

        int div = money / 7;
        int rem = money % 7;

        if (div == children)
        {
            if (rem == 0)
                return children;
            else
                return children - 1;
        }

        if (div > children)
            return children - 1;

        if (div < children)
        {
            if (rem == 3 && div == children - 1)
                return div - 1;
            else
                return div;
        }
        return -1;
    }
}
