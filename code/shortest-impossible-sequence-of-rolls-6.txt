class Solution {
public:
    int shortestSequence(vector<int>& rolls, int k) {
        set<int>st;
        int ans=1,q=0;
        for(int i=0;i<rolls.size();i++){
            if(st.find(rolls[i])==st.end())
            {
                st.insert(rolls[i]);
                if(st.size()==k)
                {
                    ans++;
                    st.clear();
                }
            }
        }
        return ans;
    }
};
