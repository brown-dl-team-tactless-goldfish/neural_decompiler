bool function(int *a){
    if(a[3]!=0||a[4]!=0||a[7]!=0){
        return false;
    }
    if(a[2]==0&&a[5]==0&&a[6]==0&&a[9]==0){
        return false;
    }
    return true;
}
int rotatedDigits(int N) {
    int ret=0;
    int n=0;
    for(int i=1;i<=N;i++){
        int array[10]={0};
        n=i;
         while(n>0){
            array[n%10]++;
            n/=10;
        }
        if(function(array)){
            ret++;
        }
    }
    return ret;
}
