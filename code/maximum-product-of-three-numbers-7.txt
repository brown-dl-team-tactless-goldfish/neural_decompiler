int cmp(const void *a, const void *b){
    return *(int *)a - *(int *)b;
}

int maximumProduct(int* nums, int numsSize){
    qsort(nums, numsSize, sizeof(int), cmp);
    int i=numsSize-1;
    int product_back=nums[i]*nums[i-1]*nums[i-2];
    int product_front=nums[0]*nums[1]*nums[numsSize-1];    
    
    return product_back>product_front?product_back:product_front;
}
