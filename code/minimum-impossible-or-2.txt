class Solution {
public:
    int minImpossibleOR(vector<int>& nums) {
        unordered_set<int>Set ;
        for(int i = 0; i < nums.size(); i++){
            Set.insert(nums[i]) ;
        }
        for(int i =0; i < 31; i++){
            int val = pow(2,i) ;
            if(Set.find(val) == Set.end())
                return val ;
        }
        return INT_MAX;
    }
};
