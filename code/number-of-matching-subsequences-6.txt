int numMatchingSubseq(char * s, char ** words, int wordsSize){
    short sRet = 0, sWordsSize = wordsSize - 1;

    bool abIsEmpty[26];
    memset(abIsEmpty, false, 26*sizeof(bool));

    while(*s)
    {
        // *** special case ***, skip checking if current char doesn't exist
        char cIndex = *s - 'a';
        if(abIsEmpty[cIndex])
        {
            s++;
            continue;
        }

        // start to check current words
        bool bIsModified = false;
        short sIndex = sWordsSize;
        while(sIndex >= 0)
        {
            // if current char of s is the same of words
            if(*s == *(words[sIndex]))
            {
                bIsModified = true;

                // end of words
                if(!(*(++words[sIndex])))
                {
                    sRet++;

                    // move from last
                    words[sIndex] = words[sWordsSize];
                    sWordsSize--;
                }
            }

            sIndex--;
        }

        // add flag if no current char
        if(bIsModified)
        {
            memset(abIsEmpty, false, 26*sizeof(bool));
        }
        else
        {
            abIsEmpty[cIndex] = true;
        }

        // move to next s
        s++;
    }

    return sRet;
}
