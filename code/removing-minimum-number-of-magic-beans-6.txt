long long minimumRemoval(vector<int>& beans) {
    sort(beans.begin(),beans.end());
    long long n=beans.size();
     long long sum=accumulate(beans.begin(),beans.end(),0l);
    long long ans=0;
    for(int i=0;i<beans.size();i++)
    {
      ans=max(ans,(1LL)*(n-i)*beans[i]);
    }
    return sum-ans;
    
    
}
