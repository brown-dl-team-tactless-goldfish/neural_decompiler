class Solution {
public:
    long long findTheArrayConcVal(vector<int>& nums) {
        long long n=nums.size(),ans=0;
        for(int i=0;i<n/2;i++)
        {
            string s1=to_string(nums[i]);
            string s2=to_string(nums[n-i-1]);
            s1=s1+s2;
            cout<<s1<<endl;
            ans+=stoi(s1);
        }
        if(n%2!=0) 
        {
            ans+=nums[n/2];
        }
        return ans;
    }
};
