int waysToSplitArray(int* nums, int numsSize){
    long int prefix[100000];
    int i, ans=0;
    prefix[0] = nums[0];
    for(i=1; i<numsSize;i++) {
        prefix[i] = prefix[i-1] + nums[i];
    }

    for(i=0; i<numsSize-1;i++) {
        if(prefix[i]>= prefix[numsSize-1]-prefix[i]){
            ans++;
        }
    }
    return ans;
}
