double average(int* salary, int salarySize){
double average,sum;
int count,temp;
for(int i=0;i<salarySize;i++){
    for(int j=i+1;j<salarySize;j++){
        if(salary[i]>=salary[j]){
            temp=salary[i];
            salary[i]=salary[j];
            salary[j]=temp;
        }
    }


}
for(int i=0;i<salarySize;i++){
    sum+=salary[i];
}
sum-=(salary[0]+salary[salarySize-1]);
sum=sum/(salarySize-2);
return sum;
}
