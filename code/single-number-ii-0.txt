int singleNumber(int* nums, int numsSize){
    int ans = 0;
    int sum ;
    for(int i = 0; i < 32; i++){
        long long mask = (long long)1 << i;
        sum = 0;
        for(int j = 0;  j < numsSize; j++){
            if(mask & nums[j])
                sum++;
        }
        if (sum % 3) ans |= mask;
    }    
    return ans;
}
