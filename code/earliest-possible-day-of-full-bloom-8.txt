class Solution {
public:
    int earliestFullBloom(vector<int>& p, vector<int>& g) {
        vector<pair<int,int>> v;
        for(int i=0;i<p.size();i++){
            v.push_back({g[i],p[i]});
        }
        sort(v.begin(),v.end());
        int cnt=0,ma=0;
        for(int i=p.size()-1;i>=0;i--){
            int k1=v[i].first;
            int k2=v[i].second;
            ma=max(ma,k2+k1+cnt);
            cnt+=k2;
        }
        return ma;
    }
};
