string reformat(string s) {
	string sResult = "";
	vector<char> chars {};
	vector<int> digits {};
	for (auto c : s)
		(isdigit(c)) ? digits.push_back(c) : chars.push_back(c);
	int charsSize = chars.size(), digitsSize = digits.size();
	if (abs(charsSize - digitsSize) > 1)
		return sResult;
	else {
		for (int i = 0; i < s.size() / 2; i++) {
			sResult.push_back(digits[i]);
			sResult.push_back(chars[i]);
		}
		if (digitsSize > charsSize) {
			sResult.push_back(digits[digitsSize - 1]);
		}
		else if(digitsSize < charsSize){
			sResult = chars[charsSize - 1] + sResult;
		}
	}
	return sResult;
}
