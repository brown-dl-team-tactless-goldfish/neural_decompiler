    public bool FindSubarrays(int[] nums)
    {
        HashSet<int> sums = new HashSet<int>();
        for (int i = 0; i < nums.Length - 1; i++)
        {
            int newSum = nums[i] + nums[i + 1];
            if (sums.Contains(newSum))
                return true;
            sums.Add(newSum);
        }
        return false;
    }
