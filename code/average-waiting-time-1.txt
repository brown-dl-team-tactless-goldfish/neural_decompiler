double averageWaitingTime(int** customers, int customersSize, int* customersColSize){
    double waitingTimeSum = 0;
    int availableTime = -1;
    for(int i = 0; i < customersSize; i++){
        if(availableTime > customers[i][0]){
            availableTime += customers[i][1];
            waitingTimeSum += (availableTime - customers[i][0]);    
        }
        else{
            availableTime = customers[i][0]+customers[i][1];
            waitingTimeSum += customers[i][1];
        }        
    }
    return (waitingTimeSum/customersSize);
}
