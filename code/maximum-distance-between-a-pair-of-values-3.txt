class Solution {
public:
    int maxDistance(vector<int>& nums1, vector<int>& nums2) {
        int i = 0 , n = nums1.size();
        int j = 0 , m = nums2.size();
        int res = 0;
        
        while(i < n && j < m){
            if(nums1[i] <= nums2[j]){
                if(i <= j)
                    res = max(res , j - i);
                j++;
            }
            else if(nums1[i] > nums2[j]){
                i++;
            }
        }
        return res;
    }
};
