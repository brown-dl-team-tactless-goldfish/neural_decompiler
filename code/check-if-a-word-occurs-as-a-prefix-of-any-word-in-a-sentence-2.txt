int isPrefixOfWord(char * sentence, char * searchWord){
int ptr=0;
int counter=1;

while (ptr<strlen(sentence)) {
    while((ptr<strlen(sentence)) && !isalpha(sentence[ptr]))
        ptr++;
    if (ptr+strlen(searchWord)>strlen(sentence)) return -1;
    if (strncmp(sentence+ptr,searchWord,strlen(searchWord))==0)
        return counter;
    counter++;
    while((ptr<strlen(sentence)) && isalpha(sentence[ptr]))
        ptr++;
}
return -1;
}
