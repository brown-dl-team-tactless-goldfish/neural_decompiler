class Solution {
public:
    int waysToSplitArray(vector<int>& nums) {
        int ans = 0;
        long long rSum = 0, lSum = 0;
        for (auto &n : nums) rSum += n;
        for (int i = 0; i < nums.size() - 1; ++i){
            lSum += nums[i];
            rSum -= nums[i];
            ans += lSum >= rSum;
        }
        return ans;
    }
};
