class Solution {
public:
    int similarPairs(vector<string>& words) {
        map<vector<int>, int> cnt;
        int ans = 0;
        for(auto w : words) {
            vector<int> alp(26);
            for(char c : w) alp[c - 'a'] = 1;
            ans += cnt[alp]++;
        }
        return ans;
    }
};
