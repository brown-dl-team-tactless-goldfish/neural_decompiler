    double tax;
    if(income>a[0][0])
        tax=a[0][0]*0.01*a[0][1];
    else
    {
        tax=income*0.01*a[0][1];
        goto b;
    }
        
    for(int i=1;i<a.size();i++)
    {
        if(income>a[i][0])
            tax+= (a[i][0]-a[i-1][0])*a[i][1]*0.01;
        else 
        {
            tax+= (income-a[i-1][0])*a[i][1]*0.01;
            break;
        }
              
    
    }
    b:
    return tax;
