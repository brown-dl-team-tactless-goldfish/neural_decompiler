#define ll long long int
class Solution {
public:
    long long minimumReplacement(vector<int>& nums) {
        ll n=nums.size();
       // ll prev=nums[0];
        ll ans=0;
        for(ll i=n-2;i>=0;i--)
        {
            if(nums[i]<=nums[i+1])
            {
                continue;
            }
            
            ll zz=ceil((double)nums[i]/(1.0*nums[i+1]));
            ans+=(zz-1);
            nums[i]/=zz;
        }
        return ans;
    }
};
