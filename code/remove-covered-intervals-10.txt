int removeCoveredIntervals(vector<vector<int>>& is) {
    sort(begin(is), end(is));
    int skips = 0, l = -1, r = -1;
    for (auto &i : is) {
        if (i[1] > r) {
            skips += i[0] == l;
            l = i[0], r = i[1];
        }
        else ++skips;
    }            
    return is.size() - skips;
}
