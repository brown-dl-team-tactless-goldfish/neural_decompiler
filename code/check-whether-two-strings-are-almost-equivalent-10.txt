public class Solution {
    public bool CheckAlmostEquivalent(string w1, string w2) {
        int[] c = new int[26];
        for (int i = 0; i < w1.Length; ++i) {
            c[w1[i] - 'a']++;
        }
        
        for (int i = 0; i < w2.Length; ++i) {
            c[w2[i] - 'a']--;
        }
        
        for (int i = 0; i < c.Length; ++i) {
            if (Math.Abs(c[i]) > 3) return false;
        }
        
        return true;
    }
}
