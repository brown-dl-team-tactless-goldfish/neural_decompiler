    int countSpecialSubsequences(vector<int>& nums) {
        const int mod = 1e9+7;
        vector<long long> dp(3,0);
        for(auto x:nums){
            if (x == 2) dp[2] = (dp[1] + 2*dp[2]) % mod;
            else if (x == 1) dp[1] = (dp[0]+ 2*dp[1]) % mod;
            else dp[0] = (2*dp[0]+1) % mod;
        }
        return dp[2];
    }
