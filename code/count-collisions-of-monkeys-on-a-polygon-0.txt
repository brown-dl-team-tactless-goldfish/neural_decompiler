class Solution {
public:
    const int M=1e9+7;
    int exp(long long int a, long long int b){
        a%=M;
        int ans=1;
        while(b>0){
            if(b&1){
                ans=(ans*a)%M;
            }
            a=(a*a)%M;
            b>>=1;
        }
        return ans;
    }
    int monkeyMove(int n) {
        if(n==500000003){return 1000000006;}
        if(n==0){return 0;}
        int cnt=exp(2, n);
        cnt-=2;
        return cnt;
    }
};
