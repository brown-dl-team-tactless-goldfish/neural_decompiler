int findKthPositive(int* arr, int arrSize, int k){
    if (k < arr[0]) return k;
    int left = 0, right = arrSize - 1;
    int mid = -1;
    while (left <= right) {
        mid = (left + right) / 2;
        if (arr[mid] - (mid + 1) < k) {
            left = mid + 1;
        }
        else {
            right = mid - 1;
        }
    }
    return left + k;
}
