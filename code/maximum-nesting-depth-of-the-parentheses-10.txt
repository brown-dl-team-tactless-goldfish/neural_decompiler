int maxDepth(char * s){
    int maxDepth = 0, depthCount = 0, index;
    for(index = 0; index < strlen(s); index++)
    {
        depthCount = (*(s+index) == '(')      ? ++depthCount : depthCount;
        depthCount = (*(s+index) == ')')      ? --depthCount : depthCount;
        maxDepth   = (depthCount > maxDepth)  ? depthCount : maxDepth;
    }
    return maxDepth;
}
