class Solution {
public:
    string largestOddNumber(string num) {
        auto it = find_if(num.rbegin(), num.rend(),[](char c){
            return c == '1' || c == '3' || c == '5' || c == '7' || c == '9';
        });
        if(it == num.rend())
            return "";
        
        return string(num.begin(), it.base());
    }
};
