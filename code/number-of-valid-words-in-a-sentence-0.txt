bool check(char* s){
    int n = strlen(s) ;
    if(s[0] == '-' || s[n-1] == '-' )
        return false ;
    
    int hyphen_num = 0;
    for(int i = 0; i < n; i++){
        if( isdigit( s[i] ) )
            return false ;
        if(s[i] == '-'){
            hyphen_num++;
            if(hyphen_num > 1)
                return false ;
            if( ( isalpha(s[i-1]) && isalpha(s[i+1]) ) == false  )
                return false ;
        }
        if(i < n - 1){
            if(s[i] == '!' || s[i] =='.' || s[i] == ',')
                return false ;
        }    
    }
    return true ;
}
int countValidWords(char * sentence){
    int n = strlen(sentence);
    int p = 0 ;
    int ans = 0;
    int begin;
    while(p < n){
        begin  = -1 ;
        for(int i =  p; i < n; i++){
            if(sentence[i] != ' '){
                begin = i;
                break ;
            }
        }
        if(begin == -1)
            break ;
        
        p = n ;
        for(int i = begin; i < n; i++){
            if(sentence[i] == ' '){
                sentence[i] = '\0' ;
                p = i + 1 ;
                break ;
            }
        }
        ans += check( &sentence[begin] ) ;
    }
    return ans;
}
