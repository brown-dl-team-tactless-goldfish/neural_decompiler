double average(int* salary, int salarySize){
    double sum=0,c=0;
    int min=salary[0],i;
    int max=salary[0];
for(int i=0;i<salarySize;i++)
{
    if(max<=salary[i]){
        max=salary[i];
    }
    if(min>=salary[i])
    {
        min=salary[i];
    }
}
for(i=0;i<salarySize;i++)
{
    if(salary[i]==min||salary[i]==max)
    {
        continue;
    }
    else{
        sum+=salary[i];
        c++;
    }

}
return (sum/c);
}
