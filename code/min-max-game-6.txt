int minMaxGame(vector<int>& nums) {
	while (nums.size() >= 2) {
		vector<int> nnums(nums.size() / 2);
		int i = 0, j = 0;
		while (i < nums.size() - 1) {
			nnums[j++] = min(nums[i], nums[i + 1]);
			i += 2;
			if (i < nums.size() - 1)
				nnums[j++] = max(nums[i], nums[i + 1]);
			i += 2;
		}
		nums = nnums;
	}
	return nums[0];
}
