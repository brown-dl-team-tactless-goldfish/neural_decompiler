class Solution:
    def isThree(self, n: int) -> bool:
        data = []
        
        for i in range(1, n // 2 + 1):
            if (n % i == 0):
                data.append(i)
                
            if (len(data) > 3):
                return False
            
        print ("data = ", data)
        
        return len(data) == 2
