class Solution {
public:
	int checkRecord(int n) {
		int MOD=1e9+7;
		long XAXL=1; //No A no L in the end
		long XAL=0;  // No A with one L in the end 
		long  XALL=0;  // No A with two L in the end 
		long  AXL=0;  // One A no L in the end
		long  AL=0;  // One A with one L in the end
		long  ALL=0;  // One A with two L in the end

		for(int i=1;i<=n;i++){
			long  N_XAXL=XAXL;
			long  N_XAL=XAL;
			long  N_XALL=XALL;
			long  N_AXL=AXL;
			long  N_AL=AL;
			long  N_ALL=ALL;

			ALL=N_AL;
			AL=N_AXL;
			XAL=N_XAXL;
			XALL=N_XAL;
			XAXL=(N_XAL+N_XALL+N_XAXL)%MOD;
			AXL=(N_AXL+N_AL+N_ALL+N_XAL+N_XALL+N_XAXL)%MOD;
		}

		return (XAXL+XAL+XALL+AXL+AL+ALL)%MOD;

	}
};
