class SmallestInfiniteSet {
    priority_queue <int,vector <int>, greater <int>> pq;
    vector <int> v;
public:
    SmallestInfiniteSet():v(1001,1){
        for(int i=1; i<=1000;i++)
            pq.push(i);
    }
    
    int popSmallest() {
        int popN = pq.top();
        pq.pop();
        v[popN] = 0;
        return popN;
    }
    
    void addBack(int num) {
        if(v[num] == 0){
            v[num] = 1;
            pq.push(num);
        }
    }
};
