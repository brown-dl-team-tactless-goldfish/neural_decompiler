int minHeightShelves(vector<vector<int>>& books, int shelf_width) {
	int n = books.size();
	books.insert(books.begin(), {0, 0});
	vector<int> dp(n + 1, INT_MAX);
	dp[0] = 0;
	for(int i = 1; i <= n; ++i)
		for(int j=i-1, w = books[i][0], h=books[i][1]; j>=0 && w<=shelf_width; h=max(h,books[j][1]), w+=books[j--][0])
			dp[i] = min(dp[i], dp[j] + h);
	return dp.back();
}
