int reverse(int n){
    int ans = 0;
    while(n){
        ans *= 10;
        ans += (n % 10);
        n /= 10;
    }
    return ans;
}

int countDistinctIntegers(int* nums, int numsSize){
    int* hashCnt = (int*)calloc(1000001, sizeof(int));
    int ans = 0;
    for(int i = 0; i < numsSize; i++){
        int rev = reverse(nums[i]);
        hashCnt[nums[i]]++;
        if(hashCnt[nums[i]]==1) ans++;
        hashCnt[rev]++;
        if(hashCnt[rev]==1) ans++;        
    }
    return ans;
}
