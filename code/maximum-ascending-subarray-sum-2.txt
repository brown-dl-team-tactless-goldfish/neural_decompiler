int maxAscendingSum(int* nums, int ns){

        int m = nums[0];
        int curr_max = nums[0];
        for(int i = 1; i < ns; i++)
        {
            if(nums[i] > nums[i-1])
            {
                curr_max += nums[i];
                m = fmax(m, curr_max);
            }
            else
                curr_max = nums[i];     
        }
        return m;
}
