class Solution {
public:
    bool checkStraightLine(vector<vector<int>>& A) {
        auto [p, q] = tie(A[0][0], A[0][1]);
        auto [u, v] = tie(A[1][0], A[1][1]);
        auto isLine = [&](auto x, auto y) -> bool {
            return (y - q) * (x - u) == (y - v) * (x - p);
        };
        for (auto c: A)
            if (!isLine(c[0], c[1]))
                return false;            
        return true;
    }
};
