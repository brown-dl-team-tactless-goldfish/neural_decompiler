int numberOfArithmeticSlices(int* nums, int numsSize) {
    if(numsSize < 3) return 0;
    int cnt = 0, ans = 0;
    for(int i = 1; i < numsSize-1; i++) {
        if(nums[i]-nums[i-1] == nums[i+1] - nums[i]) {
            cnt++;
        } else {
            if(cnt != 0) ans += cnt * (1 + cnt) / 2;
            cnt = 0;
        }
    }
    if(cnt != 0) ans += cnt * (1 + cnt) / 2;
    return ans;
}
