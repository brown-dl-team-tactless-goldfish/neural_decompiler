bool canConstruct(char * s, int k){
    if(strlen(s) < k)
        return false;
    int odd_char = 0;
    int *odd = &odd_char;
    int letters[26] = {};
    int *p = letters;
    while(*s != '\0')
        (*(letters + *s++ - 'a'))++;
    while(p != letters + 26)
        if(*p++ & 1)
            (*odd)++;
    if(*odd > k)
        return false;
    return true;
}
