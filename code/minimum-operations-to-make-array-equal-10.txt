class Solution {
public:
    int minOperations(int n) {
        int k=0;
        int x=2*n-1;
        int y=1;
        int v=0;
        if(n%2==0){
            int l=n/2;
            int k=l-1;
            v=((2*l)+1+(2*k)+1)/2;
        }
        else{
            int h=n/2;
            v=(2*h)+1;
        }
        int d=x-v,ans=0;
        while(d>0){
            ans+=d;
            d-=2;
        }
        return ans;
            
    }
};
