public class Solution {
    public int PartitionString(string s) {
        int result = 0;
        
        HashSet<char> unique = new();
        
        foreach (char c in s) 
            if (!unique.Add(c)) {
                result += 1;
                
                unique.Clear();
                unique.Add(c);
            }
                
        return result + Math.Sign(unique.Count);
    }
}
