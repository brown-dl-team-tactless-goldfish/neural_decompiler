int cmp(const void *x, const void *y)
{
    return (*(int*)y-*(int*)x);
}
int minSetSize(int* a, int n)
{
    qsort(a, n, sizeof(int), cmp);
    int res=0;
    int map[100001]={0};
    for(int i=0;i<n;i++)
        map[a[i]]++;
    qsort(map, a[0]+1, sizeof(int), cmp);
    int i;
    for( i=0;i<a[0]+1;i++)
    {
        res+=map[i];
        if(res>=n/2)break;
    }
    return i+1;
}
