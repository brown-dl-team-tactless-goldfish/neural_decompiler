bool makeEqual(char ** words, int wordsSize){
    int i,j;
    int arr[26]={0};
    for(i=0;i<wordsSize;i++){
        for(j=0;j<strlen(words[i]);j++){
            arr[words[i][j]-97]+=1;
        }
    }
    
    for(i=0;i<26;i++){
        if(arr[i]!=0 && arr[i]%wordsSize!=0) return false;
    }
    
    return true;
}
