    bool canChoose(vector<vector<int>>& groups, vector<int>& nums) {
        int i=0,j=0,k=0;
        int m=groups.size();
        while(k<nums.size() && i<m){
            if(groups[i].empty()) {i++;continue;}
            if(groups[i][j]!=nums[k]) {
                if(j) j=0; //reset
                if(groups[i][j]==nums[k]) j++;
            }
            else {
                j++;
            }
            if(j>=groups[i].size()) i++,j=0;
            k++;
        }
        return i>=m;
    }
