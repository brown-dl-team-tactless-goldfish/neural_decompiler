using LL = long long ;
class Solution {
    bool check(LL time, int n, const vector<int>& batteries){
        LL total = time * n ;
        LL sum = 0 ;
        for(int i = 0; i < batteries.size(); i++){
            sum += fmin(time, batteries[i]) ;
            if(sum >= total)
                return true ;
        }
        return false ;
    }
public:
    long long maxRunTime(int n, vector<int>& batteries) {
        LL left = 0, right = 1e14 ;
        while(left < right){
            LL mid = right - (right - left) / 2 ;
            if(check(mid, n, batteries))
                left = mid ;
            else
                right = mid - 1 ;
        }
        return left ;
    }
};
