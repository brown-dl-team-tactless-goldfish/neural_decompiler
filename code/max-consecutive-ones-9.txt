int findMaxConsecutiveOnes(int* nums, int numsSize){
    int max = 0,temp = 0;
    for(int i = 0 ; i < numsSize; i++){
        if(nums[i] == 1){
            temp++;
        }
        else if(nums[i] == 0){
            if(temp > max)max = temp;
            temp = 0;
        }
    }
    if(temp > max)max = temp;
    return max;
}
