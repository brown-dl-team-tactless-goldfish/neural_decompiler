int maximum69Number (int num){
    int arr[4],i;
    int final=0,count=0;
    
    for (i=3; i>=0; i--) {
        arr[i]=num%10;
        num=num/10;
    }
    
    for (i=0; i<4; i++) {
        if (count == 0 && arr[i] == 6) {
            arr[i]=9;
            count++;
        }
    }
    final = (arr[0] * 1000) + (arr[1] * 100) + (arr[2] * 10) + (arr[3]);
    return final;
}
