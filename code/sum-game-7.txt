class Solution {
public:
    bool sumGame(string num) {
        int diff = 0, qm = 0; 
        for (int i = 0; i < num.size(); ++i) {
            if (num[i] == '?') qm += i < num.size()/2 ? 1 : -1; 
            else diff += i < num.size()/2 ? num[i] - '0' : '0' - num[i]; 
        }
        return diff * 2 + qm * 9 != 0; 
    }
};
