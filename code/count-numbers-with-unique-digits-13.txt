public class Solution 
{
    public int CountNumbersWithUniqueDigits(int n) 
    {
        if (n == 0) return 1;
        if (n == 1) return 10;

        var sum = 10;

        for (int i = 1; i < n; i++)
        {
            var total = 9;
            for (int j = 1; j <= i; j++)
            {
                total = total * (10 - j);
            }
            sum += total;
        }

        return sum;
    }
}
