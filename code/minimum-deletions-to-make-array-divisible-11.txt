class Solution {
public:
    int minOperations(vector<int>& nums, vector<int>& d) {
        int n=nums.size();
        int x=d[0];
        int n1=d.size();
        for(int i=1;i<n1;i++)
        {
            x=__gcd(x,d[i]);
        }
        
        sort(nums.begin(),nums.end());
        
        for(int i=0;i<n;i++)
        {
            if(x%nums[i]==0) return i;
        }
        
        return -1;
    }
};
