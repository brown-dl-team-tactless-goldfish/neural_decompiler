    int minDeletionSize(vector<string>& A) {
        vector<int> dp(A[0].size(), 1);       
        for(int i=1;i<A[0].size();i++)
            for(int j=0;j<i;j++)
                if(all_of(A.begin(), A.end(), [=](auto& v){return v[j] <= v[i];})) dp[i] = max(dp[i], 1+ dp[j]);
        return A[0].size() - *max_element(dp.begin(), dp.end());
    }
