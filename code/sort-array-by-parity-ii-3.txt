int* sortArrayByParityII(int* A, int ASize, int* returnSize) {
    int* result = (int*)malloc(sizeof(int)*ASize);
    *returnSize = ASize;
    int oddbit = 0;
    int evenbit = 0;
    int i;
    for(i = 0;i<ASize;i++){
        if(A[i]%2==0)result[(evenbit++)*2]=A[i];
        else result[(oddbit++)*2+1]=A[i];
    }
    return result;
}
