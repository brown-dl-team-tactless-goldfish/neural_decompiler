#include <inttypes.h>
int oddCells(int n, int m, int** indices, int indicesSize, int* indicesColSize){
    unsigned long long row_mask=0,col_mask=0;
    for(int i=0;i<indicesSize;++i)
    {
        row_mask ^= 1ULL<<(indices[i][0]);
        col_mask ^= 1ULL<<(indices[i][1]);
    }
    int rc = __builtin_popcountll(row_mask);
    int cc = __builtin_popcountll(col_mask);
    return n*cc+m*rc-2*rc*cc;
}
