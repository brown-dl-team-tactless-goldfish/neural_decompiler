char * licenseKeyFormatting(char * s, int k){
    int n = strlen(s);
    int N = 2*n + 1;
    char* ans = malloc(N  * sizeof(char) );
    ans[N -1] = '\0';
    int a_idx = N-2;
    int cn = 0;
    for(int i = n-1; i >=0; i--){
        if(s[i] =='-')
            continue;
        cn++;
        if(s[i] >='a' && s[i] <='z')
            ans[a_idx] = s[i] -'a' +'A' ;
        else
            ans[a_idx] = s[i];
        a_idx--;
        if(cn % k == 0){
            ans[a_idx] = '-' ;
            a_idx--;
        }
    }
    if(ans[a_idx+1] == '-')
        a_idx++;
    return &ans[++a_idx ];
}
