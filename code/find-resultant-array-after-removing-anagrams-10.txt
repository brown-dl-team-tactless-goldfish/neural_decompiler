class Solution {
    bool isAnagram(string a, string b) {
        if (a.length() != b.length())
            return false;
        vector<int> alphabets(26, 0);
        for (char c : a)
            alphabets[c-'a']++;
        for (char c : b)
            alphabets[c-'a']--;
        for(int num : alphabets)
            if (num != 0)
                return false;
        return true;
    }
    
    void removeAnagrams(vector<string>& res, vector<string>& words, int d, int l) {
        if (d == words.size())
            res.push_back(words[l]);
        else if (isAnagram(words[d], words[l]))
            removeAnagrams(res, words, d+1, l);
        else {
            res.push_back(words[l]);
            removeAnagrams(res, words, d+1, d);
        }
    }
    
public:
    vector<string> removeAnagrams(vector<string>& words) {
        vector<string> res;
        removeAnagrams(res, words, 1, 0);
        return res;
    }
};
