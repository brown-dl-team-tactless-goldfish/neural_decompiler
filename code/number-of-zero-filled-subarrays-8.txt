long long zeroFilledSubarray(int* nums, int numsSize){
    long long ans = 0;
    int curCount = 0;
    for(int i = 0; i < numsSize; i++)
    {
        if(nums[i] == 0)
        {
            curCount++;
            ans += curCount;
        }
        else
            curCount = 0;
    }
    return ans;
}
