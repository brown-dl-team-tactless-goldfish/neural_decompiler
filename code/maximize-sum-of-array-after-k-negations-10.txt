class Solution:
    def largestSumAfterKNegations(self, A: List[int], K: int) -> int:
        A.sort()
        # print(" A = ", A)
        # print("K = ", K)

        if (K == 0):
            return sum(A)
        elif (K == 1):

            A[0] = 0-A[0]
            return sum(A)
        elif (K > 1):

            A[0] = 0-A[0]
            return self.largestSumAfterKNegations(A, K-1)
