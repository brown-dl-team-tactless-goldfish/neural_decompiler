class Solution {
public:
    vector<int> decode(vector<int>& encoded) {
        int n=encoded.size()+1;
        int x=0;
        for(int i=1;i<=n;i++)  x=x^i;
        for(int i=0;i<n-2;i=i+2)  x=x^encoded[i]; 
       
       
        vector<int>  ans(n);
        ans.back()=x;
        for(int i=n-2;i>=0;i--){
            ans[i]=ans[i+1]^encoded[i];
        }
        
        return ans;
        
    }
};
