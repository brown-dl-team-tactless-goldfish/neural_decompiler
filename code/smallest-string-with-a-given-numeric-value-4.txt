

char * getSmallestString(int n, int k){
    k = k - n;
    int countOfZ = k / 25;
    int value = k % 25;
    int i = n-1;
    char* ans = (char* )malloc(sizeof(char)*(n+1));
    ans[n]='\0';
    while(countOfZ--> 0 && i >= 0)
        ans[i--] = 'z';
     
    if(value > 0 && i >=0)
        ans[i--] = (char ) ('a' + value);
    while(i>=0)
        ans[i--]= 'a';
    return ans;
}
