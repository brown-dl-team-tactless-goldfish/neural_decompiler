  vector<string> getFolderNames(vector<string>& names) {
      unordered_map<string, int> mem;
        for(int i=0; i<names.size(); i++){
            string tmp(names[i]);
            while(mem.find(tmp)!=mem.end()){
                tmp = (names[i] + "("+to_string(mem[names[i]])+")");
                mem[names[i]]++;
            }
            mem[tmp]++;
            names[i]=tmp;
        }
        return names;
    }
