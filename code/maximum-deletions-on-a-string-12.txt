class Solution {
public:
    int deleteString(string s) {
        int n=s.size();
        const char* cStr=s.c_str();
        vector<int> dp(n, 1);
        for(int i=n-1; i>=0; i--) {
            int len=n-i;
            for(int j=len/2; j>0; j--) {
                //if(s.substr(i,j) == s.substr(i+j, j))
                if(0==memcmp(cStr+i, cStr+i+j, j))
                    dp[i]=max(dp[i], 1+dp[i+j]);
            }
        }
        return dp[0];
    }
};
