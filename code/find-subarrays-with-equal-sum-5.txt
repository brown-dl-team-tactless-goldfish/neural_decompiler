class Solution {
public:
    bool findSubarrays(vector<int>& nums) {
        int n = nums.size();
        if(n == 2)
            return false;
        for(int i = 1; i < n; i++)
        {
            long long int sum = nums[i] + nums[i-1];
            
            for(int j = i; j < n-1; j++)
            {
                long long int tsum = nums[j] + nums[j+1];
                if(sum == tsum)
                    return true;
            }
        }
        return false;
    }
};
