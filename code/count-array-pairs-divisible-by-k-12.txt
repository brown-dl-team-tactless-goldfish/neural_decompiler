class Solution {
public:
    long long countPairs(vector<int>& nums, int k) {
        unordered_map<int,long long> mp;
        long long ans=0;
        for(auto n:nums){
            int gcd=__gcd(n,k);
            int req=k/gcd;
            for(auto it:mp)
                if(it.first%req==0)
                    ans+=it.second;
            mp[gcd]++;
        }
        return ans;
    }
};
