void duplicateZeros(int* arr, int arrSize){
    int *tmp=malloc(sizeof(int)*(arrSize+1)),idx=0;

    for(int i=0;i<arrSize&&idx<arrSize;i++)
    {
        if(arr[i]!=0)
            tmp[idx++]=arr[i];
        else if(arr[i]==0&&((idx)<arrSize))
        {
            tmp[idx++]=0;
            tmp[idx++]=0;
        }
    }

    memcpy(arr,tmp,sizeof(int)*arrSize);
}
