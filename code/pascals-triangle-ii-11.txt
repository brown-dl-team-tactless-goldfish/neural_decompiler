
void ans(int *ret,int rowIndex,int i)
{
        if(i>rowIndex) return;
        int a,b;
        ret[0]=1;
        ret[i]=1;
        for(int j=1,a=ret[0];j<i;j++)
        {
            b=ret[j];
            ret[j]=a+ret[j];     
            a=b;

        }
        ans(ret,rowIndex,++i);

}
int* getRow(int rowIndex, int* returnSize){
    *returnSize=rowIndex+1;
    int *ret=calloc((*returnSize),sizeof(int));
    ans(ret,rowIndex,0);
    return ret;
    
}
