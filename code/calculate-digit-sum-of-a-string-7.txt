class Solution {
public:
    string digitSum(string s, int k) {
        string ans = s;
        while (ans.length() > k)
        {
            string t = "";
            for (int i = 0; i < ans.length(); i += k)
            {
                int c = 0;
                for (int j = i; j < i + k && j < ans.length(); j++)
                    c += (ans[j] - '0');
                t += to_string(c);
            }
            ans = t;
        }
        return ans;
    }
};
