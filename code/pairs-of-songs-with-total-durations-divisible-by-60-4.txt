int numPairsDivisibleBy60(int* time, int timeSize){
    int count = 0;
    int arr[60] = {0};
    for(int i=0; i<timeSize; i++){
        arr[time[i]%60]++;
    }
    count += (arr[0]*(arr[0]-1))/2;  // choose 2 from arr[0], (n choose 2) = (n*(n-1))/2
    count += (arr[30]*(arr[30]-1))/2;  // choose 2 from arr[30]
    for(int i=1; i<30; i++){
        count += arr[i] * arr[60-i];  // choose 1 from arr[i] and another 1 from arr[60-i]
    }
    return count;
}
