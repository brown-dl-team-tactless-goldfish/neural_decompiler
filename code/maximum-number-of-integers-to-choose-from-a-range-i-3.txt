public class Solution 
{
    public int MaxCount(int[] banned, int n, int maxSum) 
    {
        HashSet<int> bannedHs = new HashSet<int>(banned);
        int counter = 0;
        long currSum=0;
        for (int i = 1; i <= n; i++)
        {
            if (!bannedHs.Contains(i))
            {
                currSum += i;
                if (currSum > maxSum)
                    break;
                counter++;
            }

        }
        return counter;
    }
}
