class Solution {
public:
    int getCommon(vector<int>& nums1, vector<int>& nums2) {
        // int * a = new int[100000];
        int c = -1;
        int i=0;
        int j=0;
        while(i<nums1.size() && j<nums2.size()){
            if(nums1[i] == nums2[j]){
                c = nums1[i];
                break;
            }
            else if(nums2[j]>nums1[i]){
                i++;
            }
            else{
                j++;
            }
        }
        return c;
    }
};
