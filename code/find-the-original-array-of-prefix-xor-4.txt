class Solution {
public:
    vector<int> findArray(vector<int>& a) 
    {
        vector<int> res;
        if(a.size()==0)return res;
        if(a.size()==1)return a;
        res.push_back(a[0]);
        for(int i=1; i<a.size(); i++)
            res.push_back(a[i]^a[i-1]);
        return res;
    }
};
