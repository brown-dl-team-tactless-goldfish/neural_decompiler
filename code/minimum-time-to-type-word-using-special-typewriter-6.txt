public int MinTimeToType(string word) {
	var seconds = 0;
	var pointer = 'a';
	var count = 'z' - 'a' + 1;

	foreach (var c in word) {
		if (pointer != c) {
			var diff = ((c - pointer + count) % count);
			if (diff > count / 2) {
				diff = count - diff;
			}
			seconds += diff;
			pointer = c;
		}
		seconds++;
	}

	return seconds;
}
