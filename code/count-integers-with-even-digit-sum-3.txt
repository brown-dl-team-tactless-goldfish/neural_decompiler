 int countEven(int num)
 {
     int count =0 ;
     int sum = 0;
     int denum ;
     for ( int i = num ; i > 0 ; i-- )
     {
         denum = i;
         
           while ( denum > 0)
         {
             sum = sum + ( denum % 10);
             denum = denum /10;
         }
         if ( (sum % 2) == 0)
             
         {
             count++;
         }
         sum  = 0;
     }
     return count;
 }
