class Solution {
public:
    vector<int>bob ;
    int maxi = 0 ;
    void solve(int pos,int numArrows , vector<int>&aliceArrows ,int sum ,vector<int>&temp){
        if(numArrows < 0) return ;
        if(pos >= aliceArrows.size()){
            if(sum > maxi){
                maxi = sum ;
                bob = temp ; 
                bob[0] += numArrows ; //remaining Arrows can be thrown in 0th index as it won't affect our answer
                
            }
            return ;
        }
        //Let Bob wins 
        temp.push_back(aliceArrows[pos] + 1) ;
        solve(pos+1,numArrows-temp.back(),aliceArrows,sum+pos,temp) ;
        temp.pop_back();
        
        //Let Alice wins 
        temp.push_back(0);
        solve(pos+1,numArrows,aliceArrows,sum,temp) ;
        temp.pop_back() ;
        
    }
    vector<int> maximumBobPoints(int numArrows, vector<int>& aliceArrows) {
        vector<int>temp ; 
        solve(0,numArrows,aliceArrows,0,temp) ;
        return bob ;
    }
};
