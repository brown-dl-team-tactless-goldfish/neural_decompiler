char * kthDistinct(char ** arr, int arrSize, int k){
    int i, j, l = arrSize, flag;
    
    for(i = 0; i < l; i++){
        for(j = 0, flag = 0; j < l; j++){
            if(j != i){
                if(!strcmp(arr[i], arr[j])){
                    flag = 1;
                    break;
                }
            }
        }
        if(!flag)
            k--;
        
        if(!k)
            return arr[i];
    }
    
    return "";
}
