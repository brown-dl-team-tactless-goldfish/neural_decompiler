int* getRow(int rowIndex, int* returnSize){
    int * result = malloc(sizeof(int) * (rowIndex + 1));
    int temp1, temp2;
    int i, j, k;
    result[0] = 1;
    for (i = 1; i <= rowIndex; i++)
        result[i] = 0;
    for (k = 1; k <= rowIndex; k++){
        temp2 = 1;
        for (j = 1; j <= k; j++){
            temp1 = result[j];
            result[j] = temp2 + result[j];
            temp2 = temp1;
        }
    }
    *returnSize = rowIndex + 1;
    return result;
}
