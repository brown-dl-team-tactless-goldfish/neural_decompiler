int cmp(const void *a, const void *b)
{
    return *(int *)a - *(int *)b;
}

int maximumBags(int* capacity, int capacitySize, int* rocks, int rocksSize, int additionalRocks){
    int * remain = (int *)malloc(capacitySize * sizeof(int));
    int count = 0, idx = 0;
    for(int i = 0; i < capacitySize; i++)
        remain[i] = capacity[i] - rocks[i];
    qsort(remain, capacitySize, sizeof(int), cmp);
    while(additionalRocks > 0 && idx < capacitySize)
    {
        additionalRocks -= remain[idx];
        if(additionalRocks >= 0) 
            count++;
        idx++;
    }
    return count;
}
