class Solution {
public:
    string addSpaces(string s, vector<int>& spaces) {
        int i=0,j=0;
        string ans="";
        while(i<s.length()) {
            while(j<spaces.size() && i<spaces[j]) {
                ans.push_back(s[i++]);
            }
            if(i<s.length()){
                if(j<spaces.size()) {
                    ans+=" ";
                }
                ans.push_back(s[i]);
            }
            i++;
            j++;
        }
        return ans;
    }
};
