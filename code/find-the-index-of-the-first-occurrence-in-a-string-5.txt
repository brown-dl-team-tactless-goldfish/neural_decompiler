int strStr(char * haystack, char * needle){
    int haystack_len = strlen(haystack);
    int needle_len = strlen(needle);
    int i;
    
    if(haystack_len == 0 && needle_len == 0) return 0;
    
    for(i = 0; i<haystack_len; i++){
        int j = 0, k = i;
        while(k<haystack_len && j<needle_len && haystack[k] == needle[j]){
            k++;
            j++;
        }
        if(j == needle_len) return k-j;
    }
    
    return -1;
}
