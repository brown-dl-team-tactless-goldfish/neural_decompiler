/**
 * Definition for a Node.
 * struct Node {
 *     int val;
 *     int numChildren;
 *     struct Node** children;
 * };
 */

int maxDepth(struct Node* root) {
    if(!root)
        return 0;
    int i,temp,max=0;
    for(i=0;i<root->numChildren;i++){
        temp=maxDepth(root->children[i]);
        if(temp>max)
            max=temp;
    }
    return max+1;
}
