struct ListNode* removeNthFromEnd(struct ListNode* head, int n)
{
    
struct ListNode*l1=head;
    int count=0;
    while(l1)
    {
        count++;
        l1=l1->next;
    }
  
    if(count==1)
    {
        head=NULL;
        return head;
    }
      if(count==n)
    {
        l1=head->next;
        return l1;
    }
    count=count-n-1;
  
    l1=head;
    while(count)
    {
        count--;
        l1=l1->next;
        
    }
    l1->next=l1->next->next;
    return head;
}
