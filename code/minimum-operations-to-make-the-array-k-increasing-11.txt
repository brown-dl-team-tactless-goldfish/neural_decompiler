int kIncreasing(vector<int>& arr, int k) {
    
    
    int n=arr.size();
     vector<int> temp;
    int ans=0;
    for(int i=0;i<k;i++)
    {
      //  vector<int> temp;
        for(int j=i;j<n;j+=k)
        {
            temp.push_back(arr[j]);
        }
        int sz=temp.size();
        ans+=(sz-LIS(temp));
        temp.clear();
        
    }
    return ans;
    
}
