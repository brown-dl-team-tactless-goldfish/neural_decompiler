class Solution {
public:
    int numberOfRounds(string startTime, string finishTime) {
        int s = getmin(startTime);
        int e = getmin(finishTime);
        if (e < s) e += 1440;
        
        return e/15 - int(ceil(float(s)/15));
    }
    
    int getmin(string time){
        return int(time[0]-'0')*600+int(time[1]-'0')*60+int(time[3]-'0')*10+int(time[4]-'0');
    }
};
