

int countStudents(int* students, int studentsSize, int* sandwiches, int sandwichesSize)
{
    int no_of_students_prefer_square = 0;
    int no_of_students_prefer_circular = 0;
    int i=0;
    
    for(i=0; i<studentsSize; i++)
    {
        if(students[i] == 0)
        {
            no_of_students_prefer_circular++;
        }
        else
        {
            no_of_students_prefer_square++;
        }
    }
       
    for(i=0; i<sandwichesSize; i++)
    {
        if(sandwiches[i] == 0)
        {
            if(no_of_students_prefer_circular > 0)
            {
                no_of_students_prefer_circular--;
            }
            else
            {
                break;
            }
        }
        else
        {
            if(no_of_students_prefer_square > 0)
            {
                no_of_students_prefer_square--;
            }
            else
            {
                break;
            }
        }
    }
    return (no_of_students_prefer_circular + no_of_students_prefer_square);
}
