class Solution {
public:
    int numberOfArrays(vector<int>& differences, int lower, int upper) {
        vector<long long> v(differences.size()+1);
        v[0]=0;
        for (int i=1;i<v.size();i++) v[i]=differences[i-1]+v[i-1];
        int ans=(upper-lower)+(*min_element(v.begin(),v.end())-*max_element(v.begin(),v.end()))+1;
        if (ans<0) return 0;
        return ans;
    }
};
