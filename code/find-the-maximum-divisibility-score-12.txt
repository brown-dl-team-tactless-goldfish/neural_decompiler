class Solution {
public:
    int maxDivScore(vector<int>& nums, vector<int>& divisors) {
        vector<int> candidates;
        int max_score = 0;
        for (int divisor : divisors) {
            int score = 0;
            for (int num : nums) {
                if (num % divisor == 0) {
                    score++;
                }
            }
            if (score > max_score) {
                max_score = score;
                candidates.clear();
                candidates.push_back(divisor);
            }
           else if (score == max_score) {
                candidates.push_back(divisor);
            }
        }
        return *min_element(candidates.begin(), candidates.end());
    }
};
