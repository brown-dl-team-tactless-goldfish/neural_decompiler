class Solution {
public:
    char slowestKey(vector<int>& releaseTimes, string keysPressed) {
        int maxd = 0;
        char c;
        
        for(int i = 0; i < releaseTimes.size(); i++)
        {
            //Check the duration
            int d = (i == 0? releaseTimes[i]-0 : releaseTimes[i] - releaseTimes[i-1]);
            
            //duration is greater? for same duration char is larger?
            if(d > maxd || ( d == maxd && keysPressed[i] > c))
            {
                maxd = d;
                c = keysPressed[i];
            }            
        }
        
        return c;
    }
};
