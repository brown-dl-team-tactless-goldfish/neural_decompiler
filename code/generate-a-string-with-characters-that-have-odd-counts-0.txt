char * generateTheString(int n){
    char* ans = malloc( ( n + 1 )* sizeof(char) ) ;
    ans[n] = '\0' ;
    if(n % 2){
        for(int i = 0; i < n ;i++){
            ans[i] = 'a' ;
        }
    }
    else{
        for(int i = 0; i < n-1 ;i++){
            ans[i] = 'a' ;
        }
        ans[n-1] = 'b' ;
    }
    return ans ;
}
