int wateringPlants(int* plants, int plantsSize, int capacity){
    int count = 0, capacityTemp = capacity;
    for(int i = 0; i < plantsSize; i++)
    {
       count++;
       capacity = capacity - plants[i];
       if(i == plantsSize - 1) // to break, or else the if below will execute (and there is no plants[i +1] after the last element
       {
           break;
       }
       if(capacity - plants[i + 1] < 0)
       {
           count = count + (i + 1)*2;
           capacity = capacityTemp;
       }
    }
    return count;
}
