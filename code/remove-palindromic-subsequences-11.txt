class Solution {
    bool isPalindrome(string s)
    {
        int l = 0, r = s.size()-1;
        while(l<r)
        {
            if(s[l++] != s[r--])
                return false;
        }
        return true;
    }
public:
    int removePalindromeSub(string s) {
        if(s.size() == 0)
            return 0;
        if(isPalindrome(s))
            return 1;
        //subsequence means taking any char by deleting chars in middle 2.
        //There are max 2 operation, once selecting all a, which will be defnitely palindrome //aaaa
        // delete remaining b which are palindrome //bbbb
        return 2;
    }
};
