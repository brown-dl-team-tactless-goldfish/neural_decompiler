
class Solution
{
public:
    int numSubarraysWithSum(vector<int> &nums, int goal)
    {
        unordered_map<int, int> m;
        int sum = 0;
        int c = 0;
        for (int i = 0; i < nums.size(); i++)
        {
            sum += nums[i];
            if (sum == goal)
            {
                c++;
            }

            if (m.find(sum - goal) != m.end())
            {
                c += m[sum - goal];
            }
            m[sum]++;
        }
        return c;
    }
};
