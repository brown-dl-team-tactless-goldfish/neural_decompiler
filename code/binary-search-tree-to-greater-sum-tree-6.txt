void modify(TreeNode *root,pair<int,int> p[])
{
    if(root==NULL)
        return;
    int x = root->val;
    for(int i=0;i<k;i++)
    {
        if(p[i].first==x)
            root->val = p[i].second;
    }
    modify(root->left,p);
    modify(root->right,p);
}
TreeNode* bstToGst(TreeNode* root) {
    
    pair<int,int> p[100];
    fillvector(p,root);
    sort(p,p+k);
    for(int i=k-2;i>=0;i--)
    {
        p[i].second = p[i+1].second+p[i].second;
    }
    modify(root,p);
    return root;
}
