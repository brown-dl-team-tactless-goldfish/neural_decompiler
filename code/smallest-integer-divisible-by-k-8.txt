public class Solution {
    public int SmallestRepunitDivByK(int k) {
        
        int remainder = 0, len = 0;
        HashSet<int> set = new HashSet<int>();
        
        while(true)
        {
            remainder = (remainder * 10 + 1) % k;
            len++;
            
            if(remainder == 0)
                return len;
            else
            {
                if(set.Contains(remainder))
                    break;
                else
                    set.Add(remainder);
            }   
        }
        
        return -1;
    }
}
