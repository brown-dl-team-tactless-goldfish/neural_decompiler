class Solution {
public:
    int partitionArray(vector<int>& nums, int k) {
        sort(nums.begin(), nums.end());
        int i = 0, n = nums.size(), ans=0;
        while(i<n)
        {
            int minv = nums[i++];
            while(i<n && nums[i]-minv<=k)
               i++; 
            ans++;
        }
        return ans;
    }
};
