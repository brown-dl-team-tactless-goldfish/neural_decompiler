bool areOccurrencesEqual(char * s) {

	int alpha[26] = {0};
	register unsigned short i, ct;
	while(s[i] != '\0') {
		alpha[s[i] - 'a']++;
		i++;
	}

	register bool flag = false;
	register unsigned char j;
	for(j = 0; j < 26; j++) {
		if(alpha[j] != 0 && !flag) {
			ct = alpha[j];
			flag = true;
		} else if(flag && (alpha[j] != 0 && alpha[j] != ct)) {
			printf("%d %d\n", alpha[j], ct);
			return false;
		}
	}
	return true;
}
