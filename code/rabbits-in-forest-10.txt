int numRabbits(vector<int>& answers) {
	int res = 0, sz = answers.size() ;
	sort(answers.begin(), answers.end());
	unordered_map<int, int> mp;
	for (int i = 0; i < sz; i++) {
		int curCount = 1;
		while (i + 1 < sz && answers[i] == answers[i + 1] && curCount <= answers[i]) {
			curCount++;
			i++;
		}
		res += answers[i] + 1;
	}
	return res;
}
