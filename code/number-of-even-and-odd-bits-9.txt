class Solution {
public:
    vector<int> evenOddBit(int n) {
        vector<int> ans(2,0);
        bool ep = true;
        
        while(n)
        {
            //cout<<n<<"\n";
            if(n & 1)
            {
                if(ep)
                    ans[0]++;
                else
                    ans[1]++;
            }
            
            ep = !ep;
            n = n >> 1;
        }
        
        return ans;
        
    }
};
