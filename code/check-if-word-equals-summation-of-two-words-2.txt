#include<math.h>
int base = (int)'a';
int val(char *src, int len){
        int i=0, ans=0;
        while(src[i] != NULL){            
            ans += ( (int)*(src+i) - base ) * pow(10,len);  
            i++;
            len--;
        }
        return ans;
    }

bool isSumEqual(char * firstWord, char * secondWord, char * targetWord){
    int sum, t_int;  

     
    sum = val(firstWord, strlen(firstWord)-1) + val(secondWord, strlen(secondWord)-1);
    t_int = val(targetWord, strlen(targetWord)-1);
    return (sum==t_int) ? true : false;

}
