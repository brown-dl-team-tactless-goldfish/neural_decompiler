 int prefixCount(char ** words, int wordsSize, char * pref)
 {
     int j = 0 , counter = 0;
     for ( int i = 0 ; i < wordsSize ; i++)
     {
         while ( words[i][j] == pref[j] && j < strlen(words[i]) && strlen(pref) )
          {
              j++;
          }
         
         if ( j == strlen(pref))
             {
                 counter++;
             }
         
         j = 0;
             
     }
     
     return counter;
 
 }
