int* smallerNumbersThanCurrent(int* nums, int numsSize, int* returnSize){
int* ans = (int*)malloc(sizeof(int)*numsSize);
    *returnSize=numsSize;
    for(int i=0;i<numsSize;i++)
    {
        int count=0;
        for(int j=0;j<numsSize;j++)
        {
            if(j==i)
            {
                continue;
            }
            if(nums[j]<nums[i])
            {
                count++;
            }
        }
        ans[i]=count;
    }
    return ans;
}
