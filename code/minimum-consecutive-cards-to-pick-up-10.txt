class Solution {
public:
    int minimumCardPickup(vector<int>& cards) {
        int n = cards.size(), ans = INT_MAX;
        unordered_map<int, int> idx;
        for (int r = 0; r < n; r++)
        {
            if (idx.count(cards[r]))
            {
                int l = idx[cards[r]];
                ans = min(ans, r - l + 1);
            }
            idx[cards[r]] = r;
        }
        return (ans == INT_MAX ? -1 : ans);
    }
};
