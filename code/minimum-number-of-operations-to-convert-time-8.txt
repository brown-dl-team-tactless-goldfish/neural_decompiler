    int diff = (DateTime.Parse(correct).Hour * 60 + DateTime.Parse(correct).Minute) - (DateTime.Parse(current).Hour * 60 + DateTime.Parse(current).Minute);

    int count = 0;
    while (diff >= 60)
    {
        diff -= 60;              
        count++;                
    }

    while (diff >= 15)
    {
        diff -= 15;
        count++;
    }

    while (diff >= 5)
    {
        diff -= 5;
        count++;
    }

    return count + diff;
    
}
