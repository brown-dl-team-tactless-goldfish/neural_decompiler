inline int min(int a, int b) { return a < b ? a : b; }

int findMin(int* nums, int numsSize){
    if (numsSize == 1) return nums[0];

    int l = 0;
    int r = numsSize - 1;
    int m = (l + r)/2;
    
    int ans = nums[0];
    
    if (nums[l] >= nums[m]) {
        ans = min(ans, findMin(nums, m - l + 1));
    }

    if (nums[m] >= nums[r]) {
        ans = min(ans, findMin(nums + m + 1, r - m));
    }
    
    return ans;
}
