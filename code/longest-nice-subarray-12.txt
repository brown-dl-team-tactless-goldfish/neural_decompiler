public class Solution {
    public int LongestNiceSubarray(int[] nums) {
        int result = 0;
        
        for (int left = 0, right = 0, total = 0; right < nums.Length; total |= nums[right++]) {
            while ((total ^ nums[right]) != (total | nums[right]))
                total ^= nums[left++];
                
            result = Math.Max(result, right - left + 1);
        }
        
        return result;
    }
}
