int max(int a, int b){
    return a > b ? a : b;
}

int wiggleMaxLength(int* nums, int numsSize){
    int Nega = 1, Posi = 1;

    for(int i = 1; i < numsSize; ++i){
        if (nums[i] > nums[i-1]){
            Posi = Nega + 1;
        }else  if  (nums[i] < nums[i-1]){
            Nega = Posi + 1;
        }
    }
    return max(Posi, Nega);

}
