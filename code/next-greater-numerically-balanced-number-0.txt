bool isBalancedNum(int n){
    int* digit = calloc(10, sizeof(int)) ; 
    while(n){
        int d = n % 10 ;
        n /= 10 ;
        digit[d]++ ;
        if(d == 0 || digit[d] > d)
            return false ;
    }
    
    for(int i = 1; i < 10; i++){
        if(digit[i] > 0 && digit[i] != i )
            return false ;
    }
    return true ;
}
int nextBeautifulNumber(int n){
    n += 1 ;
    while(1){
        if(isBalancedNum(n))
            return n ;
        n++ ;
    }
    return -1 ;
}
