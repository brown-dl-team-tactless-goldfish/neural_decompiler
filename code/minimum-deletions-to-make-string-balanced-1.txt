class Solution {
public:
    int minimumDeletions(string s) {
        int n = s.size();
        int ca = 0 ; 
        int cb = 0 ;
        for(int i = 0 ; i <  n;i++)
        {
            if(s[i] == 'a')ca++;
            else cb++;
            ca = min(ca , cb);
        }
        return ca;
    }
};
