vector<int> recoverArray(vector<int>& arr) {
sort(arr.begin(),arr.end());
int n=arr.size();
    multiset<int>ms(arr.begin(),arr.end());
    
int low=arr[0];
    
    for(int i=1;i<n;i++){
        vector<int>ans;
        int high=arr[i];
        if((high-low)%2==0){
        int k=(high-low)/2;
     if(k&&solve(ms,ans,k)){
         return ans;
     }   

    }
    }
    
    return {};
}
