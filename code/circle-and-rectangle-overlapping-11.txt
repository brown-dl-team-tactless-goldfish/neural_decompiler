class Solution {
public:
    bool checkOverlap(int radius, int x_center, int y_center, int x1, int y1, int x2, int y2) {
        
        int lenX = (x1 > x_center) ? x1 : (x2 < x_center) ? x2 : x_center; 
        int lenY = (y1 > y_center) ? y1 : (y2 < y_center) ? y2 : y_center; 
        int diffX = x_center - lenX;
        int diffY = y_center - lenY;
        return diffX * diffX + diffY * diffY <= radius * radius; 
    }
};
