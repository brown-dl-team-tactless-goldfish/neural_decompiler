int partitionString(char * s){
    int n = strlen(s);
    int ans = 0;
    int* post = malloc( 26 * sizeof(int));
    for(int i = 0; i < 26; i++){
        post[i] = -1;
    }
    int start = 0, last = 0;
    for(int i = 0; i < n; i++){
        if(post[s[i] - 'a'] == -1 || post[s[i] - 'a'] < start)          
            last++;
        else{
            ans++;
            start = i;
            last = i;                
        }
        post[s[i] - 'a'] = i;
    }
    return ans + 1;
}
