class Solution {
private:
    const int MOD = 1e9 + 7;
public:
    int numMusicPlaylists(int N, int L, int K) {
        vector<vector<long>> d(L+1, vector<long>(N+1, 1));
        d[0][0] = 1;
        for (int i = 1; i <= L; ++i) {
            for (int j = 1; j <= min(i, N); ++j) {
                d[i][j] = (d[i-1][j-1] * (N-j+1) + d[i-1][j] * max(0, j-K)) % MOD;
            }
        }
        return d[L][N];
    }
};
