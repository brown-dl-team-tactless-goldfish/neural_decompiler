public class Solution {
    public int[] SortJumbled(int[] mapping, int[] nums) {
        return Enumerable.Range(0, nums.Length).OrderBy(i => Map(mapping, nums[i])).Select(i => nums[i]).ToArray();
    }
    
    public int Map(int[] mapping, int n) {
        string res = "";
        foreach(char c in n.ToString()) {
            var m = mapping[(int)(c - '0')];
            res = res + m.ToString();
        }
        return Int32.Parse(res);
    }
}
