bool isValid(char* s) {
    int len = strlen(s), top = 0;
    char* t = (char*) malloc((len + 1) * sizeof(char));
    for (int i = 0; i < len; i++) {
        t[top++] = s[i];
        if (top > 2 && (t[top - 1] == 'c' && t[top - 2] == 'b' && t[top - 3] == 'a')) top -= 3;
    }
    free(t);
    return top == 0;
}
