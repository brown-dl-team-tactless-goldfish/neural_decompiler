int kthLargestValue(vector<vector<int>>& A, int k, vector<int> v = {}) {
	for (int i = 0; i < A.size(); ++i) 
		for (int j = 0; j < A[0].size(); ++j)
			v.push_back(A[i][j] ^= (i > 0 ? A[i - 1][j] : 0) ^ (j > 0 ? A[i][j - 1] : 0) ^ (i > 0 && j > 0 ? A[i - 1][j - 1] : 0));
	sort(begin(v), end(v), greater<int>());
	return v[k - 1];
}
