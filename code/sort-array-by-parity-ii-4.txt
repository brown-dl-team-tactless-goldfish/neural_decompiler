int* sortArrayByParityII(int* A, int ASize, int* returnSize){
    *returnSize = ASize;
    int odd_idx = 1;
    int even_idx = 0;
    int *result = (int *)malloc(sizeof(int) * ASize);
    for (int i = 0; i < ASize; i++)
    {
        if (A[i] % 2)
        {
            result[odd_idx] = A[i];
            odd_idx += 2;
        }
        else
        {
            result[even_idx] = A[i];
            even_idx += 2;
        }
    }

    return result;
}
