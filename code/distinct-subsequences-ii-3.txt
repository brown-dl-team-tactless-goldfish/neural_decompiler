int distinctSubseqII(string s) { 
	int n = s.size();
	long long dp[n+1];
	dp[0] = 1;
	vector<int> count(26, 0);
	for(int i=1; i<=n; i++){
		dp[i] = 2*dp[i-1] % MOD;
	    char c = s[i-1] - 'a';
	    if(count[c]){
			int j = count[c];
			dp[i] = (dp[i] - dp[j-1] + MOD) % MOD;
	    }
		count[c] = i;
	}
	return (dp[n]-1)%MOD;
}
