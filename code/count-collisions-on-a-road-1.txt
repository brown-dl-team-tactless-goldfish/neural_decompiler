int countCollisions(char * directions){
    int n = strlen(directions);
    bool flag = false;
    int ans = 0;
    for(int i = 0; i < n; i++){
        if( directions[i] == 'S' || directions[i] =='R')
            flag = true;
        else if (flag == true)
            ans++;
    }

    flag = false;
    for(int i = n-1; i >= 0; i--){
        if( directions[i] == 'S' || directions[i] =='L')
            flag = true;
        else if(flag == true)
            ans++;
    }
    
    return ans;
}
