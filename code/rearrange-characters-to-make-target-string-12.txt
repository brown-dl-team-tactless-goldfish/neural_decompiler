int rearrangeCharacters(string s, string target) {
        map<char, int> mp;
        
        for(char c: s){
            mp[c]++;
        }
        
        map<char, int> tmp;
        for(char c : target)
            tmp[c]++;
        
        int min = INT_MAX;
        for(char c : target){
            int req = tmp[c];
            int avail = mp[c];
            int nos = avail/req;
            if(nos < min)
                min = nos;
        }
        return min;
