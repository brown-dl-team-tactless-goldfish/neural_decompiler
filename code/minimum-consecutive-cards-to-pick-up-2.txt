class Solution {
public:
    int minimumCardPickup(vector<int>& nums) {
        int result=INT_MAX; 
        unordered_map<int,pair<int,int>>mp;
     for(int i=0;i<nums.size();i++)
     {
         if(mp[nums[i]].second>=1)
             result=min(result,i-mp[nums[i]].first+1);
             
         mp[nums[i]].first=i;
        mp[nums[i]].second++;
         
     }
        return result==INT_MAX?-1:result;
    }
};

Time Complexity: O(n)
Space Complexity: O(n)
