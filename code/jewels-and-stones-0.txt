int numJewelsInStones(char * jewels, char * stones){
    bool* jew = calloc(256, sizeof(bool)) ;
    int ans = 0 ;
    for(int i = 0 ; i < strlen(jewels) ; i ++){
        jew[ jewels[i]  ] = true;
    }
    for(int i = 0 ; i < strlen(stones) ; i++){
        if( jew[ stones[i] ] )
            ans++;
    }
    return ans;
}
