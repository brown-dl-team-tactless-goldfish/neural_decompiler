    for(auto &x: arr)
    {
        int r = ((x%k)+k)%k;
        
        mp[r]++;
    }
    
    for(auto &x: arr)
    {
        int rem = ((x%k)+k)%k;
        
        if(rem == 0)
        {
            if(mp[rem] % 2 != 0)
            {
                return false;
            }
        }
        
        else if(mp[rem] != mp[k-rem])
        {
            return false;
        }
    }
    
    return true;
}
