class Solution {
public:
    int minimumRecolors(string blocks, int k) {
        int n = size(blocks), dp[101] = {};
        for (int i = 1; i <= n; i++) {
            dp[i] = dp[i-1] + (blocks[i-1] == 'W');
            if (i > k) dp[i] -= (blocks[i-k-1] == 'W');
        }
        return *min_element(begin(dp)+k, begin(dp)+n+1);
    }
};
