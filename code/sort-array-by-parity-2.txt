int* sortArrayByParity(int* A, int ASize, int* returnSize){
    int i,left=0,right=ASize-1, *r;
    *returnSize = ASize;
    r = (int*)malloc(sizeof(int)*ASize);
    for(i=0;i<ASize;i++){
        if(A[i]%2==0)
            r[left++]=A[i];
        else
            r[right--]=A[i];
    }
    return r;
}
