int i=0,j=numsSize-1,k;
while(i<j)                      //move till both pointer cross 
{
    if(nums[i]==0)              //if we found 0 then shift all element by one
    {
        for(k=i;k<j;k++)
            nums[k]=nums[k+1];
        nums[j]=0;              //and append 0 at last
        j--;                    //move j backward
    }
    else
        i++;                    //move i forwward
}
