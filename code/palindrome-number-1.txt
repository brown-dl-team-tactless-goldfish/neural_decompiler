bool isPalindrome(int x) {
    
    int digits[10];
    
    if (x<0) {
        return false;
    }
    
    int maxIndex = 0;
    while(x!=0) {
        digits[maxIndex] = x % 10;
        x /= 10;
        maxIndex++;
    }
    
    int i=0, j=maxIndex -1;
    
    while (i<j) {
        if (digits[i] != digits[j]) {
            return false;
        }
        i++;
        j--;
    }
    return true;
}
