#include<string.h>
int minFlips(char * target){
    int flip=0;
    int flag=0;
    int size=strlen(target);
    for(int i=0; i<size; i++ )
    {
        if(target[i]=='1')
        {
            flag=1;
            flip++;
            for(int j=i+1; j<size; j++)
            {
                if(target[j]!=target[j-1])
                    flip++;
            }
        }
        if(flag==1)
            break;
    }
    return flip;

}
