
int prefixCount(char ** words, int wordsSize, char * pref)
{
    int NumOfWords = 0;
    int i;
    char * destination[strlen(pref)];
    
    for (i = 0; i < wordsSize ; i++)
    {
        memset(destination,'\0',sizeof(destination));
        strncpy(destination, words[i], strlen(pref));
        if (!(strcmp(destination,pref)))
        {
             NumOfWords++;   
        }
    }
    return (NumOfWords);
}
