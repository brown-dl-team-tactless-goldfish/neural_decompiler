int findLUSlength(std::string a, std::string b) 
{
    const int da = a.size();
    const int db = b.size();
    
    if(da == db)
    {
        if(a != b) return a.size();
    }
    else if(da < db)
    {
        return db;
    }
    else if(da > db)
    {
        return da;
    }
    
    return -1;
}
