class Solution {
public:
    string getFraction(int denominator,int numerator)
    {
        return (to_string(denominator) + "/" + to_string(numerator));
    }
    int getGcd(int denominator,int numerator)
    {
        if((denominator % numerator) == 0)
            return numerator;
        return getGcd(numerator,denominator % numerator);
    }
    bool isValid(int numerator,int denominator)
    {
        return (getGcd(numerator,denominator) == 1);
    }
    vector<string> simplifiedFractions(int n) 
    {
        vector<string> simplifiedFractionVector;
        for(int i = 1 ; i <= n ; i++)
        {
            for(int j = i+1 ; j <=n ; j++ )
            {
                if(isValid(j,i))
                {
                    simplifiedFractionVector.push_back(getFraction(i,j));
                }
            }
        }
        return simplifiedFractionVector;
    }
};
