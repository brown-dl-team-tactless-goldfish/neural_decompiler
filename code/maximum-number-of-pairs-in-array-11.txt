class Solution {
public:
    vector<int> numberOfPairs(vector<int>& nums) {
        int m=0;
        int n=0;
     unordered_map<int,int>mp; 
        for(auto num:nums)
        {
         if(mp[num]%2)
         {
             n-=1;
             m+=1;
             
         }
        else
             n+=1;
            
        mp[num]++;
        }
        return {m,n};
    }
};
