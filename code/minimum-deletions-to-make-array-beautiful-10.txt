    int minDeletion(vector<int>& nums) {
        // if(nums.size() == 0)return 0;
        if(nums.size() <= 1)return nums.size();
        
         int c = 0;//Counter to count the 2nd condition 
            
            for(int i = 0;i<nums.size()-1;i++){
                if((i-c)%2 == 0 && nums[i]==nums[i+1])c++; // here i-c indicate the updates index of the current elemnt if these is any instance where 2nd condition is coming true
            }
            
            if((nums.size()-c)%2)return c+1;//if after deleteing c elements size of array is odd we will delete one more element 
            return c;
        
    }
