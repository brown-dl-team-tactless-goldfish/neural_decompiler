int splitArray(vector<int>& nums, int m) {
	int l = 0, r = std::accumulate(nums.begin(), nums.end(), 0);
	for(auto& num : nums) l = max(l, num);

	int c = 0, n = nums.size(), mid = 0, sum = 0;
	while(l<r)
	{
		mid = l + (r - l) / 2;
		sum = 0;
		c = 1;
		for(int i=0; i<n; i++)
		{
			if(sum + nums[i] <= mid)
				sum += nums[i];
			else
			{
				sum = nums[i];
				c++;
			}
		}

		if(c<=m) r = mid;
		else l = mid + 1;
	}

	return l;
}
