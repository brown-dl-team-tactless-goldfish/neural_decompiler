/**
 * Definition for a Node.
 * struct Node {
 *     int val;
 *     struct Node *left;
 *     struct Node *right;
 *     struct Node *next;
 * };
 */

void dfs(struct Node* root)
{
        if (!root->left)
                return;
        root->left->next = root->right;
        if (!root->next)
                root->right->next = NULL;
        else
                root->right->next = root->next->left;
        
        dfs(root->left);
        dfs(root->right);
}

struct Node* connect(struct Node* root)
{
        if (!root)
                return NULL;
        
        root->next = NULL;
        dfs(root);
        
        return root;
}
