char* destCity(char*** paths, int pathsSize, int* pathsColSize) {
	int i, j, chk;
	for (i = 0; i < pathsSize; i++) {
		chk = 0;
		for (j = 0; j < pathsSize; j++)
			if (!strcmp(paths[i][1], paths[j][0])) {
				chk = 1;
				continue;
			}
		if (chk)
			continue;
		return paths[i][1];
	}
	return NULL;
}
