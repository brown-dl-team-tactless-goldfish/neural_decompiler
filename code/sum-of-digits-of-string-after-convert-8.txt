int getLucky(char * s, int k){
    int sum=0,res = 0;
    for(int i=0;i<strlen(s);i++){
       sum+=(int)s[i]-96;
       while(sum!=0){
           res+=sum%10;
           sum=sum/10;
       }
    }
    for(int i=1;i<k;i++){
        int ans = 0;
        while(res!=0){
            ans += res%10;
            res=res/10;
        }
        res=ans;
    }
    return res;
}
