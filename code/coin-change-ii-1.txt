int change(int amount, int* coins, int coinsSize) {
    if(coinsSize==0){
        if(amount==0){
            return 1;
        }
        return 0;
    }
    int* dp=(int*)calloc(amount+1,sizeof(int));
    dp[0]=1;
    for(int i=0;i<coinsSize;i++){
        for(int j=coins[i];j<amount+1;j++){
            dp[j]+=dp[j-coins[i]];
        }
    }
    return dp[amount];
}
