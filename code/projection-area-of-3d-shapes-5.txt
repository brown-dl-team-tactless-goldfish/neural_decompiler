for(int i =0;i<gridSize;i++)
{
    iX[i] = 0;
}
for(int j = 0;j<*gridColSize; j++)
{
    iY[j] = 0;
}
for(int i =0;i<gridSize;i++)
{        
    for(int j = 0;j<*gridColSize; j++)
    {
        if(grid[i][j]>0)  isum++;
        if(iX[i]<grid[i][j]) iX[i]=grid[i][j];
        if(iY[j]<grid[i][j]) iY[j]=grid[i][j];
    }
}

for(int i =0;i<gridSize;i++)
{
    isum += iX[i];
}
for(int j = 0;j<*gridColSize; j++)
{
    isum += iY[j];
}

return isum;
