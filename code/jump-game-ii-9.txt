int jump(int* nums, int numsSize) {
	int max = nums[0], res = 1, tmp = 0;
	if (max >= numsSize - 1) return (numsSize == 1) ? 0 : 1;
	for (int i = 1;i < numsSize;)
	{
		while (i <= max)
		{
			tmp = (i + nums[i]) > tmp ? (i + nums[i]) : tmp;
			i++;
		}		
		res++;
		if (tmp >= numsSize - 1) return res;
		max = tmp;
	}
	return res;
}
