class Solution {
public:
    int minGroups(vector<vector<int>>& arr) {
        multiset<int> ms;
        sort(arr.begin(),arr.end());
        int g = 0;
        for(auto &vec : arr){
            int st = vec[0], en = vec[1];
            
            auto lb = ms.lower_bound(st);
            
            if(lb == ms.begin()) g++;
            else ms.erase(--lb);
            
            ms.insert(en);
        }
        return g;
    }
};
