/*
I count the number of students who wants square sandwiches and 
those who wants Circular Sandwiches and check if any student still
want the sandwich on the top if no one wants that kind I will return 
the remain students
*/
int countStudents(int* students, int studentsSize, int* sandwiches, int sandwichesSize){
    int wantCircularSandwiches = 0, wantSquareSandwiches = 0 ;
    for(int i = 0 ; i < studentsSize ; i++) {
        if(students[i] == 0) {
            wantCircularSandwiches++;
        }
        else{
            wantSquareSandwiches++;
        }
    }
    for(int i = 0 ; i < sandwichesSize ; i++) {
        
        if(sandwiches[i] == 0 && wantCircularSandwiches != 0){
            wantCircularSandwiches--;
        }
        else if (sandwiches[i] == 1 && wantSquareSandwiches!= 0){
            wantSquareSandwiches--;
        }
        else {
            break;
        }
    }
    return wantSquareSandwiches + wantCircularSandwiches ;
}
