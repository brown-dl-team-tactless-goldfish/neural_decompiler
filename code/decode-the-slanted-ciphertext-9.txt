class Solution {
public:
    
    string decodeCiphertext(string eT, int r) {
        int len = eT.length();
        if(len == 0) return eT;
        if(r == 1) return eT;
        string ans;
        int c = len/r;
        int i=0,u=0;
        while(i != c-1){
            int l = ans.length()-1;
            ans += eT[i];
            i += c+1;
            if(i>=len) i = ++u;
        }
        ans += eT[i];
        while(ans[ans.size() - 1] == ' ')  ans.pop_back();
        return ans;
    }
};
