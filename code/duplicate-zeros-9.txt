void duplicateZeros(int* arr, int arrSize){
    int i=0;
    int j=arrSize-1;
    while(i<j){  // arr[j] will be the last element in the modified array
        if(arr[i] == 0) j--;
        i++;
    }
    if(j == arrSize-1) return;
    int k = arrSize-1;
    if(i==j) arr[k--] = arr[j--];  // if i not equals to j, then arr[j] is 0 and duplicated; if i equals to j, then arr[j] is not 0 or not duplicated
    while(j<k){
        if(arr[j] == 0) arr[k--] = 0;
        arr[k--] = arr[j--];
    }
}
