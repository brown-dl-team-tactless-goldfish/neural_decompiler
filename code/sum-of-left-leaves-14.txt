void help(struct TreeNode* root, int *sum) {
    if(root == NULL) return;
    if(root->left)
        if(root->left->left == NULL && root->left->right==NULL)
            *sum += root->left->val;
    help(root->left,sum);
    help(root->right,sum);
}


int sumOfLeftLeaves(struct TreeNode* root){
    if(root==NULL) return 0;
    int sum = 0;
    help(root,&sum);
    return sum;
}
