int subtractProductAndSum(int n){
    int arr [6]; 
    int count = 0;
    int prod = 1;
    int sum = 0;
    while(n != 0)
    {
        arr[count] = n % 10;
        n = n /= 10;
        count = count + 1;
    }
    for(int i = count-1; i >= 0; i--)
    {
        prod = prod * arr[i];
    }
    for(int i = count-1; i >= 0; i--)
    {
        sum = sum + arr[i];
    }
    int solution = prod-sum;
    return solution;

}
