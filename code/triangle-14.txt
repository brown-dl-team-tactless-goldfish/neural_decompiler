#define MIN(x, y) (((x) < (y)) ? (x) : (y))

int minimumTotal(int** triangle, int triangleSize, int* triangleColSize){
    int i;
    int j;
    for (i=triangleSize-2; i>= 0; i--) {
        for (j=0; j<=i; j++) {
            triangle[i][j] += MIN(triangle[i+1][j], triangle[i+1][j+1]);
        }
    }
    return triangle[0][0];
}
