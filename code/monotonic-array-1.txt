bool isMonotonic(int* A, int ASize) {
    int i, j;
    for(i = 1; i < ASize; ++i) if(A[i] != A[i - 1]) break;
    if(A[i] > A[i - 1]){
        for(j = i + 1; j < ASize; ++j) if(A[j] < A[j - 1]) return false;
    }
    else{
        for(j = i + 1; j < ASize; ++j) if(A[j] > A[j - 1]) return false;
    }
    return true;
}
