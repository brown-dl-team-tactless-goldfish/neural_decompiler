string oddString(vector<string>& words) {
	map<vector<int>, int> mp;
	for (auto& w : words) {
		vector<int> v(w.size() - 1);
		for (int j = 0; j < w.size() - 1; j++)
			v[j] = w[j + 1] - w[j];
		mp[v]++;
	}
	for (auto& w : words) {
		vector<int> v(w.size() - 1);
		for (int j = 0; j < w.size() - 1; j++)
			v[j] = w[j + 1] - w[j];
		if (mp[v] == 1)
			return w;
	}
	return "";
}
