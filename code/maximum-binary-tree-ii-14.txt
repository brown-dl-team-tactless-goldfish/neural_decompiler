class Solution {
public:
    TreeNode* insertIntoMaxTree(TreeNode* root, int x) {
        TreeNode* t = new TreeNode;
        t->right = t->left = NULL;
        t->val = x;
        if(!root) return t;
        if(root->val < x) {
            t->left = root;
            root = t;
        } else {
            root->right = insertIntoMaxTree(root->right, x);
        }
        return root;
    }
};
