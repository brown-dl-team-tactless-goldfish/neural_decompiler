#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

class Solution {
public:
    int maximumProduct(vector<int>& nums, int k) {
      int mod=1e9+7;
        priority_queue<ll, vector<ll>, greater<long long>> pq;
        for(ll it: nums){
            pq.push(it);
        }
        for(ll i=0;i<k;i++){
            ll x=pq.top();
            pq.pop();
            pq.push(x+1);
        }
    ll prod=1;
        for(ll i=0;i<nums.size();i++){
            ll x=pq.top();
            pq.pop();
            prod=(prod*x)%mod;
        }
        return prod;
    }
};
