public class Solution {
    public IList<int> NumOfBurgers(int tomatoSlices, int cheeseSlices) {
                   
        if(tomatoSlices % 2 == 0 && tomatoSlices / 2 - cheeseSlices >= 0 && 2 * cheeseSlices - tomatoSlices / 2 >= 0 )
            return new List<int>(){tomatoSlices / 2 - cheeseSlices, 2 * cheeseSlices - tomatoSlices / 2};
        else
            return new List<int>();
    }
}
