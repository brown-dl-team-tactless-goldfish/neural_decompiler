bool canConvertString(string s, string t, int k) {
	if (s.size() != t.size())
		return false;

	vector<int> moves(26);
	for (int i = 0; i < s.size(); i++) {
		if (s[i] != t[i]) {
			int shift = t[i] - s[i];
			if (shift < 0) {
				shift += 26;
			}
			if (shift > 0 && 26 * moves[shift] + shift > k)
				return false;
			else
				moves[shift]++;
		}
	}
	return true;
}
