int minimumNumbers(int num, int k){
    if(num == 0)
        return 0 ;
    int data[11] ;
    for(int i = 1; i <= 10; i++){
        data[i] = (k*i)%10 ;
    }
    int m = num % 10 ;
    for(int i = 1; i <= 10; i++){
        if(m == data[i] && num >= k*i)
            return i ;
    }
    return -1 ;
}
