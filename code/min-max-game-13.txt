class Solution {
public:
    int minMaxGame(vector<int>& nums) {
        
        while(nums.size() > 1){
            bool mini = true ;
            vector<int>temp ;
            for(int i = 0 ; i < nums.size() - 1 ; i += 2 ){
                if(mini) temp.push_back(min(nums[i],nums[i+1])) ;
                else temp.push_back(max(nums[i],nums[i+1])) ;
                mini = !mini ;
            }
            nums = temp ;
        }
        return nums.back() ;
    }
};
