if(prices== NULL || pricesSize==0) return 0;
int max = 0;
int* first = firsthalf(prices,pricesSize);
int* last = lasthalf(prices,pricesSize);

for(int i=0;i+1+1<pricesSize;i++){
    int t = first[i]+last[i+1];
    if(t>max) max =t;
}
int edge = fmax(last[0],first[pricesSize-1]);

return fmax(max,edge);
