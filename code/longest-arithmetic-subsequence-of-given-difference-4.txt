class Solution {
public:
    int longestSubsequence(vector<int>& arr, int d) {
        unordered_map<int, int> map;
        
        int ans = 0;
        
        for (int i = arr.size() - 1; i >= 0; --i) {
            map[arr[i]] = map[arr[i] + d] + 1;
            ans = max(map[arr[i]], ans);
        }
        
        return ans;
    }
};
