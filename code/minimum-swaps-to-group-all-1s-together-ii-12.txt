class Solution {
public:
    int minSwaps(vector<int>& nums) {
        int n = nums.size(), res = INT_MAX;
        int ones = count(nums.begin(), nums.end(), 1);
        int zcnt = 0;
        if (!ones)
            return 0;
        for (int l = 0, r = 0; l < n; r = (r + 1) % n) {
            zcnt += !nums[r];
            if (r - l + 1 == ones || r < l) {
                res = min(res, zcnt);
                zcnt -= !nums[l++];
            }
        }
        return res;
    }
};
