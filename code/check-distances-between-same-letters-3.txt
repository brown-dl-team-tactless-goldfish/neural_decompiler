bool checkDistances(char * s, int* distance, int distanceSize){
    char ind=0;
    while(s[ind]!=0){
        distance[s[ind]-97]+=ind;
        ind++;
    }
    ind--;
    while (ind>0 && (distance[s[ind]-97]==-1 || distance[s[ind]-97]==ind+ind-1)){//2nd occurance can't be on s[0]
        distance[s[ind]-97]=-1;
        ind--;
    }
    return (!ind);
}
