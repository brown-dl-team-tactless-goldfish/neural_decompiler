int countSegments(char * s){
    int count = 0 , i = 0;
    bool flag = true;
    int n = strlen(s);
    while (i<n){
        if(*(s+i) == ' '){
            flag =true;
        }
        else{
            if (flag == true){
                count++;
            }
            flag = false;
        }
        ++i;
    }
    return count;

}
