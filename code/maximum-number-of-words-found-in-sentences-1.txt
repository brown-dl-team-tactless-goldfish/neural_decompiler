

int mostWordsFound(char ** sentences, int sentencesSize){
    int ans = 0;
    for (int i = 0; i < sentencesSize; i++){
        int cur = 1;
        int j = 0;
        while(sentences[i][j]){
            if (sentences[i][j] == ' ')
                cur++;
            j++;
        }
        ans = (cur > ans) ? cur : ans;
    }
    return ans;
}
