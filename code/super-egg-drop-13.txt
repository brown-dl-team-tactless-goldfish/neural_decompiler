class Solution {
public:
    int superEggDrop(int k, int n){
        if(k==0 || n==0) return 0;
        int arr[10001][101]={0};
        for(int i=1;i<=k;i++) arr[1][i]=1;
        for(int i=2;i<=n;i++){
            for(int j=1;j<=k;j++){
                arr[i][j]=1+arr[i-1][j-1]+arr[i-1][j];
                if(arr[i][j]>=n) return i;
            }
        }
        return 1;
    }
};
