int searchInsert(int* nums, int numsSize, int target){
    int i = 0;
    int j;
    for(j = 1; j <= numsSize; j++){
        if(j == numsSize){
            return numsSize;
        }
        else if((target < nums[i]) && (target < nums[j])){
            return i;
        }
        else if((target > nums[i]) && (target < nums[j])){
            return j;
        }
        else{
            i++;
        }
    }
}
