class Solution {
    public int[] findArray(int[] pref) {
        int length = pref.length;
        if (length == 1) {
            return pref;
        }
        int [] result = new int [length];
        result[0] = pref[0];
        for (int i = 1; i < length; i++) {
            result[i] = pref[i - 1] ^ pref[i];
        }
        return result;
    }
}
