class Solution:
    def reversePrefix(self, word: str, ch: str) -> str:
        if (ch not in word):
            return word
        
        n = len(word)
        idx = word.index(ch)
        
        if (idx == n - 1):
            return word[::-1]
        
        part1 = word[:idx+1]
        part2 = word[idx+1:]
        
        return part1[::-1] + part2
