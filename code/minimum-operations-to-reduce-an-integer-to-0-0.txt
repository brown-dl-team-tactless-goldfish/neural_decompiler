/* more than 2 continus 1 needs to add 1 */
int minOperations(int n){
    int contCount = 0, ans = 0;
    while(n){
        if(n & 1){
            contCount++;
            n >>= 1;
        }
        else {
            ans += contCount? 1 : 0;
            if(contCount > 1)
                n |= 1;/* do not shift add one for next round calculate */
            else
                n >>= 1;
            contCount = 0;
        }
        
    }
    return contCount > 1? ans + 2: ans + 1;
}
