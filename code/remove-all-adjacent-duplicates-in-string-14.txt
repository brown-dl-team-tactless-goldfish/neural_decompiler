

char * removeDuplicates(char * s){
    int slen=strlen(s);
    int index=0;
    int i=1;
    char* stack=malloc(sizeof(char)*slen);
    
    stack[0]=s[0];
    while(i < slen){
        if(index>=0 && stack[index]==s[i])
            index--;
        else
            stack[++index]=s[i];
        i++;
    }
    stack[index+1]='\0';
    return stack;
}
