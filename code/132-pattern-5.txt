class Solution {
public:
    bool find132pattern(vector<int>& nums) {
        
        int n = nums.size();
        stack<int>stk;
        int prev = INT_MIN;
        
        for (int i = n-1; i >= 0; i--) {
            if (nums[i] < prev)
                return true;
            while (!stk.empty() && nums[i] > stk.top()) {
                prev = stk.top();
                stk.pop();
            }
            stk.push(nums[i]);
        }
        return false;
    }
};
