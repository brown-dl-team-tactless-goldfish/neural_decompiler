class Solution:
    def minimumOperations(self, nums: List[int], start: int, goal: int) -> int:
        seen = set()
        que = deque([(start, 0)])
        
        while que:
            item, cnt = que.popleft()
            if item == goal:
                return cnt
            if item in seen:
                continue
            if item >= 0 and item <= 1000:
                for n in nums:
                    que.append((item + n, cnt + 1))
                    que.append((item - n, cnt + 1))
                    que.append((item ^ n, cnt + 1))
            seen.add(item)
        
        return -1
