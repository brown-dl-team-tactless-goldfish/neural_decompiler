int* sortedSquares(int* A, int ASize, int* returnSize)
{
    int* res = (int *)malloc(sizeof(int) * ASize);
    *returnSize = ASize;
    int *pos = A + ASize - 1, *neg = A, k = ASize - 1;
    while (pos != neg - 1)
    {
        if (*pos >= abs(*neg))
        {
            res[k] = (*pos) * (*pos);
            k--;
            pos--;
        }
        else
        {
            res[k] = (*neg) * (*neg);
            k--;
            neg++;
        }
    }
    return res;
}
