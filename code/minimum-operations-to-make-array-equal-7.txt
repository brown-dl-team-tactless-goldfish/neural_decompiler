int minOperations(int n){
    int numOfOperations = 0;
    int target, i;
    
    if(n%2 == 1){
        target = 2*(n/2)+1;
        
        for(i=0;i<n/2;i++)
            numOfOperations += target - (2*i+1);
    }
    else{
        target = 2*(n/2);
        for(i=0;i<n/2;i++)
            numOfOperations += target - (2*i+1);
    }
    return numOfOperations;
}
