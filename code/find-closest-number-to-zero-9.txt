public class Solution {
    public int FindClosestNumber(int[] nums) {
        
        if(nums == null || nums.Length == 0)
            throw new ArgumentException("Invaid Input.");
        
        int res = Int32.MinValue, minDiff = Int32.MaxValue;
        foreach(var num in nums)
        {
            int currDiff = Math.Abs(num);
            
            if(currDiff <= minDiff)
            {
                res = currDiff < minDiff? num : Math.Max(res, num);
                minDiff = currDiff;   
            }
        }
        
        return res;
    }
}
