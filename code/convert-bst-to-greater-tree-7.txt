/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     struct TreeNode *left;
 *     struct TreeNode *right;
 * };
 */

int pre;

void traversal(struct TreeNode* node){
    if(!node) return NULL;
    traversal(node->right);
    node->val+=pre;
    pre=node->val;
    traversal(node->left);
    
}
struct TreeNode* convertBST(struct TreeNode* root){
    pre=0;
    traversal(root);
    
    return root;
}
