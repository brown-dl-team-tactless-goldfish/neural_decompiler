{   
    if (*s1 == '\0')
        s1 = word1 [w1];
    if (*s2 == '\0')
        s2 = word2 [w2];  

    while (*s1 != '\0' && *s2 != '\0')
    {
        if (*s1 != *s2)
            return 0;
        s1++;
        s2++;
    }

        w1++;
    if (*s2 == '\0')
        w2++;  
}

    return 1;
