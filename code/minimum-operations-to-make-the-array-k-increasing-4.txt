    int kIncreasing(vector<int>& arr, int k) {
        int numGood = 0;
        for(int i = 0; i<k; i++){
            vector<int> v;
            for(int j=i;j<arr.size();j+=k){
                auto it = upper_bound(v.begin(), v.end(), arr[j]);
                if(it == v.end()) v.push_back(arr[j]);
                else *it = arr[j];
            }
            numGood += v.size();
        }
        return arr.size() - numGood;
    }
