class Solution:
    def removeNodes(self, head: Optional[ListNode]) -> Optional[ListNode]:
        dummy = ListNode(inf)
        stack = [dummy]
        node = head
        while node: 
            while stack[-1].val < node.val: stack.pop()
            stack[-1].next = node
            stack.append(node)
            node = node.next 
        return dummy.next
