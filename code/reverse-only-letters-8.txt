char * reverseOnlyLetters(char * s){
    char letters[100];
    int letters_size = 0, length = strlen(s);
    for(int i = 0 ; i < length; i++){
        if(isalpha(s[i])){
            letters[letters_size++] = s[i];
        }
    }
    for(int i = 0 ; i < length ; i++){
        if(isalpha(s[i])){
            s[i] = letters[--letters_size];
        }
    }
    return s;
}
