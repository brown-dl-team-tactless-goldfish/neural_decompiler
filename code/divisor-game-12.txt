bool next_iter(int N, bool turn)
{
    for (int i = 1; i < N; i++) {
        if (N%i == 0) {
            return next_iter(N-i, !turn);
        }
     }
    return !turn; 
}

bool divisorGame(int N){
    for (int i = 1; i < N; i++)
    {
        if (N%i == 0) {
            return next_iter(N-i, false);
        }
    }
    return false; 
}
