bool isThree(int n){
    int cnt = 0;
    for (int i = 1; i <= sqrt(n); i++)
    {
        /* code */
        if (n % i == 0 && i*i == n)
        {
            /* code */
            cnt += 1;
        }
        else if(n % i == 0){
            cnt+=2;
        }
        if(cnt > 3) return false;
    }
    if(cnt == 3) return true;
    return false;
}
