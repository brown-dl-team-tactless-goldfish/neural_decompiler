long long findTheArrayConcVal(int* nums, int numsSize){

    long long  out =0 ;

    int i=0;

    long long   tmp=0;

    int q=0;

    for (i=0;i<numsSize/2;i++)
    {
        q=0;
        tmp=nums[numsSize-i-1];

        while(tmp>0)
        {
            q++;
            tmp/=10;
        }

        out+= (nums[numsSize-i-1]+nums[i]*pow(10,q));
        
    }
    if(numsSize%2==1) out+= nums[numsSize/2];

    return out ;

}
