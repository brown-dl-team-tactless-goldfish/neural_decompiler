class Solution:
    def removeDigit(self, number: str, digit: str) -> str:
        n = len(number)
        
        indexes = []
        
        for idx in range(n):
            if (number[idx] == digit):
                indexes.append(idx)
        
        print ("indexes = ", indexes)
        
        res = ""
        for idx in indexes:
            if (res == ""):
                res = number[:idx] + number[idx+1:]
            else:
                tmp = number[:idx] + number[idx+1:]
                if (int(tmp) > int(res)):
                    res = tmp
                    
        return res
