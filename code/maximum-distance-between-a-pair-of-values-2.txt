int maxDistance(int* nums1, int nums1Size, int* nums2, int nums2Size){
    int i = 0, j = 0;
    int maxDistance = -1;
    while(i < nums1Size && j < nums2Size){
        if(i <= j && nums1[i] <= nums2[j]){
            if((j - i) > maxDistance)
                maxDistance = j - i;
            j++;
        }
        else if(nums1[i] > nums2[j])
            i++;
  
        if(i > j)
            j = i;
    }
    return maxDistance==-1? 0 : maxDistance;
}
