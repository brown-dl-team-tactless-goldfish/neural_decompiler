class Solution
{
public:
    int shortestSequence(vector<int> &rolls, int k)
    {
        unordered_set<int> set;
        int rv = 0;
        for (auto &n : rolls)
        {
            set.insert(n);
            if (set.size() == k)
                set.clear(), ++rv;
        }
        return 1 + rv;
    }
};
