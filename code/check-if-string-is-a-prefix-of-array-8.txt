class Solution {
public:
    bool isPrefixString(string s, vector<string>& w) 
    {
        string str="";
        for(auto it:w)
        {
            if(str.size()>s.size())
            {
                return false;
            }
            str+=it;
            if(s==str)
            {
                // cout<<str<<"1"<<endl;
                return true;
            }
        }
        if(str==s)
        {
            return true;
        }
        return false;
    }
};
