class Solution {
public:
    int minOperationsMaxProfit(vector<int>& customers, int boardingCost, int runningCost) {
        int maxProfit{0}, counter{-1}, numPeople{0}, curProfit{0};
        for(int i{0}, j{1}, size{(int)customers.size()}; i < size || numPeople; ++i, ++j) {
            if(i < size) numPeople += customers[i];
            int board{std::min(4,numPeople)};
            numPeople -= board;
            curProfit += board * boardingCost - runningCost;
            if(curProfit > maxProfit) maxProfit = curProfit, counter = j;
        }
        
        return counter;
    }
};
