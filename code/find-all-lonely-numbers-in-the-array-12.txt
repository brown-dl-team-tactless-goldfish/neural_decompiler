class Solution {
public:
    vector<int> findLonely(vector<int>& nums) {
        unordered_map<int, int> mp;
        int n = nums.size();
        vector<int> res;
        
        for(auto& num:nums) mp[num]++;
        
        for(auto& num:nums){
            if(mp[num] == 1){
                if(mp.find(num+1) == mp.end() && mp.find(num-1) == mp.end()){
                    res.push_back(num);
                }
            } 
        }
        return res;
    }
};
