    int tupleSameProduct(vector<int>& nums) {
        map<int,int>mp;
        int i,j,sum=0,x;
        for(i=0;i<nums.size();i++)
        {
            for(j=i+1;j<nums.size();j++)
            {
                mp[nums[i]*nums[j]]++;
            }
        }
        for(auto i:mp)
        {
                x=(i.second*(i.second-1))/2;
                sum+=x;
        }
        return sum*8;
    }
};
