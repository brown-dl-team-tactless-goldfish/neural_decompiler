class Solution {
public:
    string reverseParentheses(string s) {
        string res;
        for (char c : s) {
            if (c != ')') {
			    res += c;
            } else {
			    // â€œ(abbcâ€ ->   "cbba(" -> "cbba"
                int i = res.size() - 1;
                while (~i && res[i] != '(') --i;
                reverse(res.begin() + i, res.end());
                res.pop_back();
            }
        }
        return res;
    }
};
