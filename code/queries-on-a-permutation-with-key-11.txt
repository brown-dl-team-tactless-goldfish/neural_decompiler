class Solution {
public:
    vector<int> processQueries(vector<int>& queries, int m) {
        list<int>l;
        vector<int>res;
        for(int i=1;i<=m;i++){
            l.push_back(i);
        }
        for(auto q:queries){
            auto it=l.begin();
            int c=0;
            while(*it!=q){
                c++;
                it++;
            }
            res.push_back(c);
            l.erase(it);
            l.push_front(q);
        }
        return res;
    }
};
