float getSlope(int x1,int x2,int y1,int y2){
    if(!(x2-x1)){
        return (float)INT_MAX; 
    }     
    return ((float)(y2-y1)/(float)(x2-x1));
}

bool checkStraightLine(int** c, int row, int* col){
    float slope=getSlope(c[0][0],c[1][0],c[0][1],c[1][1]);
    
    for(int i=2;i<row;i++){
        if(slope != getSlope(c[i-1][0],c[i][0],c[i-1][1],c[i][1])){
            return 0;
        }
    }
    return 1;
}
