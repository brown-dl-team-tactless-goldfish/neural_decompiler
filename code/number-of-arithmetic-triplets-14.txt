int arithmeticTriplets(int* nums, int numsSize, int diff){
  int c=0;
  int i,j,k;
  for(i=0;i<=numsSize-1;i++)
  {
      for(j=i+1;j<=numsSize-1;j++)
      {
          for(k=j+1;k<=numsSize-1;k++)
          {
              if(nums[j]-nums[i]==diff && nums[k]-nums[j]==diff)
              c++;
          }
      }
  }
  return c;
}
