class Solution {
public:
   int recur(int i, string &time, int n){
        
        if(i >= n){
            int a = 0, b = 0;
            
            int f = time[0] - '0';
            int ff = time[1] - '0';
            int g = time[3] - '0';
            int gg = time[4] - '0';
        
            int d = f*10 + ff;
            int e = g*10 + gg;
            
            return (d >= 0 && d<=23 && e>=0 && e<=59);
        }
        
        if(time[i] == '?'){
            int cnt = 0;
            for(int j = 0; j<10; j++){
                time[i] = char('0' + j);
                cnt += recur(i + 1, time, n);
                time[i] = '?';
            }
            return cnt;
        }
        
        return recur(i + 1, time, n);
    }
    
    int countTime(string time) {
        int n = time.length();
        return recur(0, time, n);
    }
};
