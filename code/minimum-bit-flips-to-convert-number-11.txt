int minBitFlips(int start, int goal){
    int bitScan = 0;
    int steps = 0;
    
    while(start != goal && bitScan < 31)
    {
        int bmsk = 0x01 << bitScan;
        if((start & bmsk) ^ (goal & bmsk))
        {
            if((goal & bmsk) == bmsk)
                start = start | bmsk; 
            else
                start = start & ~bmsk; 
            steps++;
        }
        bitScan++;
    }
    return steps;
}
