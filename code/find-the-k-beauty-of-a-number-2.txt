int divisorSubstrings(int num, int k) 
{
    
 int count = 0;
    
 string nums = to_string(num);
    
 for(int i=0; i<=nums.size()-k; i++)
 {
    if(stoi(nums.substr(i,k)) != 0) 
    {
        if(num % stoi(nums.substr(i,k)) == 0)
        {
            count++;
        }
    }
}
    
return count;
    
}
