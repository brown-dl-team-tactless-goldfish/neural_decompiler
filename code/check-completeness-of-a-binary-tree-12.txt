var isCompleteTree = function (root) {
    if (!root) return true;
    let q = [root];
    let check = false;

    while (q.length) {
        let curr = q.shift();
        if (curr.left) {
            if (check) return false;
            q.push(curr.left);
        } else check = true;

        if (curr.right) {
            if (check) return false;
            q.push(curr.right);
        } else check = true;
    }
    return true;
};
