class Solution {
public:
    int maximumProduct(vector<int>& nums, int k) {
        int prod=1;
        sort(nums.begin(),nums.end());
        for(int i=nums.size()-2;i>=0;i--){
            while(k){
                nums[i]=nums[i]+1;
            }
        }
        for(int j=0;j<nums.size();j++){
            prod*=nums[j];
        }
        return prod;
    }
};
