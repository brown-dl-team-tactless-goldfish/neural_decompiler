#include<bits/stdc++.h>
class Solution {
public:
    int digArtifacts(int n, vector<vector<int>>& artifacts, vector<vector<int>>& dig) {
        int c = 0;
        vector<vector<bool>>vis(n,vector<bool>(n,false));
        
        for(auto it:dig){
            vis[it[0]][it[1]] = true;
        }
        for(auto it:vis){
            cout<<it[0]<<" "<<it[1]<<endl;
        }
        for(int i = 0 ; i < artifacts.size() ; i++){
            int startI = artifacts[i][0];
            int endI = artifacts[i][2];
            int startJ = artifacts[i][1];
            int endJ = artifacts[i][3];
            bool flag = true;
            
            
            for(int i = startI ; i <= endI ; i++){
                for(int j = startJ ; j <= endJ ; j++){
                    if(vis[i][j] == false){
                        flag = false;
                        break;
                    }
                }
            }
            if(flag)c++;
        }
        return c;
    }
};
