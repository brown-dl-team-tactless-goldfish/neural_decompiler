bool    is_vowel(char c)
{
    c = tolower(c);
    if (c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u')
        return (true);
    return (false);
}

int count_vowels(char *s, int n)
{
    int i;
    int vowels;

    vowels = 0;
    i = -1;
    while (++i < n)
        if (is_vowel(s[i]))
            vowels++;
    return (vowels);
}

bool    halvesAreAlike(char *s)
{
    int len;

    len = strlen(s);
    return (count_vowels(s, len/2) == count_vowels(s + len/2, len/2));
}
