class Solution {
public:
    int numberOfSubstrings(string s) {
        int result = 0;
        for (int i = 0, a = -1, b = -1, c = -1; i < s.length(); i++) {
            if (s[i] == 'a') {
                a = i;
            } else if (s[i] == 'b') {
                b = i;
            } else {
                c = i;
            }
            int idx = min(a, min(b, c));
            if (idx >= 0) {
                result += idx + 1;
            }
        }
        return result;
    }
};
