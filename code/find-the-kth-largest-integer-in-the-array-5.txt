class Solution {
public:
    string kthLargestNumber(vector<string>& nums, int k) {
        sort(nums.begin(), nums.end(), [](auto& a, auto& b) {
           if (a.size() == b.size()) {
               return a > b;
           } else {
               return a.size() > b.size();
           }
        });
        
        return nums[k - 1];
    }
};
