bool checkAlmostEquivalent(char * word1, char * word2){
    int n = strlen(word1);
    int alpha1[26] = {0};
    int alpha2[26] = {0};
    for(int i = 0; i < n; i++){
        alpha1[word1[i] -'a']++;
        alpha2[word2[i] -'a']++;
    }

    for(int i = 0; i < 26; i++){
        if(abs(alpha1[i] - alpha2[i]) > 3)
            return false;
    }
    return true;
}
