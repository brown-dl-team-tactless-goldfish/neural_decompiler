

char * defangIPaddr(char * address){
    int n = strlen(address) ;
    int N = n + 3*2 + 1 ;
    char* ans = malloc(N * sizeof(char) ) ;
    int a_idx = 0 ;
    for(int i = 0; i < n; i++){
        if(address[i] != '.'){
            ans[a_idx] = address[i] ;
            a_idx++;
            continue ;
        }
        ans[a_idx] = '[' ;
        a_idx++ ;
        ans[a_idx] = '.' ;
        a_idx++ ;
        ans[a_idx] = ']' ;
        a_idx++ ;
    }
    ans[a_idx] = '\0' ;
    return ans ;
}
