int maximumSum(vector<int>& arr) {
    int n=arr.size();
    int prevWithDelete=0;
    int prevNotDelete=arr[0];
    int ans=arr[0];
    
    for(int i=1;i<n;i++){
        prevWithDelete=max(prevNotDelete,prevWithDelete+arr[i]);
        prevNotDelete=max(prevNotDelete+arr[i],arr[i]);
        int current=max(prevWithDelete,prevNotDelete);
        ans=max(ans,current);                
   }
    return ans;
}
