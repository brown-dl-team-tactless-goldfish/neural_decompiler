	else
		s[j++] = ss + '0';
        ss=0;
        t=0;
    }
    if(strlen(s)-1==i&&t!=0)
    {
        if (ss > 9)
	{
		int r = 0;
		char s1[k];
		while (ss != 0)
		{
			s1[r++] = ss%10 + '0';
			ss /= 10;
		}
		r--;
		while (r >= 0)
			s[j++] = s1[r--];
	}
        else
            s[j++]=ss+'0';
    }
}
s[j] = '\0';
return digitSum(s,k);
