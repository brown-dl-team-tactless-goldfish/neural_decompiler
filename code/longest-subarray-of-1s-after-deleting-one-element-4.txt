int mex(int a,int b)
{
    if(a > b)
        return a;
    else
        return b;
}

int longestSubarray(int* nums, int numsSize){
    int j , next ;
    int n , sum;

    n = numsSize;
    next = 0;
    j = 0;
    sum = 0;
    for(int i = 0 ; i < n ; i++)
    {
        if(nums[i] == 0)
            next++;
        while(next > 1)
        {
            if(nums[j] == 0)
                next--;
            j++;
        }
        sum = mex(sum,(i - j));
    }
    
    return (sum);
}
