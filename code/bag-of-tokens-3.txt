int points=0;
int left=0, right=tokensSize-1;
int currPwr = P;
int maxPts = 0;

sort_ints(tokens, tokensSize);


while(left <= right){
    if(currPwr >= tokens[left]){
        points++;
        currPwr -=  tokens[left];
        if(points > maxPts)
            maxPts = points;
        left++;
        
    }else if(points > 0){
        points--;  
        currPwr +=  tokens[right];
        right--;
    }else{
        left++;
    }
    
}
return maxPts;
