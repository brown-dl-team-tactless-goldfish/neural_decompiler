Array.prototype.snail = function(r, c) {
    const [a, g] = [this, []]
    if (r * c != a.length)
        return g
    for (let i = 0; i < r; i++)
        g.push(Array(c))
    for (let k = 0, j = 0; j < c; j++) {
        if (j % 2 == 0)
            for (let i = -1; ++i < r;)
                g[i][j] = a[k++]
        else
            for (let i = r; --i >= 0;)
                g[i][j] = a[k++]
    }
    return g
}
