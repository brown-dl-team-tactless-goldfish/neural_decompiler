class Solution:
    def zigzagLevelOrder(self, root: TreeNode) -> List[List[int]]:

        res = []

        def helper(curNode, curLevel):
            if (curNode == None):
                return
            if (len(res) > curLevel):
                if (curLevel % 2 == 0):
                    res[curLevel].append(curNode.val)
                else:
                    res[curLevel].insert(0, curNode.val)
            else:
                res.append([curNode.val])

            # print ("1 res = ", res)
            helper(curNode.left, curLevel+1)
            helper(curNode.right, curLevel+1)

        helper(root, 0)
        # print ("res = ", res)
        return res
