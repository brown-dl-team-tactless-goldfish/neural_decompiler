class Solution {
public:
    vector<int> prevPermOpt1(vector<int>& A) {
        int i = A.size() - 1;
        while (i > 0 and A[i] >= A[i-1])
            i--;
        
        if (--i < 0)
            return A;
        
        int maxPos = i + 1;
        for (int j = i + 2; j < A.size(); j++)
            if (A[j] < A[i] and A[j] > A[maxPos])
                maxPos = j;
        
        swap(A[i], A[maxPos]);
        
        return A;
    }
};
