int partitionDisjoint(int* A, int ASize) {
    int* array=(int*)malloc(ASize*sizeof(int));
    int max=-1;
    for(int i=0;i<ASize;i++){
        max=max>A[i]?max:A[i];
        array[i]=max;
    }
    int* dp=(int*)malloc(ASize*sizeof(int));
    dp[0]=1;
    for(int i=1;i<ASize;i++){
        dp[i]=A[i]<array[dp[i-1]-1]?i+1:dp[i-1];
    }
    free(array);
    return dp[ASize-1];
}
