int* decode(int* encoded, int encodedSize, int first, int* returnSize){
    int *test = malloc(sizeof(int) * (encodedSize + 1));
    int i, count = 0;
    test[count++] = first;
    for(i = 0 ; i < encodedSize ; i++)
    {
        first = first ^ encoded[i];
        test[count++] = first;
    }
    *returnSize = count;
    return test;
}
