class Solution:
    def sortSentence(self, s: str) -> str:
        n = len(s)
        data = {}
        
        for word in s.split(" "):
            print ("word = ", word)
            digit = int(word[-1])
            data[digit] = word[:-1]
            
        print ("data = ", data)
        
        res = ""
        for i in range(1, 10):
            if (i not in data ):
                break
                
            res += data[i] + " "
            
        return res[:-1]
