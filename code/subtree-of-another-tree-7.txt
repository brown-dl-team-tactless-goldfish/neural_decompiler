/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     struct TreeNode *left;
 *     struct TreeNode *right;
 * };
 */
bool compare(struct TreeNode* s, struct TreeNode* t)
{
    if(s==NULL&&t==NULL){return true;}
    else if((s==NULL&&t!=NULL)||(s!=NULL&&t==NULL)){return false;}
    else if(s->val!=t->val){return false;}
    
    bool result=true;
    result=compare(s->left,t->left);
    if(!result){return false;}
    result=compare(s->right,t->right);
    return result;  
}
bool isSubtree(struct TreeNode* s, struct TreeNode* t) {
    if(s==NULL){return false;}
    bool result=false;
    if(s->val==t->val){result=compare(s,t);}
    if(result){return result;}
    result=isSubtree(s->left,t);
    if(result){return result;}
    result=isSubtree(s->right,t);
    return result;
}
