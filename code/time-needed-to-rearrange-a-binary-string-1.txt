int secondsToRemoveOccurrences(char * s){
    int i, len = strlen(s), flag = 1, times = 0;

    if (len == 1)
        return 0;

    while (flag) {
        flag = 0;
        for (i = 0; i < len - 1; i++) {
            if (s[i] == '0' && s[i + 1] == '1') {
                flag = 1;
                char temp = s[i];
                s[i] = s[i + 1];
                s[i + 1] = temp;
                i++;
            }
        }

        times++;
    }

    return times - 1;
}
