    int tupleSameProduct(vector<int>& nums) {
        //abcd , bacd, abdc, badc  and reverse. so for each {a,b,c,d} there are 8 tuples.
        unordered_map<int,int> cnt;
        for(int i=0;i<nums.size();i++)for(int j=0;j<i;j++) cnt[nums[i]*nums[j]]++;
        int res = 0;
        for(auto [prod, n] : cnt) res += n*(n-1)/2;
        return res * 8;
    }
