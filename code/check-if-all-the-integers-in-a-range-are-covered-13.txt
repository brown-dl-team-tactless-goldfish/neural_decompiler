    bool isCovered(vector<vector<int>>& ranges, int left, int right) {
        long long int i,x,y,j;
        set<int>s;
        for(i=0;i<ranges.size();i++)
        {
            for(j=ranges[i][0];j<=ranges[i][1];j++)
            {
                s.insert(j);
            }
        }
        for(i=left;i<=right;i++)
        {
            if(s.find(i)==s.end())
            {
                return false;
            }
        }
        return true;
    }
};
