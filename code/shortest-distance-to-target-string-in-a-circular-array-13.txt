class Solution {
public:
    int closetTarget(const vector<string>& words, const string& target, int startIndex) {
        if(words[startIndex]==target) return 0;

        int len=words.size(), dist=len, tmp;
        for(int i=0; i<len; ++i)
            if(words[i]==target) {
                tmp=abs(startIndex-i);
                dist=min(dist,min(tmp,len-tmp));
            }
        
        return dist==len?-1:dist;
    }
};
