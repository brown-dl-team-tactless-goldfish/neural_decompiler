bool hasPathSum(struct TreeNode* root, int sum)
{
    if (root == NULL) return false;
    
    if (root->left == NULL && root->right == NULL && sum-root->val == 0)
    {
        return true;
    }
    
    bool left,right;
    
    left=hasPathSum(root->left,sum-root->val);
        
    right=hasPathSum(root->right,sum-root->val);
    
    return left || right;
}
