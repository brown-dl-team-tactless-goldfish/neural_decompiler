impl Solution {
    pub fn repeated_substring_pattern(s: String) -> bool {
        for i in 1..(s.len()/2 + 1) { // the most it could repeat is 1/2 the string
            let slice = &s[0..i];
            // treat look aheads as a problem of increasing lengths
            let mut is_match = true;
            for j in (i..s.len()).step_by(slice.len()) {
                let upper = if j+slice.len() > s.len() { s.len() } else { j+slice.len() };
                let test = &s[j..upper];
                if test != slice {
                    is_match = false;
                    break;
                }
            }
            if is_match {
                return true;
            }
        }
        return false;
    }
}
