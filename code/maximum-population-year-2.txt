

int maximumPopulation(int** logs, int logsSize, int* logsColSize){
    int pop[101] = {0},maxPop = 0,maxYear = 0;
    int i,m,n,j;
    for(i = 0;i < logsSize;i++){
        m = logs[i][0] - 1950;
        n = logs[i][1] - 1950;
        for(j = m;j < n;j++){
            pop[j]++;
        }
    }
    for(i = 0;i < 101;i++){
        printf("%d - %d\n",i,pop[i]);
        if(pop[i] > maxPop){
            maxPop = pop[i];
            maxYear = i+1950;
        }
    }
    return maxYear;
}
