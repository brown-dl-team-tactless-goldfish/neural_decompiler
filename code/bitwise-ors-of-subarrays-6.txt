class Solution {
public:
    int subarrayBitwiseORs(vector<int>& arr) {
        unordered_set<int>ret, cur ;
        for(auto i : arr)
        {
            unordered_set<int>tmp = {i} ;
            for(auto j : cur)
                tmp.insert(i | j ) ;
            cur = tmp ;
            for(auto k : cur)
                ret.insert(k) ;
        }
        
        return ret.size() ;
    }
};
