    unordered_map<int,int> mp;
    
    int mask = 0, ans = 0, val;
    
    mp[0] = -1;
    
    for(int i=0; i<s.size(); i++)
    {
        val = 0;
         
        if(s[i] == 'a' || s[i] == 'e' || s[i] == 'i' || s[i] == 'o' ||
          s[i] == 'u')
        {
            val = s[i];
        }
        
        mask = mask ^ val;
        
        if(!mp.count(mask))
        {
            mp[mask] = i;
        }
        
        ans = max(ans, i-mp[mask]);
    }
    
    return ans;
    
}
