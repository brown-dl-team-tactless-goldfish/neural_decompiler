class Solution {
public:
    bool containsPattern(vector<int>& arr, int m, int k) {
        for (auto first1 = arr.begin(); !(arr.end() - first1 < m * k); ++first1) {
            auto t = k - 1;
            for (auto first2 = first1 + m; t && std::equal(first1, first1 + m, first2, first2 + m); first2 += m, --t)
                ;
            if (0 == t)
                return true;
        }
        return false;
    }
};
