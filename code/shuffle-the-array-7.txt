/**
 * Note: The returned array must be malloced, assume caller calls free().
 */
int* shuffle(int* nums, int numsSize, int n, int* returnSize){
    
    *returnSize = numsSize;
    
    int *result = (int*)malloc(numsSize*sizeof(int));

    for(int i=0, j=0; j<numsSize; j++){
        if(j%2==0){
            result[j] = nums[i];
            i++;
        }
        else
            result[j] = nums[i+n-1];
    }
    return result;    
}
