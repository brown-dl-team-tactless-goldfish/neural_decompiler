class Solution {
public:
    double findMaxAverage(vector<int>& nums, int k) {
        int maxx = INT_MAX, curr = 0;
        for(int i = 0; i < k; i++)
            curr += nums[i];
        maxx = curr;
        for(int i = 0; i < nums.size() - k; i++) {
            curr += nums[i + k];
            curr -= nums[i];
            maxx = max(maxx, curr);
        }
        return (double) maxx / k; 
    }
};
