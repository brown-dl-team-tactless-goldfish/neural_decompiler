char* mergeAlternately(char* word1, char* word2) {
    int len1 = strlen(word1), len2 = strlen(word2), arrSize = len1 + len2 + 1;
    int j = 0, k = 0;
    char* arr = (char*) malloc(arrSize * sizeof(char));
    for (int i = 0; i < arrSize - 1; i++) {
        if (i % 2 == 0) {
            arr[i] = (j < len1) ? word1[j++] : word2[k++];
        } else {
            arr[i] = (k < len2) ? word2[k++] : word1[j++];;
        }
    }
    arr[arrSize-1] = '\0';
    return arr;
}
