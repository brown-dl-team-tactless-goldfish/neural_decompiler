class Solution {
public:
    int minNumberOperations(vector<int>& target) {
        int ret = target[0];
        int last = target[0];
        for(int i=1; i<target.size(); i++)
        {
            if(target[i]>last)
                ret+=(target[i]-last);
            last = target[i];
        }
        return ret;
    }
};
