int minSwaps(string s) {
	int l = s.length(), res0 = 0, res1 = 0;
	int o = count(s.begin(), s.end(), '1'), z = l - o;
	bool cur = true;
	for (char c : s) {
		if (cur)
			(c == '0') ? res1++ : res0++;
		else
			(c == '1') ? res1++ : res0++;
		cur = !cur;
	}
	res0 = (res0 + 1) / 2;
	res1 = (res1 + 1) / 2;
	return abs(o - z) <= 1 ? (o > z ? res1 : (z > o ? res0 : min(res1, res0))) : -1;
}
