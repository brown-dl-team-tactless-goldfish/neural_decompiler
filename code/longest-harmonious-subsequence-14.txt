class Solution {
public:
    int findLHS(vector<int>& nums) {
        unordered_map<int, int> cnt;
        for (auto n : nums) {
            cnt[n] += 1;
        }
        int retVal = 0;
        for (auto v : cnt) {
            if (cnt.find(v.first+1) != cnt.end()) {
                retVal = max(retVal, cnt[v.first+1]+v.second);
            }
        }
        return retVal;
    }
};
