int sumEvenGrandparent(struct TreeNode* root){
    
    if (root == NULL) return 0;
    
    int value = 0;
    
    // check if this node is even valued:
    if (root->val % 2 == 0)
    {
        // check left child:
        if (root->left != NULL)
        {
            // check left grandchild:
            if (root->left->left != NULL)
                value += root->left->left->val;
            
            // check right grandchild:
            if (root->left->right != NULL)
                value += root->left->right->val;
        }
        
        // check right child:
        if (root->right != NULL)
        {
            // check left grandchild:
            if (root->right->left != NULL)
                value += root->right->left->val;
            
            // check right grandchild:
            if (root->right->right != NULL)
                value += root->right->right->val;
        }
    }
    
    return value + sumEvenGrandparent(root->left) + sumEvenGrandparent(root->right);
}
