class Solution {
public:vector<int> ans;
    void help(TreeNode* root,int i){
        if(root==NULL)
            return;       
        if(ans.size()==i)
            ans.push_back(root->val);       
        help(root->left, i+1);
        help(root->right, i+1);
      
    }
    
    int findBottomLeftValue(TreeNode* root) {
        help(root,0);
        int n=ans.size();
         return ans[n-1]; 
    }
};
