int pivotInteger(int n){
    if (n == 1)
        return 1;
    int ltr[1002] = {[0 ... 1001] = 0};
    int rtl[1002] = {[0 ... 1001] = 0};
    for (int i = 1; i < n; i++)
        ltr[i] = ltr[i - 1] + i;
    for (int i = n; i > -1; i--)
        rtl[i] = rtl[i + 1] + i;
    for (int i = 1; i < n; i++)
        if (ltr[i] == rtl[i])
            return i;
    return -1;
}
