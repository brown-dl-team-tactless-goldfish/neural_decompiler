bool isOdd(int num)
{
    if (!(num%2))
    {
        return false;
    }
    return true;
}

bool threeConsecutiveOdds(int* arr, int arrSize){

    int consecutiveOdd = 0;

    for (int index = 0; index < arrSize; index++)
    {
        if (isOdd(arr[index]))
        {
            consecutiveOdd++;
        }
        else
        {
            consecutiveOdd = 0;
        }

        if (consecutiveOdd >= 3)
        {
            return true;
        }
    }

    return false;
}
