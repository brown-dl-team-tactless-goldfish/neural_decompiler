bool isPerfectSquare(int num)
{
    int i=1;
    int q;
  
    if ( 1 == num )
    {
        return 1;
    }
    else
    {
        q = num/i;
        while ( ( 0 != i-q ) || ( 1 != i-q ) || ( 1 != q-i ) )
        {
            if ( 0 == i-q )
            {
                if ( 0 == (num%i) )
                {
                    return 1;
                }
                else
                {
                    return 0;
                }
            }

            if ( 1 == i-q )
            {
                return 0;
            }

            if ( 1 == q-i )
            {
                return 0;
            }
            i++;
            q = num/i;
        }
    }
    return 0;
}
