    long long repairCars(vector<int>& ranks, int cars) {
        auto able = [&](long long t, long long sum = 0){
            for(auto r:ranks) sum += (long long)sqrt(t/r);
            return sum >= cars;
        };
        long long l = 0, r = 1e16;
        while(l+1<r){
            auto mid = (l+r)/2;
            able(mid) ? r=mid : l=mid;
        }
        return r;
    }
