bool judgeCircle(char * moves){
    int x, y;
    char c;
    
    x = y = 0;
    
    while ((c = *moves++)) {
        
        if (c == 'U') {
            y++;
        } else if (c == 'D') {
            y--;
        } else if (c == 'L') {
            x--;
        } else {
            x++;
        }
        
    }
    
    return x == 0 && y == 0;
}
