int maxProduct(vector<int>& nums) {
    int n=nums.size();
    int max_prodt=nums[0];
    int max_end=nums[0];
    int min_end=nums[0];
    
    for(int i=1;i<n;i++){
       
        int temp=max_end;
        max_end=max({nums[i],nums[i]*max_end,nums[i]*min_end});
        min_end=min({nums[i],nums[i]*temp,nums[i]*min_end});
       
        max_prodt=max(max_prodt,max_end);
    }
    
    return max_prodt;
}
