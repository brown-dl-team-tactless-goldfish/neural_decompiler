class Solution {
public:
    int minNumberOfHours(int initialEnergy, int initialExperience, vector<int>& energy, vector<int>& experience) {
        int ans = 0, n = size(energy), exp = initialExperience, ener = initialEnergy;
        for (int i = 0; i < n; i++) {
            if (exp <= experience[i]) ans += experience[i] - exp + 1, exp = experience[i]+1;
            if (ener <= energy[i]) ans += energy[i] - ener + 1, ener = energy[i] + 1;
            exp += experience[i], ener -= energy[i];
        }
        return ans;
    }
};
