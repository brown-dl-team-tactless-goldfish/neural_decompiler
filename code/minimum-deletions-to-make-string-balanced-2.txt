class Solution {
public:
    int minimumDeletions(string s) {
        int a = 0, b = 0; 
        
        for (const auto& c : s) {
            if (c == 'a') {
                a += 1; 
            } else {
                b = max(a + 1, b + 1);
            }
        }
        
        return s.size() - max(a, b);
    }
};
