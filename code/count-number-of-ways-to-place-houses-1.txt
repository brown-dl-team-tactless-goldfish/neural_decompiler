class Solution {
public:
    int countHousePlacements(int n) {
            int m = 1e9+7;
        if(n == 1)
            return 4;
        if(n == 2)
            return 9;
        long long int a = 2;
        long long int b = 3;
        int i = 2;
        long long int c;
        while(i<n)
        {
             c = (a%m+b%m)%m;
             a = b;
             b = c;
             i++;
        }
        return ((c%m)*(c%m))%m;
        
    }
};
