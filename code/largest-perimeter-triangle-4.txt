class Solution {
public:
    int largestPerimeter(vector<int>& nums) {
        ios_base::sync_with_stdio(false);cin.tie(0);
        int ans = 0;
        int n = (int)nums.size();
        sort(nums.begin(), nums.end());
        int i = n-1, j = n-2, k = n-3;
        while(k>=0){
            if(nums[i]<nums[j]+nums[k]){
                return nums[i] + nums[j]+nums[k];
            }else{
                --i, --j, --k;
            }
        }
        return 0;
    }
};
