char * removeDuplicates(char * S){
    int len = strlen(S);
    
    if(len <= 0){
        return S;
    }
    
    char* chars = (char *)malloc(len+1);
    int top = -1;
    
    for ( int i = 0; i < len ; i++ ){
        if(top < 0 || chars[top] != S[i] ){
            chars[++top] = S[i];
        }else{
            top--;
        }
    }
    chars[++top] = '\0';
    return chars;
}
