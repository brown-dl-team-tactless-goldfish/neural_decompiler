class Solution {
public:
    string maximumNumber(string num, vector<int>& change) 
    {
        int f=0;
        for(int i=0;i<num.size();i++)
        {
            int val=num[i]-'0';
            int c=change[val];
            if(val<c)
            {
                num[i]=c+'0';
                f=1;
            }
            if(val>c)
            {
                if(f)
                {
                    return num;
                }
            }
        }
        return num;
    }
};
