int numSpecialEquivGroups(vector<string>& A) {
	set<string> sWords;
	for (string w : A) {
		string even = "", odd = "";
		for (int i = 0; i < w.size(); i += 2) {
			even += w[i];
			odd += w[i + 1];
		}
		sort(begin(even), end(even));
		sort(begin(odd), end(odd));
		w = even + odd;
		sWords.insert(w);
	}
	return sWords.size();
}
