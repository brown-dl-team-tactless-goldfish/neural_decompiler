		int c = 0;// second pointer for counting 1 which was changed from 0 ;
		pre[0] = m;
		for (i = 0; i < n; i++)
		{
			if (v[i] == 1)
			{
				c++;
			}
			else m--;
			pre[i + 1] = c + m;  // ith left is c and right is m
		}

		int ans = INT_MAX;
		for (i = 0; i <= n; i++)
		{
			ans = min(ans, pre[i]);
		}

		for (i = 0; i <= n; i++)
		{
			if (ans == pre[i])
			{
				ans = i;
				break;
			}
		}

		return ans;
	}
