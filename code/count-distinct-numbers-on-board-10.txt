class Solution {
public:
    int distinctIntegers(int n) {
        if(n==1) return 1;
        else return n-1;
    }

    // 2nd Approach

    int distinctIntegers(int n) {
        unordered_set<int> st;
        st.insert(n);
        for(int j=1 ; j<=n ; j++){
            for(int i=1 ; i<=n ; i++){
                for(auto it:st){
                    if(it % i == 1) st.insert(i);
                }
            }
        }
        return st.size();
    }
};
