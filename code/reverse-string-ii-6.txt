

void reverseSubStr(char *s,int i,int j)
{
    for(;i<j;i++,j--)
    {
        s[i]^=s[j]^=s[i]^=s[j];
    }
}

char * reverseStr(char * s, int k){
    int i,j,len=strlen(s);
    for(i=0;i<len;i=i+2*k)
    {
        if(i+k > len)
            reverseSubStr(s,i,len-1);
        else
            reverseSubStr(s,i,i+k-1);
    }
    return s;
}
