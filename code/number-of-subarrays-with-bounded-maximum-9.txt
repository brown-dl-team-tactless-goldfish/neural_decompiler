    int window_start=0,window_end=-1, i, count = 0;
    
    for(i=0; i<nums.size(); i++)
    {
        if(nums[i] <= right)
        {
            if(nums[i] >= left)
            {
                window_end = i;
            }
            
            count+= window_end - window_start+1;
        }
        
        else
        {
            window_end = i;
            
            window_start = i+1;
        }
       
    }
    
    return count;
    
}
