    bool winnerSquareGame(int n) {
        vector<int> vSquare;
        for (int i=1;i*i<=n;i++) vSquare.push_back(i*i);
        vector<bool> vdp(n+1, false);
        for (int i=1;i<=n;i++) for (auto x:vSquare){            
            if (x > i) break;
            if (!vdp[i-x]){
                vdp[i] = true;
                break;
            }
        }
        return vdp[n];
    }
