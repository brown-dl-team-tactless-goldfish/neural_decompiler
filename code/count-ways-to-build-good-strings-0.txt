int countGoodStrings(int low, int high, int zero, int one){
    int M = 1e9 + 7 ;
    int* dp = calloc(high+1, sizeof(int)) ;
    long long ret = 0 ;
    dp[0] = 1 ;
    for(int i = 1; i <= high; i++){
        if(i >= one)
            dp[i] = (dp[i] + dp[i-one])%M ;
        if(i >= zero)
            dp[i] = (dp[i] + dp[i-zero])%M ;
        if(i >= low)
            ret = (ret + dp[i]) % M ;
    }
    return ret ;
}
