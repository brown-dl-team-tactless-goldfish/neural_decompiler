class Solution 
{
public:
    int maxScore(vector<int>& nums) 
    {
        sort(nums.rbegin(), nums.rend());

        int ans = 0;
        long long sum = 0;
        for (auto n : nums)    
        {
            sum += n;
            ans += sum > 0;

            if (sum < 0)
            {
                break;
            }
        }

        return ans;
    }
};
