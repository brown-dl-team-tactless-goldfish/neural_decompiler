bool canBeEqual(int* target, int targetSize, int* arr, int arrSize){
    if(targetSize != arrSize) return false;
    
    int i;
    int Hash[1001]={0};
    for(i=0; i<targetSize; i++){
        Hash[target[i]]++;
        Hash[arr[i]]--;
    }
    
    int sum=0;
    for(i=0; i<1001; i++){
        sum += Hash[i]>=0?Hash[i]:-Hash[i];
    }

    return sum?false:true;
}
