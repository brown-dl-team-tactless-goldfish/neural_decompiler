#include<stdio.h>  
// size of matrix
int n ,m;

//display function to print matrix
void display(int arr[n][m]){
  for(int i=0;i<n;i++){    
    for(int j=0;j<m;j++){    
      printf("%d  ",arr[i][j]);    
    }//end of j    

    printf("\n");
  }//end of i 

  printf("\n\n---------------------\n");
}

//matrix function
void reverse_matrix(int arr[n][m]){
  //creating temporary 2D array
  int temp[n][m];

  for(int i=n-1,t = 0;i>=0;i--,t++){    
    for(int j=m-1, s = 0;j>=0;j--,s++){    
      temp[t][s] = arr[i][j];    
    }//end of j    
  }//end of i 

  
  // now copy temp elemetns to our matrix
  for(int i=0;i<n;i++){
    for(int j=0;j<m;j++)
      arr[i][j] = temp[i][j];
  }
}

int main(){ 
  n = 4;  // number of rows
  m = 5; // number of columns
  int arr[4][5]={{6,1,7,4,2},{7,5,2,3,8},{9,9,1,1,5},{1,2,3,4,5}};     
 
  printf("Before reversing matrix \n");
  //traversing 2D array    
  display(arr); 

  
  reverse_matrix(arr);
  printf("After reversing \n");
  display(arr);
  
return 0;
