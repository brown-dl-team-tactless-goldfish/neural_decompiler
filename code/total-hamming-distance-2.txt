int totalHammingDistance(int* nums, int numsSize){
    int *num;
    int result = 0, ones;
    for(int i=0; i < 32; i++){
        num = nums;
        ones = 0;
        while(num != nums + numsSize){
            if(*num & 1) ones++;
            *num >>= 1;
            num++;
        }
        result += ones * (numsSize - ones);
    }
    return result;
}
