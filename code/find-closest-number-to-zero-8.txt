class Solution {
public:
    int findClosestNumber(vector<int>& nums) {
        
        int diff = INT_MAX, ans = INT_MIN;
        for (int i = 0; i < nums.size(); i++) {
            if (abs(nums[i]) <= diff) {
                ans = (abs(nums[i]) == diff) ? max(ans, nums[i]) : nums[i];
                diff = abs(nums[i]);
            }
        }
        
        return ans;
    }
};
