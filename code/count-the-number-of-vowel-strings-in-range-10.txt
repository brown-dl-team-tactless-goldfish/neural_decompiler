public class Solution {
    public int VowelStrings(string[] words, int left, int right) 
    {
        var isVowel = new bool[128];

        foreach(var c in "aeiou") isVowel[c] = true;

        var count = 0;

        for(;left <= right; left++)
        {
            if(isVowel[words[left][0]] && isVowel[words[left][^1]])
                count++;
        }

        return count;
    }
}
