

int hammingDistance(int x, int y){
    
    
    return counter(x^y);    
    

}


int counter(int a){
    
    a=(a&0x55555555)+(0x55555555&(a>>1));
    a=(a&0x33333333)+(0x33333333&(a>>2));
    a=(a&0x0f0f0f0f)+(0x0f0f0f0f&(a>>4));
    a=(a&0x00ff00ff)+(0x00ff00ff&(a>>8));
    a=(a&0x0000ffff)+(0x0000ffff&(a>>16));
    
    
    return a;
    
    
}
