int findDuplicate(int* nums, int numsSize) {
    int Low = 1;
    int High = numsSize;
    int Mid, i, counter;
    while(Low + 1 < High){
        counter = 0;
        Mid = Low + (High - Low) / 2;
        for(i = 0; i < numsSize; i++) if(nums[i] >= Low && nums[i] <= Mid) counter++;
        if (counter > (Mid - Low + 1)) High = Mid;
        else Low = Mid;
    }
    counter = 0;
    for(i = 0; i < numsSize; i++) if(nums[i] == Low) counter++;
    if (counter > 1) return Low;
    return High;
}
