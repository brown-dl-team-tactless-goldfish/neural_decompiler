class Solution:
    def findMinimumTime(self, tasks: List[List[int]]) -> int:
        line = [0]*2001
        for i, (lo, hi, time) in enumerate(sorted(tasks, key=lambda x: x[1])): 
            cnt = sum(line[x] for x in range(lo, hi+1))
            time = max(0, time - cnt)
            for x in range(hi, lo-1, -1): 
                if time and not line[x]: 
                    line[x] = 1
                    time -= 1
        return sum(line)
