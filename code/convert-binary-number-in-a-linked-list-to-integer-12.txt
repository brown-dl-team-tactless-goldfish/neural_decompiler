int getDecimalValue(struct ListNode* head){
    
    
    int count = 1;
    
    struct ListNode *tmp = head;
    
    if (head == NULL)
        return 0;
    
    while (tmp != NULL) {
        
        tmp = tmp->next;
        count = count * 2;
    }

    
    count = count/2;
    
    int ans = 0;
    
    tmp = head;
    
    while (tmp != NULL) {
        
        ans += count * tmp->val;
        count = count/2;
        tmp = tmp->next;
    }
    
    return ans;
    
}
