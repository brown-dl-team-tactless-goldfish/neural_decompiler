bool validateStackSequences(int* pushed, int pushedSize, int* popped, int poppedSize){
    int top=-1, s=0; 
    for(int i=0; i<pushedSize; i++){
        top++;
        pushed[top]=pushed[i];
        while(top>-1){
            if(pushed[top]==popped[s]){
                top--;
                s++;
            }
            else
                break;
        }
    }
    for(int i=s; i<poppedSize; i++){
        if(pushed[top]==popped[i])
            top--;
        else
            return false;
    }
    return true;
}
