
double average(int* salary, int salarySize){
double sum = 0, average = 0;

int i, minimumSalary, maximumSalary;

minimumSalary = salary[0], maximumSalary = salary[0];

for (i = 1; i < salarySize; i++)

{

	if (salary[i] < minimumSalary)

		minimumSalary = salary[i];

	else if (salary[i] > maximumSalary)

		maximumSalary = salary[i];

}

for (i = 0; i < salarySize; i++)

{

	if (salary[i] == minimumSalary || salary[i] == maximumSalary)

		continue;

	else

		sum += salary[i];

}

average = sum / (salarySize - 2);

return average;
}
