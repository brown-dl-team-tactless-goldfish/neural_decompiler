    public int MinOperations(string s) {
        
        char[] c = new char[2] { '0', '1' };
      
        int idx0 = 0;
        int idx1 = 1;
        
        int dif0 = 0;
        int dif1 = 0;
        
        for(int i=0; i < s.Length; i++)
        {
            if (s[i] != c[idx0]) { dif0++; }
            if (s[i] != c[idx1]) { dif1++; }
            
            idx0 ^= 1;
            idx1 ^= 1;
        }
        
        return Math.Min(dif0, dif1);
    }
