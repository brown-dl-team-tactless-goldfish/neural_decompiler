typedef long long ll;

class Solution {
public:
    ll beautifulSubarrays(vector<int>& nums) {
        ll count=0,XOR=0;
        unordered_map<ll,ll>m;
        m[0]=1;
        for(int i=0;i<nums.size();i++) {
            XOR^=nums[i];
            if(m.find(XOR)!=m.end()) {
                count+=m[XOR];
            }
            m[XOR]++;
        }
        return count;
    }
};
