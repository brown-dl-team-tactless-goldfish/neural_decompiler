int partitionString(char * s){
    int output = 1;
    int cur_start = 0;
    int str_size = strlen(s);
    int hash[26];
    int i;

    for (i = 0; i < 26; i++)
        hash[i] = -1;

    for (i = 0; i < str_size; i++) {
        if (hash[s[i] - 'a'] >= cur_start) {
            output++;
            cur_start = i;
        } 
        hash[s[i] - 'a'] = i;

    }

    return output;
}
