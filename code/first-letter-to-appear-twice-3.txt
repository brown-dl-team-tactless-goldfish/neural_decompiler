class Solution:
    def repeatedCharacter(self, s: str) -> str:
        n = len(s)
        maps = {}
        
        for i in range(n):
            c = s[i]
#             print ("maps = ", maps)
#             print ("c = ", c)
            
            if (c not in maps):
                # print ("in if")
                maps[c] = 1
            else:
                # print ("now return", c)
                return c
