char * removeDuplicates(char * s){
    int top = -1;
    char temp[strlen(s)];
    temp[++top] = s[0];
    temp[top+1] = '\0';
    for(int i = 1; s[i] != '\0'; ++i){
        if(top != -1 && s[i] == temp[top]){
            temp[top--] = '\0';
        } else {
            temp[++top] = s[i];
            temp[top+1] = '\0';
        }
    }
    strcpy(s,temp);
    return s;
}
