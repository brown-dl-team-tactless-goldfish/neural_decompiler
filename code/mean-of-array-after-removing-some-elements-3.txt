int cmp(const void *x, const void *y){
    return (*(int*)x-*(int*)y);
}
double trimMean(int* arr, int arrSize){
int numfor5percent,i,k=0;
double sum=0;
qsort(arr, arrSize, sizeof(int), cmp); // sorting the array
numfor5percent=(arrSize*5)/100;
    for(i=numfor5percent;i<(arrSize-numfor5percent);i++){ // erasing 5% elements from the start and from the end of the arrray 
      sum+=arr[i];
      k++;
    }
    sum/=k;
    return sum;
}
