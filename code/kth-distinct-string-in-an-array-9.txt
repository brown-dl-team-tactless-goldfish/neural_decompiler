public class Solution {
    public string KthDistinct(string[] arr, int k) {
        
        if(arr == null || arr.Length < k)
            return "";
        
        Dictionary<string,int> dic = new Dictionary<string,int>();
        foreach(var str in arr)
            dic[str] = dic.ContainsKey(str)? ++dic[str] : 1;
        
        foreach(var str in arr)
        {
            if(dic[str] == 1 && --k == 0)
            {
                return str;
            } 
        }
        
        return "";
    }
}
