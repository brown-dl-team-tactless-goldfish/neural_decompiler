
class Solution {
public:
    #define ll long long
    const int mod = INT_MAX;
    vector<int> divisibilityArray(string word, int m) {
        
    int n = word.length();
    ll num = 0;
    vector<int> div(n, 0);
    for (int i = 0; i < n; i++) {
        num = (num * 10 + (word[i] - '0')) % m;
        if (num == 0) {
            div[i] = 1;
        }
    }
    return div;
}
};
