bool isSubsequence(char * s, char * t){
	// if s is empty, then automatically true
    if (s[0] == '\0') return true;
    
    for (int t_idx = 0, s_idx = 0; t[t_idx] != '\0'; ++t_idx) {
        if (s[s_idx] == t[t_idx]) {
            if (s[++s_idx] == '\0') return true;
        }
    } 
    return false;
}
