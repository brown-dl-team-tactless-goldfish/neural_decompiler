struct ListNode* mergeInBetween(struct ListNode* list1, int a, int b, struct ListNode* list2){

    struct ListNode* temp=list1,*p=list1;
    for(int i=0;i<b;i++)
    {
        if(i<a-1)
            temp=temp->next;
        p=p->next;
    }
    temp->next=list2;
    while(list2->next!=NULL)
    {
        list2=list2->next;
    }
    list2->next=p->next;
    return list1;
}
