class Solution {
public:
    bool isReachable(int tx, int ty) {
        return canReach(tx,ty);
    }
    private:
    bool canReach(int targetX, int targetY) {
        while(targetX%2 == 0)
        {
            targetX/= 2;
        }
        while(targetY%2 == 0)
        {
            targetY/= 2;
        }
        if (targetX < 1 || targetY < 1) return false;
        return __gcd(targetX, targetY) == 1;
    }

};
