

int findLengthOfLCIS(int* nums, int numsSize){
    if(numsSize==0 )
        return 0;
    if(numsSize==1)
        return 1;
    
    int cnt=1,mc=1;
    for(int i=0;i<numsSize-1;i++){
        if(nums[i]<nums[i+1]){
            cnt++;
            if(cnt>mc)
                mc=cnt;
        }
        else
            cnt=1;            
    }
    return mc;
}
