bool canBeIncreasing(int* nums, int numsSize){
    int count=0;
     int index=-1;
    for(int i=1;i<numsSize;i++){
        if(nums[i-1]>= nums[i]){
            count++;
            index=i;
        }
    }
    if(count>1)
        return false;
    if(count==0)
        return true;
    if(index== 1|| index==numsSize-1)
        return true;
   
    if (nums[index - 1] < nums[index + 1])
        return true;
   
    if(nums[index-2]<nums[index])
        return true;
    
    return false;
