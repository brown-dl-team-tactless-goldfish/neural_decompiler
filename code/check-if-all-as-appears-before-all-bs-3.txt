bool checkString(char * s)
{
        int flag = 0;
        int i = 0;
        while( i < strlen(s))
        {   
            
          while ( s[i] != 'a' && i < strlen(s))
            {
                flag = 1;
                i++;
            }
            
          if ( flag == 1 && i < strlen(s) )
            {
                return false;
            }
            
           i++;
        }
    
    return true;

}
