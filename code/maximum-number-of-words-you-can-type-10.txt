int canBeTypedWords(string text, string brokenLetters) {
	int res = 0;
	stringstream ss(text);
	string word;
	while (ss >> word) {
		bool bCanType = true;
		for (const char& c : brokenLetters) {
			if (word.find(c) != string::npos) {
				bCanType = false;
				break;
			}
		}
		res += bCanType & 1;
	}
	return res;
}
