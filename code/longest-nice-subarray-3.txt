    while(i<n and j<n){

        // checking if window is as per requirement
        while(!check(nums[j],vis) and i<j){
            clear(nums[i],vis);
            i++;
        }    

        // filling window
        if(check(nums[j],vis)){
            fill(nums[j],vis);
            j++;
            maxlen = max(maxlen,j-i);
        }
    }
    
    return maxlen;
}
