class Solution {
public:
    int r , c ;
    vector<vector<int>> restoreMatrix(vector<int>& rowSum, vector<int>& colSum) {
        r = rowSum.size() , c = colSum.size() ;
        vector<vector<int>>grid(r,vector<int>(c,0)) ;
        
        for(int i = 0 ; i < r ;++i ){
            for(int j = 0 ; j < c ; ++j ){
                int element = min(rowSum[i],colSum[j]) ;
                grid[i][j] = element ;
                rowSum[i] -= element ; colSum[j] -= element; 
            }
        }
        return grid ;
    }
};
