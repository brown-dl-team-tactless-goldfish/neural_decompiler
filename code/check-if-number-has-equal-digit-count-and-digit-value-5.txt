bool digitCount(char * num){
    int digitCounter[10] = { 0 };
    
    int i;
    for (i = 0; num[i] != '\0'; i++) {
        digitCounter[num[i] - '0']++;
    }
	
    for (i = 0; num[i] != '\0'; i++) {
        if (num[i] - '0' != digitCounter[i]) {
            return false;
        }
    }
    
    return true;
}
