//Example 1:
//  chips = [1,2,3]
//  =>
//      1 2 3
//      _ _ _
//  pos 1 2 3

//Example 2:
//  chips = [2,2,2,3,3]
//  =>
//        1
//        2 4
//        3 5
//      _ _ _ _ _
//  pos 1 2 3 4 5

int minCostToMoveChips(int* chips, int chipsSize){
    int odd = 0;
    int even = 0;
    for(int i=0; i<chipsSize; i++){
        if(chips[i]%2) odd++;
        else even++;
    }
    return odd<even?odd:even;
}
