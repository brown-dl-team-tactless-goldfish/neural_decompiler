int busyStudent(int* startTime, int startTimeSize, int* endTime, int endTimeSize, int queryTime){
    int index, busyStudentCount = 0;
    for(index = 0; index < endTimeSize; index++)
        busyStudentCount = (startTime[index] <= queryTime && queryTime <= endTime[index]) ? ++busyStudentCount : busyStudentCount;
    return busyStudentCount;
}
