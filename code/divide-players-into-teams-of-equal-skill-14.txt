class Solution {
public:
    long long dividePlayers(vector<int>& a) {
        sort(begin(a), end(a));
        long long i = 0, n = a.size(), res = 0, power = a[i] + a[--n];
        while(i < n){
            if(a[i] + a[n] != power) return -1;
            res += (a[i++] * a[n--]);
        }
        return res;
    }
};
