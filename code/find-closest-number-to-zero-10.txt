public class Solution {
    public int FindClosestNumber(int[] nums) {
        int res = Int32.MaxValue;
        
        foreach (var n in nums)
        {
            int cur = Math.Abs(res) >= Math.Abs(n) ? n : res;
            res = Math.Abs(res) == Math.Abs(cur) ? Math.Max(res, cur) : cur;
        }
        
        return res;
    }
}
