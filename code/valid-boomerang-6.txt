class Solution:
    def isBoomerang(self, points: List[List[int]]) -> bool:
        points.sort()
        p1, p2, p3 = points[0], points[1], points[2]
        
        if (p1 == p2 or p1 == p3):
            return False
        
        if (p2[0] - p1[0] == 0):
            return p3[0] != p1[0]
        else:
            return (p2[1] - p1[1])/(p2[0] - p1[0]) != (p3[1] - p1[1])/(p3[0] - p1[0])
