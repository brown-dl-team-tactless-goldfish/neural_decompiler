struct ListNode* oddEvenList(struct ListNode* head){
	struct ListNode odd_head, even_head;
	struct ListNode *odd = &odd_head, *even = &even_head;

	odd_head.next = NULL;
	even_head.next = NULL;

	int count = 1;
	while(head != NULL) {
		if(count % 2 == 0) {
			even->next = head;
			even = even->next;
		}else {
			odd->next = head;
			odd = odd->next;
		}
		count++;
		head = head->next;
	}
	odd->next = even_head.next;
	even->next = NULL;
	return odd_head.next;
}
