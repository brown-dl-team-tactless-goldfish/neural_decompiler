class Solution {
public:
    string convert(int ind,string nums){
        return nums.erase(ind,1);
    }
    
    string removeDigit(string nums, char digit) {
        string res;
        for(int i = 0;i < nums.size();i++){
            if(nums[i] == digit)
                res = max(res,convert(i,nums));
        }
        return res;
    }
};
