class Solution {
public:
    bool reachingPoints(int sx, int sy, int tx, int ty) {
        while (sx < tx || sy < ty) {
            if (tx > ty) {
                int k = (tx - sx)/ty; 
                if (k == 0) break; 
                tx -= k * ty; 
            } else {
                int k = (ty - sy)/tx; 
                if (k == 0) break; 
                ty -= k * tx; 
            }
        }
        return sx == tx && sy == ty; 
    }
};
