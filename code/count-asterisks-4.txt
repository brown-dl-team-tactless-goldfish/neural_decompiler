class Solution:
    def countAsterisks(self, s: str) -> int:
        res = 0
        n = len(s)
        
        num_bar = 0
        
        for c in s:
            if (c == "|"):
                num_bar += 1
                num_bar = num_bar % 2
                
            elif (c == "*"):
                if (num_bar == 0):
                    res += 1
        
        return res
