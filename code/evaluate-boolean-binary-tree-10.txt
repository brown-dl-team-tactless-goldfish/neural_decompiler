class Solution {
public:
    bool evaluateTree(TreeNode* root) {
        if(!root) return true;
        if(!root->left and !root->right) return root->val;
        bool left = evaluateTree(root->left);
        bool right = evaluateTree(root->right);
        if(root->val == 2) return left or right;
        return left and right;
    }
};
