int maximumDifference(int* nums, int numsSize){
    int i = 0;
    int diff = 0;
    int j = numsSize - 1;
    int maxNum = INT_MIN;

    for(int i = 0; i < numsSize; i++) {
        for(int j = i + 1; j < numsSize; j++) {
            if(i == j) continue;
           if(nums[i] < nums[j])  {
               diff = nums[j] - nums[i];
               maxNum = diff > maxNum ? diff : maxNum;
           }
        }
    }

    maxNum = maxNum == INT_MIN ? -1 : maxNum;
    return maxNum;
}
