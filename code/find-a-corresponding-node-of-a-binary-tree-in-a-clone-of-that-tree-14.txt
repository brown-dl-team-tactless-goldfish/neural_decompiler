class Solution {
public:
    TreeNode* getTargetCopy(TreeNode* original, TreeNode* cloned, TreeNode* target) {
        queue<TreeNode *> org, cln;
        org.push(original); cln.push(cloned);
        while(org.size()){
            original=org.front(); org.pop();
            cloned=cln.front(); cln.pop();
            if(original==target) return cloned;
            if(original->left) org.push(original->left);
            if(cloned->left) cln.push(cloned->left);
            if(original->right) org.push(original->right);
            if(cloned->right) cln.push(cloned->right);
        }
        return NULL;
    }
};
