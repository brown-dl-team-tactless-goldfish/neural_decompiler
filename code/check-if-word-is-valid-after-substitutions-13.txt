class Solution {
public:
    bool isValid(string S) 
    {
        vector<char> v;
        v.reserve(S.size());
        for (char c : S) {            
            v.push_back(c);
            const auto sz = v.size();
            if (sz >= 3 && v[sz-3] == 'a' && v[sz-2] == 'b' && v[sz-1] == 'c')
                v.resize(sz-3);
        }
        return v.empty();
    }
};
