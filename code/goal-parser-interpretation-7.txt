char * interpret(char * command){

	char *ans=malloc(100*sizeof(char));
	int i,j=0,k=0,len=strlen(command);
	for(i=0;i<len;i++)
	{
		if(command[i]=='(')
		{
			k=0;
		}
		else if(command[i]==')')
		{
			if(k==0)
			{
				ans[j++]='o';
			}
		}
		else if(command[i]=='a' || command[i]=='l')
		{
			ans[j++]=command[i];
			k++;
		}
		else if(command[i]=='G')
		{
			ans[j++]=command[i];            
		}
	}
	ans[j]='\0';
	return ans;

}
