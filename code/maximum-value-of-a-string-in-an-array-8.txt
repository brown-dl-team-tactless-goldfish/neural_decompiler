class Solution {
public:
    int maximumValue(vector<string>& strs) {
        int res = 0;
        for (const auto& s : strs) {
            int cur = 0;
            for (const auto& c : s) {
                if (isalpha(c)) {
                    cur = s.size();
                    break;
                }
                cur = cur * 10 + (c - '0');
            }
            res = max(res, cur);
        }
        return res;
    }
};
