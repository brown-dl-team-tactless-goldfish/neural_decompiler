int maxScore(char * s){
    int i,score=0,ls,rs=0;
    int n=strlen(s);
    ls=(s[0]=='0')?1:0;
    for(i=1;i<n;i++){
        if(s[i]=='1'){
            rs+=1;
        }
    }
    i=1;
    score=ls+rs;
    while(i!=n-1){
        if(s[i]=='0'){
            ls+=1;
        }
        else if(s[i]=='1'){
            rs-=1;
        }
        if(ls+rs>score){
            score=ls+rs;
        }
        i+=1;
    }
    return score;
}
