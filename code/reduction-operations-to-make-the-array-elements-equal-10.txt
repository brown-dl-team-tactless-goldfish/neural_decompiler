class Solution {
public:
    using VI = vector<int>;
    using Map = map<int, int>;
    int reductionOperations(VI& A, Map m = {}, int cnt = 0) {
        for (auto x: A)
            ++m[-x];
        while (1 < m.size()) {
            ++(next(m.begin()))->second;
            if (!--m.begin()->second)
                m.erase(m.begin());
            ++cnt;
        }
        return cnt;
    }
};
