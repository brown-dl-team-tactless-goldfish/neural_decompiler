class Solution {
public:
    bool winnerOfGame(string colors) {
        int cnt[128] = {};
        int pre = 0, len = 0;
        for(auto c:colors){
            c == pre ? len++ : len=1;
            if(len>2) cnt[c]++;
            pre = c;
        }
        return cnt['A'] > 0 && cnt['A'] > cnt['B'];
    }
};
