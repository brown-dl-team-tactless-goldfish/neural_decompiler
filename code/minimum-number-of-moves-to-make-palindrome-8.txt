class Solution {
public:
    int minMovesToMakePalindrome(string s) {
        int ret = 0 ;
        while(s.size() > 1){
            int i = s.find(s.back()) ;
            //the string in the middle
            if(i == s.size() - 1){
                ret += s.size()/2 ;
            }
            else{
                ret += i ;
                s.erase(i, 1) ;
            }
            s.pop_back() ;
        }
        return ret ;
    }
};
