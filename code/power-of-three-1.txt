bool isPowerOfThree(int n){
    switch (n % 10) {
        case 1:
        case 3:
        case 7:
        case 9:
            break;            // 3^n last digit is one of 1,3,7,9
        default:
            return false;
    }
    
    while (n > 1) {
        if (n % 3)
            return false;
        n /= 3;
    }
    
    return n == 1? true: false;
}
