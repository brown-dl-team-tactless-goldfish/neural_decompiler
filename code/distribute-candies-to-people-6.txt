int* distributeCandies(int candies, int num_people, int* returnSize){
    int idx = 0, pre = 0;
    int * result = (int *)calloc(num_people, sizeof(int));
    while(candies > 0)
    {
        result[idx] = pre + 1 < candies ? result[idx] + pre + 1 : result[idx] + candies;
        pre++;
        candies -= pre;
        idx++;
        if(idx == num_people)
            idx = 0;
    }
    *returnSize = num_people;
    return result;
}
