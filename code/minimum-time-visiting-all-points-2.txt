int minTimeToVisitAllPoints(int** points, int pointsSize, int* pointsColSize)
{
    int ans=0;
    
	// will iterate from last point to second point
    while(pointsSize---1)
        ans += fmax
        (
			// for x-coordinate
            abs(points[pointsSize][0]-points[pointsSize-1][0]),
			// for y-coordinate
            abs(points[pointsSize][1]-points[pointsSize-1][1])
        );
    
    return ans;
}
