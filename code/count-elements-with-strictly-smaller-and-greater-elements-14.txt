    int countElements(vector<int>& nums) {
        int n = nums.size();
        if(n<=2) return 0;
        int s = nums[0] , e = nums[0];
        for(auto x: nums){
            s = min(x,s);
            e = max(x,e);
        }
        
        int count = 0;
        
        if(s == e) return 0;
        
        else{
            for(auto it : nums){
                if(it != s and it != e){
                    count++;
                }
            }
        }
        return count;
    }
