bool evaluateTree(struct TreeNode* root){
    if(!root->left && !root->right)
    {
        return root->val;
    }
    if(root->val == 2)
    {
        if(evaluateTree(root->left) || evaluateTree(root->right))
        {
            return true;
        }
    }
    else if(root->val == 3)
    {
        if(evaluateTree(root->left) && evaluateTree(root->right))
        {
            return true;
        }
    }
    return false;
}
