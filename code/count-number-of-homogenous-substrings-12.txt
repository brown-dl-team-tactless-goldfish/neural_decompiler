class Solution {
public:
    int mod = 1e9 + 7;
    int countHomogenous(string s) {
        long long ans = 0 , len = 0 , n = s.length();
        for(int i = 0 ; i < n ; i++){
            if(i==0){
                len++;
            }else if(s[i]==s[i-1]){
                len++;
            }else{
                ans += (((len%mod)*(len+1)%mod)%mod)/2;
                len = 1;
            }
        }
        ans += (((len%mod)*(len+1)%mod)%mod)/2;
        return ans;
        
    }
};
