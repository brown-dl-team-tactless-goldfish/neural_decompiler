bool isUgly(int num) {
    int i,nums[3] = {2,3,5};
    bool flag;
    if(num < 1)
        return false;
    while(num > 1)
    {
        flag = false;
        for(i = 0; i < 3; i++)
            if(num % nums[i] == 0)
            {
                flag = true;
                num /= nums[i];
                break;
            }
        if(flag == false)
            return false;
    }
    return true;
}
