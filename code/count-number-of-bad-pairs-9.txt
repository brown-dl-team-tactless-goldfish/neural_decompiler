class Solution {
public:
    long long countBadPairs(vector<int>& nums) {
        unordered_map<long long, long long> maps;
        long long n = nums.size();
        for (int i = 0; i < n; i++) {
            maps[i - nums[i]]++;
        }
        long long answer = n * (n - 1) / 2;
        for (auto [key, value] : maps) {
            long long res = value * (value - 1) / 2;
            answer -= res;
        }
        return answer;
    }
};
