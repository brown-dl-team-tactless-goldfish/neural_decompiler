class Solution {
public:
    const int mod=1e9+7;
    int countHomogenous(string s) 
    {
        int count=1;
        int ans=0;
        int n=s.size();
        for(int i=0;i<n;i++)
        {
            if(i>0&&s[i-1]==s[i])
            {
                count++;
            }
            else
            {
                count=1;
            }
            ans=(ans+count)%mod;
        }
        return ans%mod;
    }
};
