class Solution {
public:
    long long beautifulSubarrays(vector<int>& nums) 
    {
        long long xr=0,ans;
        int n=nums.size();
        map<long long,int> dp{{0,1}};
        for(int i=0;i<n;i++)
        {
            xr^=nums[i];
            ans+=dp[xr]++;
        }
        return ans;
    }
};
