    bool isValid(string s) {
        string abc = "abc";
        unordered_map<char,char> map;
        for(int i = 1; i < abc.length(); i++){
            map[abc[i]] = abc[i - 1];
        }
        if(s.front() != 'a' || s.back() != 'c') return false;
        stack<char> stk;
        for(char c : s){
            if(stk.empty() && c != 'a') return false;
            if(c == 'a'){
                stk.push(c);
                continue;
            }
            if(!stk.empty()){
                auto t = stk.top();
                if(map[c] == t){
                    if(c == 'c'){
                        int k = 2;
                        while(k--) stk.pop();
                        continue;
                    }
                }
                
            }
            stk.push(c);

        }
        return stk.empty();
    }
