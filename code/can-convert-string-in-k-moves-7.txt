class Solution {
public:
    bool canConvertString(const string& s, const string& t, int k) {
        if(s.size() != t.size()) return false;
        int n{k/26}, r{k%26}, size{(int)s.size()};
        std::vector<int> f(26, n);
        f[0] += size;
        for(int i{1}; i <= r; ++i) ++f[i];
        for(int i{0}; i < size; ++i) {
            if(--f[(26 + t[i] - s[i])%26] == -1) return false;
        }
        return true;
        
    }
};
