int compare(void *a, void *b) {
    return *(int *)a - *(int *)b;
}

int findContentChildren(int* g, int gSize, int* s, int sSize){
    qsort(g, gSize, sizeof(g[0]), compare);
    qsort(s, sSize, sizeof(s[0]), compare);
    int n = 0;
    for (int i = 0, j = 0; i < gSize && j < sSize; j++) {
        if (g[i] > s[j]) continue;
        i++;
        n++;
    }
    return n;
}
