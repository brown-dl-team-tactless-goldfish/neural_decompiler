class Solution {
public:
    string getSmallestString(int n, int k) {
        string ans = "";
        char c = 'a';
        
        int req = k;
        for(int i = 1; i <= n; i++) {
            if(req-(c-'a'+1) > 26*(n-i)) {
                c++;
                i--;
            } else {
                ans += c;
                req -= c-'a'+1;
            }
        }
        
        return ans;
    }
};
