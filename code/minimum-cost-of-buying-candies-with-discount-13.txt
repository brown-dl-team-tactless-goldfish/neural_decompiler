// OJ: https://leetcode.com/contest/biweekly-contest-70/problems/minimum-cost-of-buying-candies-with-discount/
// Author: github.com/lzl124631x
// Time: O(NlogN)
// Space: O(1)
class Solution {
public:
    int minimumCost(vector<int>& A) {
        sort(begin(A), end(A), greater<>());
        int ans = 0, N = A.size();
        for (int i = 0; i < N; ++i) {
            ans += A[i++];
            if (i < N) ans += A[i++];
        }
        return ans;
    }
};
