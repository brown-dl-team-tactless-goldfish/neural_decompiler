class Solution {
public:
    bool isValid(string &word)
    {
        int hyphen=0,h_index=0,p=0,p_index=0;
        for(int i=0;i<word.length();i++)
        {
            if(word[i]>='0' && word[i]<='9')
            {
                return false;
            }
            if(word[i]=='-')
            {
                hyphen++;
                h_index=i;
            }
            if(word[i]=='!' || word[i]=='.' || word[i]==',')
            {
                p++;
                p_index=i;
            }
        }
        bool first=(hyphen==0 || (hyphen==1 && h_index<word.length()-1 && h_index>0 && word[h_index-1]>='a' && word[h_index-1]<='z' && word[h_index+1]>='a' && word[h_index+1]<='z'));
        bool second=(p==0 || (p==1 && p_index==word.length()-1));
        return first && second;
    }
    int countValidWords(string sentence) 
    {
        int count=0,i=0;
        while(i<sentence.length())
        {
            string tmp="";
            while(i<sentence.length() && sentence[i]!=' ')
            {
                tmp.push_back(sentence[i++]);
            }
            if(tmp!="" && isValid(tmp))
            {
                count++;
            }
            i++;
        }
        return count;
    }
};
