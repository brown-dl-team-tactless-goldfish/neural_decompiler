class Solution {
public:
    bool isStrictlyPalindromic(int n) {
        for(int i = 2; i <= n - 2; i++){
            string a = inbase(n, i);
            string b = a;
            reverse(b.begin(), b.end());
            if(a != b) return false;
        }
        return true;

    }
    string inbase(int a, int b){
        string res;
        while(a){
            int c = a % b;
            res += to_string(c);
            a /= b;
        }
        return res;
    }
};
