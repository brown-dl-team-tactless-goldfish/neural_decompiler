class Solution {
public:
    vector<int> lexicalOrder(int n) {
        int no=1;
        vector<int> ans(n,0);
        for(int i=0;i<n;i++){
            ans[i]=no;
            if(no*10<=n)
                no*=10;
            else
            {
                if(no>=n)
                    no/=10;
                no+=1;
                while(no%10==0)no/=10;
            }
        }
        return ans;
    }
};
