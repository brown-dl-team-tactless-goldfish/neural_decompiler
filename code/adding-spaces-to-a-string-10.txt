class Solution {
public:
    string addSpaces(string s, vector<int>& spaces) {
        
        string ans;
        
        int count = 0;
        int idx = 0;
        
        for(char c : s){
            
            if(idx < spaces.size() and count == spaces[idx]){
                idx++;
                ans += ' ';
            }
            
            ans += c;
            count++;
        }
        
        return ans;
    }
};
