public class Solution
    {
        public int[] DivisibilityArray(string word, int m)
        {
            long currentWord = 0;
            var res = new int[word.Length];

            var index = 0;
            foreach (var c in word)
            {
                var digit = c - '0';
                currentWord = (currentWord * 10 + digit) % m;
                res[index] = (currentWord == 0 ? 1 : 0);
                index++;
            }

            return res;
        }
    }
