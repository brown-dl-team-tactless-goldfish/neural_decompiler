int rob(struct TreeNode* root){
    if(root==NULL) return 0;
    if(root->val<0) return -1*root->val;   // if this node was caculated before, it should be a negative number! just return it
    if(root->left && root->right) root->val = -1*fmax(rob(root->left)+rob(root->right), root->val+rob(root->left->left)+rob(root->left->right)+rob(root->right->left)+rob(root->right->right));
    else if(root->left) root->val = -1*fmax(rob(root->left), root->val+rob(root->left->left)+rob(root->left->right));
    else if(root->right) root->val = -1*fmax(rob(root->right), root->val+rob(root->right->left)+rob(root->right->right));
    return fabs(root->val);
}
