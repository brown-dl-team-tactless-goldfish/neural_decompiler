class Solution {
public:
    vector<string> removeAnagrams(vector<string>& words) {
        unordered_map<char, int> last;
        vector<string> ans;
        for (auto& s : words){
            unordered_map<char, int> a;
            for (auto& c : s) ++a[c];
            if (last != a){
                ans.push_back(s);
                last = a;
            }
        }
        return ans;
    }
};
