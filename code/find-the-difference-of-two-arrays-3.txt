    public IList<IList<int>> FindDifference(int[] nums1, int[] nums2) {
        
        int[] intersect = nums1.Intersect(nums2).ToArray();

        IList<IList<int>> rets = new List<IList<int>>();

        rets.Add(nums1.Except(intersect).ToArray());
        rets.Add(nums2.Except(intersect).ToArray());


        return rets;
    }
