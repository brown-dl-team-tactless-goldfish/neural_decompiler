int minDeletionSize(char ** strs, int strsSize)
{
    int i;
    int col = 0;
    char *ptr = strs[0];
    int count = 0;

    while(*ptr != '\0')
    {
        for(i = 1; i < strsSize; i++)
        {
            if(*ptr > *(strs[i] + col) )
            {
                count += 1;
                break;
            }
            ptr = strs[i] + col;
        }
        col += 1;
        ptr = strs[0] + col;
    }

    return count;
}
