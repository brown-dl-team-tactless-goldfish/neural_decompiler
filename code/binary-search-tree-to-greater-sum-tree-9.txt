int myBstToGst(struct TreeNode *root, int offset)
{
    if(NULL == root)  return offset;
    root->val += myBstToGst(root->right, offset);
    return myBstToGst(root->left, root->val);
}


struct TreeNode* bstToGst(struct TreeNode* root){
    if(NULL == root)  return NULL;
    myBstToGst(root, 0);
    return root;
}
