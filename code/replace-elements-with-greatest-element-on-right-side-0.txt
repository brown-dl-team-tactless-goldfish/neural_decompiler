int* replaceElements(int* arr, int arrSize, int* returnSize)
{
	int *r;
	int i, j, k;
	int max;

	r = (int *)malloc(sizeof(int) * arrSize);
	if (r) {
		j = 0;
		for (i = 0; i < arrSize - 1; i++) {
			if (i >= j) {
				j = k = i + 1;
				max = arr[k];
				for (++k; k < arrSize; k++) {
					if (arr[k] > max) {
						max = arr[k];
						j = k;
					}
				}
			}
			r[i] = max;
		}
		r[i] = -1;
		*returnSize = arrSize;
	} else {
		*returnSize = 0;
	}
	return r;
}
