class NumArray {
	private:
	vector<int> cnt;
public:
	NumArray(vector<int>& nums) {
		for(int i=0;i<nums.size();i++){
			cnt.push_back(nums[i]);
		}
	}

	void update(int i, int val) {
		cnt[i]=val;
	}

	int sumRange(int i, int j) {
		int res=0;
		for(int v=i;v<=j;v++){
			res+=cnt[v];
		}
		return res;
	}
};
