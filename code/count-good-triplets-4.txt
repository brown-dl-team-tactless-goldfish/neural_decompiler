int countGoodTriplets(int* arr, int arrSize, int a, int b, int c){
    int counter = 0;
    
    for (int i=0; i<arrSize-2; i++){
        for (int k=i+1; k<arrSize-1; k++){
            for (int h=k+1; h<arrSize; h++){
                if ((abs(arr[i]-arr[k]) <= a) && (abs(arr[k]-arr[h]) <= b) && (abs(arr[i]-arr[h]) <= c))
                    counter++;
            }
        }
    }
    return counter;
}
