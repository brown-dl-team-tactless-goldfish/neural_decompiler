class Solution {
public:
    string largestWordCount(vector<string>& nums, vector<string>& senders) {
        map<string,int> mp;
        for(int i = 0; i<nums.size();++i){
            string &s = nums[i];  int n = 0;
            for(int j = 0; j<s.size();++j) if(s[j]==' ') n++;
            mp[senders[i]]+=n+1;
        }
        int m = 0; string re;
        for(auto&[s,val]:mp){
            if(val>=m){ m = val; re = s; }
        }
        return re;
    }
};
