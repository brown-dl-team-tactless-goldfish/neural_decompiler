public bool EscapeGhosts(int[][] ghosts, int[] target)
{
	var distance = Distance(target);

	return ghosts
		.Select(g => new[] { g[0] - target[0], g[1] - target[1] })
		.All(g => Distance(g) > distance);

	int Abs(int val) => val >= 0 ? val : -val;
	int Distance(int[] point) => Abs(point[0]) + Abs(point[1]);
}
