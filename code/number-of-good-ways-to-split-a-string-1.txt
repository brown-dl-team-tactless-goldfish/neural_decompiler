int numSplits(char * s){
    int rMap[26] = {0}, lMap[26] = {0}, gSplits = 0, rChars = 0, lChars = 0;
    
    int len = strlen(s);
    
    for(int i = 0; i < len; i++) {
        if(rMap[s[i] - 'a']++ == 0)
            rChars++;
    }
    
    for(int i = 0; i < len - 1; i++) {
        if(lMap[s[i] - 'a']++ == 0)
            lChars++;
        if(--rMap[s[i] - 'a'] == 0)
            rChars--;
        if(lChars == rChars)
            gSplits++;
    }
    return gSplits;
}
