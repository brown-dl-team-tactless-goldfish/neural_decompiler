class Solution {
public:
    vector<vector<int>> restoreMatrix(vector<int>& rowSum, vector<int>& colSum) {
        int M = rowSum.size();
        int N = colSum.size();
        vector<vector<int>> result(M, vector<int>(N, 0));
        int i = 0, j = 0;
        while (i < M && j < N) {
            result[i][j] = min(rowSum[i], colSum[j]);
            rowSum[i] -= result[i][j];
            colSum[j] -= result[i][j];
            if (rowSum[i] == 0) {
                i++;
            }
            if (colSum[j] == 0) {
                j++;
            }
        }
        return result;
    }
};
