class Solution {
public:
    int longestSubsequence(string s, int k) {
        int maxl = 0;
        int pow = 1;
        int ans= 0;
        reverse(s.begin(),s.end());
        for(auto c : s){
            int temp = ans+(c-'0')*pow;
            if(temp<=k && pow <= k){
                maxl++;
                pow*=2;
                ans = temp;
            }
            else if(c == '0'){
                maxl++;
            }
        }
        return maxl;
    }
};
