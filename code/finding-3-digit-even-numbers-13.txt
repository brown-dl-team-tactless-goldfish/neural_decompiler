vector<int> findEvenNumbers(vector<int>& digits) 
{
	vector<int> result;
	int n=digits.size(),a[10]={0};
	for(int i=0;i<n;i++) a[digits[i]]++;

	for(int i=100;i<1000;i+=2)
	{
		int b[10]={0},t1=i/100,t2=(i/10)%10,t3=i%10;
		b[t1]++,b[t2]++,b[t3]++;
		if(b[t1]<=a[t1]&&b[t2]<=a[t2]&&b[t3]<=a[t3]) result.push_back(i);
	}
	return result;
}
