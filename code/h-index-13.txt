class Solution {
public:
    int hIndex(vector<int>& citations) {
        int n=citations.size();
        vector<int> hindex(n+1);
        for(int x=0;x<n;x++){
            int h=citations[x];
            if(h>=n){
                hindex[n]++;
            }else{
                hindex[h]++;
            }
        }
        int c=0;
        for(int i=n;i>=0;i--){
            c+=hindex[i];
            if(c>=i){
                return i;
            }
        }
        return -1;
    }
};
