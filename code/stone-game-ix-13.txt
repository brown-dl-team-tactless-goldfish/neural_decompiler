class Solution {
public:
    bool stoneGameIX(vector<int>& stones) {
        vector<int> count(3, 0);
        
        for (auto val: stones) 
            ++count[val % 3];
        
        if (count[0] % 2) {
            return abs(count[1] - count[2]) > 2;
        }
        
        return count[1] != 0 && count[2] != 0;
    }
};
