int t=0;
int max=0;

for(int i=0;i<=hoursSize;i++)c[i]=-1;

for(int i=0;i<hoursSize;i++){
    if(hours[i]>8)t++;
    else t--;
    
    if(t<0 && c[abs(t)]==-1)c[abs(t)]=i;
    
    if(t>0)max=i+1;
    else if(t<=0){
        if(c[abs(t-1)]>=0 && i-c[abs(t-1)]>max)max=i-c[abs(t-1)];
    }
}
return max;
