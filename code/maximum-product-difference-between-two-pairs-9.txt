int max_num1=0,max_num2=0,min_num1=999999,min_num2=999999;

for (int i=0;i<numsSize;i++)
{
    if(nums[i]>max_num1)
    {
        max_num2=max_num1;
        max_num1=nums[i];
    }
    
    else if(nums[i]>max_num2) max_num2=nums[i];
    
    if(nums[i]<min_num1)
    {
        min_num2=min_num1;
        min_num1=nums[i];
    }
    
    else if(nums[i]<min_num2) min_num2=nums[i];
}
