int xorBeauty(int* nums, int numsSize){
    int n = numsSize;
    int* bit = calloc(32, sizeof(int));
    for(int i = 0; i < n; i++){
        int val = nums[i];
        int idx = 0;
        while(val){
            if(val % 2)
                bit[idx]++;
            idx++;
            val /= 2;
        }
    }
    int ans = 0;
    long long t = 1;
    for(int i = 0; i < 32; i++){
        if(bit[i] %2){
            ans += t;
        }
        t = t << 1;
    }
    return ans;
}
