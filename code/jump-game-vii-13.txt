bool canReach(string s, int minJump, int maxJump) 
{
	int n=s.length(),current=0,total[n+1];
	memset(total,0,sizeof(total));
	if(s.back()=='1') return false;
	for(int i=0;i<n;i++)
	{
		current+=total[i];
		if(i==0||(current&&s[i]=='0'))
		{
			total[min(i+minJump,n)]++;
			total[min(i+maxJump+1,n)]--;
		}
	}
	return current;
}
