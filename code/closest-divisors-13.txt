public class Solution {
    public int[] ClosestDivisors(int num) {
        int div11 = find(num+1), div21 = find(num+2);
        int div12 = (num+1)/div11, div22 = (num+2)/div21;
        return Math.Abs(div11-div12) < Math.Abs(div21-div22) ? new int[]{div11, div12} : new int[]{div21,div22};
    }    
    private int find(int num)
    {
        int div = (int)Math.Sqrt(num);
        while(num%div != 0) div--;
        return div;
    }
}
