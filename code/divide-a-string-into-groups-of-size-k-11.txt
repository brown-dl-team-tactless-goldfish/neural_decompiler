public class Solution {
    public string[] DivideString(string s, int k, char fill) {
        
        List<string> list = new();
        StringBuilder sb = new();
        
        foreach (char c in s)
        {
            sb.Append(c);
            if (sb.Length == k)
            {
                list.Add(sb.ToString());
                sb.Clear();
            }
        }
        
        // Add fill if required
        if (sb.Length > 0)
        {
            while (sb.Length < k)
            {
                sb.Append(fill);
            }
            list.Add(sb.ToString());
        }
        
        return list.ToArray();
    }
}
