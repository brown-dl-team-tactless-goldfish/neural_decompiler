    class local:
        sum = 0
    
    
    def helper(curNode):
        if curNode == None:
            return
        helper(curNode.left)
        local.sum += curNode.val
        helper(curNode.right)
    helper(root)
    # print "local.sum = ", local.sum
    
    def helper2(curNode):
        if curNode == None:
            return
        helper2(curNode.left)
        local.sum -= curNode.val
        curNode.val += local.sum
        
        helper2(curNode.right)
    helper2(root)
    
    return root
