int subarraysDivByK(int* nums, int numsSize, int k){
    int modGroup[k], cumSum = 0;
    for (int i = 0; i < k; i++){
        modGroup[i] = 0;
    }
    for (int i = 0; i < numsSize; i++){
        cumSum += nums[i];
        modGroup[((cumSum % k) + k) % k] += 1;
    }
    int result = modGroup[0];
    for (int i = 0; i < k; i++){
        result += (modGroup[i] * (modGroup[i] - 1))/2;
    }
    return result;
}
