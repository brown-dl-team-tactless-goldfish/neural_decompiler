public class Solution {
    public string FindDifferentBinaryString(string[] nums) {

        HashSet<string> set = new HashSet<string>(nums);
        HashSet<string> combinationSet = new HashSet<string>();
        
        dfs(nums.Length, "", combinationSet);
        foreach(string str in combinationSet)
        {
            if(!set.Contains(str))
                return str;
        }
        
        return "";
    }
    
    private void dfs(int length, string str, HashSet<string> set)
    {
        if(str.Length == length)
        {
            set.Add(str);
            return;
        }
        
        dfs(length, str + '0', set);
        dfs(length, str + '1', set);
    }
}
