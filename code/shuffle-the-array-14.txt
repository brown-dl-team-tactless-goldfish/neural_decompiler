/**
 * Note: The returned array must be malloced, assume caller calls free().
 */
int* shuffle(int* nums, int numsSize, int n, int* returnSize)
{
    *returnSize=numsSize;
    int *returna, i, j;
    returna=(int*)malloc(numsSize*sizeof(int));
    for(i=0, j=0;i<n, j<numsSize;i++, j=j+2)
    {
        returna[j]=nums[i];
    }
    for(i=n, j=1;i<numsSize, j<numsSize;i++, j=j+2)
    {
        returna[j]=nums[i];
    }
    return returna;


}
