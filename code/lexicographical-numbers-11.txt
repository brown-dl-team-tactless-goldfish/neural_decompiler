	void back(int n, vector<int>& vec, int a) {
        if(a!=0) {
            if(a>n) return;
            vec.push_back(a);
        }
        for(int i=0; i<=9; i++) {
            if(a==0 and i==0) i++;
            back(n, vec, a*10 + i);
        }
    }
    vector<int> lexicalOrder(int n) {
        vector<int> vec;
        back(n, vec, 0);
        return vec;
    }
