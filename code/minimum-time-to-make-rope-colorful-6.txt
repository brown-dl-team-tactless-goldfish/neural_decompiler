
int minCost(char * colors, int* neededTime, int neededTimeSize){
    if(neededTimeSize == 1) return 0;

    int slow = 0;
    int fast = 0;
    int ans = 0;

    while(fast < neededTimeSize && slow < neededTimeSize){
        int currTotal = 0, currMax = 0;

        while(fast < neededTimeSize && colors[slow] == colors[fast]){
            currTotal += neededTime[fast];
            currMax = fmax(currMax, neededTime[fast]);
            fast++;
        }
        ans += currTotal - currMax;
        slow = fast;
    }
    return ans;
}
