int smallestRangeI(int* A, int ASize, int K) {
    int max, min, v;
    
    if (K == 0) {
        max = min = A[0];
        for (int i=1; i<ASize; i++) {
            if (A[i] > max)
                max = A[i];
            if (A[i] < min)
                min = A[i];
        }
        return max - min;
    }
    
    v = smallestRangeI(A, ASize, K-1);
    if (v <= 2)         // v <- 0 | 1 | 2
        return 0;
    else
        return v - 2;
}
