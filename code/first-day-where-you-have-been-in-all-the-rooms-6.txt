class Solution {
public:
    int firstDayBeenInAllRooms(vector<int>& nextVisit) {
        vector<long> dp(nextVisit.size());
        dp[1] = 2;
        for (int i = 2; i < nextVisit.size(); i++) {
            dp[i] = (dp[i-1] + dp[i-1] - dp[nextVisit[i-1]] + 1000000009) % 1000000007;
        }
        return static_cast<int>(dp[dp.size()-1]);
    }
};
