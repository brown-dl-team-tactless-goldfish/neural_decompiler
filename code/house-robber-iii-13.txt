    int rob(TreeNode* root) {
        auto ret = dfs(root);
        return std::max(ret.first, ret.second);
    }
    
private:
    std::pair<int, int> dfs(TreeNode* root) // <rob root, don't rob root>
    {
        if (!root)
        {
            return {0, 0};
        }
        auto l = dfs(root->left);
        auto r = dfs(root->right);
        return { root->val + l.second + r.second, std::max(l.first, l.second) + std::max(r.first, r.second) };
    }
