    int head = 1,increment = 1;
    
    bool left = true;
    
    while( n > 1)
    {
        
        if(left)
        {
            
            head = head + increment;
        }
        else{
            
            if(n%2 == 1)
            {
                head = head + increment;
                
            }
        }
        
        n = n / 2;
        
        increment = increment * 2;
        
        left = !left;
        
    }
    
    return head;
    
}
