class Solution {
public:
    int maximumGap(vector<int>& nums, int maxGap = 0)
    {
        if ( nums.size() > 1 )
        {
            sort( nums.begin(), nums.end() );
            for ( auto prev = nums.begin(), curr = next(prev);  curr != nums.end();  ++prev, ++curr )
            {
                maxGap = max( maxGap, *curr - *prev );
            }
        }
        return maxGap;
    }
};
