bool isValid(char*s){
    int cntH=0;
    int len=strlen(s);
    int x=0;
    while(x<len-1){
        if(!isalpha(s[x]))if(s[x]!='-'||cntH++>0||x==0||!isalpha(s[x-1])||!isalpha(s[x+1]))return false;
        x++;
    }
    if(!isalpha(s[x])&&(isdigit(s[x])||s[x]=='-'))return false;
    return true;
}

int countValidWords(char * sentence){
    char tok[2]=" ";
    char* token=strtok(sentence,tok);
    int cnt=0;
    while(token){
        if(isValid(token))cnt++;
        token=strtok(NULL,tok);
    }
    return cnt;
}
