int minDeletionSize(char ** strs, int strsSize){

    int cnt = 0;
    int stlen = strlen(*(strs));
    int strslen = strsSize - 1;
    
    for(int i = 0; i < stlen; ++i)
    {
        for(int j = 0; j < strslen; ++j)
        {
            if(*(*(strs+j)+i) > *(*(strs+j+1)+i)) {
                cnt++;
                break;
            }
        }
    }
    return cnt;
}
