class Solution {
public:
    int alternateDigitSum(int n) {
        int altSum = 0, sign = 1;   // initialize alternate sum to 0 and sign to 1
        while(n > 0) {              // loop through the digits of n from right to left
            altSum += sign * (n % 10);  // add the current digit with the sign flipped to alternate sum
            n /= 10;                // remove the last digit
            sign *= -1;             // flip the sign
        }
        return -1 * sign * altSum;  // multiply the alternate sum by the sign and return it
    }
};
