public class Solution {
    private static bool IsVowel(char c) => 
        c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u';

    public int VowelStrings(string[] words, int left, int right) => words
        .Skip(left)
        .Take(right - left + 1)
        .Count(item => IsVowel(item[0]) && IsVowel(item[^1]));   
    
}
