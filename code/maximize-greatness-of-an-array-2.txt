class Solution {
public:
    int maximizeGreatness(vector<int>& nums) {
        sort(nums.begin(), nums.end()) ;
        queue<int>q ;
        int ret = 0 ;
        int n = nums.size() ;
        for(int i = n-1; i >= 0; i--){
            if(q.front() > nums[i]){
                ret++ ;
                q.pop() ;
            }
            q.push(nums[i]) ;
        }
        return ret ;
    }
};
