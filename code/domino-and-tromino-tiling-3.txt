int numTilings(int n){
    if (n <= 1)
        return 1;
    long *dp = (long*)malloc((n + 1) * sizeof(long));
    int modul = 1e9 + 7;
    
    dp[0] = 1;
    dp[1] = 1;
    dp[2] = 2;
    
    for (int i = 3; i <= n; i++)
        dp[i] = (dp[i - 1] * 2 + dp[i - 3]) % modul;
    
    modul = dp[n];
    free(dp);
    
    return modul;
}
