static bool comp(vector<int>&a,vector<int>&b){
    if(a[0]==b[0]) return a[1]>b[1];
    return a[0]<b[0];
}
int removeCoveredIntervals(vector<vector<int>>& nums) {
    sort(nums.begin(),nums.end(),comp);
    int c = 1,x = 0;
    for(int i=1;i<nums.size();i++){
        if(nums[i][1]<=nums[x][1]) continue;
        c++;
        x = i;
    }
    return c;
}
