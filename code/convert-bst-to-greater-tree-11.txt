void func(struct TreeNode *root, int *t)
{
        if(root != NULL)
        {
            func(root->right, t);
            *t = *t + root->val;
            root->val = *t;
            func(root->left, t);
        }
}
struct TreeNode* convertBST(struct TreeNode* root)
{
    int t=0;
    func(root, &t);
    return root;
}
