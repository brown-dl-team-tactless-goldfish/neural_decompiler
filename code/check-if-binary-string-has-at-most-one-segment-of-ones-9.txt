public class Solution
{
    public bool CheckOnesSegment(string s)
    {
        bool zeroAlreadyFound = false;
        for (int index = 0; index < s.Length; index++)
        {
            if (s[index] is '0')
            {
                zeroAlreadyFound = true;
            }
            else if (zeroAlreadyFound)
            {
                return false;
            }
        }

        return true;
    }
}
