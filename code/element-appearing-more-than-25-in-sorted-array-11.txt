class Solution {
public:
    int findSpecialInteger(vector<int>& arr) {
	    // Append a number bigger than the last to the end to reduce case checking. 
        arr.push_back(arr.back() + 1);
        int size = arr.size();
        for(int i = 0, j = 0; i < size; i ++) {
            if(arr[j] != arr[i] && (i - j) * 4 >= size) return arr[j];
            if(arr[j] != arr[i]) j = i;
        }
        return 0;
    }
};
