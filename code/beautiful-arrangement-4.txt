class Solution {
public:
    int ans=0;
    int req(int i,int j,int n){
        if(i<=0){
            return 1;
        }
        for(int k=1;k<=n;k++){
            if((j&(1<<k))==0 && (k%i==0 || i%k==0)){
                ans+=req(i-1,j|(1<<k),n);
            }
        }
        return 0;
    }

    int countArrangement(int n) {
        req(n,0,n);
        return ans;
    }
};
