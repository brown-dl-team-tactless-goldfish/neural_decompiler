bool checkIfExist(int* arr, int arrSize){
    int N = 0, count = 0;
    
    for (int i = 0; i < arrSize; i++) {
        if (arr[i] == 0)
            count++;
    }
    if (count >= 2)
        return true;
    for (int i = 0; i < arrSize; i++) {
        N = arr[i];
        for (int j = 0; j < arrSize; j++)
            if (N % 2 == 0 && N != 0)
                if ((N/2) == arr[j])
                    return true;
    }
    return false;
}
