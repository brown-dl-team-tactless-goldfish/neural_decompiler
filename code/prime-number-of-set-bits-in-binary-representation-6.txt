int SetBits(int i){        //helper function to count the number of set bits
    int bits = 0;
    while(i){
        bits += (i & 1);
        i = i >> 1;
    }
    return bits;
}

int countPrimeSetBits(int left, int right){   
    int count = 0;
    for(int i = left; i <= right; i++){
        switch(SetBits(i)){   //if number of set bits is prime number, add the count
            case 2:           //hint says we just need to consider these prime numbers
            case 3:          
            case 5:
            case 7:
            case 11:
            case 13:
            case 17:
            case 19:
                count++;
                break;
            default:
                break;
        }
    }
    return count;
}
