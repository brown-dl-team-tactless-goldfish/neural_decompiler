char * reverseOnlyLetters(char * S){
    char* l = S;
    char* r = S + strlen(S) - 1;
    
    while (l < r) {
        if (!isalpha(*l)) {
            ++l;
        }
        else if (!isalpha(*r)) {
            --r;
        }
        else {
            char t = *l;
            *l = *r;
            *r = t;
            
            ++l;
            --r;
        }
    }
    
    return S;
}
