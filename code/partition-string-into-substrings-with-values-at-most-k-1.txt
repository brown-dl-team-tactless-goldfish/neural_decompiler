int minimumPartition(char * s, int k){
    int idx = 0, len = strlen(s), ans = 0;
    long prev = 0, curr = 0;
    while(idx < len){
        prev = curr;
        curr *= 10;
        curr += (s[idx] - '0');
        if(curr > k){
            ans++;
            curr = (s[idx] - '0');
            if(prev > k || !prev)
                return -1;
        }
        idx++;
    }
    if(curr > k) return -1;
    ans++;
    return ans;
}
