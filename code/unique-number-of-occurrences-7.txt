

bool uniqueOccurrences(int* arr, int arrSize){
    int time[2001]={0};//arr[i]+1000;
    int*uniqueArr=(int*)calloc(arrSize,sizeof(int));
    int index=0;
    for(int i=0;i<arrSize;i++){
        if(time[arr[i]+1000]==0)
            uniqueArr[index++]=arr[i];  
        time[arr[i]+1000]++;
    }
    for(int i=0;i<index;i++){
        for(int j=i+1;j<index;j++){
            if(time[uniqueArr[i]+1000]==time[uniqueArr[j]+1000])
                return 0;
        }
    }
    return 1;
}
