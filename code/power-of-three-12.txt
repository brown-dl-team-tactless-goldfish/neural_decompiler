bool isPowerOfThree(int n){
    
    if(n<=0) return false;
    else if(n==1) return true;
    else {
        int y=0, i=-1;
        int temp=n;
// dividing the n value by 3 and storing power value in i.
// for 27 i=3
        while(y==0 && n>0)
        {
            int y=n%3;
            n=n/3;
            i++;
        }
        int power=1;
// keeping i as range finding the 3 to ith power value and storing them in power
        for(int a=0;a<i;a++)
        {
            power=power*3;
        }
//if given n and power are same then return true. store the n value in temp since we make changes to n
        if(power==temp) return true;
        else return false;
    }

}
