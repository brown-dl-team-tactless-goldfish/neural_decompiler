int findMiddleIndex(int* nums, int numsSize){
    int i, n, left, right, f;
    f = -1;
    for(i=0; i<numsSize; i++){
        right=0;
        left=0;
        for(n=0; n<i; n++)
        right+=nums[n];
        for(n=i+1; n<numsSize; n++)
        left+=nums[n];
        if(left==right){
            f=i;
            break;
        }
    }
    return f;

}
