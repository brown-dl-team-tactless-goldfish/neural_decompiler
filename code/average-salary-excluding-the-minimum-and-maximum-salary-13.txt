double average(int* salary, int salarySize){
    int i;
    int min=salary[0];
    int max=salary[0];
    for(i=0;i<salarySize;i++)
    {
        if(salary[i]<min)
            min=salary[i];
        if(salary[i]>max)
            max=salary[i];
    }
    double sum=0;
    for(i=0;i<salarySize;i++)
    {
        if(salary[i]!=min && salary[i]!=max)
            sum=sum+salary[i];
    }
    return sum/(salarySize-2);
}
