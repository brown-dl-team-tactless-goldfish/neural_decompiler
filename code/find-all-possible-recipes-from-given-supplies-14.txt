public class Solution {
    public IList<string> FindAllRecipes(string[] recipes, IList<IList<string>> ingredients, string[] supplies) {    
        HashSet<string> res = new HashSet<string>();
        HashSet<string> supp = new HashSet<string>(supplies);
        bool making = true;
        while (making) {
            making = false;
            for (int i = 0; i < recipes.Length; ++i) {
                if (supp.Contains(recipes[i])) continue;
                List<string> ing = (List<string>)ingredients[i];
                bool hasAll = true;
                for (int j = 0; j < ing.Count; ++j) {
                    if (!supp.Contains(ing[j])) {
                        hasAll = false;
                        break;
                    }
                }
                
                if (hasAll) {
                    making = true;
                    res.Add(recipes[i]);
                    supp.Add(recipes[i]);
                }
            }
        }
        
        return res.ToList();
    }
}
