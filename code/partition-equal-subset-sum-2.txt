bool canPartition(int* nums, int numsSize){
    int sum = 0;
    for(int i = 0; i < numsSize; i++)
        sum += nums[i];
    if(sum % 2 == 1)return false;
    sum = sum >> 1;
    bool dp[sum + 1];
    memset(dp, 0, (sum + 1) * sizeof(bool));
    dp[0] = true;
    for(int i= 1; i <= numsSize; i++)
        for(int j = sum; j >= nums[i-1]; j--)
            dp[j] = dp[j] || dp[j - nums[i - 1]];
    return dp[sum];
}
