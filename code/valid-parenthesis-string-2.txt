public class Solution {
    public bool CheckValidString(string s) {
        
        if(string.IsNullOrEmpty(s))
            return false;
        
        int balance = 0, wild = 0;
        
        for(int i = 0; i < s.Length; i++)
        {
            if(s[i] == '*')
                wild++;
            else if(s[i] == '(')
                balance++;
            else
            {
                balance--;
                if(balance < 0)
                {
                    if(wild > 0)
                    {
                        wild--;
                        balance++;
                    }
                    else
                        return false;
                }
            }
        }
        
        balance = 0;
        wild = 0;
        for(int i = s.Length - 1; i >= 0; i--)
        {
            if(s[i] == '*')
                wild++;
            else if(s[i] == ')')
                balance++;
            else
            {
                balance--;
                if(balance < 0)
                {
                    if(wild > 0)
                    {
                        wild--;
                        balance++;
                    }
                    else
                        return false;
                }
            }
        }
        
        return true;
    }
}
