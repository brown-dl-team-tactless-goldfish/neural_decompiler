int findUnsortedSubarray(int* nums, int numsSize){
    int i, j;
    int start=-2, end=-2;
    for(i=0, j=1; i<numsSize-1; ++i, ++j){
        if(nums[i] > nums[j]){
            if(start == -2) start = i;
            end = j;
            while(start >= 0 && nums[j] < nums[start]) start--;
            nums[j] = nums[i];
        }
    }
    return end - start;
}
