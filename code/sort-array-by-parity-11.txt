int* sortArrayByParity(int* A, int ASize, int* returnSize){

    int *ptr=(int*)malloc(ASize*sizeof(int));
    *returnSize=ASize;
    
    int i,k=0;
    
    for(i=0;i<ASize;i++)
    {
        if(A[i]%2==0)
        {
            ptr[k]=A[i];
            k=k+1;
        }
    }
    
    for(i=0;i<ASize;i++)
    {
        if(A[i]%2!=0)
        {
            ptr[k]=A[i];
            k=k+1;
        }
    }
    
    return ptr;
}
