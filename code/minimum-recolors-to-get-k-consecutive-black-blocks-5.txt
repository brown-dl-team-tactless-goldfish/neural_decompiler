class Solution 
{
public:
    int minimumRecolors(string blocks, int k) 
    {
        int c = count_if(blocks.begin(),blocks.begin()+k,[](char ch)
        {
            return ch=='W';
        });
        int minOpe = c, n = blocks.size()-k+1;
        for(int i=1;i<n;i++)
        {
            c = c - (blocks[i-1]=='W') + (blocks[i+k-1]=='W');
            minOpe = min(minOpe,c); 
        }
        return minOpe;
    }
};
