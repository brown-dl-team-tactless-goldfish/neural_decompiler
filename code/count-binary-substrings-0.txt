int countBinarySubstrings(char * s){
    int ptr = 0;
    int cur = 1;
    int ans = 0;
    for(int i = 1; i < strlen(s); i++){
        if(s[i] == s[i-1])
            cur++;
        else{
            ans += fmin(ptr, cur);
            ptr = cur;
            cur = 1;
        }
    }
    ans += fmin(ptr, cur);
    return ans;
}
