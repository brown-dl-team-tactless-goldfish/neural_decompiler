class Solution {
public:
    string maxValue(string s, int x) {
        int i,n=s.size(),idx=n;
        if(s[0]=='-'){
            for(i=n-1;i>=1;i--){
                if(s[i]-'0'>x) idx=i;
            }
        }
        else{
            for(i=n-1;i>=0;i--){
                if(s[i]-'0'<x) idx=i;
            }
        }
        s.insert(s.begin()+idx,x+'0');
        return s;
    }
};
