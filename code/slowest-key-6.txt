char slowestKey(int* releaseTimes, int releaseTimesSize, char * keysPressed){
    int iele = releaseTimes[0];
    char cele = keysPressed[0];
    
    for(int i=1; i<releaseTimesSize; i++){
        if (releaseTimes[i]-releaseTimes[i-1] > iele){
            iele = releaseTimes[i] - releaseTimes[i-1];
            cele = keysPressed[i];
        }
        else if (releaseTimes[i]-releaseTimes[i-1] == iele){
            if (keysPressed[i] > cele)
                cele = keysPressed[i];
        }
    }
    return cele;
}
