public class Solution 
{
    public int SumFourDivisors(int[] nums) 
    {
        return nums.Select(x => SumFourDivisors(x)).Sum();
    }
    
    public int SumFourDivisors(int num)
    {
        var hashSet = new HashSet<int>(new int[] {1, num});
        for(int i = 2; i <= Math.Sqrt(num); i++)
            if(num % i == 0)
                hashSet.UnionWith(new int[] {i, num / i});
        return hashSet.Count == 4 ? hashSet.Sum() : 0;
    }
}
