    vector<int> res(num_people, 0);
    
    if(candies == 0){
        return res;
    }
    
    int i = 1;
    int j = 0;
    
    while(candies){
          if(j == num_people){
              j = 0;
          }
        
          if(i > candies){ // Condition when last candies are left
              res[j] += candies;
              break;
          }
        
          res[j] += i;
          candies -= i;
          i++;
          j++;
    }
    
    
    return res;
}
