    def minMoves(self, A, k):
        A = [i for i, a in enumerate(A) if a]
        n = len(A)
        B = [0] * (n + 1)
        res = float('inf')
        for i in xrange(n):
            B[i + 1] = B[i] + A[i]
        for i in xrange(len(A) - k + 1):
            res = min(res, B[i + k] - B[k / 2 + i] - B[(k + 1) / 2 + i] + B[i])
        res -= (k / 2) * ((k + 1) / 2)
        return res
