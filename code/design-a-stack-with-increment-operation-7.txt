class CustomStack:

    def __init__(self, maxSize: int):
        self.data = []
        self.max_size = maxSize
        

    def push(self, x: int) -> None:
        curLen = len(self.data)
        
        if (curLen >= self.max_size):
            return
        
        self.data.append(x)

    def pop(self) -> int:
        curLen = len(self.data)
        
        if (curLen == 0):
            return -1
        
        res = self.data.pop()
        return res

    def increment(self, k: int, val: int) -> None:
        
        if (k > len(self.data)):
            k = len(self.data)
            
        for i in range(k):
            self.data[i] += val
