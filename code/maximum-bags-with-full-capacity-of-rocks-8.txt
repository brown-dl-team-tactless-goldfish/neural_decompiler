public class Solution {
    public int MaximumBags(int[] capacity, int[] rocks, int ar) {
        int [] h=new int[rocks.Length]; int cnt=0; 
        for(int i=0;i<h.Length;i++){
            h[i]=capacity[i]-rocks[i];
            if(h[i]==0){cnt++;}
            Console.WriteLine(h[i]);
        } 
        Array.Sort(h);
        for(int i=0;i<h.Length;i++){
            if(ar>=h[i] && h[i]!=0 && ar>0){
            ar-=h[i]; 
            cnt++;
            }
        }
        return cnt;
    }
}
