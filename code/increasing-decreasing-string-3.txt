char * tmp_string=new_string;

while(string_len > 0){
	// go from step 1 to step 3;
	// increasing the string
	int ii=0;
	for(;ii<128;ii++){
		if(ascii[ii]>0){
			*tmp_string=(char)ii;
			ascii[ii]--;
			string_len--;
			tmp_string++;
		}
	}
	// go from step 4 to 6
	// decreasing the string
	ii=127;
	for(;ii>0;ii--){
		if(ascii[ii]>0){
			*tmp_string=(char)ii;
			ascii[ii]--;
			string_len--;
			tmp_string++;
		}
	}

}
return new_string;
