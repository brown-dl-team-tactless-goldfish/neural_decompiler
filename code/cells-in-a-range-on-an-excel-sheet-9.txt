class Solution {
public:
    vector<string> cellsInRange(string s) {
        char a = 0, b = 0, c = 0, d = 0;
        sscanf(s.c_str(), "%c%c:%c%c", &a, &b, &c, &d);
        vector<string> res;
        for (char i = a; i <= c; ++i) {
            for (char j = b; j <= d; ++j) {
                stringstream ss;
                ss << i << j;
                res.push_back(ss.str());
            }
        }
        return res;
    }
};
