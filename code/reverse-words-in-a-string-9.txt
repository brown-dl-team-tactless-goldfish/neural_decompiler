char * reverseWords(char * s){
    char *p[10001];
    char *ans = (char *)malloc(strlen(s)+2);
    p[0] = strtok(s, " ");
    int pos = 1;
    while ((p[pos] = strtok(NULL, " "))) {
        pos++;
    }
    
    memset(ans, 0, strlen(s)+2);
    for (int i=pos-1; i>=0; i--) {
        sprintf(ans+strlen(ans), "%s ", p[i]);
    }
    ans[strlen(ans)-1] = '\0';
    return ans;
}
