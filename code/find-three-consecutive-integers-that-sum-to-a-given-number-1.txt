long long* sumOfThree(long long num, int* returnSize) {
    *returnSize = 0;
    long long* arr = (long long*) malloc(3 * sizeof(long long));
    if (num % 3 == 0) {
        num /= 3;
        arr[0] = num - 1;
        arr[1] = arr[0] + 1;
        arr[2] = arr[1] + 1;
        *returnSize = 3;
    }
    return arr;
}
