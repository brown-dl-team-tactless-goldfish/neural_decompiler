class Solution {
public:
    long long maxKelements(vector<int>& nums, int k) {
        priority_queue <long long> pq;
        for(auto &it:nums)
            pq.push(it);
        long long score=0;
        long double tmp;
        while(k--){
            tmp = pq.top();
            score += tmp;
            pq.pop();
            pq.push(ceil(tmp/3));
        }
        return score;
    }
};
