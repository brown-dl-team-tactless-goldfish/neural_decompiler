vector<int> filterRestaurants(vector<vector<int>>& restaurants, int veganFriendly, int maxPrice, int maxDistance) {
	vector<vector<int>> res;
	for(auto& rest : restaurants){
		if(veganFriendly && rest[2]==0 || rest[3]>maxPrice || rest[4]>maxDistance) continue;
		res.push_back(rest);
	}

	sort(res.begin(), res.end(), [](auto& a, auto& b){return a[1]==b[1]?a[0]>b[0]:a[1]>b[1];});
	vector<int> ret;
	for(auto& v: res)
		ret.push_back(v[0]);

	return ret;
}
