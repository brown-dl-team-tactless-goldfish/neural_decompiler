 map<int,int> m;
int s;
SmallestInfiniteSet() {
    for(int i=1;i<=1000;i++){
        m[i]=1;
    }
    s=1;
}

int popSmallest() {
    while(m.find(s)==m.end())
        s++;
        m.erase(s);
        s++;
        return s-1;
}

void addBack(int num) {
    if(m.find(num)==m.end()){
        m[num]=1;
        s=min(s,num);
    }
