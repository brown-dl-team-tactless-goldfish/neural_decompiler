char * makeFancyString(char * s){
    int n = strlen(s) ;
    char* ans = malloc((n + 1));
    int idx = 0;
    int repeat = 1;
    ans[idx] = s[0];
    idx++;
    for(int i = 1; i < n; i++){
        if(s[i] == s[i-1]){
            repeat++;
        }
        else{
            repeat = 1;
        }
        if(repeat < 3){
            ans[idx] = s[i];
            idx++;
        }
    }
    ans[idx] = '\0';
    return ans;

}
