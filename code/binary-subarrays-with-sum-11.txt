class Solution {
public:
    int numSubarraysWithSum(vector<int>& A, int S) {
        int sum = 0 , count = 0;
        unordered_map<int,int> u;
        for(int i=0; i <A.size() ; i++)
        {
            sum += A[i];
            if(sum == S)
            {
                count++;
            }
            if(u.find(sum-S)!=u.end())
            {
                count += u[sum-S];
            }
            u[sum]++;
        }
        return count;
    }
};
