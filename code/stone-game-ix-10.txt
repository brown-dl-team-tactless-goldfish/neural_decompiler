	bool stoneGameIX(vector<int>& stones) {
    int n=stones.size();
    int arr[3]={0};
    for(int i=0;i<n;i++)
        arr[stones[i]%3]++;
    if(arr[0]%2==1)
        if(abs(arr[1]-arr[2])>=3)return true;
        else return false;
    else
        if(arr[1]==0 || arr[2]==0)return false;
        else return true;
    }
