class Solution {
public:
    vector<int> arrayRankTransform(vector<int>& arr) {
        
        set<int> s(arr.begin(), arr.end());
        
        unordered_map<int, int> m;
        
        int r = 0;
        
        vector<int> ans;
        
        for (const auto & a : s) m[a] = ++r;
        
        for (const auto & a : arr)
        {
            ans.push_back(m[a]);
        }
                            
        return ans;
    }
};
