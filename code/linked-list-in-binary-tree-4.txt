bool check(struct ListNode* head, struct TreeNode* root);
bool isSubPath(struct ListNode* head, struct TreeNode* root)
{
    bool flag;
    if(!root)
    {
        return false;
    }
    
    flag=check(head,root);
    
    
    return flag|| isSubPath(head,root->left)||isSubPath(head,root->right);  
    
}
bool check(struct ListNode* head, struct TreeNode* root)
{
    
  if(!root)
  {
      return false;
  }
  if(!head->next)
  {
      return head->val==root->val;
  }
    
 return head->val==root->val&&(check(head->next,root->left)||check(head->next,root->right)); 
}
