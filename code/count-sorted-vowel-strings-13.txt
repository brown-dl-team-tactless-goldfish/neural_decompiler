

int countVowelStrings(int n){
    int tmp[5]={1,1,1,1,1};
    int sum=0;
    
    for(int i=1;i<n;i++)
    {
        tmp[1]+=tmp[0];
        tmp[2]+=tmp[1];
        tmp[3]+=tmp[2];
        tmp[4]+=tmp[3];
    }
    
    for(int i = 0;i < 5;i++)
    {
        sum+=tmp[i];
    }
    
    return sum;
}
