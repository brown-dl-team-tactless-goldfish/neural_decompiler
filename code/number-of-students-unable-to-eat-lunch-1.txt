int count=0;

int countStudents(int* students, int studentsSize, int* sandwiches, int sandwichesSize){

//first student is *students
//first sandwiches is *sandwiches


if(*students==*sandwiches)
{
    //printf("1");
    for(int i=0;i<sandwichesSize-1;i++)
           {
               *(students+i)=*(students+i+1);
           }
    for(int i=0;i<sandwichesSize-1;i++)
           {
               *(sandwiches+i)=*(sandwiches+i+1);
           }
    studentsSize--;
    sandwichesSize--;
    count=0;
}
else if(*students!=*sandwiches)
{
    //printf("2");
    int temp;
    temp=*students;
    for(int i=0;i<sandwichesSize-1;i++)
           {
               *(students+i)=*(students+i+1);
                //printf("3");
           }
    *(students+sandwichesSize-1)=temp;
    count++;
}
    if(count!=sandwichesSize)
    {
        countStudents(students,studentsSize,sandwiches,sandwichesSize);
        //repeat until count is equal to the studentsSize.
    }
    if(count==sandwichesSize)
    {
        return count;
    }
return count;
