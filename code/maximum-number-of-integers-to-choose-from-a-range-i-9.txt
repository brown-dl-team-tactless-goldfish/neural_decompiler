class Solution {
public:
    int maxCount(vector<int>& banned, int n, int maxSum) {
        map<int, int> mp;
        int sum = 0, cnt = 0;
        for(auto i : banned) mp[i]++;
        for(int i = 1; i <= n; i++) {
            if(mp.find(i) == mp.end() && (sum + i <= maxSum)) {
                cnt++;
                sum += i;
            }
        }
        return cnt;
    }
};
