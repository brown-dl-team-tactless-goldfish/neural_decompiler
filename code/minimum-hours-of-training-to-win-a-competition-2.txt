int minNumberOfHours( int &ieng, int &iex,
 const vector<int>& energy,const vector<int>& experience) {
    int size = energy.size(), cnt = 0,ac = 0;

    for(int i = 0; i < size; ++i) {
        ac += energy[i];
        if(iex <= experience[i]) {
            cnt += (experience[i] - iex +  1);
            iex += (experience[i] - iex +  1);
        }
        
        iex += experience[i];
    }
    cnt += (ac >= ieng ? ac - ieng + 1 : 0 );  

    return cnt; 
}
