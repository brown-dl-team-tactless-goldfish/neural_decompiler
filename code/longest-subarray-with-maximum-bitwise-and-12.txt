class Solution {
public:
    int longestSubarray(vector<int>& nums) 
    {
        // 1. find out the max value of nums
        // 2. count the appreances of max value for subarrays and return max one
        int n = nums.size();
        
        int mx_val = INT_MIN;
        for(auto it : nums) 
            mx_val = max(mx_val,it);
        
        int cnt = 1;
        int ans = 1;
        for(int i=0;i<n-1;i++)
        {
            if(nums[i]==mx_val && nums[i]==nums[i+1])
                cnt++;
            else
                cnt=1;
            
            ans = max(ans,cnt);
        }
        return ans;
    }
};
