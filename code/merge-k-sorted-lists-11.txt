
struct ListNode* mergeKLists(struct ListNode** lists, int listsSize){
    
    
    struct ListNode dummy, *temp;
    
    temp = &dummy;
    dummy.next = NULL;
    int count = 0;
    int index = 0;
    int val = INT_MAX;
    int *rem;
    
    rem = calloc(sizeof(int),listsSize);
    
    while (true) {
        val = INT_MAX;
        
        if(listsSize == count)
            return dummy.next;
        
        for(int i = 0; i < listsSize; i++) {
        
            if(lists[i] == NULL && rem[i] == 0) {
                count++;
                rem[i] = 1;
            if(listsSize == count)
                return dummy.next;
            continue;
            }
    
             if(lists[i] != NULL && val > lists[i]->val) {
            
                index = i;
                val = lists[i]->val;
             }
      }

        temp->next = lists[index];
        lists[index] = lists[index]->next;
        temp = temp->next;
    }
    
    free(rem);

    return dummy.next;
}
