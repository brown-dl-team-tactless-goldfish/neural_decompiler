int minCharacters(string a, string b) {
    int ca[26] = {}, cb[26] = {}, sza = a.size(), szb = b.size();
    for (auto c : a)
        ++ca[c - 'a'];
    for (auto c : b)
        ++cb[c - 'a'];
    int suma = ca[0], sumb = cb[0], res = sza - ca[0] + szb - cb[0];
    for (auto c = 1; c < 26; ++c) {
        res = min({res, suma + szb - sumb, sumb + sza - suma, 
                   sza - ca[c] + szb - cb[c]});
        suma += ca[c];
        sumb += cb[c];
    }
    return res;
}
