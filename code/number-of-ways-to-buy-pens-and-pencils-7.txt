class Solution {
public:
    long long waysToBuyPensPencils(int total, int cost1, int cost2) {
        long long result = 0;
        
        int cost1Ps = total/cost1;
        int cost2Ps = 0;
        int tempTotal = 0;
        
        for(int i = 0; i <= cost1Ps; i++) {
            tempTotal = total - i*cost1;
            cost2Ps = tempTotal/cost2;
            result += cost2Ps+1;
        }
        
        return result;
    }
};
