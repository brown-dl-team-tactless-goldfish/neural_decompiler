/* int hIndex(int* citations, int citationsSize) */
int hIndex(int* c, int cS)
{
  int dp[cS+1];
  for(int i = 0; i<cS+1; i++) dp[i] = 0;

  for(int i = 0; i<cS; i++)
    if(c[i] < cS)
      dp[c[i]]++;
    else
      dp[cS]++;

  /* count all the buckets backward */
  for(int i = cS, acc = 0; i>=0; i--)
  {
    acc+=dp[i];
    if(acc>=i) return i;
  }
  return 0;
}
