 int maxAscendingSum(int* nums, int numsSize)
 {
     int sum = nums[0] , max = sum;
     
     
     for ( int i = 1 ; i < numsSize ; i++)
     {
                
         if ( nums[i] <= nums[i-1])
         {
             sum = 0;
         }
         
         sum = nums[i] + sum;
         
        // printf("sum = %d\n",sum);
         
         if ( max < sum)
          {
            max = sum;
             
          }
     }
     
     
     return max;
 
 }
