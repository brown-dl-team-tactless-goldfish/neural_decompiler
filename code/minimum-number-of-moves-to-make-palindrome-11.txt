public class Solution {
    public int MinMovesToMakePalindrome(string s) {
        List<char> strList = s.ToList();

        int result=0;
        while( strList.Any())
        {
            //Find last element in list
            char lastElement = strList[strList.Count-1];
            int index = strList.FindIndex(ch=> ch== lastElement);

            //if element match not found and it is last element
            //then move that element to center of string to make palindrome
            if(index == strList.Count -1)
                result += (index/2);
            //add i index to result and remove the item;
            else
            {
                result +=index;
                strList.RemoveAt(index);
            }
            strList.RemoveAt(strList.Count-1);
        }
        return result;
    }
}
