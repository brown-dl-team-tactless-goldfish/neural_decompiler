public class Solution 
{
    public bool IsGoodArray(int[] nums) 
    {
        var gcd = nums[0];
        for(int i = 1; i < nums.Length; i++)
            gcd = GCD(gcd, nums[i]);
        return gcd == 1;
    }
    
    private int GCD(int x, int y)
    {
        return y == 0 ? x : GCD(y, x % y);
    }
}
