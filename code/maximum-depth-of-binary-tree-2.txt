/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     struct TreeNode *left;
 *     struct TreeNode *right;
 * };
 */


int maxDepth(struct TreeNode* root){

    if(root == NULL) return 0;
     register uint32_t hr = 1, hl = 1;
     hr += maxDepth(root->right);
     hl += maxDepth(root->left);
    
    if ( hr > hl) return hr;
    else return hl;
    
    
}
