
/* Function to convert upper to lower case */
void uptolow(char *str)
{
    int i;
    for(i=0;str[i]!='\0';i++)
    {
        if(str[i]>=65 && str[i]<=90)
            str[i]+=32;
    }
}

bool isPalindrome(char * s){
    int i,j,l=strlen(s);
    i=0;
    j=l-1;
    uptolow(s);                                                             //convert upper to lower case
    while(i<j)
    {
        /* now check if there is any illegal characters in strings and skip that for compare */
        while(((s[i]<97 || s[i]>122) && (s[i]<48 ||s[i]>57))  && i<l) 
            i++;
        while(((s[j]<97 || s[j]>122) && (s[j]<48 ||s[j]>57))  && j>0)
            j--;
        
        /* if string only contains illegal characters then return true */
        if(i==l && j==0)
            return true;
       
        /* for any mismatch return false */
        if(s[i]!=s[j])
            return false; 
        i++;
        j--;
    }
    return true;            //else return true
}
