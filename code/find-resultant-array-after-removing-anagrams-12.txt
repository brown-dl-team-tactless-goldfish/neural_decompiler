class Solution {
public:
    vector<string> removeAnagrams(vector<string>& nums) {
        vector<string> arr = nums;
        for(int i = 0; i<nums.size();++i) sort(nums[i].begin(),nums[i].end());
        vector<string> re;
        re.push_back(arr[0]);
        for(int i = 1; i<nums.size();++i){
            if(nums[i]!=nums[i-1]) re.push_back(arr[i]);
        }
        return re;
    }
};
