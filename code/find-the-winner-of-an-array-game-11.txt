class Solution {
public:
    int getWinner(vector<int>& arr, int k) {
        int n = arr.size(), i, c=0, mx=arr[0];
        for(i=1; i<n; i++){
            if(mx > arr[i]){
                c++;
            }
            else{
                c = 1;
                mx = arr[i];
            }
            if(c==k){return mx;}
        }
        return mx;
    }
};
