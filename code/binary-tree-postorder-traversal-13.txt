void getSize(struct TreeNode* root, int *size)
{
    if(!root)
        return;
    
    if(root->left)
        getSize(root->left, size);
        
    if(root->right)
        getSize(root->right, size);
 
    (*size)++;
}

void helper(struct TreeNode* root, int *index, int *res)
{
    if(!root)
        return;
    
    if(root->left)
        helper(root->left, index, res);
    if(root->right)
        helper(root->right, index, res);
    
    res[(*index)++] = root->val;
}

int* postorderTraversal(struct TreeNode* root, int* returnSize){

    int size=0, index = 0;
    int *res = NULL;
    
    getSize(root, &size);
    
    res = (int*)calloc(size, sizeof(int));
    
    helper(root, &index, res);
    
    *returnSize = size;
    return res;
}
