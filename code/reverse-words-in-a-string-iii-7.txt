void reverse(char *s, int l, int r)
{
	char tmp;
	while(l < r)
	{
		tmp = s[l];
		s[l] = s[r];
		s[r] = tmp;
		l ++;
		r --;
	}
}

char* reverseWords(char* s) {
	int l, r, len=strlen(s);
	l=r=0;
	while(1){
		if(r >= len-1)	return s;
		while(s[l] == ' ') l++;
		r = l;
		while(s[r] != ' ' && r < len) r++;
		reverse(s, l, r-1);
		l=r;
	}
}
