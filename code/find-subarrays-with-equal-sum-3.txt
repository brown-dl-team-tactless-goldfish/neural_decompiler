#define MAX 1000

bool findSubarrays(int* nums, int numsSize){
    int V_sum[MAX] = {0},count=0;
    for(int i=0;i<numsSize-1;i++){
        V_sum[i] = nums[i] + nums[i+1];
        for(int j=0;j<count;j++){
            if(V_sum[j]==V_sum[i]) return true;
        }
        count++;
    }
    return false;
}
