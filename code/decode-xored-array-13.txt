int* decode(int* encoded, int encodedSize, int first, int* returnSize){
    int* newArr = (int*) malloc(sizeof(int)*(encodedSize+1));
    int i=0;
    newArr[0] = first;
    while(i!=encodedSize){
        newArr[i+1] = encoded[i]^newArr[i];

        i++;
    }
    *returnSize = encodedSize+1;
    return newArr;
}
