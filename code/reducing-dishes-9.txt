int maxSatisfaction(int* satisfaction, int satisfactionSize){
    int temp, tempsum, satisfactionsum = 0;
    for(int i = 0; i < satisfactionSize; i ++){
        for(int j = i + 1; j < satisfactionSize; j++){
            if(satisfaction[i] > satisfaction[j]){
                temp = satisfaction[i];
                satisfaction[i] = satisfaction[j];
                satisfaction[j] = temp;
            }
        }
    }
    for(int i = 0; i < satisfactionSize; i++){
        tempsum = 0;
        for(int j = satisfactionSize; j > i; j--){
            tempsum = tempsum + (satisfaction[j - 1] * (j - i));
        }
        if(tempsum > satisfactionsum){
            satisfactionsum = tempsum;
        }
    }
    return satisfactionsum;
}
