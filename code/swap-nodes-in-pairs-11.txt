struct ListNode* swapPairs(struct ListNode* head){
    if(!head)
        return NULL;
    struct ListNode *p1 = head, *p2 = head->next, *prev = NULL;
    while(p2)
    {
        p1->next = p2->next;
        p2->next = p1;
        if(!prev)
        {
            head = p2;
            prev = p1;
        }
        else
        {
            prev->next = p2;
            prev = p1;
        }
        
        if(p1->next) {p1 = p1->next;}
        p2 = p1->next;

    }
    return head;
}
