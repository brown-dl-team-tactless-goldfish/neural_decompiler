
char * removeDuplicates(char * S){
    int len=strlen(S),idx=0,i=1;
    char *stack=malloc(sizeof(char)*len);
    stack[0]=S[0];
    while(i<len)
    {
        if(idx>=0&&stack[idx]==S[i])
            idx--;     
        else
            stack[++idx]=S[i];
        
        i++;
    }
    stack[idx+1]='\0';
    return stack;
}
