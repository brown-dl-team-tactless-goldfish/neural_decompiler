	long long wonderfulSubstrings(string word) {
        int m[1<<10];
        memset(m, 0, sizeof(m));
        m[0] = 1;
        long long count=0, mask=0;
        for(auto &c : word) {
            mask ^= 1<<(c-'a');
            for(int i=0; i<10; i++) count += m[mask^(1<<i)];
            count += m[mask]++;
        }
        return count;
    }
