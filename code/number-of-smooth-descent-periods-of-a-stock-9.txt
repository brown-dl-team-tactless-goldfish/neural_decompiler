long long getDescentPeriods(vector<int>& nums) {
        int i=0;
        int n=nums.size();
        long long ans=0;   
        
        while(i<n){
            long long count=0,j;            
            for(j=i+1; j<n; j++){
                if(nums[j] == nums[j-1]-1){
                    count++;  
                }else
                    break;
            }
            ans+=(count*(count+1))/2;
            i=j;
        }
        return ans+n;
    }
