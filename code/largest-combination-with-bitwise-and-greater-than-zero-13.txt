class Solution {
public:
    int largestCombination(vector<int>& a) {
        int i,n=a.size(), j, x=1,cnt=0, ans=0;
        for(i=0;i<31;i++) {
            cnt=0;
            for(j=0;j<n;j++) {
                if(x&a[j]) cnt++;
            }
            ans=max(ans, cnt);
            x=x<<1;
        }
        return ans;
    }
};
