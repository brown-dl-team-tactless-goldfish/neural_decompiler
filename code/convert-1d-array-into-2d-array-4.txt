public class Solution {
    public int[][] Construct2DArray(int[] o, int m, int n) {
        if (o.Length != m * n) return new int[0][];
        
        int[][] res = new int[m][];
        for (int i = 0; i < m; ++i) res[i] = new int[n];
        
        for (int i = 0; i < o.Length; ++i) {
            
            res[i / n][i % n] = o[i];
        }
        
        return res;
    }
}
