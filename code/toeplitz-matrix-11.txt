bool isToeplitzMatrix(int** matrix, int matrixRowSize, int *matrixColSizes) {
    for (int r = 1; r < matrixRowSize; ++r) {
        int* a   = matrix[r - 1];
        int* b   = matrix[r] + 1;
        int  len = (matrixColSizes[r] - 1)*sizeof(int);
        
        if (memcmp(a, b, len)) return false;
    }
    
    return true;
}
