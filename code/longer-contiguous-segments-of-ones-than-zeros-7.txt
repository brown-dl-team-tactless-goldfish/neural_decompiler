class Solution:
    def checkZeroOnes(self, s: str) -> bool:
        data1 = s.split("0")
        data0 = s.split("1")
        
        data1.sort(key=len)
        data0.sort(key=len)
        
        print ("data1 = ", data1)
        print ("data0 = ", data0)
        
        return len(data1[-1]) > len(data0[-1])
