class Solution {
public:
    int countPrefixes(vector<string>& words, string s) {
        int c = 0;
        unordered_map<string,int> hmp;
        int l = s.length();
        for(int i=0; i<words.size(); i++) {
            hmp[words[i]]++;
        }
        for(int i=1; i<=l; i++) {
            string str = s.substr(0,i);
            if(hmp.find(str)!=hmp.end())
                c += hmp[str];
        }
        return c;
    }
};
