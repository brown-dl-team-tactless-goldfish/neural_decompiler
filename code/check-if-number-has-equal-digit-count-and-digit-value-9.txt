bool digitCount(char * num){

    int* q = malloc(10*sizeof(int)); memset(q,0,10*4);
    for(char* t = num; *t != '\0'; ++t)
        q[*t - '0']++;
    
    int i = 0;
    for(char* t = num; *t != '\0'; ++t, ++i)
    {
        if( (int) *t-'0' != q[i])
           return false;    
    }
    
    return true;
     
}
