/**
 * Note: The returned array must be malloced, assume caller calls free().
 */
int* sumZero(int n, int* returnSize){
    *returnSize = n;
    int* res = malloc(*returnSize * sizeof(int));
    if(n%2){
        res[n/2] = 0;
        for(int i=1; i<=n/2; i++){
            res[n/2+i] = i;
            res[n/2-i] = -i;
        }
    }else{
        for(int i=1; i<=n/2; i++){
            res[n/2-1+i] = i;
            res[n/2-i] = -i;
        }
    }
    return res;
}
