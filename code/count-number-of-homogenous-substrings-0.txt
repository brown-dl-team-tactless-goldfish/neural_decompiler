

int countHomogenous(char * s){
    int ans  = 0;
    int mod = 1e9 + 7;
    int n = strlen(s);
    int cn =1;
    for(int i = 1; i < n; i++){
        if(s[i] == s[i-1])
            cn++;
        else{
            long val = (1+cn) * cn /2;
            ans = (ans + val % mod) % mod;
            cn = 1;
        }
    }
    long long val = (long long)(1+cn) * cn /2;
    ans = (ans + val % mod) % mod;    
    return ans;
}
