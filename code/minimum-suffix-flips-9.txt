class Solution {
public:
    int minFlips(string target) {
        int n = target.size(), cnt = 0;
        bool flag = false;
        for(int i = 0; i < n; i++)
        {
            char c = '0';
            if(flag)
                c = '1';
            if(c != target[i])
            {
                flag = !flag;
                cnt += 1;
            }
        }
        return cnt;
    }
};
