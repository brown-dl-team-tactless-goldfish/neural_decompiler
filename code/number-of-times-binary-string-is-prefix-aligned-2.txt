int numTimesAllBlue(int* flips, int n)
{  
   int ret=0;
   int i=0,start=-1,end=-1;
    
    for(i=0;i<n;i++)
    {
       if(start==-1)
       {
           end=start=flips[i];
       }
       else
       {
           if(flips[i]>end)
           {
               end=flips[i];
           }
           if(flips[i]<start)
           {
               start=flips[i];
           }
       }
      if(i+1>=end&&start==1)
      {
          ret++;
      }
    }
    
    return ret;
}
