public class Solution {
    public long SubArrayRanges(int[] nums) {
        long r = 0;
        
        for (int i = 0; i < nums.Length - 1; ++i) {
            int max = nums[i], min = nums[i];
            for (int j = i + 1; j < nums.Length; ++j) {
                max = Math.Max(max, nums[j]);
                min = Math.Min(min, nums[j]);
                r += Math.Abs(max - min);
            }
        }
        
        return r;
    }
}
