class Solution {
public:
    int findKthNumber(int m, int n, int k) {
        int l = 1, r = m * n + 1, mid;
        while(l < r){
            mid = l + (r - l) / 2;
            if(lex(m, n, mid) >= k) r = mid;
            else l = mid + 1;
        }
        return l;
    }
    //Returns # of elements in the m*n table that are <= x
    int lex(int m, int n, int x){
        int count = 0;
        for(int i = 1; i <= m; i++)
            count += min(n, x / i);
        return count;
    }
};
