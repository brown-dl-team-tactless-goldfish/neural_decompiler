    while(maxi < time)
    {
        for(int i=0; i <clips.size(); i++)
        {
            int left = clips[i][0];
            
            int right = clips[i][1];
            
            if(left <= mini && right > maxi)
            {
                maxi = right;
            }
            
        }
         if(mini == maxi)
            {
                return -1;
            }
            
            mini = maxi;
            
            total++;
    }
    
    return total;
}
