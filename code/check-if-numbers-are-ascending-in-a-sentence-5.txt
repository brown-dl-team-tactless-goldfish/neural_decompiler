bool areNumbersAscending(char * s){
    int i, j = -1;
    
    char *token = strtok(s, "abcdefghijklmnopqrstuvwxyz ");
    while(token){
        i = atoi(token);
        
        if(i <= j)
            return 0;
        
        j = i;
        token = strtok(NULL, "abcdefghijklmnopqrstuvwxyz ");
    }
    
    return 1;
}
