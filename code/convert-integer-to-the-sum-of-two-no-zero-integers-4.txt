int* getNoZeroIntegers(int n, int* returnSize){
    int m = n;
    int a = 0;
    int d = 1;
    
    while(n > 1)
    {
        if(n % 10 == 1)
        {
            n -= 2;
            a += d * 2;
        }
        else
        {
            n -= 1;
            a += d;
        }
        
        n /= 10;
        d *= 10;
    }
    
    int *ans = malloc(sizeof(int) * 2);
    ans[0] = a;
    ans[1] = m - a;
    
    *returnSize = 2;
    return ans;
}
