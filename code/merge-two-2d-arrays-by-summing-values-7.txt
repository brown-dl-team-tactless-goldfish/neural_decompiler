class Solution {
public:
    vector<vector<int>> mergeArrays(vector<vector<int>>& nums1, vector<vector<int>>& nums2) {
        map <int,int> ump;
        for(auto &it:nums1)
            ump[it[0]] += it[1]; 
        for(auto &it:nums2)
            ump[it[0]] += it[1]; 
        nums1.clear();
        nums2.clear();
        for(auto &it:ump)
            nums1.push_back({it.first,it.second});
        return nums1;
    }
};
