public class Solution {
    public int PartitionArray(int[] nums, int k) {
        Array.Sort(nums);
        
        int result = 1;
        int prior = nums[0];
        
        foreach (int current in nums) 
            if (current - prior > k) {
                prior = current;
                
                result += 1;
            }
                
        return result;
    }
}
