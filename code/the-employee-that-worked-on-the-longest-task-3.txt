    public int HardestWorker(int n, int[][] logs)
    {
        int longestTask = logs[0][1], id = logs[0][0];
        for (int i = 1; i < logs.Length; i++)
        {
            int curr = logs[i][1] - logs[i - 1][1];
            if (curr > longestTask || curr == longestTask && id > logs[i][0])
            {
                longestTask = curr;
                id = logs[i][0];
            }
        }
        return id;
    }
