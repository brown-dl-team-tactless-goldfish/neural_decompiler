void travel(struct TreeNode* root,int* max,int cnt,int *ret)
{
    if(!root)
        return;
    if(cnt>*max)
    {
        *max=cnt;
        *ret=root->val;
    }

    travel(root->left,max,cnt+1,ret);
    travel(root->right,max,cnt+1,ret);
    return;
}

int findBottomLeftValue(struct TreeNode* root){
    int max=0,ret=root->val;
    travel(root,&max,0,&ret);
    return ret;
}
