int numOfSubarrays(int* arr, int arrSize, int k, int threshold){
    
    int sum = 0;
    int count_arrays = 0;
    for(size_t i = 0 ; i < arrSize ; i++)
    {
        sum += arr[i];
        
        // After we reach lenght of array that we need - start checking
        if(i >= k -1)
        {
            if((sum / k) >= threshold)
            {
                count_arrays++;
            }   
            sum -= arr[i - (k-1)];
        }
    }

    return count_arrays;
}
