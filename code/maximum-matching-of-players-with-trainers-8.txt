class Solution {
public:
    int matchPlayersAndTrainers(vector<int>& p, vector<int>& t) {
        sort(p.begin(),p.end());
        sort(t.begin(),t.end());
        int cnt = 0;
        for(int i = p.size() - 1 , j = t.size() - 1 ; i >= 0 && j >= 0 ;){
            if(p[i] <= t[j]) i--,j--,cnt++;
            else if(p[i] > t[j]) i--;
        }
        return cnt;
    }
};
