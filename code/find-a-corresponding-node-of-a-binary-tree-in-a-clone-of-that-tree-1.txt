# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def getTargetCopy(self, original: TreeNode, cloned: TreeNode, target: TreeNode) -> TreeNode:
                        
        self.res = None
        def helper(curNode1, curNode2):
            if (curNode1 == target):
                self.res = curNode2
                return
            
            if (self.res != None):
                return
            
            if (curNode1 != None and curNode2 != None):
                helper(curNode1.left, curNode2.left)
                helper(curNode1.right, curNode2.right)
            
        helper(original, cloned)
        return self.res
