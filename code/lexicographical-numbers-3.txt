    private IList<int> res = new List<int>();
    
    public IList<int> LexicalOrder(int n) {
        Backtrack(n, 0);
        
        return res;
    }
    
    private void Backtrack(int n, int cur)
    {
        for (int i = 0; i < 10; i++)
        {
            if (cur == 0 && i == 0)
                continue;
            
            if (cur * 10 + i <= n)
            {
                res.Add(cur * 10 + i);
                Backtrack(n, cur * 10 + i);
            }
        }
    }
