int arrayNesting(int* nums, int numsSize){
    int max = 0;
    int counter = 0;
    for(int i = 0; i < numsSize; i++)
    {
        if(nums[i] != -1)
        {
            int j = nums[i]; 
            counter = 0;
            while(j != i)
            {
                counter++;
                int temp = nums[j];
                nums[j] = -1;
                j = temp;
            }
            counter++;
            if(counter > max)
                max = counter;
        }
            
    }
    return max;
}
