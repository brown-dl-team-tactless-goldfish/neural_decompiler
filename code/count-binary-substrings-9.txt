int countBinarySubstrings(char * s){
    int i, prev = 0;
    int n = strlen(s);
    int ans = 0;
    int *count = (int*) malloc (n * sizeof(int));
    i = 1;
    count[0] = 1;
    while(i < n) {
        if (s[i] != s[i-1]) {
            prev = count[i-1];
            count[i] = 1;
        } else {
            count[i] = count[i-1] + 1;
        }
        if (count[i] <= prev)
            ans++;
        i++;
    }
    return ans;
}
