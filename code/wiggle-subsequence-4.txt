int wiggleMaxLength(int* nums, int numsSize){
    if(numsSize <= 1){
        return numsSize;
    }
    int curDiff = 0;
    int preDiff = 0;
    int result = 1;
    for(int i = 0; i < numsSize - 1; i++){
        curDiff = nums[i + 1] - nums[i];
        if((curDiff > 0 && preDiff <= 0) || (preDiff >= 0 && curDiff < 0)){
            result++;
            preDiff = curDiff;
        }
    }
    return result;
}
