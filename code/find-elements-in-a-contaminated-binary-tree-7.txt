class FindElements {
public:
    FindElements(TreeNode* root) {
        go(root);
    }
    
    bool find(int target) {
        return seen.find(target) != seen.end();
    }
private:
    unordered_set<int> seen;
    void go(TreeNode* root, int val=0) {
        seen.insert(val);
        root->val = val;
        if (root->left) go(root->left, 2*val+1);
        if (root->right) go(root->right, 2*val+2);
    }
};
