public class Solution 
{
    public int LongestDecomposition(string text) 
    {
        StringBuilder left = new StringBuilder();
        StringBuilder right = new StringBuilder();
        
        int count = 0;
        
        for(int i = 0; i < text.Length; i++)
        {
            left.Append(text[i]);
            right.Insert(0, text[text.Length - i - 1]);  //prepend as need to check palindrome
            
            if(left.ToString().Equals(right.ToString(), StringComparison.Ordinal))
            {
                count ++;
                left.Clear();
                right.Clear();
            }            
        }
        
        return count;
    }
}
