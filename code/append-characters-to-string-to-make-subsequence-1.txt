#include <string.h>

int appendCharacters(char * s, char * t){
    int ans = 0;
    int t_idx = 0;
    int s_len = strlen(s), t_len = strlen(t);
    
    for (int i = 0; (i < s_len) && (t_idx < t_len); i++) {
        if (s[i] == t[t_idx]) {
            t_idx++;
        }
    }
    
    ans = t_len - t_idx;
    
    return ans;
}
