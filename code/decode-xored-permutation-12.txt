class Solution {
public:
    vector<int> decode(vector<int>& e)
    {
        int n = e.size() + 1;
        vector<int> p(e.size() + 1);
        for (int i = 1; i <= n; i++) p[0] ^= i;
        for (int i = 1; i < e.size(); i += 2) p[0] ^= e[i];
        for (int i = 0; i < e.size(); i++) p[i + 1] = p[i] ^ e[i];
        return p;
    }
};
