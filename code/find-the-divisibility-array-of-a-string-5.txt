class Solution {
public:
    vector<int> divisibilityArray(string word, int m) {
        vector<int> result;
        long long temp=0;
        for(char c : word){
            temp = temp*10+(c-'0');
            if(temp%m==0) result.push_back(1);
            else result.push_back(0);
            temp%=m;
        }
        return result;
    }
};
