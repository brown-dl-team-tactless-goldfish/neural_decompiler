int maxCount(int m, int n, int** ops, int opsSize, int* opsColSize){
	if(!ops || opsSize == 0) return m*n;

	int row_min = INT_MAX;
	int col_min = INT_MAX;

	for(int i = 0; i < opsSize; i++)
	{
		row_min = row_min < ops[i][0] ? row_min : ops[i][0];
		col_min = col_min < ops[i][1] ? col_min : ops[i][1];
	}

	return row_min * col_min;
}
