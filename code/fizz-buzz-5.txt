char **fizzBuzz(int n, int* returnSize) {
	char** modified = (char**)malloc(n * sizeof(char*));
	*returnSize = n;
	for (int i = 1; i <= n; i++) {
		int num = i;
		modified[i - 1] = (char*)malloc(8 * sizeof(char));
		if (num % 3 == 0 && num % 5 == 0) modified[i-1] = "FizzBuzz";
		else if (num % 3 == 0) modified[i-1] = "Fizz";
		else if (num % 5 == 0) modified[i-1] = "Buzz";
		else sprintf(modified[i - 1], "%d", num);
	}
    return modified;
}
