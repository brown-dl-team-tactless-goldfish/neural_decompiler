    int minOperations(int n) {
        int count = 0; // number of operations
        
        while (n) {
            int low = n & 3; // take lowest 2 bits to check the lowest bit
            if (low == 3) { // lowest bit 1 has continuous bit 1
                // apply one operation: add 1 bit
                ++count;
                ++n;
            } else if (low == 1) { // lowest bit 1 is isolated
                // apply one operation: remove 1 bit
                ++count;
            }
            
            n >>= 1; // check next bit
        }
        
        return count;
    }
