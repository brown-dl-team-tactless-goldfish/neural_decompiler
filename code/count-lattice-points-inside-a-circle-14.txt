    set<pair<int,int>> st;
    for(int i=0;i<c.size();i++)
    {
        int x=c[i][0];
        int y=c[i][1];
        int r=c[i][2];
        for(int i=x-r;i<=x+r;i++)
        {
            for(int j=y-r;j<=y+r;j++)
            {
                if((i-x)*(i-x)+(j-y)*(j-y)<=r*r)
                    st.insert({i,j});
            }
        }
    }
    return st.size();
}
