int longestPalindrome(char * s){
    int* CHAR_HOLDER = (int*)calloc(58,sizeof(int));
    int MAX_LEN_PALINDROME = 0;
    //copy ll the characters and count number of occurance 
    for(int idx = 0 ; idx < strlen(s); idx++){CHAR_HOLDER[s[idx] - 'A']++;}
    for(int table_idx = 0 ; table_idx < 58; table_idx++){
        MAX_LEN_PALINDROME += (CHAR_HOLDER[table_idx]/2)*2;
    }
    free(CHAR_HOLDER);
    //add termination character if the palindrome length is less than the input string
    if(MAX_LEN_PALINDROME < strlen(s)){MAX_LEN_PALINDROME += 1;}
    return MAX_LEN_PALINDROME;
}
