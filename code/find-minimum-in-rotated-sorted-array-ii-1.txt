int findMin(int* nums, int numsSize){
    int i, count=0;
    for(i=0;i<numsSize-1;i++)
    {
        if(nums[i]<=nums[i+1])
        {
            count++;
            continue;
        }
        else
        {
            return nums[i+1];
        }
    }
    if(count+1==numsSize) //rotated by numssize
    {
        return nums[0];
    }
    return 0;
}
