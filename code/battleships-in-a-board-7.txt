#define IS_HIT(c) ((c) == 'X')

int countBattleships(char** bd, int row, int col) {
    int count = 0;
    
    for (int i = 0; i < row; ++i) {
        for (int j = 0; j < col; ++j) {
            if (IS_HIT(bd[i][j])) {
                bool left = i == 0 || !IS_HIT(bd[i-1][j]);
                bool top = j == 0 || !IS_HIT(bd[i][j-1]);
                if (left && top) {
                    ++count;
                }
            }    
        }
    }
        
    return count;    
}
