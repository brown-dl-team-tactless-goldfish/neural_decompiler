int longestOnes(int* A, int ASize, int K) {
    int i = 0;
    int j;
    int numOfZero = 0;
    int max = 0;
    int current = 0;
    for(j = 0;j<ASize;j++){
        if(A[j]==0){
            numOfZero++;
        }
        if(numOfZero>K){
            if(A[i] == 0){
                numOfZero--;
            }
            i++;
        }
        current = j-i+1;
        if(current>max){
            max = current;
        }      
    }
    return max;
}
