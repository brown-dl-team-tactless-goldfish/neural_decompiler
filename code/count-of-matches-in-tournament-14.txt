int numberOfMatches(int n)
{
    int match,advance,sum = 0,team;
    team = n;
    for (int i=0;i<10;i++)
    {

        if( team%2 == 0){
            match = team/2;
            advance = team-match;
            team = advance;
        }
        else{
            match = (team-1)/2;
            advance = team-match;
            team = advance ;
        }
            sum = sum + match;
    }
    return sum;
}
