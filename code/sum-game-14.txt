class Solution {
public:
    bool sumGame(string num) {
        int a=0;
        int b=0;
        int count1=0;
        int count2=0;
        int n = num.size();
        for(int i=0;i<num.length()/2;i++){
            if(num[i]!='?'){
                a+=num[i]-'0';
            }
            else{
                count1++;
            }
            if(num[n-1-i]!='?'){
                b+=num[n-1-i]-'0';
            }
            else{
                count2++;
            }
        }
        if((count1+count2)==0&&a==b)return false; //No scope of changes, if sum of two sides is already equal then return false
        if((count1+count2)%2==1){ // Alice got one more chance so she will definitely gonna win
                return true;
            }
        
        if((a-b)==9*(count2-count1)/2||(b-a)==9*(count1-count2)/2){ //case when bob can make the sum equal by choosing 9-x, where alice has choosen x. This will only happen when the sum is more on one side and more number of '??..' is on the other side.
            // this condition only holds when the difference is 9 times the no. of moves each player plays.
            return false;
        }
    return true; //this will make alice win in case the difference is not 9 times the no. of effective moves each player plays. Effective in the sense we cancell '?' on one side with other side upto the time when '?' is only on one side
    }
};
