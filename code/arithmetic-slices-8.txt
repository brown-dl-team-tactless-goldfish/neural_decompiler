int numberOfArithmeticSlices(int* A, int ASize){
    if(ASize<2) return 0;
    int i, d, n=1, sum=0;
    d = A[1]-A[0];
    
    for(i=1; i<ASize-1; i++){
        if(A[i+1]-A[i] == d) n++;
        else{
            d = A[i+1]-A[i];
            sum += (n-1)*(n)/2;
            n=1;
        }
        
    }
    if(A[i]-A[i-1] == d) sum += (n-1)*(n)/2;
    return sum;
}
