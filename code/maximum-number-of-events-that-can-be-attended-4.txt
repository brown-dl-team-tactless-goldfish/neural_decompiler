    int maxEvents(vector<vector<int>>& e) {
        priority_queue<int, vector<int>, greater<int>> pq;
        sort(e.begin(), e.end());
        
        int res = 0;
        for(int d = 1, idx = 0; d <= 100000; d++) {
            while(idx < e.size() && e[idx][0] == d) pq.push(e[idx++][1]);
            while(!pq.empty() && pq.top() < d) pq.pop();
            if(!pq.empty()) {
                pq.pop();
                res++;
            }
        }
        return res;
    }
