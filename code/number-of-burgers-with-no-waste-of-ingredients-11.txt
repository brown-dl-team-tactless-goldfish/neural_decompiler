class Solution {
public:
    vector<int> numOfBurgers(int t, int c) {
        if (t > c * 4 || t < c * 2 || t % 2 == 1) return {};
        vector<int> ans(2);
        int minDif = c * 2;
        int extra = t - minDif;
        ans[0] = extra / 2;
        ans[1] = c - ans[0];
        return ans;
    }
};
