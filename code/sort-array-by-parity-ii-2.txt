int* sortArrayByParityII(int* A, int ASize, int* returnSize)
{
    int *res = malloc(sizeof(int) * ASize);
    *returnSize = ASize;
    int *odd, *even, i = 0;
    
    odd = res + 1;
    even = res;
    
    while(i < ASize)
    {
        if(A[i] % 2)
        {
            *odd = A[i];
            odd +=2;
            i++;
            
        }
        else
        {
            *even = A[i];
            even += 2;
            i++;
        }
    }
    return res;
}
