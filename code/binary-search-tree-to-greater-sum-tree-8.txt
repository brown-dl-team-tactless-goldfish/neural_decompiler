struct TreeNode* bstToGst(struct TreeNode* root) {
    if (root == NULL) {
        return root;
    }
    bstToGstRecursive(root, 0);
    return root;
}

int bstToGstRecursive(struct TreeNode* node, int sum) {
    if (node == NULL) {
        return sum;
    }
    sum = node->val + bstToGstRecursive(node->right, sum);
    node->val = sum;
    sum = bstToGstRecursive(node->left, sum);
    return sum;
}
