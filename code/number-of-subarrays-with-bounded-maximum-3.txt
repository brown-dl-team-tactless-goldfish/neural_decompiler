public class Solution {
    public int NumSubarrayBoundedMax(int[] nums, int left, int right) {
        
        if(nums == null || nums.Length == 0)
            return 0;
        
        int cnt = 0, invalidIdx = -1, validSubarrayCnt = 0;
        for(int i = 0; i < nums.Length; i++)
        {
            if(nums[i] < left)
                validSubarrayCnt = validSubarrayCnt;
            else if(nums[i] > right)
            {
                validSubarrayCnt = 0;
                invalidIdx = i;
            }
            else
                validSubarrayCnt = i - invalidIdx;
            
            cnt += validSubarrayCnt;
        }
        
        return cnt;
    }
}
