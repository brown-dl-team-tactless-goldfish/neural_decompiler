class Solution:
    def makeFancyString(self, s: str) -> str:
        n = len(s)
        if (n < 3):
            return s
        
        res = s[:2]
        for i in range(2, n):
            if (s[i] == s[i-1] and s[i] == s[i-2]):
                continue
            else:
                res += (s[i])
                
        return res
