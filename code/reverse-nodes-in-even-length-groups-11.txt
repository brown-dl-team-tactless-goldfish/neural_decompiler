ListNode* relgInt(ListNode* head, int len) {
    if (!head)
        return head;
    ListNode *tail = head;
    int real_len = 1;
    for (; real_len < len && tail->next; tail = tail->next, real_len++);
    if (real_len % 2 == 1) {
        tail->next = relgInt(tail->next, len+1);
        return head;
    }
    ListNode dummy, *next;
    dummy.next = tail->next;
    tail = head;
    for (int i = 0; i < real_len && head; i++) {
        next = head->next;
        head->next = dummy.next;
        dummy.next = head;
        head = next;
    }
    tail->next = relgInt(tail->next, len+1);
    return dummy.next;
}

ListNode* reverseEvenLengthGroups(ListNode* head) {
    return relgInt(head, 1);
}
