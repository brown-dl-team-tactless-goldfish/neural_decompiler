class Solution {
    int zero(int n){
        int c=0;
        while(n>0){
            if(n%10 ==0) c++;
            n /= 10;
        }
        return c;
    }
public:
    vector<int> getNoZeroIntegers(int n) {
        for(int i=1; i<=10000; ++i){
            if(zero(i)==0 && zero(n-i)==0){
                return {i,n-i};
            }
        }
        return {};
    }
};
