int maximumTop(int* nums, int numsSize, int k){
    if(numsSize == 1 && k%2 == 1)
        return -1;
    
    int max = 0;
    for(int i = 0; i < k-1 ; i++){
        if(i >= numsSize)
            break;
        if(nums[i] > max)
            max = nums[i];
    }
    
    if(numsSize > k)
        max = fmax(max, nums[k]);
    return max;
}
