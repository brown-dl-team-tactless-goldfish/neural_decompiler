class Solution {
public:
    int numSubarraysWithSum(vector<int>& v, int s){
        int ans=0;
        unordered_map<int,int>mp;
        mp[0]=1;
        int sum=0;
        for(int i=0;i<v.size();i++){
            sum+=v[i];
            ans+=mp[sum-s];
            mp[sum]++;
        }
        return ans;
    }
};
