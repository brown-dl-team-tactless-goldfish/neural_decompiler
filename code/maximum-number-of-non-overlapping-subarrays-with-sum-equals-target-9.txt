    int maxNonOverlapping(vector<int>& nums, int target) {
        int cnt = 0, sum = 0;
        unordered_set<int> sums;
        for(auto x:nums){
            sum += x;
            if(sums.count(sum-target) || sum == target){
                cnt++;
                sum=0;
                sums.clear();
            }
            else sums.insert(sum);
        }
        return cnt;
    }
