int firstUniqChar(char * s){
    int alph[26]={0};
    
    if (s==NULL)
        return -1;
    
    for (int i=0;s[i]!='\0';i++)
    {
        alph[s[i] - 'a'] = alph[s[i] - 'a']+1;
    }
    
    for(int i=0;s[i]!='\0';i++)
    {
        if (1 == alph[s[i] - 'a'])
            return i;
    }
    
    return -1;
}
