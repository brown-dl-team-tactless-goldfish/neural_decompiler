class Solution {
public:
    string maximumBinaryString(string b) {
        int n=b.size();
        int index=-1;
        for(int i=0;i<n;i++){
            if(b[i]=='0'){
                index=i;
                break;
            }
        }
        if(index==-1)return b;
        int nb0=0;
        for(int i=index;i<n;i++){
            if(b[i]=='0'){
                nb0++;
                b[i]='1';
            }    
        }
        b[index+nb0-1]='0';

        return b;
    }
};
