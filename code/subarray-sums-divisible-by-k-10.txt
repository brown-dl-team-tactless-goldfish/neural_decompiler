class Solution {
public:
    int subarraysDivByK(vector<int>& nums, int k) {
        int n       = nums.size();
        int count = 0;
        int   sum = 0;
        unordered_map<int, int> mp;
        mp[0] = 1;
        for(int &num : nums) {
            sum += num;
            int remain = ((sum%k)+k)%k;
            if(mp.find(remain) != mp.end()) {
                count += mp[remain];
            }
            mp[remain] += 1;
        }
        
        return count;
    }
};
