class Solution {
public:
    int addMinimum(string word) {
        
        int len = word.length();

    string temp = word;

    if (temp[0] == 'b')
    {
        temp = "a" + temp;
    }
    if (temp[0] == 'c')
    {
        temp = "ab" + temp;
    }
    len = temp.length();
    if (temp[len - 1] == 'a')
    {
        temp = temp + "bc";
    }
    if (temp[len - 1] == 'b')
    {
        temp = temp + "c";
    }

    for (int i = 0; temp[i + 1] != '\0'; i++)
    {
        if (temp[i] == 'a' && temp[i + 1] != 'b')
        {
            temp.insert(i + 1, "b");
        }
        if (temp[i] == 'b' && temp[i + 1] != 'c')
        {
            temp.insert(i + 1, "c");
        }
        if (temp[i] == 'c' && temp[i + 1] != 'a')
        {
            temp.insert(i + 1, "a");
        }
    }

    int ans = temp.length() - word.length();
    return ans;
    }
};
