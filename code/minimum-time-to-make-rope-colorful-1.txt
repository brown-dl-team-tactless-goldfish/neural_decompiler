int minCost(char * colors, int* neededTime, int neededTimeSize){
    if(neededTimeSize == 1) return 0;
    int sum = 0, maxTime = 0;
    for(int i = 0; i < neededTimeSize; i++)
    {
        if(i > 0 && colors[i] != colors[i - 1])
            maxTime = 0;
        sum += maxTime < neededTime[i] ? maxTime : neededTime[i];
        maxTime = maxTime > neededTime[i] ? maxTime : neededTime[i];
    }
    return sum;
}
