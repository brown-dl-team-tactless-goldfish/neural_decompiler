int* postorder(struct Node* root, int* returnSize) {
    int sz = 10000, *a = malloc(sizeof(int [sz])), i = sz - 1, p = 0, j;
    struct Node *s[10000] = { root }, *c;
    while (root && p >= 0)
        for (a[i--] = (c = s[p--])->val, j = 0 ; j < c->numChildren ; s[++p] = c->children[j++]);
    return memcpy(a, a + i + 1, sizeof(int [*returnSize = sz - i - 1]));
}
