class Solution {
public:
    int deleteGreatestValue(vector<vector<int>>& grid) {
        for(vector<int>& row: grid){
            sort(row.begin(), row.end(), greater<int>());
        }

        int res = 0;
        for(int i=0;i<grid[0].size();i++){
            int tmp = INT_MIN;
            for(int j=0;j<grid.size();j++){
                tmp = max(tmp, grid[j][i]);
            }

            res += tmp;

        }

        return res;
    }
};
