int minSteps(char * s, char * t){
    int arr_s[26] = {0}, arr_t[26] = {0}, i, answer = 0,s_len = strlen(s),t_len = strlen(t);
    for(i = 0; i < s_len; i++){
        arr_s[s[i] - 'a']++;
    }
    for(i = 0; i < t_len; i++){
        arr_t[t[i] - 'a']++;
    }
    for(i = 0; i < 26; i++){
        answer += abs(arr_s[i] - arr_t[i]);
    }
    return answer;
}
