public class Solution {
    public double AngleClock(int hour, int minutes) {
        double hourAngle = (hour == 12 ? 0 : hour) * 30 + minutes * 0.5,
               minutesAngle = minutes * 6;
        
        return Math.Min(Math.Min(360 - minutesAngle + hourAngle, Math.Abs(hourAngle - minutesAngle)), 360 - hourAngle + minutesAngle);
    }
}
