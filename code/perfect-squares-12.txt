void numSquaresub(int n, int count, int *min) {
    
    if(n == 0){
        *min = *min < count?  *min : count;
    }   
    for(int i = 1; i*i <= n ; i++){
            numSquaresub(n-i*i, count+1, min);
    }
}


int numSquares(int n) {
    int min = INT_MAX;
    
    numSquaresub(n, 0, &min);
    
    return min;
}
