int reverse(int val){
    int ret = 0 ;
    while(val){
        ret = ret*10 + val%10 ;
        val /= 10 ;
    }
    return ret ;  
}
bool sumOfNumberAndReverse(int num){
    for(int i = num/2 ; i <= num; i++){
        if((i + reverse(i)) == num)
            return true ;
    }
    return false ;
}
