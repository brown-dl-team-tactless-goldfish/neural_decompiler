int countTriples(int n){
    int count = 0, i, j;
    // All integer combination below 250. 
    int reg[39] = {5, 13, 17, 25, 29, 37, 41, 53, 61, 65, 65, 73, 85, 85, 89, 97, 101, 109, 113, 125, 137, 145, 145, 149, 157, 169, 173, 181, 185, 185, 193, 197, 205, 205, 229, 221, 221, 233, 241};
    
    for(i = 5; i <= n; i++){
        for(j = 0; j < 39; j++){
            if(i % reg[j] == 0)
                count++;
        }
    }
    
    return count * 2;
}
