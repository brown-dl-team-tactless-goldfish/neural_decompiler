
int getWinner(int* arr, int arrSize, int k){
    int i,j;
    int count;
    // search for a winner inside the array.
    // if not found the answer is the biggest element.
    for(i=0,j=1,count=0; j<arrSize; ++j) {
        if (arr[i] > arr[j]) 
            ++count;
        else {
            count = 1; i = j;
        }
        if (count == k) break;
    }
    return arr[i];
}
