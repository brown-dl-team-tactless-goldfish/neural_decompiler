class Solution {
public:
    bool findSubarrays(vector<int>& nums) {
        unordered_map<int,int> map;
        int n=nums.size();
        for(int i=1;i<n;i++){
            int s=0;
            s=nums[i]+nums[i-1];
            if(map.find(s)!=map.end())return true;
            map[s]=i-1;
        }
        return false;
    }
};
