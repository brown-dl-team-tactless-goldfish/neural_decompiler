class Solution {
public:
    int partitionDisjoint(vector<int>& a) {
        int minnum=a[0],maxnum=a[0];
        int index=0;
        for(int i=1;i<a.size();i++){
            if(a[i]<minnum){
                index=i;
                minnum=maxnum;
            }
            maxnum=max(a[i],maxnum);
        }
        return index+1;
    }
};
