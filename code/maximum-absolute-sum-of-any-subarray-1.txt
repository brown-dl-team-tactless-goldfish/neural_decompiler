int maxAbsoluteSum(int* nums, int numsSize){
        int maxsofar=0,maxendinghere=0;
        int minsofar=0,minendinghere=0;
        int temp;
        for(int i=0;i<numsSize;i++){
                temp=maxendinghere+nums[i];
                maxendinghere=(temp>0)?temp:0;
                maxsofar=(maxsofar>maxendinghere)?maxsofar:maxendinghere;
        }
        for(int i=0;i<numsSize;i++){
                temp=minendinghere+nums[i];
                minendinghere=(temp<0)?temp:0;
                minsofar=(minsofar<minendinghere)?minsofar:minendinghere;
        }
        temp=fabs(minsofar);
        return (temp>maxsofar)?temp:maxsofar;
}
