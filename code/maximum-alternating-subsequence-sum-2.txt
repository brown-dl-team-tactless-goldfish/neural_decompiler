class Solution {
public:
    long long maxAlternatingSum(vector<int>& nums) {

        long long res = 0;

        long long prefix_max = 0;
        for(int i=0;i<nums.size();i++) {
            res = max(res, prefix_max + nums[i]);

            if(i>0 && nums[i-1] > nums[i]) {
                prefix_max += nums[i-1] - nums[i];
            }

            res = max(res, prefix_max);
            res = max(res, (long long)nums[i]);
        }

        return res;
    }
};
