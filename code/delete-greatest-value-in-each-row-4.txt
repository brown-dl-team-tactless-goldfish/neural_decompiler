class Solution {
public:
  int deleteGreatestValue(std::vector<std::vector<int>>& grid) {
    int row = grid.size(), col = grid[0].size();
    int result = 0;

    for (auto&& row : grid) {
      std::sort(row.begin(), row.end());
    }

    for (int c = 0; c < col; ++c) {
      int max = std::numeric_limits<int>::min();
      for (int r = 0; r < row; ++r) {
        if (max < grid[r][c]) {
          max = grid[r][c];
        }
      }
      result += max;
    }
    return result;
  }
};
