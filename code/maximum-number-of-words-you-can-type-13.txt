class Solution {
public:
    int canBeTypedWords(string text, string brokenLetters) {
        vector<bool> brokenLetterHash(26, false);
        for (auto c : brokenLetters) brokenLetterHash[c - 'a'] = true;
        
        string token;
        stringstream ss(text);
        int canBeTypedCount = 0;
        while (getline(ss, token, ' ')) {
            int n = token.size();
            
            bool flag = true;
            for (auto c : token) {
                if (brokenLetterHash[c - 'a']) {
                    flag = false;
                    break;
                }
            }
            canBeTypedCount += flag;
        }
        return canBeTypedCount;
    }
};
