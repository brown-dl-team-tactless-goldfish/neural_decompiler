public class Solution {
    public int MinNumber(int[] nums1, int[] nums2) {
        int[] common = nums1.Intersect(nums2).ToArray();
        if(common.Any())
            return common.Min(); //smallest number in common elements
        
        int min1 = nums1.Min();
        int min2 = nums2.Min();
        
        return (Math.Min(min1, min2)) * 10 + (Math.Max(min1, min2)); //build smallest 2 digit number
        
    }
}
