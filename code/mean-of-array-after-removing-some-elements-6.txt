class Solution:
    def trimMean(self, arr: List[int]) -> float:
        n = len(arr)
        arr.sort()
        
        print ("n = ", n)
        print ("arr = ", arr)
        
        removed_number = n // 20
        
        arr = arr[removed_number:n-removed_number]
        
        print ("arr = ", arr)
        
        return sum(arr)/len(arr)
