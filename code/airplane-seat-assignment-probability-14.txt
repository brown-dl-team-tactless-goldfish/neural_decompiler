class Solution {
public:
	
	// Probability of a person getting his own seat when 2 or more seats are available - 0.5
	// Probability of a person not getting his own seat  - 0.5

	// If there is only one seat, he will get his own seat.
	// this sums up the required solution.
	
    double nthPersonGetsNthSeat(int n) {
        return n == 1 ? 1 : 0.5;
    }
};
