    int minKBitFlips(vector<int>& A, int K) {
        queue<int> q;
        int cnt = 0;
        for(int i=0;i<A.size();i++){
            while(!q.empty() && q.front() + K - 1 < i) q.pop();
            if((A[i] - q.size()) % 2 == 1) continue;
            if(i+K > A.size()) return -1;
            cnt++;
            q.push(i);
        }
        return cnt;
    }
