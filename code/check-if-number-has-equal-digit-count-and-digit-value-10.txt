bool digitCount(char * num){
    // int + '0' is used to convert the int to a char
    for (int i = 0; i < strlen(num); ++i) {
        int count = 0;
        for (int j = 0; j < strlen(num); ++j) {
            if ((i + '0') == *(num + j))
                count++;
        }

        if (*(num + i) != (count + '0'))
            return false;
    }
    return true;
}
