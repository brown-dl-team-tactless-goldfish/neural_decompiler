class Solution {
public:
    int minSteps(string s, string t) {
        int cnt1[26] = {0}; // counter for s
        int cnt2[26] = {0}; // counter for t
        for (auto c : s) {
            cnt1[c - 'a']++;
        }
        for (auto c : t) {
            cnt2[c - 'a']++;
        }
        int ans = 0;
        for (int i = 0; i < 26; i++) {
            ans += abs(cnt1[i] - cnt2[i]);
        }
        return ans;
    }
};
