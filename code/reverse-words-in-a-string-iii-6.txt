char * reverseWords(char * s){

    long int len = strlen(s);
    char* arry = calloc(len, sizeof(char));
    int cnt = 0;
    
    for(int i = 0; i <= len; ++i)
    {
        if((*(s+i) == ' ') || (*(s+i) == '\0'))
        {
            while (cnt)
            {
                *(s + i - cnt) = *(arry + cnt - 1);
                cnt--;
            }
        } else {
            *(arry + cnt++) = *(s + i);
        }
    }
    
    free(arry);
    return s;
}
