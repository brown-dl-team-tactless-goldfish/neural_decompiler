class Solution(object):
    def minOperations(self, grid, x):
        v = []
        for i in range(len(grid)):
            for j in range(len(grid[0])):
                v.append(grid[i][j])

        v.sort()
        m, ans = v[len(v) / 2], 0
        for i in v:
            if abs(i - m) % x: return -1
            ans += abs(i - m)
        
        return ans / x
