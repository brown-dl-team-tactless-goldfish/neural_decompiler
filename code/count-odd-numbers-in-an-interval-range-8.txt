int countOdds(int low, int high){
    int count=(high-low);
    if(count%2==0)
    {
        count= count/2;
        if(high%2!=0 || low%2!=0)
        {
            count++;
            return count;
        }
        else
        {
            return count;
        }
    }
    else
    {
        count=((high-low+1)/2);
        return count;
    }

}
