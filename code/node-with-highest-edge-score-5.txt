class Solution {
public:
    int edgeScore(vector<int>& edges) {
        map<int, long long> mp;
        for(int i = 0; i < edges.size(); i++){
            mp[edges[i]] += i;
        }
        int a = 0;
        int b = 0;
        for(auto it = mp.begin(); it != mp.end(); ++it)
            if(it->second > a){
                a = it->second;
                b = it->first;
            }
            return b;
    }
};
