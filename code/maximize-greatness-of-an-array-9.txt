class Solution {
public:
    int maximizeGreatness(vector<int>& nums) {
        multiset<int> ms;
        int ans = 0;
        
        for(auto x : nums)
            ms.insert(x);
        
        for(auto x : nums)
        {
            auto it = ms.upper_bound(x);
            
            if(it != ms.end())
            {
                //cout<<x<<" "<<*it<<"\n";
                ans++;
                ms.erase(it);
            }
        }
        
        return ans;
        
    }
};
