class Solution {
public:
    int maxNumOfMarkedIndices(vector<int>& nums) {
        sort(nums.begin(),nums.end());
        int ans = 0, i = 0, j, n = nums.size();
        j = n-1;
        for(i = n/2-1; i>=0;i--){
            if(nums[j]>=nums[i]*2){
                ans += 2;
                j--;
            }
            
        }
        return ans;
    }
};
