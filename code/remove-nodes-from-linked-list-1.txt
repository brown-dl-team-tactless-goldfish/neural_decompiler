struct ListNode* check(struct ListNode* head, int* now){
    if (head == NULL){
        return NULL;
    }
    head->next = check(head->next, now);
    if (head->val >= *now){
        *now = head->val;
        return head;
    } 
    return head->next;
}

struct ListNode* removeNodes(struct ListNode* head){
    int now = 0;
    return check(head, &now);
}
