#include <stdio.h>
#include<stdlib.h>

typedef struct node
{   int val;
    struct node *left;
    struct node *right;
} node;

node *root=NULL;

void insert(node *root1,int value)
{
    if(root==NULL)
    {
        node *temp=(node*)malloc(sizeof(node));
        temp->val=value;
        temp->left=NULL;
        temp->right=NULL;
        root=temp;
        root1=root;
        return;
    }


    if(root1==NULL && root!=NULL)
    {
        node *temp=(node*)malloc(sizeof(node));
        temp->val=value;
        temp->left=NULL;
        temp->right=NULL;
        root1=temp;
        return;
    }


    if(root1->val >value)
    {
        insert(root1->left, value);
    }
    else
    {
        insert(root1->right, value);
    }

    return;
}


void display(node *root1)
{
    if(root1==NULL)
    {
        return;
    }
    while(root1 !=NULL)
    {
        printf("%d\n", root1->val);
        display(root1->left);
        display(root1->right);
        return;
    }
}


int main()
{
    insert(root,4);
    insert(root,12);
    insert(root,2);
    insert(root,55);
    display(root);
    return 0;
}
