struct ListNode* deleteMiddle(struct ListNode* head){
    if (!head || !head->next)
        return NULL;
    
    struct ListNode *__return, *prev;
    __return = head;
    
    // Calculate length
    int n = 0;
    for (struct ListNode *i = head; i; i = i->next)
        n++;
    
    int mid = n / 2;
    prev = head;
    head = head->next; 
    
    while (--mid) {
        prev = head;
        head = head->next;
    }
    
    prev->next = head->next;
    return __return;
}
