int sort(const void *a, const void *b)
{
    return *(int*)a - *(int*)b;
}

int maximumGap(int* nums, int numsSize){
    if(numsSize<2)
    {
        return 0;
    }
    int d=INT_MIN;
    qsort(nums,numsSize,sizeof(int),sort);
    int i=0,j=1;
    for(;j<numsSize;i++,j++)
    {
        if(nums[j]-nums[i]>d)
        {
            d=nums[j]-nums[i];
        }
    }
    return d;
}
