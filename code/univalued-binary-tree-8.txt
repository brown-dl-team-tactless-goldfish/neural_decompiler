bool isUnivalTreeUtil(struct TreeNode *node, int val) {
    if(!node) return true;
    
    if(node->val != val) return false;
    
    return isUnivalTreeUtil(node->left, val) && isUnivalTreeUtil(node->right, val);
}

bool isUnivalTree(struct TreeNode* root){
    return isUnivalTreeUtil(root, root->val);    
}
