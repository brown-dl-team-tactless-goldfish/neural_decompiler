int maxRepeating(string seq, string word) {
    int pos = seq.find(word), res = 0, sz = word.size();
    while (pos != string::npos) {
        int i = pos + sz, j = 0;
        while (i < seq.size() && seq[i] == word[j % sz]) {
            ++i;
            ++j;
        }
        res = max(res, 1 + j / sz);
        pos = seq.find(word, pos + 1);
    }
    return res;
}
