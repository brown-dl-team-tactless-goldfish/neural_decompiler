public class Solution {
    public IList<int> Postorder(Node root) {
        Stack<Node> nodes = new Stack<Node>();
        List<int> results = new List<int>();
        nodes.Push(root);
        
        while (nodes.Count != 0){
            var n = nodes.Pop();
            if (n == null){
                continue;
            }
            else if (n.children == null){
                results.Add(n.val);
            }
            else{
                nodes.Push(n);
                foreach (var c in n.children.Reverse()){
                    nodes.Push(c);
                }    
                
                n.children = null;
            }
        }
        
        return results;
    }
}
