char * removeStars(char * s){
char* return_string = (char*)calloc(strlen(s)+1,sizeof(char));
int return_string_ptr =0;
for(int idx =0 ;idx < strlen(s); idx++){
    if(s[idx] == '*'){
       if(return_string_ptr > 0){
           return_string[--return_string_ptr] = '\0';
        }
    }
    else if(s[idx] != '*'){
        return_string[return_string_ptr++] = s[idx];
    }
}
return return_string;
}
