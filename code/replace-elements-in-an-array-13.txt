public:
    
    vector<int> arrayChange(vector<int>& nums, vector<vector<int>>& op) {
        unordered_map<int,int> map;
        
        
        for(int i = 0; i < nums.size(); i++) map[nums[i]] = i;
        
        
        for(int i = 0; i < op.size(); i++) {
            nums[map[op[i][0]]] = op[i][1];
            map[nums[map[op[i][0]]]] = map[op[i][0]];
        }
        
        return nums;
    }
    
};
