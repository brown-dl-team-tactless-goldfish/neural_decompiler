class Solution {
public:
    bool isValid(string s) {
        stack<char>st;
        char currtp;
        for(auto ch:s)
        {
            if(ch=='a' || ch=='b')
            st.push(ch);
            else
            {
                // ch is c
                if(st.size()<2)
                return false;
                currtp= st.top();
                if(currtp != 'b')
                return false;
                st.pop();
                currtp= st.top();
                if(currtp != 'a')
                return false;
                st.pop();
            }
        }
        if(st.empty())
        return true;
        return false;
    }
};
