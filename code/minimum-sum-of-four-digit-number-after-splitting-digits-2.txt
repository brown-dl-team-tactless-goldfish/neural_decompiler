int cmp(const void *a,const void *b)
{
    return(*(int *)a-*(int *)b);
}
int minimumSum(int n)
{
    int a=n%1000;
    int b=a%100;
    int *x=(int *)malloc(4*sizeof(int));
    x[0]=n/1000;
    x[1]=a/100;
    x[2]=b/10;
    x[3]=b%10;
    qsort(x,4,sizeof(int),cmp);
    int sum=(x[0]*10)+x[2]+(x[1]*10)+x[3];
    return sum;

}
