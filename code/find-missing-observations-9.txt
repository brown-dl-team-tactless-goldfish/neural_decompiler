    vector<int> missingRolls(vector<int>& rolls, int mean, int n) {
        int sum = mean*(n+rolls.size()) - accumulate(rolls.begin(), rolls.end(), 0);
        if(sum < n || sum > 6*n) return {};
        vector<int> res;
        for(int i=n;i>0;i--){
            int avg = sum / i;
            res.push_back(avg);
            sum -= avg;
        }
        return res;
    }
