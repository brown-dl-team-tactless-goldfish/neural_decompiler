int rotatedDigits(int N) {
    int i, Num, Dig, Sum = 0;
    int FlagA, FlagB;
    
    for(i = 1; i <= N; i++)
    {
        Num = i;
        FlagA = 0;
        FlagB = 0;
        while(Num > 0)
        {
            Dig = Num % 10;
            if(Dig == 2 || Dig == 5 || Dig == 6 || Dig == 9) FlagA = 1;
            if (Dig == 3 || Dig == 4 || Dig == 7) FlagB = 1;
            Num = Num / 10;
        }
        if(FlagA == 1 && FlagB == 0) Sum++;
    }
    return Sum;
}
