  void converstion_SLL(struct TreeNode* root, struct TreeNode** pre){
       
        if(!root)
            return;
            
        converstion_SLL(root->left,pre);
        
        struct TreeNode *tmp = root->right;   
        
    
    
        if(root->left){
            (*pre)->right=root->right;
            root->right=root->left;
            root->left=NULL;
        }else
            (*pre) = root;
        
        converstion_SLL(tmp,pre);   //root->right is tmp 
        
    }


void flatten(struct TreeNode* root) {
    
    struct TreeNode* prev=NULL;
    
    converstion_SLL(root,&prev);


}
