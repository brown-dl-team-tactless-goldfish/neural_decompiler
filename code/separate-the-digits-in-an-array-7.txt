public class Solution {
    public int[] SeparateDigits(int[] nums) {
        List<int> list = new List<int>();
            foreach (int num in nums)
            {
                string numString = num.ToString();
                foreach (char c in numString)
                {
                    list.Add(int.Parse(c.ToString()));
                }
            }
            return list.ToArray();
    }
}
