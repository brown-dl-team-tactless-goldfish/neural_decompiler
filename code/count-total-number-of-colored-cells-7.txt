class Solution {
public:
    long long coloredCells(int n) {
        if(n == 1) return 1;
        if(n == 2) return 5;
        if(n == 3) return 13;
        long long sum = 13;
        for(int i = 4; i <= n; i++){
            sum += pow(i, 2);
            sum -= pow(i-2, 2);
        }
        return sum;
    }
};
