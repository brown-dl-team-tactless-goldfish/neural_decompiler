# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class FindElements:

    def __init__(self, root: TreeNode):

        curVal = 0
        self.data = []

        def helper(curNode, curVal):
            if (curNode == None):
                return

            helper(curNode.left, 2*curVal+1)
            curNode.val = curVal
            self.data.append(curVal)

            helper(curNode.right, 2*curVal+2)

        helper(root, 0)
        print("self.data = ", self.data)

    def find(self, target: int) -> bool:
        return (target in self.data)
