bool isPossibleDivide(vector<int>& nums, int k, unordered_map<int, int> mp = {}) {
	for (int n : nums) ++mp[n];
	while (mp.size()) 
		for (int j = 0, n = min_element(begin(mp), end(mp))->first; ++j <= k; ++n) 
			if (j < k && !mp.count(n + 1)) return false;
			else if (--(mp[n]) == 0) mp.erase(n);
	return true;
}
