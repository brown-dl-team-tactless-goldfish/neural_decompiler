class Solution {
public:
    int maxSumDivThree(vector<int>& nums) {
        
        vector<int>dp = {0, 0, 0};
        
        for (int i = 0; i < nums.size(); i++) {
            vector<int>ndp = dp;
            for (int j =  0; j < 3; j++) {
                int x = dp[j] + nums[i];
                ndp[x%3] = max(ndp[x%3], x);
            }
            dp = ndp;
        }
        
        return dp[0];
    }
};
