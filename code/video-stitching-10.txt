    int videoStitching(vector<vector<int>>& clips, int T) {
        sort(clips.begin(), clips.end());
        int cur = 0, nxt = 0, count = 1;
        for(auto& e : clips) {
            if(e[0] > nxt) return -1;
            if(e[0] > cur) {
                cur = nxt;
                count++;
            }
            nxt = max(nxt, e[1]);
            if(nxt >= T) return count;
        }
        return -1;
    }
