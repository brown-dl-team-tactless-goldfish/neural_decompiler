public class Solution {
    public int MinDays(int n)
    {
        return runDepthFirstSearch(n, new Dictionary<int, int>());
    }

    private int runDepthFirstSearch(int n, Dictionary<int,int> memo)
    {
        if (n < 3)
        {
            return n;
        }
        
        if (memo.ContainsKey(n))
        {
            return memo[n];
        }
        
        memo[n] = Math.Min(runDepthFirstSearch(n / 2, memo) + runDepthFirstSearch(n % 2, memo), runDepthFirstSearch(n / 3, memo) + runDepthFirstSearch(n % 3, memo)) + 1;
        
        return memo[n];
    }
}
