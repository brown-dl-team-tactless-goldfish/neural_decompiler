        break;
    }
    else if(temp==head){
        
        temp->next=nn->next;
        nn->next=temp;
        head=nn;
        pn=temp;
        
    }
    else{
        
        temp->next=nn->next;
        nn->next=temp;
        pn->next=nn;
    }
    pn=temp;
    temp=temp->next;
}
return head;
