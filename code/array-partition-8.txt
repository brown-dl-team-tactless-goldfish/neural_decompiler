int cmp(const void *x, const void *y){
    return (*(int*)x-*(int*)y);
}
int arrayPairSum(int* nums, int numsSize){
int i,sum=0;
qsort(nums, numsSize, sizeof(int), cmp); // sorting the array 
    for(i=0;i<numsSize;i+=2){
        sum=sum+nums[i]; // the pattern is that after you've sorted the array you only need the sum of elements at 2k position 
    }
    return sum;
}
