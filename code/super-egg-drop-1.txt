#define MAX_EGGS (100+1)
#define MAX_FLOORS (10000+1)
static int tbl[MAX_FLOORS][MAX_EGGS] = {0};

int egg_drop_dp_nk(int eggs, int floors)
{
    int drop = 0, egg = 0;
    while (tbl[drop][eggs] < floors)
    {
        ++drop;
        for (egg = 1; egg <= eggs; ++egg)
            tbl[drop][egg] = 1 + tbl[drop-1][egg-1] + tbl[drop-1][egg];
    }
    
    return drop;
}

#define egg_drop egg_drop_dp_nk
int superEggDrop(int eggs, int floors)
{
    return egg_drop(eggs, floors);
}
