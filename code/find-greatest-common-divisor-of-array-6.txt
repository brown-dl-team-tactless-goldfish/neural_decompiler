int findGCD(int* nums, int numsSize){
    int max=0, min=nums[0], i, divisor;
    for(i=0; i<numsSize; i++){
        if(max<nums[i]){
            max=nums[i];
        }
        if(min>nums[i]){
            min=nums[i];
        }
    }
    for(i=min; i>=1; i--){
        if((max%i==0) && (min%i==0)){
            divisor=i;
            break;
        }    
    }
    return divisor;
}
