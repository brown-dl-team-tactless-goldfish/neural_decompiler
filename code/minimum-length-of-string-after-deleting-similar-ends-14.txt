int minimumLength(string s)
{
    if(s.size() <= 1)
        return s.size();
    
    int l = 0, r = s.size()-1;
    
    while(l < r and s[l] == s[r])
    {
        char c = s[l];
        while(l < r and s[l] == c)
            l++;
        
        while(r >= l and s[r] == c)
            r--;
    }
    
    return r - l + 1;
}
