int longestContinuousSubstring(char * s){
    int n=strlen(s);
    int ans=1, max=0;
    for(int i=0; i<n;i++)
    {
        if(s[i]+1 == s[i+1])
        {
            ans+=1;
        }
        if(ans>max)
        {
            max=ans;
        }
        if(s[i]+1 != s[i+1])
        {
            ans=1;
        }
    }
    return max;
}
