int minDeletionSize(char **A, int ASize) {
    // *(*(A+行)+列)
    int j = 0;
    int sum = 0;
    while (*(*(A) + j) != 0) {
        for (int i = 1; i < ASize; i++) {
            if (*(*(A + i) + j) < *(*(A + i - 1) + j)) {
                sum++;
                break;
            }
        }
        j++;
    }
    return sum;
}
