int minimumDeletions(string s) {
        int invalid = 0;
        stack<char> st;
        for(char c : s){
            if(c == 'b'){
                st.push(c);
            }
            if(c == 'a' and st.empty()){
                continue;
            }
            if(c == 'a' and !st.empty()){
                invalid +=1;
                st.pop();
            }
        }
        return invalid;
    }
