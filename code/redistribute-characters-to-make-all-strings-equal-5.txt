// Memory Usage: 11.7 MB, less than 65.77% of C++ online submissions for Redistribute Characters to Make All Strings Equal.
bool makeEqual(vector<string>& words) {
	vector<int> count(26, 0);
	int wordsCount = 0;
	for (const string& word : words) {
		wordsCount++;
		for (const char& c : word)
			++count[c - 'a'];
	}
	for (const int& c : count)
		if (c % wordsCount)
			return false;
	return true;
}
