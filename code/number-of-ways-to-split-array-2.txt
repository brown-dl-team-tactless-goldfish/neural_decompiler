class Solution {
public:
    int waysToSplitArray(vector<int>& nums) {
        int n = nums.size();
        long long lst = 0;
        for(int i = 0; i < n; i++)
        {
            lst += nums[i];
        }
        long long lft = 0, rht, ans = 0;
        for(int i = 0; i < n-1; i++)
        {
            lft += nums[i];
            rht = lst-lft;
            if(lft >= rht)
                ans++;
        }
        return ans;
    }
};
