/**
 * Note: The returned array must be malloced, assume caller calls free().
 */
int* getAverages(int* nums, int numsSize, int k, int* returnSize){
    
    int *ret_arr = (int*)calloc(numsSize,sizeof(int));
    memset(ret_arr,-1,sizeof(int) *numsSize);
    long long int sum = 0;
    int avg = 0;
    
    for(size_t i = 0; i < numsSize ; i++)
    {
        
        sum += nums[i];
        
        // traverse until double radius
        if(i >= (k*2))
        {
            // fill middle position with avarage
           ret_arr[i - k] = sum / ((k*2) + 1);
            // delete first element
           sum -= nums[i - (k*2)];
            
        }
    }
    *returnSize = numsSize;
    return ret_arr;

}
