public int reverse(int x) {
	int neg = 1, degree = 0, rep = 0;
	long r = 0l;
	if (x == 0 || x == Integer.MIN_VALUE)
		return 0;
	neg = x < 0 ? -1 : 1;
	x = x < 0 ? x * -1 : x;
	degree = (int) Math.pow(10, (int) Math.log10(x));
	// main loop after lots of crap
	while (x != 0) {
		int t = x / degree;
		r += (long) Math.pow(10, rep++) * t;
		if (r > Integer.MAX_VALUE)
			return 0;
		x %= degree;
		degree /= 10;
	}
	return (int) r * neg;
}
