int combinationSum4(int* nums, int numsSize, int target){
    unsigned long int* ret=(unsigned long int*)malloc((target+1)*sizeof(unsigned long int));
    ret[0]=1;
    ret[1]=1;
    for(int i=1;i<=target;i++)
        ret[i]=0;
    for(int i=1;i<=target;i++)
        for(int j=0; j<numsSize;j++)
            if(nums[j]<=i) ret[i]+=ret[i-nums[j]];
    return (int)ret[target];
}
