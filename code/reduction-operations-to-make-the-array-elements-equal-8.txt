class Solution {
public:
    int reductionOperations(vector<int>& nums) {
        map<int, int> cnt;
        for(auto n : nums) ++cnt[n];
        int ans = 0;
        while(cnt.size() > 1) {
            auto it = cnt.end(); --it;
            ans += it->second;
            int k1 = it->first;
            int c1 = it->second;
            auto it2 = it;
            --it2;
            it2->second += c1;
            cnt.erase(k1);
        }
        return ans;
    }
};
