int jump(int* nums, int size){
    if(size == 1)  return 0;
    int so_far = nums[0], time = 0, max = 0;
    for(int i=0; i<size-1; i++){
        max = (i + nums[i] > max) ? i+nums[i] : max;
        if(i == so_far){
            time++;
            so_far = max;
        }
    }
    return time+1;  //+1 : final jump
}
