class Solution {
public:
    bool isCircularSentence(string sentence) {
        string cur;
        char last='#';
        stringstream ss(sentence);
        while(ss>>cur){
            if(last!='#'&&last!=cur[0])return false;
            last=cur[cur.size()-1];
        }
        return sentence[0]==sentence[sentence.size()-1];
    }
};
