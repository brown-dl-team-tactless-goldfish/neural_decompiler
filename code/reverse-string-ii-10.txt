void reverse(char * s, int start, int end)
{
    int stop = end < strlen(s)-1 ? end : strlen(s)-1;
    for(int i=start; i<=(start+stop)/2; i++)
    {
        char temp = s[i];
        s[i] = s[start+stop-i];
        s[start+stop-i] = temp;
    }
}

char * reverseStr(char * s, int k){
    if(k == 1) return s;
    for(int i=0; i<strlen(s); i+=2*k)
        reverse(s, i, i+k-1);
    return s;
}
