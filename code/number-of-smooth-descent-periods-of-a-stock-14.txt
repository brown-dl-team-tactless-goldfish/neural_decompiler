class Solution {
public:
    long long getDescentPeriods(vector<int>& vec) {
        long long ans = 0;
        int n = vec.size();
        long long cnt = 1;

        for(int i = 1;i<n;i++){
            if(vec[i-1]==vec[i]+1){
                cnt++;
            }
            else{
                ans = ans + cnt*(cnt+1)/2;
                cnt = 1;
            }
        }

        if(cnt!=0) ans = ans + cnt*(cnt+1)/2;

        return ans;
    }
};
