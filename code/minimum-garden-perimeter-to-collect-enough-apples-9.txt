class Solution {
public:
    long long minimumPerimeter(long long neededApples) {
        auto apple = [](size_t n) {
            return 2 * (n*(n+1)*(2*n + 1));
        };
        
        size_t l = 0, r = 600000;
        while (l < r) {
            int m = l + (r - l) / 2;
            if (apple(m) < neededApples) {
                l = m + 1;
            } else {
                r = m;
            }
        }
        return l * 8;
    }
};
