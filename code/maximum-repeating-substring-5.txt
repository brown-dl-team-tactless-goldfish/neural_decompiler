class Solution:
    def maxRepeating(self, sequence: str, word: str) -> int:
        l,m = len(word), len(sequence)
        if(sequence==word): return 1
        ma=0
        for i in range(m-l+1):
            c=0
            for j in range(i,m,l):
                t = sequence[j:l+j]
                if(t == word):
                    c += 1
                else: break
            ma = max(ma,c)
        return ma
