class Solution {
public:
    int countOrders(int n) {
        
        
        /* P1 D1 n = 1;
        n=2 P2 can be placed at 3 postionns
        
        _ p1 d1,  p1 _ d1, p1 d1_
        
        similary d2 can be placed for all (3+ 2+1) = 6 places
        
        now from this we conclude for_new_element = old_permuation * (
            new_positon arrangements
        )
        */
        
        
        long long ans = 1 ;
        
        for(int i= 2 ; i <= n ; i++) {
            int val = 2*i - 1; // total postions are 2*i one is already fixed
            long long sum = ((val * 1ll* (val + 1)) / 2);
                
            sum = sum % 1000000007;
            
            ans = (ans % 1000000007 * 1ll * (sum) )% 1000000007;
        }
        return ans;
    }
};
