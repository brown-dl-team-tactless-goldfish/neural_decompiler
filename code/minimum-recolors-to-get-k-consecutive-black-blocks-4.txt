class Solution:
    def minimumRecolors(self, blocks: str, k: int) -> int:
        start_index = 0
        length = len(blocks)
        res = float("inf")
        count = 0
        for i in range(start_index, k):
            if blocks[i] == "W":
                count += 1
        res = min(res, count)
        
        for i in range(k, length):
            if blocks[start_index] == "W":
                count -= 1
            if blocks[i] == "W":
                count += 1
            
            res = min(res, count)
            start_index += 1
        return res
