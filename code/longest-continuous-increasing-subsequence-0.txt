int findLengthOfLCIS(int* nums, int numsSize){
    int cur = 1;
    int max = 1;
    for(int i = 1; i < numsSize; i++){
        if(nums[i] > nums[i-1])
            cur++;
        else{
            if(cur > max)
                max = cur;
                cur = 1;
        }
    }
        
    if(cur > max)
        max = cur;
    
    return max;
}
