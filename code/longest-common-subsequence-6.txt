#define MAX(a, b) (a > b ? a : b)

int longestCommonSubsequence(char * s1, char * s2){
    int l1=strlen(s1),l2=strlen(s2);
    int DP[l1+1][l2+1];
    for (int i=0;i<=l1;i++) {
        for (int j=0;j<=l2;j++) {
            DP[i][j]=(i==0||j==0)?0:(s1[i-1]==s2[j-1])?(DP[i-1][j-1]+1):MAX(DP[i][j-1],DP[i-1][j]);
        }
    }
    return DP[l1][l2];
}
