int rearrangeCharacters(char * s, char * target){
    int* alphaS = calloc(26, sizeof(int));
    int* alphaT = calloc(26, sizeof(int));
    for(int i = 0; i < strlen(s); i++){
        alphaS[s[i] - 'a']++;
    }
    for(int i = 0; i < strlen(target); i++){
        alphaT[target[i] -'a']++;
    }
    int ans = INT_MAX;
    for(int i = 0; i < 26; i++){
        if( alphaT[i] ){
            ans = fmin(ans, alphaS[i]/alphaT[i]);
        }
    }
    free(alphaS);
    free(alphaT);
    return ans;
}
