class Solution {
public:
    
    bool isVowel(char c) {
        return (c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u');
    }
    
    long long countVowels(string word) {
        
        int n = word.size();
        long long count = 0;
        
        for(int i=0; i<n; i++) {
            if(isVowel(word[i]))
                count += (long long)(i+1) * (long long)(n-i);
        }
        
        return count;
    }
};
