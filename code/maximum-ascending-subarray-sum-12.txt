int maxAscendingSum(int* nums, int numsSize){
    int maxSum = 0;
    int i = 0;
    while (i < numsSize) {
        int j = i + 1;
        int ascendingSum = nums[i];
        while (j < numsSize && nums[j] > nums[j-1]) {
            ascendingSum += nums[j];
            j++;
        }
        if (ascendingSum > maxSum) maxSum = ascendingSum;
        i = j;
    }
    return maxSum;
}
