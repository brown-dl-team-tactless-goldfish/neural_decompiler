int maxProfit(int* prices, int pricesSize){
    if (pricesSize == 0)
        return 0;
    int min = prices[0];
    int profit = 0;
    int max = 0;
    for (int i = 1; i < pricesSize; i++){
        min = (prices[i] < min)?prices[i]:min;
        profit = prices[i] - min;
        max = (max > profit)?max:profit;
    }
    return max;
}
