    bool closeStrings(string word1, string word2) {
        vector<int>a(26),b(26);
        for(char c : word1)
            a[c-'a']++;
        for(char c : word2)
            b[c-'a']++;
        
        set<char> s1(word1.begin(),word1.end()), s2(word2.begin(),word2.end());
        if(s1!=s2) return false;
        
        sort(a.begin(),a.end());
        sort(b.begin(),b.end());
        
        if(a!=b) return false;
        
        return true;
    }
