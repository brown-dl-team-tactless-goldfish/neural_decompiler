int* decode(int* encoded, int encodedSize, int first, int* returnSize){

    
    int *arr = malloc(sizeof(int)*(encodedSize+1));
    int i;
    arr[0]=first;
    
    int prev=first;
    for(i=1; i<encodedSize+1; i++) {
        
        arr[i] = prev = prev ^ encoded[i-1];
        
    }
    
    *returnSize = encodedSize+1;
    return arr;
}
