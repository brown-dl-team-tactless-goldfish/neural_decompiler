int rows, cols;
    std::unordered_map<int, std::unordered_set<int>> m;
    Solution(int n_rows, int n_cols) {
        rows = n_rows;
        cols = n_cols;
    }
    
    vector<int> flip() {
        int r, c;
        do {
            r = rand()%rows;
        } while (m[r].size() == cols);
        do {
            c = rand()%cols;
        } while (m[r].find(c) != m[r].end());
        
        m[r].insert(c);
        return {r, c};
    }
    
    void reset() {
        m.clear();
    }
