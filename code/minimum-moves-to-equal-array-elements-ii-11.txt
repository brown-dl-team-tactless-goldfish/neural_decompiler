
int compare(const void*a, const void*b){
    return *(const int*)a - *(const int*)b;
}

int minMoves2(int* nums, int numsSize){
    int sum=0;
    qsort(nums, numsSize, sizeof(nums[0]), compare); //Sort.
    
    int mid = nums[numsSize/2]; //Get mid value.
    
    for(int i=0;i<numsSize;++i){ //Sum of diff between all of nums value and the mid value.
        sum += abs(mid-nums[i]);
    }
    
    return sum;
}
