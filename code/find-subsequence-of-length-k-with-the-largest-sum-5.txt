 vector<int> maxSubsequence(vector<int>& nums, int k) {
        int n=nums.size();
        vector<pair<int,int>> v;  //to store value with index
        for(int i=0;i<n;i++) v.push_back({nums[i],i});
        sort(v.begin(),v.end());
        map<int,int> myap;  //to sort according to the index
        for(int i=n-1;i>=n-k;i--){
            myap[v[i].second]=v[i].first;
        }
        vector<int> res;
        for(auto it:myap) res.push_back(it.second);
        return res;
    }
