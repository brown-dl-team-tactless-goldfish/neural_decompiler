int longestConsecutive(vector<int>& nums) {
        unordered_set<int> numSet;
        int answer=0;
        
        for(auto i:nums)
            numSet.insert(i);
        
        while(numSet.size() > 0){
            
            int count = 1;
            int element = *numSet.begin();
            numSet.erase(element);
            
            int bigger = element+1;
            int smaller = element-1;
            
            while(numSet.count(bigger)){
                numSet.erase(bigger);
                bigger+=1;
                count++;
            }
            
            while(numSet.count(smaller)){
                numSet.erase(smaller);
                smaller-=1;
                count++;
            }
            
            answer = max(answer, count);
        }
        return answer;
    }
