// Function to check palindrome string
bool is_pallindrome(char *str,int l)
{
    int i=0,j=l-1;
    while(i<j)
    {
        if(str[i]!=str[j])
            return false;
        i++;
        j--;
    }
    return true;
}


char * firstPalindrome(char ** words, int wordsSize){
    int i;
    for(i=0;i<wordsSize;i++)
    {
        if(is_pallindrome(words[i],strlen(words[i])))       //check the palindrome of each words
            return words[i];
    }
    return "";
}
