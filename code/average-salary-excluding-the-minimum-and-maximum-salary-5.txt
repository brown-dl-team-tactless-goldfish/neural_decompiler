double average(int* salary, int salarySize){
    double sum=0;
    int min=0;
    int max=0;

    min=salary[0];
    max=salary[0];

    for(int i=0;i<salarySize;i++){
        if(min>salary[i]){
            min=salary[i];
        }
        if(max<salary[i]){
            max=salary[i];
        }
    }

    for(int i=0;i<salarySize;i++){
        if((salary[i]!=min)&&(salary[i]!=max)){
            sum+=salary[i];
        }
    }

    return sum/(salarySize-2);
}
