int rob(int* nums, int numsSize){
    if(numsSize == 1){
        return nums[0];
    }
    int box[100];
    box[0] = nums[0];
    box[1] = nums[0] > nums[1] ? nums[0] : nums[1];
    for(int i = 2 ; i < numsSize ; i++){
        int takelast = box[i-2]+nums[i];
        box[i] = takelast > box[i-1] ? takelast : box[i-1];
    }
    int max = box[numsSize-1];
    return max;
}
