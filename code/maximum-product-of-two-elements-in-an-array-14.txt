

int maxProduct(int* nums, int numsSize){
    int secondLargest = 0;
    int Largest = 0;
    
    for(int i = 0; i < numsSize; i++)
    {
       if(nums[i] > Largest){
           nums[i] = nums[i] ^ Largest;
           Largest = nums[i] ^ Largest;
           nums[i] = nums[i] ^ Largest;
       } 
       if(nums[i] > secondLargest)
       {
           secondLargest = nums[i];
       }          
    }
    
    return  (secondLargest - 1) * (Largest - 1);
}
