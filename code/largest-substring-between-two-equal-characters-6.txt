#define MAX(x,y) (x) > (y) ? (x) : (y)

int maxLengthBetweenEqualCharacters(char * s){
    int table[26] = {0};
    int result = 0;
    int *temp;
    for(int i = 1;*s;i++,s++){
        temp = &table[*s -'a'];
        if(*temp){
            result = MAX(i - *temp ,result) ; 
        }else{
            *temp = i;
        }
    }
    return result - 1;
    
}
