double averageWaitingTime(int** customers, int customersSize, int* customersColSize){
    long long sum  = 0;
    int begin = customers[0][0];
    for(int i = 0; i < customersSize; i++){
        if(customers[i][0] < begin)
            sum += begin - customers[i][0];
        sum += customers[i][1];
        begin = fmax(customers[i][0] ,begin ) + customers[i][1]; 
    }
    return (double)sum/customersSize;
}
