class Solution {
public:
    long long beautifulSubarrays(vector<int>& nums) {
        long long ret = 0 ;
        int cur = 0 ;
        unordered_map<int, int>Map ;
        Map[cur]++ ;
        for(int i = 0; i < nums.size(); i++){
            cur ^= nums[i] ;
            if(Map.find(cur) != Map.end()){
                ret += Map[cur] ;
            }
            Map[cur]++ ;
        }
        return ret ;
    }
};
