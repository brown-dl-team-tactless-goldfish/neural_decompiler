class Solution {
    public int kItemsWithMaximumSum(int numOnes, int numZeros, int numNegOnes, int k){
        int sum=0;
       if((numOnes+numZeros)>=k)
       {
           return Math.min(numOnes,k);
       }
       else if((numOnes+numZeros)<k)
       {
           int m=k-(numOnes+numZeros);
           sum=(numOnes*(1)+numZeros*(0))-m;
       }
       return sum;
    }
}
