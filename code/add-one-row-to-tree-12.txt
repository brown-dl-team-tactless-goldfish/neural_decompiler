 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    
    TreeNode* aux(TreeNode* root, int val, int depth, int cur, int dir) {
        
        
        if(cur == depth) {
           if(!dir) {
               TreeNode* temp = new TreeNode(val);
               temp -> left = root;
               return temp;
           } else {
               TreeNode* temp = new TreeNode(val);
               temp -> right = root;
               return temp;
           }
        }
        if(!root) return NULL;
        root -> left = aux(root -> left, val, depth, cur + 1, 0);
        root -> right = aux(root -> right, val, depth, cur + 1, 1);
        
        return root;
    }
    TreeNode* addOneRow(TreeNode* root, int val, int depth) {
         return aux(root, val, depth, 1, 0);    
    }
};
