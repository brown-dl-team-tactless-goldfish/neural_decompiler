int* nextLargerNodes(struct ListNode* head, int* returnSize){
    int * ret = malloc(sizeof(int) * 10000);
    struct ListNode *p = head;
    
    int m, k = 0;
    while (p)
    {
        m = 0;
        struct ListNode *q = p->next;
        while (q)
        {
            if (q->val > p->val && q->val > m) {
                m = q->val;
                break;
            }
            q = q->next;
        }
        ret[k++] = m;
        p = p->next;
    }
    *returnSize = k;
    return ret;
}
