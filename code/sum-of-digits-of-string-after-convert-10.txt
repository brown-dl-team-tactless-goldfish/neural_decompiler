public class Solution {
    public int GetLucky(string s, int k) {
        
        int sum = 0;
        foreach(char c in s)
        {
            int num = c - 'a' + 1;
            sum += num / 10 + num % 10;
        }
        
        while(--k > 0 && sum > 9)
        {
            int tmp = 0;
            while(sum > 0)
            {
                tmp += sum % 10;
                sum /= 10;
            }
            sum = tmp;
        }
        
        return sum;
    }
}
