int countEven(int num){
    int digit = 1;
    int digitCount = 0;

    while (digit <= num){
        int divDigit = digit;
        int digitSum = 0;
        while (divDigit != 0){
            digitSum = digitSum + divDigit % 10;
            divDigit = divDigit / 10;
        }
        if (digitSum % 2 == 0){
            digitCount++;
        }
        digit++;        
    }

    return digitCount;
}
