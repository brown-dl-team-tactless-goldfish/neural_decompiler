

char * toHex(int num){
    
    unsigned int n = num;
    char convert[17] = "0123456789abcdef";
    char rem[100]; 
    char *ans;
    int j;

    int i = 0; 
    if(n == 0)
        return "0";
    while(n!=0) 
    {    

            rem[i] = convert[n%16]; 
            i++; 

            n = n/16; 
    } 
    
    ans = malloc(sizeof(char)*(i+1));
    for(j =0 ; j < i; j++)
          ans[j] = rem[i-j-1];
    
    ans[j] = '\0';
    
    return ans;

}
