int cmp(const void *a, const void *b){
    return(*(int *)a - *(int *)b);
}

int minStartValue(int* nums, int numsSize){
    
    int n = numsSize, min, startvalue = 1;
    int *prefixsum = (int *)malloc(n * sizeof(int));
    
    prefixsum[0] = nums[0];
    
    for(int i = 1; i < n; i++)
        prefixsum[i] = prefixsum[i-1] + nums[i];
    
    qsort(prefixsum, n, sizeof(int), cmp);
    min = prefixsum[0];
    
    if(min < 0) startvalue = abs(min) + 1;
    
    return startvalue;
}
