class OrderedStream {
public:
    vector<string> orderedStream;
    int currentOutputId;
    OrderedStream(int n) : orderedStream(vector<string>(n+1,"#")), currentOutputId(1)
    {
    }
    vector<string> insert(int id, string value) 
    {
        vector<string> outputStringVector;
        orderedStream[id] = value;
        while(currentOutputId < orderedStream.size() && orderedStream[currentOutputId] != "#" )
        {
            outputStringVector.push_back(orderedStream[currentOutputId++]);
        }
        return outputStringVector;
    }
};
