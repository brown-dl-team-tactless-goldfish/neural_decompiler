#include <stdlib.h>
#include <string.h>


int compare(const void *p1, const void *p2) {
    
    return (*(const char*)p1 - *(const char*)p2);
}


char findTheDifference(char * s, char * t){

    int len_s = strlen(s);
    int len_t = strlen(t);
    
    qsort(s, len_s, sizeof(char), compare);
    qsort(t, len_t, sizeof(char), compare);
    
    for (; *s; s++, t++) {
        if (*s != *t)
            break;
    }
    return *t;
}
