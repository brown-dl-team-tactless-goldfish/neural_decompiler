int sumOfDigits(int num){
    int sumDigits = 0;
    while(num != 0){
        sumDigits+= num % 10;
        num /=10;
    }
    return sumDigits;
}

int countBalls(int lowLimit, int highLimit){
    int boxes[45] = {};
    
    int mostBalls = 0;

    for(int i = lowLimit; i <= highLimit; i++){
        int boxNumber = sumOfDigits(i) - 1;
       mostBalls = ++boxes[boxNumber] > mostBalls ? boxes[boxNumber]: mostBalls;
    }

    return mostBalls;
}
