bool isalpha_numeric(char c) {
    if ((c >= 'A' && c <= 'Z') || (c >= 'a' && c <= 'z') ||
       (c >= '0' && c <= '9'))
        return 1;
    return 0;
}

int mytolower(char *c) {
    if (*c >= 'A' && *c <='Z') {
        *c = *c + ('a' - 'A');
        return 1;
    }
    return 0;
}

int mystrlen(char *s)
{
    if (!s)
        return 0;
    char *ptr = s;
    int len = 0;
    while (*ptr++)
        len++;
    return len;
}
bool isPalindrome(char * s){
    if (s == NULL)
        return 0;
    int i = 0;
    int j = mystrlen(s) -1;
    while (i <= j) {
        if (!isalpha_numeric(s[i])) {
            i++;
            continue;
        }
        if (!isalpha_numeric(s[j])) {
            j--;
            continue;
        }
        mytolower(&s[i]);
        mytolower(&s[j]);
        if (s[i] != s[j])
            return 0;
        i++;
        j--;
    }
    return 1;
}
