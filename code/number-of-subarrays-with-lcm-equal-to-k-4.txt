class Solution {
public:
	int subarrayLCM(vector<int>& nums, int k) {
		//if(nums.size()==1){if(nums[0]==k){return 1;}else{return 0;}}
		int cnt=0;
		for(int i=0; i<nums.size(); i++){
			unsigned int a=nums[i];
			for(int j=i; j<nums.size(); j++){
				a=lcm(a, nums[j]);
				if(a==k){cnt++;}
			}
		}
		return cnt;
	}
};
