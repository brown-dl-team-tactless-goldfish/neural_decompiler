class Solution {
public:
    int maximizeWin(vector<int>& arr, int k) {
        //for every index store the result
        vector<int> dp(arr.size());
        
        //dp[i] stores the max pos covered by any single segment of length k
        int res=0;
        int i=0;
        int maxiSingle=0;
        
        for(int j=0;j<arr.size();j++)
        {
            while(arr[i]< arr[j]-k)
                i++;
            int dist=j-i+1;
            maxiSingle=max(maxiSingle,dist);
            dp[j]=maxiSingle;   
            //arr[i..j] consits of one segment  and add the max ans for subarray [0..i-1] 
            int curr=dist;
            if(i>0)
                curr+=dp[i-1];
            res=max(res,curr);
            
        }
        return res;
    }
};
