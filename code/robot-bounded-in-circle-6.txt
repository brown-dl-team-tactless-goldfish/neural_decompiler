bool isRobotBounded(char * instructions){
    int dir = 0, x = 0, y = 0;
    for(int i = 0; i < strlen(instructions); i++)
    {
        if(instructions[i] == 'L')
            dir = (dir + 1) % 4;
        else if(instructions[i] == 'R')
            dir = (dir + 3) % 4;
        else if(dir == 0)
            y++;
        else if(dir == 1)
            x--;
        else if(dir == 2)
            y--;
        else
            x++;
    }   
    return dir != 0 || (x == 0 && y == 0);
}
