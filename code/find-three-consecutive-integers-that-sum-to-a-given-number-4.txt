class Solution {
public:
    vector<long long> sumOfThree(long long num) {
        vector<long long> v;
        long long c;
        if(num==0) return {-1,0,1};
        if(num<3) return v;
        if(num % 3 ==0){    
            c = num/3;
            v.push_back(c-1);
            v.push_back(c);
            v.push_back(c+1);
        }
        return v;
    }
};
