public:
    vector<int> getNoZeroIntegers(int n) {
        int i = n;
        while (i--) {
            if (valid(i) && valid(n - i))
                return {i, n-i};
        }
        
        return {};
    }
    
    bool valid(int n) {
        while (n) {
            if (n % 10 == 0)
                return false;
            
            n /= 10;
        }
        return true;
    }
