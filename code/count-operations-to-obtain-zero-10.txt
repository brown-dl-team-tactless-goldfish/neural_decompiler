int countTimes(int num1,int num2,int count){
    if(num1 == 0 || num2 == 0){
        return count;
    }
    if(num1 >= num2){
        return countTimes(num1-num2,num2,count+1);
    }else{
        return countTimes(num1,num2-num1,count+1);
    }
}
int countOperations(int num1, int num2){
    int count = 0;
    return countTimes(num1,num2,count);
}
