public class Solution {
    public bool[] CanEat(int[] candiesCount, int[][] queries) {
        
        var count = new long[candiesCount.Length];
        int pos = 0;
        
        foreach(var c in candiesCount)
        {
            count[pos] = pos == 0 ? c : c + count[pos-1];
            pos++;
        }
        
        var res = new List<bool>();                
        foreach(var query in queries)
        {
            int candyType = query[0];
            int day = query[1];
            int max = query[2];
            
            long minCandies = day;
            long maxCandies = day * max;
            
            long maxDayLimit = count[candyType];
            long minDayLimit = (candyType == 0 ? 0 : count[candyType-1])/max;
            
            res.Add(day >= minDayLimit && day < maxDayLimit);
        }
        
        return res.ToArray();
    }
}
