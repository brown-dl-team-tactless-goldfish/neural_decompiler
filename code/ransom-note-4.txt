

bool canConstruct(char * ransomNote, char * magazine){
int temp[26];
	for(int i = 0; i < 26; i++){
		temp[i] = 0;
	}
	while(*magazine!='\0'){
		temp[*magazine - 'a']++;
		magazine++;
	}
	while(*ransomNote!='\0'){
	   temp[*ransomNote - 'a']--;
		ransomNote++; 
	}
	for(int i = 0; i < 26; i++){

		if(temp[i] < 0){
			return false;
		}
	}
	return true;
}
