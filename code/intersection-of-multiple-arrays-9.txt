class Solution {
public:
    vector<int> intersection(vector<vector<int>>& nums) {
        vector<int>freq(1001,0) ;
        for(auto &arr : nums){
            for(auto &x : arr) ++freq[x] ;
        }
        
        vector<int> ans ;
        for(int i = 0 ; i < freq.size() ; ++i ){
            if(freq[i] == nums.size()) ans.push_back(i) ;
        }
        return ans ;
    }
};
