class Solution {
public:
    int minimumEffort(vector<vector<int>>& tasks) {
        for(auto &task : tasks)
            task[0]=task[1]-task[0];
        sort(tasks.begin(),tasks.end());
        int res=0;
        for(auto &task : tasks){
            int actual=task[1]-task[0];
            res+=actual;
            if(task[1]>res)
                res=task[1];
        }
        return res;
    }
};
