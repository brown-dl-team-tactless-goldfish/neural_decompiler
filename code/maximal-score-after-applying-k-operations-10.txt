class Solution {
public:
    long long maxKelements(vector<int>& nums, int k) {
        long long ans=0;
        priority_queue<int>pq(begin(nums),end(nums));
        while(size(pq) && k--){
            int x=pq.top();
            pq.pop();
            ans+=x;
            x=ceil((double)x/3.0);
            pq.push(x);
        }
        return ans;
    }
};
