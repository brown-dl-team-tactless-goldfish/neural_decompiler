int minDeletionSize(vector<string>& A) {
	if (A.empty()) return 0;
	vector<int> cnt(A[0].size(), 1);
	int maxLen = 1;
	for (int i = 0; i < A[0].length(); ++i) {
		for (int j = i - 1; j >= 0; --j) {
			if (cnt[j] + 1 > cnt[i]) {
				int t = 0;
				for (; t < A.size() && A[t][j] <= A[t][i]; ++t);
				if (t == A.size()) cnt[i] = cnt[j] + 1;
			}
		}
		maxLen = max(maxLen, cnt[i]);
	}
	return (int)A[0].length() - maxLen;
}
