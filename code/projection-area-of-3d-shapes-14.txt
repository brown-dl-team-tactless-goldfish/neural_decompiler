int projectionArea(int** grid, int gridRowSize, int *gridColSizes) {
	int i = 0, row = 0, col = 0;
	int RowMax, ColMax;
	int topArea = 0, frontArea = 0, profileArea = 0;
	int *EachRowMax = (int *)malloc(sizeof(int) * gridRowSize);
	int *EachColMax = (int *)malloc(sizeof(int) * (*gridColSizes));

	for(row = 0; row < gridRowSize; row++)      //store each row max data, and calculate top view area
	{
		RowMax = grid[row][0];
		for(col = 0; col < *gridColSizes; col++)
		{    
			if(grid[row][col] > 0)              
				topArea++;

			if(grid[row][col] > RowMax)
				RowMax = grid[row][col];   
		}
		EachRowMax[row] = RowMax;
	}

	for(col = 0; col < *gridColSizes; col++)    //store each column max data
	{
		ColMax = grid[0][col];
		for(row = 0; row < gridRowSize; row++)
		{
			if(grid[row][col] > ColMax)
				ColMax = grid[row][col];
		}
		EachColMax[col] = ColMax;
	}

	for(i = 0; i < gridRowSize; i++)        //calculate front plane  area
		frontArea += EachRowMax[i];
	for(i = 0; i < *gridColSizes; i++)      //calculate profile plane area
		profileArea += EachColMax[i];

	free(EachRowMax);
	free(EachColMax);

	return frontArea+profileArea+topArea;
}
