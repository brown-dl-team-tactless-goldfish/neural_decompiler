
/**
 * Note: The returned array must be malloced, assume caller calls free().
 */
int* sortArrayByParityII(int* A, int ASize, int* returnSize){
    int i=0,j=1;
    int temp = 0;
    for(i=0;i<ASize;i=i+2)
    {
        if(A[i]%2 == 1)
        {
            while(A[j]%2 == 1)
            {
                j = j+2;
            }
            temp = A[j];
            A[j] = A[i];
            A[i] = temp;
        }
        
    }
    *returnSize = ASize;
    return A;
}
