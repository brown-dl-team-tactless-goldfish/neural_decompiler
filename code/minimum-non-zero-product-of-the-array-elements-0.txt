typedef long long LL ;
LL quickMul(LL base, LL k, LL M){
    LL ret = 1 ;
    if(k == 0)
        return 1 ;
    LL half = quickMul(base, k/2, M) ;
    ret = half * half % M ;
    if(k%2)
        ret = (ret * (base % M)) % M ;
        
    return ret ;
}
int minNonZeroProduct(int p){
    LL M = 1e9 + 7 ;
    LL k = (((LL)1<< (p-1)) - 1) ; 
    LL base = (((LL)1 <<p) - 2) ;
    
    return quickMul(base, k, M) * ((base + 1) % M) % M ;
}
