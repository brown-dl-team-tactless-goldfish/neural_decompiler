char * reversePrefix(char * word, char ch){
    int index = 0; 
    char tmp;
    for(int i = 0; word[i] != '\0'; i++){
        if(word[i] == ch){
            index = i;
            break;
        }
    }
	//if ch at 1st character or not found, return original word
    if(index != 0)
        for(int j = 0; j < index; j++, index--){
			//swap
            tmp  = word[j];
            word[j] = word[index];
            word[index] = tmp;
        }
    return word;
}
