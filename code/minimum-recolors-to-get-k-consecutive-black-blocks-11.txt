class Solution {
public:
    int minimumRecolors(string blocks, int k) {
        int w = 0 , ans = INT_MAX ;
        for(int s = 0 ,e = 0 ;  e < blocks.size() ; ++ e){
            w += (blocks[e] == 'W') ;
            if(e >= k - 1) ans = min(ans,w) , w -= (blocks[s++] == 'W') ;
        }
        
        return ans ;
    }
};
