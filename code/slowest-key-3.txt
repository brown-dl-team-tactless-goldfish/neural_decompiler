char slowestKey(int* releaseTimes, int releaseTimesSize, char * keysPressed){
    int maxTime = releaseTimes[0];
    char result = keysPressed[0];
    for(int i=1; i<releaseTimesSize; i++)
        if(releaseTimes[i]-releaseTimes[i-1] > maxTime)
        {
            maxTime = releaseTimes[i]-releaseTimes[i-1];
            result = keysPressed[i]; 
        }
        else if(releaseTimes[i]-releaseTimes[i-1] == maxTime && keysPressed[i] > result)
        {
            result = keysPressed[i]; 
        }
    return result;
}
