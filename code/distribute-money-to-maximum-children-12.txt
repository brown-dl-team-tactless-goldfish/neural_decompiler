class Solution {
public:
    int distMoney(int money, int children) {
        if(children > money)
            return -1;
        int ans = money/8 - 1;
        money -= ans*8;
        children -= ans;
        if(children <= 0)
            return ans+children-1;
        if(money < children)
            return ans-(children-money)/8-1;
        if(children == 1 && money > 8)
            return ans;
        if(money-children < 7)
            return ans;
        if(children == 2 && money == 12)
            return ans;
        return ans+1;
    }
};
