class Solution {
public:
    int shortestSequence(vector<int>& rolls, int k) {
        int len = 0 ;
        unordered_set<int>seen ;
        for(int i = 0; i < rolls.size(); i++){
            seen.insert(rolls[i]) ;
            if(seen.size() == k){
                len++ ;
                seen.clear() ;
            }
        }
        return len+1 ;
    }
};
