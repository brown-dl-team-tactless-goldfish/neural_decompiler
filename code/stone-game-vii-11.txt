class Solution 
{
    public:
    int rec(vector<int>& stones, int start, int end, int sum) 
    {
        if (start == end) 
            return 0;
        
        if (dp[start][end]) 
            return dp[start][end];
        
        int remove_first = sum - stones[start] - rec(stones, start+1, end, sum-stones[start]);
        int remove_last = sum - stones[end] - rec(stones, start, end-1, sum-stones[end]);
        
        dp[start][end] = max(remove_first, remove_last);
        return dp[start][end];
        
    }
    int stoneGameVII(vector<int>& stones) 
    {
        dp.assign(stones.size(), vector<int>(stones.size(), 0));
        int sum = 0;
        sum = accumulate(stones.begin(), stones.end(), sum);
        return rec(stones, 0, stones.size()-1, sum);
    }
    private:
    vector<vector<int>> dp;
};
