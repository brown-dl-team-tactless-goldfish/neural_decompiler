int countGoodSubstrings(char * s){
    int count = 0;
    int str_len = strlen(s);
    
    if (str_len < 3) {
        return 0;
    }
    
    char prev_prev = s[0];
    char prev = s[1];
    
    for (int i=2; i < str_len; i++) {
        if ((s[i] != prev) && (s[i] != prev_prev) && (prev != prev_prev)) {
            count++;
        }
        prev_prev = s[i-1];
        prev = s[i];
    }
    
    return count;
}
