int* getAverages(int* nums, int numsSize, int k, int* returnSize){
    int* ans = (int*)calloc(numsSize, sizeof(int));
    int r = 0, l = 0;
    int average = 2 * k + 1;
    long sum = 0;
    *returnSize = numsSize;
    while(r < numsSize)
    {
        sum += nums[r];
        if((r < k) || (r > numsSize - k - 1))
            ans[r] = -1;
    
        if(r >= 2 * k){
            ans[r - k] = (sum / average);
            sum -= nums[l++];
        }
        r++;
    }
    return ans;
}
