void dfs(struct TreeNode* root, struct TreeNode** cur) {
    if (root == NULL) return;

    dfs(root->left, cur);

    // In-order
    root->left    = NULL;
    (*cur)->right = root;
    (*cur)        = root;

    dfs(root->right, cur);
}

struct TreeNode* increasingBST(struct TreeNode* root) {
    struct TreeNode   ans = {};
    struct TreeNode** cur = &ans;
    
    dfs(root, &cur);
    
    return ans.right;    
}
