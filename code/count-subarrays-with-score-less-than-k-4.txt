class Solution {
public:
    long long countSubarrays(vector<int>& nums, long long k) {
        int i=0;
        int j=0;
        long long count=0;
        long long prefix_sum=0;
        
        // Variable size Sliding Window Problem
        while(j<nums.size())
        {
            // Do pre-calculation 
            prefix_sum+=nums[j];
            
            long long val=prefix_sum*(long long)(j-i+1);
            
            // Play with Condition
            while(val>=k)
            {
                // Remove the calculation for i
                prefix_sum-=nums[i];
                i++;
                val=prefix_sum*(long long)(j-i+1);
            }
           
            // Store result
            count+=(j-i+1);
            // Slide Window
            j++;
        }
        return count;
    }
};

Time Complexity: O(2*n)
Space Complexity: O(1)
