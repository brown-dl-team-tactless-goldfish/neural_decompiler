struct ListNode* removeNthFromEnd(struct ListNode* head, int n){
	  struct ListNode* start = head;
	  int size=0, order;

	  //get size of list
	  while(start)
	  {
		start = start->next;  
		size++;
	  }

	  // reset head to first address of list
	  start = head; 

	  //handle boundary case
	  if(size==1)
	  {
		return NULL;
	  }

	  order = size-n;
	  if(order==0)
	  {
	   // Remove head;
		return head->next;
	  }

	  while(--order)
	  {
		start = start->next;
	  }
	  //current Node -> next Node -> next_next Node
	  //current Node  x next Node 
	  //        link---------------> next_next Node
	  start->next = start->next->next;
	  return head;
  }
