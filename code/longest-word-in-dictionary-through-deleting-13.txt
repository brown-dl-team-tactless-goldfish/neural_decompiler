class Solution {
public:
    string findLongestWord(string s, vector<string>& d) {
        string ret;
        for (int i = 0, xpos = 0; i < d.size(); i++, xpos = 0) for (char& c : s) if (c == d[i][xpos]) if (++xpos == d[i].length()) if (d[i].length() > ret.length() || (d[i].length() == ret.length() && d[i] < ret)) ret = d[i];
        return ret;
    }
};
