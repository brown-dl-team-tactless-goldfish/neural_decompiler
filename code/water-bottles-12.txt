class Solution:
    def numWaterBottles(self, numBottles: int, numExchange: int) -> int:
        
        self.res = 0
        def helper(curFull, curEmpty):
            # print ("curFull = ", curFull, " curEmpty = ", curEmpty)
            self.res += curFull
            
            if (curFull == 0):
                if (curEmpty < numExchange):
                    return
                else:
                    helper(curEmpty // numExchange,  curEmpty % numExchange)
                
            else:
                helper(0, curFull + curEmpty)

            
        helper(numBottles, 0)
        print ("self.res = ", self.res)
        return self.res
