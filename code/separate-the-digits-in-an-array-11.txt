class Solution {
public:
    vector<int> separateDigits(vector<int>& nums) {
        vector<int> q ;
        for(int i = 0; i < nums.size(); i++){
            int val = nums[i] ;
            vector<int> v ;
            while(val){
                v.push_back(val%10) ;
                val /= 10 ;
            }
            for(int i = v.size()-1; i >= 0 ; i--){
                q.push_back(v[i]) ;
            }
        }
        return q ;
    }
};
