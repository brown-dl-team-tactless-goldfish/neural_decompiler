public class Solution {
    public string DestCity(IList<IList<string>> paths) {
        
        Dictionary<string,int> outdegree = new Dictionary<string,int>();
        foreach(var path in paths)
        {
            string source = path[0];
            string destination = path[1];
            
            if(outdegree.ContainsKey(source))
                outdegree[source]++;
            else
                outdegree.Add(source,1);
            
            if(!outdegree.ContainsKey(destination))
                outdegree.Add(destination,0);
        }
        
        foreach(var kv in outdegree)
        {
            if(kv.Value == 0)
                return kv.Key;
        }
        
        return "";
    }
}
