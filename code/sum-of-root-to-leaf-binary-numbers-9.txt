void trav(struct TreeNode *root, int *i, int *s){
    if (root) {
        *i = (*i << 1) | root->val;
        !root->left && !root->right ? *s += *i : 0;
        trav(root->left, i, s);
        trav(root->right, i, s);
        *i >>= 1;        
    }
}

int sumRootToLeaf(struct TreeNode* root){
    int i = 0, s = 0;
    trav(root, &i, &s);
    return s;
}
