class Solution {
public:
    int minFlips(string t) {
        t.erase(unique(t.begin(),t.end()),t.end());
        int i = 0, c = 0;
        while(t[i++]=='0');
        if(i == t.size()+1) return 0;
        while(i < t.size()) c += t[i++]=='0';
        return t[t.size()-1]=='0' ? 2*c : 2*c+1;
    }
};
