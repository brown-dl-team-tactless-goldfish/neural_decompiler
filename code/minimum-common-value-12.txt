public class Solution {
    public int GetCommon(int[] nums1, int[] nums2) {

        int idx1 = 0, idx2 = 0;
        
        while (idx1 < nums1.Length && idx2 < nums2.Length) {
            if (nums1[idx1] == nums2[idx2])
                return nums1[idx1];
            else if (nums1[idx1] > nums2[idx2])
                idx2++;
            else 
                idx1++;
        }
        
        return -1;        
    }
}
