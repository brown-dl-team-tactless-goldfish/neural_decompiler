char * destCity(char *** paths, int pathsSize, int* pathsColSize){
    for(int i=0; i<pathsSize; i++){
        for(int j=0; j<pathsSize; j++){
            if(strcmp(paths[i][1], paths[j][0])==0){
                break;
            }
            if (j==pathsSize-1){
                return paths[i][1];
            }
        }
    }
    return paths[pathsSize-1][1];
}
