int findKthPositive(int* arr, int arrSize, int k){
   
    
    int arrCount=0;
    int count=0;
    for(int i=1;i<arr[arrSize-1];i++)
    {
      if(arr[arrCount]==i)
          arrCount++;
       else
       {
           count++;
           if(count==k)
               return i;
       }    
      
    }
    
    
    if(count<k)
        return arr[arrSize-1]+k-count;
    
    return 0;
    
    
}
