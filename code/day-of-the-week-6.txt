class Solution {
public:
    string dayOfTheWeek(int day, int month, int year) {
        struct tm myTime;
        myTime.tm_isdst=-1;
        myTime.tm_sec=1;
        myTime.tm_min=0;
        myTime.tm_hour=0;
        myTime.tm_mday=day;
        myTime.tm_year=year-1900;
        myTime.tm_mon=month-1;
        mktime(&myTime);
        string days[]={"Sunday", "Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday"};
        return days[myTime.tm_wday];
    }
};
