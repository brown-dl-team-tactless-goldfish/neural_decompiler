int maxLengthBetweenEqualCharacters(char * s){
    int len=-1,max_len=-1;
    for(int i=0;i< strlen(s);i++){
        for(int j=i;j< strlen(s);j++){
            if(s[i]==s[j])
                len = j-i-1;
            if(len>max_len)
                max_len = len;
        }
    }
    return max_len;
}
