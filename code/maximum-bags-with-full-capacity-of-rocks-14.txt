class Solution {
public:
    int maximumBags(vector<int>& capacity, vector<int>& rocks, int add) {
        vector<int>diff;
        int c = 0;
        for(int i=0;i<capacity.size();i++){
            diff.push_back(capacity[i] - rocks[i]);
        }
        
        sort(diff.begin(), diff.end());
        for(int i=0;i<diff.size();i++){
            if(add >= diff[i]){
                add -= diff[i];
                c++;
            }
        }
        
        return c;
    }
};
