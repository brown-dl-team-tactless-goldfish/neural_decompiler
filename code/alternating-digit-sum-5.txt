class Solution {
public:
    int alternateDigitSum(int n) {
        // Convert integer n to string and store it in a variable
        string num_str = to_string(n);

        // Reverse the string using the reverse function
        reverse(num_str.begin(), num_str.end());

        // Convert the reversed string back to an integer
        n = stoi(num_str);

        int sum=0,m=n;
        if(n<10)
            return n;
        int count=1;

        // Traverse through the digits of the number from right to left
        while(n) 
        {    
            // If the count is even, subtract the digit from the sum
            if(count%2==0)
            {
                sum-=n%10;
                n=n/10;
            }
            // If the count is odd, add the digit to the sum
            else{
                sum+=n%10;
                n=n/10;
            }
            count++;
        }
        // Return the final sum
        return sum;
    }
};
