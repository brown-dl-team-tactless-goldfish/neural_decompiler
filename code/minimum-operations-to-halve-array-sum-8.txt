class Solution {
public:
    int halveArray(vector<int>& nums) {
        double tSum = accumulate(nums.begin(), nums.end(), 0.0), tempSum = tSum;
        priority_queue<double> maxh(nums.begin(), nums.end());
        
        int ans = 0;
        for(; tempSum >  tSum / 2; ans++) {
            double a = maxh.top(); maxh.pop();
            tempSum -= a / 2;
            maxh.push(a / 2);
        }
        return ans;
    }
};
