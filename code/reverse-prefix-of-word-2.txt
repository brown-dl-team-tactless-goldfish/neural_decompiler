void reverse(char* s, int left, int right)
{
    char tmp;
    while(left < right)
    {
        tmp = s[left];
        s[left++] = s[right];
        s[right--] = tmp;
    }
}

char * reversePrefix(char * word, char ch){
    int slow = 0, fast = 0;
    while(fast < strlen(word))
    {
        if(word[fast] == ch)
        {
            reverse(word, slow, fast);
            break;
        }
        fast++;
    }
    return word;
}
