class Solution {
public:
    vector<int> rearrangeArray(vector<int>& nums) {
        sort(nums.begin(), nums.end());
        vector<int> ans(nums.size());
        int n = nums.size();
        int j = 0;
        for(int i = 0; i < (n+1)/2; i++) {
            ans[j] = nums[i];
            j+=2;
        }
        j = 1;
        for(int i = (n+1)/2; i < n; i++) {
            ans[j] = nums[i];
            j+=2;
        }
        return ans;
    }
};
