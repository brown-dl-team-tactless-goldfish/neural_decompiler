int longestSubsequence (int* arr, int arrSize, int difference) {
    int  i, idx, max = 0;
    int  cnt[20001] = {0};
    
    for (i = 0; i < arrSize; i++) {
        idx = 10000 + arr[i];
        if (max < cnt[idx]) {
            max = cnt[idx];
        }
        if (0 <= idx + difference && idx + difference <= 20000) {
            cnt[idx + difference] = cnt[idx] + 1;
        }
    }
    
    return max + 1;
}
