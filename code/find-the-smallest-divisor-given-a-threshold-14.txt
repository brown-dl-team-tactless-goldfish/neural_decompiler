    int smallestDivisor(vector<int>& a, int threshold) {
        int l = 1, h = 1e6;
        while(l < h) {
            int m = l + (h-l)/2, sum = 0;
            for(auto e : a) sum += (e + m-1)/m;//ceil(e/(double)m);
            if(sum <= threshold) h = m;
            else l = m + 1;
        }
        return l;
    }
