class Solution {
public:
    long long sumofdigit(long long n)
    {
        long long sum=0;
        while(n)
        {
            sum+=n%10;
            n/=10;
        }
        return sum;
    }
    long long makeIntegerBeautiful(long long n, int target) {
        long long mod=1;
        long long ans=0;
        while(sumofdigit(n)>target)
        {
            while(n%mod==0)  // to avoid 0 such that after substraction only non zero digits becomes zero
            {
                mod*=10;
            }
            ans+=(mod-n%mod);
            n+=(mod-n%mod);
        }
        return ans;
    }
};
