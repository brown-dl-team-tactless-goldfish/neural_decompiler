static int gcd(int n,int m) {
    if (m == 0) {
        return n;
    } else {
        return gcd(m, n%m);
    }
}

int findGCD(int* nums, int numsSize){
    int max = nums[0];
    int min = nums[0];
    for (int i = 0; i < numsSize; ++i) {
        if (nums[i] < min) min = nums[i];
        if (nums[i] > max) max = nums[i];
    }
    return gcd(max,min);
}
