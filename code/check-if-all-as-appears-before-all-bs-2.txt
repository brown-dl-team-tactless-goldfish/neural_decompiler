

bool checkString(char * s){
    int lock=0,i;
    
    for(i=0;s[i];i++)
    {
        if(lock==0 && s[i]=='b')                //once we encountered 'b we'll set lock
            lock=1;
        else if(lock==1 && s[i]=='a')           // if 'a' appears in lock then return false
            return false; 
    }
    
    return true;                                //else return true
}
