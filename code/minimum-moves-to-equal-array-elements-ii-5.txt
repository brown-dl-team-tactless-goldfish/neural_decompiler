unsigned long long findCnt(int* nums, int numSize, int target){
    unsigned long long count = 0;
    for(int x = 0; x < numSize; x++){
        if(nums[x] >= target)count += (nums[x] - target);
        else count += (target - nums[x]);
    }
    return count;
}
