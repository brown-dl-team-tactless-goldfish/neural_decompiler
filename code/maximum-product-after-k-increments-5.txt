class Solution {
public:
    int maximumProduct(vector<int>& nums, int k) {
        long long ans = 1, mod = 1e9 + 7;
        priority_queue<int, vector<int>, greater<int>> pq(nums.begin(), nums.end());
        
        while(k--) {
            int a = pq.top(); pq.pop();
            pq.push(a + 1);
        }
        
        while(pq.size()) {
            ans *= pq.top(); pq.pop();
            ans %= mod;
        }
        return ans;
    }
};
