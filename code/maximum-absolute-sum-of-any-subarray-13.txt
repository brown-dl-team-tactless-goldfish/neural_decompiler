
class Solution {
public:
    int maxAbsoluteSum(vector<int>& nums) {
        int mxPos = 0, mxNeg = 0, ans = 0;
		for (int i = 0; i < nums.size(); i++) {
			mxPos += nums[i];
			mxNeg += nums[i];
			mxPos = max(0, mxPos);
			if (mxNeg > 0) mxNeg = 0;
			ans = max({ans, mxPos, -mxNeg});
		}
		return ans;
    }
};
