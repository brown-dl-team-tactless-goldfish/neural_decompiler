bool divideArray(int* nums, int numsSize){
    int pairs = numsSize/2;
    int res=0;
    int h[501]={0};
    for(int i=0;i<numsSize;i++)
        h[nums[i]]++;
    for(int i=0;i<numsSize;i++)
    {
        if(h[nums[i]]%2!=0)
            return false;
        res=res+h[nums[i]]/2;
        h[nums[i]]=0;
    }
    return res==pairs;
}
