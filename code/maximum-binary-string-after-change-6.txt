class Solution {
public:
    string maximumBinaryString(string binary) {
        int n = binary.size(), first = -1, cnt = 0;
        for(int i=n-1; i>=0; i--) {
            cnt += binary[i] == '0';
            if(binary[i] == '0') first = i;
            binary[i] = '1';
        }
        if(!cnt) return binary;
        binary[first + cnt - 1] = '0';
        return binary;
    }
};
