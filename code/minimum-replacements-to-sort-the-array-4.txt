    public long MinimumReplacement(int[] nums)
    {
        long result = 0;
        int currMin = nums.Last();
        for (int i = nums.Length - 2; i >= 0; i--)
        {
            int div = (int)Math.Ceiling((decimal)nums[i] / currMin);
            result += div - 1;
            currMin = nums[i] / div;
        }
        return result;
    }
