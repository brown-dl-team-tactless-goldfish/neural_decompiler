class Solution {
public:
    long long getDescentPeriods(vector<int>& prices) {
        long long result=0;
        int n=prices.size();
        vector<long long> dp(n,1);
        result+=1;
        for(int i=1;i<n;i++)
        {
            if(prices[i]==prices[i-1]-1)
                dp[i]+=dp[i-1];
            result+=dp[i];
        }
        return result;
    }
};
