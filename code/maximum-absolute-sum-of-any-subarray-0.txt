int maxAbsoluteSum(int* nums, int numsSize){
    int curSum = nums[0], maxSum = nums[0];
    
    for(int i = 1; i < numsSize; i++) {
    // if including currentNum is -ve then currentSum start with currentNum
        curSum = fmax(nums[i], curSum + nums[i]);
        maxSum = fmax(maxSum, curSum);
    }
    
    curSum = nums[0];
    int minSum = nums[0];
    
    for(int i = 1; i < numsSize; i++) {
    // if including currentNum is -ve then currentSum start with currentNum
        curSum = fmin(nums[i], curSum + nums[i]);
        minSum = fmin(minSum, curSum);
    }
    
    if(abs(maxSum) > abs(minSum))
        return abs(maxSum);
    else
        return abs(minSum);
}
