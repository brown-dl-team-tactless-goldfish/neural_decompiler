double angleClock(int hour, int minutes){
    double maffhr=(double)minutes*30/60; //calculating the angle by which the hour hand is affected due to the minutes hand
    double hrangle=(double)hour*30+maffhr; //calculating the angle made by hour hand with 12
    double miangle=(double)minutes*30/5; //calculating the angle made by minutes hand with 12
    double ang=fabs(hrangle-miangle); //calculating the angle difference between the hour and minutes hand
    return fmin(ang,360-ang); //returns the minimum angle as been told in the description
}
