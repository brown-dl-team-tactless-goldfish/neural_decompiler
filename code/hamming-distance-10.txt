 int hammingDistance(int x, int y)
 
 {
         int  count = 0 ;
     for ( int i = 0 ; i < 32 ; i++)
     {
         //get bit of x
         
         if ( x  & ( 1u << i))
         {
             if ( !(y & ( 1u << i)))
             {
                 count++;
             }
             
         }
         else
         {
             
             if ( (y & ( 1u << i)))
             {
                 count++;
             }
             
         }
     }
     
     return count;
 }
