int numWays(int steps, int arrLen) {
    vector<int> v1(max(steps, arrLen + 2)), v2(max(steps, arrLen + 2));
    v1[1] = 1;
    while (steps-- > 0) {
        for (auto i = 1; i <= arrLen; ++i) {
            v2[i] = ((long long)v1[i] + v1[i - 1] + v1[i + 1]) % 1000000007;
            if (v2[i] == 0) break;
        }
        swap(v1, v2);
    }
    return v1[1];
}
