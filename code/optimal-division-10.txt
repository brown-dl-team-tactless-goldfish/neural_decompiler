	// edge cases
	
    if(nums.size() == 1) return std::to_string(nums[0]);
    if(nums.size() == 2) return std::to_string(nums[0]) + "/" + std::to_string(nums[1]);

    // rest cases

    std::string res;

    res = std::to_string(nums[0]) + "/(";
    
    for(size_t j = 1; j < nums.size(); j++)
    {
        res += std::to_string(nums[j]) + "/";
    }
    
    const size_t DIM = res.size();
    res.resize(DIM-1);
    res.push_back(')');
    
    return res;
}
