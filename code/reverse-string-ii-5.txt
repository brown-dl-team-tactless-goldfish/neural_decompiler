char* reverseStr(char* s, int k) {
    int size = strlen(s);
    int twoK = k << 1;
    int bound;
    for(int i = 0; i < size; i+= twoK)
    {
        if (i + k <= size)
        {
            bound = i + k - 1; 
        }
        else
        {
            bound = size - 1;
        }
        for(int j = i, l = bound; j < l; j++, l--)
        {
                s[j] ^= s[l];
                s[l] ^= s[j];
                s[j] ^= s[l];
        }
    }
    return s;
}
