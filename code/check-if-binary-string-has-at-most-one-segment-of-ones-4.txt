class Solution:
    def checkOnesSegment(self, s: str) -> bool:
        data = s.split("0")

        print ("data = ", data)
        
        newData = []
        for d in data:
            if (d != ""):
                newData.append(d)
                
        print ("newData = ", newData)
        
        if (len(newData) == 1):
            return True
        else:
            return False
