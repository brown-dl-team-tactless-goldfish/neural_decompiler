int splitNum(int num){
    int* digit = (int*)calloc(10, sizeof(int));
    while(num){
        digit[num % 10]++;
        num /= 10;
    }
    int nums[2] = {0, 0};
    int idx = 0, flag = 0;
    while(idx < 10){
        if(digit[idx]){
            nums[flag] *= 10;
            nums[flag] += idx;
            flag ^= 1;
            digit[idx]--;
        }
        else
            idx++;
    }
    return nums[0]+nums[1];
}
