int secondHighest(char * s){
    int* digit = calloc(10, sizeof(int)) ;
    for(int i = 0; i < strlen(s); i++){
        if(s[i] >= '0' && s[i] <= '9'){
            int d = s[i] -'0' ;
            digit[d]++;
        }    
    }
    int cn = 0;
    for(int i = 9; i >=0; i--){
        if(digit[i]){
            cn++ ;
            if(cn == 2)
                return i;
        }
    }
    return -1;
}
