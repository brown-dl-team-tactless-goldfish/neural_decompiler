int compare(const void *x, const void *y)
{
    return *((int *)x) > *((int *)y);
}

int maximumProduct(int *nums, int numsSize)
{
    qsort(nums, numsSize, sizeof(int), compare);
    int t1 = nums[0] * nums[1] * nums[numsSize - 1];
    int t2 = nums[numsSize - 3] * nums[numsSize - 2] * nums[numsSize - 1];
    return t1 > t2 ? t1 : t2;
}
