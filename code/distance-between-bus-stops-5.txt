int min(int a, int b){
	return a<b? a:b;
}

int distanceBetweenBusStops(int* distance, int distanceSize, int start, int destination){

	int i=0;
	int s_d_sum=0, total_sum=0;

	if(start>destination){ // swap start and destination
		start=start+destination;
		destination=start-destination;
		start=start-destination;
	}

	for(i=0;i<distanceSize;i++){
		if(i>=start && i<destination)
				s_d_sum+=distance[i];

		total_sum+=distance[i];
	}
	return min(s_d_sum, (total_sum-s_d_sum));
}
