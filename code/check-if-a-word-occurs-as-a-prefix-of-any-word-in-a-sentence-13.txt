int isPrefixOfWord(char * sentence, char * searchWord){
    int i = 0;
    char *s;
    
    // Use strtok() to split a string using one or more delimeters.
    s = strtok(sentence, " ");
    while(s != NULL){
        // strncmp() only compares the first n characters.
        if(!strncmp(s, searchWord, strlen(searchWord)))
            return i + 1;
        
        i++;
        s = strtok(NULL, " ");
    }
    return -1;
}
