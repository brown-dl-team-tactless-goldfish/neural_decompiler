int is_prime (int n)
{
    int c;
    for (c = 0; n; c++)
        n &= n - 1;
    if (c % 2)
        return (c == 3 || c == 5 || c == 7 || c == 11 || c == 13|| c == 17|| c == 19|| c == 23|| c == 29|| c == 31);
    return (c == 2);
}

int countPrimeSetBits(int left, int right)
{
    int count = 0;
    for (int i = left; i <= right; i++)
        count += is_prime (i);

    return count;
}
