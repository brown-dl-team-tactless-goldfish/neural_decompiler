public class Solution {
    public int FindBottomLeftValue(TreeNode root) {
        Queue<TreeNode> nodes = new Queue<TreeNode>();
        nodes.Enqueue(root);
        
        int value = 0;
        while (nodes.Count != 0){
            int count = nodes.Count;
            
            for (int i = 0; i < count; i++){
                var c = nodes.Dequeue();
                if (c.left != null){
                    nodes.Enqueue(c.left);
                }
                
                if (c.right != null){
                    nodes.Enqueue(c.right);
                }
                
                if (i == 0){
                    value = c.val;
                }
            }
        }
        
        return value;
    }
}
