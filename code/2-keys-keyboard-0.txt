int minSteps(int n)

{

 int count=0,copy=0,A=1;
    
    while(A!=n)
    {
        
     if(copy==0)
     {
         copy++;
         count+=2;
         A++;
     }
     else if(((n-A)%(A))==0)
     {
         copy=A;
         A+=copy;
         count+=2;
     }
     else
     {
         A+=copy;
         count++;
     }
            
    }
     
   return count; 
     
}
