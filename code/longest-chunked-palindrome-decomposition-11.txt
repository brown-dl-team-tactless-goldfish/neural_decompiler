public class Solution {
    public int LongestDecomposition(string text)
        {
            return longestDe(text, 0);
        }
        public int longestDe(string text, int c)
        {
            int length = text.Length;
            if (length == 0)
            {
                return c;
            }
            for (int i = 1; i <= length / 2; i++)
            {
                if (text.Substring(0,i).Equals(text.Substring(length - i, i)))
                {
                    return longestDe(text.Substring(i, length - 2 * i), c + 2);
                }
            }
            return c + 1;
        }
}
