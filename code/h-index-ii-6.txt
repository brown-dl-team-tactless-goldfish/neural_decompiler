/*
    Time: O(n)
    Space: O(1)
*/

public class Solution {
    public int HIndex(int[] citations) {
        
        // at index i, paper i is cited citations[i] times
        // the number of papers who have at least citations[i] is (n - i)
        // h-index is the max (n - i) value, so we need to find the smallest i that meet the criteria
        
        int n = citations.Length; 
        
        for(int i = 0; i < n; i++)
        {
            // (n-i) papers have at least citation[i]
            if(citations[i] >= n - i)
                return n - i;
        }

        return 0;
    }
}
