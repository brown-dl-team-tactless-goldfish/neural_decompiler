while (cnt < arrSize)
{
    val = arr[cnt];
    for (int i = 0; i < arrSize; i++)
    {
        if ((val == arr[i])&&(val != NULL))
        {
            tmp[cnt] += 1;
            arr[i] = NULL;
        }
    }
    val = tmp[cnt];
    for (int i = 0; i < cnt; i++)
    {
        if ((val == tmp[i])&&(val != 0)&&(cnt != i))
        {
            return false;
        }
    }
    cnt++;
}
