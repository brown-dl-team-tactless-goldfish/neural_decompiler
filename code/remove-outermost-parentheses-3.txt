char * removeOuterParentheses(char * S){
    int Slen = strlen(S);
    char* newString = (char*)calloc(Slen + 1, sizeof(char));
    int value = 0;
    int flag = 0;
    int i;
    int cur = 0;
    for (i = 0; i < Slen; i++)
    {
        if (value)
            flag = 1;
        if (S[i] == '(')
            value += 1;
        else
            value -= 1;
        if (flag && value)
            newString[cur++] = S[i];
        flag = 0;
    }
    return newString;
}
