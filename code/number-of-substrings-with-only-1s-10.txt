class Solution {
public:
    int numSub(string s) {
       const int mod = 1e9 + 7;
       unsigned long long n = s.length() , c = 0 , ans = 0;
       for(int i = 0 ; i < n ; i++)
       {
           if(s[i] == '1')
               ++c;
           if(s[i] == '0')
           {
               ans += (((c*(c+1)) % mod)/2)%mod;
               c = 0;
           }
       }
        if(c)
            ans += (((c*(c+1)) % mod)/2)%mod;
        return ans;
    }
};
