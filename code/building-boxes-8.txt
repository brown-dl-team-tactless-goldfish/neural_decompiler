public class Solution {
    public int MinimumBoxes(int n) {
        int x = 1;
        int boxes = 0;
        int max = 1;
        int toAdd = 1;
        while (x <= n) {
            if (toAdd < max) {
                x += toAdd;
                toAdd++;
                boxes++;
            }
            else {
                toAdd = 1;
                max++;
            }
        }
        return boxes;
    }
}
