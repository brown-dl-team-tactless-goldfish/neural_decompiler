/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     struct ListNode *next;
 * };
 */

int help(struct ListNode *next, int * ans) {
    
    int power;
    if(next == NULL) 
        return 1;
    
    power = help(next->next, ans);
    *ans += (next->val * power);
    return power*2;
}

int getDecimalValue(struct ListNode* head){
    
    int ans = 0;
    
    help(head, &ans);
    
    return ans;

}
