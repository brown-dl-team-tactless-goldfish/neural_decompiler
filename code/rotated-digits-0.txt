bool IsGoodInteger(int val){
    bool flag = false ;
    int d ;
    while(val){
        d = val % 10 ;
        if(d == 3 || d== 4 || d== 7)
            return false ;
        if(d == 2 || d == 5 || d == 6 || d== 9)
            flag = true ;
        val /= 10 ;
    }
    return flag ;
}
int rotatedDigits(int n){
    int ans = 0 ;
    for(int i = 1; i <= n ; i++){
        ans += IsGoodInteger(i) ;
    }
    return ans ;
}
