class Solution {
public:
    vector<vector<int>> construct2DArray(vector<int> & original_numbers, int row, int col) {
        int r, c;
        vector<vector<int>> converted_numbers;
        
        if (row * col == original_numbers.size()) {
            converted_numbers = vector<vector<int>>(row, vector<int>(col));
            
            r = c = 0;
            for (vector<int>::const_reference original_number : original_numbers) {
                converted_numbers[r][c++] = original_number;
                
                if (c == col) {
                    c = 0;
                    r++;
                }
            }
        }
        
        return converted_numbers;
    }
};
