class Solution {
public:
    int findSmallestInteger(vector<int>& nums, int value) {
        
        vector<int>mod(value) ;
        for(auto &x : nums){
            int m = (x % value + value) % value ;
            mod[m]++ ;
        }
        int minimum = INT_MAX ;
        int idx = -1 ;
        for(int i = 0; i < value; i++){
            if(mod[i] < minimum){
                minimum = mod[i] ;
                idx = i ;
            }
        }
        return minimum * value + idx ;
    }
};
