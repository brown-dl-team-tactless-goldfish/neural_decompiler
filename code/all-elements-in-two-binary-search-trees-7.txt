# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def getAllElements(self, root1: TreeNode, root2: TreeNode) -> List[int]:
        self.res = []
        def inOrder(curNode):
            if (curNode == None):
                return
            

            inOrder(curNode.left)
            self.res.append(curNode.val)
            inOrder(curNode.right)
            
        inOrder(root1)
        inOrder(root2)
        self.res.sort()
        return self.res
