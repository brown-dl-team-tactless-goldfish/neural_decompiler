public class Solution {
    public int TimeRequiredToBuy(int[] tickets, int k) {
        
        int res  = 0;
        
        for(int i = 0; i < tickets.Length; i++)
        {
            if(i <= k)
                res += Math.Min(tickets[i], tickets[k]);
            else if(i > k)
                res += Math.Min(tickets[i], tickets[k] - 1);
        }
       
        return res;
    }
}
