class Solution {
public:
    int minOperations(vector<int>& nums, vector<int>& numsDivide) {
        int g=numsDivide[0];
        for(auto &i:numsDivide){
            g=__gcd(g,i);
        }
        sort(nums.begin(),nums.end());
        for(int i=0;i<nums.size();i++){
            if(g%nums[i]==0)return i;
        }
        return -1;
    }
};
