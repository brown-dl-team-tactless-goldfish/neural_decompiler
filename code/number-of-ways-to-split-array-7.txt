class Solution {
public:
    int waysToSplitArray(vector<int>& nums) {
        int n = nums.size();
        long sum = 0;
        for(int i=0;i<n;i++){
            sum += nums[i];
        }
        int ans = 0;
        long curr = 0;
        for(int i=0;i<n-1;i++){
            curr+=nums[i];
            if(curr>= (sum - curr))
                ans++;
        }
        return ans;
    }
};
