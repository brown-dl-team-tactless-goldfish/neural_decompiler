/**
 * Definition for an interval.
 * struct Interval {
 *     int start;
 *     int end;
 * };
 */
int cmp(const void* a,const void* b){
        return ((struct Interval*)a)->end-((struct Interval*)b)->end;
}
int eraseOverlapIntervals(struct Interval* intervals, int intervalsSize) {
    qsort(intervals,intervalsSize,sizeof(intervals),cmp);
    int count=0,i=0,j;
    while(i<intervalsSize){
        j=i;
        count++;
        while(i<intervalsSize&&intervals[++i].start<intervals[j].end);
    }
    return intervalsSize-count;
}
