int maximumGroups(int* grades, int gradesSize){
    int left = 1, right = (gradesSize/2 + 1);
    int mid, sum;
    while(left < right){
        mid = right - (right - left)/2;
        sum = (1 + mid)* mid / 2;
        if(sum == gradesSize)
            return mid;
        if(sum > gradesSize)
            right = mid - 1;
        else
            left = mid;
    }    
    return left;
}
