bool isPossibleDivide(vector<int>& nums, int k) {
        unordered_map<int,int> freq;
        if(nums.size()%k!=0) return false;
        sort(nums.begin(),nums.end());
        for(auto n:nums) freq[n]++;
        for(auto n:nums){
            if(freq[n]>0){
                int c = freq[n];
                for(int j=n+k-1;j>=n;--j) {
                    freq[j]-=c;
                    if(freq[j]<0) return false;
                }
            }
        }
        return true;
    }
