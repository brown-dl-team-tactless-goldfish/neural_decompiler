class Solution {
public:
    int matchPlayersAndTrainers(vector<int>& p, vector<int>& t) {
        sort(p.begin(),p.end());
        sort(t.begin(),t.end());
        int ans = 0, i = 0;
        for(auto ele : p){
            while(i < t.size()){
                if(t[i] >= ele) {ans++; i++; break;}
                i++;
            }
        }
        return ans;
    }
};
