class Solution {
public:
    int minOperationsMaxProfit(vector<int>& customers, int boardingCost, int runningCost) {
        int rot = -2, mxm = 0, waiting = 0, n = customers.size(), profit = 0, i = 0;
        while (waiting > 0 or i < n) {
            if (i < n) 
                waiting += customers[i];
            profit = profit + min(4, waiting) * boardingCost - runningCost;
            if (profit > mxm) {
                mxm = profit;
                rot = i;
            }
            waiting -= min(4, waiting);
            ++i;
        }
        return rot+1;
    }
};
