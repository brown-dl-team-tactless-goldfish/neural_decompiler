class Solution {
public:
    int minMaxDifference(int num) {
        string s = to_string(num);
        string t = s;
        int l=s.size(), i=0;
        if(s[i]=='9'){
            while(i<l && s[i]=='9') ++i;
        }
        char c = s[i];
        for(int j=i; j<l; ++j){
            if(s[j] == c) s[j] = '9';
        }
        char d = t[0];
        for(int j=0; j<l; ++j){
            if(t[j] == d) t[j] = '0';
        }
        return stoi(s)-stoi(t);
    }
};
