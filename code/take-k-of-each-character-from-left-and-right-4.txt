    int takeCharacters(string s, int k) {
        vector<int> cnt(128, -k);
        for(auto c:s) cnt[c]++;
        for(char c : {'a', 'b', 'c'}) if(cnt[c] < 0) return -1;
        int longestMid = 0;
        for(int l=0,r=0,n=s.size(); r<n; cnt[s[l++]]++){
            while(r<n && cnt[s[r]] > 0) cnt[s[r++]]--;
            longestMid = max(longestMid, r-l);
        }   
        return s.size() - longestMid;
    }
