public List<Integer> intersection(int[][] nums) {
        Map<Integer, Integer> countMap = new HashMap<>();
        Arrays.stream(nums).forEach(a -> Arrays.stream(a).forEach(x -> countMap.put(x, countMap.getOrDefault(x, 0) + 1)));
        return countMap.entrySet().stream().filter(e -> e.getValue() == nums.length).map(Map.Entry::getKey).sorted().collect(Collectors.toList());
    }
