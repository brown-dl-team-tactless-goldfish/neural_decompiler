bool isMonotonic(int* nums, int numsSize)
{
    int  increase = 0 , decrease = 0;
    for ( int i = 0 ; i < numsSize-1 ; i++)
    {
        if ( (nums[i] <= nums[i+1]))
        {
            increase++;
        }
        
        if ( (nums[i] >= nums[i+1]) )
        {
            decrease++;
        }
    }

    
    if ( increase ==  numsSize -1 ||  decrease == numsSize -1 )
        return true;
    else
        return false;
}
