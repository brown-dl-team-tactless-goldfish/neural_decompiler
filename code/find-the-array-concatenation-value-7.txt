class Solution {
    long long concatenate(int a, int b)
    {
        int c = b;
        long long res = a;
        while (c > 0)
        {
            res = res * 10;
            c = c / 10;
        }
        return res + b; 
    }
public:
    long long findTheArrayConcVal(vector<int>& nums) {
        long long res = 0;
        auto iterForward = nums.begin();
        auto iterBackward = nums.end();
        iterBackward--;
        while (iterForward <= iterBackward)
        {
            if (iterForward == iterBackward)
            {
                res += *iterForward;
            }
            else
            {
                res += concatenate(*iterForward, *iterBackward);
            }
            iterForward++;
            iterBackward--;
        }
        return res;
    }
};
