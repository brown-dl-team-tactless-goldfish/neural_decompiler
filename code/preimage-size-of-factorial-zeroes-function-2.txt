public class Solution 
{
    public int PreimageSizeFZF(int n) 
    {
       return (int) (FindFirst(n + 1) - FindFirst(n));
    }
    
    private long FindFirst(long count)
    {
        long start = 0, end = long.MaxValue;
        while(start < end)
        {
            long mid = start + (end - start) / 2;
            if(TrailingZerosInFactorial(mid) < count)
                start = mid + 1;
            else
                end = mid;
        }
        
        return (int) start;
    }
    
    private long TrailingZerosInFactorial(long n)
    {
        return n == 0 ? 0 : n / 5 + TrailingZerosInFactorial(n / 5);
    }
}
