char * addSpaces(char * s, int* a, int n)
{
    int k=0,l=0;
    char *res=(char*)malloc(sizeof(char)*(strlen(s)+n+1));
    if(a[0]==0)
    {
        res[k++]=' ';
        l=1;
    }
    for(int i=0;i<strlen(s);i++)
    {
        res[k++]=s[i];
        if(l!=n&&i+1==a[l])
        {
            res[k++]=' ';
            l++;
        }
    }
    res[k]='\0';
    return res;
}
