class Solution {
public:
    int shortestSequence(vector<int>& rolls, int k) {
        int ans = 0;
        set<int> st;
        for(int &it: rolls) {
            st.insert(it);
            if(st.size() == k) {
                ans += 1;
                st.clear();
            }
        }
        return ans + 1;
    }
};
