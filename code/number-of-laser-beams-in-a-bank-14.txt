class Solution {
public:
    int numberOfBeams(vector<string>& bank) {
    
    int prev = 0;
    int curr = 0;
    for (int i=0; i<bank.size(); i++){
        auto c = std::count( bank[i].begin(), bank[i].end(), '1');
        if (c!=0) {
            curr += prev * c;
            prev = c;
        }
    }
    return curr;
    }
};
