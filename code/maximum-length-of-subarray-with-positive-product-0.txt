int getMaxLen(int* nums, int numsSize){
    int i = 0, positives = 0, negatives = 0, temp, maxLen = 0;

    for (; i < numsSize; i++) {
        if (nums[i] == 0)
            positives = negatives = 0;
        else if (nums[i] > 0) {
            positives++;
            if (negatives > 0)
                negatives++;
        } else {
            temp = positives;
            positives = negatives > 0 ? negatives + 1 : 0; // Tricky line : To keep track of both positives and negatives sum if there are even number of negatives in the subarray
            negatives = temp + 1;
        }

        maxLen = positives > maxLen ? positives : maxLen;
    }

    return maxLen;
}
