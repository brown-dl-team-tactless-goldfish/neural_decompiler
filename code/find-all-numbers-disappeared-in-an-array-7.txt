int* findDisappearedNumbers(int* nums, int numsSize, int* returnSize){
    
    int hash[100001]={0};
    int i;
    int *res = (int *)malloc(sizeof(int) * numsSize);
    memset(res,0,sizeof(int) * numsSize);
    for (i=0; i < numsSize; i++)
    {
        hash[nums[i]]=1;
    }
    int j=0;
    for (i=1; i < numsSize+1; i++)
    {
        if (hash[i] == 0)
        {
            res[j] = i;
            j++;
        }
    }
    *returnSize = j;
     return &res[0];
}
