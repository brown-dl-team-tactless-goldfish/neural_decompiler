class Solution {
public:
    int countValidWords(string sentence) {
        int cnt = 0;
        // split the string into words
        for (int i = 0, goodChars = 0; i < sentence.size(); ++i) {
            if (isspace(sentence[i])) {
                if (goodChars > 0) {
                    auto word = sentence.substr(i - goodChars, goodChars);
                    cnt += isValidWord(word);
                }
                goodChars = 0;
            } else {
                ++goodChars;
                if (i == sentence.size() - 1) {
                    // special case for the last char
                    auto word = sentence.substr(i - goodChars + 1, goodChars);
                    cnt += isValidWord(word);
                }
            }
        }
        return cnt;
    }
    
    static bool isValidWord(const string& w) {
        bool hasHyphen = false;
        
        for (int i = 0; i < w.size(); ++i) {
            char c = w[i];
            if (isdigit(c) || isspace(c)) {
                return false;
            }
            if (isalpha(c)) {
                continue;
            }
            if (c == '-') {
                if (hasHyphen) {
                    // not allowed to have 2+ hyphens
                    return false;
                }
                hasHyphen = true;
                // There is at most one hyphen '-'. 
                // If present, it must be surrounded by lowercase characters 
                // ("a-b" is valid, but "-ab" and "ab-" are not valid).
                if (i == 0 || i == w.size() - 1) {
                    return false;
                }
                if (!isalpha(w[i-1]) || !isalpha(w[i+1])) {
                    return false;
                }
            } else if (c == '!' || c == '.' || c == ',') {
                // punctuation has to be the last char
                if (i != w.size() - 1) {
                    return false;
                }
            }
        }
        // cout << "valid word: " << w << endl;
        return true;
    }
};
