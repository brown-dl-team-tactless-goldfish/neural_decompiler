int helper(struct TreeNode* root, int* max)
{
    int left = 0, right = 0;
    
    if(root == NULL)
        return 0;
    
    left = helper(root->left, max);
    right = helper(root->right, max);
    
    // only count the edge, not the nodes
    if((*max) < (left+right))
        *max = left+right;
    
    // +1 is for the edge
    if(left > right)
        return left + 1;
    else
        return right + 1;
}

int diameterOfBinaryTree(struct TreeNode* root){
    int max = 0;
    
    helper(root,&max);
    
    return max;
}
