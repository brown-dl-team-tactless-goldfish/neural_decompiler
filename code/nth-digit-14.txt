class Solution {
public:
    int findNthDigit(int n) {
        int i = 1;
        long long s = 9;
        int pre = 0;
        while(n>s){
            i++;
            pre = s;
            s += 9*pow(10,i-1)*i;
        }
        int num = (n - pre -1)/i + pow(10,i-1);
        string tmp = to_string(num);
        int pos = (n - pre -1) % i;
        return tmp[pos] - '0';
        
    }
};
