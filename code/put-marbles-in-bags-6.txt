class Solution {
public:
    long long putMarbles(vector<int>& weights, int k) {
        vector<long long> v;
        for(int i=0;i<weights.size()-1;i++) v.push_back(weights[i] + weights[i+1]); 
        sort(begin(v), end(v));
        return accumulate(end(v)-(k-1), end(v), 0LL) - accumulate(begin(v), begin(v)+(k-1), 0LL);
    }
};
