    int digitsum(int x)
    {
        int n = 0;
        while (x > 0) {
            n += x%10;
            x /= 10;
        }
        return n;
    }
    
    int countEven(int num) {
        int ans = 0;
        for (int i = 1; i <= num; i++) {
            if (digitsum(i) %2 == 0) ans++;
        }
        return ans;
    }
