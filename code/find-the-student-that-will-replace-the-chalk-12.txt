class Solution {
public:
    int chalkReplacer(vector<int>& chalk, int k) {
     
        //first get the total chalks used for one repetetion by all students
       long long int total=0;
        for(auto i:chalk)
            total+=i;
            
         k=k%total;
         
           //we can find ans in one pass
        for(int i=0;i<chalk.size();i++)
        {
            if(chalk[i]>k)
            {
                return i;
            }
            k-=chalk[i];
        }
        return 0;
    }
};
