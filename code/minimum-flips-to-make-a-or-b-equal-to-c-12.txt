int minFlips(int a, int b, int c){
    int count = 0;
    for (int i = 0; i < 32; i++) {
        int cb = (c >> i) & 0x1;
        int ab = (a >> i) & 0x1;
        int bb = (b >> i) & 0x1;
        
        if (cb == 0)
            count += ab + bb;
        else if (ab == 0 && bb == 0)
            count++;
    }
    return count;
}
