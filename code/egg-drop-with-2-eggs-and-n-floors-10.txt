class Solution {
    int solve(vector<int>& dp, int n) {
        if (dp[n] != 1001) {
            return dp[n];
        }
        
        for (int i = 1; i <= n; i++) {
            dp[n] = min(dp[n], 1 + max(i - 1, solve(dp, n - i)));
        }
        
        return dp[n];
    }
public:
    int twoEggDrop(int n) {
        vector<int> dp(n+1, 1001);
        dp[0] = 0;
        return solve(dp, n);
    }
};
