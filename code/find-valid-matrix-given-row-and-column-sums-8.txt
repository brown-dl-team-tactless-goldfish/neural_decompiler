public class Solution {
    public int[][] RestoreMatrix(int[] rowSum, int[] colSum) {
        
        int m = rowSum.Length, n = colSum.Length;
        int[][] res = new int[m][];
        for(int k = 0; k < m; k++)
            res[k] = new int[n];
        
        int row = 0, col = 0;
        while(row < m && col < n)
        {
            int min = Math.Min(rowSum[row], colSum[col]);
            res[row][col] = min;
            
            rowSum[row] -= min;
            if(rowSum[row] == 0)
                row++;
            
            colSum[col] -= min;
            if(colSum[col] == 0)
                col++;
        }
    
        return res;
    }
}
