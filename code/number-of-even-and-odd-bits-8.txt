class Solution {
public:
    vector<int> evenOddBit(int n) {
        vector<int> res;
        bitset<32> b(n);
        string a=b.to_string();
        int ev=0,od=0;
        reverse(a.begin(),a.end());
        for(int i=0;i<=31;i++){
            if(a[i]=='1'){
                od+=(i&1);
                ev+=(!(i&1));
            }
        }
        res.push_back(ev);
        res.push_back(od);
        return res;
    }
};
