class Solution {
public:
    long long maximumSubsequenceCount(string text, string pattern) {
        int countF = 0, countS = 0;
        for (auto& c : text){
            countF += c == pattern[0];
            countS += c == pattern[1];
        }
        long long ans = max(countF, countS);
        for (auto& c : text){
            countF -= c == pattern[0];
            countS -= c == pattern[1];
            if (c == pattern[0]) ans += countS;
        }
        return ans;
    }
};
