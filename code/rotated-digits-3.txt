

int rotatedDigits(int n){
    int i,c=0;
    bool flg=false;
    char* is_valid[]={"same","same","valid","invalid","invalid","valid","valid","invalid","same","valid"};
    for(i=2;i<=n;i++)
    {
        int r=i,flg=false;
        while(r>0){
            int t=r%10;
            if(is_valid[t]=="invalid") {flg=false;break;}
            if(is_valid[t]=="valid") flg=true;
            r=r/10;
        }
        if(flg) c++;
    }
    return c;
}
