int minimumRecolors(char * blocks, int k){
    int minCnt = k;
    int len = strlen(blocks);
    int r = 0, l = 0, whiteCnt = 0;
    while(r < len)
	{
        while(r-l<k)
            whiteCnt += (blocks[r++]=='W')? 1 : 0;
   
        if(whiteCnt == 0)
            return 0;
        else if(whiteCnt < minCnt)
            minCnt = whiteCnt;
        whiteCnt -= (blocks[l++]=='W')? 1 : 0;
    }
    return minCnt;
}
