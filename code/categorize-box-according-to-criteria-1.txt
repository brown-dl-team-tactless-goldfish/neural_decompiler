char * categorizeBox(long int length,long int width,long int height,long int mass){
 int a=(int)pow(10,4)-1;
        int b=(int)pow(10,9)-1;
        int x=(length>a || width>a||height>a ||(length*width*height)>b);
        int y= mass>=100;
        return (x==1 && y==1)?"Both":(x==1)?"Bulky":(y==1)?"Heavy":"Neither";
}
