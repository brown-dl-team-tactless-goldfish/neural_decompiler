void solve(TreeNode *root, int l, int &ml, vector<int> &v){
        if(!root)
            return;
        if(ml < l){
            v.push_back(root->val);
            ml = l;
        }
        solve(root->right, l+1, ml, v);
        solve(root->left , l+1, ml, v);
    }
    vector<int> rightSideView(TreeNode* root) {
        vector<int> v;
        
        int ml = 0;
        solve(root, 1, ml, v);
        
        return v;
    }
