    int maxSatisfied(vector<int>& customers, vector<int>& grumpy, int X) {
       int n = grumpy.size(), sum = 0, maxSum = 0, len = 0, delta = 0;
       for (int i = 0; i < n; i++) sum += grumpy[i] ? 0 : customers[i];
       for (int i = 0; i < n; i++) {
          delta += grumpy[i] ? customers[i] : 0;
          if (len < X) len++;
          else delta -= grumpy[i - X] ? customers[i - X] : 0;
          maxSum = max(maxSum, sum + delta);
       }
       return maxSum;
    }
