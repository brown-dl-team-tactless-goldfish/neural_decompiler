    int ans = INT_MIN;
    
    for(auto &x: points)
    {
        while(!pq.empty() && x[0] - pq.top().second > k)
        {
            pq.pop();
        }
        
        if(!pq.empty())
        {
            ans = max(ans, x[0]+x[1] +pq.top().first);
        }
        
        pq.push({x[1] - x[0], x[0]});
        
    }
    
    return ans;
    
}
