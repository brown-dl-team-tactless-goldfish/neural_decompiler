class Solution {
public:
    vector<int> findSmallestSetOfVertices(int n, vector<vector<int>>& edges) {
        
        unordered_map<int, int> indegree;
        vector<int> result;
        for(auto edge:edges)
        {
            int from = edge[0];
            int to = edge[1];
            
            indegree[to]++;
        }
        
        for(int i=0;i<n;i++)
        {
            if(indegree.find(i)==indegree.end())
            {
                result.push_back(i);
            }
        }
        
        return result;
    }
`};
