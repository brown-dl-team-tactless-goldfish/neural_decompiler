
void rotate(int** matrix, int matrixSize, int* matrixColSize)
{ 
    int n = matrixSize;
    for( int i = 0; i < (n+1) / 2; i++)
    {
      for(int j = 0; j < n/2; j++)
      {
          int a = matrix[i][j];
          int b = matrix[j][n-i-1];
          int c = matrix[n-i-1][n-j-1];
          int d = matrix[n-j-1][i]; 
       
           matrix[j][n-i-1] = a;
           matrix[n-i-1][n-j-1] = b;
           matrix[n-j-1][i] = c;
           matrix[i][j] = d;
      }
    }
}
