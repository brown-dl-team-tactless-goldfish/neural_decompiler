class Solution {
public:
    int superEggDrop(int K, int N) {
        vector<int> col(N, 0);
        iota(col.begin(), col.end(), 1);
        while (K-- > 1) {
            vector<int> buf{1};
            while (buf.back() < N) buf.push_back(1 + buf.back() + col.at(buf.size() - 1));
            swap(buf, col);
        }
        return col.size();
    }
};
