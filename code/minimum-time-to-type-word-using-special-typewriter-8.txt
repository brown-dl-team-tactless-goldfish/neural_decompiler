public class Solution {
    public int MinTimeToType(string word) {
        
        int count = 0;
        var cur = 'a';
        foreach(var c in word)
        {
            var diff = Math.Min(Math.Abs(c - cur), 26 - Math.Abs(c - cur));
            count += diff + 1;
            cur = c;
        }
        return count;
    }
}
