int balancedString(string s) {

	unordered_map<char, int> m;
	for (char& c : s) ++m[c];
	int res = INT_MAX;
	int t = s.length()/4;

	for (int i = 0, j = -1; i < s.length(); ++i) {
		--m[s[i]];
		while (j <= i && m['Q'] <= t && m['W'] <= t && m['E'] <= t && m['R'] <= t) {
			res = min(res, i - j);
			++j;
			++m[s[j]];
		}
	}

	return res;
}
