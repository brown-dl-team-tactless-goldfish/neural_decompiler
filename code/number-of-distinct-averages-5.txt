public class Solution 
{
    public int DistinctAverages(int[] nums) 
    {
        var ls = nums.OrderBy(x=>x).ToList();
        var hash = new HashSet<double>();
        while(ls.Count>=2)
        {
                hash.Add((ls[0] +ls[ls.Count-1])/2.0);
                ls.RemoveAt(ls.Count-1);
                ls.RemoveAt(0);
        }
        return hash.Count;
    }
}
