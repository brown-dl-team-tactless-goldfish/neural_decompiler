    for(int j=n-1 ; j>=0 ; j--)
    {
        char temp;
        int val;
        if(k - (j) > 26)
        {
            temp = 'z';
            val = 26;
        }
        else
        {
            temp = char(k-(j)+96);
            val = k-(j);
        }
        s[j] = temp;
        k -= val;
        if(k == j)
            break;
    }
    
    
    return s;
}
