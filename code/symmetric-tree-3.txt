class Solution {
public:
    bool util(TreeNode* root1, TreeNode* root2){
        if(root1 == NULL && root2 == NULL)
            return true;
        else if(root1 == NULL || root2 == NULL)
            return false;
        
        if(root1->val == root2->val && util(root1->left, root2->right) && util(root1->right, root2->left))
            return true;
        return false;
    }
    
    bool isSymmetric(TreeNode* root) {
        
        return util(root, root);
    }
};
