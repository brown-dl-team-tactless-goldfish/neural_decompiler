#include <stdbool.h>

char * bestHand(int* ranks, int ranksSize, char* suits, int suitsSize){
    bool all_same_suit = true;
    
    //check Flush
    for (int i = 0; i < suitsSize; i++) {
        if (i != 0 && (suits[i] != suits[i-1])) {
            all_same_suit = false;
        }
    }
    if (all_same_suit == true) {
        return "Flush";
    }
    
    //check Three of a Kind
    int rank_cnt[13] = {0};
    for (int i = 0; i < ranksSize; i++) {
        rank_cnt[ranks[i]-1]++;
    }
    
    for (int i = 0; i < 13; i++) {
        if (rank_cnt[i] >= 3) {
            return "Three of a Kind";
        }
    }
    
    //check Pair
    for (int i = 0; i < 13; i++) {
        if (rank_cnt[i] == 2) {
            return "Pair";
        }
    }
    
    return "High Card";
}
