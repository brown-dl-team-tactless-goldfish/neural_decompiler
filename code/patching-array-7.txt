class Solution {
public:
    int minPatches(vector<int>& nums, long long n) {
        sort(nums.begin(),nums.end());
        long long h = 0, re = 0;
        for(int i = 0;i<nums.size() && h<n;++i){
            long long a = min(n+1,(long long)nums[i]);
            while(h<a-1){ h = 2*h+1; re++; }
            h+=a;
        }
        while(h<n-1){ h = 2*h+1; re++; }
        return re;
    }
};
