 bool checkRecord(char * s)
 {
 
   int A = 0 ,  L = 0 ;
     for(int i = 0 ; i < strlen(s) ; i++)
     {
       if ( s[i] == 'A')
       {
          A++;
           if (A > 1)
               return false;
       }
         
     if ( s[i] == 'L')
       {
          L++;
             
           if ( L == 3)
               return false;
         continue;
       }
         
         L = 0;
         
         
 }
  return true;
 }
