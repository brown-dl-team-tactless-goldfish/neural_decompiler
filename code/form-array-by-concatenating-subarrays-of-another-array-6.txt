bool canChoose(vector<vector<int>>& groups, vector<int>& nums) {
	int curGroup = 0, curInd = 0, i = 0;
	while (i < nums.size()) {
		if (nums[i] == groups[curGroup][curInd]) {
			if (curInd + 1 < groups[curGroup].size())
				curInd++;
			else {
				curGroup++;
				curInd = 0;
			}
			i++;
		}
		else {
			if (curInd > 0)
				curInd = 0;
			else
				i++;
		}
		if (curGroup == groups.size())
			return true;
	}
	return curGroup == groups.size();
}
