class Solution {
public:
    string sortSentence(string s) {
        vector<string>a(10);
        string w,d;
        for(int i=0;i<s.size();i++){
            if(s[i]>=48 && s[i]<=57){
                a[s[i]-48]=w+" ";
                w="";
                i++;
            }else
            w+=s[i];
        }
        for(string b:a){
            d+=b;
        }
        d=d.substr(0,d.size()-1);
        return d;
    }
};
