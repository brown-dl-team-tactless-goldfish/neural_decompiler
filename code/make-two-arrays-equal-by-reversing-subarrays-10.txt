bool canBeEqual(int* target, int targetSize, int* arr, int arrSize){
    int hash_map[1001];
    memset(hash_map, 0, 1001 * sizeof(int));
    for (int i = 0; i < targetSize; i++)
    {
        hash_map[target[i]] ++;
    }

    for (int i = 0; i < arrSize; i++)
    {
        if (--(hash_map[arr[i]]) < 0)
        {
            return false;
        }
    }

    return true;
}
