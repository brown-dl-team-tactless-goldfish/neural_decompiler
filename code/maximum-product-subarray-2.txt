int maxProduct(int* nums, int s){
    int max = nums[0];
    int sum;
    
    for (int i=0; i<s; i++){
        sum = nums[i];
        printf ("sum: %d\n", sum);
        if (sum >= max)
                max = sum;
        for (int k=i+1; k<s; k++){
            sum *= nums[k];
            if (sum >= max)
                max = sum;
        }
    }
    return max;
}
