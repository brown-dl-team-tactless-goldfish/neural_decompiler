int maxAscendingSum(int* nums, int numsSize){
    if(numsSize == 1)
        return nums[0];
    int sum = nums[0];
    int max = 0;
    for(int i = 1; i < numsSize; i++){
        if(nums[i - 1] < nums[i])
            sum += nums[i];
        else{
            if(sum > max)
                max = sum;
            sum = nums[i];
        }
    }
    if(sum > max)
        max = sum;
    return max;
}
