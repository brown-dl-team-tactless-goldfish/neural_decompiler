class Solution {
public:
    int maxVowels(string s, int k, int cnt = 0, int max = 0) {
        auto isVowel = [](auto c) { return c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u'; };
        for (auto i{ 0 }; i < s.size(); ++i) {
            if (isVowel(s[i]))
                ++cnt;
            if (k <= i && isVowel(s[i - k]))
                --cnt;
            max = std::max(max, cnt);
        }
        return max;
    }
};
