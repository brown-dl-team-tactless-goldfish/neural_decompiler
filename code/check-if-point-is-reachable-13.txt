class Solution {
public:
    bool isReachable(int x, int y) {
        do {
            while ((x & 1) == 0)
                x >>= 1;
            while ((y & 1) == 0)
                y >>= 1;
            if (x < y)
                y += x;
            else if (x > y)
                x += y;
        } while (x != y);
        return x == 1 && y == 1;
    }
};
