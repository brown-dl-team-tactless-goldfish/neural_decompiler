
int countBinarySubstrings(char * s){
    
    int curLen = 0;
    int prevLen = 0;
    int len = strlen(s);
    int count = 0;
    int p = 0;
    int curState = s[p];
    
    while(p<len){
        
        if(s[p] == curState ){
            curLen++;
        }
        else{
            curState = s[p];
            prevLen = curLen;
            curLen = 1;
        }
        if(prevLen>=curLen) count++;
        p++;
       
    }
    return count;
    
}
