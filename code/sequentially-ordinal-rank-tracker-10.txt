public class SORTracker {
    
    int counter;
    SortedList<string, (string, int)> pq = new();

    public SORTracker() {
       counter = 0; 
    }
    
    public void Add(string name, int score) {
        string key = (int.MaxValue-score).ToString();
        pq.Add($"{key}_{name}", (name, score));
    }
    
    public string Get() {
        counter++;
        return pq.Values[counter-1].Item1;
    }
}
