#include<stdlib.h>

char * generateTheString(int n){
    char * s = calloc(n + 1, sizeof (char));
    if (n % 2 == 0) {
        while (n > 0) {
            s[--n] = 'a';
        }
        s[0] = 'b';        
    }
    else {
        while (n > 0) {
            s[--n] = 'a';
        }
    }
    
    return s;
}
