int cmp(const void* a, const void* b) {
    return *(const int*) a - *(const int*) b;
}

int minMoves2(int* nums, int numsSize) {
    qsort(nums, numsSize, sizeof(int), cmp);
    int ans = 0, tar = nums[numsSize/2];
    for(int i = 0; i < numsSize; i++) {
        ans += abs(tar - nums[i]);
    }
    return ans;
}
