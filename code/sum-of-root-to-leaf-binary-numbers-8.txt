void sumLeaf(struct TreeNode *root, int *sum, int pre)
{
    if(root == NULL) return;
    pre = (pre << 1) | root->val;
    if(root->left == NULL && root->right == NULL)
    {
        *sum += pre;
        return;
    }
    sumLeaf(root->left, sum, pre);
    sumLeaf(root->right, sum, pre);
}

int sumRootToLeaf(struct TreeNode* root){
    int result = 0;
    sumLeaf(root, &result, 0);
    return result;
}
