// Runtime: 18 ms, faster than 50.33% of C++ online submissions for Maximum Depth of N-ary Tree.
// Memory Usage: 10.7 MB, less than 74.99% of C++ online submissions for Maximum Depth of N-ary Tree.
class Solution {
public:
    int maxDepth(Node* root) {
        int depth = 0;
        traverse(root, 1, depth);
        return depth;
    }
private:
    void traverse(Node* root, int level, int& d) {
        if (root) {
            d = max(d, level);
            for (const auto& c : root->children)
                traverse(c, level + 1, d);
        }
    }
};
