class Solution {
public:
    int minStoneSum(vector<int>& piles, int k) {
        priority_queue<int> q;
        for(auto x: piles) q.push(x);
        while(k--) {
            int u = q.top();
            q.pop();
            int x = round(1.0 * u / 2);
            q.push(x);
        }
        int ans = 0;
        while(!q.empty()) {
            ans += q.top();
            q.pop();
        }
        return ans;
    }
};
