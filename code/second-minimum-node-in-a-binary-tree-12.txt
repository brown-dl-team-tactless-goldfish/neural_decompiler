int findSecondMinimumValue(struct TreeNode* root){
    struct TreeNode *m1 = NULL, *m2 = NULL, *n[25] = { root };
    for (int f = 0, b = 1 ; f < b ; f++) {
        n[f]->left ? n[b++] = n[f]->left : 0;
        n[f]->right ? n[b++] = n[f]->right : 0;
        !m1 || m1->val > n[f]->val ? m2 = m1, m1 = n[f] :
        (!m2 || m2->val > n[f]->val) && m1->val != n[f]->val ? m2 = n[f] : 0;
    }
    return m2 ? m2->val : -1;
}
