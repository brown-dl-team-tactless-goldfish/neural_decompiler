class Solution {
public:
    string decodeCiphertext(string str, int r) {
        int c = str.size() / r; 
        string ret; 
        for(int i = 0;i<c;i++) {
            for(int j = 0;j<r&&i+j<c;j++) {
                ret.push_back(str[j*c + i + j]);
            }
        }
        ret.erase(ret.find_last_not_of(' ')+1);
        return ret; 
    }
};
