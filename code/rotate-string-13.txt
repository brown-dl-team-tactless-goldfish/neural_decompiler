bool rotateString(char* A, char* B) {
    int length = strlen(A);
    char buff;
    
    for(int i = 0; i < length; i++){
        A = shift_left(A);
        if(A = B)
            return true;
        
    }
    return false;
}

void shift_left(char *A){
    char buff = A[0];
    for(int i = 0; i<strlen(A); i++){
        A[i]=A[i+1];
    }
    A[strlen(A)] = buff;
    
    return A;
}
