char * defangIPaddr(char * ad){
    char* r = (char*)malloc(22 * sizeof(char));
    int i;
    int ind = 0;
    for(i=0; *(ad + i) != '\0'; i++){
        char c = *(ad+i);
        if( c == '.'){
            r[ind++] = '[';
            r[ind++] = '.';
            r[ind++] = ']'; 
        }else r[ind++] = c;
       
    }
    r[ind]='\0';
    
    return r;
}
