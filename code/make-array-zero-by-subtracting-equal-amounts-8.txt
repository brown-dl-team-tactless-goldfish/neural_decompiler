class Solution {
public:
    int minimumOperations(vector<int>& nums) {
        sort(begin(nums),end(nums)) ;
        
        int ans = 0 , rem = 0 ;
        for(int i = 0 ; i < nums.size() ; ++i ){
            if(!nums[i] || rem >= nums[i]) continue ;
            nums[i] -= rem ;
            ++ans ;
            rem += nums[i] ;
        }
        return ans ;
    }
};
