int minFlips(int a, int b, int c){
    int res = 0;
    
    while (c != 0 || a != 0 || b != 0) {
        if (c&1) {
            res += 1 - ((a | b)&1);
        } else {
            res += (a&1) + (b&1);
        }
        
        a = a >> 1;
        b = b >> 1;
        c = c >> 1;
    }
    
    return res;
}
