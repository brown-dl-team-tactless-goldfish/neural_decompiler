int arithmeticTriplets(int* nums, int numsSize, int diff){
    int* count = (int*)calloc(201, sizeof(int));
    int ans = 0;
    for(int i = 0; i < numsSize; i++)
        count[nums[i]]++;
    
    int diff_1, diff_2, curr;
    for(int i = 0; i < numsSize; i++)
    {
        curr = nums[i];
        diff_1 = curr + diff;
        diff_2 = diff_1 + diff;
        ans += (diff_2 <= 200 && count[curr] && count[diff_1] && count[diff_2])?1:0;
    }
    return ans;
}
