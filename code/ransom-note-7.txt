bool canConstruct(char * ransomNote, char * magazine){
    bool tmp = true;
    
    while(*ransomNote){
        tmp = false;
        for(char *it=magazine;*it;it++){
            if(*ransomNote == *it){
                *it = ' ';
                tmp = true;
                break;
            }
        }
        
        if(!tmp) return false;
        
        ransomNote++;
    }
    
    return true;
}
