class Solution {
public:
    int islandPerimeter(vector<vector<int>>& grid) {
        int rows = grid.size(), cols = grid[0].size(), result = 0;
        for(int i = 0; i < rows; i ++) {
            for(int j = 0; j < cols; j ++) {
                if(!grid[i][j]) continue;
                result += 4;
                if(j && grid[i][j - 1]) result -= 2;
                if(i && grid[i - 1][j]) result -= 2;
            }
        }
        return result;
    }
};
