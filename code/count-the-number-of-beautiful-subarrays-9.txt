class Solution {
public:
    long long beautifulSubarrays(vector<int>& nums) {
        vector <int> prefixXor(nums.size()+1);
        prefixXor[0] = 0;
        for(int i=1;i<prefixXor.size();i++){
            prefixXor[i] = nums[i-1] ^ prefixXor[i-1];  
        }
        unordered_map <int,int> ump;
        long long ans = 0;
        for(auto &it:prefixXor){
            if(ump[it] > 0)
                ans += ump[it];
            ump[it]++;
        }
        return ans;
    }
};
