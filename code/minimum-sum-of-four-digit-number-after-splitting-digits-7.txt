int minimumSum(int num){
    int digits[4];
    int rem = 0;
    int p = 0;
    while (num != 0) {
       rem = num % 10; 
       num = num / 10;
       digits[p++] = rem;
    }
    qsort(digits, p, sizeof(*digits), cmp);
    return digits[0] * 10 + digits[3] + digits[1] * 10 + digits[2];
}

int cmp(const void *a, const void *b) {
    int *x = (int*) a;
    int *y = (int*) b;
    return *x - *y;
}
