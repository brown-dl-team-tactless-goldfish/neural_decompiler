

int tribonacci(int n){
int f[38];
   f[0] = 0;
    f[1] = f[2] = 1;
    int i,c=0;
    if (n==0)
        return f[0];
    if (n==1 || n==2)
        return f[2];
    for (i=3;i<=n;i++) {
       c = f[2] + f[1] + f[0];
        f[0] = f[1];
        f[1] = f[2];
        f[2] = c;
    }
    return c;
}
