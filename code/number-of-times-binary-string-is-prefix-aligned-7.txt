class Solution {
public:
    int numTimesAllBlue(vector<int>& light) {
        
        int n = light.size();
        int sum1 = 0, sum2 = 0, count = 0;
        for (int i = 0; i < n; i++) {
            sum1 += light[i];
            sum2 += (i+1);
            if (sum1 == sum2)
                count++;
        }
        
        return count;
    }
};
