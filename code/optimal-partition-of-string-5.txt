int partitionString(char * s){
    int L = strlen(s), alpha[26] = {0};
    int numPartitions = 1, i, j;
    i = 0;
    while (i < L) {
  		if (alpha[s[i] - 'a'] == 0) {
  			alpha[s[i] - 'a'] = 1;
  			++i;
  		} else {
  			for (j = 0; j < 26; ++j) {
  				alpha[j] = 0;
  			}
  			++numPartitions;
      }
    }
    return numPartitions;
}
