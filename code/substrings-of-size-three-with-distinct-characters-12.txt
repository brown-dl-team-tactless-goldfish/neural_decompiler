int countGoodSubstrings(char * s){
    int len = strlen(s);
    if(len < 3) return 0;
    int ans = 0;
    int bitmask = 0, j;
    for(int i = 0; i < len-2; i++){
        bitmask = 0;
        for(j = i; j < (i+3); j++){
            if(bitmask & (1<<(s[j]-'a')))
                break;
            bitmask |= (1<<(s[j]-'a'));
        }
        if(j==i+3)
            ans++;
    }
    return ans;
}
