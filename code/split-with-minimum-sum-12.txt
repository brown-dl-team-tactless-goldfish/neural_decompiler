public class Solution {
    public int SplitNum(int num) {
        int[] arr = num.ToString().ToCharArray()
            .Select(x => x - '0')
            .ToArray();
        Array.Sort(arr);
        int num1 = 0, num2 = 0;
        
        for(int i = 0; i < arr.Length; i++)
        {
            if(i % 2 == 0)
                num1 = num1 * 10 + arr[i];
            else
                num2 = num2 * 10 + arr[i];
        }
        
        return num1 + num2;
    }
}
