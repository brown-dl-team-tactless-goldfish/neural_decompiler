int minDeletionSize(char** A, int ASize) {
    int i;
    int len = strlen(A[0]);
    int num=0;
    for(i=0;i<len;i++){
        char c ='a';
        int j;
        for(j=0;j<ASize;j++){
            if(A[j][i]<c){
                num++;
                break;
            }else{
                c = A[j][i];
            }
        }
        
    }
    return num;
}
