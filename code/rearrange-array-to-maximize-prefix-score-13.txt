class Solution {
public:
    int maxScore(vector<int>& nums) {
        sort(nums.begin(),nums.end(),[](int a,int b){
           return a>b; 
        });
        long long ans=0,cnt=0;
        for(int i=0;i<nums.size();i++){
            ans += nums[i];
            if(ans > 0) cnt+=1;
        }
        return cnt;
    }
};
