public class Solution {
    public int MaximumValue(string[] strs) {
        var max = 0;
        foreach(var s in strs){
            max = Math.Max(max, GetLength(s));
        }
        return max;
    }

    private int GetLength(string w){
        if(w.Any(c=>c <'0' || c > '9')){
            return w.Length;
        }
        return int.Parse(w);
    }
}
