bool solve(int *arr, int arrSize, int start, bool* sw){
    if (start >= 0 && start < arrSize && arr[start] < arrSize && sw) {
        int jump = arr[start];
        arr[start] += arrSize;
        return jump == 0 || solve(arr, arrSize, start + jump, sw) || solve(arr,arrSize, start - jump, sw);
    }
    return false;
}

bool canReach(int* arr, int arrSize, int start){
     bool sw = false;
    
    return solve(arr, arrSize, start, &sw);
}
