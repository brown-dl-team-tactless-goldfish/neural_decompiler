class Solution {
public:
    bool checkSubarraySum(vector<int>& nums, int k) {
        
        unordered_map<int, int>hm;
        hm[0] = -1;
        
        int sum = 0;   
        for (int i = 0; i < nums.size(); i++) {
            sum += nums[i];
            if (k != 0) sum %= k;
            if (hm.find(sum) != hm.end()) {
                if (i - hm[sum] > 1) return true;
            } else {
                hm[sum] = i;
            }
        }
        
        return false;
    }
};
