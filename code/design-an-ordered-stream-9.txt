class OrderedStream {
public:
    vector<string> stream ;
    int prev = 0 ;
    OrderedStream(int n) {
        stream.resize(n,"") ;
    }
    
    vector<string> insert(int idKey, string value) {
        vector<string> ans ; stream[idKey - 1] = value ;
        for(int i = prev ; i < stream.size() ; ++i){
            if(stream[i].empty()) break ;
            ans.push_back(stream[i]) ;
            prev = i + 1 ;
        }
        
        return ans ;
    }
} ;
