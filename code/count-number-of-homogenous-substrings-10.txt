class Solution {
public:
    int countHomogenous(string s) {
        int n = s.size(), idx = 0;
        int res = 0, temp = 0, val = 1e9 + 7;
        while(idx < n) {
            res = (res%val + (++temp)%val)%val;
            char c = s[idx++];
            while(idx < n and s[idx] == c) {
                res = (res%val + (++temp)%val)%val;
                idx++;
            }
            temp = 0;
        }
        return res;
    }
};
