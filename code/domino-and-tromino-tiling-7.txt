int numTilings(int n){    
    long F[n+1], P[n+1];
    int mod = pow(10,9) + 7;
    
    if(n<=1) return 1;
    
    for(int i=0; i<=2; i++)
    {
        F[i] = P[i] = i;    /*F[0]=0, F[1]=1, F[2]=2;
                              P[0]=0, P[1]=1, P[2]=2;*/
    }
    
    for(int i=3;i<=n;i++)
    {
        F[i] = (F[i-1] + F[i-2] + 2*P[i-2]) % mod;
        P[i] = (P[i-1] + F[i-1]) % mod;
    }
    return F[n];
}
