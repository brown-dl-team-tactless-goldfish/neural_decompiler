#define FIVE_DOLLARS 5
#define TEN_DOLLARS 10
#define TWENTY_DOLLARS 20

bool valid_billsSize(int);

bool lemonadeChange(int* bills, int billsSize){
    int i;
    int five_bills = 0;
    int ten_bills = 0;
    
    int current_bill;
    
    if(!valid_billsSize(billsSize) || !bills) return false;
    
    for (i = 0 ; i < billsSize; i++)
    {
        current_bill = *(bills + i);
        if (current_bill == FIVE_DOLLARS)
        {
            five_bills += 1;
        }
        else if (current_bill == TEN_DOLLARS)
        {
            if (five_bills) 
            {
                five_bills -= 1;
                ten_bills += 1;
            }
            else
                return false;
        }
        else if (current_bill == TWENTY_DOLLARS)
        {
            if (ten_bills && five_bills)
            {
                ten_bills -= 1;
                five_bills -= 1;
            }
            else if (five_bills >= 3)
            {
                five_bills -= 3;
            }
            else
                return false;
        }
        else
        {
            return false;
        }
        
    }
    
    return true;
}

bool valid_billsSize(int billsSize)
{
    return billsSize>0;
}
