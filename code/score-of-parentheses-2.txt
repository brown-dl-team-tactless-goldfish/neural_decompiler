int scoreOfParentheses(char * s){
    int result = 0, count = 0;
    for(int i = 0; i < strlen(s); i++)
    {
        if(s[i] == '(')
            count++;
        else
        {
            count--;
            if(s[i - 1] == '(')
                result += 1 << count;
        }
    }
    return result;
}
