class Solution {
public:
    int maxDepth(Node* root) {
        if(root == NULL) return 0;
        int max_h = 0;
        if(root->children.size() == 0) return 1;
        for(auto c : root->children){
            int ns = maxDepth(c) + 1;
            max_h = max(max_h,ns);
        }
        return max_h;
    }
};
