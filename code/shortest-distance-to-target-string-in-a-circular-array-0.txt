#define MIN(a,b) (a<b?a:b)
#define MINIMAL(a,b,c) MIN(MIN(a,b),c)

int closetTarget(char ** words, int wordsSize, char * target, int startIndex){
    int len = strlen(target), ans = INT_MAX;
    for(int i = 0; i < wordsSize; i++){
        if(!strncmp(words[i], target, len)){
            ans = MINIMAL(abs(wordsSize+i-startIndex)%wordsSize,abs(wordsSize+startIndex-i)%wordsSize,ans);
        }
    }
    return (ans == INT_MAX)? -1 : ans;
}
