int minimumSize(int* nums, int numsSize, int max){
    int big = 1e9, small = 1, mid, count;
    while(small < big){
        mid = (big + small)/2;
        count = 0;
        for(int i = 0; i < numsSize; i++){
            count += (nums[i] - 1)/ mid;
        }
        if(count > max){
            small = mid + 1;
        }
        else{
            big  = mid;
        }
    }
    return small;

}
