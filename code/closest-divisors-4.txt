int* closestDivisors(int num, int* returnSize){
    int* output = malloc(sizeof(int) * 2);
    *returnSize = 2;
    for(int i = sqrt(num + 2); i > 0; i--){
        if((num + 1) % i == 0){
            output[0] = i;
            output[1] = (num + 1) / i;
            return output;
        }
        if((num + 2) % i == 0){
            output[0] = i;
            output[1] = (num + 2) / i;
            return output;
        }
    }
    return output;
}
