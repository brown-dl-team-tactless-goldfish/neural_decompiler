bool uniqueOccurrences(int* arr, int arrSize){
    int checkNum[1000] = {0};
    int count[2001] = {0};
    for(int i = 0; i < arrSize; i++)
    {
        count[arr[i] + 1000]++;
    }
    for(int i = 0; i <= 2000; i++)
    {
        if(count[i] != 0)
        {
            if(checkNum[count[i]] == 0)
                checkNum[count[i]] = 1;
            else
                return false;
        }
    }
    return true;
}
