void duplicateZeros(int* arr, int arrSize){

    int i=0;
    while (i < arrSize)
    {
        if (arr[i] == 0) 
        {
            for (int j=arrSize-1; j>=i+2; j--)
                arr[j] = arr[j-1];
            arr[i+1] = 0;
            i += 2;
            if (i == arrSize-1) break;
        }
        else i++;
        if (i == arrSize-1) break;
     }
}
