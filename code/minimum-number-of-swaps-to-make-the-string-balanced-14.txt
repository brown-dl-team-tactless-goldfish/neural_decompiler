int minSwaps(string s) {
    int o = 0, c = 0, res = 0;
    int n = s.size(), j = n - 1;
    for(int i = 0; i < n; i++) {
        if(s[i] == '[') o++;
        else c++;
        if(c > o) {
            while(j >= 0 && s[j] != '[') j--;
            swap(s[i],s[j]);
            j--;
            c--;
            o++;
            
            res++;
        }
    }
    return res;
}
