class Solution {
public:
    vector<int> getAverages(vector<int>& nums, int k) {
    vector<int> ans(nums.size(),-1);
    int n=nums.size(),k2=2*k;
    long long sum=0;
    for(int i=0;i<nums.size();i++)
    {
        if(i<k2) sum+=nums[i];
        else 
        {
            sum+=nums[i];
            ans[i-k]=sum/(k2+1);
            sum-=nums[i-k2];
        }
    }
    return ans;
    }
};
