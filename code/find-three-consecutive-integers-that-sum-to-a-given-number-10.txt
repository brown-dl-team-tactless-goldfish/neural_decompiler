vector<long long> sumOfThree(long long num) {
        long long a = 0,b = 0,c = 0;
        b = (num / 3);
        a = b - 1;
        c = b + 1;
        vector<long long> res;
        if(a + b + c == num) {
            res.push_back(a);
            res.push_back(b);
            res.push_back(c);
        }
        return res;   
    }
