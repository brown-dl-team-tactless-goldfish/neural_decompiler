bool checkRecord(char* s) {
    int countA=0,countL=0;
    while(*s!='\0')
    {
        if(*s=='A')
        {
            countA++;
            countL=0;
            if(countA>1)
            {return false;}
        }else if(*s=='L')
        {
            countL++;
            if(countL>2)
            {return false;}
        }else
        {
          countL=0;  
        }  
        s++;
    }
    return true;
}
