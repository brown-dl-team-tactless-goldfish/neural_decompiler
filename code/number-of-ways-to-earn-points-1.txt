    def waysToReachTarget(self, target, types):
        dp = [1] + [0] * target
        mod = 10 ** 9 + 7
        for c,m in types:
            for i in range(target,-1,-1):
                for k in range(1, min(c, i // m) + 1):
                    dp[i] = (dp[i] + dp[i - m * k]) % mod
        return dp[-1]
