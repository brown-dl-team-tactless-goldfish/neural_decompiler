    bool checkAlmostEquivalent(string word1, string word2) {
        int m[26] = {0};
        for(char& c : word1) m[c-'a']++;
        for(char& c : word2) m[c-'a']--;     
        for(int i = 0; i < 26 ; i++ ){
          if( (m[i] >=0 ? m[i] : -m[i]) > 3) return false;
        }
        return true;        
    }
