int maxProductDifference(int* nums, int numsSize){
int larger, largest, smaller, smallest;
    int i, j;
    larger = largest = 0;
    smaller = smallest = 10001;
    
    for(i=0; i<numsSize; i++) {
        j = nums[i];
        
        if (j > largest) {
            larger = largest;
            largest = j;
        } else if (j > larger) {
            larger = j;
        }
        
        if (j < smallest) {
            smaller = smallest;
            smallest = j;
        } else if (j < smaller) {
            smaller = j;
        }
        
        
    }
        int z=(larger * largest) - (smaller * smallest);
    return z;
}
