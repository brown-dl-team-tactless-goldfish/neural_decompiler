class Solution {
public:
    vector<int> intersection(vector<vector<int>>& nums) {
        map<int, int> counts;
        for(int i = 0; i < nums.size(); i++){
            for(int j = 0; j < nums[i].size(); j++)
                counts[nums[i][j]]++;
        }
        vector<int> sol = {};
        for(auto it = counts.begin(); it != counts.end(); it++)
            if(it->second == nums.size()) sol.push_back(it->first);
        
        return sol;
    }
};
