int minAddToMakeValid(char* S) {
    int len=strlen(S);
    int count=0;
    int sum=0;
    for(int i=0;i<len;i++){
        if(S[i]=='('){
            if(count<0){
                sum+=-count;
                count=0;
            }
            count++;
        }else{
            count--;
        }
    }
    return sum+(count>0?count:-count);
}
