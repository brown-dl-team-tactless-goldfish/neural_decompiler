class Solution {
public:
    int divisorSubstrings(int num, int k) {
        string s = to_string(num) ;
        int ans = 0 ;
        for(int i = 0 ; i <= s.size() - k ; ++i ){
            string temp = s.substr(i,k) ;
            int divisor = stoi(temp) ;
            if(divisor > 0 and num % divisor == 0) ++ans ;
        }
        return ans ;
    }
};
