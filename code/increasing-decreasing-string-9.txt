char * sortString(char * s){
    char *r = malloc(1000), *p = r, *t = NULL, m[128] = { 0 };    
    while (*s && ++m[*s++]);
    while (p != t && (t = p)) {
        for (int i = 'a' ; i <= 'z' ; m[i++]-- > 0 ? *p++ = i - 1 : 0);
        for (int i = 'z' ; i >= 'a' ; m[i--]-- > 0 ? *p++ = i + 1 : 0);
    }
    return *p = 0, r;
}
