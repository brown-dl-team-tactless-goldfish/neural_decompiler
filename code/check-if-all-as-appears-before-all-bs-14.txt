char checkNext;
int len = strlen(s), i;
len = len - 2;
for (i = 0; s[i] != '\0'; i++)
{
    if (strlen(s) == 1)     // if array contain only one element then it will always be ordered
            return true;
    if (i <= len)
        checkNext = s[i+1];   // checkNext keeps track of next char and if its second last element of array becoz s[i+1] will give last element.
    if (checkNext < s[i])
        return false;
}
return true;
