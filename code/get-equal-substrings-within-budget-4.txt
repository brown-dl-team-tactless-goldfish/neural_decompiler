int equalSubstring(string s, string t, int mx) {
    int n = s.size();
    int i = 0,j = 0,x = 0,c = 0;
    while(j<n){
        x+=abs(s[j]-t[j]);
        if(x<=mx){
            c = max(c,j-i+1);
            j++;
        }
        else if(x>mx){
            while(x>mx){
                x-=abs(s[i]-t[i]);
                i++;
            }
            j++;
        }
    }
    return c;
}
