bool isMonotonic(int* A, int ASize) {
    int i,j;
    j=i=0;
    while(i<ASize-1){
        if (A[i]<=A[i+1])
            i++;
        else
            break;
    }
    if(i==ASize-1)
        return true;
    while(j<ASize-1){
        if(A[j]>=A[j+1])
            j++;
        else
            break;
    }
    if(j==ASize-1)
        return true;
    return false;
    
    
}
