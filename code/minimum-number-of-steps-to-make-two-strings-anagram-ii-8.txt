class Solution {
public:
    int minSteps(string s, string t) {
       int n = s.size();
       int m = t.size();
       unordered_map<char, int> mp;
       int count = 0;
       for(char c : s){
               mp[c]++;
       }
       for(char c : t){
               if(mp.find(c) != mp.end()){
                  count++; 
                  mp[c]--;
               }
               if(mp[c] == 0) mp.erase(c);
       }
        return n + m - 2*count;
    }
};
