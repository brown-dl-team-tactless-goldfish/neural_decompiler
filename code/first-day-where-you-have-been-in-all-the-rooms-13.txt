class Solution {
public:
    int firstDayBeenInAllRooms(vector<int>& nextVisit)
    {
        int n=nextVisit.size();
        vector<long long>dp(n,0);
        long long mod=1e9+7;
        for (int i=1;i<n;i++)
        {
            if(nextVisit[i-1]==i-1)
            dp[i]=(dp[i-1]+2+mod)%mod;
            else{
                dp[i]=(dp[i-1]+2+dp[i-1]-dp[nextVisit[i-1]]+mod)%mod;
            }
        }
        return dp[n-1];
    }
};
