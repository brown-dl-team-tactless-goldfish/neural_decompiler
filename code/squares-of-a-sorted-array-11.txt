int* sortedSquares(int* nums, int numsSize, int* returnSize){
    int *value = (int*)malloc(numsSize*sizeof(int)), s=0, e=numsSize-1, p=numsSize-1;
    while(s<=e&&p>=0){
        if((nums[s]>=0?nums[s]:-nums[s]) >= (nums[e]>=0?nums[e]:-nums[e]))value[p--] = nums[s]*nums[s++];
        else value[p--] = nums[e]*nums[e--];
    }
    *returnSize = numsSize;
    return value;
}
