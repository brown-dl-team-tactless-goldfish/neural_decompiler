int* sumZero(int n, int* returnSize)
{
int *nums=malloc(sizeof(int)*n) , j=1;
*returnSize=n;
if(n%2==0)
{
for(int i=0 ; i<=n-1 ; i+=2)
    {
	   nums[i]=j;
       nums[i+1]=-j;
       j++;
    }
}
if(n%2!=0)
{
 nums[0]=0;
for(int i=1 ; i<=n-2 ; i=i+2)
    {
	  nums[i]=j;
      nums[i+1]=-j;
      j++;
	}
}
return nums;
}
