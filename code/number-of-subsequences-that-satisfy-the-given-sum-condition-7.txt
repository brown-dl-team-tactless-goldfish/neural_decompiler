    int n = nums.size(), left = 0, right = n-1, mod = 1e9+7,ans = 0;
    
    vector<long> pow(n,1);
    
    for(int i=1; i<n; i++)
    {
        pow[i] = pow[i-1]*2 % mod;
    }
    
    while(left <= right)
    {
        if(nums[left] + nums[right] > target)
        {
            right--;
        }
        else
        {
            ans = (ans +pow[right-left])% mod;
            
            left++;
            
        }
    }
    
    return ans%mod;
    
}
