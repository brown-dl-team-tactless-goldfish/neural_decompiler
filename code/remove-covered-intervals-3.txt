class Solution:
    def removeCoveredIntervals(self, intervals: List[List[int]]) -> int:
        intervals.sort()
        # print ("after sort, intervals = ", intervals)
        
        n = len(intervals)
        if (n < 2):
            return n
        
        self.res = [intervals[0]]
        for idx in range(1, n):
            candidate = intervals[idx]
            last_in_res = self.res[-1]
            if (last_in_res[0] <= candidate[0] <= candidate[1] <= last_in_res[1]):
                continue
            else:
                self.res.append(candidate)
                
        # print ("end self.res = ", self.res)
        return len(self.res)
