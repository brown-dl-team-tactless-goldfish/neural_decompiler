class Solution {
public:
    unordered_map<long long, unordered_map<long long, long long> > dp;
    long long M = 1e9 + 7;
    long long rearrangeSticks(long long n, long long k) {
        if(k == 0) return 0;
        if(n == k) return 1;
        if(dp[n].count(k)) return dp[n][k];
        return dp[n][k] = ((n - 1) * rearrangeSticks(n - 1, k) + rearrangeSticks(n - 1, k - 1)) % M;
    }
};
