char * interpret(char * command){
    char* a=malloc(sizeof(char)*(strlen(command)+1));
    int n=0;
for(int i=0;i<strlen(command);i++){
    if(command[i]=='G'){
        a[n++]='G';
    }
    else if(command[i]=='('){
        if(command[i+1]==')'){
        a[n++]='o';
        i++;
        }
        else{
        a[n++]='a';
        a[n++]='l';
        i+=2;
        }   
    }
}
a[n]='\0';
return a;
}
