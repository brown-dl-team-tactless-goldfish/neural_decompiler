class Solution {
public:
    int partitionArray(vector<int>& nums, int k) {
        sort(nums.begin(), nums.end());
        int ans = 0, t = nums[0] + k;
        auto p = nums.begin();
        while (p != nums.end())
        {
            auto c = upper_bound(p, nums.end(), t);
            if (c != nums.end())
                t = *c + k;
            ans++;
            p = c;
        }
        return ans;
    }
};
