

char * getSmallestString(int n, int k){

    char* s = (char*) malloc (sizeof(char)*(n+1));
    s[n] = '\0';
    memset (s, 'a', n);

    int i = strlen(s) -1;
    k = k -n;
    while (i >= 0) {
        if (k == 0)
            break;
        if (k >= 25){
            s[i--] = 'z';
            k -= 25;
            continue;
        }
        if (k < 25) {
            s[i--] = (char)('a'+ k);
            k -= k;
        }
    }
    return s;
}
