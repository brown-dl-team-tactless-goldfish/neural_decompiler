class Solution {
public:
    string kthDistinct(vector<string>& arr, int k) {
        unordered_map<string,int>mp;
        int c = 0;
        for(auto a:arr)
        {
            mp[a]++;
        }
        for (int i=0;i<arr.size();i++)
        {
            if(mp[arr[i]]==1)
            {
                c++;
            }
                if(c==k)
                    return arr[i];
        }
        return "";
        
       
    }
};
