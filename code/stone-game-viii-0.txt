int max(int a, int b){
    return a > b ? a : b;
}

int stoneGameVIII(int* stones, int stonesSize){
    int list[100000];
    int maxpoint[100000];
    list[0] = stones[0];
    for (int i = 1 ; i < stonesSize ; i++){
        list[i] = stones[i]+list[i-1];
    }
    maxpoint[stonesSize-2] = list[stonesSize-1]; 
    for (int i = stonesSize-3 ; i >= 0 ; i--){
        maxpoint[i] = max(list[i+1]-maxpoint[i+1], maxpoint[i+1]);
    }
    return maxpoint[0];
}
