class Solution {
public:
    vector<int> leftRigthDifference(vector<int>& nums) {
        int n = size(nums);
        vector<int> left(n),right(n),ans(n);
        for(int i=0; i<n-1; ++i)
            left[i+1] = left[i] + nums[i];
        for(int i=n-1; i>0; --i)
            right[i-1] = right[i] + nums[i];
        for(int i=0; i<n; ++i)
            ans[i] = abs(left[i] - right[i]);
        return ans;
    }
};
