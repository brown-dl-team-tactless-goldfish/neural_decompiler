bool areOccurrencesEqual(char * s){
    char c, letters[26] = {0};
    int i, j, found=0;
    
    while((c = *s++)) {
        letters[c - 97]++;
    }
    
    for(i=0; i<26; i++) {
        
        j = letters[i];
        
        if (!j) continue;
        
        if (!found) found = j;
        
        if (found != j) {
            return false;
        }
    }
    
    return true;
}
