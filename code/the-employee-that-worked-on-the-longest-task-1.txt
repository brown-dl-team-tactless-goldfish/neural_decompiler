int hardestWorker(int n, int** logs, int logsSize, int* logsColSize){
    int time = 0, retId = n+1, maxTask = INT_MIN, currTask = 0;
    for(int i = 0; i < logsSize; i++){
        currTask = logs[i][1] - time;
        time += currTask;
        if(currTask == maxTask && logs[i][0]<retId)
            retId = logs[i][0];        
        else if(currTask > maxTask){
            maxTask = currTask;
            retId = logs[i][0];
        }
    }
    return retId;
}
