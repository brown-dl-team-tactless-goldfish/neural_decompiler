WITH CTE AS (SELECT DISTINCT VISIT_ID FROM TRANSACTIONS)

SELECT 
    CUSTOMER_ID, 
    COUNT(CUSTOMER_ID) AS COUNT_NO_TRANS 
FROM 
    VISITS V
LEFT JOIN 
    CTE C 
ON 
    V.VISIT_ID = C.VISIT_ID       
WHERE 
    C.VISIT_ID IS NULL    
GROUP BY 
    CUSTOMER_ID;
