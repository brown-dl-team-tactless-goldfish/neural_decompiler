class Solution {
   public:
      string getSmallestString(int n, int k) {
         string s;
         s.reserve(n);
         for (int i = 0; i < n; ++i) {
            s.push_back('a');
         }
         k -= n;
         int pos = n - 1;
         while (k > 0) {
            int c = min(k + 1, 26);
            s[pos--] = static_cast<char>(c + 'a' - 1);
            k -= c - 1;
         }
         return s;
      }
};
