char *minRemoveToMakeValid(char *s) {
    int length = strlen(s);

    int cnt1 = 0;
    int cnt2 = 0;
    for(int i=0;i<length;i++) {
        if(s[i] == '(') {
            cnt1++;
        }
        else if(s[i] == ')') {
            if(cnt1 == 0) {
                s[i] = '.';
            }
            else {
                cnt1--; // stack pop
            }
        }

        if(s[length-1-i] == ')') {
            cnt2++;
        }
        else if(s[length-1-i] == '(') {
            if(cnt2 == 0) {
                s[length-1-i] = '.';
            }
            else {
                cnt2--;
            }
        }
    }

    // printf("=> %s\n", s); 
    // remove '.'
    int i_temp = 0;
    for(int i=0;i<length;i++) {
        if(s[i] != '.') {
            s[i_temp++] = s[i];
        }
    }
    if(i_temp < length) s[i_temp] = '\0';

    return s;
}
