int countDigit(int digit){
    int i=0;
    int count=0;
    while(digit>0){
        count++;
        digit=digit/10;        
    }
    return count;
}

int findNumbers(int* nums, int numsSize){
    int sum=0;
    int i=0;
    while(i<numsSize){
        int temp = countDigit(nums[i]);
        if(temp%2==0){
            sum++;
        } 
        i++;
    }
    return sum;   
}
