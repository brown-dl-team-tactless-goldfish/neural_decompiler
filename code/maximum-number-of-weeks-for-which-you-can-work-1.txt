int cmp(const void *a,const void *b){
    return *(int*)a-*(int*)b;
}
long long numberOfWeeks(int* milestones, int milestonesSize){
    qsort(milestones,milestonesSize,sizeof(int),cmp);
    long long other = 0;
    for(int i=milestonesSize-2;i>=0;i--){
        other+=milestones[i];
    }
    if(other>=milestones[milestonesSize-1])
        return other+milestones[milestonesSize-1];
    else
        return 2*other+1;
}
