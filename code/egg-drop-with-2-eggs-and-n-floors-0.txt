int twoEggDrop(int n){
    if(n == 1)
        return 1;
    if(n == 2)
        return 2;

    int* dp = malloc((n+1) * sizeof(int));
    dp[1] = 1;
    dp[2] = 2;
    for(int i = 3; i <= n; i++){
        dp[i] = INT_MAX;
        for(int j = 1; j <= i; j++){
            dp[i] = fmin(dp[i], 1 + fmax(j-1, dp[i-j]));
        }
    }
    int ans = dp[n];
    free(dp);
    return ans;
}
