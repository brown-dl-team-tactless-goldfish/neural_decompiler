class Solution {
public:
    int monkeyMove(int n) {
        long long int mod= 1e9+7, ans =1;
        
        // each monkey has 2 choices : either move 
        // 1. CLOCKWISE
        // 2. ANTI-CLOCKWISE
        // so we have total of 2^n combinations of movements of monkeys
        // out of these 2^n movement combinations, we only have 2 special combination movements
        // which leads to no collision ( i.e., when all the monkeys simultaneously move:
        // 1. either CLOCKWISE
        // 2. or ANTI-CLOCKWISE)
        
        // so total combinations of movements = (2^n) -2;
        for(int p=60; p>=1; p--){
            while( n>= p){
                ans = ans%mod * ((long long)pow(2,p)%mod);
                n-=p;
            }
        }
        
        return (ans-2)%mod;
    }
};
