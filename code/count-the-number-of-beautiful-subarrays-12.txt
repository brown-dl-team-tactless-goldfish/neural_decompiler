class Solution {
public:
    long long beautifulSubarrays(vector<int>& nums) {
        long long ans = 0;
        unordered_map<int, int> h;
        for (int i = 0; i < nums.size(); i++) {
            if (i > 0)
                nums[i] ^= nums[i - 1];
            ans += (nums[i] == 0) + h[nums[i]];
            h[nums[i]]++;
        }
        return ans;
    }
};
