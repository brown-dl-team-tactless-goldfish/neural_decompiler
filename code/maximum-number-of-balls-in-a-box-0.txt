int countBalls(int lowLimit, int highLimit){
    int* box = calloc(50, sizeof(int)) ;
    int max = 0 ; 
    for(int i = lowLimit ; i <= highLimit ; i++){
        int val = i ;
        int d =  0 ;
        while(val){
            d += val % 10 ;
            val /= 10 ;
        }
        box[d] += 1;
        max = fmax(max , box[d]) ;
    }
    return max ;
}
