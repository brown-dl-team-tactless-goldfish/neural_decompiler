class Solution:
    def maximumDifference(self, nums: List[int]) -> int:
        n = len(nums)
        self.res = float('-inf')
        
        curMin = nums[0]
        
        for idx in range(1, n):
            curNum = nums[idx]
            print ("idx = ", idx, " curNum = ", curNum)
            
            if (curNum < curMin):
                curMin = curNum
            else:
                self.res = max(self.res, curNum - curMin)
        
        if (self.res == float('-inf') or self.res == 0):
            return -1
        else:
            return self.res
