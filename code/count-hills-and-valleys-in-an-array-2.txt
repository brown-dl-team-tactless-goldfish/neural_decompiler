int countHillValley(int* nums, int numsSize){
    int ansCnt = 0, ansIdx = 0;
    for(int i = 1; i < numsSize; i++){
        if(nums[i] > nums[i-1])
            nums[ansIdx++] = 1;
        else if(nums[i] < nums[i-1])
            nums[ansIdx++] = 0;
    }
    for(int i = 0; i < ansIdx - 1; i++)
        ansCnt += (nums[i]^nums[i+1]);
    return ansCnt;
}
