class Solution {
public:
    bool areAlmostEqual(string s1, string s2)
    {
        if(s1.size()!=s2.size())return false;
        int k=0,index[2];
        for(int i =0;i<s1.size();i++)
        {
            if(s1[i]!=s2[i])
            {
                if(k+1==3)return false;
                index[k++]=i;
            }
        }
        if(k==2&&s1[index[0]]==s2[index[1]]&&s1[index[1]]==s2[index[0]]||k==0)return true;
        return false;
    }
};
