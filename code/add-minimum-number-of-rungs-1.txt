int max(int a, int b)
{
    if(a > b) return a;
    return b;
}

int addRungs(int* rungs, int rungsSize, int dist){
    int ans = max(0, (rungs[0]-1)/dist);  //for the first element
    for(int i=1; i<rungsSize; i++)
        ans += max(0, (rungs[i]-rungs[i-1]-1)/dist); //for the remaining elements
    return ans;
}
