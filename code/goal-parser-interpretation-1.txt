#include<string.h>
char * interpret(char * command){
  char c;
  int N=strlen(command);
  int  i=0,f=0;
  for(f=0;f<N;f++)
  {
    c = command[f];
    if(c == '(')
    {
      if(command[f+1] == ')')
      {
        command[i] = 'o';
        f=f+1;
      }
      else
      {
        command[i]='a';
        command[++i]='l';
        f+=3;
      }
    }
    else
    {
       command[i]=c;
    }
    i++;
  }
  command[i]='\0';
  return command;
}
