int* findSmallestSetOfVertices(int n, int** edges, int edgesSize, int* edgesColSize, int* returnSize){
    int * temp = (int *)calloc(n, sizeof(int));
    int * result = (int *)malloc(n * sizeof(int));
    int idx = 0;
    for(int i = 0; i < edgesSize; i++)
        temp[edges[i][0]] = 1;
    for(int i = 0; i < edgesSize; i++)
        temp[edges[i][1]] = 0;
    for(int i = 0; i < n; i++)
        if(temp[i] == 1)
            result[idx++] = i;
    *returnSize = idx;
    return result;
}
