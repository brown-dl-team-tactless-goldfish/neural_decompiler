class Solution {
public:
    long long numberOfWeeks(vector<int>& m) {
        long long mx = INT_MIN;
        long long total = 0;
        
        for(auto &x : m){
            total += x;
            if(mx<x)
                mx=x;
        }
        
        long long sm = total - mx;
        long long ans = 0;
        
        if((sm + 1) < mx) {
            ans = (2 * sm) + 1;
        }
        else {
            ans = total;
        }
        
        return ans;
    }
};
