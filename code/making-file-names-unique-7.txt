class Solution {
public:
    using VS = vector<string>;
    using Set = unordered_set<string>;
    VS getFolderNames(VS& A, Set seen = {}, VS ans = {}) {
        for (auto& s: A) {
            auto cand{ s };
            auto cnt{ 0 };
            while (seen.find(cand) != seen.end())
                cand = s + "(" + to_string(++cnt) + ")";
            ans.push_back(cand);
            seen.insert(cand);
        }
        return ans;
    }
};
