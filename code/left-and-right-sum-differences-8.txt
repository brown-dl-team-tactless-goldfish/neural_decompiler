class Solution {
public:
    vector<int> leftRigthDifference(vector<int>& nums) {
        int lefts=0;
        int rsum=accumulate(nums.begin(),nums.end(),0);
        vector<int> ans;
        for(int i=0;i<nums.size();i++){
            int temp=abs(lefts-(rsum-nums[i]-lefts));
            ans.push_back(temp);
            lefts+=nums[i];
        }
        return ans;
    }
};
