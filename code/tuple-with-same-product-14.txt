class Solution {
public:
    int tupleSameProduct(vector<int>& nums) {
        
        unordered_map<int,int> freq;
        for(int i:nums) 
        for(int j:nums)
        freq[i*j]++;
        
        int ans=0;
        
        for(auto [num, f] : freq){
           if(f>1)
           {
              if(f&1) f--;
               ans=ans + f*(f-2);
           }    
           
        }
        return ans;
    }
};
