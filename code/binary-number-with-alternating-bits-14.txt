//693. Binary Number with Alternating Bits
//4ms

bool hasAlternatingBits(int n){
    int compareBit=1;
    bool tmp,preTmp;
    int i;
    tmp=compareBit&n; //i=0ï¼Œbecause there is no preTmp
    preTmp=tmp;
    n>>=1;
    for(i=1;n>0;i++){ //â˜… use n>0 to judge whether need to continue â˜…
        tmp=compareBit&n;
        if(tmp==preTmp)return false;
        preTmp=tmp;
        n>>=1;
    }
    return true;
}
