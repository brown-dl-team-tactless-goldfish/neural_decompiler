
double myPow(double x, int n){
    if(n==0)
        return 1;
    else if(n==1)
        return x;
    else if(n==-1)
        return 1/x;
    else if(n%2==0)
        return myPow(x,n/2)*myPow(x,n/2);
    else if(n<0)
        return myPow(x,n/2)*myPow(x,n/2)*1/x;
    else
        return myPow(x,n/2)*myPow(x,n/2)*x;
}
