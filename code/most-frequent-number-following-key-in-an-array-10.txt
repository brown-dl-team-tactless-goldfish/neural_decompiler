class Solution {
public:
    int mostFrequent(vector<int>& nums, int key) {
        unordered_map<int, int> f;
        int n = nums.size();
        for (int i = 0; i < n - 1; i++)
            if (nums[i] == key)
                f[nums[i + 1]]++;
        int mf = 0, ans = -1;
        for (auto& p : f)
            if (p.second > mf)
            {
                mf = p.second;
                ans = p.first;
            }
        return ans;
    }
};
