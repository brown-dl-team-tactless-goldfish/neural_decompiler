class Solution {
public:
    int countTriplets(vector<int>& A) {
        unordered_map<int,int>f;
        int n=A.size();
        for(int i=0;i<n;i++){
            for(int j=0;j<n;j++){
                f[A[i]&A[j]]++;
            }
        }
        int ans=0;
        for(auto v:f){
            for(int i=0;i<n;i++){
                if((v.first&A[i])==0){
                    ans+=v.second;
                }
            }
        }
        return ans;
    }
};
