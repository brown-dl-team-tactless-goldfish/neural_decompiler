int mostWordsFound(char ** sentences, int sentencesSize){
    int spaces_max =0;
    int spaces =0;
    char *c;
    
    for (int i=0; i<sentencesSize; i++) {
        c = sentences[i];
        spaces =0;
        while (*c++) {
            if (*c == ' ') {
                spaces++;
            }
            
        }
        
        if (spaces > spaces_max) {
            spaces_max = spaces;
        }
    }
    
    return spaces_max + 1;
}
