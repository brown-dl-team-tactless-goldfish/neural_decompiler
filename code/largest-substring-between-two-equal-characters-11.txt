int maxLengthBetweenEqualCharacters(char * s){
    
    int charsMap[256];
    memset(charsMap,-1,sizeof(int)*256);
    int max = -1;
    for (int i = 0; s[i]; i++)
    {
        if (charsMap[s[i]] == -1)
        {
            charsMap[s[i]] = i;
            continue;
        }
        
        int newLen = i - charsMap[s[i]] - 1;
        
        if (newLen > max)
            max = newLen;
        
    }
    
    return max;
}
