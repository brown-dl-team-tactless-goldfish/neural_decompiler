

int minDeletionSize(char ** strs, int strsSize){
    int i,j=0,count=0;
    int len=strlen(strs[0]);
    
    for(i=0;i<len;i++){
        char c ='a';
        
        for(j=0;j<strsSize;j++){
            if(strs[j][i]<c){
                count++;
                break;
            }else{
                c = strs[j][i];
            }
        }
        
    }
    return count;
}
