    public int NumberOfRounds(string startTime, string finishTime) {
        int sh=int.Parse(startTime.Substring(0,2));
        int sm=int.Parse(startTime.Substring(3,2));
        int fh=int.Parse(finishTime.Substring(0,2));
        int fm=int.Parse(finishTime.Substring(3,2));
        int h=sh<=fh?fh-sh:fh+24-sh;
        if(sh==fh&&fm<sm)h=24;
        int m=fm/15-(sm+14)/15;
        return h*4+m;
    }
