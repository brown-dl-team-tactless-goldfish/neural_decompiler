#include <ctype.h>
#include <string.h>

bool isPalindrome(char * s){

    int len = strlen(s);
    int write_idx = 0;
    
    // Remove non-alpha chars and lower alpha chars in-place.
    for (int i = 0; i < len; i++) {
        if (isalnum(s[i]))
            s[write_idx++] = (char)tolower(s[i]);
    }
    s[write_idx] = 0;
    printf("%s\n", s);
    
    char *s_end = s + strlen(s) - 1;
    char *s_middle = (s_end - s) / 2;
    
    // Check to see if palindrome.
    while (s <= s_end) {
        if (*s != *s_end)
            return false;
        s++;
        s_end--;
    }
    return true;
}
