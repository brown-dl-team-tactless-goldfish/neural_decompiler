class Solution {
public:
	bool divideArray(vector<int>& nums) {
		int n = nums.size();
		int c =0;
		map<int,int>m;
		for(int i=0;i<nums.size();i++){
			m[nums[i]]++;
		}
	   for(auto j : m){
		   c+=(j.second/2);
	   }
		  return (c==n/2);
	}
};
