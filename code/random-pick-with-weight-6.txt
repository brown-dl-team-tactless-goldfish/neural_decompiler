class Solution {
public:
    Solution(vector<int>& w) {
        for (int weight : w) {
            if (m_vFreq.empty())
                m_vFreq.push_back(weight);
            else
                m_vFreq.push_back(weight + m_vFreq.back());
        }
    }
    
    int pickIndex() {
        int val = rand() % m_vFreq.back();
        auto resIt = upper_bound(m_vFreq.begin(), m_vFreq.end(), val);
        return resIt - m_vFreq.begin();
        
    }
private:
    vector<int> m_vFreq;
};
