bool checkAlmostEquivalent(char* word1, char* word2) {
    int map[26] = {0}, w1len = strlen(word1), w2len = strlen(word2);
    for (int i = 0; i < w1len; i++) {
        map[word1[i] - 'a']++;
    }
    for (int i = 0; i < w2len; i++) {
        map[word2[i] - 'a']--;
    }
    for (int i = 0; i < 26; i++) {
        if (map[i] > 3 || map[i] < -3) return false;
    }
    return true;
}
