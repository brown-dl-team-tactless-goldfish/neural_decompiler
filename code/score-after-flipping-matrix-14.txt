typedef long long ll;
#define rep(i,s,n)for(ll i = s;i<n;i++)
class Solution {
public:
	int matrixScore(vector<vector<int>>& m) {
		int R = m.size();
		int C = m[0].size();

		int z = pow(2,C-1)*R;

		rep(c, 1, C) {
			int bit = 0;
			rep(r, 0, R) {
				bit += (m[r][c] ^ m[r][0]);
			}
			z += pow(2, C-c-1) * max(bit, R - bit);
		}

		return z;
	}
};
