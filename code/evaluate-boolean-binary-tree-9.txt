class Solution {
public:
    bool evaluateTree(TreeNode* root) {
        
        function<bool(TreeNode*)> fn = [&](TreeNode* node) {
            if (node->val == 0 || node->val == 1) return (bool) node->val; 
            if (node->val == 2) return fn(node->left) || fn(node->right); 
            return fn(node->left) && fn(node->right); 
        }; 
        
        return fn(root); 
    }
};
