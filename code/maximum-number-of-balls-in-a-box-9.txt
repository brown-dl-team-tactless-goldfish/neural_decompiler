int sumDigit(int n){
    int sum = 0;
    while (n)
        {
            /* code */
            sum += n%10;
            n/=10;
        }
    return  sum;
}

    int arr[50] = {0};
    for (int i = lowLimit; i <= highLimit; i++)

        /* code */
        arr[sumDigit(i)]++;
    }
