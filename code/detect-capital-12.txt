bool detectCapitalUse(char * word)
{
    int i;
    int upperCount = 0, lowerCount = 0;

    for(i = 0; i < strlen(word); i++)
    {
        if(islower(word[i]) != 0)
            lowerCount += 1;
        if(isupper(word[i]) != 0)
            upperCount += 1;
    }

    if(upperCount == strlen(word))
        return true;
    
    if(upperCount == 1 && isupper(word[0]) != 0)
        return true;
    
    if(lowerCount == strlen(word))
        return true;
    
    return false;
}
