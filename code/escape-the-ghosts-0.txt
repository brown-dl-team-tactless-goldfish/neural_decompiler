bool escapeGhosts(int** ghosts, int ghostsRowSize, int *ghostsColSizes, int* target, int targetSize) {
    int minDistance=abs(target[0])+abs(target[1]);
    for(int i=0;i<ghostsRowSize;i++){
        if(minDistance>=abs(ghosts[i][0]-target[0])+abs(ghosts[i][1]-target[1])){
            return false;
        }
    }
    return true;
}
