class Solution {
public:
    int minSwaps(string s) {
        int cl = 0;
        int op = 0;
        
        int ans = 0;
        
        for(char ch: s){
            if(ch=='[') op++;
            else cl++;
            if(cl > op) ans = max(ans, cl - op);
        }
            return (ans + 1) / 2;
    }
};
