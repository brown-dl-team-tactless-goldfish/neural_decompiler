int calPoints(char ** ops, int opsSize){
    if(opsSize==1)return atoi(ops[0]);
    int i,count,ret;
    int *a=(int *)calloc(opsSize,sizeof(int)),*b;
    for(i=0,b=a,count=0; i<opsSize; i++){
        if(ops[i][0]=='+')(*b++)=*(b-1)+*(b-2);
        else if(ops[i][0]=='D')(*b++)=(*(b-1))*2;
        else if(ops[i][0]=='C')--b;
        else (*b++)=atoi(ops[i]);
        count+=ops[i][0]=='C'?-1:1;
    }
    for(i=0,ret=0;i<count;ret+=a[i],i++);
    free(a);
    return ret;
}
