class Solution {
public:
    vector<long long> sumOfThree(long long num) {
        long long left = 0;
        long long right = num;
        // left <= target < right
        while (left + 1 < right) {
            long long mid = left + (right - left) / 2;
            auto sum = 3 * mid;
            if (sum > num) {
                right = mid;
            } else {
                left = mid;
            }
        }
        return left * 3 == num ? vector<long long>{left - 1, left, left + 1} : vector<long long>{};
    }
};
