class Solution {
public:
    int countWords(vector<string>& words1, vector<string>& words2) {
        unordered_map<string,int>mp,mp1;
        for(string x:words1){
            mp[x]++;
        }
        for(string x:words2){
            mp1[x]++;
        }
        int cnt=0;
        for(string x:words2){
            if(mp.find(x)!=mp.end()){
                if(mp[x]==1 && mp1[x]==1)cnt++;
            }
        }
        return cnt;
    }
};
