class Solution {
public:
    int numRabbits(vector<int>& answers) {
		// Creates a vector with 1000 elements, as by the problem definition the max rabbit answer is 999.
		// This will keep track of the frequency of answers/
        vector<int> bkts(1000, 0);
		
		// Calculate frequency of each rabbit answer.
        for(auto i: answers) ++bkts[i];
        
        int total = 0;
        for(int i = 0; i < bkts.size(); i++){
            const int maxCycleSize = (i+1);
            const int rabbitCount = bkts[i];
            // This is just the int ceil of (rabbitCount / maxCycleSize).
            const int minRabbitCycles = (rabbitCount / maxCycleSize) + (rabbitCount % maxCycleSize != 0);
            // Each cycle has cycleSize members in it by construction
            total += minRabbitCycles * maxCycleSize;
        }
        
        return total;
    }
};
