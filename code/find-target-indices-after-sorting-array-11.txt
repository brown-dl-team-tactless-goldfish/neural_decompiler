int* targetIndices(int* nums, int numsSize, int target, int* returnSize){
    int numLessOrEqualTarget = 0;
    int rs = 0;
    for (int i = 0; i < numsSize; i++){
        if (nums[i] < target) {
            numLessOrEqualTarget++;
        }
        if (nums[i] == target) {
            numLessOrEqualTarget++;
            rs++;
        }
    }
    int *ans = (int*)malloc(rs * sizeof(int));
    
    for (int i = 0; i < rs; i++){
        ans[i] = numLessOrEqualTarget - rs + i;
    }
    *returnSize = rs;
    return ans;
}
