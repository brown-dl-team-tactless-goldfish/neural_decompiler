class Solution {
public:
    long long gridGame(vector<vector<int>>& g) {
        long long sum0 = accumulate(begin(g[0]), end(g[0]), 0LL), sum1 = 0, res = LONG_MAX;
        for(int i = 0; i < g[0].size(); i++) {
            sum0 -= g[0][i];
            res = min(max(sum0, sum1), res);
            sum1 += g[1][i];
        }
        return res;
    }
};
