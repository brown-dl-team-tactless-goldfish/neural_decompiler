
int cmpFunc(const void *a, const void *b){
    return (*(int*)a - *(int*)b);
}


double trimMean(int* arr, int arrSize){
    int i,start,end,sum,newSize;
    
    qsort(arr,arrSize,sizeof(int),cmpFunc);
    
    start = arrSize * 0.05;
    end = arrSize-(arrSize*0.05);
    newSize = arrSize-(start*2);
    sum = 0;
    
    for(i=start;i<end;i++){
        sum += arr[i];
    }
    //printf("%d %d %d",sum,newSize,arrSize);
    return (double)sum/newSize;
}
