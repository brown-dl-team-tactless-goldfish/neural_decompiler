int findComplement(int num){
    int c = 0;
    int n = num;
    int mask = 1;
    while(n) {
        bool isOne = num & (mask << c);
        if(isOne) {
            num &= ~ (mask << c);
        } else {
            num |= (mask << c);
        }
        n >>= 1;
        c++;
    }
    return num;
}
