int hardestWorker(int n, int** logs, int logsSize, int* logsColSize){
    int max = logs[0][1];
    int max_id = logs[0][0];
    for(int i = 1; i < logsSize; i++){
        int tmp = logs[i][1] - logs[i-1][1];
        if(tmp > max){
            max = tmp;
            max_id = logs[i][0];
        }
        else if(tmp == max)
            max_id = fmin(max_id, logs[i][0]) ;
    }
    return max_id;
}
