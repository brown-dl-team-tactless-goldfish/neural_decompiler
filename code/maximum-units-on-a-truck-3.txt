int comp(void* u, void* v)
{
    return -*(*((int**)u)+1) + *(*((int**)v)+1);
}

int maximumUnits(int** boxTypes, int boxTypesSize, int* boxTypesColSize, int ts){

    qsort(boxTypes,  boxTypesSize, sizeof(int*),   comp);
    
    int ct = 0;
    int i = 0;
    while(ts > 0 && i < boxTypesSize)
    {
        ct += fmin(ts, boxTypes[i][0] ) * boxTypes[i][1];
        ts -= boxTypes[i][0];
        i++;
    }
    return ct; 
}
