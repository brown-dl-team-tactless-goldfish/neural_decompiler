int alternateDigitSum(int n){
    int data[32] = {0};
    int idx = 0;
    while(n){
        data[idx] = n % 10 ;
        n /= 10 ;
        idx++;
    }
    int ans = 0;
    int k = 0;
    for(int i = idx-1 ;i >= 0; i--){
        if(k % 2 == 0)
            ans += data[i] ;
        else
            ans -= data[i] ;
        k++ ;
    }
    return ans ;
}
