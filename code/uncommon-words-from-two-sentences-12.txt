 vector<string> uncommonFromSentences(string s1, string s2) {
    
        unordered_map<string,int> mp;
    
        stringstream p(s1),q(s2);
        string word;
    
        while(p >> word or q >> word)   mp[word]++;
    
        vector<string> ans;
    
        for(auto i:mp)
                 if(i.second==1) ans.push_back(i.first);
    
        return ans;
    
}
