class Solution {
public:
    long long mostPoints(vector<vector<int>>& q) {
        vector<long long> dp(q.size() + 1, 0);
        for (int i = q.size() - 1; i >= 0; i--) {
            int index = min(i + q[i][1] + 1, (int)q.size());
            dp[i] = max(dp[i + 1], q[i][0] + dp[index]);
        }
        return dp[0];
    }
};
