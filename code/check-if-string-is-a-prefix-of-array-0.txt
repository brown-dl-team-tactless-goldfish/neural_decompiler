bool isPrefixString(char * s, char ** words, int wordsSize){
    int i = 0;
    char *pch;
    while (i < wordsSize && *s) {
        for (pch = words[i]; *pch; pch++) {
            if (*pch != *s++) {
                return false;
            }
        }
        
        i++;
    }
    
    return *s == '\0';
}
