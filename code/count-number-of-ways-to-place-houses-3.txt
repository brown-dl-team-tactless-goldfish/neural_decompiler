public class Solution
{
    public int CountHousePlacements(int n)
    {
        long[] dp = new long[n + 1];
        dp[0] = 2;
        dp[1] = 3;
        long mod = (long)(1e9 + 7);

        for (int i = 2; i < n; i++) {
            dp[i] = dp[i - 1] % mod + dp[i - 2] % mod;
        }

        return (int)((dp[n - 1] * dp[n - 1]) % mod);
    }
}
