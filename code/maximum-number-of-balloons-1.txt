int maxNumberOfBalloons(char* text) {
    char ans[8] = "balloon";
    int size = strlen(text), cnt = INT_MAX;
    int arr[26] = {0};
    for(int i = 0; i < size; i++) {
        if(text[i] == 'a' || text[i] == 'b' || text[i] == 'l' || text[i] == 'o' || text[i] == 'n') arr[text[i] - 'a']++;
    }
    arr['l'-'a'] /= 2;
    arr['o'-'a'] /= 2;
    for(int i = 0; i < 7; i++) {
        cnt = (arr[ans[i] - 'a'] < cnt) ? arr[ans[i] - 'a'] : cnt;
    }
    return cnt == INT_MAX ? 0 : cnt;
}
