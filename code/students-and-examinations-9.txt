with cte as
(
select student_id,student_name,subject_name
from students s cross join subjects s1 
) 

select c.student_id,student_name,c.subject_name, isnull(attended_exams,0) as attended_exams 
from cte c left join (select student_id, subject_name,count(subject_name) as attended_exams 
                      from examinations
                      group by student_id,subject_name) e 
                      on e.student_id = c.student_id and e.subject_name = c.subject_name
order by c.student_id,subject_name
