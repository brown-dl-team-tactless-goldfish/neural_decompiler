class Solution {
public:
    vector<long long> maximumEvenSplit(long long num) 
    {
        if(num%2==1)
        {
            return {};
        }
        vector<long long>ans;
        long long sum=0;
        for(int i=1;;i++)
        {
            if(sum+i*2<=num)
            {
                ans.push_back(i*2);
                sum+=i*2;
            }
            else
            {
                break;
            }
        }
        ans.back()+=(num-sum);
        return ans;
    }
};
