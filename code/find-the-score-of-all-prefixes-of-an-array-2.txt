/**
 * Note: The returned array must be malloced, assume caller calls free().
 */

long long* findPrefixScore(int* nums, int numsSize, int* returnSize)
{
    int i;
    long max=0;
    long long *new=(long long*)malloc(sizeof(long long)*numsSize);
    for(i=0;i<numsSize;i++)
    {
        if(max<nums[i])
        max=nums[i];
        new[i]=nums[i]+max;
        
    }
    long long temp=0;
    for(i=0;i<numsSize;i++)
    {
        temp+=new[i];
        new[i]=temp;
    }
    *returnSize=numsSize;
    return new;    
}
