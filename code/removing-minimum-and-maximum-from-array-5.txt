public class Solution {
    public int MinimumDeletions(int[] nums) {
        
        if(nums == null || nums.Length == 0)
            return 0;
        
        int minIdx = -1, maxIdx = -1, min = Int32.MaxValue, max = Int32.MinValue;
        for(int i = 0; i < nums.Length; i++)
        {
            if(nums[i] < min)
            {
                min = nums[i];
                minIdx = i;
            }
            if(nums[i] > max)
            {
                max = nums[i];
                maxIdx = i;
            }
        }     
        
        int smallerIdx = Math.Min(minIdx, maxIdx);
        int largerIdx = Math.Max(minIdx, maxIdx);
        
        int operation1 = largerIdx + 1;
        int operation2 = nums.Length - smallerIdx;
        int operation3 = smallerIdx + 1 + nums.Length - largerIdx;
        
        return Math.Min(Math.Min(operation1, operation2), operation3);
    }
}
