vector<int> constructArray(int n, int k) {
	vector<int> res {};
	int i = 1, j = n;
	while (i <= j) {
		if (k > 0) {
			res.push_back(k-- % 2 ? i++ : j--);
		}
		else
			res.push_back(i++);
	}
	return res;
}
