int passThePillow(int n, int time){
    int k= 0;
    bool flag=true;
    for(int i= 0;i<time;i++){
        if(flag){
            k++;
            if(k==n-1){
                flag = false;
            }
        }
        else if(!flag){
            k--;
            if(k==0){
                flag = true;
            }
        }
    }
    return k+1;
}
