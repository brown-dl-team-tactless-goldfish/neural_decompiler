long long interchangeableRectangles(vector<vector<int>>& a) {
        map<pair<int, int>, int> mp;
        long long result=0;
        for(auto &g:a){
            int gcd = __gcd(g[0],g[1]);
            double ratio = (double) g[0] / (double) gcd;
            pair<int, int> key = {g[0]/gcd, g[1]/gcd};
		    if(mp.find(key) != mp.end()) result += mp[key];
		    mp[key]++;
        }
        return result;
    }
