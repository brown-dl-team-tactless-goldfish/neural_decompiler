char ** divideString(char * s, int k, char fill, int* returnSize){
    int n = strlen(s);
    int N = n/k + ((n%k)? 1 : 0) ;

    char** ans = malloc( N * sizeof(char*));
    for(int i = 0; i < N; i++){
        ans[i] = malloc((k+1) * sizeof(char));
        ans[i][k] = '\0';
        strncpy(ans[i], &s[i*k], k);
    }
    if(n%k){
        for(int i = (n%k); i < k; i++){
            ans[n/k][i] = fill;    
        }
    }
    *returnSize = N;
    return ans;
}
