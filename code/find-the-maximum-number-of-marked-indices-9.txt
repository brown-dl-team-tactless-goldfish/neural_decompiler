class Solution {
public:
    int maxNumOfMarkedIndices(vector<int>& nums) {
        sort(begin(nums),end(nums));
        long long j = (nums.size()+1)/2;
        long long i = 0,ans=0;
        while(j<nums.size()){
            if((2*nums[i]<=nums[j])){
                ans+=2;
                i+=1;
            }
            j+=1;
        }
        return ans;
    }
};
