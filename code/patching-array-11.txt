public class Solution {
    public int MinPatches(int[] nums, int n) {
        int i = 0, result = 0, len = nums.Length;
        long max = 0;
        
        while (max < n) {
            if (i < len && nums[i] <= (max + 1)) {
                max += nums[i];
                i++;
            }
            else {
                max += (max + 1);
                result++;
            }
        }
        
        return result;
    }
}
