int numMatchingSubseq(char* s, char** words, int wordsSize) {
    int cnt = 0;
    for (int i = 0; i < wordsSize; i++) {
        int len = strlen(words[i]);
        bool check = true;
        char* str = s;
        for (int j = 0; j < len; j++) {
            str = strchr(str, words[i][j]);
            if (!str) {
                check = false;
                break;
            }
            str++;
        }
        if (check) cnt++;
    }
    return cnt;
}
