class Solution {
public:
    int minMoves2(vector<int>& nums) {
        
        int n = nums.size();
        
        sort(nums.begin(), nums.end());
        int median = (n%2) ? nums[n/2] : (nums[(n-1)/2] + nums[(n+1)/2])/2;
        int res = 0;
        
        for (int i = 0; i < n; i++) {
            res += abs(nums[i]-median);
        }
        
        return res;
    }
};
