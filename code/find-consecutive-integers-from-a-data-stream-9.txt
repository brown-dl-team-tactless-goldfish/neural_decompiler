class DataStream {
    int val,K,cnt=0;
public:
    DataStream(int value, int k) {
        val = value;
        K = k;
    }
    bool consec(int num) {
        if(num == val)
            cnt++;
        else
            cnt=0;
        if(cnt >= K)
            return true;
        return false;
    }
};
