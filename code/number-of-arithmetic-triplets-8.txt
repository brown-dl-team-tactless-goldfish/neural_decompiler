int arithmeticTriplets(int* nums, int numsSize, int diff) {
    unsigned int i, j, k, ans;
    for (i = ans = 0; i < numsSize; i++) {
        for (j = i + 1; j < numsSize; j++) {
            if (nums[j] - nums[i] == diff) {
                for (k = j + 1; k < numsSize; k++) {
                    if(nums[k] - nums[j] == diff)
                        ans++;
                }
            }
        }
    }
    return ans;
}
