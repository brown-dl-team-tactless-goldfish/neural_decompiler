char * longestCommonPrefix(char ** strs, int strsSize)
{
    char *stack = (char*)malloc(sizeof(char));
    stack[0] = '\0';

    for(int i=0 ; i < strlen(strs[0]) ; i++)
    {
        char c = strs[0][i];

        for(int j=1 ; j < strsSize ; j++)
        {
            if(c != strs[j][i])
                return (stack);
        }

        stack = realloc(stack,(i+2) * sizeof(char));
        stack[i] = c ;
        stack[i+1] = '\0';
    }
    return (stack);
}
