#define NEGATIVE(val) (0-val)
int* sumZero(int n, int* returnSize){

    int index, value;
    value = ((n % 2) == 0) ? NEGATIVE(n/2) : NEGATIVE((n-1)/2);
    int *array = (int*)calloc(n, sizeof(int));
    
    for(index = 0; index < n; index++)
    {
        if(((n % 2) == 0) && (value ==  0))
            value++;        
        array[index] = value++;
    }
    *returnSize = n;
    return array;
}
