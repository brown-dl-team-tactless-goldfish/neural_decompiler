char * removeDuplicates(char * s){
    int n = strlen(s);
    char* stack = malloc((n+1) * sizeof(char));
    int idx = 0;
    for(int i = 0; i < n; i++){        
        stack[idx] = s[i];
        while(1){
            if(idx <= 0)
                break;
            if(stack[idx] == stack[idx-1])
                idx -= 2;
            else
                break;
        }
        idx++;
    }
    stack[idx] = '\0';
    return stack;
}
