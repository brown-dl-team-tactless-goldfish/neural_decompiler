int compare(const void *a , const void *b) {
    int A =*((int *)a) ;
    int B =*((int *)b) ;
    return B-A ;   
}
int minimumCost(int* cost, int costSize){
    qsort(cost , costSize , sizeof(int) , compare) ;
    int total = 0; ;
    for(int i=1 ; i <= costSize ; i++) {
        if(!(i % 3)) continue ;
        total += cost[i - 1] ;
    }

    return total ;

}
