char * toHex(int num){
    if(num == 0) return "0";
    int idx = 8;
    char* ans = (char*)calloc(idx+1, sizeof(char));
    ans[idx--] = '\0';
    while(idx >= 0 && num){
        int currHex = (num & 0xF);
        if(currHex <= 9)
            ans[idx--] = '0' + currHex;
        else
            ans[idx--] = 'a' + currHex - 10;
        num >>= 4;
    }
    return &ans[idx+1];
}
