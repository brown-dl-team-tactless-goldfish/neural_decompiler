bool init_ed = false;
bool is_sum[100001];

int rev(int x)
{
    int t = 0;
    while (x) {
        t = t * 10 + x % 10;
        x /= 10;
    }
    return t;
}

void init()
{
    init_ed = true;
    for (int i = 0; i <= 100000; ++i) {
        int x = i + rev(i);
        if (x <= 100000) {
            is_sum[x] = true;
        }
    }
}

bool sumOfNumberAndReverse(int num)
{
    if (!init_ed) {
        init();
    }
    return is_sum[num];
}
