class Solution {
public:
    int minSteps(string s, string t) {
        vector<int> first(26);
        vector<int> second(26);
        
        for(char& c : s){
            first[c - 'a']++;
        }
        for(char& c : t){
            second[c - 'a']++;
        }
        
        int ans = 0;
        for(int i = 0; i < 26; i++){
            ans += abs(first[i] - second[i]);
        }
        return ans;
    }
};
