#define MAX(A,B) ((A)>(B)?(A):(B))
int digitsum(int digit){
    int sum=0;
    while(digit/10) {
        sum+=digit%10;
        digit/=10;
    }
    sum+=digit;
    return sum;
}
int maximumSum(int* nums, int numsSize){
    int count[100]={0};
    int max=-1,sum;
    int i;
    for(i=0;i<numsSize;i++) {
        sum=digitsum(nums[i]);
        if(count[sum] != 0){
            max=MAX(max,count[sum]+nums[i]);
        }
        count[sum] = MAX(count[sum],nums[i]);
    }
    return max;

}
