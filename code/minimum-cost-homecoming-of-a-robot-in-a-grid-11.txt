public class Solution {
    public int MinCost(int[] sP, int[] hP, int[] rC, int[] cC) {
        if (sP[0] == hP[0] && sP[1] == hP[1]) return 0;
        
        int i = sP[0], j = sP[1], k = hP[0], l = hP[1], r = 0;
        
        while (i != k) {
            i += (k - i) / Math.Abs(k - i);
            r += rC[i];
        }
        
        while (j != l) {
            j += (l - j) / Math.Abs(l - j);
            r += cC[j];
        }
                
        return r;
    }
}
