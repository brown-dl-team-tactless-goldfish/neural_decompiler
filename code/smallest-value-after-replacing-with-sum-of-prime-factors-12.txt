class Solution {
public:
    int smallestValue(int n) {
        int p = primeSum(n);
        while(n != p){
            n = p;
            p = primeSum(n);
        }
        return n;
    }

    int primeSum(int n){
        int lim = n;
        int sum = 0;
        for(int i = 2; i <= lim; i++){
            
            while(n % i == 0){
                    n /= i;
                    sum += i;
                }

        }
        return sum;
    }
};
