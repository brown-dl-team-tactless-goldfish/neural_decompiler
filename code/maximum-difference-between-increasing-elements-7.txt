int maximumDifference(int* nums, int numsSize){
    int max_diff = -1;
    
    for(int i=0;i<numsSize-1;i++){
        for(int j=i;j<numsSize;j++){
            if(nums[i] < nums[j]){
                if(nums[j]-nums[i] > max_diff){
                    max_diff = nums[j]-nums[i];
                }
            }
        }
    }
    
    return max_diff;
}
