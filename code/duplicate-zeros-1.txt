void duplicateZeros(int* arr, int arrSize){
int* temp_buffer = malloc(arrSize * sizeof(int));
int read_ptr=0;
int write_ptr=0;
int i;
    
while (write_ptr<arrSize)
    if (arr[read_ptr]!=0)
        temp_buffer[write_ptr++]=arr[read_ptr++];
      else {
          read_ptr++;
          temp_buffer[write_ptr++]=0;
          if (write_ptr==arrSize) break;
          temp_buffer[write_ptr++]=0;
      }

for (i=0; i<arrSize; i++) arr[i]=temp_buffer[i];
    
return;
}
