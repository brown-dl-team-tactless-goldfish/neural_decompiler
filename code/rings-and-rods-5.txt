 int countPoints(char * rings)
 {
     unsigned int arr[10] = {0};
     int sum = 0;
     unsigned int rod = ( 1u << ( 'B' -'A'))  | ( 1u << ( 'R'-'A')) | ( 1u << ( 'G' -'A'));
     
     for ( int i = 0 ; i < strlen(rings)-1 ; i = i+ 2)
     {
         arr[rings[i+1] - '0'] =  arr[rings[i+1]-'0'] | ( 1u << ( rings[i] -'A'));
         
     }
     
     for ( int i = 0 ; i < 10 ; i++)
     {
         if ( arr[i] == rod)
         {
             sum++;
         }
 
     }
     return sum;
 }
