    bool isIdealPermutation(vector<int>& nums) {
        vector<int>v(nums.size());
        int i,n=nums.size();
        v[n-1]=nums[n-1];
        for(i=n-2;i>=0;i--)
        {
            v[i]=min(nums[i],v[i+1]);
        }
        for(i=0;i<n-2;i++)
        {
            if(nums[i]>v[i+2])
            {
                return false;
            }
        }
        return true;
    }
};
