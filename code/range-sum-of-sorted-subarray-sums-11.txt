class Solution {
public:
    int MOD = 1e9 + 7 ;
    int rangeSum(vector<int>& nums, int n, int left, int right) {
        vector<int> arr ;
        for(int i = 0 ; i < nums.size() ; ++i ){
            int sum = 0 ;
            for(int j = i ; j < nums.size() ; ++j ) sum += nums[j] , arr.push_back(sum) ;   
        }
        
        sort(begin(arr),end(arr)) ;
        int ans = 0 ;
        for(int i = left - 1 ; i <= right - 1 ; ++i ) ans += arr[i] , ans %= MOD ;
        return ans ;
    }
};
