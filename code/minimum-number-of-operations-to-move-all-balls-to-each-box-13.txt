int* minOperations(char * boxes, int* returnSize){
    int n=strlen(boxes),*dl=(int*)malloc(n*sizeof(int)),*dr=(int*)malloc(n*sizeof(int)),i,j;
    int *arr=(int*)malloc(n*sizeof(int));
    for(i=j=0;i<n;i++){
        dl[i]=+j;
        if(boxes[i]=='1')j++;
        if(i!=0)dl[i]+=dl[i-1];
    }
    for(i=n-1,j=0;i>=0;i--){
        dr[i]=+j;
        if(boxes[i]=='1')j++;
        if(i!=n-1)dr[i]+=dr[i+1];
    }
    for(i=0;i<n;i++)
        arr[i]=dl[i]+dr[i];
    *returnSize=n;
    return arr;
}
