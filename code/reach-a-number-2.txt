int reachNumber(int target) {
    if(target<0){
        target=-target;
    }
    int sum=0;
    int step=0;
    while(1){
        step++;
        sum+=step;
        if(sum==target){
            break;
        }
        if(sum>target&&(sum-target)%2==0){
             break;
        }
    }
    return step;
}
