class Solution {
public:
    int longestContinuousSubstring(string s) {
        int n=s.size();
        int i=0;
        int res=0;
        while(i<n){
            int j=i+1;
            while(j<n and (int)(s[j]-s[j-1])==1){
                j++;
            }
          
            res=max(res,j-i);
            i=j;
        }
        return res;
    }
};
