public class Solution {
    public string LongestNiceSubstring(string s) {
        string res = "";
        for(int i=0; i<s.Length; i++)
        {
            string tmp = "";
            for(int j=i; j<s.Length; j++)
            {
                tmp += s[j];
                if(tmp.IsNiceSubstring() && tmp.Length > res.Length)
                {
                    res = tmp;
                }
            }
        }
        return res;
    }
}

public static class Ex{
    public static bool IsNiceSubstring(this string s){
        foreach(char c in s)
        {
            if(c-'a' >= 0 && !s.Contains(Char.ToUpper(c)))
            {//lower case
                return false;
            }
            if(c-'a' < 0 && !s.Contains(Char.ToLower(c)))
            {
                return false;
            }
        }
        return true;
    }
}
