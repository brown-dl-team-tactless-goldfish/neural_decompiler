int* getConcatenation(int* nums, int numsSize, int* returnSize){
    int twiceLarger = 2 * numsSize;
    int *result = malloc(sizeof(int) * twiceLarger);
    
    for (int i = 0; i < numsSize; ++i){
        result[i] = nums[i];
        result[i + numsSize] = nums[i];
    }
        
    *returnSize = twiceLarger;
    
    return result;
}
