int findMiddleIndex(int* nums, int numsSize)
{
    for (int i = 1; i < numsSize; i++)
        nums[i] += nums[i - 1];
    
    if (nums[0] == nums[numsSize - 1])
        return 0;
    
    for (int i = 1; i < numsSize; i++)
        if (nums[i - 1] == nums[numsSize - 1] - nums[i])
            return i;
    
    return -1;
}
