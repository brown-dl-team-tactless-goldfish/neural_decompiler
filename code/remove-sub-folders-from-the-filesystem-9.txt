public IList<string> RemoveSubfolders(string[] folders) {
	var set = new HashSet<string>();
	string parent = null;

	foreach (var folder in folders.OrderBy(x => x)) {
		if (parent != null && IsSubFolder(folder, parent)) continue;
		set.Add(folder);
		parent = folder;
	}

	return set.ToArray();

	bool IsSubFolder(string f, string p) =>
		f.StartsWith(p) && f.Split('/').Length != p.Split('/').Length;
}
