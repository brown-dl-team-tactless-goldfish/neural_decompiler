class Solution {
public:
    int maximizeGreatness(vector<int>& nums) {
        sort(nums.begin(), nums.end());
        int count = 0;
        int l = nums.size()-2;
        int r = nums.size()-1;
        while (l >= 0 && r >= 0){
            if(nums[l] < nums[r]){
                count++;
                l--;
                r--;
            }
            else l--;
        }
        return count;
    }
};
