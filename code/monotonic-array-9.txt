    public boolean isMonotonic(int[] A) {
        int i;
        boolean flag;
        for(i=1; i<A.length && A[0]==A[i]; i++);
        if(i== A.length) return true;
        else if(A[0]<A[i]) flag = true;
        else flag = false;
        for(; i<A.length; i++){
            if(flag && A[i] < A[i-1]) return false;
            if(!flag && A[i] > A[i-1]) return false;
        }
        return true;
    }
