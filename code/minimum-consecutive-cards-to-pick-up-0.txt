int minimumCardPickup(int* cards, int cardsSize){
    int *map = calloc((1e6+1), sizeof(int)); // initialized 0 (no occurrence)
    int ans = cardsSize;
    for(int i = 0; i < cardsSize; i++){
        if (map[cards[i]] != 0){
            int temp = i+1 - map[cards[i]] + 1;
            ans = (temp < ans? temp: ans);
        }
        map[cards[i]] = i+1; //+1 to avoid confuse of "index 0" and "absence of the number"
    }
    free(map);
    return (ans == cardsSize? -1: ans);
}
