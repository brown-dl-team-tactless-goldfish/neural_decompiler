public class Solution {
    public IList<int> TwoOutOfThree(int[] n1, int[] n2, int[] n3) {
        HashSet<int> r = new HashSet<int>();
        
        HashSet<int> v = new HashSet<int>(n1);
        
        foreach (int n in n2) 
            if (v.Contains(n)) 
                r.Add(n);
        
        v.UnionWith(new HashSet<int>(n2));
        
        foreach (int n in n3) 
            if (v.Contains(n)) 
                r.Add(n);
        
        return r.ToList();
    }
}
