int numEquivDominoPairs(vector<vector<int>>& nums) {
    map<pair<int,int>,int>m;
    int c = 0;
    for(auto x:nums){
        int u = x[0];
        int v = x[1];
        if(m.find({u,v})!=m.end()){
            c+=m[{u,v}];
        }
        if(u!=v){
        if(m.find({v,u})!=m.end()){
            c+=m[{v,u}];
        }
        }
        m[{u,v}]++;
    }
    return c;
}
