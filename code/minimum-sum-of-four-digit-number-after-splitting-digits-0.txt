int cmp(const void* a, const void* b) {
    return *(const int*) a - *(const int*) b;
}

int minimumSum(int num) {
    int ans = 0;
    int* arr = (int*) malloc(4 * sizeof(int));
    arr[0] = num / 1000;
    num %= 1000;
    arr[1] = num / 100;
    num %= 100;
    arr[2] = num / 10;
    arr[3] = num % 10;
    qsort(arr, 4, sizeof(int), cmp);
    ans = 10 * arr[0] + arr[3] + 10 * arr[1] + arr[2];
    free(arr);
    return ans;
}
