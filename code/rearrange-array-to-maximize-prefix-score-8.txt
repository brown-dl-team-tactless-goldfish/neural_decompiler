class Solution {
public:
    int maxScore(vector<int>& nums) {
//Greatest element at the end
        sort(nums.begin(),nums.end());
 int c=0;
//To avoid integer overflow
       long long int sum=0;
        
for(int i=nums.size()-1;i>=0;i--){
sum+=nums[i];
    if(sum>0){
c++;
    }
    else{
break;
    }

}
        return c;
        
       
    }
};
