int secondHighest(char * s){
    int len = strlen(s);
    int max1st = -1, max2nd = -1;
    for(int i = 0; i < len; i++){
        if(s[i]<='9')
        {
            int digit = s[i]-'0';/*0-9*/
            if(digit>max1st){
                max2nd = max1st;
                max1st = digit;
            }
            else if(digit>max2nd && digit<max1st)
                max2nd = digit;
        }
    }
    return max2nd;
}
