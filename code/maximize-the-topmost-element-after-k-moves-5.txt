    if(k > n and n > 1) return *max_element(nums.begin(),nums.end());
    
    if(k % 2 == 0 and n == 1) return nums[0];
    else if(k % 2 != 0 and n == 1) return -1;
    
    
    if(k == 0) return nums[0];
    if(n == 1) return -1;
    
    int c = k,ma = INT_MIN;
    
    
    
    
    while(i < n){
        if(c == 1) break;
        if(nums[i] > ma) ma = nums[i];
        c--;
        i++;
    }
    
    
    if(nums[i + 1] > ma and i != n - 1) return nums[i + 1];
    else return ma;
}
