char * removeOuterParentheses(char * s) {
    int slen = strlen(s), cnt = 0, copylen = 0, start = 0;
    char* res = (char*) calloc(slen + 1, sizeof(char));
    for (int i = 0; i < slen; i++) {
        copylen++;
        if (s[i] == ')') cnt--;
        if (s[i] == '(') cnt++;
        if (cnt == 0) {
            strncpy(res + start, s + i - copylen + 2, copylen - 2);
            start += copylen - 2;
            copylen = 0;
        }
    }
    return res;
}
