int cmp(const void* a, const void* b){
    return *(int*)a - *(int*)b;
}
bool asteroidsDestroyed(int mass, int* asteroids, int asteroidsSize){
    int n = asteroidsSize; 
    long cur = mass;
    qsort(asteroids, n, sizeof(int), cmp) ;
    for(int i = 0; i < n; i++){
        if(cur < asteroids[i])
            return false;
        else
            cur += asteroids[i];
    }
    
    return true;
}
