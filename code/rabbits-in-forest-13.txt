    int n=answers.size(), i, group_size,group_needed,reported_rabit,ans=0;
    
    unordered_map<int,int> m;
    
    for(auto x:answers)
    {
        m[x]++;
    }
    
    for(auto x: m)
    {
       group_size = x.first+1;
        
       reported_rabit = x.second;
        
       group_needed = ceil((double) reported_rabit /  group_size);
        
       ans = ans + group_needed*group_size;
        
    }
    
    return ans;
