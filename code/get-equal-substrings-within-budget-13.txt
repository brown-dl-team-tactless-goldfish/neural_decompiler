class Solution {
public:
    int equalSubstring(string s, string t, int maxCost, int cost=0, int ans=0) {
        auto ham = [](auto a, auto b) {
            return abs(a - b);
        };
        for (int i=0, j=0, n=s.size(); i < n && j < n; ++j) {
            for (cost += ham(s[j], t[j]); i <= j && cost > maxCost; ++i)
                cost -= ham(s[i], t[i]);
            ans = max(ans, j+1-i); // +1 for i..j inclusive
        }
        return ans;
    }
};
