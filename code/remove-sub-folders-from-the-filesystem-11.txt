    sort(folder.begin(),folder.end());
    
    vector<string> res;
    
    res.push_back(folder[0]);
    
    for(int i=1; i<folder.size(); i++)
    {
        string root = res[res.size()-1] + "/";
            
        if(folder[i].substr(0,root.size()) != root)
        {
            res.push_back(folder[i]);
        }
    }
    
    return res;
    
}
