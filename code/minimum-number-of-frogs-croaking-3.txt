int minNumberOfFrogs (char *croakOfFrogs) {
	int  i, idx, cnt = 0, max = 0;
	char index[128] = {['c'] = 0, ['r'] = 1, ['o'] = 2, ['a'] = 3, ['k'] = 4};
	int  state[5] = {0};

	for (i = 0; croakOfFrogs[i]; i++) {
		idx = index[croakOfFrogs[i]];
		if (idx == 0) {
			cnt++;
			if (max < cnt - state[4]) {
                max = cnt - state[4];
            }
		} else if (state[idx-1]) {
			state[idx-1]--;
		} else {
			return -1;
		}
		state[idx]++;
	}

	return cnt == state[4] ? max : -1;
}
