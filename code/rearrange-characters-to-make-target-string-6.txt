public class Solution {
    public int RearrangeCharacters(string s, string target) {
        
        int[] freq = new int[26];
        foreach(var c in s)
            freq[c - 'a']++;
        
        int cnt = 0;
        while(true)
        {
            foreach(var c in target)
            {
                freq[c - 'a']--;
                if(freq[c - 'a'] < 0)
                    return cnt;
            }
            cnt++;
        }
        
        return 0;
    }
}
