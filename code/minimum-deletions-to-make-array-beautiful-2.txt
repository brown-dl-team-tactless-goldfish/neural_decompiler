class Solution {
public:
    int minDeletion(vector<int>& a) {
        int c = 0 ; int j  = 0 ; 
        for(int i = 0 ; i < a.size()-1;i++)
        {
            if((i-c)%2==0 && a[i]==a[i+1])
            {
                c++ ; 
            }
        }
        if((a.size()-c)%2!=0)c++;
        return c ; 
    }
};
