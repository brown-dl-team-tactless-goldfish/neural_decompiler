class Solution {
public:
    int lim=1950;
    int maximumPopulation(vector<vector<int>>& logs)
    {
        vector<int>pop(101,0);
        for(auto &v:logs)
        {
            pop[v[0]-lim]++;
            pop[v[1]-lim]--;
        }
        int k=1950,res=pop[0];
        for(int i=1;i<=100;i++)
        {
            pop[i]+=pop[i-1];
            if(pop[i]>res)
            {
                res=pop[i];
                k=i+lim;
            }
        }
        return k;
    }
};
