int longestNiceSubarray(int* nums, int numsSize){
    int left = 0, right = 1, cnt = 1, maxCnt = 1, mask = nums[0];
    while(right<numsSize){
        if(mask & nums[right])
        {
            if(cnt > maxCnt) maxCnt = cnt;
            cnt--;
            mask ^= nums[left++];
        }
        else
        {
            mask |= nums[right++];
            cnt++;
        }
    }
    return (cnt>maxCnt)? cnt : maxCnt;
}
