public class Solution 
{
    public bool CanArrange(int[] arr, int k) 
    {
        var mods = new int[k];
        foreach(var num in arr)
        {
            var mod = num % k;
            if(mod < 0) mod += k;
            mods[mod]++;
        }

        if(mods[0] % 2 != 0)
            return false;
        for(int i = 1; i < k; i++)
            if(mods[k - i] != mods[i])
                return false;
        
        return true;
    }
}
