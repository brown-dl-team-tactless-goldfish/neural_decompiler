public class Solution
{
    public int MinElements(int[] nums, int limit, int goal)
    {
        long sum = 0;
        for (int i = 0; i < nums.Length; i++)
            sum += nums[i];
        
        long diff = Math.Abs(goal - sum);
        var count = 0;
        if (diff % limit != 0)
            count++;
        
        count += (int)(diff / limit);
        return count;
    }
}
