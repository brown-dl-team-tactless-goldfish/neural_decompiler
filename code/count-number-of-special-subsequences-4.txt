class Solution {
public:
    int countSpecialSubsequences(vector<int>& nums) {
        const int m = 1000000007;
        int a=0,b=0,c=0;
        for(int i=0;i<nums.size();++i){
            a=nums[i]==0 ? (a*2+1)%m : a;
            b=nums[i]==1? (b*2%m+a)%m : b;
            c=nums[i]==2 ? (c*2%m+b)%m : c;
        }
        return c;
    }
};
