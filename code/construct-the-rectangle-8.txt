int findCube(int *nums, int size, int area) {
    int i, j, nth;
    int cases;
    int width;
    int length;
    int diffmin;
    int diffminwidth;

    cases = 1;
    for(i=0; i<size; i++) {
        cases = cases *2;
    }

//printf("cases %d, size %d\n", cases, size);
    j = 0; diffmin = INT_MAX;
    for(i=0; i<cases; i++) {
        nth = i; width = 1;
        for(j=0; j<size; j++) {
            if(nth==0) break;
            if(nth%2 != 0) {
                width *= nums[j];
//                printf("%d\t", nums[j]);
            }
//printf("i %d, j %d, nth = %d\n", i, j, nth);
            nth /= 2;
        }
//        printf("\n");
        length = area / width;
        if(diffmin > abs(width-length)) {
//printf("diffmin %d, width %d\n", diffmin, width);

            diffmin = abs(width-length);
            diffminwidth = width;
        }
    }
    
    return diffminwidth;
}
