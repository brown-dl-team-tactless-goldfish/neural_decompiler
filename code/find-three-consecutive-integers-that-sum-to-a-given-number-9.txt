class Solution {
public:
    vector<long long> sumOfThree(long long num) {
        vector<long long>res;
        long long b = num/3;
        long long a = b-1;
        long long c = b+1;
        if(num == (a+b+c))
        {
            res.push_back(a);res.push_back(b);res.push_back(c);
        }
        return res;
    }
};
