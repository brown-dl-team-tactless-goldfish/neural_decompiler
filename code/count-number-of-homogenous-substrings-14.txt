public class Solution {
    public int CountHomogenous(string s) {

        int n = s.Length;
        int modulo = 1_000_000_007;
        int ans = 0;
        
        char prev = '@';
        int c = 0;
        for(int i=0; i<n; i++){
            
            if(s[i] != prev){
                prev = s[i];
                c = 0;
            }
            c += 1;
            ans = (ans + c) % modulo;
        }

        return ans;
    }
}
