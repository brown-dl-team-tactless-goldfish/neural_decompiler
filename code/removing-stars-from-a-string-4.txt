char * removeStars(char * s){
    int len=strlen(s);
    char *stack=(char*)malloc(sizeof(char)*(len+1));
    int top=-1;
    for(int i=0;i<len;i++){
        if(top==-1 && s[i]=='*') continue;
        else if(top!=-1 && s[i]=='*'){
            top--;
        }else{
            stack[++top]=s[i];
        }
    }
    stack[++top]='\0';
    return &stack[0];
}
