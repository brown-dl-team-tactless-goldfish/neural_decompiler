int findMax(int *nums, int numsSize, int *index){
    int i, max = 0;
    int ind = *index;
    for(i=0; i<numsSize; i++){
        if(nums[i] > max && i!=ind){
            max = nums[i];
            *index = i;
        }
    }
    return max;
}

int maxProduct(int* nums, int numsSize){

    int maxValue, secBigValue, product;

    int max = -1;
    int *m = &max; 

    maxValue = findMax(nums,numsSize,m);
    secBigValue = findMax(nums,numsSize,m);
   
    product = ((maxValue-1)*(secBigValue-1));

    free(nums);
    return product;   
}
