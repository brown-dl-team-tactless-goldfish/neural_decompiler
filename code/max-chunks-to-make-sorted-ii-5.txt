    int count = 0, n = arr.size();
    
    vector<int> right_min(n+1);
    
    right_min[n] = INT_MAX;
    
    for(int i = n-1; i>=0; i--)
    {
        right_min[i] = min(right_min[i+1],arr[i]);
    }
    
    int left_max = INT_MIN;
    
    for(int i=0; i<n; i++)
    {
        left_max = max(left_max,arr[i]);
        
        if(left_max <= right_min[i+1])
        {
            count++;
        }
    }
    
    
    
    return count;
    

}
