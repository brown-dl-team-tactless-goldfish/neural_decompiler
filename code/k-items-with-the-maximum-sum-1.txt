/* decision: where k is 
 * 0  - - - - - - - - - - - k
 * |--one--|--zero--|--NegOne--|
 * */
int kItemsWithMaximumSum(int numOnes, int numZeros, int numNegOnes, int k){
    int ans = 0;
    if(k <= numOnes)
        return k;
    else if(k <= (numOnes+numZeros))
        return numOnes;
    else
        k -= (numOnes+numZeros);
    return (numOnes-k);
}
