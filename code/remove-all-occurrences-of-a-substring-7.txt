class Solution:
    def removeOccurrences(self, s: str, part: str) -> str:
        
        len_part = len(part)
        def helper(curS):
            if (part in curS):
                idx = curS.index(part)
                return helper(curS[:idx] + curS[idx+len_part:])
            else:
                return curS
            
        return helper(s)
