class Solution {
public:
    bool PredictTheWinner(vector<int>& nums) {
        int dp[21][21]{};
        for (int i = nums.size() - 1; i >= 0; i--) {
                dp[i][i] = nums[i];
                int sum = nums[i];
            for (int j = i + 1; j < nums.size(); j++) {
                sum += nums[j];
                dp[i][j] = sum - min(dp[i][j - 1], dp[i + 1][j]);
            }
        }
        return dp[0][nums.size() - 1] * 2 >= accumulate(nums.begin(), nums.end(), 0);
    }
};
