int maxProfit(int* nums, int n){
   int sum=0;
   int flag=0;
   for(int i=0;i<n-1;i++){
        if(flag==0 && nums[i]<nums[i+1])
        {
            sum=sum-nums[i];
            flag=1;
        }
        else
            if(flag==1 && nums[i]>nums[i+1])
            {
                sum+=nums[i];  
                flag=0;
            }  
        else continue;
        
    }
    if(flag==1)
        sum=sum+nums[n-1];
    return sum;
    
}
