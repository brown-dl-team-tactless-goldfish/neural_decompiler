class Solution {
public:
    int minMaxGame(vector<int>& nums) {
        int n = nums.size();
        for (int i = n; i >= 1; i /= 2)
            for (int j = 0; j < (n / 2); j++)
            {
                if (j % 2)
                    nums[j] = max(nums[2 * j], nums[2 * j + 1]);
                else
                    nums[j] = min(nums[2 * j], nums[2 * j + 1]);
            }
        return nums[0];
    }
};
