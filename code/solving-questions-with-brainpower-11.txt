    public long MostPoints(int[][] questions) {
        long[] dp=new long[questions.Length];
        return dfs(questions,0,dp);  
    }
    public long dfs(int[][] questions,int id,long[] dp)
    {
        if(id>=questions.Length)return 0;
        if(dp[id]>0)return dp[id];
        int skip=questions[id][1];
        long ret=questions[id][0]+dfs(questions,id+skip+1,dp);
        ret=Math.Max(ret,dfs(questions,id+1,dp));
        return dp[id]=ret;
    }
