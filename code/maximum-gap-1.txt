int cmp(const void *x, const void *y)
{
    return *(int*)x - *(int*)y;
}
int maximumGap(int* nums, int numsSize)
{
    if(numsSize < 2)
    {
        return 0;
    }
    else
    {
        int count = 0;
        qsort(nums, numsSize, sizeof(int), cmp);
        for(int i = 1; i < numsSize; i++)
        {
            if(abs(nums[i] - nums[i-1]) > count)
            {
                count = abs(nums[i] - nums[i-1]);
            }
        }
        return count;
    }
}
