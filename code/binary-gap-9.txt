//868. Binary Gap

int binaryGap(int N){
    int d; //除數
    int tmpN=N; //copy一份N
    int f=-1,r=0,max=0; //front;rear;max差距
    
    while(tmpN){//當被除數還有就繼續
    d=tmpN%2;//找出餘數
    tmpN/=2;//產生下次的被除數
    if(d==1){//若餘數==1，紀錄並看是否為新max
        if(r-f>max && f!=-1)
            max=r-f;
        f=r;
    }
    r++;//繼續往後掃描
    }
    return max;
}
