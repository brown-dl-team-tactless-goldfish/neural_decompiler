int* findDuplicates(int* nums, int numsSize, int* returnSize){
    int i=0,count=0;
    int *ret,tmp[100000]={0};
    ret=(int*)malloc(sizeof(int)*numsSize);
    
    if(numsSize<2){
        *returnSize=NULL;
        return NULL;
    }
        
    do{
        tmp[nums[i]]+=1;
        if(tmp[nums[i]]>1){
            ret[count]=nums[i];
            count+=1;
        }
        i+=1;
    }while(i!=numsSize);
    
    *returnSize=count;
   return ret; 
}
