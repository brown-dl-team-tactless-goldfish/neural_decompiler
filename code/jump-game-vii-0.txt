bool canReach (char * s, int minJump, int maxJump) {
    int i, j;
    
    s[0] = '2';
    for (i = 1, j = 0; s[i]; i++) {
        if (s[i] == '0') {
            while (j < i && (s[j] != '2' || j + maxJump < i)) {
                j++;
            }
            if (s[j] == '2' && j + minJump <= i) {
                s[i] = '2';
            }
        }
    }
    
    return s[i-1] == '2';
}
