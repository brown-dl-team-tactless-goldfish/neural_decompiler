int* leftRigthDifference(int* nums, int numsSize, int* returnSize){
    int right=0;
    int left=0;
    *returnSize = numsSize;
    int* ans = malloc(sizeof(int)* numsSize);
    
    for(int i=0; i<numsSize; i++){
        right+=nums[i];
    }

    for(int i=0; i<numsSize; i++){
        right -= nums[i];
        ans[i] = abs(left - right);
        left += nums[i];
    }

    return ans;
}
