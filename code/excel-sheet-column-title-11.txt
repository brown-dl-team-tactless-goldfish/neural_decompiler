char* convertToTitle(int n) {
    int size=0;
    for(int i=n-1;i>=0;){
        ++size;
        i=i/26-1;
    }
    
    char *str=(char*)malloc(sizeof(char)*(size+1));
    
    for(int i=n-1,j=size-1;j>=0;j--){
        str[j]=i%26+'A';
        i=i/26-1;
    }
    
    str[size]='\0';
    
    return str;
}
