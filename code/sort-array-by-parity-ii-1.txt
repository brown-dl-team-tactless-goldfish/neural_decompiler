int *sortArrayByParityII(int const *A, int ASize, int *returnSize) {
    int odd = 1, even = 0;
    int *ret = (int *) malloc(sizeof(int) * (*returnSize = ASize));
    for (int i = 0; i < *returnSize; ++i) {
        if (A[i] % 2 == 0) {
            ret[even] = A[i];
            even += 2;
        } else {
            ret[odd] = A[i];
            odd += 2;
        }
    }
    return ret;
}
