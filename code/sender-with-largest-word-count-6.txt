	public string LargestWordCount(string[] messages, string[] senders) {
        var dict = new Dictionary<string, int>();
        for (int i=0; i<messages.Length; i++) {
            var count = messages[i].Count(c => c == ' ') + 1;
            if (!dict.TryAdd(senders[i], count))
                dict[senders[i]] += count;
        }
        return dict.OrderByDescending(x => x.Value)
                   .ThenByDescending(x => x.Key, StringComparer.Ordinal) // use StringComparer.Ordinal for lexicographical order 
                   .First().Key;
    }
