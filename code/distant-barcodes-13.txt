class Solution {
public:
    vector<int> rearrangeBarcodes(vector<int>& barcodes) {
        //we use max heap to store pairs
        //pair.first will contain the frequency of element
        //pair.second will contain the element value
        priority_queue<pair<int,int>> pq;
        
        vector<int> freq(10001,0);
        for(int i=0;i<barcodes.size();i++)
        {
            freq[barcodes[i]]++;
        }
        for(int i=0;i<10001;i++){
		//if the freq of element is greater than zero then push it into pq
            if(freq[i])
                pq.push({freq[i],i});
        }
        vector<int> res;
        pair<int,int> prev={0,0};
        while(!pq.empty())
        {
            auto [val,ele]=pq.top();
            pq.pop();
            res.push_back(ele);
            if(prev.first!=0) //if the previous popped out element has frequency greater than zero
                pq.push(prev);
            prev={--val,ele};
        }
        return res;
    }
};
