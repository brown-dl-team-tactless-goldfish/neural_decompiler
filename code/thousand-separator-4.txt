char * thousandSeparator(int n){
    char numStr[14];
    char * output;
    int i = sizeof(numStr) - 1;
    numStr[i] = '\0';
    
    do {
        if((sizeof(numStr) - i) % 4 == 0) {
            numStr[--i] = '.';
        }
        numStr[--i] = n % 10 + '0';
        n /= 10; 
    } while (n != 0);
    
    output = malloc(sizeof(numStr) - i); 

    return strcpy(output, numStr + i);
}
