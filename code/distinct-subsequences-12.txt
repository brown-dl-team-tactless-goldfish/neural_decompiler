class Solution {
public:
    int numDistinct(string s, string t) {
        vector<size_t> dp(t.size() + 1);
        dp[0] = 1;
        for (auto c : s) {
            for (int i = t.size(); i; i--) {
                if (c == t[i - 1]) {
                    dp[i] += dp[i - 1];
                }
            }
        }
        return dp.back();
    }
};
