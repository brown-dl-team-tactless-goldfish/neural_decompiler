class Solution {
public:
    int countServers(vector<vector<int>>& grid) {
        
        int r = grid.size();
        int c = grid[0].size();
        
        // store number of server in each row and column
        vector<int>rows(r,0);
        vector<int>cols(c,0);
        
        for(int i=0;i<r;i++)
            for(int j=0;j<c;j++)
                if(grid[i][j])rows[i]++,cols[j]++;

        // number of connected servers
        int count = 0;
        
        // If a grid position has a server and its row or column has more than 1 server then
        // the server is connected to some other server and so we increment connected server count
        for(int i=0;i<r;i++)
            for(int j=0;j<c;j++)
                if(grid[i][j]==1 && (rows[i]>1 || cols[j]>1))count++;
        
        return count;

    }
};
