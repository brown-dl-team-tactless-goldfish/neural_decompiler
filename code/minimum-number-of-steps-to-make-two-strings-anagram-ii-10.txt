class Solution {
public:
    int minSteps(string s, string t) {
        unordered_map<char,int> m;
        int c=0;
        for(auto& i : s){
            m[i]++;
        }
        for(auto& i : t){
            m[i]--;
        }
        for(auto& itr : m){
            c += fabs(itr.second);
        }
        return c;
    }
};
