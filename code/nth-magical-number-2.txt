int nthMagicalNumber(int N, int A, int B){

    int d,g,h,x,y,i,gcd,z,t,res;
    A>B?(t=A,A=B,B=t):0;
    
    for (gcd=i=1;i<=A;i++)
        (A%i==0 && B%i==0)? gcd=i: 0;
    x=A/gcd, y=B/gcd, z=x+y-1, res=t=N%z;
    
    for(g=h=1,d=0;d<t;d++)
        (x*g < y*h)?(res=x*g++):(res=y*h++);    
    return (int)((((N/z)*((long)x*y)+res)*gcd)%((long)1000000000+7));
}
