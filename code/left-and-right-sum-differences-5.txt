int* leftRigthDifference(int* n, int s, int* rs)
{
    *rs = s;
    long leftSum = 0, rightSum = 0;
    int v;

    for(int i = 0; i < s; i++) rightSum += n[i];
    
    for(int i = 0; i < s; i++)
    {
        v = n[i];
        rightSum -= v;
        n[i] = abs(leftSum - rightSum);
        leftSum += v;
    }

    return n;
}
