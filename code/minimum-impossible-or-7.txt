class Solution {
public:
    int minImpossibleOR(vector<int>& nums) {
        int n=nums.size();
        unordered_map<int,int> y;
        for(int i=0;i<n;i++){
            if((nums[i]&(nums[i]-1))==0) y[nums[i]]=1;
        }
        int i=1;
        while(true){
            if(y[i]==0) return i;
            i*=2;
        }
    }
};
