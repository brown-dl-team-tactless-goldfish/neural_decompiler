  int findCenter(vector<vector<int>>& t) {
         int a = t[0][0];
         int b = t[0][1];
         int c = t[1][0];
         int d = t[1][1];
        if(a==c){
            return a;
        }
        if(a==d){
            return a;
        }
        if(b==c){
            return b;
        }
        if(b==d){
            return b;
        }
        return 0;
    }
