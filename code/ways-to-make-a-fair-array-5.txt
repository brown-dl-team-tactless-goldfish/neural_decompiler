int waysToMakeFair (int* nums, int numsSize) {
	int i, left_odd = 0, left_even = 0, right_odd = 0, right_even = 0, way = 0;

	for (i = 0; i < numsSize; i++) {
		if (i & 1) left_odd += nums[i];
		else left_even += nums[i];
	}

	for (i = numsSize-1; i >= 0; i--) {
		if (i & 1) left_odd -= nums[i];
		else left_even -= nums[i];

		if (left_odd + right_odd == left_even + right_even) way++;

		if (i & 1) right_even += nums[i];
		else right_odd += nums[i];
	}

	return way;
}
