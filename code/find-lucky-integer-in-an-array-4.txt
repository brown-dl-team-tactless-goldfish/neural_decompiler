
int findLucky(int* arr, int arrSize){
    int mx = 0;
    for (int i=0;i<arrSize;i++){
        if (arr[i] > mx)
        {
            mx = arr[i];
        }
    }
    int* arr2 = (int *)calloc(mx+1 , sizeof(int));
    //printf("%d" , mx);
    for (int i=0;i<arrSize;i++)
        arr2[arr[i]]++;
    for (int i=mx ;i>0;i--){
        //printf("%d" , arr2[i]);
        if (arr2[i] == i)
        {
            free(arr2);
            return i;
        }
    }
    free(arr2);
    return -1;
}
