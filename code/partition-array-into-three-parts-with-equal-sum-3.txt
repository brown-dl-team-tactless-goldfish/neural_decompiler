

bool canThreePartsEqualSum(int* A, int ASize){


    int i,s=0,s1=0,c=0,ans=0,temp=0;

    for(i=0;i<ASize;i++)
    {
        s=s+A[i];
    }

    temp=s/3;

    if(s%3!=0)
    ans=0;
    else if(s%3==0)
    {
        s1=0;
    for(i=0;i<ASize;i++)
    {
        s1=s1+A[i];
        if(s1==temp)
        {
            ++c;
            s1=0;
        }
    }
    if(c>=3)
    ans=1;
    else
    ans=0;
    }
    
    return ans;
}
