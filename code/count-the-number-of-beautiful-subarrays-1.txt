class Solution {
public:
    long long beautifulSubarrays(vector<int>& nums) 
    {
        int n=nums.size();
        long long ans=0;
        unordered_map<int,int> mp;
        int c=0;
        mp[c]++;
        for(int i=0;i<n;i++)
        {
            c=c^nums[i];
            if(mp.count(c)==1)
            {
                ans+=(mp[c]);
            }
            mp[c]++;
        }
        return ans;
    }
};
