int difference = (a | b) ^ c;
int num = difference;
int i= 0;
int flips = 0;

while (num){
    /* calculate only for misplaced bits */
    if(difference & (1<<i)){
        // case for 1 condition and 0 condition
        if(c & (1<<i)){
            flips += 1;
        }
        else{
            if(a & (1<<i))
                flips += 1;
            if(b & (1<<i))
                flips += 1;   
        } 
    }  
    i += 1;
    num >>= 1;
}

return flips;
