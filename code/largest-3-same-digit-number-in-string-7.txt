	string largestGoodInteger(string num) {
        string s = "";
        char c = '0';
        for(int i = 0; i < num.size()-2; i++){
            if(num[i] == num[i+1] && num[i] == num[i+2]){
                if(num[i] >= c){
                    s = num.substr(i, 3);
                    c = num[i];
                }
            }
        }
        return s;
    }
