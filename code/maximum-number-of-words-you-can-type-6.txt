int canBeTypedWords(char * text, char * brokenLetters) {

	bool alpha[26] = {0};
	register unsigned char i = 0;
	while(brokenLetters[i] != '\0') {
		alpha[brokenLetters[i] - 'a'] = 1;
		i++;
	}

	bool flag = true;
	register unsigned short j = 0, sum = 0;
	while(text[j] != '\0') {
		if(!isspace(text[j]) && alpha[text[j] - 'a'] == 1)
			flag = false;
		if(isspace(text[j]) && flag)
			sum++;
		else if(isspace(text[j]) && !flag)
			flag = true;
		j++;
	}

	if(flag)
		sum++;

	return sum;
}
