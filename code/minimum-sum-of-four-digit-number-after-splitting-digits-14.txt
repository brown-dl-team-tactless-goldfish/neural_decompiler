int cmpfunc (const void * a, const void * b) {
   return ( *(int*)a - *(int*)b );
}

int minimumSum(int num){
int a,b,c,d;
    int *digit=(int*)malloc(sizeof(int)*4);
    a=num/1000;
    b=(num/100)%10;
    c=(num/10)%10;
    d=num%10;
    digit[0]=a;
    digit[1]=b;
    digit[2]=c;
    digit[3]=d;
    qsort(digit, 4, sizeof(int), cmpfunc);
    int f = digit[0] * 10 + digit[2];
    int s = digit[1] * 10 + digit[3];
        
    return f + s;
    
}
