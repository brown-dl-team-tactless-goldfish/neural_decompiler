char* makeGood(char* s) {
    int slen = strlen(s), top = 0;
    if (slen == 1) return s;
    
    char* res = (char*) calloc(slen + 1, sizeof(char));
    for (int i = 0 ; i < slen; i++) {
        res[top] = s[i];
        if (top > 0 && (res[top] + 32 == res[top-1] || res[top] - 32 == res[top-1])) top -= 2;
        top++;
    }
    res[top] = '\0';
    return res;
}
