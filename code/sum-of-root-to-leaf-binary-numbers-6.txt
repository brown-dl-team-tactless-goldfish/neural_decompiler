/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     struct TreeNode *left;
 *     struct TreeNode *right;
 * };
 */


int sumRecursion(struct TreeNode* root,int sum){
    if(root){
        sum=(sum<<1)|root->val;
        if(!root->left&&!root->right){
            return sum;
        }
        return( sumRecursion(root->left,sum)+sumRecursion(root->right,sum));
    }
    return 0;
}


int sumRootToLeaf(struct TreeNode* root){
return sumRecursion(root,0);
}
