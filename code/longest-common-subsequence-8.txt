int m=strlen(text1);
int n=strlen(text2);

int dp[m+1][n+1];
for(int i=0;i<m+1;i++){
    for(int j=0;j<n+1;j++){
        dp[i][j]=0;
    }
}


for(int i=0;i<m;i++){
    for(int j=0;j<n;j++){
        if(text1[i]==text2[j]){
            dp[i+1][j+1]=dp[i][j]+1;
        }
        else{
            dp[i+1][j+1]=max(dp[i][j+1],dp[i+1][j]);
        }
    }
}

return dp[m][n];
