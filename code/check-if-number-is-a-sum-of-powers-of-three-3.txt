class Solution {
public:
    bool checkPowersOfThree(int n) {
        int x = n;
        int count = 0;
        while(n > 3)
        {
           while(x > 1)
           {
            if(x == 2)
            {
                return false;
            }
            x = x/3;
            count++;
           }
        n = n - pow(3, count);
        x = n;
        count = 0; 
        }
        if(n == 2)
            return false;
        return true;
    }
};
