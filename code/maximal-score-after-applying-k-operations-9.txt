class Solution {
public:
    long long maxKelements(vector<int>& nums, int k) {
        priority_queue<int, vector<int>> pq;
        for(int i = 0; i < nums.size(); i++) pq.push(nums[i]);
        long long ans = 0;
        for(int i = 0; i < k; i++){
            int n = pq.top();
            pq.pop();
            ans += n;
            pq.push((n+2) / 3);
        }
        return ans;
    }
};
