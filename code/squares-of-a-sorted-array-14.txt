int* sortedSquares(int* nums, int numsSize, int* returnSize){
for(int i=0;i<numsSize;i++)
{
    nums[i]=pow(nums[i],2);
}
    int cmpfunc(const void *a,const void  *b)
    {
        return *(int *)a-*(int *)b;
