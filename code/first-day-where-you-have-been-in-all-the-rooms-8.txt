class Solution {
public:
    int firstDayBeenInAllRooms(vector<int>& nextVisit) {
        int n = nextVisit.size();
        int v[n];//v[i] is the first # for room i
        v[0] = 0;
        int ans = 1;
        for(int i=0; i<n-1; i++)
        {
            ans = ((ans - v[nextVisit[i]] + 1000000007) % 1000000007 + ans + 1) %(1000000007);
            v[i+1] = ans - 1;
        }
        return (ans-1+ 1000000007)% 1000000007;
    }
};
