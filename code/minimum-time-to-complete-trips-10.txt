long long minimumTime(vector<int>& nums, int t) {
    sort(nums.begin(),nums.end());
    long long c = 1e16;
    long long l = 0,r = 1e16;
    while(l<=r){
        long long mid = l+(r-l)/2;
        long long x = 0;
        for(int i=0;i<nums.size();i++){
            x += (mid/nums[i]);
            if(x>t) break;
        }
        if(x>=t){
            c = min(c,mid);
            r = mid-1;
        }else{
            l = mid+1;
        }
    }
    return c;
}
