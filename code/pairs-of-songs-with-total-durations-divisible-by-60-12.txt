public class Solution {
    public int NumPairsDivisibleBy60(int[] time) {
        
        if(time == null || time.Length == 0)
            return 0;
        
        int[] remainder = new int[60];
        int cnt = 0;
        foreach(int t in time)
        {
            if(t % 60 == 0)
                cnt += remainder[0];
            else
                cnt += remainder[60 - t % 60];
            
            remainder[t % 60]++;
        }
        
        return cnt;
    }
}
