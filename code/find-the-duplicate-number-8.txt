int findDuplicate(int* nums, int numsSize){

    // scores[i] keeps track of how many occurrences of the value i
    int* scores = calloc(sizeof(int), numsSize);
    
    for (int i = 0; i < numsSize; i++) {
        // Read array element:
        int val = nums[i]; 
        
        // Update its score (number of occurrences):
        scores[val]++;
        
        // Check if duplicate:
        if (scores[val] > 1) {
		    // Duplicate: bail out immediately
            return val;
        }
    }
    
    // (this shouldn't happen if the input is according to the problem constraints)
    return -1;
}
