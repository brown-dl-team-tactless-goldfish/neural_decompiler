class Solution {
public:
    long long countBadPairs(vector<int>& nums) {
        unordered_map<long long int,long long int>um;
        for(int i{0};i<nums.size();++i){
            um[nums.at(i)-i]++;
        }
        long long len=nums.size();
        long long ans=(len*(len-1))/2;
        for(auto i:um){
            ans-=(i.second*(i.second-1))/2;
        }         
        return ans;
    }
};
