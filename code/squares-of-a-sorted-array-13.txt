int* sortedSquares(int* A, int ASize, int* returnSize){
    *returnSize = ASize;
    int i,nL=0,nR=0;
    int *res = (int*) malloc(ASize * sizeof(int));
    
    for(i = 0; i < ASize; i++){
        if(abs(A[nL])>abs(A[ASize-1-nR])) {
            res[ASize-1 -i] = A[nL]*A[nL];
            nL++;
        }else{
            res[ASize-1 -i] = A[ASize-1-nR]*A[ASize-1-nR];
            nR++;
        }
    }
    
    return res;
}
