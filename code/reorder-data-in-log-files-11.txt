public class Solution {
    public string[] ReorderLogFiles(string[] logs) {
        
        if(logs == null || logs.Length == 0)
            return new string[0];
        
        List<string> res = new List<string>();
        List<string> letterLogs = new List<string>();
        List<string> numberLogs = new List<string>();
        foreach(var log in logs)
        {
            string content = log.Split(" ", StringSplitOptions.RemoveEmptyEntries)[1];
            if(IsNumberLog(content))
                numberLogs.Add(log);
            else
                letterLogs.Add(log);
        }
        
        // sort the letter logs
        letterLogs.Sort((l1, l2) =>
                   {
                       int idx1 = l1.IndexOf(' ');
                       string id1 = l1.Substring(0, idx1);
                       string content1 = l1.Substring(idx1 + 1);
                       
                       int idx2 = l2.IndexOf(' ');
                       string id2 = l2.Substring(0, idx2);
                       string content2 = l2.Substring(idx2 + 1);
                       
                       if(content1 == content2)
                           return id1.CompareTo(id2);                     
                       else
                           return content1.CompareTo(content2);         
                   });
        
        
        foreach(var log in letterLogs)
            res.Add(log);
        
        res.AddRange(numberLogs);
        
        return res.ToArray();
    }
    
    private bool IsNumberLog(string log)
    {
        foreach(var c in log)
        {
            if(c < '0' || c > '9')
                return false;
        }
        return true;
    }
}
