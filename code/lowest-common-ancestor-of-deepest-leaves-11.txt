int get(TreeNode* root)
{
    if(!root)
    {
        return 0;
    }
    
    int left = get(root->left);
    int right = get(root->right);
    
    return 1 + max(left, right);
}

TreeNode* lcaDeepestLeaves(TreeNode* root) 
{
    if(!root)
    {
        return 0;
    }
    
    int left_hight = get(root->left);
    
    int right_hight = get(root->right);
    
    if(left_hight == right_hight)
    {
        return root;
    }
    
    return left_hight > right_hight?lcaDeepestLeaves(root->left):lcaDeepestLeaves(root->right);
