

#define max(a, b) ((a) > (b) ? (a) : (b))

int maximumSum(int* arr, int arrSize)
{
    if (arrSize == 1)
        return arr[0];
    int ans = arr[0];
    int drop = 0, nodrop = arr[0];
    for (int i = 1; i < arrSize; i++) {
        drop = max(drop + arr[i], nodrop);
        nodrop = max(nodrop + arr[i], arr[i]);
        if (drop > ans)
            ans = drop;
        if (nodrop > ans)
            ans = nodrop;
    }
    return ans;
}
