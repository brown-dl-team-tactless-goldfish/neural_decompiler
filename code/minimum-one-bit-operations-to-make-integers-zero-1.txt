int minimumOneBitOperations(int n){
    int tmp_n = n;
    int sign = 1;
    int count = 0;
    int bit_pos = 29;
    while(tmp_n){
        if(tmp_n & (1<<bit_pos)){
            count += sign*((1<<(bit_pos+1)) -1);
            tmp_n &= ~(1<<bit_pos);
            sign *= -1;
        }
        bit_pos--;
    }
    return count;
}
