char change(unsigned int n)
{
    if(n<10)
    {return n+'0';}
    else
    {return n-10+'a';}
}
char* toHex(int num) {
    char* result=(char*)malloc(sizeof(char)*9);
    int count=8;
    result[8]='\0';
    if(num==0)
    {
        result[7]='0';
        return &result[7];
    }
    unsigned int a;
    if(num<0)
    {a=num+4294967296;}
    else
    {a=num;}
    while(a!=0)
    {
        count--;
        result[count]=change(a%16);
        a=a/16;
    }
    return &result[count];
}
