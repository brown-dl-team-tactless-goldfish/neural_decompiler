bool canConstruct(char * s, int k){
    int len = strlen(s) ;

    if(len < k)
        return false;
    
    int* character = calloc( 26 , sizeof(int));
    for(int i = 0; i < len; i++){
        character[s[i] - 'a']++;
    }
    
    int odd = 0;
    for(int i = 0; i < 26; i++){
        if(character[i] % 2 == 1)
            odd++;
    }
    if(odd > k)
        return false;
    
    return true;

}
