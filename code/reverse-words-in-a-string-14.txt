for(int i=len-1;i>=0;i--){
    if(s[i]!=' '){
        temp[j++]=s[i];
        isSpace = true;
    }   
    else if (s[i] == ' '){ 
        j--;
        for(;j>=0;j--){
            rev[++m]= temp[j];
        } 
     
        //Reset temp
        j=0;
        temp[j]='\0';
        
        if(isSpace){
            rev[++m] = s[i];
            isSpace=false;
        }
     }  
}
j--;
for(;j>=0;j--){
    rev[++m]= temp[j];
}
rev[++m] = '\0';
trim(rev);
return rev;
