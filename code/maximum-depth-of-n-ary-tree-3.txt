int dfs(struct Node* curr, int depth, int *ret){
    if(!curr)
        return 0;
    
    if(depth> *ret)
        *ret = depth;
    
    int i;
    for(i=0;i<curr->numChildren;i++){
        dfs(curr->children[i], depth+1, ret);
    }
    return 0;
}

int maxDepth(struct Node* root) {
    if(!root)
        return 0;
    int ret = 0;
    dfs(root, 1, &ret);
    
    return ret;
}
