// Just try to form more number of "abc".
// Creater two pointers i and j for string word and "abc" respectively.
// if any of chars in word does not match with abc then increase the count else move the i pointer.

class Solution {
	public:
	int addMinimum(string word) {

		string abc="abc";
		int count=0,j=0;

		for(int i=0;i<word.size();)
		{
			j=j%3;

			if(word[i]==abc[j])
				i++;
			else
				count++;

			j++;
		}

		count +=(3-j);        // if last letter was not c then add count of remaining chars 
		return count;
	 }
};
