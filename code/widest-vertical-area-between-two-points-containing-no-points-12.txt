    public class Solution
    {
        public int MaxWidthOfVerticalArea(int[][] points)
        {
            Array.Sort(points, (p1, p2) => { return p1[0].CompareTo(p2[0]); });
            int res = 0;

            for (int i = 0; i < points.Length - 1; i++)
            {
                var diff = points[i + 1][0] - points[i][0];
                res = Math.Max(res, diff);
            }

            return res;
        }
    }
