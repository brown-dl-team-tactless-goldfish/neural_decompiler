long long minimumMoney(int** transactions, int transactionsSize, int* transactionsColSize){
    long long ans = 0;
    int mostback = 0;
    int makemoney = -1;
    for (int i = 0 ; i < transactionsSize ; i++){
        if (transactions[i][0] > transactions[i][1]){
            ans += transactions[i][0]-transactions[i][1];
            if (mostback < transactions[i][1]){
                mostback = transactions[i][1];
            }
        } else {
            if (makemoney < transactions[i][0]){
                makemoney = transactions[i][0];
            }
        }
    }
    
    if (mostback < makemoney){
        ans += makemoney;
    } else {
        ans += mostback;
    }
    return ans;
}
