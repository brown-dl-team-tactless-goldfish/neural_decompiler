cmpfunc(void *v1, void *v2){
    return ( *(int*)v2 - *(int*)v1);
}

int specialArray(int* nums, int numsSize){
    qsort(nums, numsSize, sizeof(int), cmpfunc);
    if(!nums[0]) return -1;
    for(int i=0; i< numsSize; i++)
        if(nums[i] < i)
            return  (i <= nums[i-1]) ? i : -1;
    return numsSize;
}
