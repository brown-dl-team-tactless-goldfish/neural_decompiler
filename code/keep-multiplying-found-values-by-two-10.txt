bool isexist(vector<int>& nums, int s)
{
    for(int i=0;i<nums.size();i++)
    {
        if(nums[i]==s)
        {
            return true;
        }
    }
    return false;
}

int findFinalValue(vector<int>& nums, int original) {
    int nu=original;
    while(isexist(nums,original))
    {
        original=original*2;
        nu=original;
    }
    return nu;
}
