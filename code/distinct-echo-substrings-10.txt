	public int DistinctEchoSubstrings(string text) {
        var hashSet = new HashSet<string>();
        for (int n = 1; n <= text.Length/2; n++) {
            for (int i=0; i<=text.Length-2*n; i++) {
                if (AreEqualSubstrings(text, i, n+i, n)) // compare current and next substrings of length n
                    hashSet.Add(text.Substring(i, n));
            }
        }
        return hashSet.Count;
    }
    
    bool AreEqualSubstrings(string text, int start1, int start2, int length) {
        for (int i=0; i<length; i++) {
            if (text[start1+i] != text[start2+i])
                return false;
        }
        return true;
    }
