public class Solution {
    public int CountOrders(int n) {
        long result = 1;
        int MOD = 1000000007;
        
        for (int i = 1; i <= 2 * n; i++) {
            result *= i;
            
            if (i % 2 == 0)
                result /= 2;
            
            result %= MOD;
        }
        
        return (int)result;
    }
}
