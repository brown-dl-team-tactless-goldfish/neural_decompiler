class Solution {
public:
    int minIncrementForUnique(vector<int>& nums) {
        sort(nums.begin(),nums.end());
        int count{0};
        int left{0};
        while(left<nums.size())
        {
            if(left>0 and nums[left-1]>=nums[left])
            {
                count+=nums[left-1]-nums[left]+1;
                nums[left]=nums[left-1]+1;
            }
            left++;
        }
        return count;
    }
};
