class Solution {
public:
    long long countGood(vector<int>& A, int k) {
        long long res = 0, N = A.size();
        unordered_map<int, int> mp;
        for(int i = 0, j = 0 , cur = 0; i < N; i++) {
            cur += mp[A[i]]++;
            while(cur >= k) {
                res += N-i;
                cur -= (mp[A[j++]]--) -1;
            }
        }
        return res;
    }
};
