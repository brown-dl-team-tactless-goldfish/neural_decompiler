int minAddToMakeValid(char * s){
    int ctr = 0;
    int numOpening = 0; // pseudo-stack
    while (*s){
        if (*(s++) == '(') numOpening++; // "push" onto stack
        else if (numOpening > 0) numOpening--; // "pop" from stack
        else ctr++; // increment counter

    }
    return (numOpening + ctr);
}
