int partitionDisjoint(int* nums, int numsSize){
    int lmax, rmax;
    int pos = 0;
    int i;
    
    lmax = rmax = nums[0];
    for(i = 1; i < numsSize; i++)
    {
        if(nums[i] > rmax) rmax = nums[i];
        if(nums[i] < lmax)
        {
            lmax = rmax;
            pos = i;
        }
    }
	// pos is index, we need to add 1 for length
    return pos + 1;
}
