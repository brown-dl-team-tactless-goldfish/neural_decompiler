    sort(nums.begin(),nums.end());
    
    
    vector<int> res;
    
    for(int i=1; i<nums.size(); i++)
    {
        nums[i] += nums[i-1];
    }
    
    for(auto &x: queries)
    {
        int tmp=upper_bound(nums.begin(),nums.end(),x) - nums.begin();
        
        res.push_back(tmp);
   }
    
 return res;
