int findLucky(int* arr, int arrSize){
    
    int i,j,sum=-1;
    
    for(i=0;i<arrSize;i++)
    {
        int sun=0;
        for(j=0;j<arrSize;j++)
        {
            if(arr[i]==arr[j])
            {
                sun=sun+1;
            }
        }
        if(sun==arr[i])
        {
            if(arr[i]>sum)
            {
                sum=arr[i];
            }
        }
    }
    
    return sum;

}
