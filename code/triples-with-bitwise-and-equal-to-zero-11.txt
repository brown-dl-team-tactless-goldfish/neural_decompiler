class Solution {
public:
    int countTriplets(vector<int>& A) {
        // using map
        int n = A.size();
        unordered_map<int, int> m;
        for (int i = 0; i < n; i++)
            for (int j = 0; j < n; j++)
                m[A[i]&A[j]]++;
        int ans = 0;
        for (auto x: m){
            for (int i = 0; i < n; i++)
                if ((x.first & A[i])==0)
                    ans += x.second;
        }
        return ans;
    }
};
