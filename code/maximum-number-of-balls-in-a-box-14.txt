public class Solution {
    public int CountBalls(int lowLimit, int highLimit) {
        
        Dictionary<int,int> dic = new Dictionary<int,int>();
        
        int maxCnt = 0;
        for(int i = lowLimit; i <= highLimit; i++)
        {        
            int currNum = i;
            int boxIndex = 0;
            while(currNum > 0)
            {                
                int digit = currNum % 10;
                boxIndex += digit;
                currNum = currNum / 10;
            }
            
            if(dic.ContainsKey(boxIndex))
                dic[boxIndex]++;
            else
                dic.Add(boxIndex,1);
            
            maxCnt = Math.Max(maxCnt, dic[boxIndex]);
        }
        
        return maxCnt;
    }
}
