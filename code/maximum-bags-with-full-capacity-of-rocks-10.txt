int comp(int *a, int *b){
    return *a-*b;
}

int maximumBags(int* capacity, int capacitySize, int* rocks, int rocksSize, int additionalRocks){
    int ans=0;
    int *res = malloc(sizeof(int)*capacitySize);
    for(int i=0;i<capacitySize;i++){
        res[i]=capacity[i]-rocks[i];
    }
    qsort(res, capacitySize, sizeof(int), comp);
    for(int i=0;i<capacitySize;i++){
        if(res[i]==0){
            ans++;
        }
        else{
            if(additionalRocks>=res[i]){
                additionalRocks = additionalRocks - res[i];
                ans++;
            }
            else{
                break;
            }
        }
    }
    free(res);
    return ans;
}
