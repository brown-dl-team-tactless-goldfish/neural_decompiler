class Solution {
public:
    long long numberOfWeeks(vector<int>& milestones) {
        long long sum = 0, mx = 0;
        for (long long i : milestones) {
            sum += i;
            mx = max(mx, i);
        }
        if (sum - mx < mx)
            return 2 * (sum - mx) + 1;
        return sum;
    }
};
