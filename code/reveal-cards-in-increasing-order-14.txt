public class Solution {
    public int[] DeckRevealedIncreasing(int[] deck) {
        
        Array.Sort(deck);
        Queue<int> queue = new Queue<int>();
        for(int i = 0; i < deck.Length; i++)
            queue.Enqueue(i);
        
        int[] res = new int[deck.Length];
        for(int i = 0; i < deck.Length; i++)
        {
            res[queue.Dequeue()] = deck[i];
            if(queue.Count > 0)
                queue.Enqueue(queue.Dequeue());
        }
        
        return res;
    }
}
