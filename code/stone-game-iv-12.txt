class Solution {
    bool dp[100001];
public:
    bool winnerSquareGame(int n) {
        
        if(n==0)return 0;
        
        if(dp[n])return 1;
        
         bool f=0;
        for(int i=1;i*i<=n;i++)
        {
            f|=(!winnerSquareGame(n-i*i));   
        }
         return dp[n]=f;   
    }
};
