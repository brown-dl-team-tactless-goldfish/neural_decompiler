class Solution {
public:
    char findKthBit(int n, int k) {
        string ret = "0" ;
        for(int i = 2; i <= n && ret.size() < k; i++){
            string tmp ;
            for(int i = ret.size() - 1; i >=0 ; i-- ){
                tmp.push_back('1' - ret[i] + '0' ) ;
            }
            ret = ret + "1" + tmp ;
        }
        return ret[k-1] ;
    }
};
