class Solution {
public:
    int mat[10][10];
    int numEquivDominoPairs(vector<vector<int>>& dominoes) {
        int ans=0;
        for(int i=0;i<dominoes.size();i++){
            int mx = max(dominoes[i][0],dominoes[i][1]);
            int mn = min(dominoes[i][0],dominoes[i][1]);
            ans += mat[mn][mx]++;
        }
        return ans;
    }
};
