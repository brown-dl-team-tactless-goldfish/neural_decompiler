class Solution {
public:
    double angleClock(int hour, int minutes) 
    {
        double hour_angle = hour * 30.0f + minutes * 0.5f ;
        double minutes_angle = minutes * 6.0f;
        
        if(hour_angle > minutes_angle)
            return min(hour_angle - minutes_angle, 360 - hour_angle + minutes_angle);
        else
            return min(minutes_angle - hour_angle, 360 + hour_angle - minutes_angle);
        
    }
};
