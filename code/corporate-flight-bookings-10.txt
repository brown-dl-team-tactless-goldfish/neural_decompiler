    for(auto &x: bookings)
    {
        int from = x[0];
        int to = x[1];
        int val = x[2];
        
        res[from-1] += val;
        
        if(to < n)
        {
          res[to] -= val;
            
        }
    }
    
    for(int i=1; i<n; i++)
    {
        res[i] += res[i-1];
    }
    
    res.pop_back();
    
    return res;
}
