bool evaluateTree(struct TreeNode* root){
    if (root->val == 0){ //estoy en una hoja false
        return false;
    }
    else if (root->val == 1){ //estoy en la hoja true
        return true;
    }
    else{ //no estoy en una hoja
        if (root->val == 2){ //si es un OR
            return evaluateTree(root->left) || evaluateTree(root->right); //aplico la recusion a ambos lados
        }
        else{ //si es un AND
            return evaluateTree(root->left) && evaluateTree(root->right); //aplico la recusion a ambos lados
        }
    }
}
