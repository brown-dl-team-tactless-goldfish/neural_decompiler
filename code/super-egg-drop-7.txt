    int superEggDrop(int K, int N) {
        if(K==1) return N;
        if(N==1) return 1;
        const int M = 200;
        vector<vector<int>> dp(M, vector<int>(K+1,0));
        for(int j=1;j<K+1;j++) dp[1][j] = 1;
        for(int i=2;i<M;i++){
            for(int j=1;j<K+1;j++) dp[i][j] = 1 + dp[i-1][j-1] + dp[i-1][j];
            if(dp[i][K] >= N) return i;
        }       
    }
