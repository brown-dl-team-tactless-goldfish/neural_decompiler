class Solution {
public:
  vector<int> pivotArray(vector<int>& nums, int pivot) {
    vector<int>mini,maxi;
    int p = 0;
    for(auto n: nums)
      if(n < pivot)mini.push_back(n);
      else if(n > pivot)maxi.push_back(n);
      else p++;
    
    while(p--)mini.push_back(pivot);
    for(auto n: maxi) mini.push_back(n);
    
    return mini;
  }
};
