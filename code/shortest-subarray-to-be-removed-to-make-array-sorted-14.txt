    int findLengthOfShortestSubarray(vector<int>& arr) {
        int n = arr.size(), r = n-1;
        for (; r>0 && arr[r-1] <= arr[r]; r--){}
        int res = r;
        for (int l=0;l<r;l++){
            if (l>0 && arr[l] < arr[l-1]) break;
            while (r<n && arr[r] < arr[l]) r++;
            res = min(res, r-l-1);
        }
        return res;
    }
