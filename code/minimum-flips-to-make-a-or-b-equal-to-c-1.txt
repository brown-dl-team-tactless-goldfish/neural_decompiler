//注意运算符的优先级，不明确就加括号
int minFlips(int a, int b, int c){
    int count = 0;
    while(a>0 || b>0 || c>0){
        if((c&1)==0){
            if(((a&1)==1) && ((b&1)==1))
                count+=2;
            else if(((a&1)==1) || ((b&1)==1))
                count++;
        }else{
            if(((a&1)==0) && ((b&1)==0)){
                count++;
            }
        }
        a = a>>1;
        b = b>>1;
        c = c>>1;
    }
    return count;
}
