    while(res.size() < n)
    {
        vector<int> tmp;
        
        for(auto &x: res)
        {
            if(2*x - 1 <= n)
            {
                tmp.push_back(2*x-1);
            }
        }
        
        for(auto &x: res)
        {
            if(2*x  <= n)
            {
                tmp.push_back(2*x);
            }
        }
        
        res = tmp;
    }
    
    return res;
