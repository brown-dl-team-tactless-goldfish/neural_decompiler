    bool hasAllCodes(string s, int k) {
        int n=pow(2,k);
        unordered_set<string> v;
        for(int i=0;i<s.size();i++)
        {
            if(i+k-1>=s.size())break;
            string str=s.substr(i,k);
            v.insert(str);
        }
        if(v.size()==n)return true;
        return false;
    }
