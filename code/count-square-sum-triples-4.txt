class Solution:
    def countTriples(self, n: int) -> int:
        cnt = 0
        aSet = set()
        for i in range(1, n + 1):
            aSet.add(i**2)
            
        for a in range(1, n + 1):
            for b in range(a+1, n + 1):
                if a**2 + b**2 in aSet:
                    cnt += 1
            
        return cnt * 2
