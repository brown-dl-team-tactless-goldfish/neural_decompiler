compare(const void* xp, const void* yp){
    return *((int*)xp) - *((int*)yp);
}
int minimumDifference(int* nums, int numsSize, int k){
    qsort(nums, numsSize, sizeof(int), compare);
    int i = 0;
    int j = k - 1;
    int diff = nums[k - 1] - nums[0];
    while(j < numsSize){
        if((nums[j] - nums[i]) < diff){
            diff = nums[j] - nums[i];            
        }
        i++;
        j++;
    }
    return diff;
}
