class Solution {
    public int ladderLength(String beginWord, String endWord, List<String> wordList) {
        if(wordList.size() == 0) return -1;
        
        Set<String> words = new HashSet(wordList);
        Set<String> visited = new HashSet();

        Queue<String> q = new LinkedList();

        q.add(beginWord);

        int level = 0;

        while(!q.isEmpty()) {

            String s = q.poll();

            level++;

            visited.add(s);
            Set<String> neighbors = findNeighbors(s, words);
            for(String neighbor: neighbors) {
                if(neighbors.contains(endWord)) {
                    return level;
                }
                if (!visited.contains(neighbor)) {
                    q.add(neighbor);
                }
            }
        }

        return 0;
    }

    public Set<String> findNeighbors(String beginWord, Set<String> words) {
        Set<String> retSet = new HashSet();
        char[] chs = beginWord.toCharArray();
        for (int i = 0; i < chs.length; i++) {
            char temp = chs[i];
            for (char c = 'a'; c <= 'z'; c++) {
                chs[i] = c;
                String tryWord = new String(chs);
                if(words.contains(tryWord)) {
                    retSet.add(tryWord);
                }
            }
            chs[i] = temp;
        }
        return retSet;
    }
}
