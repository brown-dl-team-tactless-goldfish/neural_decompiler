char * breakPalindrome(char * palindrome){
    int length = strlen(palindrome);
    if(length == 1)return "";
    int index = 0;
    bool flag = true;
    char original_char = 'A';//random char
    while(index < length && flag){
        if(palindrome[index] != 'a'){
            original_char = palindrome[index];
            palindrome[index] = 'a';
            flag = false;
            break;
        }
        else index++;
    }
    if(flag){
        palindrome[length - 1] = 'b';
    }
    else if(length % 2 == 1 && index == length / 2){
        palindrome[index] = original_char;
        palindrome[length - 1] = 'b';
    }
    return palindrome;
}
