class Solution {
public:
    bool validTicTacToe(vector<string>& board) {
        
        int countX{0};
        int countO{0};
        
        for(auto&row:board)
        {
            for(auto&coord:row)
            {
                if(coord=='X')
                {
                    countX++;
                }
                if(coord=='O')
                {
                    countO++;
                }
            }
        }
        if(abs(countX-countO)>1 or countO>countX)
        {
            return false;
        }
    
        if(isWin(board,'X') and isWin(board,'O') )
        {
            return false;
        }
        if(isWin(board,'X') and countO==countX or isWin(board,'O') and countO<countX)
        {
            return false;
        }
         return true;
    }
    
    bool isWin(vector<string>& board,char player)
    {
        int count{0};
        for(int i=0;i<3;i++)//diagonal
        {
            if(board[i][i]==player)
            {
                count++;
            }
        }
        if(count==3)
        {
            return true;
        }
        count=0;
        for(int i=2,j=0;i>=0 and j<3;i-- and j++)//anti diagnonal
        {
            if(board[i][j]==player)
            {
                count++;
            }
        }
        if(count==3)
        {
            return true;
        }
        for(int i=0;i<3;i++)
        {
            count=0;
            for(int j=0;j<3;j++)//row by row
            {
                if(board[i][j]==player)
                {
                    count++;
                }
            }
            if(count==3)
            {
                return true;
            }
            count=0;
            for(int j=0;j<3;j++)//col by col
            {
                if(board[j][i]==player)
                {
                    count++;
                }
            }
            if(count==3)
            {
                return true;
            }
        }
        return false;
    }
};
