class Solution:
    def isAlienSorted(self, words: List[str], order: str) -> bool:
        
        mapped = []
        for w in words:
            tmp = []
            for c in w:
                tmp.append(order.index(c) - 0)
            mapped.append(tmp)
        # print ("mapped = ", mapped)
        
        n = len(mapped)
        for i in range(1, n):
            prv = mapped[i-1]
            cur = mapped[i]
            
            if (len(cur) < len(prv) and words[i-1].startswith(words[i]) ==True):
                return False
            
            for idx in range(min(len(prv), len(cur))):
                # print ("idx = ", idx)
                tmpPrv = prv[idx]
                tmpCur = cur[idx]
                # print ("tmpPrv = ", tmpPrv, " tmpCur = ", tmpCur)
                if (tmpPrv< tmpCur):
                    break
                elif(tmpPrv == tmpCur):
                    continue
                else:
                    # print ("False")
                    return False
                
        return True
