void rotate(int* nums, int numsSize, int k){
    
    k %= numsSize;
    if(numsSize < 2 || k==0)
        return;
    int i,n = k;
    int t[n];    
    for (i = n-1; i>=0; i--)
        t[i] = nums[numsSize-k+i];
    
    for(i = numsSize-n-1; i>=0; ) {
        int c = k;
        while(--c>=0 && i>=0) {
            nums[i+k] = nums[i];
            i--;
        }
    }
    for (i = 0; i<k; i++)        
        nums[i] = t[i];
    
}
