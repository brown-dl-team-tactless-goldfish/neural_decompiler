int repeatedNTimes(int* A, int ASize){
    int count=0;
    int num;
    for(int i=0; i<ASize; i++)
    {
        count=1;
        num=A[i];
        for(int j=i+1; j<ASize; j++)
        {
            if(A[j]==num)
                count++;
        }
        if(count==ASize/2)
            return num;
    }
    return 0;
}
