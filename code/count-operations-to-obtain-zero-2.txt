//helper function to count till it reaches 0
void helper_countOps(int nums1,int nums2, int* return_count){
    //termination when nums1 or nums2 reache 0
    if(nums1 == 0 || nums2 == 0){return;}
    //if nums1 > nums2
    if(nums1 >= nums2){
        (*return_count)++;   //increment return_count
        //nums1 = nums1 - nums2 and nums2 = nums2
        helper_countOps(nums1 - nums2,nums2,return_count);
    }
    else{
        (*return_count)++;   //increment return_count
        //nums1 = nums1 and nums2 = nums2 - nums1
        helper_countOps(nums1,nums2 - nums1,return_count);
    }
}



int countOperations(int num1, int num2){
int return_ops = 0;
//helper function to count the number of operations
helper_countOps(num1,num2,&return_ops);
return return_ops;
}
