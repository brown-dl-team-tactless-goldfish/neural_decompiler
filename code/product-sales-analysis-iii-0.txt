SELECT
b.product_id,
b.year as first_year,
b.quantity,
b.price
FROM
(
SELECT product_id,
year,
RANK() OVER (PARTITION BY product_id ORDER BY year) AS year_rank,
quantity,
price
FROM Sales
    ) b
    INNER JOIN Product c ON (c.product_id=b.product_id)
WHERE b.year_rank=1
GROUP BY
b.product_id,
b.year,
b.quantity,
b.price;
