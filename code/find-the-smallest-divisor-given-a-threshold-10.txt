    while(left < right)
    {
        int  mid = left + (right-left)/2;
        
        int sum = 0;
        
         for(auto& x: nums)
         {
             sum = sum +(x +  mid-1) / mid;
         }
        
        if(sum > threshold)
        {
            left = mid + 1; //divisor is too small increase left value
        }
        else
        {
            right = mid;
        }
    }
    
    return left;
}
