long long sumof(long long t)
{
    long long sum=0;
    while(t!=0)
    {
        sum = sum + t%10;
        t/=10;
    }
    return sum;
}




long long makeIntegerBeautiful(long long n, int target){
    long k=1;
    long long t=n;
    while(sumof(t)>target)
    {
        t=t/10+1;
        k=k*10;
    }
    return k*t-n;
   

}
