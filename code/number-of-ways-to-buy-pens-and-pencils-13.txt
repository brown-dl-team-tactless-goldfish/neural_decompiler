class Solution {
public:
    long long waysToBuyPensPencils(int total, int cost1, int cost2) {
        int original = total;
        long long numofpen = total/cost1;
        long long result = 0;
        for(long long i=0;i<=numofpen;i++){
            total = original - (i*cost1);
            long long numofpencil = total/cost2;
            result += 1 + numofpencil;
        }
        return result;
    }
};
