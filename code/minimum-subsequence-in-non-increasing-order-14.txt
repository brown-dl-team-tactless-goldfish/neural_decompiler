class Solution {
public:
    vector<int> minSubsequence(vector<int>& nums) {        
        int sum = accumulate(nums.begin(), nums.end(), 0);
        sort(nums.rbegin(), nums.rend());
        
        vector<int> result;
        int i = 0, curSum = 0;
        
        while (i < nums.size() and curSum <= sum) {
            curSum += nums[i];
            sum -= nums[i];
            
            result.push_back(nums[i++]);
        }   
        
        return result;
    }
};
