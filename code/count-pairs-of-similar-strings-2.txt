class Solution {
public:
    int similarPairs(vector<string>& w) {
      
int ans =0;
for(int i =0;i<w.size();i++){
set<char> s ;
set<char> s1;
for(int j  =0; j<w[i].size();j++){
    s.insert(w[i][j]);
}
for(int  k = i +1; k<w.size();k++){
    for(int l = 0;l<w[k].size();l++){
        s1.insert(w[k][l]);
    }
    if(s1 == s){
        ans ++;
    }
    s1.clear();
}
}
    return ans ;
    }
// do upvote
};
