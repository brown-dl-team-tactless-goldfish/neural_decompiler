int findCircleNum(int** isConnected, int isConnectedSize, int* isConnectedColSize){
    
    bool *visited = (bool *)malloc(sizeof(bool) * isConnectedSize);
    memset(visited,0,sizeof(bool) * isConnectedSize);
    int i;
    int c=0;
    for (i=0; i < isConnectedSize; i++)
    {
        if (visited[i] == 0)
        {
            visited[i] = 1;
            c++;
            dfs(i,isConnectedSize,visited,isConnected);
        }
    }
    return c;
}

void dfs(int src, int size, bool *visited, int** isConnected)
{
    int d;
    for (d=0; d < size; d++)
    {
        if (isConnected[src][d] == 1 && visited[d] != 1)
        {
            visited[d] = 1;
            dfs(d,size,visited,isConnected);
        }
    }
    return; 
}
