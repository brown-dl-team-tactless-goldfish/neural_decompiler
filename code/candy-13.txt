class Solution {
public:
    int candy(vector<int>& r) {
        int n=r.size();
        vector<int> check(n,1);
        for(int i=0;i<n-1;i++)
        {
            if(r[i+1]>r[i])
            {
                check[i+1]=check[i]+1;
            }
        }
        for(int i=n-1;i>0;i--)
        {
            if(r[i]<r[i-1] && check[i-1])
            {
                check[i-1]=check[i]+1;
            }
            
        }
        return accumulate(check.begin(),check.end(),0);
    }
};
