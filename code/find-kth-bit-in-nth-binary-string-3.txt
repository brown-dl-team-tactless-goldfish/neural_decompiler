char findKthBit(int n, int k) {
        if(n == 1 && k == 1)
            return '0';
        int length = pow(2, n);
        int mid = length/2;
        if(k < mid) {
            return findKthBit(n-1, k);
        }
        else if(k > mid) {
            char c = findKthBit(n-1, length-k);
            return c == '0'  ? '1' : '0';
        }
        else {
            return '1';
        }
    }
