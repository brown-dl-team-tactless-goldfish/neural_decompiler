bool increasingTriplet(int* nums, int numsSize) {
    if(numsSize<3){
        return false;
    }
    int a=nums[0];
    int b=INT_MAX;
    for(int i=1;i<numsSize;i++){
        if(nums[i]<=a){
            a=nums[i];
        }else if(nums[i]<=b){
            b=nums[i];
        }else{
            return true;
        }
    }
    return false;
}
