int subarraySum(int* nums, int numsSize, int k){
    
    // int *cum = (int*)malloc(sizeof(int)*(numsSize+1));
    int cum[numsSize+1];
    cum[0]=0;
    for(int i =1; i <=numsSize; ++i){
        cum[i]=cum[i-1]+nums[i-1];
    }
    int count = 0;
    for(int i =0; i <=numsSize; ++i){
        for(int j=0; j < i; ++j){
            if ((cum[i]-cum[j])==k){
                count+=1;
            }
        }
    }
   // free(cum);
    
    return count;
}
