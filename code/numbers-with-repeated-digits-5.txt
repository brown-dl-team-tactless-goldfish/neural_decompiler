    int ans = 1;
    
    for(int i=0; i<10; i++)
    {
        if(i == 0 && mask == 0)
        {
            continue;
        }
        
        if(mask & (1 << i))
        {
            continue;
        }
        
        ans += dfs(curr*10+i, mask+ (1<<i), n);
    }
    
    return ans;
   
}

int numDupDigitsAtMostN(int n)
{
    return n+1 - dfs(0,0,n);
}
