class Solution {
    public:
    bool reachingPoints(int sx, int sy, int tx, int ty) {
        if (tx < sx || ty<sy)
             return false;
        if( sx==tx && sy==ty)
             return true;
    
        if(tx>ty)
            tx= min( tx,tx-(tx/ty-sx/ty-1)*ty) ;
        else
            ty= min( ty,ty-(ty/tx-sy/tx-1)*tx) ;
   
        return reachingPoints(sx, sy, tx - ty, ty) || reachingPoints(sx, sy, tx, ty - tx);
    } 
};
