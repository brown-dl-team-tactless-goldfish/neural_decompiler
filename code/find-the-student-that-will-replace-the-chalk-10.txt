public class Solution {
    public int ChalkReplacer(int[] chalk, int k) {
        
        long sum = 0;
        foreach(var c in chalk)
            sum += c;
        
        k = (int)(k % sum);
        
        for(int i = 0; i < chalk.Length; i++)
        {
            k -= chalk[i];
            if(k < 0)
                return i;
        }
        
        return -1;
    }
}
