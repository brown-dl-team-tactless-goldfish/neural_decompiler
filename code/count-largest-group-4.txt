int countLargestGroup(int n){
    int count[37]={0},i,j,sum,max=0;
    for(i=1;i<=n;i++){
        j=i,sum=0;
        while(j){
            sum+=(j%10);
            j/=10;
        }
        count[sum]++;
        if(count[sum]>max)
            max=count[sum];
    }
    sum=0;
    for(i=0;i<27;i++)
        if(count[i]==max)
            sum++;
    return sum;
}
