class LUPrefix {
public:
    int next = 1;
    unordered_set<int> set;
    int size;
    LUPrefix(int n) {
        this->size = n;
    }
    
    void upload(int video) {
        set.insert(video);
    }
    
    int longest() {
        while(set.count(next) > 0) {
            next += 1;
        }

        return next-1;
    }
};

/**
 * Your LUPrefix object will be instantiated and called as such:
 * LUPrefix* obj = new LUPrefix(n);
 * obj->upload(video);
 * int param_2 = obj->longest();
 */
