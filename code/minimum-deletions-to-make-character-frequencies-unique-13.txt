class Solution {
public:
    int minDeletions(string s) {
        int cnt[26]{}, res = 0;
        for(auto c : s) cnt[c-'a']++;
        sort(begin(cnt), end(cnt));
        for(int i = 24, prev = cnt[25]; i >= 0 && cnt[i] > 0; i--) {
            prev = min(cnt[i], max(0, prev-1));
            res += cnt[i]-prev;
        }
        return res;
    }
};
