    public class Solution
    {
        public string OptimalDivision(int[] nums)
        {
            if (nums.Length == 1)
            {
                return nums.First().ToString();
            }

            string res = string.Join("/", nums.Select(n => n.ToString()).ToArray());
            if (nums.Length > 2)
            {
                res = res.Insert(res.IndexOf('/') + 1, "(");
                res += ")";
            }
            return res;
        }
    }
