int minMoves(int target, int maxDoubles){
    int val = target;
    int step = 0;
    while(val > 1 && maxDoubles > 0){
        if(val %2 ){
            val -= 1;
        }    
        else{
            val /= 2;
            maxDoubles--;
        }
        step++;
    }
    step += (val-1);
    return step;
}
