char * removeOuterParentheses(char * S){
    
    int n = 0, idx=0;
    char* out = malloc((strlen(S)+1) * sizeof(char));
   
    for(char* it=S; *it; ++it){
        if(*it == '('){
            ++n;
            if(n!=1) out[idx++]=*it;
        }
        
        if(*it == ')'){
            --n;
            if(n!=0) out[idx++]=*it;
        }
        
    }
    out[idx]='\0';
    return out;
}
