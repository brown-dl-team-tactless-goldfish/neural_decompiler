int closestCost(int* baseCosts, int baseCostsSize, int* toppingCosts, int toppingCostsSize, int target){
    int ret = INT_MAX ;
    int diff = INT_MAX ;
    
    for(int i = 0; i < baseCostsSize; i++){        
        for(int j = 0; j < pow(3, toppingCostsSize); j++){
            int cost = baseCosts[i] ;
            int val = j ;
            int k = 0 ;
            while(val){
                cost += toppingCosts[k] * (val%3 ) ;
                val /= 3 ;
                k++ ;
            }
            int new_diff = abs(cost - target) ;
            if(diff > new_diff){
                diff = new_diff ;
                ret = cost ;
            }
            else if(diff == new_diff && ret > cost)
                ret = cost ;
        }    
    }
    return ret ;
}
