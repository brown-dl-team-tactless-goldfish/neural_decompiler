int projectionArea(int** grid, int gridRowSize, int *gridColSizes) {
    (void)(gridColSizes);   // NxN is assumed
    
    int area = 0;
    for (int i = 0; i < gridRowSize; ++i) {
        int zx = 0;
        int yz = 0;
        for (int j = 0; j < gridRowSize; ++j) {
            area += (grid[i][j] > 0);
            zx = zx > grid[i][j] ? zx : grid[i][j];
            yz = yz > grid[j][i] ? yz : grid[j][i];
        }

        area += (zx + yz);
    }

    return area;    
}
