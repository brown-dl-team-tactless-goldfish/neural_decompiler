bool isLongPressedName(char* name, char* typed) {
    unsigned int i = 0, j = 0;
    int len_typed = strlen(typed);
    int len_name = strlen(name);
    while(j < len_typed)
    {
        if(name[i] == typed[j])
        {
            i++;
            j++;
        }
        else j++;
    }
    if(i == len_name) return true;
    else              return false;
}
