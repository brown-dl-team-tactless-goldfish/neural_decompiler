long long averaging(int* time, int timeSize, double search){
    long long return_val = 0;
    for(int ptr=0; ptr < timeSize; ptr++){
        return_val += search / time[ptr]; 
    }
    return return_val;
}

long long minimumTime(int* time, int timeSize, int totalTrips){
    long long left =0; 
    long long right = 1e14;
    long long mid;
    //binary search
    while(left < right){
        mid = (left+right)/2;
        if(averaging(time,timeSize,mid) >= totalTrips){
            right = mid;
        }else{
            left = mid + 1;
        }
    }
    if(averaging(time,timeSize,mid) == totalTrips){
        return left;
    }
    else{
        return right;
    } 
}
