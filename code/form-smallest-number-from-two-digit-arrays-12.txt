class Solution {
public:
    int minNumber(vector<int>& n1, vector<int>& n2) {
        sort(n1.begin(),n1.end());
        sort(n2.begin(),n2.end());
        set<int>s;
        for(int i=0;i<n1.size();i++){
            s.insert(n1[i]);
        }
        for(int i=0;i<n2.size();i++){
            if(s.count(n2[i])!=0){
                return n2[i];
            }
        }
        if(n1[0]<n2[0])
            return n1[0]*10+n2[0];
        else
            return n2[0]*10+n1[0];
    }
};
