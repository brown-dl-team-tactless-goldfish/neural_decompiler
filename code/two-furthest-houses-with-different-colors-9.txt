class Solution {
public:
    int maxDistance(vector<int>& c) {
        
        int mx=0;
        vector<int>v;
        for(int i=0;i<c.size();i++)
        {
            for(int j=0;j<c.size();j++)
            {
                if(c[i]!=c[j])
                    v.push_back(abs(i-j));
                    
            }
        }
        sort(v.begin(),v.end());
        return v[v.size()-1];
    }
};
