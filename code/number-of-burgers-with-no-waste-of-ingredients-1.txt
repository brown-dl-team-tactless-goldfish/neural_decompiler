class Solution:
    def numOfBurgers(self, t: int, c: int) -> List[int]:
        
        x = (t-2*c)//2
        
        y = (4*c-t)//2
        
        # print ("x = ", x, " y = ", y)
        
        if (x<0 or y<0):
            return []
        
        if (4*x+2*y != t or x+y!= c):
            return []
        return [x, y]
