    def splitPainting(self, segs):
        d = collections.defaultdict(int)
        for i, j, c in segs:
            d[i] += c
            d[j] -= c
        res = []
        i = 0
        for j in sorted(d):
            if d[i]:
                res.append([i, j, d[i]])
            d[j] += d[i]
            i = j
        return res
