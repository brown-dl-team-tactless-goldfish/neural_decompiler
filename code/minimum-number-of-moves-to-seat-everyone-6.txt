int minMovesToSeat(int* seats, int seatsSize, int* students, int studentsSize)

{

int i,stu[101]={0},sea[101]={0},j=1,count=0;
    
for(i=0;i<seatsSize;i++)
{
    sea[seats[i]]++;
}
for(i=0;i<studentsSize;i++)
{
    stu[students[i]]++;
}
i=1;    
while(i<101)
{
  if(stu[i]!=0)
  {
      while(j<101&&sea[j]==0)
      {
          j++;
      }
     count+=abs(j-i);
     stu[i]--;
     sea[j]--;
  }
  else
  {
      i++;
  }
}
    
  return count;
}
