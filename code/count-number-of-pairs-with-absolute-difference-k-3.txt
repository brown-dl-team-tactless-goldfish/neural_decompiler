int countKDifference(int* nums, int numsSize, int k)
{
    int diff[2];
    int count = 0;
    for (int i = 0; i < numsSize-1; i++)
    {
        diff[0] = nums[i] + k;
        diff[1] = nums[i] - k;
        for (int j = i + 1; j < numsSize; j++)
            if (nums[j] == diff[0] || nums[j] == diff[1])
                count++;
    }
    return count;
}
