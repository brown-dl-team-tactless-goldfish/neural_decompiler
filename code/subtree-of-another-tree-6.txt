bool isMatch(struct TreeNode* s, struct TreeNode* t)
{
    if (!s && !t)
        return true;
    else if (!s || !t)
        return false;
    
    if (s->val == t->val)
        return isMatch(s->left, t->left) & isMatch(s->right, t->right) & true;
    else
        return false;
}


bool isSubtree(struct TreeNode* s, struct TreeNode* t)
{
    bool self;
    bool left;
    bool right;

    if (!s && !t)
        return true;
    else if (!s || !t)
        return false;
    
    self = isMatch(s, t);
    left = isMatch(s->left, t);
    right = isMatch(s->right, t);
    
    if (self || left || right)
        return true;
    else
        return isSubtree(s->left, t) | isSubtree(s->right, t);
}
