int max(int a, int b){
    if(a > b){
        return a;
    }
    return b;
}

int maxProfit(int* prices, int pricesSize){
    int dp[pricesSize + 2][2];
    memset(dp, 0, sizeof dp);
    for(int i = pricesSize - 1; i >= 0; i--){
        
            dp[i][1] = max(dp[i + 1][1], dp[i+2][0] + prices[i]);
            dp[i][0] = max(dp[i + 1][0], dp[i+1][1] - prices[i]);
        
    }
    return dp[0][0];
}
