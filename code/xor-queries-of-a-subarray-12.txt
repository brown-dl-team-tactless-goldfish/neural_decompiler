    vector<int> prefix(n);
    
    vector<int> ans;
    
    prefix[0] = arr[0];
    
    for(int i=1; i<n; i++)
    {
        prefix[i] = prefix[i-1]^arr[i];
    }
    
    for(int i=0; i<queries.size(); i++)
    {
        int L = queries[i][0];
        int R = queries[i][1];
        
        if(L != 0)
        {
            t = prefix[L-1]^prefix[R];    
        }
        else
        {
            t = prefix[R];
        }
        
        ans.push_back(t);
    }
    
    return ans;
}
