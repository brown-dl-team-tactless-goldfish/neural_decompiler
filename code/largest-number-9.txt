string largestNumber(vector<int>& nums) {
    string ans="";
    vector<string> store;
    int n=nums.size();
    for(int i=0;i<n;i++){
        store.push_back(to_string(nums[i]));
    }
    
    sort(store.begin(),store.end(),[](string &x , string &y){
        string s1 = x+y;
        string s2 = y+x;
        return s1>s2;
    });
    
    if(store[0]=="0") return store[0];
    
    for(int i=0;i<n;i++){
        ans+=store[i];
    }
    return ans;
}
