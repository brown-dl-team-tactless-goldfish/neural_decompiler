char* convertToTitle(int n) {
    int* table = malloc(sizeof(int));
    int i = 0;
    int j = 0;
    
    while (n != 0)
    {
        table = realloc(table, sizeof(int)*(i+1));
        table[i] = n % 26 == 0 ? 26 : n % 26;
        n = (n-table[i++])/26;        
    }
    
    char* ans = malloc(sizeof(char)*(i));

    for(i--;i>=0;i--)
    {
        ans[j++] = table[i] + 'A' - 1;        
    }
    return ans;
}
