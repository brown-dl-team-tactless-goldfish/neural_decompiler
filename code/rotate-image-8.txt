void rotate(int** matrix, int matrixSize, int* matrixColSize){
	int c=0;
	int n=matrixSize-1;

	for( c=0 ; c <= n ; c++, n=--) {
		for( int i=c, j=n ; i<n; i++, j--) {
		   	int tmp = matrix[c][j];
			matrix[c][j] = matrix[i][c];
			matrix[i][c] = matrix[n][i];
			matrix[n][i] = matrix[j][n];
			matrix[j][n] = tmp;
     	}
	}
}
