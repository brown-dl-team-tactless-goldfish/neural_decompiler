class Solution {
public:
    int wateringPlants(vector<int>& plants, int capacity) {
        int n = plants.size();
        int c = capacity ;
        int st = 1;
        for(int i = 0;i < n-1; i++){
            capacity -= plants[i];
            if(capacity < plants[i+1]){
                st += i+1+i+1;
                capacity = c;
            }
            st++;
        }
        return st;
    }
};
