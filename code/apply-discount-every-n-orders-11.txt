class Cashier {
public:
    Cashier(int n_, int discount_, vector<int>& products_, vector<int>& prices_):
    customerNum(0), n(n_), discount(discount_) {
        size_t size = products_.size();
        for(int i = 0; i < size; ++i) productPrice[products_[i]] = prices_[i];
    }
    
    double getBill(vector<int> product, vector<int> amount) {
        ++customerNum;
        double cost = 0;
        size_t size = product.size();
        for(int i = 0; i < size; ++i) cost += productPrice[product[i]] * amount[i];
        if(customerNum == n) {
            cost = cost - cost * ((double) discount / 100);
            customerNum = 0;
        }
        return cost;
    }
    int customerNum;
    const int n;
    const int discount;
    unordered_map<int, int> productPrice;
};

 * Your Cashier object will be instantiated and called as such:
 * Cashier* obj = new Cashier(n, discount, products, prices);
 * double param_1 = obj->getBill(product,amount);
