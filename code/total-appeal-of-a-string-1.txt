long long appealSum(char * s){
    int len = strlen(s);
    long long ans = 0;
    int count[128] = {0};
    int last[128];
    int LEN = len;
    for (int i = 0 ; i < len ; i++){
        if (count[s[i]] == 0){
            ans += (i+1)*(LEN);   
        } else {
            ans += (i-last[s[i]])*(LEN);
        }
        last[s[i]] = i;
        count[s[i]]++;
        LEN--;
    }
    return ans;
}
