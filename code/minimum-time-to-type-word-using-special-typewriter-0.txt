int minTimeToType(char * word){
    int n = strlen(word);
    int ans = 0;
    int d = fabs(word[0] - 'a');
    if(d > 13)
        d = 26- d;
    ans += d + 1;
    for(int i = 1; i < n; i++){
        d = fabs(word[i] - word[i-1]);  
    if(d > 13)
        d = 26- d;
        ans += d + 1;
    }
    return ans;
}
