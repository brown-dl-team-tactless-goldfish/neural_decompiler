public class Solution {
    public int WordCount(string[] startWords, string[] targetWords) {
        HashSet<string> keys = new HashSet<string>();
        int res = 0;
        foreach(string startWord in startWords)
        {
            keys.Add(this.GetKey(startWord));
        }
        
        foreach(string targetWord in targetWords)
        {
            StringBuilder sb = new StringBuilder(targetWord);
            for(int i = 0; i < targetWord.Length; i++)
            {
                string newWord = sb.Remove(i, 1).ToString();
                string key = this.GetKey(newWord);
                if (keys.Contains(key))
                {
                    res++;
                    break;
                }
                
                sb = new StringBuilder(targetWord);
            }
        }
        
        return res;
    }
    
    private string GetKey(string str)
    {
        char[] chars = new char[26];
        foreach(char c in str)
        {
            chars[c-'a']++;
        }
        
        return new string(chars);
    }
}
