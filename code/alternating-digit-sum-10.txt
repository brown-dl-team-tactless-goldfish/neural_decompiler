class Solution {
public:
    int alternateDigitSum(int n) {
        int ans=0;
        string s;
        while(n){
            s+=to_string(n%10);
            n/=10;
        }
        reverse(s.begin(), s.end());
        for(int i=0; i<s.size(); i++){
            if(i&1){
                ans+=(-1)*(s[i]-'0');
            }
            else{
                ans+=(s[i]-'0');
            }
        }
        return ans;
    }
};
