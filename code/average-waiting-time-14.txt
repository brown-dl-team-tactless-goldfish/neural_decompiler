class Solution {
public:
    double averageWaitingTime(vector<vector<int>>& customers) {
        int n=customers.size();
        double ans=0, start=customers[0][0];
        
        for(int i=0; i<n; i++){
            start = max(start, (double)customers[i][0]);
            start += customers[i][1];
            ans += abs(start - customers[i][0]);
        }
        
        double result = (ans/n);
        return result;
    }
};
