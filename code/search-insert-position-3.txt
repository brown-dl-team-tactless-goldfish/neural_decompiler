int binarysearch(int *nums,int first,int end,int target)
{
    int mid=first+(end-first)/2;
      if(nums[mid]==target)
        return mid;
    if(first==end)
        return -2;
  
    if(nums[mid]<target)
        return binarysearch(nums,mid+1,end,target);
    if(nums[mid]>target)
        return binarysearch(nums,first,mid,target);
    return -2;
}
int searchInsert(int* nums, int numsSize, int target)
{
int a=binarysearch(nums,0,numsSize-1,target);
 if(a<0)
 {
     if(target>nums[numsSize-1])
     {
         a=numsSize;
         return a;
     }
     if(target<nums[0])
     {
         a=0;
         return a;
     }
     while(a<0)
     {
         target=target-1;
        a=binarysearch(nums,0,numsSize-1,target)+1;  
     }
 }
    
    
   return a; 
}
