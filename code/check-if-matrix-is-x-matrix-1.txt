bool checkXMatrix(int** grid, int gridSize, int* gridColSize){
    int sIdx = 0,  eIdx = gridSize - 1;
    for(int i = 0; i < gridSize; i++){
        for(int j = 0; j < gridSize; j++)
        {
            if(j == sIdx || j == eIdx){
                if(!grid[i][j])
                    return false;
            }
            else if(grid[i][j])
                return false;
        }
        sIdx++;
        eIdx--;
    }
    return true;
}
