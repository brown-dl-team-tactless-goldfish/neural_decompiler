/**
 * Note: The returned array must be malloced, assume caller calls free().
 */
int* decompressRLElist(int* nums, int numsSize, int* returnSize){
int i, j;
int* ret_array;
int ptr=0;
*returnSize = 0;
for (i=0; i<numsSize; i+=2) *returnSize += nums[i];
ret_array = malloc(*returnSize * sizeof(int));
for (i=0; i<numsSize; i+=2)
    for (j=0; j<nums[i]; j++) ret_array[ptr++]=nums[i+1];
return ret_array;
}
