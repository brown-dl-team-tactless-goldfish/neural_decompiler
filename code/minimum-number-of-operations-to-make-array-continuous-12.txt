class Solution {
public:
    int minOperations(vector<int>& nums) {
        int n = nums.size(), i = 0, ans = INT_MIN;
        sort(nums.begin(), nums.end());
        nums.erase(unique(nums.begin(), nums.end()), nums.end());
        for(int j = 0; j < nums.size(); ++j){
            while(i <= j && (nums[j] - nums[i] + 1) > n) i++;
            ans = max(ans, j - i + 1);
        }
        // cout<<ans<<endl;
        return n - ans;
    }
};
