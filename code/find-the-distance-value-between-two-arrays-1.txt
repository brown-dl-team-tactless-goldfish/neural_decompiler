int findTheDistanceValue(int* arr1, int s1, int* arr2, int s2, int d){
    int c = 0, k = 0;
    
    for (int i=0; i<s1; i++){
        c = 0;
        for (int j=0; j<s2; j++){
            if (abs (arr1[i]-arr2[j]) > d){
                c++;
            }
        }
        if (c == s2){
            k++;
        }
    }
    return k;
}
