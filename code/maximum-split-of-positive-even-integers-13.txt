vector<long long> maximumEvenSplit(long long finalSum) {
        vector<long long> ans;
        if (fs&1) return ans;
        
        long even = 2;
        while (finalSum){
            if (finalSum == even || finalSum-even >= even+2){
                ans.push_back(even);
                finalSum -= even;
            }else{
                ans.push_back(finalSum);
                break;
            }
            even += 2;
        }
        
        return ans;
    }
