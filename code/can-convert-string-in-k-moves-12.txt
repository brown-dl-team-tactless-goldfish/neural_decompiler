class Solution {
public:
    bool canConvertString(string s, string t, int k) {
        if (s.length() != t.length()) return false;
        vector<int> v(26);
        for (int i = 1; i < 26; ++i) {
            v[i] = i;
        }
        for (int i = 0; i < t.size(); ++i) {
            const int x = (t[i] + 26 - s[i]) % 26;
            if (x) {
                if (v[x] > k) return false;
                v[x] += 26;
            }
        }
        return true;
    }
};
