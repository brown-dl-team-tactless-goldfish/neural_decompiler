class Solution {
public:
    int jump(vector<int>& nums) {
        int current=0;
        int furthest=0;
        int jump=0;
        for(int i=0;i<nums.size()-1;i++){
            furthest=max(furthest,nums[i]+i);
            if(current==i){
                current=furthest;
                jump++;
            }
        }return jump;
        
    }
};
