

int countGoodRectangles(int** rectangles, int rectanglesSize, int* rectanglesColSize){
    
    int min=0;
    int maxInMinSide =0;
    int count=0;
    
    for(int i =0 ;i<rectanglesSize;i++){
        min=0;
        min =rectangles[i][0];
        if(rectangles[i][1]<min){
            min=rectangles[i][1];
        }
        if(min>maxInMinSide){
             maxInMinSide= min;
             count=0;
        }
        if(min==maxInMinSide){
            count++;
        }
    }
    return count;
}
