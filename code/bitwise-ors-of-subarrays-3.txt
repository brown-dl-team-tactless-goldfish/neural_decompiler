public class Solution {
    public int SubarrayBitwiseORs(int[] arr) {
        HashSet<int> ans = new (), current = new ();
        
        for (int i = 0; i < arr.Length; i++) {
            HashSet<int> newCurrent = new ();
            newCurrent.Add(arr[i]);
            ans.Add(arr[i]);

            foreach (int c in current) {
                int r = arr[i] | c;
                newCurrent.Add(r);
                ans.Add(r);
            }
            current = newCurrent;
        }
        
        return ans.Count;
    }
}
