class Solution {
public:
    vector<long long> maximumEvenSplit(long long f) {
        vector<long long> res;
        if (f%2 == 1) return res;
        long long curr = 2;
        long long totalSum = 0;
        while (totalSum + curr <= f) {
            res.push_back(curr);
            totalSum += curr;
            curr += 2;
        }
        totalSum -= res.back();
        res.pop_back();
        res.push_back(f - totalSum);
        return res;
    }
};
