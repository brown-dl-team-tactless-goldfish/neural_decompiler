int numOfSubarrays(int* arr, int arrSize, int k, int threshold){
    int target = k * threshold;
    int sum = 0;
    for(int i = 0; i < k; i++){
        sum += arr[i];
    }
    int ans = 0;
    if(sum >= target)
        ans++;
    for(int i = k; i < arrSize; i++){
        sum += arr[i] - arr[i-k];
        if(sum >= target)
            ans++;
    }
    return ans;
}
