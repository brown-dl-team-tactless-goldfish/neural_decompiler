public class Solution 
{
    public int MinimumPartition(string s, int k) 
    {
        int counter = 1;
        long curSubstring = 0;
        foreach (char c in s)
        {
            int num = int.Parse(c.ToString());
            if (num > k)
                return -1;
            if (curSubstring * 10 + num <= k)
                curSubstring = curSubstring * 10 + num;
            else
            {
                counter++;
                curSubstring = num;
            }
        }
        return counter;
    }
}
