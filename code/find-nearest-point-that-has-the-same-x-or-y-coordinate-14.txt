int nearestValidPoint(int x, int y, int** points, int pointsSize, int* pointsColSize){
    
    int smallestManhattan = INT_MAX;
    int smallestIndex = -1;
    
    for (int i = 0; i < pointsSize; i++){
        int a = points[i][0];
        int b = points[i][1];
        
        if (x == a || y == b) {
            int md = abs(x - a) + abs(y - b);
            
            if (md < smallestManhattan) {
                smallestManhattan = md;
                smallestIndex = i;
            }
        }
    }
    
    return smallestIndex;
}
