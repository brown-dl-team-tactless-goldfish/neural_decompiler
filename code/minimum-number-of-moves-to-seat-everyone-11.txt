// OJ: https://leetcode.com/problems/minimum-number-of-moves-to-seat-everyone/
// Author: github.com/lzl124631x
// Time: O(NlogN)
// Space: O(1)
class Solution {
public:
    int minMovesToSeat(vector<int>& A, vector<int>& B) {
        sort(begin(A), end(A));
        sort(begin(B), end(B));
        int ans = 0;
        for (int i = 0; i < A.size(); ++i) {
            ans += abs(A[i] - B[i]);
        }
        return ans;
    }
};
