class Solution {
public:
    long long taskSchedulerII(vector<int>& tasks, int space) {
        unordered_map<long long, long long> m;
        
        long long d = 1, i = 0;
        while(i < tasks.size()) {
            if(m.find(tasks[i]) == m.end()) m[tasks[i]] = d++, i++;
            else {
                if(d - m[tasks[i]] - 1 < space) d += space - (d - m[tasks[i]] - 1);
                else m[tasks[i]] = d++, i++;
                
            }
        }
        return d - 1;
    }
};
