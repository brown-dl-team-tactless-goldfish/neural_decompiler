

int integerReplacement(long long int n){
    int x=0;
    while(n!=1){
        if (n%2==0){
            n=n/2;
        }
        else if (n==3){
            n=n-1;
        }
        else if((n&3)==3){
            n=n+1;
        }
        else if ((n&3)==1){
            n=n-1;
        }
        x+=1;
    }
    return x;

}
