int cmp(const void* a, const void* b){
    return *(int*)a - *(int*)b;
}
int minDifference(int* nums, int numsSize){
    int n = numsSize;
    if(n <= 3)
        return 0;
    qsort(nums, n, sizeof(int), cmp);
    int a , b , c, d;
    a = nums[n-4] - nums[0];
    b = nums[n-3] - nums[1];
    c = nums[n-2] - nums[2];
    d = nums[n-1] - nums[3];
    return fmin( fmin(a,b), fmin(c, d) );
}
