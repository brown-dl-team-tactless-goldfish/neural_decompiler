#include <string.h>

int prefixCount(char ** words, int wordsSize, char * pref){
    int ans = 0, cmp_res = 0;
    
    for (int i = 0; i < wordsSize; i++) {
        //compare the first n letters in words[i] with pref (n == the length of the prefix)
        cmp_res = strncmp(pref, words[i], sizeof(char) * strlen(pref));
        if (cmp_res == 0) {
            ans++;
        }
    }
    
    return ans;
}
