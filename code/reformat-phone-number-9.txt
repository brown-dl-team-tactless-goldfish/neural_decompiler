string reformatNumber(string number) {
	string nos = "", ans = "";
	for(char c: number){
		if(c >= '0' && c <= '9')
			nos += c;
	}
	int n = nos.size();
	for(int i=0; i < n; ){
		if(n-i > 4){
			ans += nos.substr(i,3);
			i += 3;
		}else{
			if(n-i == 4){
				ans += nos.substr(i,2) + "-" + nos.substr(i+2);
			}else{
				ans += nos.substr(i);
			}
			break;

		}
		if(i < n)
			ans += "-";
	}
	return ans;
}
