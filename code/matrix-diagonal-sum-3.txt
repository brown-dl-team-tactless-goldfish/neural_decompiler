int diagonalSum(int** mat, int r, int* c){
    int ret=0;
    //primary diagonal
    for(int i=0,j=0;i<r && j<(*c);i++,j++){
        ret+=mat[i][j];
    }
    //secondary diagonal
    for(int i=0,j=(*c)-1;i<r && j>=0;i++,j--){
        ret+=mat[i][j];
    }
    
    if(r==(*c) && r%2){
        ret-=mat[r/2][(*c)/2];
    }
    return ret;
}
