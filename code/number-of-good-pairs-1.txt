int numIdenticalPairs(int* nums, int numsSize) {
    int max = nums[0];
    for(int i = 0 ; i < numsSize; i++) {
        max = (nums[i] > max) ? nums[i] : max;
    }
    max++;
    int* arr = (int*) calloc(max, sizeof(int));
    for(int i = 0; i < numsSize; i++) {
        arr[nums[i]]++;
    }
    int ans = 0;
    for(int i = 0; i < max; i++) {
        if(arr[i] > 1) ans += (arr[i] * (arr[i] - 1) / 2);
    }
    free(arr);
    return ans;
}
