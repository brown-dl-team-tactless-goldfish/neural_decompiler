public class Solution 
{
    public int[] RowAndMaximumOnes(int[][] mat) 
    {
        int maxOnes = 0;
        int rowIndex = 0;
        for (int i = 0; i < mat.Length; i++)
        {
            int countOnes = mat[i].Count(n => n==1);
            if (countOnes > maxOnes)
            {
                maxOnes = countOnes;
                rowIndex = i;
            }
        }

        return new int[] { rowIndex, maxOnes };
    }
}
