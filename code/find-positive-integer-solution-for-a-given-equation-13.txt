std::vector<std::vector<int>> 
    findSolution(CustomFunction& cf, int z) 
{
    constexpr int dim = 1000;
    std::vector<std::vector<int>> res;
    
    for(int x = 1; x <= dim; x++)
    {
        for(int y = 1; y <= dim; y++)
        {
            if(cf.f(x, y) == z)
            {
                res.push_back({x, y});
            }    
        }
    }
    
    return res;
}
