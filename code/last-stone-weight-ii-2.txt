int lastStoneWeightII(vector<int>& arr) {
        int n = arr.size(), total = 0, sum = 0;
        
        for(int& a : arr) total += a;
        
        sum = total/2;
        
        vector<int>dp(sum + 1);
        
        for(int a : arr) {
            for(int c = sum; c >= a; c--) {
                dp[c] = max(dp[c], a + dp[c - a]);
            }
        }
        
        return total - 2*dp[sum];
    }
