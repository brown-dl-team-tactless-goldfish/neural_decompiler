int maximumValue(char ** strs, int strsSize){
  int res = 0;
  int now = 0;
  for (size_t i = 0; i < strsSize; i++)
  {
    for (size_t j = 0; j < strlen(strs[i]); j++)
    {
      if (strs[i][j] < 48 || strs[i][j] > 57)
      {
        now = strlen(strs[i]);
        break;
      }
      else
      {
        now = now * 10 + strs[i][j] - 48;
      }
    }
    if (res < now)
    {
      res = now;
    }
  }
  return res;
}
