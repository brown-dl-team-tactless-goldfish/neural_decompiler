/**
 * Note: The returned array must be malloced, assume caller calls free().
 */
int* constructRectangle(int area, int* returnSize){
    int max = 0;
    int i;
    int l = area;
    int w = 1;
    int* ret = (int* ) malloc(sizeof(int)*2);
    *returnSize = 2;
    
    for(i=1;i<area/4+2;i++){
        if(area%i == 0){
            if(abs(l-w)>abs(area/i -i)){
                w = i;
                l = area/w;
            }
        }
    }
    
    ret[0] = l;
    ret[1] = w;
    return ret;
}
