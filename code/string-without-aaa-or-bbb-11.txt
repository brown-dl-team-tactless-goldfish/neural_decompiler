class Solution {
public:
    string strWithout3a3b(int A, int B) {
        string temp = "";
        while(A||B){
            if(A > B+1 && B){
                temp+="aab";
                A-=2,B--;
            }
            else if(B > A+1 && A){
                temp+="bba";
                A--,B-=2;
            }
            else{
                if(A) temp+='a',A--;
                if(B) temp+='b',B--;
            }
        }
        return temp;
    }
};
