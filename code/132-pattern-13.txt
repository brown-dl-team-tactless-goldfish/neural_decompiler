    public bool Find132pattern(int[] nums) {
        
        return nums.Select((s, i) => new { s, i }).Select(
                x => nums.Select((s, i) => new { s, i }).Select(
                y => nums.Select((s, i) => new { s, i }).Count(z => x.i < y.i && y.i < z.i && x.s < z.s && z.s < y.s)
                ).Sum()
                ).Sum() > 0;
    }
