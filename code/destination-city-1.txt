char * destCity(char *** paths, int pathsSize, int* pathsColSize){
    int i,j, flag=0;
    for(i = 0; i < pathsSize; i++){
        flag = 0;
        for(j = 0; j < pathsSize; j++){
            if(i == j) continue;
            if(strcmp(paths[i][1], paths[j][0])==0){
                flag=1;
                break;
            }
        }
        if(flag==0) break;
    }
    return paths[i][1];
}
