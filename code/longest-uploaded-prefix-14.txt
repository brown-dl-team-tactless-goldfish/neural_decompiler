class LUPrefix {
    set<int> Set;
public:
    LUPrefix(int n) {
        n++;
        for(int i = 1; i <= n; i++) {
            Set.insert(i);
        }
    }
    
    void upload(int video) {
        Set.erase(video);
    }
    
    int longest() {
        return *Set.begin() - 1;
    }
};

/**
 * Your LUPrefix object will be instantiated and called as such:
 * LUPrefix* obj = new LUPrefix(n);
 * obj->upload(video);
 * int param_2 = obj->longest();
 */
