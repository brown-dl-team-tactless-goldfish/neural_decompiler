class Solution:
    def maximumTastiness(self, price: List[int], k: int) -> int:
        price.sort()
        lo, hi = 0, price[-1] - price[0]
        while lo < hi: 
            mid = lo + hi +1 >> 1
            val = price[0]
            cnt = 0 
            for x in price: 
                if x >= mid + val: 
                    cnt += 1
                    val = x
            if cnt >= k-1: lo = mid 
            else: hi = mid - 1
        return lo
