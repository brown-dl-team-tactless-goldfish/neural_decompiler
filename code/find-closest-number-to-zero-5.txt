bool isNumber(int* nums, int numsSize, int find){
    for (int i = 0; i < numsSize; i++){
        if (nums[i] == find) return true;
    }
    return false;
}

int findClosestNumber(int* nums, int numsSize){
    int closestNum = INT_MAX, minDist = INT_MAX;
    for (int i = 0; i < numsSize; i++){
        if (abs(nums[i]) < minDist){
            minDist = abs(nums[i]);
            closestNum = nums[i];
        }
    }
    return (closestNum < 0 && isNumber(nums, numsSize, -closestNum)) ? -closestNum : closestNum; 
}
