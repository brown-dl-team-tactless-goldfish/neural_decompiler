int encoded_element = 0, temp_element = 0;          //VARIABLES TO STORE XORED VALUES
int *arr = (int *)malloc((prefSize)*sizeof(int));   //MALLOC ARR OF SIZE SAME AS PREF[]
arr[0] = pref[0];                                   //FIRST ELEMENT OF ARR[]

for(int i=0; i<prefSize-1; i++){
    
    encoded_element = encoded_element^pref[i+1];    //GET THE PREF[] ELEMENT
    temp_element = temp_element^arr[i];             //STORE ALL THE PREVIOUS XORS OF ARR[]
    encoded_element = encoded_element^temp_element; //FIND THE ELEMENT OF ARR[]

    arr[i+1] = encoded_element;
    encoded_element = 0;                            //RESET
    
}

*returnSize = prefSize;

return arr;
