int max(int a,int b)
{
    return a > b ? a : b;
}
int largestAltitude(int* gain, int gainSize){
    int dp[gainSize + 1];
    int maximum = 0,current = 0;
    dp[0] = 0;
    for(int i = 0;i < gainSize; i++)
    {
        current += gain[i];
        dp[i+1] = current;
        maximum = max(maximum,current);
    }
    return maximum;
}
