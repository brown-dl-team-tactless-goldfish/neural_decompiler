/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     struct ListNode *next;
 * };
 */
void deleteNode(struct ListNode* p){
    struct ListNode* tmp;
    tmp = p->next;
    p->next = p->next->next;
    free(tmp);
}

struct ListNode* mergeNodes(struct ListNode* head){
    struct ListNode* p;
    p=head;
    while(p){
        if(p->next->val == 0){
            deleteNode(p);
            p = p->next;
        }
        else{
            p->val += p->next->val;
            deleteNode(p);
        }
    }
    return head;
