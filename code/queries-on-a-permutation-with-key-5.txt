int* processQueries(int* queries, int queriesSize, int m, int* returnSize){
    int n = queriesSize;
    *returnSize = n;
    int*result = (int*)malloc(n*sizeof(int));
    int*index = (int*)malloc((m+1)*sizeof(int));
    for(int i=0;i<=m;i++){index[i]=i-1;}
    for(int i=0;i<n;i++){
        result[i] = index[queries[i]];
        for(int j=0;j<=m;j++)
            index[j] += (result[i]>index[j])? 1 : 0;
        index[queries[i]]=0;
    }
    return result;
}
