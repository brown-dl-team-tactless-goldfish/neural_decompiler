public class CustomStack 
{
    Stack<int> stack;
    int maxSize;

    public CustomStack(int maxSize) 
    {
        stack = new Stack<int>();
        this.maxSize = maxSize;
    }
    
    public void Push(int x) 
    {
        if(stack.Count < maxSize)
            stack.Push(x);
    }
    
    public int Pop() 
    {
        if(stack.Count == 0)
            return -1;
        return stack.Pop();
    }
    
    public void Increment(int k, int val) 
    {
        Stack<int> temp = new Stack<int>();
        while(stack.Count != 0)
        {
            int pop = stack.Pop();
            int num = stack.Count < k ? pop + val : pop;
            temp.Push(num);
        }
        
        while(temp.Count != 0)
        {
            stack.Push(temp.Pop());
        }
    }
}
