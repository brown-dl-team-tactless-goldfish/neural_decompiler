class Solution {
public:
    int longestSubarray(vector<int>& nums, int limit) {
        int res = 0, size = nums.size();
        multiset<int> numSet;
        for (int i = 0, j = 0; j < size; ++j)
        {
            numSet.insert(nums[j]);
            while (i < j && *(--numSet.end()) - *numSet.begin() > limit)
            {
                numSet.erase(numSet.lower_bound(nums[i++]));
            }

            res = max(res, j - i + 1);
        }
        
        return res;
    }
};
