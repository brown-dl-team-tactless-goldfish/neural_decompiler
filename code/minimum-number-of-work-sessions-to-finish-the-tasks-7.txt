public class Solution {
	private bool CanAssignWork(int[] tasks, int[] availability, int taskId){
		if(taskId == tasks.Length)
			return true;

		for(int i=0;i<availability.Length;i++){
			if(taskId<i)
				continue;
			if(availability[i]>=tasks[taskId]){
				availability[i] -= tasks[taskId];
				if(CanAssignWork(tasks,availability,taskId+1))
					return true;
				availability[i] += tasks[taskId];
			}
		}

		return false;
	}
	public int MinSessions(int[] tasks, int sessionTime) {
		for(int i=1;i<=tasks.Length;i++){
			int[] availability = new int[i];
			for(int j=0;j<i;j++){
				availability[j]=sessionTime;
			}
			if(CanAssignWork(tasks,availability,0))
				return i;
		}
		return tasks.Length;
	}
}
