int findTilt(struct TreeNode* root) {
	int result=0;
	int lnodeTilt=0,rnodeTilt=0,lnodeVal=0,rnodeVal=0;
	if(!root) return result;
	else{
	if(root->left) lnodeTilt=findTilt(root->left),lnodeVal=root->left->val,root->val+=lnodeVal;
	if(root->right) rnodeTilt=findTilt(root->right),rnodeVal=root->right->val,root->val+=rnodeVal;
	result=lnodeTilt+rnodeTilt+abs;
	return result;
}
