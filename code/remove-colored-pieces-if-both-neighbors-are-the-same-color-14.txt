class Solution {
public:
    bool winnerOfGame(string C) {
        int cntA = 0, cntB = 0;
        for(int i = 0, cnt = 0; i <= C.size(); i++) {
            if(i == C.size() || i > 0 && C[i] != C[i-1]) {
                if(cnt > 2 && C[i-1] == 'A') cntA += cnt-2;
                if(cnt > 2 && C[i-1] == 'B') cntB += cnt-2;
                cnt = 0;
            }
            cnt++;
        }
        return cntA > cntB;
    }
};
