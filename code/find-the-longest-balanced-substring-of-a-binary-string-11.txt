class Solution {
public:
    int findTheLongestBalancedSubstring(string s) {
        int cnt0=0,cnt1=0,mx=0;
        int n=s.length();
        for(int i=0;i<n;i++){
            if(i!=0 && s[i-1]=='1' && s[i]=='0'){
                cnt0=1;
                cnt1=0;
            }
            else{
                if(s[i]=='0') cnt0++;
                else cnt1++;
                mx=max(mx,min(cnt1,cnt0)*2);
            }
        }
        return mx;
    }
};
