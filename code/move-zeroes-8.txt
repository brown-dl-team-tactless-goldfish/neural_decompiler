void moveZeroes(int* nums, int numsSize){
int j = 0;
        // move all the nonzero elements advance
        for (int i = 0; i < numsSize; i++) {
            if (nums[i] != 0) {
                nums[j++] = nums[i];
            }
        }
        for (;j < numsSize; j++) {
            nums[j] = 0;
        }
}
