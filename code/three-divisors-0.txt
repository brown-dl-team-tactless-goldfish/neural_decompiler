bool isThree(int n){
    int cn = 2;
    for(int i = 2; i <= n/2; i++){
        if(n % i == 0){
            cn++;
            if(i != n/i)
                return false;
            if(cn > 3)
                return false;
        }
    }
    if(cn == 3)
        return true;
    else
        return false;

}
