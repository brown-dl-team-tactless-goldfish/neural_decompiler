int cmp(const void* a, const void* b){
    return *(int*)a - *(int*)b ;
}
    
int getMaximumConsecutive(int* coins, int coinsSize){
    qsort(coins, coinsSize, sizeof(int), cmp) ;
    int CurMax = 0 ;
    for(int i = 0; i < coinsSize; i++){
        if((CurMax + 1) < coins[i])
            break ;
        CurMax += coins[i] ;
    }
    return CurMax + 1 ;
}
