class Solution {
public:
    int maxSumAfterPartitioning(vector<int>& arr, int k) {
        int n = arr.size();
        vector<int> dp(n + 1);
        for(int i = 1; i <= n; i++) {
            int max_value = arr[i - 1];
            for(int j = i; j > 0 && i - j + 1 <= k; j--) {
                max_value = max(max_value, arr[j - 1]);
                dp[i] = max(dp[i], dp[j - 1] + (i - j + 1) * max_value);
            }
        }
        return dp[n];
    }
};
