class Solution {
public:
    long long appealSum(string s) {
        vector<int> last_index(26, -1);
        vector<long long> dp(s.length(), 1);
        
        last_index[s[0]-'a'] = 0;
        long long ans = 1;
        for(int i = 1; i < s.length(); i++) {
            dp[i] = dp[i-1] + i - last_index[s[i]-'a'];
            last_index[s[i]-'a'] = i;
            ans += dp[i];
        }
        
        return ans;
    }
};
