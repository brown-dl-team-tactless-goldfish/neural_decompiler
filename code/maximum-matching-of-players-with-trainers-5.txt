    public int MatchPlayersAndTrainers(int[] players, int[] trainers) 
    {
        Array.Sort(players);
        Array.Sort(trainers);
        int result = 0;
        int playerIndex = 0;
        for (int i = 0; i < trainers.Length; i++)
        {
            if (players[playerIndex] <= trainers[i])
            {
                result++;
                playerIndex++;
                if (playerIndex >= players.Length)
                    break;
            }
        }
        return result;
    }
