bool checkPalindrome(char* s, int n){
    if(n <= 1)
        return true;
    for(int i = 0; i < n/2; i++){
        if(s[i] != s[n-1-i])
            return false;
    }
    return true;
}
bool validPalindrome(char * s){
    int n = strlen(s);
    for(int i = 0; i < n/2; i++){
        if(s[i] != s[n-1-i]){
            return checkPalindrome(&s[i], n-1-2*(i)) || checkPalindrome(&s[i+1], n-1-2*(i)) ;
        }    
    }
    return true;
}
