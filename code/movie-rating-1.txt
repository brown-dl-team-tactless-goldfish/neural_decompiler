with x as
    (select u.name, m.title,  mr.rating, mr.created_at 
    from Movie_Rating as mr
    left join Movies  as m on mr.movie_id = m.movie_id
    left join Users  as u on mr.user_id = u.user_id),
    
    x1 as 
    (select name, count(*) as c  from x 
     group by name
    order by c desc, name 
    limit 1),


    x2 as
    (select  title  ,  created_at
     from x
     where month(created_at) = 2 and year(created_at) = 2020
     group by title
    order by avg(rating) desc , title 
    limit 1)

select name as results from x1
union
select title as results from x2
