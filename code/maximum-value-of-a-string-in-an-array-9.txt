public class Solution {
    public int MaximumValue(string[] strs) {
        int maxVal = 0, val = 0;
        
        foreach(var str in strs)
        {
            val = ValOfS(str);
            maxVal = Math.Max(maxVal, val);
        }
        
        return maxVal;
    }
    
    private int ValOfS(string s)
    {
        foreach(var c in s)
        {
            if(Char.IsLetter(c))
                return s.Length;
        }
        return Convert.ToInt32(s);
    }
}
