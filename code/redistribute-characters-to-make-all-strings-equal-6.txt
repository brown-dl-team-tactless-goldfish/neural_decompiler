class Solution {
public:
    bool makeEqual(vector<string>& w) {
         int ch[26]={0};
         int n=w.size();
         for(int i=0; i<n; i++) {
              for(char c : w[i]) ch[c-'a']++;
         }
         for(int i=0; i<26; i++) {
               if(ch[i]%n) return false;
         }
        return true;
    }
};
