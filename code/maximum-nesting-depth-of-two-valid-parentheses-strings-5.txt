 vector<int> maxDepthAfterSplit(string seq) {
        vector<int> res(seq.length());
        
        int dep = 0;
        for (int i = 0; i < seq.length(); ++i){
            const char c = seq[i];
            if (c == '('){
                dep ++;
                res[i] = dep % 2;
            }else {
                res[i] = dep % 2;
                dep --;
            }
        }
        
        return res;
    }
