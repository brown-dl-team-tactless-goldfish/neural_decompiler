char * firstPalindrome(char ** words, int wordsSize){

for(int i=0;i<wordsSize;i++)
{
    
    if(strlen(words[i])==1)
    {
        return words[i];
    }
    
    
    for(int j=0;j<strlen(words[i])/2;j++)
    {
        if(words[i][j]!=words[i][strlen(words[i])-1-j])
        {
            break;
        }  
        if(j==strlen(words[i])/2-1)
        {
            return words[i];
        }
    }

}
    
    
    
return "";

}
