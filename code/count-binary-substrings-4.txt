int countBinarySubstrings(char* s)
{
    char standard = s[0];
    int i=0, pre_cnt=0, cur_cnt=1;
    int answer = 0;
    int s_len = strlen(s);
    
    for(i=1; i<s_len; i++)
    {
        if(standard == s[i]) cur_cnt++;
        else
        {
            standard = s[i];
            pre_cnt = cur_cnt;
            cur_cnt = 1;
        }
        
        if(pre_cnt >= cur_cnt)answer++;
    }
    return answer;
}
