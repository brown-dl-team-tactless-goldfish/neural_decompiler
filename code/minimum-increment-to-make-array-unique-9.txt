int minIncrementForUnique (int* A, int ASize) {
    int i, max, cnt = 0;
    
    if (ASize < 2) {
        return 0;
    }
    
	 // use your own quick_sort()
    quick_sort(A, 0, ASize-1);
    
    max = A[0];
    for (i = 1; i < ASize; i++) {
        if (A[i] > max) {
            max = A[i];
        } else {
            max += 1;
            cnt += max - A[i];
        }
    }
    
    return cnt;
}
