#define MAX(A,B)(A>B?A:B)

int findMaxConsecutiveOnes(int* nums, int numsSize) {
    int i;
    int count=0,max=0,temp;
    for(i=0;i<numsSize;i++){
        if(nums[i]==1) count++;
        else{
            temp=count;
            max=MAX(max,temp);
            count=0;
        }
    }
    return MAX(count,max);
}
