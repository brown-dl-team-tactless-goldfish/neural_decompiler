int getMaximumGenerated(int n){
    int res[101]={0};
    res[0]=0;
    res[1]=1;
    int ans=0;
    if(n==1)ans=1;
    for(int i=2;i<=n;i++){
        if(i%2==0){
            res[i]=res[i/2];
        }
        else{
            res[i]=res[i/2]+res[i/2+1];
        }
        if(res[i]>ans){
            ans=res[i];
        }
    }
    return ans;
}
