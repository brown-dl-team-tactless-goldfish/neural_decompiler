typedef pair<int, int> PII;
class Solution {
public:
    int destination;
    
    double arr_time(PII car)
    {
        return (destination - car.first) / (double)car.second;
    }
    
    int carFleet(int target, vector<int>& position, vector<int>& speed) {
        int n = position.size();
        destination = target;
        
        vector<PII> combined;
        stack<PII> st;                                                  
                                                                         
        for (int i = 0; i < n; i ++)
        {
            combined.push_back({position[i], speed[i]});
        }
        
        sort(combined.begin(), combined.end(), greater<PII>());
        
        for (PII car: combined)
        {
            if (st.empty() || arr_time(car) > arr_time(st.top()))
                st.push(car);
        }
        
        return st.size();
    }
};
