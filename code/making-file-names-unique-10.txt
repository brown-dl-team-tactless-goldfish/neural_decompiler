 public class Solution
    {
        public string[] GetFolderNames(string[] names)
        {
            List<string> res= new List<string>(names.Length);

            IDictionary<string, int> name2suff = new Dictionary<string, int>();
            foreach (var name in names)
            {
                name2suff[name] = 1;
            }

            ISet<string> usedNames = new HashSet<string>();

            foreach (var name in names)
            {
                if (usedNames.Add(name))
                {
                    res.Add(name);
                    continue;
                }

                while (true)
                {
                    var newName = $"{name}({name2suff[name]})";
                    if (usedNames.Add(newName))
                    {
                        name2suff[name]++;
                        res.Add(newName);
                        break;
                    }
                    name2suff[name]++;
                }
            }

            return res.ToArray();
        }
    }
