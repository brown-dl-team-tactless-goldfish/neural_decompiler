struct ListNode* middleNode(struct ListNode* head){

    struct ListNode* start;
    int len=0,i;
    
    if (!head)
        return NULL;
    
    start = head;
    while(start)
    {
        start = start->next;
        len++;
    }
    i=0;
    while(i<len/2)
    {
        head=head->next;
        i++;
    }
    return head;
}
