class Solution {
public:
    #define ll long long
    int fact[1010];
    int mod= 1e9 +7;
    long long binpow(long long a, long long b, long long m) {
        a %= m;
        long long res = 1;
        while (b > 0) {
            if (b & 1)
                res = res * a % m;
            a = a * a % m;
            b >>= 1;
        }
        return res;
    }
    int fac(int n){
        if(n==0 || n==1) return 1;
        if(fact[n]) return fact[n];
        return fact[n] =(1ll * n * fac(n-1)) % mod;
    }
    int countOrders(int n) {
        ll num = fac(n * 2);
        int dem = binpow(2,n,mod);
        cout<<num<<" "<<dem<<endl;
        ll ans = (num % mod * binpow(dem,mod-2,mod) % mod) % mod;
        return (int)ans;
    }
};
