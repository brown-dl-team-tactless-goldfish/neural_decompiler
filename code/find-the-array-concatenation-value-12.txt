class Solution {
public:
    long long findTheArrayConcVal(vector<int>& nums) {
        long long res = 0;
        for(int i=0, j=nums.size()-1;i<=j;i++, j--){
            if(i == j) res += nums[i];
            else res += stoll(to_string(nums[i]) + to_string(nums[j]));
        }
        return res;
    }
};
