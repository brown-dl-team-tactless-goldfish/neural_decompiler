int minMoves(int target, int maxDoubles){
    int count=0, i=maxDoubles;
    if(maxDoubles==0)
    return target-1;
    while(target!=1){
        if(target/2>=1 && i!=0 && target%2==0){
            target=target/2;
            i--;
            count++;
        }
        else{
            target-=1;
            count++;
        }
    }
    return count;
}
