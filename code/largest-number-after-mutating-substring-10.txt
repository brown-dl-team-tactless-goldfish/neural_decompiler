class Solution {
public:
    string maximumNumber(string num, vector<int>& change) {
        bool v = false;
        for(char& c:num)
        {
            int temp = c - '0';
            if(change[temp]>temp || (v && change[temp]==temp))
            {
                c = '0' + change[temp];
                v = true;
                continue;
            }
            if(v)
            {
                break;
            }
        }
        return num;
    }
};
