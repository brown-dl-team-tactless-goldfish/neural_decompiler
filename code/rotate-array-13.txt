void rotate(int* nums, int numsSize, int k){
    k %= numsSize;
    if(k == 0) return;
    int tmp[k];
    for(int i = 0; i < k; i++)
    {
        tmp[i] = nums[numsSize-1-i];
    }
    for(int i = numsSize - 1; i >= k; i--)
    {
        nums[i] = nums[i - k];
    }
    for(int i = 0; i < k; i++)
    {
        nums[i] = tmp[k-1-i];
    }
    return;
}
