class Solution {
public:
    int minimumPartition(string s, int k) {
        long long int ans = 0;
        long long int c = 0;
        bool work = 0; 
        for(auto i: s){
            if(i - '0' > k) return -1;
        }
        for(int i  = 0; i < s.size(); ++i){
            ans = (ans * 10) + (s[i] - '0');
            if(ans > k){
                ans =  s[i] - '0';
                c++;
            }
        }
        c++;
        return c;
        
    }
};
