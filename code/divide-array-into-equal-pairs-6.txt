bool divideArray(int* nums, int numsSize){
    int stupid[500];
    memset(stupid, 0, 500 * sizeof(int));
    
    int i;
    for (i = 0 ; i < numsSize; i++) {
        stupid[nums[i] - 1]++;
    }
    
    for (i = 0; i < 500; i++) {
        if ( stupid[i]&1 ) {
            return false;
        }
    }
    
    return true;
    
}
