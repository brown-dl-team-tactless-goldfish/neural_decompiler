int cmpfunc (const void * a, const void * b) {
   return ( *(int*)b- *(int*)a );
}
int maxCoins(int* piles, int pilesSize){
  qsort(piles, pilesSize, sizeof(int), cmpfunc);
  int output=0;
   int times= pilesSize/3;
   

   for(int i=1;i<pilesSize;i=i+2){
    output=output+piles[i];
    times--;
    if(times==0){
        break;
    }
   }
return output;
}
