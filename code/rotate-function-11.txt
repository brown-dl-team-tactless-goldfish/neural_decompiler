class Solution {
public:
    int maxRotateFunction(vector<int>& A) {
        long f = 0, sum = 0, n = A.size();
        for (int i=0; i<n; ++i) {
            sum += A[i];
            f += i*A[i];
        }
        int res = f;
        for (int i=1; i<n; ++i) {
            f = f + sum - n*A[n-i];
            res = max(res, (int)f);
        }
        return res;
    }
};
