int cur_frogs=0;
int res_frogs=0;
while(*start != '\0'){
	map[*start]++;
	if(!check_map(map)){
		return -1;
	}
	if(*start == 'c'){
		cur_frogs++;
		res_frogs=MAX(res_frogs, cur_frogs);
	}else if(*start == 'k'){
		char * string= frog_sound;
		while(*string !='\0'){
			if(map[*string]< 1){
				return -1;
			}
			update_map(map, *string);
			string++;
		}
		cur_frogs--;
	}
	start++;
}

if(cur_frogs > 0){
	return -1;
}

return res_frogs;
