

bool canBeEqual(int* target, int targetSize, int* arr, int arrSize){
    int j;
    for(int i=0; i<arrSize; i++){
        for(j=0; j<arrSize;j++){
            if(arr[i]==target[j]){
                target[j]=0; // updating taget array to 0 is value found in arr array
                break;
            }
        }
    }
    int sum=0;
    for(int i=0; i<arrSize; i++){
        sum += target[i]; // finding sum of all values of taget array is all values with arr matched then sum must be 0 
    }
    if(sum==0)
        return true;
    else
        return false;
}
