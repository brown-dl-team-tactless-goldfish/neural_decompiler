int minNumber(int* nums1, int nums1Size, int* nums2, int nums2Size){
    int ds1 = 0, ds2 = 0;
    for (int i = 0; i < nums1Size; ++i) ds1 |= 1 << nums1[i];
    for (int i = 0; i < nums2Size; ++i) ds2 |= 1 << nums2[i];
    int common = ds1 & ds2;
    if (common != 0) return __builtin_ctz(common);
    int mn1 = __builtin_ctz(ds1);
    int mn2 = __builtin_ctz(ds2);
    if (mn1 < mn2) return mn1 * 10 + mn2;
    return mn2 * 10 + mn1;
}
