class Solution {
public:
	using VI = vector<int>;
	int maxScoreSightseeingPair(VI& A, int max = -1) {
		auto score = [&](auto i, auto j) { return A[i] + A[j] + i - j; };
		for (auto i{ 0 }, j{ 1 }; j < A.size(); ++j) {
			max = std::max(max, score(i, j));
			if (A[i] + i < A[j] + j)
				i = j; // best i found at j
		}
		return max;
	}
};
