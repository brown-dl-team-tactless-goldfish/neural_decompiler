int majorityElement(int* nums, int numsSize){
    int number = -1;
    int appears = 0;
    for(int i = 0;i<numsSize;i++){
        if(appears==0){
            number = nums[i];
            appears = 1;
        }
        else{
            if(number == nums[i]){
                appears++;
            }
            else{
                appears--;
            }
        }
    }
   
    return number;
}
