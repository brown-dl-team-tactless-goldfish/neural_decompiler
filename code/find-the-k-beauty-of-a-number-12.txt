class Solution {
public:
    int divisorSubstrings(int num, int k) {
        
        int count = 0;
        
        string num_str = to_string(num);
        
        for (int i = 0; i + k <= num_str.size(); i++)
        {
            string substr = num_str.substr(i,k);
            int substr_num = stoi(substr);
            
            if (substr_num == 0)
                continue;
            
            if (num % substr_num == 0)
                count++;
        }
        
        return count;
    }
};
