// Runtime: 15 ms, faster than 49.75% of C++ online submissions for Smallest Index With Equal Value.
// Memory Usage: 22.1 MB, less than 40.95% of C++ online submissions for Smallest Index With Equal Value.
int smallestEqual(vector<int>& nums) {
	for (int i = 0; i < nums.size(); i++)
		if (i % 10 == nums[i])
			return i;
	return -1;
}
