public class Solution 
{
    public long[] FindPrefixScore(int[] nums) 
    {
        long[] result = new long[nums.Length];
        int max = 0; 
        long score = 0;
        long conversion = 0;
        for (int i = 0; i < nums.Length; i++)
        {
            max = Math.Max(max, nums[i]);
            conversion = nums[i] + max;
            score += conversion;
            result[i] = score;
        }

        return result;
    }
}
