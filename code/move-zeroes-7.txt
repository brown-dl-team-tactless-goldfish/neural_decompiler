void moveZeroes(int* nums, int numsSize){
   int i, k;
   
   if(numsSize <= 1) {
      return;
   }
   for(i = 0, k = 0; i < numsSize; i++) {
      if(nums[i] != 0) {
         nums[k] = nums[i];
         k++;
      }
   }
   /* Make rest of the array 0 */
   for(; k < numsSize; k++) {
      nums[k] = 0;
   }
   return;
}
