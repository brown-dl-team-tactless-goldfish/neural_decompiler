bool escapeGhosts(vector<vector<int>>& ghosts, vector<int>& target) {
    
    
 int my_distance = abs(target[0])+abs(target[1]);
    
    for(auto x: ghosts)
    {
        int dist = abs(x[0]-target[0])+abs(x[1] - target[1]);
        
        if(my_distance >= dist)
        {
            return false;
        }
    }
    
    return true;
    
}
