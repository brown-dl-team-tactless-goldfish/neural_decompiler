    int maxNonOverlapping(vector<int>& nums, int target) {
        unordered_map<int, int> m;
        m[0] = -1;
        int res = 0;
        int sum = 0;
        int prev = -1;
        for(int i =0; i < nums.size(); ++i)
        {
            sum += nums[i];
            if(m.count(sum-target) && m[sum-target]+1 > prev)
            {
                res++;
                prev = i;
            }
            m[sum]=i;
        }
        return res;
    }
