public class Solution {
    public bool IsIdealPermutation(int[] A) {
        
        int currMax = 0;
        
        for(int i = 0; i < A.Length - 2; i++)
        {
            currMax = Math.Max(currMax, A[i]);
            if(currMax > A[i + 2])
                return false;
        }
        
        return true;
    }
}
