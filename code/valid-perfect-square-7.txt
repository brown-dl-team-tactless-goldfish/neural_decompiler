int mySqrt(int x) {
    int lo,hi;
    long mid;
    lo = 0;
    hi = x;
    while(lo<=hi){
        mid = lo + (hi-lo)/2;
        if(mid*mid>x){
            hi = mid-1;
        } else if ((mid+1)*(mid+1)>x) {
            return mid;
        } else{
            lo = mid+1;
        }
    }
    return lo;
}

bool isPerfectSquare(int num) {
    int sqrt = mySqrt(num);
    return sqrt*sqrt == num;
}
