class Solution {
public:
    int minimumOneBitOperations(int n) {
	     int zero = 0, one = 0;
	     for (int i = 1; n; n >>= 1, i <<= 1) {
		      one += i;
		 if (n % 2) swap(one, zero);
	}
	return zero;
  }
};![image](https://assets.leetcode.com/users/images/925de01b-9655-4e53-bd2e-d3f6f21e103c_1657197710.7105587.png)
