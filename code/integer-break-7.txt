

int integerBreak(int n){
    
 
    if(n ==  2)
        return 1;
    if(n == 3)
        return 2;
    if(n == 4)
        return 4;
    
    int t = n;
    int k = n;
    t = n / 3;
    k = k % 3;
    int result = 1;
    while(t)
    {
        result = result * 3;
        t--;
    }
    if(k == 1)
    {
        return result/3 * 4;
    }
    if(k == 2)
    {
        return result * 2;
    }
    
    
    return result;
        
    

}
