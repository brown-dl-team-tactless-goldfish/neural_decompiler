void rotate(int *arr,int n){
    int k=arr[0];
    for(int i=0;i<n-1;i++)
        arr[i]=arr[i+1];
    arr[n-1]=k;
}



int sorted(int *arr,int n){
    for(int i=0;i<n-1;i++)
        if(arr[i]>arr[i+1])
            return 0;
    return 1;
}

bool check(int* nums, int numsSize){
    for(int i=0;i<numsSize;i++){
        rotate(nums,numsSize);
        if(sorted(nums,numsSize))
            return true;
    }
    return false;    
}```
