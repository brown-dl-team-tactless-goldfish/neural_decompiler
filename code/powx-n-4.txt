double myPow(double x, int n){
    if (n == 0 || x == 1)
        return 1;
    if (x == -1){
        if (n % 2 == 1)
            return -1;
        else
            return 1;
    }
    if (n < 0){
        x = 1 / x;
    }
    double result = 1;
    if (n == INT_MIN){
        result = result * x;
        n = -(n + 1);
    }
    else
        n = abs(n);
    while (n > 0){
        result = result * x;
        if (result == 0)
            return 0;
        n--;
    }
    return result;
}
