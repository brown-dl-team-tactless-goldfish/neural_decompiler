class Solution {
public:
    int equalSubstring(string s, string t, int maxCost) {
        int n = s.size(), j = 0, c = 0, ans = 0;
        for(int i=0; i<n; i++)
        {
            c += abs(s[i]-t[i]);
            while(j<=i && c>maxCost)
            {
                c-=abs(s[j]-t[j]);
                j++;
            }
            if (c <= maxCost && j<=i)
                ans = max(i-j+1,ans);
        }
        return ans;
    }
};
