class LUPrefix {
	public:
		vector<int> v;
		int it=0;
		LUPrefix(int n) {
			vector<int> temp(n,-1);
			v=temp;
		}

		void upload(int video) {
			v[video-1]=1;
			while(it<v.size()){
				if(v[it]==-1) break;
				it++;
			}
		}

		int longest() {
			return it;
		}
	};

/**
 * Your LUPrefix object will be instantiated and called as such:
 * LUPrefix* obj = new LUPrefix(n);
 * obj->upload(video);
 * int param_2 = obj->longest();
 */
