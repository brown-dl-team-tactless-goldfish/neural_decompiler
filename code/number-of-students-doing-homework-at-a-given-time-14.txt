int busyStudent(int* startTime, int startTimeSize, int* endTime, int endTimeSize, int queryTime){
    if(startTimeSize == 1  && *(startTime) == queryTime){
        return 1;
    }
    int is = 0;
    for(int j = 0; j < startTimeSize; j++){
        for(int i = *(startTime + j); i <= *(endTime + j); i++){
            if(i == queryTime){
               is += 1; 
                break;
            }
        }
    }
    return is;
}
