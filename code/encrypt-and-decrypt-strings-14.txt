class Encrypter {
public:
    map<string, int> cnt;
    map<char, string> kv;
    Encrypter(vector<char>& keys, vector<string>& values, vector<string>& dict) {
        
        for(int i=0; i<keys.size(); i++)
            kv[keys[i]]=values[i];
        
        for(auto d : dict)
            cnt[encrypt(d)]++;
    }
    
    string encrypt(string word1) {
        string encp;
        for(auto c : word1)
            encp+=kv[c];
        return encp;
    }
    
    int decrypt(string word2) {
        return cnt[word2];
    }
};
