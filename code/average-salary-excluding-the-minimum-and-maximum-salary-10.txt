double average(int* salary, int salarySize){
    int max=0,min=salary[0];
    double sum=0;
    for(int i=0;i<salarySize;++i){
        if(salary[i]>max) max=salary[i];
        else if(salary[i]<min) min=salary[i];
        sum+=salary[i];
    }
    double output=((sum-max-min)/(salarySize-2));
    return output;
}
