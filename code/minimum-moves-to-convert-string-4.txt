int minimumMoves(char * s){
  int count = 0;
  for (size_t i = 0; i < strlen(s); i++)
  {
    if (s[i] == 'X')                 // 當遇到'X'
    {
      count++;                       // count += 1
      for (size_t j = 0; j < 3; j++) // 將含他的三個變成'O'
      {
        if (i + j < strlen(s))
            s[i + j] = 'O';          // 如果index < strlen(s)
      }
    }
  }
  return count;
}
