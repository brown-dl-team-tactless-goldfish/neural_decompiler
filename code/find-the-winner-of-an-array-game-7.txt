class Solution 
{
    public:
    int getWinner(vector<int>& arr, int k) 
    {
        int n=arr.size(),c=0,current=arr[0];//c for counting number of wins for current maximum, current for recording the current maximum
        for(int i=0;i<n-1;i++)
        {
            if(current>arr[i+1]) c++;//increase the counter for current maximum
            else c=1,current=arr[i+1];//change the curreent maximum to arr[i+1] and reset the counter
            if(c==k) return current;//current maximum reaches k wins
        }
        return current;//if not yet reaches k wins, return the maximum number (it will always be at the front)
    }
};
