    int n = nums.size();
    
    vector<int> temp = nums;
    
    sort(temp.begin( ),temp.end( ));
    

    
    int i = n-1;
    
  int  j = i / 2;
    
    int k = 0;
    
    while(k < n)
    {
        if(k & 1){
            
            nums[k++] = temp[i--];
        }
        else{
            
            nums[k++] = temp[j--];
        }
        
    }
    
}
