int step_size(int* u, int* v)
{
    return fmax(fabs(*v - *u),fabs(*(v+1)-*(u+1)));
}

int minTimeToVisitAllPoints(int** points, int pointsSize, int* pointsColSize){

    int ct = 0;
    
    for(int i = 0; i < pointsSize -1; i++)
        ct += step_size(points[i],points[i+1]);
    
    return ct;
}
