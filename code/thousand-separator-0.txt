char * thousandSeparator(int n){
    if(n == 0)
        return "0";
    char* ans = malloc(14 * sizeof(char));
    ans[13] = '\0';
    int val = n;
    int idx = 12;
    int k = 0;
    while(val)
    {
        if(k > 0 && k%3 ==0){
            ans[idx] ='.';
            idx--;
        }
        ans[idx] = val%10 +'0';
        val /= 10;
        idx--;
        k++;
    }
    return &ans[idx+1];
}
