    vector<int> q;
    int i;
	PeekingIterator(const vector<int>& nums) : Iterator(nums) {
	    q=nums;
        i=0;
	}
	
   
	int peek() {
        return q[i];
	}
	
	int next() {
	    i++;
        return q[i-1];
           
	}
	
	bool hasNext() const {
	    if(i<q.size()) return true;
        return false;
	}
};
