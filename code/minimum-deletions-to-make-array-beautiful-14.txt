class Solution {
public:
    int minDeletion(vector<int>& nums) {
        int idx = 0, ans = 0;
        for(int i=0;i<nums.size();){
            int ele = nums[i];
            int j = i+1;
            for(;j<nums.size();j++){
                if(nums[j] != ele) break;
                ans++;
            }
            if(j < nums.size()) idx += 2;
            else idx++;
            i = j + 1;
        }
        if(idx % 2) ans++;
        return ans;
    }
};
