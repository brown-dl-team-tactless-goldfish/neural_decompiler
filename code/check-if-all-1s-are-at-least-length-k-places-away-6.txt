bool kLengthApart(int* nums, int numsSize, int k)
{
    int places = 0;
    for(int i = 0; i < numsSize; i++)
    {
        if (nums[i] == 1)
        {
            if (places > 0)
                return false;
                
            places = k;
        }
        else
        {
            places--;
        }
    }
    
    return true;
}
