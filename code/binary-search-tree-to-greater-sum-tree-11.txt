int sum = 0;
void inorder(struct TreeNode* root);
struct TreeNode* bstToGst(struct TreeNode* root){
    inorder(root);
    sum = 0;// return the sum to zero for next test case
    return root;
}

void inorder(struct TreeNode* root){
    if(root == NULL) return;
    inorder(root->right);
    sum += root->val;
    root->val = sum;
    inorder(root->left);
}
