char * categorizeBox(int length, int width, int height, int mass){
    if ((length >= pow(10, 4) || width >= pow(10, 4) || height >= pow(10, 4)|| length >= (pow(10, 9) / (height * width))) && mass >= 100)
        return "Both";

    else if (length >= pow(10, 4) || width >= pow(10, 4) || height >= pow(10, 4) || length >= (pow(10, 9) / (height * width)))

        return "Bulky";

    else if (mass >= 100)
        return "Heavy";

    return "Neither";
}
