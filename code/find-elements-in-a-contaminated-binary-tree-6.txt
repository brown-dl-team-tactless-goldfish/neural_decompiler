public class FindElements 
{
    private TreeNode root;
    
    public FindElements(TreeNode r) 
    {
        this.root = r;
    }
    
    public bool Find(int target) 
    {
        return Find(root, target, 0);
    }
    
    private bool Find(TreeNode node, int target, int index)
    {
        if(target == index) return true;
        var left = node.left == null ? false : Find(node.left, target, index * 2 + 1); 
        var right = node.right == null ? false : Find(node.right, target, index * 2 + 2);
        return left || right;
    }
}
