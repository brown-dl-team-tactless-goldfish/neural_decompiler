/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     struct TreeNode *left;
 *     struct TreeNode *right;
 * };
 */
void order(struct TreeNode* root){
    if(root==NULL){
        return NULL;
    }
    order(root->left);
    order(root->right);
    if(root->left!=NULL){
        struct TreeNode* q=root->right;
        struct TreeNode* p=root->left;
        root->right=root->left;
        while(p->right!=NULL){
            p=p->right;
        }
        p->right=q;
        root->left=NULL;
    }
}
void flatten(struct TreeNode* root) {
    order(root);
    return root;
}
