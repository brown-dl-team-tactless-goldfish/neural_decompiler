class Solution {
public:
    int maxScoreSightseeingPair(vector<int>& values) {
        int n=values.size();
        int help=values[n-1]-n+1;
        int ans=INT_MIN;
        for(int i=n-2;i>=0;i--){
            ans=max(ans,i+values[i]+help);
            help=max(values[i]-i,help);
        }return ans;
    }
};
