int chalkReplacer(int* chalk, int chalkSize, int k){
    long long sum = 0;
    int i;
    for(i = 0; i < chalkSize; i++){
        sum += chalk[i];
    }
    
    k %= sum;
    
    for(i = 0; i < chalkSize; i++){
        k -= chalk[i];
        if(k < 0)
            return i;
    }
    
    return 0;

}
