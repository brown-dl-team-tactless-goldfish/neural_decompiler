int arrayNesting(int* a, int n)
{
    int max=INT_MIN;
    for(int i=0;i<n;i++)
    {
        if(a[i]!=-1)
        {
        int res=0,k=i,p;
        p=a[k];
        while(a[p]>=0)
        {
            int t=p;
            p=a[p];
            a[t]=-1;
            res++;
        }
        if(res>max)max=res;
        }
    }
    return max;
}
