class Solution {
public:
    int convertTime(string current, string correct) {
        int t1 = stoi(current.substr(0,2))*60 + stoi(current.substr(3,2));
        int t2 = stoi(correct.substr(0,2))*60 + stoi(correct.substr(3,2));
        int k = t2-t1,re=0;
        re = k/60 + (k%60)/15 + ((k%60)%15)/5 + ((k%60)%15)%5;
        return re;
    }
};
