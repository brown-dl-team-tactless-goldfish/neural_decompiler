/**
 * Note: The returned array must be malloced, assume caller calls free().
 */
int* evenOddBit(int n, int* returnSize){

    *returnSize = 2;
    int cnt = 0;
    int even = 0;
    int odd = 0;

    int* arr = malloc(sizeof(int)*2);
    
    while (cnt < 16)
    {
        int x = (n>>cnt & 1);
        
        if(cnt%2 == 0)
            even += x;
        else
            odd += x;

        ++cnt;
    }
    
    arr[0] = even;
    arr[1] = odd;

    return arr;
}
