int sort(const* a,const* b){
    return *(int*)a - *(int*)b;
}
bool* kidsWithCandies(int* candies, int candiesSize, int extraCandies, int* returnSize){
    int *nums = (int*)malloc(candiesSize *sizeof(int));
    for(int i = 0; i < candiesSize; i++){
        nums[i] = candies[i];
    }
    qsort(nums,candiesSize,sizeof(int),sort);
    bool *arr = (bool*)malloc(candiesSize * sizeof(bool));
    for(int i = 0; i < candiesSize; i++){
        if(candies[i] + extraCandies >= nums[candiesSize - 1]){
            arr[i] = true;
        }
        else{
            arr[i] = false;
        }
    }
    *returnSize = candiesSize;
    return arr;
}
