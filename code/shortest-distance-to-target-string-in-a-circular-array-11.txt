class Solution {
public:
    int closetTarget(vector<string>& words, string target, int startIndex) {
        int idx = -1,n = words.size(),mn = n;
        for(int i=0;i<n;i++){
            if(words[i] == target){
                idx = i;
                mn = min(mn,min(((n + startIndex - idx) % n),((n - startIndex + idx) % n)));
            }
        }
        if(idx == -1)
            return -1;
        return mn; 
    }
};
