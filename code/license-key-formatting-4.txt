temp_array[index] = '\0';

for (i = strlen(s) - 1; i >= 0; i--)
{
    if (curK == k)
    {
        temp_array[--index] = '-';
        curK = 0;
    }
    
    if (s[i] == '-')
    {
        continue;
    }
    else if ((s[i] >= 'a') && (s[i] <= 'z'))
    {
        temp_array[--index] = s[i] - 'a' + 'A';
        curK++;
    }
    else
    {
        temp_array[--index] = s[i];
        curK++;
    }
}
if (temp_array[index] == '-')
    return &temp_array[index+1];
else
    return &temp_array[index];
