class Solution {
public:
    int maxCount(vector<int>& banned, int n, int maxSum) {
        set <int> st;
        for(auto &it:banned)
            st.insert(it);
        int cnt = 0,sum = 0;
        for(int i=1;i<=n;i++){
            if(st.find(i) == st.end())
                cnt++,sum += i;
            if(sum > maxSum){
                cnt--;
                break;
            }
        }
        return cnt;
    }
};
