class Solution {
public:
    bool stoneGameIX(vector<int>& stones) {
        vector<int> cnt(10, 0);
        for (auto val : stones) cnt[val%3]++;
        
        if (!cnt[1]&&!cnt[2]) return false;
        if (cnt[0] & 1)       return abs(cnt[1]-cnt[2])>=3;
        else                  return min(cnt[1],cnt[2])!=0;
    }
};
