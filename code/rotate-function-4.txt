int maxRotateFunction(int* A, int ASize){

	if(ASize == 0) {
		return 0;
	}

	int max;
	int total = 0;
	int temp = 0;

	for(int i = 0; i < ASize; i++) {
		if(A[i] == INT_MAX || A[i] == INT_MIN) {
			return A[i];
		}
		temp += i * A[i];
		total += A[i];
	}

	max = temp;

	for(int i = ASize - 1; i > 0; i--) {
		temp += total - ASize * A[i];
		if(temp > max) {
			max = temp;
		}
	}
	return max;
}
