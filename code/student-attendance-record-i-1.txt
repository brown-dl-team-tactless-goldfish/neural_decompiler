bool
checkRecord(char *s)
{
	int absent, late;

	absent = late = 0;
	for (;;)
		switch (*s++) {
			case '\0':
				return true;
			case 'L':
				if (++late > 2)
					return false;
				break;
			case 'A':
				if (++absent > 1)
					return false;
				/* FALLTHROUGH */
			default:
				late = 0;
		}

	/* UNREACHED */
}
