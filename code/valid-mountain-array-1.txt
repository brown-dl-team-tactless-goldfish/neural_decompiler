bool validMountainArray(int* A, int ASize) {
    bool up = false, down = false;
    for(int i = 0; i < ASize-1; i++)
    {
        if(A[i] < A[i+1])
        {
            if(down) return false;
            else up = true;
        }
        else if (A[i] > A[i+1])
        {
            if(!up) return false;
            else down = true;
        }
        else
        {
            return false;
        }
    }
    return (up && down);
}
