long long maximumImportance(int n, vector<vector<int>>& r) {
    int m = r.size();
    vector<int>adj[n];
    for(int i=0;i<m;i++){
        int u = r[i][0];
        int v = r[i][1];
        adj[u].push_back(v);
        adj[v].push_back(u);
    }
    vector<long long>v;
    for(auto x:adj){
        v.push_back(x.size());
    }
    sort(v.begin(),v.end(),greater<int>());
    long long c = 0;
    for(auto x:v){
        c+=(x*(long long)n);
        n--;
    }
    return c;
}
