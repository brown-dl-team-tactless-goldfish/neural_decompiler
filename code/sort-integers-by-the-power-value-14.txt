class Solution {
    int power(int n) {
        int ans = 0;
        while(n > 1) {
            n % 2 ? n = n * 3 + 1 : n /= 2;
            ans++;
        }
        return ans;
    }
public:
    int getKth(int lo, int hi, int k) {
        priority_queue<pair<int, int>> pq;

        for(int i=lo; i<=hi; i++) {
            pq.push({power(i), i});
            if(pq.size() > k) pq.pop();
        }
        return pq.top().second;
    }
};
