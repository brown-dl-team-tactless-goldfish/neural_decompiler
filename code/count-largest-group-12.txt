    public int CountLargestGroup(int n) {
        
        Dictionary<int,int> dic = new Dictionary<int,int>();
        int largestSize = -1, count = 0;
        
        for(int i = 1; i <= n; i++)
        {         
            int num = i, digitSum = 0;
            while(num > 0)
            {
                digitSum += num % 10;
                num /= 10;
            }
                
            if(dic.ContainsKey(digitSum))
                dic[digitSum]++;
            else
                dic.Add(digitSum, 1);
            
            largestSize = Math.Max(largestSize, dic[digitSum]);                                    
        }
        
        foreach(var kv in dic)
        {
            if(kv.Value == largestSize)
                count++;
        }
        
        return count;
    }
