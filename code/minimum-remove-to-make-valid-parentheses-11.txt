char * minRemoveToMakeValid(char * s){
    int ln=strlen(s);
    int stack[ln];
    int top=-1;
    for(int i=0;i<ln;i++)
    {
        if(s[i]=='(')
        {
            stack[++top]=i; 
        }
        if(s[i]==')')
        {   
            if(top>-1)
            {   
                if(s[stack[top]]=='(')
                {
                    stack[top]=NULL;
                    top--;
                }

                else{
                    if(i==ln-1)
                    {
                        s[i]=NULL; 
                        ln--;
                    }
                    else
                    {
                        for(int j=i;j<ln-1;j++)
                        {
                            s[j]=s[j+1];
                        }
                        s[ln-1]=NULL;
                        ln--;
                    }
                }
            }
            else{
                if(i==ln-1)
                {
                    s[i]=NULL; 
                    ln--;
                }
                else
                {
                    for(int j=i;j<ln-1;j++)
                    {
                        s[j]=s[j+1];
                    }
                    s[ln-1]=NULL;
                    ln--;
                    i--;
                }
            }
        }
    }
    
    for(int i=top;i>-1;i--)
    {
        s[stack[i]]=NULL;
        if(stack[i]==ln)
        {
            stack[top]=NULL;
        }
        else{
            for(int j=stack[i];j<ln-1;j++)
            {
                s[j]=s[j+1];   
            }
            s[ln-1]=NULL;
            ln--;
        }
    }
    return s;
}
