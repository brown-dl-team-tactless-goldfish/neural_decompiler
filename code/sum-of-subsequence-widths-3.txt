class Solution {
public:
	int sumSubseqWidths(vector<int>& A) {
		int kMod=1e9+7;
		vector<long> powOf2(20001);
		powOf2[0]=1;
		for(int i=1;i<=20000;i++)
		{
			powOf2[i]=(powOf2[i-1]*2)%kMod;
		}
		sort(A.begin(),A.end());
		int ans=0;
		int n=A.size();
		for(int i=0;i<A.size();i++)
		{
			ans-=A[i]*(powOf2[n-1-i+1-1]-1)%kMod;
			ans+=A[i]*(powOf2[i]-1)%kMod;
			ans%=kMod;

		}
		return ans;

	}
};
