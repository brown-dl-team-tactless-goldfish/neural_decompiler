class Solution {
public:
    int minNumberOfFrogs(string croakOfFrogs) {
        const char *croak = "croak";
        vector<int> v(5);
        int r = 0;
        for (char c :  croakOfFrogs) {
            const int temp =  strchr(croak, c) - croak;
            if (temp == 0 && v[0] == 0) {
                ++v[0];
                ++r;
            }
            if (--v[temp] < 0) return -1;
            ++v[(temp + 1) % 5];
        }
        return (v[0] == r) ? r : (-1);
        
    }
};
