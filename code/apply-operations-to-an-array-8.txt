    int cnt = 0;
    
    for(int i=0; i<n-1; i++)
    {
        if(v[i] == v[i+1])
        {
            v[i] = 2*v[i];
            
            v[i+1] = 0;
        }
    }
    
    vector<int> ans;
    
    for(auto &x: v)
    {
        if(x == 0)
        {
            cnt++;
            
        }
        else
        {
            ans.push_back(x);
        }
    }
    
    for(int i=0; i<cnt; i++)
    {
        ans.push_back(0);
    }
    
    return ans;
}
