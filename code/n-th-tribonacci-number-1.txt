int tribonacci(int n){
    int x;
    int a = 0;
    int b = 1;
    int c = 1;
    int count = 2;
        if(n==0)
            return 0;
        if(n==1 || n==2)
            return 1;
    
        while(n>=3 && count!=n){
          x= a+b+c;
            a=b;
            b=c;
            c=x;
            count++;
            
        }
   
       return x;     
    }
