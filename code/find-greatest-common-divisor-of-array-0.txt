

int findGCD(int* nums, int numsSize){
int key, j;
    
    for (int i = 1; i <= numsSize-1; i++)
    {
        key = nums[i];
        j = i-1;
               while(j>=0 && nums[j] > key){
            nums[j+1] = nums[j];
            j--;
        }
        nums[j+1] = key;
    }
int a=nums[0];
int b=nums[numsSize-1];
 int r;
do
{
   r= b%a;
   if (r>0)
   {
    b=a;
    a=r;
   }
   else if(r==0)
   break;
   } 
    while(r>=0);
   return a;
}
