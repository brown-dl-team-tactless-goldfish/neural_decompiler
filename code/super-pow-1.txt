int func(int a,int b){
    int ret=1;
    for(int i=0;i<b;i++){
        ret=ret*a%1337;
    }
    return ret;
}
int myPow(int a,int b,int zeroNum){
    int ret=func(a,b);
    for(int i=0;i<zeroNum;i++){
        ret=func(ret,10);
    }
    return ret;
}
int superPow(int a, int* b, int bSize){
    int ret=1;
    a%=1337;
    for(int i=0;i<bSize;i++){
        ret=ret*myPow(a,b[i],bSize-i-1)%1337;
    }
    return ret;
}
