struct pair {
    double dis;
    double spd;
    double sec
};

int cmp(const void* a, const void* b) {
    struct pair pa = *(const struct pair*) a;
    struct pair pb = *(const struct pair*) b;
    return pa.sec > pb.sec;
}

int eliminateMaximum(int* dist, int distSize, int* speed, int speedSize) {
    struct pair* arr = (struct pair*) malloc(distSize * sizeof(struct pair));
    for(int i = 0; i < distSize; i++) {
        arr[i].dis = dist[i];
        arr[i].spd = speed[i];
        arr[i].sec = arr[i].dis / arr[i].spd;
    }
    qsort(arr, distSize, sizeof(struct pair), cmp);
    int ans = 0;
    for(int i = 0; i < distSize; i++) {
        double tmp = ans * arr[i].spd;
        if(arr[i].dis - tmp <= 0) break;
        ans++;
    }
    free(arr);
    return ans;
}
