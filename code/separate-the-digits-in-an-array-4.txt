int* separateDigits(int* nums, int numsSize, int* returnSize) {

	char a[6500] = "";
	char buf[7] = "";
	register unsigned short i, j = 0, answer_len;
	for(i = 0; i < numsSize; i++) {
		snprintf(buf, 7, "%d", nums[i]);
		strcat(a, buf);
	}

	answer_len = strlen(a);
	*returnSize = answer_len;
	int * answer = (int*)malloc(answer_len * sizeof(int));

	while(a[j] != '\0') {
		answer[j] = a[j] - '0';
		j++;
	}

	return answer;
}
