public class Solution {
    public long AppealSum(string s) {
        long appeal = 0;
        int[] prev = new int[26];
        Array.Fill(prev, -1);
        for(int i = 0; i < s.Length; i++){
            appeal += (i-prev[s[i]-'a'])*(s.Length-i);
            prev[s[i]-'a'] = i;
        }
        return appeal;
    }
}
