void rotate(int* nums, int numsSize, int k) {
    k=k%numsSize;
    if(k==0){return ;}
    int *p=(int*)malloc(sizeof(int)*numsSize);
    int count=0;
    for(int i=0;i<numsSize-k;i++){
        p[count++]=nums[i];
    }
    for(int i=0;i<k;i++){
        nums[i]=nums[numsSize-k+i];
    }
    count=0;
    for(int i=k;i<numsSize;i++){
        nums[i]=p[count++];
    }
}
