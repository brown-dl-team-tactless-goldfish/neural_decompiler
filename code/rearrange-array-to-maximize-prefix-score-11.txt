class Solution {
public:
    int maxScore(const vector<int>& nums) {
        int count = 0;
        long long sum = 0;
        priority_queue<int> neg;

        for(int i : nums) {
            if(i > 0) {
                sum += i;
                ++count;
            }
            else neg.push(i);
        }

        while(!neg.empty()) {
            sum += neg.top();
            neg.pop();
            if(sum > 0) ++count;
            else break;
        }

        return count;
    }
};
