class Solution {
public:
    int distinctAverages(vector<int>& nums) {
        set<double> st;
        sort(nums.begin(), nums.end());
        
        int i = 0, j = nums.size()-1;
        while(i < j) {
            double a = nums[i], b = nums[j];
            st.insert((a+b)/2);
            // cout << (a+b)/2 << endl;
            i++, j--;
        }
        
        return st.size();
    }
};
