bool* prefixesDivBy5(int* A, int ASize, int* returnSize){
    *returnSize = ASize;
    bool *res = (bool *)malloc(ASize * sizeof(bool));

    int q0 = 0, q1 = 1, q2 = 2, q3 = 3, q4 = 4;
    int DFA[5][2] = { {q0, q1}, {q2, q3}, {q4, q0}, {q1, q2}, {q3, q4} };

    int status = q0;
    for (int i = 0; i < ASize; i++) {
        status = DFA[status][A[i]];
        res[i] = status == q0;
    }

    return res;
}
