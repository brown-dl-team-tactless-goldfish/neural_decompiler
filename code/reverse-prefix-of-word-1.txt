char* reversePrefix(char* word, char ch)
{
    int to = (-1), len;
    char tmp;
    for (len = 0; word[len]; len++)//finds len and char occurence at the same time
        if (to<0 &&  word[len] == ch )
            to = len ;
    if (to<1)//to not found or to==0
        return word;
    for (int i = 0; i < (to / 2)+1; i++)
    {//simple swap-around-mid-pivot
        tmp = word[i];
        word[i] = word[to - i];
        word[to-i] = tmp;
    }
    return word;
}
