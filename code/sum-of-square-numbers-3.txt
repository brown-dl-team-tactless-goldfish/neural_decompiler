class Solution {
public:
    bool judgeSquareSum(int c) {
	
        long long temp = sqrt(c);
        if(temp*temp == c) return true;
		// These two lines check if number is perfect sq
		
        long long i = 1;
        while(i*i < c) {
            temp = c - i*i; // Decrementing 1,4,9,16,25... from number
            long long sq = sqrt(temp);
            if(sq*sq == temp) return true;
			// Above two line check if the remaining part is perfect sq or not
			
            i++; // If its not a perfect square we will incremenet number we have subtracted first and then again check if remaining part is perfect sq
        }
        return false;
    }
};
