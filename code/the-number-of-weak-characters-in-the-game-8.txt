public int numberOfWeakCharacters(int[][] properties) {
  int[][] props = properties.clone();
  Arrays.sort(props, (lhs, rhs) -> (lhs[0] != rhs[0]) ? rhs[0] - lhs[0] : lhs[1] - rhs[1]);
  int ret = 0;
  int def = 0;
  for (int[] prop : props) {
    if (prop[1] < def) {
      ++ret;
    } else {
      def = prop[1];
    }
  }
  return ret;
}
