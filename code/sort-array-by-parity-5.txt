int* sortArrayByParity(int* nums, int numsSize, int* returnSize){
    int *ans = malloc(numsSize * sizeof *ans);
    *returnSize = 0;
    for (int i = 0; i < numsSize; i++) 
        if (nums[i] % 2 == 0) 
            ans[(*returnSize)++] = nums[i];
    for (int i = 0; i < numsSize; i++) 
        if (nums[i] % 2) 
            ans[(*returnSize)++] = nums[i];
    return ans;
}
