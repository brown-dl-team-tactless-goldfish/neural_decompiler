class Solution {
public:
    long long taskSchedulerII(vector<int>& t, int space) {
        long ans=0;
        unordered_map<long,long>lastDay;
        for(auto & type:t){
            // if this type is seen before then check latest day from which we can start
            if(lastDay.count(type))
                ans=max(ans,lastDay[type]);
            // do that job using 1 second
            ans++;
            // update lastday
            lastDay[type]= ans+space;
        }
        return ans;
    }
};
