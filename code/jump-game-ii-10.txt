int jump(int* nums, int numsSize){
    if (numsSize == 1) {return 0;}
    int res = 0, curPos = 0, curFar = 0;
    for (int i = 0; i < numsSize - 1; ++i) {
        curFar = (curFar < i + nums[i])? i + nums[i] : curFar;
        if (i == curPos) {
            res++;
            curPos = curFar;
        }
    }
    return res;
}
