class Solution {
public:
    vector<int> mostCompetitive(vector<int>& A, int k) {
        vector<int> res;
        deque<int> dq;
        for(int i = 0; i < A.size(); i++) {
            while(!dq.empty() && A[i] < A[dq.front()]) dq.pop_front();
            dq.push_front(i);
            if(A.size()-i <= k) {
                res.push_back(A[dq.back()]);
                dq.pop_back();
                k--;
            }
        }
        return res;
    }
};
