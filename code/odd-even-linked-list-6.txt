void change(struct ListNode* root, struct ListNode* second, struct ListNode* lastend, int count){
    if(root->next == NULL){
        if(count % 2 == 0){
            root->next = second;
            return;
        } else {
            lastend->next = second;
            return;
        }
    }
    struct ListNode* copy = root->next;
    root->next = copy->next;
    change(copy,second,root,count+1);
    return;
}

struct ListNode* oddEvenList(struct ListNode* head){
    if(head == NULL){
        return head;
    }
    struct ListNode* second = head->next;
    change(head,second,head,0);
    return head;
}
