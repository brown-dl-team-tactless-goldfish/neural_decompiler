int* sumZero(int n, int* returnSize){
    *returnSize = n;
    int x = 1;
    int *finalArray = malloc(sizeof(int) * n);
    if (n % 2 != 0){
        finalArray[0] = 0;
        for (int i = 1; i < n; ++i){
            finalArray[i] = x;
            finalArray[++i] = -x;
            ++x;
        }
    }
    
    else{
        for (int i = 0; i < n; ++i){
            finalArray[i] = x;
            finalArray[++i] = -x;
            ++x;
        }
    }
    
    return finalArray;
}
