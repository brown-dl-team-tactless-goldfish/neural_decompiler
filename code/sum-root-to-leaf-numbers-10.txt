int sm(struct TreeNode* root, int ver)
{
    int l,r;
    l=r=0;
    if(root->left==NULL && root->right==NULL)
        return ver;

    if(root->left!=NULL)
        l=sm(root->left, ver*10+root->left->val);

    if(root->right!=NULL)
        r=sm(root->right, ver*10+root->right->val);
    
    return l+r;
}
