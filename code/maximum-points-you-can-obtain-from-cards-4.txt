int maxScore(int* cardPoints, int cardPointsSize, int k){
	int windowsize = cardPointsSize - k;
	int min = INT_MAX;
	int sum = 0;
	int tmp = 0;
	for(int i = 0; i < cardPointsSize; i++){
		sum += cardPoints[i];
	}
	if(k == cardPointsSize) return sum;

	for(int i = 0; i < windowsize; i++)
	{
		tmp += cardPoints[i];
	}
	min = min > tmp ? tmp : min;
	for(int i = 1; i < cardPointsSize - windowsize + 1; i++){
		tmp = tmp - cardPoints[i - 1] + cardPoints[i + windowsize - 1] ;
		min = min > tmp ? tmp : min;
	}

	return sum - min;
}
