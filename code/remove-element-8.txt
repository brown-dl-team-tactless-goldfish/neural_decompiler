int removeElement(int* nums, int numsSize, int val){
    register unsigned char i, temp = 0, sum = numsSize;
	for(i = 0; i < numsSize; i++) {
		if(nums[i] == val) {
			temp = nums[i];             // swap first with last
			nums[i] = nums[numsSize-1]; //
			nums[numsSize-1] = temp;    //
            numsSize--;                 // last to second last..
            i--;                        // check the i'th after swap, maybe equal to val
			sum--;                      // don't check all the val's at the end
		}
	}
	return sum;
}
