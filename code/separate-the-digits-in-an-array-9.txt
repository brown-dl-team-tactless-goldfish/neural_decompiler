class Solution {
public:
    vector<int> separateDigits(vector<int>& nums) {
        vector<int> res;
        int i = 0;
        for(int n : nums) {
            vector<int> tmp;
            int cnt = 0;
            while(n) {
                res.insert(res.begin() + i, n % 10);
                n /= 10;
                cnt++;
            }
            i += cnt;
        }
        return res;
    }
};
