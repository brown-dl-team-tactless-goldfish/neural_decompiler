class Solution {
public:
    int minimizeSum(vector<int>& nums) {
        sort(nums.begin(),nums.end());
        int n = nums.size(),mn = INT_MAX;
        int i = max(0,n-3);
        mn = min(mn,nums[i]-nums[0]);
        i = max(0,n-2);
        mn = min(mn,nums[i]-nums[1]);
        i = max(0,n-1);
        mn = min(mn,nums[i]-nums[2]);
        return mn;
    }
};
