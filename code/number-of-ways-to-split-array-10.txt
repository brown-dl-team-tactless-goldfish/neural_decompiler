class Solution {
public:
    int waysToSplitArray(vector<int>& nums) {
        long long ls = 0, rs = 0, ans = 0, n = nums.size();
        for(int i : nums) rs += i;

        for(int i=0; i<n-1; i++) {
            ls += nums[i];
            rs -= nums[i];
            ans += ls >= rs;
        }
        return ans;
    }
};
