    unordered_map<int, stack<int>> map;
    unordered_map<int, int> freq;
    int max_freq = 0;
public:
    FreqStack() {
        
    }
    
    void push(int x) {
        max_freq = max(max_freq, ++freq[x]);
        map[freq[x]].push(x);
    }
    
    int pop() {
        int res = map[max_freq].top();
        map[max_freq].pop();
        --freq[res];
        if(map[max_freq].empty()) max_freq--;
        return res;
    }
