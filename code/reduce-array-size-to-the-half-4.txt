int cmp(int *a, int *b) {
    return *b - *a;
}

int minSetSize(int* arr, int arrSize){
    int a[100001] = {0};
    
    for (int i=0; i<arrSize; i++) {
        a[arr[i]]++;
    }
    
    qsort(a, 100001, sizeof(int), cmp);
    
    int total = 0;
    int i;
    for (i=0; i<100001; i++) {
        total += a[i];
        if (total >= arrSize / 2) {
            break;
        }
    }
    
    return i + 1;
}
