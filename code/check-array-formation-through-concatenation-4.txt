bool canFormArray(int* arr, int arrSize, int** pieces, int piecesSize, int* piecesColSize){
    int ans = false;
    int* mapping = (int*)calloc(101, sizeof(int));
    /* record indx into mapping array */
    for(int i = 0; i < arrSize; i++)
        mapping[arr[i]] = i + 1;
    int startIdx;
    /* read the first element of piece then 
        check the index correct or not*/
    for(int i = 0; i < piecesSize; i++){
        startIdx = mapping[pieces[i][0]];
        if(!startIdx) return false; /* index isn't exist */
        for(int j = 0; j < piecesColSize[i]; j++)
            if(mapping[pieces[i][j]] != (startIdx + j))
                return false;   
    }
    return true;
}
