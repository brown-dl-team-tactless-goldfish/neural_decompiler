    int ans = 0, id = 0, range = startFuel;
    
    while(range < target){
    
    while(id < stations.size() && stations[id][0] <= range)
    {
        pq.push(stations[id++][1]);
    }
        
    if(pq.empty())
    {
        return -1;
    }
    
    range += pq.top();
    pq.pop( );
    
    ans++;
        
 }
    
    return ans;
