int totalHammingDistance(int* nums, int numsSize){

    int total_dist = 0;
    int bits = 0;
    for (int i = 0; i < 32; i++) {
        for (int j = 0; j < numsSize; j++) {
            bits += (nums[j] >> i) &1;
        }
        total_dist += bits*(numsSize-bits);
        bits = 0;
    }
    return total_dist;
}
