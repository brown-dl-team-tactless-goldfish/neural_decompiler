/**
 * Note: The returned array must be malloced, assume caller calls free().
 */
int* divisibilityArray(char * word, int m, int* returnSize){
    int n = strlen(word) ;
    *returnSize = n ;
    int* ans = malloc(n * sizeof(int)) ;
    unsigned long long val = 0 ;
    for(int i = 0; i < n; i++){
        val = val * 10 + word[i] - '0' ;
        val %= m ;
        if(val ==0)
            ans[i] = 1 ;
        else 
            ans[i] = 0 ;
    }
    return ans ;
}
