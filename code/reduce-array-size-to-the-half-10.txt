int cmpfunc (const void * a, const void * b)
{
   return -( *(int*)a - *(int*)b );
}

int minSetSize(int* arr, int arrSize){
    int map[100001] = {0}, i, half = arrSize / 2, pass = 0;
    for (i = 0; i < arrSize; ++i) ++map[arr[i]];
    qsort(map, 100001, sizeof(int), cmpfunc);
    while (arrSize > half) {
        arrSize -= map[pass++];
    }
    return pass;
}
