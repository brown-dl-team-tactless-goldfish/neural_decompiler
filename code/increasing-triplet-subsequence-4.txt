bool increasingTriplet(int* nums, int numsSize) {
    int a, b;
    if (numsSize < 3) return 0;

    a = b = nums[0];
    for (int i = 1; i < numsSize; i++) {
        if (b > a && nums[i] > b) {
            return 1;
        }
        if (nums[i] <= a) {
            if (b == a) {
                b = nums[i];
            }
            a = nums[i];
        } else {
            b = nums[i];
        }
    }

    return 0;
}
