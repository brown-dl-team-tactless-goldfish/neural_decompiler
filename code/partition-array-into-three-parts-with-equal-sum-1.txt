bool canThreePartsEqualSum(int* arr, int arrSize){
    int arrsum = 0, psum = 0, c = 0;
    for (int i = 0; i < arrSize; i++) arrsum += arr[i];
    if (!(arrsum % 3))
    {
        arrsum /= 3;
        for (int i = 0; i < arrSize; i++)
        {
            psum += arr[i];
            if (psum == arrsum)
            {
                psum = 0;
                c++;
            }
        }
        
    }
    else return false;
    
    if (c < 3) return false;
    else return true;
}
