int findPairs(int* nums, int numsSize, int k){
    char a[200000] = { 0 }, *m = &a[500];
    for (int i = 0, c = 0, *n = nums ; i < numsSize || (numsSize = c, 0) ; i++)
        for (int j = i, p ; ++j < numsSize ; abs(n[i] - n[j]) == k && !m[p = n[i] + n[j]] ? c += m[p] = 1 : 0);
    return numsSize;
}
