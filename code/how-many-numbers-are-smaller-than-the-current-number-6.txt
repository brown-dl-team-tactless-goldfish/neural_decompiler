int* smallerNumbersThanCurrent(int* nums, int numsSize, int* returnSize){
int *ans= (int*)malloc(sizeof(int)*numsSize);
    *returnSize= numsSize;
    int i,j;
    for(i=0;i<numsSize;i++){
      int count=0;
        for(j=0;j<numsSize;j++){
        if(nums[i]>nums[j]){
            count++;
        }
    }
        ans[i]= count;
    }
    return ans;
}
