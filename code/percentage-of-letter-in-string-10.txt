public class Solution {
    public int PercentageLetter(string s, char letter) {
        
        if(string.IsNullOrEmpty(s))
            throw new ArgumentException("Invalid Input.");
        
        int cnt = 0;
        foreach(var c in s)
        {
            if(c == letter)
                cnt++;
        }
        
        return cnt * 100 / s.Length ;
    }
}
