int oddCells(int m, int n, int** A, int S, int* C){
    bool* v = (bool*)calloc(m+n,sizeof(bool));
    int r ,c;
    r = c = 0;
    for(int i = 0 ; i < S ; i++){
        v[A[i][0]] = !v[A[i][0]];
        v[m+A[i][1]] = !v[m+A[i][1]];
    }
    for(int i = 0 ; i < m ; i++){
        if(v[i]) r++;
    }
    for(int i = m ; i < n+m ; i++){
        if(v[i]) c++;
    }
    free(v);
    return r*(n-c) + c*(m-r);
}
