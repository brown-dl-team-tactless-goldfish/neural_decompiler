class Solution {
public:
    int wateringPlants(vector<int>& plants, int capacity) {
        long ans = 0, C = capacity;
        for (int i = 0; i < plants.size(); i++) {
            if (plants[i] <= capacity) {
                capacity -= plants[i];
            } else {
                ans+= (2*i);
                capacity = C;
                capacity -= plants[i];
            }
            ans ++;
        }
        return ans;
    }
};
