int maximumDifference(int* nums, int numsSize){
   int left = 0;
   int right = 1;
 
   int maxDiff = 0;
                                                                                                         
   while( right < numsSize ){
 
      if ( nums[left] > nums[right]) left = right;
 
      if ( nums[right]-nums[left] > maxDiff ) maxDiff = nums[right]-nums[left];
 
      right++;
   }
 
   return ( maxDiff == 0 ) ?  -1 : maxDiff ;

}
