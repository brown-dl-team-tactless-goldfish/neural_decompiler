    float m1 = (points[0][0] == points[1][0]) ? INT_MAX : (float)(points[0][1]-points[1][1])/(points[0][0] - points[1][0]) ;
    float m2 = (points[2][0] == points[1][0]) ? INT_MAX : (float)(points[2][1]-points[1][1])/(points[2][0] - points[1][0]) ;
    
    if( points[0][0] == points[1][0] &&  points[0][1] == points[1][1])
        return false ;
    if( points[2][0] == points[1][0] &&  points[2][1] == points[1][1])
        return false ;
    if( points[0][0] == points[2][0] &&  points[0][1] == points[2][1])
        return false ;

    if( m1 == m2 )
        return false ;
    return true;
}
