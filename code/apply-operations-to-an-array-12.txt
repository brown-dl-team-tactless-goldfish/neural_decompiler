class Solution {
public:
    vector<int> applyOperations(vector<int>& A) {
        int j = 0;
        for(int i = 0; i < A.size(); i++) {
            if(i+1 < A.size() && A[i] == A[i+1] && A[i]) {
                A[j++] = 2*A[i+1];
                A[i+1] = 0;
            } else if(A[i]) {
                A[j++] = A[i];
            }
        }
        for(int i = j; i < A.size(); i++) A[i] = 0;
        return A;
    }
};
