int removeElement(int* nums, int numsSize, int val){
    printf("numsSize = %d\n",numsSize);
    int count = 0;
    int tail = numsSize-1;
    for(int i=0;i<numsSize;i++){
        if(nums[i]==val){
            while(tail>0 && nums[tail]==val)
                tail--;
            if(i<tail){
                nums[i] = nums[tail];
                tail--;
            }
        }else{
            count++;
        }
    }
    return count;
}
