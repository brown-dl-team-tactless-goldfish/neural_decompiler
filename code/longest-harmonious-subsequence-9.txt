class Solution {
public:
    int findLHS(vector<int>& nums) {
        int res = 0;
        unordered_map<int, int> m;
        for (int num : nums) ++m[num];
        for (auto a : m) {
            if (m.count(a.first+1)) 
                res = max(res, m[a.first] + m[a.first+1]);
        }
        return res;
    }
};
