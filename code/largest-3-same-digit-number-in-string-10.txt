class Solution {
public:
    string largestGoodInteger(string num) {
        vector<string> vec;
        for(int i=0;i<num.length()-2;i++){
            string a = "";
            if(num[i]==num[i+1] && num[i+1]==num[i+2]){
                int c = 3;
                while(c--) a+=num[i];
            }
            if(a!="") vec.push_back(a);
        }
        sort(vec.begin(),vec.end());
        return vec.size()>0?vec[vec.size()-1]:"";
    }
};
