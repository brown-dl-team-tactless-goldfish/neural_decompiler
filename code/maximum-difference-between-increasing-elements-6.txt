
int top=-1,arr[10001];
int maximumDifference(int* nums, int numsSize){
    int max=-1;
    top++;arr[top]=nums[0];
    for(int i=1;i<numsSize;i++){
        if(top!=-1&&arr[top]<nums[i]){
            if(max<nums[i]-arr[top])max=nums[i]-arr[top];
            
        }
        else{
            top++;
            arr[top]=nums[i];
        }
    }
    return max;
}
