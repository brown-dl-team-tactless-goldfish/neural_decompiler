double findMedianSortedArrays(int* nums1, int nums1Size, int* nums2, int nums2Size){
    int array[nums1Size+nums2Size];  
    memset(array,0,sizeof(int)*(nums1Size+nums2Size));
    int index1=0,index2=0;
    for(int i=0;i<nums1Size+nums2Size;i++){  //merged two array in sorted order
        if(index1==nums1Size){//no remain elements in nums1
            array[i]=nums2[index2];
            index2++;
        }else if(index2==nums2Size){//no remain elements in nums2
            array[i]=nums1[index1];
            index1++;
        }else if(nums1[index1]>nums2[index2]){// take the smaller one merge to the result array
            array[i]=nums2[index2];
            index2++;
        }else{
            array[i]=nums1[index1];
            index1++;
        }
    }
    double result=0;
    if((nums1Size+nums2Size)%2!=0){   //count the median
        result=array[(nums1Size+nums2Size)/2];
    }else{
        result=array[(nums1Size+nums2Size)/2]+array[(nums1Size+nums2Size)/2-1];
        result/=2;
    }
    return result;
    
    
}
