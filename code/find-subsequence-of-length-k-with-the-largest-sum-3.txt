    sort(nums.begin(),nums.end());
    
    vector<int> temp(nums.end()-k, nums.end());
    
    unordered_map<int,int> mp;
    
    vector<int> ans;
    
    for(auto &x: temp)
    {
        mp[x]++;
    }
    
    for(auto &x: v)
    {
        if(mp[x]--> 0)
        {
            ans.push_back(x);
        }
    }
    
    return ans;
}
