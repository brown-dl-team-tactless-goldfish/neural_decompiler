class Solution {
public:
    int minimumDifference(vector<int>& nums, int k) {
        sort(nums.begin(),nums.end());
        int ans=INT_MAX;
        int start=0;
        int end=0;
        while(end<nums.size())
        {
            if(end-start+1==k)
            {
                ans=min(ans,nums[end]-nums[start]);
                start++;
            }
            end++;
        }
        return ans;
    }
};
