
char * mergeAlternately(char * word1, char * word2){
    
    int length = strlen(word1)+strlen(word2);
    char *merge = (char *)malloc((length+1) * sizeof(char));
    int i,j,k=0;
    for(i=0,j=0;i<strlen(word1)&&j<strlen(word2);i++,j++){
        merge[k++] = word1[i];
        merge[k++] = word2[j];
    }
    while(i<strlen(word1))
        merge[k++] = word1[i++];
    while(j<strlen(word2))
        merge[k++] = word2[j++];
    merge[k] = '\0';
    return merge;
}
