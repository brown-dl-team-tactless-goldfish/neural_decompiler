bool checkZeroOnes(char * s)

{
    
    int i,c0=0,c1=0,len1=0,len0=0;
    
    
    
    for(i=0;i<strlen(s);i++)
    {
        if(s[i]=='0')
        {
            c0++;
            c1=0;
         len0=c0>len0?c0:len0;
        }
    
         else{
             
             
            c1++; 
            c0=0;
         len1=c1>len1?c1:len1;
         }
    
    }
    
    
       return len1>len0?true:false;   
    }
