int largestInteger(int num) {
    string s = to_string(num);
    
    for(int i=0;i < s.size();i++)
    {
        for(int j=i+1;j < s.size();j++)
        {
            if(s[i] < s[j] and ((s[i] - '0') % 2) == ((s[j] - '0') % 2))
            {
                char c = s[i];
                s[i] = s[j];
                s[j] = c;
            }
        }
    }
    return stoi(s);
}
