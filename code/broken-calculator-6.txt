int brokenCalc(int X, int Y){
    int currY;
    int currX;
    int nbOp = 0;
    int minMove;

    currY = Y;
    while(currY > X){

     nbOp++;

     if((currY&1) == 0)
     {
         currY = currY/2;
     }else
     {
         currY++;
     }
    };

    minMove = nbOp + (X - currY);
    
   return minMove;
}
