int minSwaps (char * s) {
	int i, odd1 = 0, even1 = 0;

	for (i = 0; s[i]; i++) {
		if (s[i] == '1') {
			if (i & 1) {
				odd1++;
			} else {
				even1++;
			}
		}
	}

	if (odd1 + even1 < i / 2 || (i + 1) / 2 < odd1 + even1) {
		return -1;
	}

	if (i & 1) {
		return odd1 + even1 == i / 2 ? even1 : odd1;
	}

	return odd1 < even1 ? odd1 : even1;
}
