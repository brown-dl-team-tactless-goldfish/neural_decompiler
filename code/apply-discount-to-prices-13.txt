using System.Text.RegularExpressions;

public class Solution {
    public string DiscountPrices(string sentence, int discount) {
        return Regex.Replace(sentence, @"((?<=\s)|^)\$(?<money>[0-9]+(\.[0-9]+)?)((?=\s)|$)", m => {
            double price = double.Parse(m.Groups["money"].Value);
            
            return $"${price - price * discount / 100.0:f2}";
        });
    }
}
