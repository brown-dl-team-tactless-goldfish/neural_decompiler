int secondHighest(char * s){
    int hash[10]={0},count=0;
    for(int i=0;i<strlen(s);i++)
    {
        if(s[i]>='0' && s[i]<='9')
        {
            hash[s[i]-48]++; //ascii
        }
    }
    for(int i=9;i>=0;i--)
    {
        if(hash[i]>0)
        {
            count++;
            if(count==2)
            {
                return i;
            }
        }
    }
    return -1;
}
