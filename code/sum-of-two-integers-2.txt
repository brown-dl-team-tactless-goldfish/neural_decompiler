int getSum(int a, int b){
    int sum;
    
    while (b) // while carry over value is not 0
    {
        sum = a ^ b; // Get sum without think about carry over
        b = (unsigned)(a&b)<<1; // Compute new carry over value to be sum in next loop
        a = sum; // Update sum to 'a' for next addition 
    }
    return a;
}
