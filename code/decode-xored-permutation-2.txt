class Solution {
public:
    vector<int> decode(vector<int>& encoded) {
        int n = encoded.size() + 1;
        vector<int> res = vector<int>(n, 0);
        res[0] = 1;
        for (int v = 2; v <= n; v++) {
            res[0] ^= v;
        }
        for (int i = 1; i < n - 1; i += 2) {
            res[0] ^= encoded[i];
        }
        for (int i = 1; i < n; i++) {
            res[i] = (res[i - 1] ^ encoded[i - 1]);
        }
        return res;
    }
};
