class Solution {
public:
    int distMoney(int money, int children) {
        int ans=0;
        if(children>money)return -1;
        
        vector<int> v(children,1);  //Give each child a dollar
        money-=children;
        
        for(int i=0;i<children-1;i++){
            if(money>6){
                ans++;
                v[i]+=7;     //Give 7 dollarrs to children if possible
                money-=7;
            }
        }
        if(money==7)ans++;    //for the last child if we have 7 dollars left

        else if (money==3){ 
            if(v[children-2]==8){
                ans--;       //if we are left with 3 dollars and all
            }                //other children have 8 dollars, it would
        }                    //disturb our count of children with 8 dollars        
        return ans;
    }
};
