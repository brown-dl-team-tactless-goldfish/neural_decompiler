class Solution {
public:
    long long waysToBuyPensPencils(int total, int cost1, int cost2) 
    {
        long long ans=0;
        for(int i=0;cost1*i<=total;i++)
        {
            int val=total;
            if(i==0)
            ans+=1+total/cost2;
            else
            {
                
                val-=cost1*i;
                ans+=1+(val/cost2);
            }
        }
        return ans;
    }
};
