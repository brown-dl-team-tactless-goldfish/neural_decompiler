bool areAlmostEqual(char * s1, char * s2){
  int diff1 = -1;
  int diff2 = -1;
  for (size_t i = 0; i < strlen(s1); i++)
  {
    if (s1[i] != s2[i])
    {
      if (diff1 == -1)
        diff1 = i;
      else if (diff2 == -1)
        diff2 = i;
      else
        return false;
    }
  }
  if (diff1 == -1)
    return true;
  else if (diff2 == -1)
    return false;
  else if (s1[diff1] != s2[diff2] || s1[diff2] != s2[diff1])
    return false;
  return true;
}
