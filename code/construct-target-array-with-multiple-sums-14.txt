    bool isPossible(vector<int>& t) {
        auto sum = accumulate(t.begin(), t.end(), (long)0);
        priority_queue<int> pq(t.begin(), t.end());
        
        while(pq.top()*2 > sum) {
            auto m = pq.top();
            pq.pop();
            pq.push(2*m - sum);
            sum = m;
        }
        return pq.top() == 1;
    }
