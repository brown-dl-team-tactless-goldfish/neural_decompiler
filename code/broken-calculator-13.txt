class Solution {
public:
    int brokenCalc(int startValue, int target) {
        int c = 0;
        int s = startValue,t = target;
        while(s<t)
        {
            if(t%2 == 1)
                t++;
            else
                t = t/2;
            c++;
        }        
        return c+s-t;   
    }
};
