int minCut(string s) {
  vector<int> dp(s.size() + 1);
  iota(begin(dp), end(dp), 0);
  for (int m = 0; m < s.size(); ++m) {
    for (int i = m, j = m + 1; i >= 0 && j < s.size() && s[i] == s[j]; --i, ++j) 
        dp[j + 1] = min(dp[j + 1], 1 + dp[i]);
    for (int i = m, j = m; i >= 0 && j < s.size() && s[i] == s[j]; --i, ++j) 
        dp[j + 1] = min(dp[j + 1], 1 + dp[i]);
  }
  return dp[s.size()] - 1;
}
