char * freqAlphabets(char * s){
    char *result = (char*)calloc(strlen(s) + 1, sizeof(char));
    int idx = 0, len = 0;
    while(idx < strlen(s))
    {
        if(idx + 2 < strlen(s) && s[idx + 2] == '#')
        {
            result[len++] = (s[idx] - '0') * 10 + (s[idx + 1] - '0') - 10 + 'j';
            idx += 3;
        }
        else
        {
            result[len++] = s[idx] - '1' + 'a';
            idx++;
        }
    }
    return result;
}
