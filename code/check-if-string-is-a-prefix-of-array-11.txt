    public bool IsPrefixString(string s, string[] words) {
    
        string test = "";
        for(int i = 0; i < words.Length; i++){
            test += words[i];
            
            if(test.Length > s.Length) return false;
            else if(test.Length == s.Length) return s == test;
        }
        return false;
    }
