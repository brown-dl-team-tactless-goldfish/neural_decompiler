typedef struct Node node;
void cnt(node *root);

node *connect(node *root) {
    if (!root)
        return NULL;
    
    cnt(root);
    return root;}

void cnt(node *root){
    if (!root->left)
        return;
    
    root->left->next = root->right;
    if (root->next)
        root->right->next = root->next->left;
    else
        root->right->next = NULL;
    
    cnt(root->left);
    cnt(root->right);}
