vector<int> nextGreaterElements(vector<int>& nums) {
	vector<int> ans;
	for(int i=0;i<nums.size();++i){
	  auto found = find_if(nums.begin()+i,nums.end(),[&](auto &rhs){return rhs>nums[i];});
	  if(found!=nums.end()) ans.push_back(*found);
	  else{
		found = find_if(nums.begin(),nums.begin()+i,[&](auto &rhs){return rhs>nums[i];});
		found==nums.begin()+i ? ans.push_back(-1) : ans.push_back(*found);
	  }
	}
	return ans;
}
