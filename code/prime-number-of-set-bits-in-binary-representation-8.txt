bool checkPrimeBits(int n)
{
    int count = 0;
    while(n > 0)
    {
        count += n & 1;
        n >>= 1;
    }
    if(count == 2 || count == 3 || count == 5 || count == 7 || count == 11 || count == 13 || count == 17 || count == 19)
        return true;
    return false;
}

int countPrimeSetBits(int left, int right){
    int result = 0;
    for(int i=left; i<=right; i++)
        if(checkPrimeBits(i)) result++;
    return result;
}
