int **transpose(int **A, int ARowSize, int const *AColSizes, int **columnSizes, int *returnSize) {
    int **result = malloc(sizeof(int *) * (*returnSize = *AColSizes));
    *columnSizes = malloc(sizeof(int) * (*AColSizes));
    for (int i = 0; i < (*AColSizes); ++i) {
        result[i] = malloc(sizeof(int) * (*(*columnSizes + i) = ARowSize));
        for (int j = 0; j < ARowSize; ++j) {
            result[i][j] = A[j][i];
        }
    }
    return result;
}
