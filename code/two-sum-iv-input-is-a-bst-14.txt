    dic = {}
    list.append(root)
    while list:
        curNode = list.pop()
        
        diff = k - curNode.val
        # print "curNode.val = ", curNode.val
        # print "need diff = ", diff
        # print "dic = ", dic
        
        if curNode.val in dic:
            return True
        else:
            dic[diff] = curNode.val
            
        if curNode.left:
            list.append(curNode.left)
        if curNode.right:
            list.append(curNode.right)
            
    return False
