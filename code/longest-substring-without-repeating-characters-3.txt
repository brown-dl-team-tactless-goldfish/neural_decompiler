
int lengthOfLongestSubstring(char * s)
{
    int a[256];
    int len = strlen(s);
    int j = -1;
    int i = 0;
    int max = 0;
    memset(a,0,256*sizeof(int));
    for (i=0; i<len; i++)
    {
        a[s[i]]++;               // -1,0
        if (a[s[i]] <2)
        {
            if (i-j > max)
                max = (i-j);
        }
        else 
        while (a[s[i]] > 1 && j<i)
        {
            j++;
            a[s[j]]--;
        }        
    }
    return max;
}
