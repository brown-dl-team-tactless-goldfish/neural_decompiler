class SmallestInfiniteSet 
{
    public:
    set<int> v;
    SmallestInfiniteSet() 
    {
        for(int i=1;i<1001;i++) v.insert(i);
    }
    
    int popSmallest() 
    {
        int result=*v.begin();
        v.erase(v.begin());
        return result;
    }
    
    void addBack(int num) 
    {
        v.insert(num);
    }
};
