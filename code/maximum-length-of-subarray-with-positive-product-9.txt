class Solution {
public:
    int getMaxLen(vector<int>& nums) {
        int ans = 0, neg = 0, pos = 0;
        for (auto t : nums) {
            neg = neg ? neg + 1 : 0;
            pos++;
            if (t < 0) swap(pos, neg);
            if (!t) neg = pos = 0;
            ans = max(ans, pos);
        }
        return ans;
    }
};
