#define mod 1000000007
class Solution {
public:
    long long power(long long x,long long y)
    {
        if (y==0) return 1;
        if (y==1) return x%mod;
        long long ans=power(x,y/2);
        ans*=ans;
        ans%=mod;
        if (y&1) ans*=x;
        ans%=mod;
        return ans;
    }
    int monkeyMove(int n1) {
        long long n=n1;
        long long ans=power(2,n);
         ans-=2;
        return (ans+mod)%mod;
    }
};
