class Solution {
public:
    int addMinimum(string word) {
        stack <int> st;
        for(char c : word) st.push(c-'a');
        int res = 0;

        while( !st.empty() ){

            // stack would have the string reversed . so "cba" is the string to be concatenated
            // finding "cba" patterns 
            
            if( !st.empty() and st.top() == 2) st.pop();
            else res++;

            if( !st.empty() and st.top() == 1) st.pop();
            else res++;

            if( !st.empty() and st.top() == 0) st.pop();
            else res++;
        }

        return res;
    }
};
