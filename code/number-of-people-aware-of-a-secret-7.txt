    int peopleAwareOfSecret(int n, int d, int f, long sm=1) {
        vector<int> dp(n+1,0);       // dp[i] people learned the secret on day i
        dp[1]=1;
        for (int i=2; i<=n; ++i) {
            for (int j=max(1, i-f+1); j<=i-d; ++j) // loop over days when secret is shared
                dp[i]=((long)dp[i]+dp[j])%1000000007;
            sm+=dp[i]-dp[max(0, i-f)];    // add new people and subtract those that forgot on this day
        }
        return sm%1000000007; 
    }
