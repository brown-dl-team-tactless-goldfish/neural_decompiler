int minIncrementForUnique(int* nums, int numsSize){
    int hist[200001] = {0};
    int ans = 0;
    int stack = 0;
    for(int i=0;i<numsSize;i++)
        hist[nums[i]]++;
        
    for(int i=0;i<200001;i++){
        ans+=stack;
        if(hist[i]>1)
            stack+=hist[i]-1;
        else if(stack>0)
            stack+=hist[i]-1;    
    }
        
    return ans;
}
