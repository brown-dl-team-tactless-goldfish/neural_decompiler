

int strStr(char * haystack, char * needle){
    int i,j,occurrence;
    for(i=0;haystack[i]!='\0';i++)                      //Iterate through a haystack
        {
            occurrence=1;                               //flag=1
            for(j=0;needle[j]!='\0';j++)                //Iterate through needle
            {
                if(needle[j]!=haystack[i+j]){           //and if we observe any mismatch then flag=0 and break through needle iteration
                    occurrence=0;
                    break;
                }
            }
        if(occurrence==1)                   // at end if flag==1 then its a perfect match so return starting index which is i.
            return i;
        }
    return -1;                              //else return -1
}
