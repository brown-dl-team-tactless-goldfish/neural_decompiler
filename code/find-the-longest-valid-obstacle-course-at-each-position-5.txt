class Solution {
public:
    vector<int> longestObstacleCourseAtEachPosition(vector<int>& obstacles) {
        
        vector<int> res(obstacles.size(), 1);
        
        vector<int> mono;
        
        for (int i = 0; i < obstacles.size(); ++i) {
            
            int a = obstacles[i];
            
            if (mono.empty() || mono[mono.size()-1] <= a ) {
                
                mono.push_back(a);
                
                res[i] = mono.size();
            } else {
                
                int index = upper_bound(mono.begin(), mono.end(), a) - mono.begin();
                
                mono[index] = a;
                
                res[i] = index + 1;
            }
        }
        return res;
    }
};
