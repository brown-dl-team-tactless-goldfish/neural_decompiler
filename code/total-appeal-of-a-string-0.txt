long long appealSum(char * s){
    int n = strlen(s) ;
    int last[26] ;       
    memset(last, -1, 26*sizeof(int));        
    long long ret = 0;
    for (int i=0; i<n; i++)
    {
        int idx = s[i] - 'a' ;
        ret += (i - last[idx] ) * (n-i) ;
        last[idx] = i;
    }       
    return ret;  
}
