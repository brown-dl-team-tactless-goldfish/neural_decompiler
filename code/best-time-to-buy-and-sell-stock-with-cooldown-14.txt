int maxProfit(int* prices, int pricesSize){
    int p0, p1, p2, t0, t1;
    p0 = 0;
    p1 = -prices[0];
    p2 = 0;

    for (int i = 1; i < pricesSize; i++){
        t0 = p0;
        t1 = p1;
        p0 = (p0 < p2) ? p2 : p0;
        p1 = (t0 - prices[i] < p1) ? p1 : t0 - prices[i];
        p2 = t1 + prices[i];}
    
    int maxp = (p0 < p1) ? p1 : p0;
    maxp = (p2 < maxp) ? maxp : p2;
    return maxp;}
