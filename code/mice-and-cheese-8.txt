class Solution {
public:
    int miceAndCheese(vector<int>& r1, vector<int>& r2, int k) {
        int res = 0;
        vector<tuple<int,int,int>> v;
        for(int i=0;i<r1.size();i++) v.emplace_back(r2[i] - r1[i], r1[i], r2[i]);
        sort(v.begin(), v.end());
        for(int i=0;i<v.size();i++) res += (i<k) ? get<1>(v[i]) : get<2>(v[i]);
        return res;
    }
};
