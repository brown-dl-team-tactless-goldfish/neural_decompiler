class Solution {
public:
    bool haveConflict(vector<string>& event1, vector<string>& event2) {
        auto intoTime = [&](string& e) {
            int res = 0;
            for(char c: e) res = res * 10 + c - '0';
            return res;
        };
        
        return !(intoTime(event1[1]) < intoTime(event2[0]) || intoTime(event1[0]) > intoTime(event2[1]));
    }
};
