class LUPrefix
{
    public:
        vector<int> dp;
    int len = 1;
    int ans = 0;
    LUPrefix(int n)
    {
        len = 1;
        ans = 0;
        dp = vector<int> (n + 10, 0);
    }

    void upload(int video)
    {
        dp[video] = 1;
        if (video == len)
        {
            ans = len;
            len++;
        }
        while (dp[len])
        {
            ans = len;
            len++;
        }
    }

    int longest()
    {
        return ans;
    }
};
