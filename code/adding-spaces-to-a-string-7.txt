    string addSpaces(string s, vector<int>& spaces) {

        int cnt=0;
        int i=0;
        string str;

        for(auto c : s){
            if(i<spaces.size() && cnt == spaces[i]){
                str.push_back(' ');
                ++i;
            }
            str.push_back(c);
            ++cnt;
        }        
        return str;
    }
