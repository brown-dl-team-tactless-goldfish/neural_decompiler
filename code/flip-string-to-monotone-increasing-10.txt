#define MIN(a,b) (a<b?a:b)

int minFlipsMonoIncr(char * s){
    
    int len = strlen(s);
    int ans = 0;
    int ones = 0;
    
    for(int i = 0; i < len; ++i)
    {
        if(s[i] == '1')
        {
            ++ones;
        } else {
            ++ans;
        }
        ans = MIN(ones,ans);
    }   
    return ans;
}
