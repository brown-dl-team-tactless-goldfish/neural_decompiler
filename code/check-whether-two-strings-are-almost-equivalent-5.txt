bool checkAlmostEquivalent(char * word1, char * word2){
    int V_w1[26]={0},V_w2[26]={0};
    int n=strlen(word1);
    for(int i=0;i<n;i++){
        V_w1[word1[i]-'a']++;
        V_w2[word2[i]-'a']++;
    }
    for(int i=0;i<26;i++){
        if(abs(V_w1[i]-V_w2[i])>3)
            return 0;
    }
    return 1;
}
