public class Solution {
    public int KthFactor(int n, int k) {
        List<int> l = new List<int>();
        l.Add(n);
        for(int i=1;i<=n/2;i++) {
            if(n%i == 0) {
                l.Add(i);
            }
        }
        l.Sort();
        return (k-1 < l.Count()) ? l[k-1] : -1;
        
    }
}
