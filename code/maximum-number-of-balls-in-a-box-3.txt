
int sumofball(int ball)
{
    int box=0;
    while(ball>0)
    {
        int r =ball%10;
        box+=r;
        ball=ball/10;
    }
    return box;
}
int countBalls(int lowLimit, int highLimit){
    int i;
    int box[50]={0};
    for(i=lowLimit;i<=highLimit;i++)
    {
        
        int r=sumofball(i);
        // printf("i : %d, r : %d\n",i,r);
        box[r]++;
    }
    int k,max=0;
    for(k=0;k<50;k++)
        max=max>box[k]?max:box[k];
    return max;
}
