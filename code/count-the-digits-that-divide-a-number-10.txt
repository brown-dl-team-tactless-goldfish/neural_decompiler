class Solution {
public:
    int countDigits(int num) {
        int currentDigit;
        int cpy = num;
        int res = 0;

        while (cpy != 0) {
            currentDigit = cpy % 10;
            if (num % currentDigit == 0)
                res++;
            cpy = cpy / 10;
        }
        return res;
    }
};
