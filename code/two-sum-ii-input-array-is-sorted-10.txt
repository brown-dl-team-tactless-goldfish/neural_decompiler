
int* twoSum(int* numbers, int numbersSize, int target, int* returnSize)
{
        *returnSize=2;
        int *a=malloc(sizeof(int)*(*returnSize)),sum=0;
        int *low=numbers,*high=numbers+numbersSize-1;
        while((*low + *high)!=target)
        {
                sum=*low + *high;
                if(sum>target)
                        high--;
                else
                        low++;
        }
        a[0]=low-numbers+1;
        a[1]=high-numbers+1;
        return a;
}
