class Solution {
public:
    string addSpaces(string s, vector<int>& sp) {
        string a = {s.begin(),s.begin()+sp[0]};
        for(int i=1;i<sp.size();i++){
            string c = {s.begin()+sp[i-1],s.begin()+sp[i]};
            a+=' ';
            a+=c;
        }
        string c = {s.begin()+sp[sp.size()-1],s.end()};
        a+=' ';
        a+=c;
        return a;
    }
};
