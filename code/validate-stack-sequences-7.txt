bool validateStackSequences(int* pushed, int pushedSize, int* popped, int poppedSize){
    int pushTop = -1, popTop = 0;
    for (int i = 0; i < pushedSize; i ++) {
        if (popTop) {
            pushed[ ++pushTop ] = pushed[i];
        }
        else {
            ++pushTop;
        }
        if (pushed[pushTop] == popped[popTop]) {
            popTop ++;
            pushTop --;
            while (pushTop >= 0) {
                if (pushed[pushTop] == popped[popTop]) {
                    popTop ++;
                    pushTop --;
                }
                else {
                    break;
                }
            } 
        }
    }
    if (popTop == poppedSize) {
        return true;
    }
    return false;
}
