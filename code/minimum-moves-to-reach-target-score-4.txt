int minMoves(int target, int maxDoubles) {
        int cnt=0;
        while(target>1){
            if(maxDoubles==0){
                cnt=cnt+target-1;
                break;
            }
            
            if(target%2==0){
                maxDoubles--;
                target/=2;
                cnt++;
            }
            else{
                cnt++;
                target--;
            }
        }
    
        return cnt;
    }
