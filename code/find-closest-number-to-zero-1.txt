  int findClosestNumber(int* nums, int numsSize)
  {
      int min = INT_MAX ;
      for ( int i = 0 ; i < numsSize ; i++)
      {
          if ( abs(nums[i] - 0) - 0 < abs(min - 0) )
          {
                  min = nums[i];
  
  
          }
          else if ( abs(nums[i] - 0) - 0 == abs(min - 0) )
          {
              if ( min < nums[i])
                  min = nums[i];
          }
          
      }
  
      return min;
  
  }
