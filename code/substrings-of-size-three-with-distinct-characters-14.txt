

int countGoodSubstrings(char * s){
    int arr[26] = {0}, count = 0, j = 0;
    if(strlen(s) < 3) return 0;
    for(int i = 0 ; i < 3 ; i++)
    {
        arr[s[i] - 'a']++;
    }
    if(arr[s[0] - 'a'] == 1 && arr[s[2] - 'a'] == 1) count++;
    for(int i = 3 ; i < strlen(s) ; i++)
    {
        arr[s[j] - 'a']--;
        arr[s[i] - 'a']++;
        j++;
        if(arr[s[i] - 'a'] == 1  && arr[s[j] - 'a'] == 1) count++;
    }
    return count;
}
