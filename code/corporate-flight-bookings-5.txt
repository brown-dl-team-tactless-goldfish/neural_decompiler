public int[] CorpFlightBookings(int[][] bookings, int n) {
	var changes = new int[n + 1];
	foreach (var booking in bookings) {
		changes[booking[0] - 1] += booking[2];
		changes[booking[1]] -= booking[2];
	}

	var count = 0;
	var result = new int[n];
	for (var i = 0; i < n; i++) {
		count += changes[i];
		result[i] = count;
	}

	return result;
}
