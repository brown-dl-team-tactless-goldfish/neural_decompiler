class Solution {
public:
    int minimumDeletions(string s) {
        int n = s.size();
        
        int dp = 0;
        int currb = 0;
        for(int i = 0; i < n; i++) {
            if(s[i] == 'a') dp = min(currb, dp + 1);
            currb += s[i] == 'b';
        }
        
        return dp;
    }
};
