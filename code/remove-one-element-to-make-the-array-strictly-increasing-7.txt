public class Solution {
    public bool CanBeIncreasing(int[] nums) {
        int c = 0;
        for (int i = 1; i < nums.Length && c < 2; ++i) {
            if (nums[i - 1] >= nums[i]) {
                c++;
                if (i > 1 && nums[i - 2] >= nums[i]) {
                    nums[i] = nums[i - 1];
                }
            }
        }
        return c < 2;
    }
}
