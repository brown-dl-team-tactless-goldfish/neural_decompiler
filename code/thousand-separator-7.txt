char * thousandSeparator(int n){
    char buf[14];
    char * outPtr = buf + sizeof(buf) - 1;
    char * retPtr;
    
    *outPtr = '\0';
    
    do{
        if((buf + sizeof(buf) - outPtr) % 4 == 0)
            *(--outPtr) = '.';

        *(--outPtr) = "0123456789"[n % 10];
    }while(n /= 10);
    
    retPtr = malloc((buf + sizeof(buf) - outPtr) + 1);
    
    return strcpy(retPtr, outPtr);
}
