class ProductOfNumbers {
private:
    vector<int> prefix = {1};
public:
    ProductOfNumbers() {
        
    }
    
    void add(int num) {
        if (num == 0)
            prefix = {1};
        else
            prefix.push_back(prefix.back() * num);
    }

    int getProduct(int k) {
        int size = prefix.size() - 1;
        if (k > size) return 0;
        
        return prefix.back()/prefix[size - k];
    }
};
