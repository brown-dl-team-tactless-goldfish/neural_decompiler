class FindSumPairs {
public:
    FindSumPairs(vector<int>& nums1, vector<int>& nums2) : n1(nums1), n2(nums2) {
        for (auto n : n2) um[n]++;
    }
    
    void add(int index, int val) {
        if(!--um[n2[index]]) um.erase(n2[index]);
        um[n2[index] += val]++;
    }
    
    int count(int tot) {
        int ans = 0;
        for (auto n : n1) if (um.count(tot - n)) ans += um[tot - n];
        return ans;
    }
private:
    vector<int> &n1, &n2;
    unordered_map<int, int> um;
};
