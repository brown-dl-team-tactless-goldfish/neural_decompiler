char * reverseVowels(char * s){
int head, tail;
char swap;
char* vowels="aeiouAEIOU";
if (strlen(s)<2) return s;
head=0;
tail=strlen(s)-1;
while (head<tail) 
    if (!strchr(vowels,s[head])) head++;
      else if (!strchr(vowels,s[tail])) tail--;
             else {
                 swap=s[head];
                 s[head++]=s[tail];
                 s[tail--]=swap;
             }
return s;
}
