class Solution {
public:
    int minimumRounds(vector<int>& tasks) {
        unordered_map<int, int> cnt;
        for (auto& t : tasks)
            cnt[t]++;
        int ans = 0;
        for (auto& t : cnt)
        {
            if (t.second == 1)
                return -1;
            int c = t.second / 3, r = t.second % 3;
            if (r == 0)
                ans += c;
            else
                ans += (c + 1);
        }
        return ans;
    }
};
