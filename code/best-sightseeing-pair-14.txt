    int maxScoreSightseeingPair(vector<int>& values) {
        // the problem is the same as finding the largest values[i] + i + values[j] - j
        // since every elements in values are positive, so the problem turns out to be 
        // find two elements, values[i] + i and values[j] - j, (i < j) that their sum is maximum
        // so values[i] + i should be the maximum value before j
        int preMax = values[0];
        int res = 0;
        for (int i = 1; i < values.size(); i++) {
            int subValue = values[i] - i;
            res = max(res, preMax + subValue);
            preMax = max(preMax, values[i] + i);
        }
        return res;
    }
