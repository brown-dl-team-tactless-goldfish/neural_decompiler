int ans = 0;
void solve(int curr, vector<int>& topping, int start, int goal){
    if(abs(goal - curr) < abs(goal - ans))
        ans = curr;
    else if(abs(goal - curr) == abs(goal - ans))
        ans = min(ans,curr);
    if(start == topping.size())
        return;
    solve(curr,topping,start+1,goal);
    solve(curr + topping[start],topping,start+1,goal);
    solve(curr + topping[start] * 2 , topping,start + 1, goal);
}
int closestCost(vector<int>& base, vector<int>& topping, int target) {
    ans = base[0];
    for(auto& b : base)
        solve(b,topping,0,target);
    return ans;
}
