class Solution {
public:
    vector<int> evenOddBit(int n) {
        int counter = 0;
        int even = 0, odd = 0;

        while(n) {
            int res = (n & 1);
            if(counter & 1) {
                if(res == 1) odd++;
            }
            else {
                if(res == 1) even++;
            }
            counter++;
            n = (n >> 1);
        }

        return {even, odd};

    }
};
