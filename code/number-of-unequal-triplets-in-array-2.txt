int unequalTriplets (int* nums, int numsSize) {
    int i, j, k, c;
    for (i = c = 0; i < numsSize; i++) {
        for (j = i + 1; j < numsSize; j++) {
            if (nums[i] != nums[j]) {
                for (k = j + 1; k < numsSize; k++) {
                    if (nums[j] != nums[k]) {
                        if (nums[i] != nums[k]) {
                            c++;
                        }
                    }
                }
            } 
            
        }
    }
    return c;
}
