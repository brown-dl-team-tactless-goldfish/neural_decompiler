"""
# Definition for a Node.
class Node:
    def __init__(self, val, left, right, next):
        self.val = val
        self.left = left
        self.right = right
        self.next = next
"""
class Solution:
    def connect(self, root: 'Node') -> 'Node':
        level_order = {}

        def helper(curNode, curV):
            if (curNode == None):
                return
            helper(curNode.left, curV+1)

            curNode.next = None
            if (curV not in level_order):
                level_order[curV] = [curNode]
            else:
                level_order[curV][-1].next = curNode
                level_order[curV].append(curNode)

            helper(curNode.right, curV+1)

        helper(root, 0)
        return root
