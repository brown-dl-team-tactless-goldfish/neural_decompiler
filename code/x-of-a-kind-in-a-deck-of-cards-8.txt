for (int i = 0; i < deckSize; i++)
{
	if (deck[i] > max)
		max = deck[i];

	if (temp[deck[i]] == 0)
	{
		count++;
	}
	temp[deck[i]]++;
}

result = calloc(count, sizeof(int));
count = 0;
for (int i = 0; i <= max; i++)
{
	if (temp[i] != 0)
	{
		result[count++] = temp[i];
		if (temp[i] < min)
			min = temp[i];
	}
}

for (int i = 2; i <= min; i++)
{
	flag = 0;
	for (int j = 0; j < count; j++)
	{
		if (result[j] % i != 0)
		{
			flag = 1;
		}
	}

	if (flag == 0)
	{
		return 1;
	}
}

return 0;
