int* preorderTraversal(struct TreeNode* root, int* returnSize){
    int *ret_buf=(int *)malloc(100*sizeof(int));
    *returnSize=0;
    
    memset(ret_buf,0,(100*sizeof(int)));
    preorder(root, ret_buf, returnSize);
           
    return ret_buf;
}
void preorder(struct TreeNode* root, int *ret_buf, int *returnSize) {
    if(root == NULL)
        return;
    ret_buf[(*returnSize)++] = root->val;
    preorder(root->left, ret_buf, returnSize);
    preorder(root->right, ret_buf, returnSize);
}
