class Solution:
    def prefixesDivBy5(self, A: List[int]) -> List[bool]:
        res = []
        s = ""
        tmpInt = 0
        for bit in A:
            tmpInt = tmpInt*2 + int(bit)
            # print ("tmpInt = ", tmpInt)
            if (tmpInt % 5 == 0):
                res.append(True)
            else:
                res.append(False)
                
        return res
