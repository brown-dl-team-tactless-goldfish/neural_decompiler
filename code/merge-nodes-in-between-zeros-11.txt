struct ListNode* mergeNodes(struct ListNode* head){
    struct ListNode *cur,*fut;
    cur=NULL;
    fut=head;
    int temp=0;
    while (fut != NULL){
        if (fut->val ==0 ){
            fut->val=temp;
            if (cur!=NULL)
                cur->next = fut;
            cur=fut;
            temp =0;
        }
        else{
            temp = temp + fut->val;
            cur->next =fut;
        }
        fut=fut->next;
    }
    head= head->next;
    return head;
}
