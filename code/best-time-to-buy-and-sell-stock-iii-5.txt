int maxProfit(int* prices, int pricesSize){
    int minCost1 = INT_MAX, minCost2 = INT_MAX, maxProfit1 = 0, maxProfit2 = 0;
    
    for(int i = 0; i < pricesSize; i++) {
        minCost1 = fmin(minCost1, prices[i]);
        maxProfit1 = fmax(maxProfit1, prices[i] - minCost1);
        
        // for 2nd transaction we can apply profit from 1st transaction
        minCost2 = fmin(minCost2, prices[i] - maxProfit1);
        maxProfit2 = fmax(maxProfit2, prices[i] - minCost2);
    }
    return maxProfit2;
}
