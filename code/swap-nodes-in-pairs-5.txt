/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     struct ListNode *next;
 * };
 */

struct ListNode* swapPairs(struct ListNode* head){
  if(!head || !head->next)
    return head;
  struct ListNode* new_head = head->next;
  struct ListNode* curr_odd = head;
  struct ListNode* curr_even = head->next;
  struct ListNode* next_odd = head;
  struct ListNode* next_even = head->next;
  while(next_even){
      next_odd = curr_even->next;
      next_even = (next_odd)? next_odd->next:NULL;
      curr_odd->next = (next_even)? next_even:next_odd;
      curr_even->next = curr_odd;
      curr_odd = next_odd;
      curr_even = next_even;
  }
  return new_head;
}
