    long long makeSimilar(vector<int>& nums, vector<int>& target) {
        auto v3 = vector(2, vector(2, vector<int>()));
        for(auto x:nums) v3[x%2][0].push_back(x);
        for(auto x:target) v3[x%2][1].push_back(x);
        for(auto& v2:v3) for(auto& v:v2) sort(v.begin(), v.end());
        long long diff = 0;
        for(auto& v2:v3) for(int i=0;i<v2[0].size();i++) diff += abs(v2[0][i] - v2[1][i]);
        return diff/4;
    }
