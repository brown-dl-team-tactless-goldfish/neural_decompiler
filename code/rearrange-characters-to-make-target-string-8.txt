class Solution {
public:
    int rearrangeCharacters(string s, string target) {
        int tar[26] = {0};
        for (auto c : target) {
            ++tar[c - 'a'];
        }
        
        int cur[26] = {0};
        for (auto c : s) {
            ++cur[c - 'a'];
        }
        
        int ans = INT_MAX;
        for (int i = 0; i < 26; ++i) {
            if (tar[i] > 0) {
                ans = min(ans, cur[i] / tar[i]);
            }
        }
        
        return ans;
    }
};
