class Solution:
    def maxScore(self, nums1: List[int], nums2: List[int], k: int) -> int:
        nums2, nums1 = zip(*sorted(zip(nums2, nums1), reverse=True))
        pq = [] 
        ans = total = 0 
        for i, (x, y) in enumerate(zip(nums1, nums2)): 
            total += x 
            heappush(pq, x)
            if i >= k: total -= heappop(pq)
            if i >= k-1: ans = max(ans, total * y)
        return ans
