public class Solution {
    public int LongestContinuousSubstring(string s) {
        int result = 1;
        
        char prior = '\0';
        int current = 0;
        
        foreach (char c in s) {
            if (c - prior == 1)
                result = Math.Max(result, ++current);
            else
                current = 1;
            
            prior = c;
        }
        
        return result;
    }
}
