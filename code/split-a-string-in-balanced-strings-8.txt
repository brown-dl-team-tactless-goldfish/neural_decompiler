int balancedStringSplit(char * s){
    int idx=1,len=strlen(s),count=0;
    char pivot=s[0];
    for(int i=1;i<len;i++)
    {
        if(pivot!=s[i])
            idx--;
        else
            idx++;
        if(idx==0)
        {
            count++;
            if(i+1<len)
                pivot=s[i];
        }
    }   
    return count;
}
