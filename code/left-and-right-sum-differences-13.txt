public class Solution 
{
    public int[] LeftRigthDifference(int[] nums) 
    {
        int n = nums.Length;
        int[] right = new int[n];
        int[] result = new int[n];

        int sum = 0;
        for (int i = n-1; i >= 0; i--)
        {
            right[i] = sum;
            sum += nums[i];
        }

        sum = 0;
        for (int i = 0; i < n; i++)
        {
            result[i] = Math.Abs(sum - right[i]);
            sum += nums[i];
        }

        return result;
    }
}
