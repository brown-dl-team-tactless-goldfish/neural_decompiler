class Solution {
public:
    int countPalindromicSubsequence(string s) {
        int count = 0;
        for (char c = 'a'; c <= 'z'; ++c) {
            int l = s.find(c);
            int r = s.find_last_of(c);
            unordered_set<char> map;
            for (int i = l + 1; i < r; ++i) {
                map.insert(s[i]);
            }
            if (l != -1) {
                count += map.size();
            }
        }
        return count;
    }
};
