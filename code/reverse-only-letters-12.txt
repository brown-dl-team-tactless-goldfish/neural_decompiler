char * reverseOnlyLetters(char * s){
int l=0;
    int r=strlen(s)-1;
    while(l<r){
        if(!isalpha(s[l])){
            l++;
        }
           else if(!isalpha(s[r])){
               r--;
           }
        else{
            char temp=s[r];
            s[r]=s[l];
            s[l]=temp;
            l++;
            r--;
        }
    }
    return s;
}
