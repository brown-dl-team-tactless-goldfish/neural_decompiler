class Solution {
public:
    int dp[(1<<16)+1][4];
    int sol(vector<int>& nums, int count, int cur){
        if(count==4){return cur==0;}
        if(dp[cur][count]!=-1)return dp[cur][count];
        int ans=0;
        for(int i=0;i<nums.size();i++){
            ans+=sol(nums,count+1,(cur&nums[i]));
        }
                     return dp[cur][count]=ans;
    }
    int countTriplets(vector<int>& nums) {
        memset(dp,-1,sizeof dp);
        return sol(nums,1,pow(2,16)-1);
    }
};
