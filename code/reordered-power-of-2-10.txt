  long getCounter(long N){
        long res=0;
        while(N){
            res+=pow(10,N%10);
            N/=10;
        }
        return res;
    }
    
    bool reorderedPowerOf2(int N) {
        long c = getCounter(N);
        for(long i=1;i<=N*10ll; i*=2){
            if(i<N/10) continue;
            if(getCounter(i)==c)
                return true;
        }
        return false;
    }
