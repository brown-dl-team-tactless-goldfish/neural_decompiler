int* createTargetArray(int* nums, int numsSize, int* index, int indexSize, int* returnSize){
    int *r = (int *) malloc(numsSize * sizeof(int));
    int i, j;
    *returnSize = numsSize;
    for (i = 0; i < numsSize; i++) {
        memmove(r + index[i] + 1, r + index[i], sizeof(int) * (numsSize - index[i] - 1));
        r[index[i]] = nums[i];
    }
    return r;
}
