#define MOD 1000000007
class Solution {
public:
    
   long long int fib(long long int n)
{
   long long int a = 0, b = 1, c, i;
    if( n == 0) return a;
    for(i = 2; i <= n; i++)
    {
       c = a%MOD + b%MOD;
       a = b;
       b = c;
    }
    return b;
}
 

    int countHousePlacements(int n) {
        
        long long int ans= fib(n+2);
        ans= (ans * ans)%MOD;
        
        
        return (int)ans;
        
    }
};
