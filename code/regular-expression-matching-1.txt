bool f(char* s, char* p){
	if(*s == '\0' && *p =='\0') return 1;
	if(*s != '\0' && *p == '\0') return 0;

	if(*(p + 1) == '*')
	{
		if(*p == *s || (*p == '.' && *s != '\0') ){
			return f(s + 1, p) || f(s, p + 2) || f(s + 1, p + 2);  
		}
		else{
			return f(s, p + 2);
		}
	}

	if(*s == *p || (*p == '.' && *s != '\0')){
		return f(s + 1, p + 1);
	}
	return 0;
}

bool isMatch(char * s, char * p){
	if (!s || !p) return 0;

	return f(s, p);
}
