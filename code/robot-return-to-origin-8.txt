  bool judgeCircle(char * moves)
  {
      
      int count = 0;
      
      for ( int i = 0 ; i < strlen(moves) ; i++)
      {
          if ( moves[i] == 'R')
              count++;
          else if ( moves[i] == 'L')
              count--;
          else if ( moves[i] == 'U')
              count += 3;
          else
              count -= 3;
  
      }
      
      
      return !count;
  
  }
