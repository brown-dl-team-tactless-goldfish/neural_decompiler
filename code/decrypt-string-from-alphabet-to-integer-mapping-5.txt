char * freqAlphabets(char * s){
    int p=0,i=0, len=strlen(s);
    char *str = (char*)malloc(sizeof(char)*len+1);
    for(p=0;p<len;p++)
        if(p<(len-2)&&s[p+2]=='#'){
            str[i++]= (s[p]-48)*10 + (s[p+1]-48) +96;
            p=p+2;
        }else
            str[i++]=s[p]+48;
    str[i]='\0';
    return str;
}
