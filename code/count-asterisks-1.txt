

int countAsterisks(char * s){
    int i,count=0;
    bool unlock=true;                   //lock variable
    for(i=0;s[i];i++)
    {
        if(s[i]=='|')                   // if we detecect any '|' then we'll lock section untill we foun another '|'
            unlock=true^unlock;
        else if(unlock && s[i]=='*')    //if it is unlocked and character is '*' then count++
            count++;
    }
    return count;                       //return count
}
