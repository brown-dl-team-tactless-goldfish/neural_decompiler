int wiggleMaxLength(int* nums, int numsSize){
    int up   = 0;
    int down = 0;
    for ( int i=1; i<numsSize; i++ ) {
        if ( nums[i] > nums[i-1] ) {
            up = down + 1;   
        }
        if ( nums[i] < nums[i-1] ) {
            down = up + 1;
        }
    }
    
    return (up > down ? up : down) + 1;
}
