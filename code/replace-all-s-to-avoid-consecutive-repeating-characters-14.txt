string modifyString(string s) {
	s = '_' + s + '_';
	string res = "";
	int n;
	char c;
	for (int i = 1; i < s.length() - 1; i++) {
		if (s[i] == '?') {
			n = 97, c = char(n);
			while (c == res.back() || c == s[i + 1])
				n++, c = char(n);
			res += c;
		} else {
			res += s[i];
		}
	}
	return res;
}
