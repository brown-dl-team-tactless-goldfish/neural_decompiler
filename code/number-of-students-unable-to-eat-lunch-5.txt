int countStudents(int* students, int studentsSize, int* sandwiches, int sandwichesSize){
    int table[2] = {0,0};
    int i;
    for(i = 0; i < studentsSize; table[students[i++]]++)
        ;
    for(i = 0; i < sandwichesSize && table[sandwiches[i]] > 0; table[sandwiches[i++]]--)
        ;
    return sandwichesSize - i;
}
