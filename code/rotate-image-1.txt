void rotate(int** mtx, int n, int* ColSize){
    for(int i=0; i < n>>1; i++){
        for(int j=i; j < n-1-i; j++){
            int a = mtx[i][j], b = mtx[j][n-1-i], c = mtx[n-1-i][n-1-j], d = mtx[n-1-j][i];
            mtx[i][j] = d;
            mtx[j][n-1-i] = a; 
            mtx[n-1-i][n-1-j] = b;
            mtx[n-1-j][i] = c;
        }
    }
}
