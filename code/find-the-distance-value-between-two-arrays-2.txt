int findTheDistanceValue(int* arr1, int arr1Size, int* arr2, int arr2Size, int d){
    
    int i,j,sun=0;
    
    for(i=0;i<arr1Size;i++)
    {
        int sum=0;
        for(j=0;j<arr2Size;j++)
        {
            if(abs(arr1[i]-arr2[j])<=d)
            {
                sum=sum+1;
            }
        }
        if(sum==0)
        {
            sun=sun+1;
        }
    }
    
    return sun;

}
