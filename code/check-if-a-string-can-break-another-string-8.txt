public bool CheckIfCanBreak(string s1, string s2) {
	s1 = Sort(s1);
	s2 = Sort(s2);

	return CanBreak(s1, s2) || CanBreak(s2, s1);

	static string Sort(string str) =>
		string.Concat(str.OrderBy(c => c));

	static bool CanBreak(string str1, string str2) =>
		str1.Select((c, i) => (c, i)).All(t => t.c >= str2[t.i]);
}
