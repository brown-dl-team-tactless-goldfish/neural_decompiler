class Solution {
public:
	vector<int> findKDistantIndices(vector<int>& a, int key, int k) {
		int n = a.size();
		vector<int>ans;
		for (int i = 0; i < n; i++) {
			if (a[i] == key) {
				for (int j = max(i - k, (ans.size() ? ans.back() + 1 : 0)); j <= min(i + k, n - 1); j++) {
					ans.push_back(j);
				}
			}
		}
		return ans;
	}
};
