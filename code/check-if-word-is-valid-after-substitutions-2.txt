bool isValid(char* S) {
    int len = strlen(S);
    if(len<3){
        return false;
    }
    
    char* x =(char*)malloc(sizeof(char)*(len+1));
    strcpy(x,S);
    int flag = 1;
    while(flag==1){
        int i;
        flag =0;
        
        for(i = 0;i<len-2;i++){
            if(x[i]=='a'&&x[i+1]=='b'&&x[i+2]=='c'){
                int j;
                for(j = i;j<=len-3;j++){
                    x[j] = x[j+3];
                }
                len-=3;
                flag =1;
                break;
                
            }
        }
        
    }
    if(len==0){
        return true;
    }
    return false;
}
