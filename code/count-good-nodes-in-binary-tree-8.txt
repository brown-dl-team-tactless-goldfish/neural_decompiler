/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     struct TreeNode *left;
 *     struct TreeNode *right;
 * };
 */


int max(int x, int y){
    return (x>=y)? x:y;
}

int goodNodes_func(struct TreeNode* root,int high){
    
    if(!root) return 0;
    return (root->val >= high) + goodNodes_func(root->left,max(root->val,high)) + goodNodes_func(root->right,max(root->val,max(root->val,high))); 
    }
    

int goodNodes(struct TreeNode* root){
    return goodNodes_func(root,INT_MIN);
}
