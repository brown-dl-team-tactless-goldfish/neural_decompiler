// Add all values to a max heap
// Pop 2 values, add their cost to final ans
// Pop the third value, and dont add the cost

class Solution {
public:
    int minimumCost(vector<int>& cost) {
        priority_queue<int> pq;
        for(int c : cost)
            pq.push(c);
        
        int ans = 0;
        while(pq.size())
        {
            int a = pq.top();
            pq.pop();
            ans+=a;
            if(pq.size())
            {
                int b = pq.top();
                pq.pop();
                ans+=b;
                if(pq.size())
                    pq.pop();
            }
        }
        return ans;
    }
};
