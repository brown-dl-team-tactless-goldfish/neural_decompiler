class Solution {
public:
    vector<int> productQueries(int n, vector<vector<int>>& q) {
        vector<int> nums,re(q.size(),1);
        for(int i = 0;i<32;++i) if(n&(1<<i)) nums.push_back(1<<i);
        for(int i = 0; i<q.size();++i){
            for(int j = q[i][0]; j<=q[i][1];++j){
                re[i] = (long long)re[i]*nums[j]%(1000000007);
            }
        }
        return re;
    }
};
