int compare(const void *a, const void *b){
    return (*(int *)a > *(int *)b)? 1: -1;
}

int getMaximumConsecutive(int* coins, int coinsSize){
    qsort(
        (void *)coins,
        coinsSize,
        sizeof(int),
        &compare
    );
    int max_num = 0;
    for(int i = 0; i < coinsSize; i++){
        if(max_num - coins[i] >= -1){
            max_num += coins[i];
        } else {
            break;
        }
    }
    return max_num + 1;
}
