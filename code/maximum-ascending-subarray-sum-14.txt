class Solution:
    def maxAscendingSum(self, nums: List[int]) -> int:
        n = len(nums)
        
        dp = []
        
        for i in range(n):
            if (i == 0):
                dp.append(nums[i])
            else:
                if (nums[i] > nums[i-1]):
                    dp.append(dp[i-1] + nums[i])
                else:
                    dp.append(nums[i])
                    
        print ("dp = ", dp)
        
        return max(dp)
