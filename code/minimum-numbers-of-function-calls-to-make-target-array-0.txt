int checkBit(int val){
    int count = 0 ;
    while(val){
        if(val & 1)
            count++ ;
        val >>= 1 ;
    }
    return count ;
}
int minOperations(int* nums, int numsSize){
    int op1 = 0 ; 
    int op2 = 0 ;
    
    for(int i = 0; i < numsSize; i++){
        op1 += checkBit(nums[i]) ;
        op2 = fmax(op2, log2(nums[i]) ) ;
    }

    return op1 + op2 ;
}
