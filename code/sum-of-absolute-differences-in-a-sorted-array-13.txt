    vector<int> getSumAbsoluteDifferences(vector<int>& nums) {
        int n=nums.size(), sumL = 0, sumR = accumulate(nums.begin(), nums.end(), 0);
        vector<int> res;
        for(int i=0;i<n;i++){
            sumR -= nums[i];
            int left = i, right = n-i-1;
            int sumDiff = abs(left*nums[i] - sumL) + abs(right*nums[i] - sumR);
            res.push_back(sumDiff);
            sumL += nums[i];
        }
        return res;
    }
