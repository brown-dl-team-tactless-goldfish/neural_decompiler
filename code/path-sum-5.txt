 return false;
else if (root->left==NULL && root->right==NULL)
{
    s=s+root->val;
    if (s==sum)
        return true;
    else
        return false;
}
else
{
    bool l=Sum(root->left,s+root->val,sum);
    bool r=Sum(root->right,s+root->val,sum);
    if (l==true ||r==true)
        return true;
    else
        return false;
}
