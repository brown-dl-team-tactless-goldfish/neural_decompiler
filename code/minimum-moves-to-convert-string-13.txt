class Solution {
public:
    int minimumMoves(string s) {
        int count=0;
        for(int i=0;i<s.size()-2;i++)
        {
            if(s[i]=='X')
            {
                count++;
                s[i]='O';
                s[i+1]='O';
                s[i+2]='O';
            }
        }
        if(s[s.size()-1]=='X' || s[s.size()-2]=='X')
            count++;
        return count;
    }
};
