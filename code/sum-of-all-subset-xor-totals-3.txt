 int subsetXORSum(vector<int>& nums) {
        int n = nums.size(), count=0, ans=0;
        int no_of_subset = pow(2, n);
    
        
        for(int i=1; i<no_of_subset; i++){
            count=i;
            int j=0, out=0;
            while(count > 0){
                if(1 & count){
                    out ^= nums[j];
                }
                j++;
                count >>= 1;
            }
            ans += out;
        }
        return ans;
    }
