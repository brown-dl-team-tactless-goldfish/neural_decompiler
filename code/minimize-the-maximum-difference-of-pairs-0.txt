class Solution {
public:
    int minimizeMax(vector<int>& nums, int p) {
        sort(nums.begin(),nums.end());
        int low = 0, high = INT_MAX,ans;
        while(low <= high){
            int mid = (low + high)/2;
            int c = 0;
            for(int i=1;i<nums.size();i++){
                if(abs(nums[i]-nums[i-1]) <= mid) c++, i++;
            }
            if(c >= p) ans = mid, high = mid-1;
            else low = mid+1;
        }
        return ans;
    }
};
