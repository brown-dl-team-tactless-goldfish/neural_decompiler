class Solution {
public:
    int minimumNumbers(int num, int k) {
        if (num == 0) {
            return 0;
        }
        
        if (k == 0) {
            return num % 10 ? -1 : 1;
        }

        for (int i = 1; i <= min(10, num / k); ++i) {
            int v = num - i * k;
            if (v % 10 == 0) {
                return i;
            }
        }
        return -1;
    }
};
