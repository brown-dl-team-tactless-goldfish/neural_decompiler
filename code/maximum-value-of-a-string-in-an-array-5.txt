bool isNumeric(char* str) {
  for (int i = 0; i < strlen(str); i++) {
    if (!isdigit(str[i])) {
      return false;
    }
  }
  return true;
}
int maximumValue(char ** strs, int strsSize){
     int max_value = 0;

  // Iterate over the array of strings.
  for (int i = 0; i < strsSize; i++) {
    // Check if the string is numeric.
    if (isNumeric(strs[i])) {
      // Convert the string to an integer and compare it to max_value.
      // If the integer is greater than max_value, update max_value to the integer.
      int value = atoi(strs[i]);
      if (value > max_value) {
        max_value = value;
      }
    }
    else {
      // If the string is not numeric, compare the length of the string to max_value.
      // If the length of the string is greater than max_value, update max_value to the length of the string.
      int value = strlen(strs[i]);
      if (value > max_value) {
        max_value = value;
      }
    }
  }

  // Return the maximum value of any string in the array.
  return max_value;

}
