public class Solution {
    public TreeNode InsertIntoMaxTree(TreeNode root, int val) {
        if(root == null) return root;
        
        return AppendNode(root, val);
    }
    
    private TreeNode AppendNode(TreeNode root, int val){
        var newNode = new TreeNode(val);
        
        if(val > root.val){
            newNode.left = root;
            
            return newNode;
        }
        
        root.right = root.right == null ? newNode : AppendNode(root.right, val);
        
        return root;
    }
}
