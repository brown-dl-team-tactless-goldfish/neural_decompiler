public class Solution {
    public int MinDeletion(int[] nums) {
        int cnt = 0;
        int last = -1;
        bool even = false;
        for (int i = 0; i < nums.Length; i++) {
            if (even && last == nums[i]) {
                cnt++;
                continue;
            }

            even = !even;
            last = nums[i];
        }
        
        return cnt + (nums.Length - cnt) % 2;
    }
}
