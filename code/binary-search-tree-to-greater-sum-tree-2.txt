 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     struct TreeNode *left;
 *     struct TreeNode *right;
 * };
 */

struct TreeNode* bstHelper(struct TreeNode*, int *);

struct TreeNode* bstToGst(struct TreeNode* root){
    int *current_total = 0;
    return bstHelper(root, &current_total);
}

struct TreeNode* bstHelper(struct TreeNode* root, int * current_total)
{
    if (root == NULL) return;
    bstHelper(root->right, current_total);
    *current_total += root->val;
    root->val = *current_total;
    bstHelper(root->left, current_total);
    return root;
}```
