// first solution
	var result = new List<int>();
	HashSet<int> map = new HashSet<int>();
	for (int i = 0; i < nums.Length; i++)
	{
		if (map.Contains(nums[i]))
		{
			map.Remove(nums[i]);
			result.Add(nums[i]);
		}
		else
			map.Add(nums[i]);
	}
	return result;

	var result = new List<int>();
	for(int i = 0; i < nums.Length; i++)
	{
		int index = Math.Abs(nums[i]) - 1;
		if (nums[index] < 0) result.Add(index + 1);
		nums[index] = -nums[index];
	}
	return result;
