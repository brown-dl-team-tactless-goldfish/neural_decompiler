int* replaceElements(int* arr, int arrSize, int* returnSize){

    int *ptr=(int*)malloc(arrSize*sizeof(int));
    *returnSize=arrSize;
    
    int i,j;
    
    for(i=0;i<arrSize-1;i++)
    {
        int sum=arr[i+1];
        for(j=i+1;j<arrSize;j++)
        {
            if(arr[j]>sum)
            {
                sum=arr[j];
            }
        }
        if(i!=arrSize-1)
        {
            ptr[i]=sum;
        }
     
    }
    ptr[arrSize-1]=-1;
    
    return ptr;
}
