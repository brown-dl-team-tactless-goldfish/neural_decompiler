

bool validMountainArray(int* A, int ASize){
    if(ASize<3){
        return false;
    }
    int ramp = 0;
    int count = 1;
    int pre = A[0];
    if(A[0]>=A[1])
    {
        return false;
    }
    for(count; count<ASize; count++){
        // on the ramp up stage
        if(ramp == 0)
        {
            if(A[count] > pre)
            {
                pre = A[count];
                continue;
            }
            else if(A[count] == pre)
            {
                return false;
            }
            else if(A[count] < pre)
            {
                ramp = 1;
                pre = A[count];
                continue;
            }
        }
        else if(ramp == 1)
        {
            if(A[count] < pre)
            {
                pre = A[count];
                continue;
            }else
            {
                return false;
            }
        }
    }
    if(ramp!=1)
    {
        return false;
    }
    return true;

}
