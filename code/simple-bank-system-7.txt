public class Bank {
    public long[] AcctBal;
    public Bank(long[] balance) {
        AcctBal = balance;
    }
    
    public bool Transfer(int account1, int account2, long money) {
            if(account1>AcctBal.Length||account2>AcctBal.Length)
            return false;    

        if(AcctBal[account1-1]>=money)
        {
            AcctBal[account1-1] -= money;
            AcctBal[account2-1] += money;
            return true;
        }
        
        return false;
    }
    
    public bool Deposit(int account, long money) {
        if(account>AcctBal.Length)
            return false;
        
        AcctBal[account-1] += money;
        return true;
    }
    
    public bool Withdraw(int account, long money) {
        if(account>AcctBal.Length)
            return false;
        if(AcctBal[account-1]>=money)
        {
            AcctBal[account-1] -= money; 
            return true;
        }
        return false;
    }
}
