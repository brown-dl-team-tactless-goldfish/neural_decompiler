class Solution {
public:
    int addMinimum(string word) {
        int n = word.size() ;
        int idx = 0 ;
        int ret = 0 ;
        while(idx < n){
                if(idx+2 < n){
                    if(word.substr(idx, 3) == "abc"){
                        idx += 3 ;
                        continue ;
                    }
                }
                if(idx+1 < n){
                    if(word.substr(idx, 2) == "ab" || word.substr(idx, 2) == "ac" || word.substr(idx, 2) == "bc"){
                        idx += 2 ;
                        ret += 1 ;
                        continue ;
                    }
                }
                ret += 2 ;
                idx++ ;       
        }
        return ret ;
    }
};
