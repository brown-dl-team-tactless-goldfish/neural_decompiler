class Solution {
public:
    #define ll long long
    long long waysToBuyPensPencils(int total, int pens, int pencils) {
        ll numPens=total/pens;
        ll res=0;
        for(ll i=0;i<=numPens;i++)
        {
            // if i buy i pens then with remaining amt find number of pencils you can buy
            ll curr=total-(pens*i);
            res+=(curr)/pencils;
            res++;
        }
        return res;
    }
};
