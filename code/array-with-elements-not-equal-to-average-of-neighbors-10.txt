    public int[] RearrangeArray(int[] nums) {
        Array.Sort(nums);
        
        int i = 0;
        int j = nums.Length-1;
        List<int> result = new List<int>();
        int h=0;
        
        while(i < j)
        {
            result.Add(nums[i++]);
            result.Add(nums[j--]);
            if (i == j)
            {
                result.Add(nums[i++]);
                 
            }
        }
        return result.ToArray(); 
    }
