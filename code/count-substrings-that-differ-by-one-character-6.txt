int countSubstrings(string s, string t) {
    int count=0, sLen=s.length(), tLen=t.length();
    for(int i=0; i<sLen; i++)
        for(int j=0; j<tLen; j++)
		{
            int flag=0, a=i, b=j;
            while(a<sLen && b<tLen && flag<2)
			{
                if(s[a++]!=t[b++])
                    flag++;
                count+=(flag==1)?1:0;;
            }
        }
    return count;
}
