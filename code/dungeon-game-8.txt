public class Solution 
{
    public int CalculateMinimumHP(int[][] d)
    {
        int c = d.Length, r = d[0].Length;
        int[] dp = Enumerable.Repeat(int.MaxValue, r + 1).ToArray();
        dp[r - 1] = 1;
        for (int i = c - 1; i >= 0; i--)
            for (int j = r - 1; j >= 0; j--)
                dp[j] = Math.Max(1 , Math.Min(dp[j], dp[j + 1]) - d[i][j]);

        return dp[0];
    }
}
