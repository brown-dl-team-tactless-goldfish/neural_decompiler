class Solution {
public:
    int scoreOfParentheses(string S) {
        
        int c = 0, ans = 0;
        for (int i = 0; i < S.length(); i++) {
            if (S[i] == '(')
                c++;
            else {
                c--;
                if (S[i-1] == '(')
                    ans += (1 << c);
            }
        }
        
        return ans;
    }
};
