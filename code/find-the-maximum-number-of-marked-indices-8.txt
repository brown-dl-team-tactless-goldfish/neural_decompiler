class Solution {
public:
    int maxNumOfMarkedIndices(vector<int>& nums) {
        if(nums.size() == 1) return 0;
        sort(nums.begin(), nums.end());
        int j = nums.size() - 1;
        int i = nums.size() / 2 - 1;
        int ans = 0;
        while(j >= 0 and i >= 0){
            if(nums[i] * 2 <= nums[j]){
                ans += 2;
                j--;
                i--;
            }
            else i--;
        }
        return ans;
    }
};
