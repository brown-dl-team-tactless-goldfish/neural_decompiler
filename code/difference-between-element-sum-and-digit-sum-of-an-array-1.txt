int differenceOfSum(int* nums, int numsSize){
    int esum=0,dsum=0;
    for(int i=0;i<numsSize;i++)
    {
        esum+=nums[i];
        if(nums[i]<10)
            dsum+=nums[i];
        else
        {
            int t=nums[i];
            while(t!=0)
            {
                dsum+=t%10;
                t/=10;
            }
        }
    }
    return(abs(esum-dsum));
}
