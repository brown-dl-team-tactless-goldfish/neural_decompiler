class Solution {
public:
    int destroyTargets(vector<int>& nums, int space) {
        vector<int>tmp = nums;
        unordered_map<int,int>ump;
        
        for(int & t : tmp)
        {
            t = t%space;
            ump[t]++;
        }
        //cal most count of most occured element
        int most_occured_count = INT_MIN;
        
        for(auto & m : ump)
        {
            most_occured_count = max(most_occured_count,m.second);
        }
        
        int res = INT_MAX;
        
        for(int i = 0;i<tmp.size();i++)
        {
            if(ump[tmp[i]]==most_occured_count)
            {
                res = min(res,nums[i]);
            }
        }
        return res;
    }
};
