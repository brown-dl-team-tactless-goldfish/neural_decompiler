bool isUgly(int num) {
    if(num<=0) return false;
    if(num==1) return true;
    while(true){
        if(num==1) return true;
        if(num%2==0){
            num=num/2;
            continue;
        }
        else if(num%3==0){
            num=num/3;
            continue;
        }
        else if(num%5==0){
            num=num/5;
            continue;
        }
        else break;
    }
    return false;
}
