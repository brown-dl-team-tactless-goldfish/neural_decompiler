class Solution {
public:
    int minDeletions(string s) {
      vector<int> chars(26);
      for (auto c : s) {
        chars[c - 'a']++;
      }
      int ret = 0;
      unordered_set<int> st;
      for (auto cnt : chars) {
        while (cnt != 0) {
          if (st.count(cnt) == 0) {
            st.insert(cnt); 
            break;
          } else {
            cnt--; 
            ret++;    
          } 
        } 
      }
      return ret;
    }
};
