class Solution {
public:
    int hardestWorker(int n, vector<vector<int>>& logs) {
        int prev = 0, longestTime = 0, ans = 0;
        for (auto& v : logs){
            if (longestTime < v[1] - prev)
                longestTime = v[1] - prev, ans = v[0];
            else if (longestTime == v[1] - prev)
                ans = min(ans, v[0]);
            prev = v[1];
        }
        return ans;
    }
};
