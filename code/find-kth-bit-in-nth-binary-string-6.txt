class Solution {
public:
    string generate(int n){
        if(n==1){
            return "0";
        }
        string s=generate(n-1);
        string changed=s;
        for(char& c : changed){
            c^=1;
        }
        reverse(changed.begin(),changed.end());
        return s+"1"+changed;
    } 
    char findKthBit(int n, int k) {
        return generate(n)[k-1];
    }
};
