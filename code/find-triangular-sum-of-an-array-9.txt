class Solution {
public:
    int triangularSum(vector<int>& nums) 
    {
        int n = nums.size();
        
        for(int i=1; i<=nums.size(); i++)
        {
            n--;
            
            for(int j=0; j<n; j++)
            {
                nums[j] = (nums[j]+nums[j+1])%10;  
            }
        }
        
        return nums[0];
    }
};
