    int maxTime = 0;
    char c = '0';
    for(int i=0;i<n;i++)
    {
        if(releaseTimes[i] > maxTime)
        {
            maxTime = releaseTimes[i];
            c = keyPressed[i];
        }
        else if(releaseTimes[i] == maxTime)
        {
            if(keyPressed[i] > c)
                c = keyPressed[i];
        }
    }
    return c;
}
