vector<int> corpFlightBookings(vector<vector<int>>& bookings, int n) {
    vector<int> result(n);
    vector<int> diffs(n, 0);
    for(auto& b: bookings) {
        diffs[b[0] - 1] += b[2];
        if (b[1] < n) diffs[b[1]] -= b[2];
    }

    for(uint i = 0; i < n; ++i)
        result[i] = i > 0 ? result[i - 1] + diffs[i] : diffs [i];

    return result;
}
