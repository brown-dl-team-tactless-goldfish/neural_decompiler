
class Solution
{
public:
    int countNumbersWithUniqueDigits(int n)
    {

        if (n == 0)
            return 1;

        int res = 0;
        
        
        //Starting from 1 so only 0 is skipped 
        for (int N = 1; N <= n; N++)
        {
            int temp = 9;
            int c = 9;
            for (int i = 2; i <= N; i++)
            {
                temp *= c--;
            }

            res += temp;
        }

        return res+1;
    }
};
