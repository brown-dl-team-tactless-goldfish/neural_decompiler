class Solution {
public:
    int splitNum(int num) {
        string s = to_string(num);
        sort(s.begin(), s.end());
        string num1 = "";
        string num2 = "";
        for(int i = 0; i < s.size(); i++){
            if(i % 2 == 0) num1.push_back(s[i]);
            else num2.push_back(s[i]);
        }
        return stol(num1) + stol(num2);
    }
};
