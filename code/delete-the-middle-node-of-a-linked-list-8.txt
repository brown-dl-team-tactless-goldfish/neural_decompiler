struct ListNode* deleteMiddle(struct ListNode* head)
{
    if(head->next==NULL)
        return NULL;
    struct ListNode *p=head;
    int k=0;
    while(p)
    {
        k++;
        p=p->next;
    }
    k=k/2-1;
    p=head;
    while(k--)
        p=p->next;
    p->next=p->next->next;
    return head;
}
