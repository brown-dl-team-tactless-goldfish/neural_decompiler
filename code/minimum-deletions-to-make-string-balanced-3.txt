class Solution {
public:
    int minimumDeletions(string s) {
        int left_b = 0, right_a = 0;
        for(auto c : s){
            if(c == 'a') right_a++;
        }

        int res = right_a;

        for(auto c : s){
            if(c == 'a') right_a--;
            if(c == 'b') left_b++;

            res = min(res, right_a + left_b);
        }

        return res;
    }
};
