class NumberContainers {
public:
    map<int, int> m;
    NumberContainers() {
        
    }
    
    void change(int index, int number) {
        m[index] = number;
    }
    
    int find(int number) {
        for(auto i : m) {
            if(i.second == number) return i.first;
        }
        return -1;
    }
};
