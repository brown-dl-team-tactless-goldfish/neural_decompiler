class Solution:
    def makeSubKSumEqual(self, arr: List[int], k: int) -> int:
        n=len(arr)
        g=gcd(n,k)
        ans=0
        nums=[]
        for i in range(g):
            lol=[]
            for j in range(i,n,g):
                lol.append(arr[j])
            lol.sort()
            nums.append(lol[len(lol)//2])
        for i in range(n):
            ans+=abs(nums[i%g]-arr[i])
        return ans
