int minDeletion(int* nums, int numsSize) {
    if (numsSize == 1) return 1;
    int top = 1, tmp = nums[0];
    for (int i = 1; i < numsSize; i++) {
        if (top % 2 == 1) {
            while (i < numsSize && tmp == nums[i]) {
                i++;
            }
        }
        if (i < numsSize) {
            top++;
            tmp = nums[i];
        }
    }
    if (top % 2 == 1) top--;
    return numsSize - top;
}
