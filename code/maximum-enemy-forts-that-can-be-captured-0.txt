int captureForts(int* forts, int fortsSize){
    int ans = 0;
    int id = 0;
    int a = 0;
    for(int i = 0; i < fortsSize; i++){
        if(forts[i] != 0){
            if( a * forts[i] < 0)
                ans = fmax(ans, i-id-1);
            a = forts[i];
            id = i;
            
        }
    }
    return ans;

}
