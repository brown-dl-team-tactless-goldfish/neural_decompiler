class Solution {
public:
    vector<bool> canEat(vector<int>& candies, vector<vector<int>>& queries) {
    vector<long> ps(1);
    vector<bool> res;
    for (auto c : candies)
        ps.push_back(ps.back() + c);
    for (auto &q : queries) {
        long fav_c = q[0], fav_d = q[1], daily = q[2];
        res.push_back(ps[fav_c + 1] > fav_d && ps[fav_c] < daily * (fav_d + 1));
    }
    return res;
}
};
