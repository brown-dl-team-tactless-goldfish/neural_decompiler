public class Solution 
{
    public int AddMinimum(string word)
    {
        int n = word.Length;
        int index = 0;
        int result = 0;
        while (index < n)
        {
            foreach (char c in "abc")
            {
                if (index >= n || word[index] != c)
                    result++;
                else
                    index++;
            }
        }

        return result;
    }
}
