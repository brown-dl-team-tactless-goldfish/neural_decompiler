int* decompressRLElist(int* nums, int numsSize, int* returnSize){
    int *newarr=malloc(sizeof(int)*(numsSize/2)*100);
    *returnSize=0;
    for(int i=0;i<numsSize;i=i+2){
        for(int j=0;j<nums[i];j++){
            newarr[*returnSize]=nums[i+1];
            (*returnSize)++;}}
    return newarr;}
