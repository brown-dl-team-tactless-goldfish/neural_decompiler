class Solution {
public:
    double trimMean(vector<int>& arr) {
    sort(arr.begin() , arr.end());
    double x=.05*arr.size();
    for(int i=0 ; i<x ; i++)
        arr.erase(arr.begin());
    for(int i=0 ; i<x ; i++)
         arr.erase(arr.end()-1);
    x=(double)accumulate(arr.begin() , arr.end(),0);
    return (x/arr.size());
        
    }
};
