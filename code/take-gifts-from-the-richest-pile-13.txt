class Solution {
public:
    long long pickGifts(const vector<int>& gifts, int k) {

        priority_queue<int> pq(gifts.begin(),gifts.end());
        
        while(k--) {
            pq.push(floor(sqrt(pq.top())));
            pq.pop();
        }

        long long sum = 0;
        while(!pq.empty()) {
            sum += pq.top();
            pq.pop();
        }

        return sum;
    }
};
