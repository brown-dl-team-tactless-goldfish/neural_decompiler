class Solution {
public:
    int countDistinctIntegers(vector<int>& nums) {
        set<int> st;
        for(auto it : nums){
            st.insert(it);
        }
        for(auto it : nums){
            int t=it,rev=0;
            while(t){
                rev = (rev*10)+(t%10);
                t/=10;
            }
            st.insert(rev);
        }
        return st.size();
    }
};
