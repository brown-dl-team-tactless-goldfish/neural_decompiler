class Solution {
public:
    int maxSatisfied(const vector<int>& customers, const vector<int>& grumpy, int X) {
        
        int res=0, cust=0, maxCust=0;
        for(int i=0, k=-X; i<grumpy.size(); ++i, ++k) {
            grumpy[i]?cust+=customers[i]:res+=customers[i];
            if(k>-1 and grumpy[k]) cust-=customers[k];
            maxCust=max(maxCust,cust);
        }
        return res+maxCust;   
    }
};
