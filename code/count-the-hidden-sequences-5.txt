    int numberOfArrays(vector<int>& differences, int lower, int upper) {
        long long xmin = 0, xmax = 0, sum = 0;
        for(int x:differences){
            sum+=x;
            xmin = min(xmin, sum);
            xmax = max(xmax, sum);
        }
        return max<long long>(0, upper - lower - (xmax -xmin) + 1);
    }
