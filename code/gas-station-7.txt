int canCompleteCircuit(int* gas, int gasSize, int* cost, int costSize){
    
    int overallFuel = 0;
    int overallFuel_2 = 0;
    int start = 0;
    int pos = 0;
    for(int i=0;i<gasSize;i++){
        overallFuel += gas[i] - cost[i];
        overallFuel_2 += gas[i] - cost[i];
        if(overallFuel<0){
            overallFuel = 0;
            start = i+1;
        }

    }
    return overallFuel_2>=0?start:-1;

}
