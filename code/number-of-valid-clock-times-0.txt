int countTime(char * time){
    int ans = 1;
    if(time[0] =='?' && time[1] =='?') 
        ans *= 24;
    else{
        if(time[1] =='?'){
            if( time[0] == '1'|| time[0] == '0') 
                ans *= 10;
            else 
                ans *= 4;
        }
        else if( time[0] == '?'){
            if( time[1] >= '4') 
                ans *= 2;   //0 , 1, 
            else 
                ans *= 3;   //0 , 1 , 2
        }
    }
    if( time[3] =='?' && time[4] =='?') 
        ans *= 60;
    else{
        if( time[3]  =='?') 
            ans *= 6 ;
        else if( time[4] == '?')
            ans *= 10;
    }
        
    return ans;

}
