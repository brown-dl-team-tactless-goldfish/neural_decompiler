class Solution
{
public:

    int longestOnes( vector<int>& A, int K, int res=0 )
    {
        for( auto L{ A.begin() }, R{ L }; R != A.end(); res = max( res, (int) distance( L,R ) ) )
            if( *R == 1 )
                ++R;
            else
            if( *R == 0 && K > 0 )
                ++R, --K;
            else
            if( *R == 0 && K == 0 && *L++ == 0 )
                ++K;
        return res;
    }

};
