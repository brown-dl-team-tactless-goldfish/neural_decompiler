
int maxScore(char * s){
    int max = 0;
    int len = strlen(s);
    int one, zero;
    one = zero = 0;
    for (int i = 0; i < len; i++)
        if (s[i] == '1')
            one++;
    for (int i = 0; i < len - 1; i++)
    {
        if (s[i] == '0') 
            zero++;
        if (s[i] == '1')
            one--;
        if (one + zero > max)
            max = one + zero;
    }
    return max;
}
