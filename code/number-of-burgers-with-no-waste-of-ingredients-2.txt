vector<int> numOfBurgers(int t, int c) {
	if (t % 2 != 0 || 4 * c < t || 2 * c > t)
		return {};
	// x - jumbo, y - small
	// 1. t = 4 * x + 2 * y;
	//    c = x + y = > y = (c - x)
	// 2. t = 4 * x + 2 (c - x) => 4 * x + 2 * c - 2 * x =>
	// t = 2 * x + 2 * c =>
	// x = (t - 2 * c) / 2
	// 3. y = c - x
	int jumbo = (t - 2 * c) / 2;
	int small = c - jumbo;
	return { jumbo, small };
}
