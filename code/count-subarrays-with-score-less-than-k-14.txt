    long long countSubarrays(vector<int>& nums, long long k) {
        using ll = long long;
        int l = 0;
        ll sum = 0;
        ll res = 0;
        for (int r = 0; r < nums.size(); r++) {
            sum += nums[r];
            while (sum * (r-l+1) >= k) {
                sum -= nums[l++];
            }
            res += r - l + 1;
        }
        return res;
    }
