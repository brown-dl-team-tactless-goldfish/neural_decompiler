int jump(int* nums, int numsSize) {
    int next_r = nums[0];
    int r = next_r;
    int jumps = 0;
    
    for(int i=1;i<numsSize;++i){
        if(--next_r < nums[i]) 
            next_r = nums[i];
        if(i == r || i == numsSize-1){
            jumps++;
            r += next_r;
        }
    }
    return jumps;
}
