class Solution {
public:
    int partitionArray(vector<int>& nums, int k) 
    {
        sort(nums.begin(),nums.end());
        int cnt = 0;
        int flag = 1;
        int n = nums.size();
        int mn = INT_MAX;
        
        for(int i=0;i<n;i++)
        {
            // if our flag is 1 it means we are ready with new subsequence
            if(flag==1)
            {
                flag = 0;
                mn = nums[i];
                cnt++;
            }
            
            // if our consecutive diff is greater than k then we need to start new subsequence
            if(nums[i]-mn > k)
            {
                flag = 1;               // made flag 1 to start the new subsequence
                mn = INT_MAX;           // make again mn as max
                i--;                    // move curr pointer , 1 step behind
            }
        }
        return cnt;
        
    }
};
