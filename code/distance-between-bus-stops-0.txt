

int distanceBetweenBusStops(int* distance, int distanceSize, int start, int destination){
    int total = 0;
    int gap = 0;
    int min = fmin(start, destination);
    int max = fmax(start, destination);
    for(int i = 0; i < distanceSize; i++){
        total += distance[i] ;
        if(i >= min && i < max)
            gap += distance[i];
    }
    return fmin(gap, total - gap);
}
