int rob(int* nums, int numsSize){
    int result = 0;
    int max[numsSize];
    for ( int i = 0 ; i < numsSize ; i++ )
    {
        int opt1, opt2;
        if ( (i - 3) < 0 ) opt1 = 0;
        else opt1 = max[i-3];
        if ( (i - 2) < 0 ) opt2 = 0;
        else opt2 = max[i-2];
        
        max[i] = ( opt1 > opt2 ) ? opt1 + nums[i] : opt2 + nums[i];
        if ( max[i] > result ) result = max[i];
    }
    
    return result;
}
