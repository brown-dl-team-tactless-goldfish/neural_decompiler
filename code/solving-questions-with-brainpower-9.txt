class Solution {
public:
    long long dp[100001];
    long long f(vector<vector<int>> &a, int ind, int n) {
        if(ind>=n) return 0;
        if(dp[ind] != -1) return dp[ind];
        long long int x=f(a, ind+1, n);
        long long int y=a[ind][0]+f(a, ind+a[ind][1]+1, n);
        return dp[ind] = max(x,y);
    }
    long long mostPoints(vector<vector<int>>& a) {
        int n=a.size();
        if(n==1) return a[0][0];
        memset(dp, -1, sizeof(dp));
        return f(a,0,n);
    }
};
