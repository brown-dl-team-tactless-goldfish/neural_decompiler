int* decompressRLElist(int* nums, int numsSize, int* returnSize){
    
    *returnSize = 0;
    int *result = malloc(numsSize / 2 * sizeof(int) * 100);
        
    for (int i = 0; i < numsSize; i += 2)
        for (int j = 0; j < nums[i]; j++)
            result[(*returnSize)++] = nums[i+1];
    
    return result;
}
