int* sortArrayByParityII(int* nums, int numsSize, int* returnSize){
    int * result = (int *)malloc(numsSize*sizeof(int));
    int idxEven = 0, idxOdd = 1;
    for(int i=0; i<numsSize; i++)
        if(nums[i] % 2 == 0)
        {
            result[idxEven] = nums[i];
            idxEven += 2;
        }
        else
        {
            result[idxOdd] = nums[i];
            idxOdd += 2;
        }
    *returnSize = numsSize;
    return result;
}
