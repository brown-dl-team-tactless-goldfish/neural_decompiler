void postorder(struct TreeNode *root, int* out, int* size)
{
    if(root == NULL) return;
    postorder(root->left, out, size);   
    postorder(root->right, out, size);
    out[(*size)++] = root->val;
}
int* postorderTraversal(struct TreeNode* root, int* returnSize){
    int * result = malloc(1000*sizeof(int));
    *returnSize = 0;
    postorder(root, result, returnSize);
    return result;
}
