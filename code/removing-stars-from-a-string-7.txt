char * removeStars(char * s){
    int top = -1, len = strlen(s);

    for (int i = 0; i < len; i ++) {
        if (s[i] == '*') {
            if (top == -1)
                continue;
            top --;
        }
        else {
            s[++ top] = s[i];
        }
    }
    s[top + 1] = '\0';
    return s;
}
