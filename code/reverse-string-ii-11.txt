

void rev(char *s, int l, int k, int size ){
    
    int r = (l + k) - 1 <= size ? (l + k) - 1 : size;
    
    while(l < r){
        char temp = s[l];
        s[l] = s[r];
        s[r] = temp;
        l++;
        r--;
    }
}
char * reverseStr(char * s, int k){
    
    int len = strlen(s);
    
    for(int i = 0; i < len; i += 2*k){
        
        rev(s, i, k, len - 1);
        
    }
    
    return s;
}
