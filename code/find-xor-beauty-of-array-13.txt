class Solution {
public:
    int xorBeauty(vector<int>& nums) {
        if(nums.size()==1)return nums[0];
        int i;
        for(i = 0; i < nums.size()-1; i++){
            if(nums[i]!=nums[i+1])break;
        }
        if(i==nums.size()-1)return 0;
        int ans = 0;
        for(auto &i: nums) ans |= i;
        for(auto &i: nums) ans &= i;
        for(auto &i: nums) ans ^= i;
        return ans;
    }
};
