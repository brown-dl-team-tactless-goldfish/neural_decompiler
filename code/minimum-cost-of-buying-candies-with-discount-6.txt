class Solution {
public:
    int minimumCost(vector<int>& cost) {
        sort(cost.begin(),cost.end());
        int count=0;
        for(int i=cost.size()-1; i>=0; i-=3){
            if(i-2>=0){
                count+=(cost[i]+cost[i-1]);
            }
            else if(i-1>=0){
                count+=(cost[i]+cost[i-1]);
            }
            else{
                count+=cost[i];
            }
        }
        return count;  
    }
};
