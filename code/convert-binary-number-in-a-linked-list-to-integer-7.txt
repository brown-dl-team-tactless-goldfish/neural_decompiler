int getDecimalValue(struct ListNode* head){
    struct ListNode* ptr=head;
    int sum=0,len=0;
    
    if (!head)
        return 0;

    while(ptr)
    {
        ptr=ptr->next;
        len++;
    }
    ptr=head;
    while(len && ptr)
    {
        len--;
        sum = sum + pow(2,len)*ptr->val;
        ptr=ptr->next;
    }
    return sum;
}
