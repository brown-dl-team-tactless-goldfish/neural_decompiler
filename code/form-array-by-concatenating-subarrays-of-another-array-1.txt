

bool canChoose(int** groups, int groupsSize, int* groupsColSize, int* nums, int numsSize){
    
    for (int g = 0, i = 0; i < numsSize && g < groupsSize; i++) {
        
        // Get the size of the current group.
        
        int gs = groupsColSize[g];
        if (gs + i > numsSize) {
            break;
        }
        
        // Check if the current group is a subarray of nums,
        // starting from position "i".
        
        int found = 1;
        for (int ii = 0; ii < gs; ii++) {
            if (groups[g][ii] != nums[i + ii]) {
                found = 0;
                break;
            }
        }
        
        // If the group is a subarray, move onto the next group
        // and move "i" to the end of the subarray.
        
        if (found) {
            
            g += 1;
            i += gs - 1;
            
            // Check if all the groups have been placed.
            
            if (g >= groupsSize) {
                return true;
            }
            
        }
    }
    
    return false;
    
}
