class Solution {
public:
    vector<int> divisibilityArray(string word, int m) {
        long long num = 0;
        vector <int> ans;
        for(int i=0;i<word.length();i++){
            num = (num*10 + (word[i]-'0')) % m;
            if(num % m)
                ans.push_back(0);
            else
                ans.push_back(1);
        }
        return ans;
    }
};
