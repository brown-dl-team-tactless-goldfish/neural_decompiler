int findSpecialInteger(int* arr, int arrSize){
    int percentual = ((arrSize * 25) / 100);
    int number = arr[0];
    int count;
    
    for (int i = 0; i < arrSize; i++) {
        count = 1;
        for (int j = i+1; j < arrSize; j++)
            if (arr[i] == arr[j])
                count++;
        if (count > percentual)
            number = arr[i];
    }
    return number;
}
