class Solution {
public:
    string removeOccurrences(string s, string part) {
        string ans;
        for(char c : s) {
            ans += c;
            if(ans.size() < part.size()) continue;
            if(ans.substr(ans.size() - part.size()) == part) ans.erase(ans.size() - part.size(), part.size());
        }
        return ans;
    }
};
