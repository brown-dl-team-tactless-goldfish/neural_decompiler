public class Solution {
    public int WateringPlants(int[] p, int c) {
        int r = 0, w = c;
        for (int i = 0; i < p.Length; ++i) {
            if (p[i] <= w) {
                w -= p[i];
                r++;
                
            } else {
                w = c - p[i];
                r += 2 * i + 1;
            }
        }
        return r;
    }
}
