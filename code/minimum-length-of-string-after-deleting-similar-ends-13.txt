class Solution {
public:
    int minimumLength(string s) {
        int n = s.length();
        int i = 0, j = n - 1;
        while (i < j && s[i] == s[j]) {
            char c = s[i];
            while (s[i] == c && i < j) i++, n--;
            while (j >= i && s[j] == c) j--, n--;
        }
        return n;
    }
};
