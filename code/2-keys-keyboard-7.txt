    dp = [0 for i in range(n+1)]
    # print("dp = ", dp)

    for i in range(2, n+1):
        # print("i = ", i)
        dp[i] = i
        for j in range(i-1, 1, -1):
            # print("j = ", j)
            if i % j == 0:
                dp[i] = dp[j] + int(i/j)
                break

    # print("end dp = ", dp)
    return dp[n]
