class Solution {
public:
    int secondHighest(string s) {
        vector<bool> num(10);
        for (auto& c : s)
            if (isdigit(c)) num[c - '0'] = true;
        bool foundOne = true;
        for (int i = num.size(); i > -1; --i){
            if (num[i]){
                if (foundOne) foundOne = false;
                else return i;
            }
        }
        return -1;
    }
};
