public class Solution {
    public double AngleClock(int hour, int minutes) {
        
        double hourHandAnglePerHour = 360.0 / 12;
        double minuteHandAnglePerMin = 360.0 / 60;
        
        double hourHandAngle = (hour % 12 + minutes / 60.0) * hourHandAnglePerHour;
        double minuteHandAngle = minutes * minuteHandAnglePerMin;
    
        double res = Math.Abs(hourHandAngle - minuteHandAngle);
        return res > 180? 360.0 - res : res;
    }
}
