int minFlipsMonoIncr(char* S) {
    int count = 0, max = 0, ret = 0, front1 = 0;
    int len = strlen(S);
    for( int i = 0; i < len; i++ ) {
        if( S[i] == '0' ) {
            count++;
            ret++;
        } else {
            count--;
            front1++;
        }
        if( max < count ) {
            max = count;
            ret = front1;
        }
    }
    return ret;
}
