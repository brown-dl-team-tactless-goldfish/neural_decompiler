TreeNode* reverseOddLevels(TreeNode* root) 
{
	deque<TreeNode*> q={root};
	int c=0;
	while(!q.empty())
	{
		c++;
		int s=q.size();
		for(int i=0;i<s;i++)
		{
			TreeNode* p=q.front();
			q.pop_front();
			if(p->left&&p->right) q.push_back(p->left),q.push_back(p->right);
		}
		if(c%2==1)
		{
			int t=q.size();
			for(int i=0;i<t/2;i++) swap(q[i]->val,q[t-i-1]->val);
		}
	}
	return root;
}
