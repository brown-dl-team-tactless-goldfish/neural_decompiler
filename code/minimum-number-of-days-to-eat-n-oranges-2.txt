map<int,int> dp;


int help(int n)
{
    if(dp.count(n))
    {
        return dp[n];
    }
    
    int res = INT_MAX;
    
    if(n % 2 == 0)
    {
        res = min(res,help(n/2));
    }
    
    if(n % 3 == 0)
    {
        res = min(res,help(n/3));
    }
    
    if(n % 2 != 0 || n % 3 != 0)
    {
        res = min(res,help(n-1));
    }
    
    dp[n] = res+1;
    
    return dp[n];
}

int minDays(int n) 
{
    dp[0] = 0;
    dp[1] = 1;
    
    return  help(n);
    
    
}
