class Solution {
public:
    vector<long long> sumOfThree(long long num) {
        vector<long long>res;
        
       
        if((num-3)%3!=0)
        {
            return res;
        }
        long long ansi=(num-3)/3;
        res.push_back(ansi);
        res.push_back(ansi+1);
        res.push_back(ansi+2);
        return res;
    }
};
