class Solution {
private:
    vector<int> v;    
public:
    Solution(vector<int>& w) {
        double sum = 0;
        for (int num : w) sum += num;
        for (int i = 0; i < w.size(); i++) {
            double num = w[i];
            int cnt = ceil((num/sum)*100.0);
            for (int j = 0; j < cnt; j++) v.push_back(i); 
        }
    }
    
    int pickIndex() {
        return v[rand()%v.size()];
    }
};
