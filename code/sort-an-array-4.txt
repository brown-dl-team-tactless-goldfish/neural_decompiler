int cmpfunc(const void* a,const void* b){
    return (*(int*)a - *(int*)b);
}


int* sortArray(int* nums, int numsSize, int* returnSize){
    
    int *res = (int*)calloc(numsSize,sizeof(int));
    
    memmove(res,nums,sizeof(int)*numsSize);
    
    qsort(res,numsSize,sizeof(int),cmpfunc);
    
    *returnSize = numsSize;
    return res;
}
