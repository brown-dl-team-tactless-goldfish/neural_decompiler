int maxScore(char * s){
    int left = 0;
    int right = 0;
    int maxSum = 0;
    
    for(int i = 0; i < strlen(s) - 1; i++){
        if(s[i] == '0'){
            left++;
        }
        
        for(int j = i + 1; j < strlen(s); j++){
            if(s[j] == '1'){
                right++;
            }
        }
        
        if((left + right) > maxSum){
            maxSum = left + right;
        }
        
        right = 0;
    }
    
    return maxSum;
}
