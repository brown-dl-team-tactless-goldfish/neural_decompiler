int cmp(const void *a, const void *b){ 
    if((*(int*)a) >= (*(int*)b))
        return 1;
    else
        return -1;
}
bool findSubarrays(int* nums, int numsSize){
    for(int i = 0; i < numsSize-1; i++)
        nums[i] += nums[i+1];
    qsort(nums, numsSize-1, sizeof(int), cmp);
    for(int i = 0; i < numsSize - 2; i++){
        if(nums[i]==nums[i+1])
            return true;
    }
    return false;
}
