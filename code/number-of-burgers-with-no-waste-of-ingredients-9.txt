class Solution:
    def numOfBurgers(self, tomatoSlices: int, cheeseSlices: int) -> List[int]:
        v = []
        t = (tomatoSlices-2*cheeseSlices)
        c = (4*cheeseSlices-tomatoSlices)
        if(t>=0 and c>=0):
            if(t%2==0 and c%2==0):
                v.append(t//2)
                v.append(c//2)
        return v
