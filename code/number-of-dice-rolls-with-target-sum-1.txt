int numRollsToTarget(int n, int k, int target){
    int mod = 1e9 + 7;
    long long dp[n + 1][target + 1];
    memset(dp, 0, sizeof(long long)*(n + 1)*(target + 1));
    dp[0][0] = 1;
    for(int i = 1; i <= n; i++){
        for(int j = 1; j <= k; ++j){
            for(int f = j; f <= target; ++f){
                dp[i][f] = (dp[i][f] + dp[i - 1][f - j]) % mod;
            }
        }
    }
    return (int)dp[n][target]; 
}
