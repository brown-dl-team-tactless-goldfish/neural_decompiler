int alternateDigitSum(int n){
    int sum = 0, countDigits = (int)log10(n) + 1;
    bool isNegative = (countDigits % 2 == 0) ? true : false;
    while (n != 0){
        int digit = n % 10;
        n = n / 10;
        sum += (isNegative) ? -digit : digit;
        isNegative = !isNegative; 
    }
    return sum;
}
