int findKthPositive(int* arr, int arrSize, int k){
    int i=0;
    int x=1;
    int missingNumbers=arr[arrSize-1]-arrSize;
    if(missingNumbers<k)return (k-missingNumbers+arr[arrSize-1]);
    while(1){
        if(arr[i]==x)i++;
        else k--;
        if(k==0)return x;
        x++;
    }
    return -1;
}
