

int maxProduct(int* nums, int numsSize){
   
    if(numsSize ==1)return nums[0];
    int product=nums[0];
    int p;
    for(int i =0; i<numsSize; i++)
        
    { 
      p = nums[i];  // initial product
     if(product<p) product=p;
     for (int j=1+i, t=0; j<numsSize; j++) // next element index
     {
         
             p *= nums[j]; // mjultiplying the subarray to next element
         if(product<p)
             product =p;
             
         }
     
   

        
    }
    
return product;
    
}
