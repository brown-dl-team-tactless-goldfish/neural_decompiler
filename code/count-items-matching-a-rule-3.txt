         #define TYPE 0
         #define COLOR 1
         #define NAME   2
         
         int countMatches(char *** items, int itemsSize, int* itemsColSize, char * ruleKey, char * ruleValue)
         
         {
             char key= 0;
             int count = 0;
            // check the rule key
          
             switch(ruleKey[0])
             {
                 case 't':  key = TYPE; break;
                 case 'c':  key = COLOR; break;
                 case 'n':  key = NAME; break;
             }
             
             // loop for each item to check the value with the item value
             for ( int i = 0 ; i < itemsSize ; i++)
             {
                 if ( !strcmp( *(*(items + i)+key) , ruleValue ))
                 {
                     count++;
                 }
             }
             
             return count;
         }
