#define N 1001
int findMaxK (int* nums, int numsSize){
    int map[N] = {0};
    int max = -1;
    
    for (int i = 0; i < numsSize; i++) {
        if (map[abs(nums[i])] == -nums[i]) {
            if (abs(nums[i]) > max)
                max = abs(nums[i]);
        }
        map[abs(nums[i])] = nums[i];   
    }
    return max;
}
