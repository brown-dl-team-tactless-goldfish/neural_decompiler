class Solution {
public:
    int numSubarraysWithSum(vector<int>& A, int S) {
        
        int n = A.size();
        int sum = 0, res = 0;
        unordered_map<int, int>hm;
        hm[0] = 1;
        
        
        for (int i = 0; i < n; i++) {
            sum += A[i];
            if (hm.find(sum-S) != hm.end())
                res += hm[sum-S];
            hm[sum]++;
        }
        
        return res;
    }
};
