int getWinner(int* arr, int arrSize, int k){
    int max = arr[0] ;
    int cn = 0 ;
    
    for(int i = 1; i < arrSize; i++){
        if(arr[i] < max){
            cn++ ;
        }
        else{
            cn = 1 ;
            max = arr[i] ;
        }
        if(cn == k)
            return max ;
    }
    
    return max;
}
