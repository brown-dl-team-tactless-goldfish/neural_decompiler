#define min(a, b) a < b ? a : b;

int distributeCandies(int* candyType, int candyTypeSize){
    int i, hashCandyTypes[200001] = {0}, differentCandyTypes = 0;

    for (i = 0; i < candyTypeSize; i++)
        if ((hashCandyTypes[candyType[i] + 100000]+=1) == 1)
            differentCandyTypes++;        

    return min(differentCandyTypes, candyTypeSize / 2);
}
