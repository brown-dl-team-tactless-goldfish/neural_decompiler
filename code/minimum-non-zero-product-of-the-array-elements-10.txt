    public int MinNonZeroProduct(int p) {
        
        const long MOD = 1000000007;
        
        long count = (1L << p) - 1;
        
        long product = (long)BigInteger.ModPow(count - 1, count >> 1, MOD);
        product *= (count % MOD);
        
        
        return (int)(product % MOD);
    }
