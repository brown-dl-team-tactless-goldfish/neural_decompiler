int numPairsDivisibleBy60(int* time, int timeSize){
    int i = 0, j = 0, count = 0;
    int store[60] = {0};
    
    for (i = 0; i < timeSize; i++) {
        store[time[i] % 60]++;
    }
    
    count += (store[0] * (store[0]-1))/2;
    
    for (i = 1; i < 30; i++) {
        count += store[i] * store[60 - i];
    }
    
    count += (store[30] * (store[30]-1))/2;
    
    return count;
}
