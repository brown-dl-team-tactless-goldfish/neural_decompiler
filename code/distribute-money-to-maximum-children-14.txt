
class Solution {
public:
    int distMoney(int m, int c) {
        if(c>m) return -1;
        if(m<8) return 0;
        int d = m/8,  r = m%8;
        if(c==d && r==0) return d;
        if(d>=c) return c-1;
        if(r==4 && (c-d)==1) return c-2;
        if(r>=(c-d)) return d;
        int l=c-d;
        while(r<l){
            l++; r+=8; d--;
            }
        
        return d;
    }
};
