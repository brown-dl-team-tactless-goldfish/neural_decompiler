class Solution {
public:
    int longestSubarray(vector<int>& nums) {
        int mxel = INT_MIN, ans = 0, curr = 0;
        for(int i : nums) mxel = max(mxel, i);
        
        for(int i=0; i<nums.size(); i++) {
            while(i < nums.size() && nums[i] == mxel) curr++, i++;
            ans = max(ans, curr);
            curr = 0;
        }
        return ans;
    }
};
