int minSteps(char * s, char * t){
    
    int sa[26]={0};
    int i=0,j=0;
    int idx;
    int r=0;
    while (s[i]!=NULL)
    {
        idx = s[i]-'a';
        sa[idx]++;
        idx = t[i]-'a';
        sa[idx]--;
        i++;
    }
    
    for (i=0; i < 26; i++)
    {
        if (sa[i] > 0)
        {
            r+=sa[i];
        }
    }
    
    return r;
}
