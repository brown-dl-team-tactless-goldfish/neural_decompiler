int no_of_distinct_digits = 0;

for(auto s : digits)
    no_of_distinct_digits++;

int temp = n, no_of_digits_in_ub = 0;

while(temp>0){
    temp/=10;
    no_of_digits_in_ub++;
}

int res = 0;

for(int i=1; i<no_of_digits_in_ub; ++i)
    res += (int)pow(no_of_distinct_digits, i);

string s = "";

while(n>0){
    char c = n%10 + '0';
    s = c + s;
    n/=10;
}

// s = string form of n

res += helper(digits, s, no_of_distinct_digits);

return res;
