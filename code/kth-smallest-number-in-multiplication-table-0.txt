int helper(int m, int n, int p){
    int sum = 0 ;
    for(int i = 0; i < m; i++){
        sum += fmin(n, p/(i+1)) ;
    }
    return sum ;    
}
int findKthNumber(int m, int n, int k){
    if(k == 1 || k== m*n)
            return k ;
    int left = 1, right = m*n;
    while(left < right){
        int mid = left + (right-left)/2 ;
        int count = helper(m, n, mid) ;
        if(count >= k)
            right = mid ;
        else
            left = mid + 1 ;
    }
    return left ;
}
