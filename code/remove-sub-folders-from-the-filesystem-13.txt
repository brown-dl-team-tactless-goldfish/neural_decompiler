class Solution {
public:
    vector<string> removeSubfolders(vector<string>& folder) {
        int n = folder.size();
        
        vector<string> ret;
        
        std::sort(folder.begin(), folder.end());
        
        for (int i = 0; i < n; ++i) {
            string & currStr = folder[i];
            if ( ret.size() == 0 || (currStr.rfind(ret.back() + "/", 0) != 0) ) {
                ret.push_back(currStr);
            }
        }
        
        return ret;
    }
};
