struct TreeNode* increasingBST(struct TreeNode* root){
    if(root->right)
        root->right=increasingBST(root->right);
    if(root->left){
        root->left=increasingBST(root->left);
        struct TreeNode*temp=root->left;
        struct TreeNode*next=temp->right;
        while(next){
            temp=next;
            next=next->right;
        }
        next=root->left;
        temp->right=root;
        root->left=NULL;
        return next;
    }
    return root;
}
