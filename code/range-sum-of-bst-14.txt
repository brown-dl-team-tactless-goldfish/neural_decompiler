/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     struct TreeNode *left;
 *     struct TreeNode *right;
 * };
 */
void    inorder(struct TreeNode *root, int *result, int low, int high)
{
    if (root)
    {
        inorder(root->left, result, low, high);
        if (root->val >= low && root->val <= high)
            *result += root->val;
        inorder(root->right, result, low, high);
    }
}

int rangeSumBST(struct TreeNode* root, int low, int high){
    int result;
    
    result = 0;
    inorder(root, &result, low, high);
    return(result);
}
