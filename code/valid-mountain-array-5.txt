bool validMountainArray(int* arr, int arrSize){
    int count =0; 
    int count1 =0; 
    if(arrSize<3) return false ; 
    
    for( int i =0 ; i < arrSize ; i++){
          if(i+1 >=arrSize)
              continue; 
        
            
        if ( count ==0){
            if(arr[i] <= arr[i+1]){
                if ( arr[i] < arr[i+1]){ 
                    count1++;
                    continue;
                } 
                else return false ;
            }
            else count++; 
        }
        
         if( count==1){
             if (arr[i] > arr[i+1]) continue; 
             else return false ;
         }
        
    }
    
    return    (count==1 && count1>0) ? true : false  ; 
}
