int so(const *a,const *b){
    return *(int*)a - *(int*)b;
}

int findPeakElement(int* nums, int numsSize){
    int *arr = (int*)malloc(numsSize * sizeof(int));
    for(int i = 0; i < numsSize; i++){
        arr[i] = nums[i];
    }
    qsort(arr,numsSize,sizeof(int),so);
    int x = arr[numsSize - 1];
    int q;
    for(int i = 0; i < numsSize; i++){
        if(nums[i] == x){
            q = i;
        }
    }
    return q;
}
