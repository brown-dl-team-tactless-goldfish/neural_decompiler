class Solution:
    def areOccurrencesEqual(self, s: str) -> bool:
        data = {}
        for c in s:
            if (c not in data):
                data[c] = 1
            else:
                data[c] += 1
                
        print ("data = ", data)

        
        val = float('inf')
        for k, v in data.items():
            if (val == float('inf')):
                val = v
            elif v != val:
                return False
            
        return True
