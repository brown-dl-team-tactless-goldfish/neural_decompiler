class Solution {
public:
    vector<string> simplifiedFractions(int n) 
    {
        vector<string>v;
        for(int i=1;i<=n;i++)
        {
            for(int j=1;j<i;j++)
            {
                if(__gcd(i,j)<=1)
                {
                    v.push_back(to_string(j)+"/"+to_string(i));
                }
            }
        }
        return v;
    }
};
