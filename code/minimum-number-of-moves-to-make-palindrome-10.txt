public class Solution {
    public int MinMovesToMakePalindrome(string s) {
        List<char> list = s.ToList();
        int moves = 0;
        while(list.Count > 2){
            char target = list[0];
            int i = list.FindLastIndex(ch => ch == target);
            if(i != 0){
                moves += list.Count-i-1;
                list.RemoveAt(i);
                list.RemoveAt(0);
            } else{
                target = list[list.Count-1];
                i = list.FindIndex(ch => ch == target);
                moves += i;
                list.RemoveAt(list.Count-1);
                list.RemoveAt(i);
            }
        }
        return moves;
    }
}
