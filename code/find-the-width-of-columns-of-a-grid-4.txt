public class Solution {
    public int[] FindColumnWidth(int[][] grid) {
        int m = grid.Length, n = grid[0].Length;
        int[] res = new int[n];
        
        for(int j = 0; j < n; j++)
        {
            int max = int.MinValue;
            for(int i = 0; i < m; i++)
            {
                max = Math.Max(max, grid[i][j].ToString().Length);
            }
            res[j] = max;
        }
        
        return res;
    }
}
