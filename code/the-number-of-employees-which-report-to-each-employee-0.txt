with c as (
select e1.employee_id, count(e2.reports_to) reports_count, round(avg(e2.age),0) average_age
from employees e1
join employees e2
on e1.employee_id = e2.reports_to
group by e1.employee_id)
select c.employee_id, e3.name, c.reports_count, c.average_age
from employees e3
join c
using(employee_id)
order by employee_id
