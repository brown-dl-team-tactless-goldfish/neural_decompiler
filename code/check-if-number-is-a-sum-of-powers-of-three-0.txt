

bool checkPowersOfThree(int k){
    int count=0;
    int n=k;
    while(n!=0)
    {
        n/=3;
        count++;
    }
    int A[count];
    int f=1;
    int a=0;
    while(a<count)
    {
        A[a++]=f;
        f=f*3;
    }
    for(int j=count-1;j>=0;j--)
    {
        if(k==0||k==1)
        {
            return true;
        }
        if(k>=A[j])
        {
            k=k-A[j];
        }
    }
   return false; 

}
