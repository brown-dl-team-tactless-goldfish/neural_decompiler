int findPoisonedDuration(int* arr, int arrSize, int length){
    int total = 0;
    for(int i = 0; i < arrSize - 1; i++){     //Skip the last element.
        if(arr[i + 1] <= arr[i] + length - 1) //If two intervals intersect,
            total += arr[i + 1] - arr[i];     //add non-intersecting part of the first interval.
        else
            total += length;                  //Otherwise add full length of the first interval.
    }
    return total + length;                    //Add the length of the last skiped element.
}
