char * decodeMessage(char * key, char * message){

    int* q = malloc(26*sizeof(int));
    memset(q, -1, 26*sizeof(int));
    
    int i = 0;
    for(char* k = key; *k != '\0'; k++)
    {
        if(*k != ' ' && q[*k - 'a'] == -1)
            q[*k - 'a'] = i++; 
            
    }
    
    for(char* m = message; *m != '\0'; ++m)
    {
        if(*m != ' ')
            *m = 'a' + q[*m - 'a'];
    }
    
    return message;
}
