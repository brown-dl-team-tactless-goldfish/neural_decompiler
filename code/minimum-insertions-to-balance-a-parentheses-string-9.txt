 int minInsertions(string s) {
		stack<char> temp;
		int res = 0,len;
		len = s.length();

		for(int i =0;i<s.length();i++)
		{
			if(s[i]=='(')
				temp.push('(');
			else
			{
				if(!temp.empty() && temp.top()=='(')
					temp.pop();
				else
					res++;

				if(i<len-1 && s[i+1] ==')')
					i++;
				else
					res++;
			}                
		}

		res += (temp.size()*2);
		return res;
	}
