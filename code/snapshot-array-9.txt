class SnapshotArray {
private:
    unordered_map<int, unordered_map<int,int>> mp;
    int snapCount;
public:
    SnapshotArray(int length) {
       snapCount=-1;
    }
    void set(int index, int val) {
        mp[-1][index] = val;
    }
    int snap() {
       snapCount++;
       mp[snapCount] = mp[-1];
       return snapCount;
    }
    int get(int index, int snap_id) {
       if (!mp[snap_id].count(index))
           mp[snap_id][index] = 0;
       return mp[snap_id][index];
    }
};
