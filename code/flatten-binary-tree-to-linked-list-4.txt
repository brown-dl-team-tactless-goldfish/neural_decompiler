void help(struct TreeNode *root, struct TreeNode **prev) {
    
    if(root == NULL)
        return;
    
    struct TreeNode *right =  root->right;
    
    if(*prev)
        (*prev)->right = root;
    
    *prev = root;
    help(root->left, prev);
    help(right, prev);
    root->left = NULL;
    
}

void flatten(struct TreeNode* root){
    struct TreeNode *prev = NULL;
    help(root,&prev);
    return root;

}
