    int  n = s.size();
    
   // vector<long> dp(n+1,0);
    
    long dp[n+1];
    
    memset(dp,0,sizeof(dp));
    
    long long ans = 0;
    
    vector<int> last_seen(26,-1);
    
    for(int i=0; i<n; i++)
    {
        int idx = s[i] - 'a';
        
        dp[i+1] = dp[i] + i - last_seen[idx];
        
        ans += dp[i+1];
        
        last_seen[idx] = i;
    }
    
    return ans;
    
}
