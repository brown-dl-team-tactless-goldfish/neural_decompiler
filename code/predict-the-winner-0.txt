int getScore(int* nums, int l, int r){
    if(l == r)
        return nums[l];
    return fmax( nums[l] - getScore(nums, l+1, r), nums[r] - getScore(nums, l, r-1) );
}
bool PredictTheWinner(int* nums, int numsSize){
    if(getScore(nums, 0, numsSize-1) >= 0)
        return true;
    else
        return false;
}
