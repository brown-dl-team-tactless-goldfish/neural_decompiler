
char * convertToTitle(int columnNumber){
     int m = columnNumber;
     int counter = 0; 
      while(m>0)
      {
          m = (m-1)/26;
          counter++;
      }

      int n = counter+1; // counter add one for '\0'
      char *c = (char*)malloc(sizeof(char)*n); 
      c[n-1] = '\0'; 
    
      for(int i=n-2;i>=0;i--)
      {
       
            c[i] = (char) ((columnNumber-1)%26 + 65);
            columnNumber=(columnNumber-1)/26;
      }
    
    return c; 
}
