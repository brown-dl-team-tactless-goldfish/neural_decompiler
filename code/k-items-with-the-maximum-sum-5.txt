public class Solution {
    public int KItemsWithMaximumSum(int numOnes, int numZeros, int numNegOnes, int k) {
        if(k == (numOnes + numZeros + numNegOnes))
            return (numOnes - numNegOnes);
        
        List<int> list = new();
        
        for(int i = 0; i < Math.Min(numOnes, k); i++)
            list.Add(1);
        
        if(list.Count == k)
            return list.Sum();
        
        k -= numOnes;
        for(int i = 0; i < Math.Min(numZeros, k); i++)
            list.Add(0);
        
        if(list.Count == numOnes + k)
            return list.Sum();
        
        k -= numZeros;
        for(int i = 0; i < Math.Min(numNegOnes, k); i++)
            list.Add(-1);
			
        return list.Sum();
    }
}
