/**
 * Note: The returned array must be malloced, assume caller calls free().
 */
int* rearrangeArray(int* nums, int numsSize, int* returnSize){
    int *arr=malloc(sizeof(int)*numsSize);
    *returnSize=numsSize;
    int c=0;
    for(int i=0;i<numsSize;i++){
        if(nums[i]>0){
            arr[c]=nums[i];
            c+=2;
        }
    }
    c=1;
    for(int i=0;i<numsSize;i++){
        if(nums[i]<0){
            arr[c]=nums[i];
            c+=2;
        }
    }
    return arr;
}
