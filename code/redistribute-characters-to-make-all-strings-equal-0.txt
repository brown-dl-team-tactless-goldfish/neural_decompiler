bool makeEqual(char ** words, int wordsSize){
    int n = wordsSize ;
    int* alpha = calloc(26, sizeof(int) ) ;
    for(int i = 0 ; i < n ; i++){
        int l = strlen(words[i]) ;
        for(int j = 0 ; j < l; j++){
            alpha[words[i][j] - 'a']++;
        }
    }
    for(int i = 0; i < 26; i++){
        if(alpha[i] % n)
            return false;
    }
    return true;
}
