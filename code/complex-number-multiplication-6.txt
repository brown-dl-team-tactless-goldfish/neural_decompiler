   string complexNumberMultiply(string a, string b) {
        int aReal, aIm, bReal, bIm;
        sscanf (a.c_str(),"%d+%di", &aReal, &aIm);
        sscanf(b.c_str(),"%d+%di", &bReal, &bIm);
        int last = -1 * aIm *bIm;
        int first = aReal *bReal;
        return ( to_string(first + last) + '+' + to_string(aReal * bIm +  bReal*aIm) + "i");
    }
