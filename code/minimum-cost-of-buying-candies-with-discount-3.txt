//step 1 : sorting 
// bubble sort !?

int i,j;
int out=0;

for (i=0;i<costSize;i++)
{
    for (j=0;j<costSize-1;j++)
    {
        if (cost[j]<cost[j+1])
        {
            cost[j]^=cost[j+1];
            cost[j+1]^=cost[j];
            cost[j]^=cost[j+1];
        }
    }
}
for (i=0;i<costSize;i++)
{
    if ((i+1)%3!=0)
    {
        out+=cost[i];
    }
}
return out;
