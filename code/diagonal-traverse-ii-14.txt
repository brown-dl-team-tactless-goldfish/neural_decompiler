class Solution {
public:
vector<int> findDiagonalOrder(vector<vector<int>>& nums) {
	map<int,list<int>> m;
	vector<int> res;
	
    for(int i=0;i<nums.size();i++){
        for(int j=0;j<nums[i].size();j++){
            m[i+j].push_front(nums[i][j]);	
        }		
    }
	
     for(auto x:m){
       for(auto val:x.second){
            res.push_back(val);
        }
     }
    return res;
}
};
