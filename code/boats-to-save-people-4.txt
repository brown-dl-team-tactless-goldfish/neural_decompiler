int cmp(const void *a,const void *b){
    return (*(int*)a-(*(int*)b));
}

int numRescueBoats(int* people, int peopleSize, int limit){
    qsort(people,peopleSize,sizeof(int),cmp);
    int boats=0;
    int start=0,end=peopleSize-1;
    while(start<=end){
        boats++;
        if(people[start]+people[end]<=limit){
            start++;
        }
        end--;
    }
    return boats;
}
