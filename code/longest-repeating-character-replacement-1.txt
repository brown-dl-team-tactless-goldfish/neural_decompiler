int characterReplacement(char * s, int k){
    
    int sLen=strlen(s);
    
    unsigned table[255] = {0};

    unsigned diff=0;
    unsigned max_same=0;
    for( unsigned left = 0, right=0; right<sLen; right++){
        
        // Move right pointer 
        table[s[right]]++;
        
        // Update max_same
        if(table[s[right]]>max_same)
           max_same = table[s[right]];
        
        // Or update diff
        else if(diff<k) diff++;
        
        // Or move left pointer
        else table[s[left++]]--; 
        
    }    
    return max_same+diff;
}
