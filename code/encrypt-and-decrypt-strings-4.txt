class Encrypter {
public:
    unordered_map<char, int> key;
    
    vector<string> value;
    
    vector<string> dictionary;
    
    Encrypter(vector<char>& keys, vector<string>& values, vector<string>& dictionarys) {
        
        for(int i = 0; i < keys.size(); i++)
        {
            key[keys[i]] = i;
        }
        
        value = values;
        
        dictionary = dictionarys;
    }
    
    string encrypt(string word1) {
        
        int n = word1.size();
        
        string res = "";
        
        for(int i = 0; i < n; i++)
        {
            char c = word1[i];
            
            int idx = key[c];
            
            string str = value[idx];
            
            res += str;
        }
        
        return res;
    }
    
    int decrypt(string word2) {
        
        int res = 0;
        
        for(auto x : dictionary)
        {
            if(encrypt(x) == word2)
            {
                res++;
            }
        }
        
        return res;
    }
};
