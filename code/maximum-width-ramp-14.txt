class Solution {
public:
    int maxWidthRamp(vector<int>& b){
        vector<pair<int,int>>v(b.size());
        for(int i=0;i<b.size();i++){
            v[i]=make_pair(b[i],i);
        }
        sort(v.begin(),v.end());
        int ans=0;
        int m=v[0].second;
        for(int i=1;i<v.size();i++){
            ans=max(ans,v[i].second-m);
            m=min(m,v[i].second);
        }
        return ans;
    }
};
