class RLEIterator {
public:
    vector<int> arr;
    int i,len;
    RLEIterator(vector<int> A) {
        i = 0;
        len = A.size();
        arr.assign(A.begin(), A.end());
    }
    int next(int n) {
        while (i<len&&n>arr[i]) {
            n -= arr[i];
            i += 2;
        }
        if (i>=len) return -1;
        arr[i] -= n;
        return arr[i+1];
    }
};
