bool recursivePalindrome(char *s, int size) {
    if (size < 2) return true;
    return s[0] == s[size - 1] && recursivePalindrome(&s[1], size - 2);
}

bool recursiveValidPalindrome(char *s, int size) {
    if (size < 2) return true;
    if (s[0] == s[size - 1]) return recursiveValidPalindrome(&s[1], size - 2);
    return recursivePalindrome(s, size - 1) || recursivePalindrome(&s[1], size - 1);
}

bool validPalindrome(char *s){
    int len = strlen(s);
    return recursiveValidPalindrome(s, len);
}
