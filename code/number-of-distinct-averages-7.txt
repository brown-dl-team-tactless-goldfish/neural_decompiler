class Solution {
public:
   int distinctAverages(vector<int>& nums) {
        unordered_set<string> s;
        sort(nums.begin(), nums.end());
        
        int i = 0, j = size(nums)-1;
        while (i < j) 
            s.insert(to_string( (float)(nums[i++] + nums[j--])/2 ));
        
        return size(s);
    }
};
