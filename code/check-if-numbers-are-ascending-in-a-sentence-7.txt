
bool areNumbersAscending(char * s){
    
    const char del[2] = " ";
    char * token = strtok(s,del);
    int c=0;
    int tmp[101];
    while(token!=NULL)
    {
        printf("%s ",token);
        if(token[0]>=48 && token[0]<=57)
        {
            tmp[c]=atoi(token);
            c++;
        }
        token=strtok(NULL,del);
    }
    
    
    for(int i = 0; i < c-1; i++)
    {
        if(tmp[i]>=tmp[i+1])    return false;
    }
    
    return true;
    
}
