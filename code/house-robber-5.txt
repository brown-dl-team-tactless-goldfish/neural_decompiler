#define max(a,b) (a>b?a:b)

int rob(int* nums, int numsSize){
    
    int esum = 0, osum = 0;
    
    for(int i = 0; i < numsSize; ++i)
    {
        if(i%2==0)
            esum = max(esum + *(nums+i), osum);
        else
            osum = max(osum + *(nums+i), esum);
    }

    return max(esum,osum);
}
