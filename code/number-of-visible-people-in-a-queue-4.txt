public class Solution 
{
    public int[] CanSeePersonsCount(int[] heights) 
    {
        var n = heights.Length;
        
        var answer = new int[n];
        var stack = new List<int>();
        
        for (int i = n - 1; i >= 0; i--)
        {            
            var c = heights[i];
            if (stack.Count > 0)
                answer[i] = VisibleCount(stack, c);
            else
                answer[i] = 0;
            
            while (stack.Count > 0 && c > stack.Last())
                stack.RemoveAt(stack.Count - 1);

            stack.Add(c);
        }
        
        return answer;
    }
    
    private int VisibleCount(List<int> stack, int c)
    {
        if (stack.First() < c)
            return stack.Count;
        if (stack.Last() > c)
            return 1;
        
        var left = 0;
        var right = stack.Count - 1;
        
        while (right - left > 1)
        {
            var mid = (left + right) / 2;
            if (stack[mid] > c)
                left = mid;
            else
                right = mid;
        }
        
        return stack.Count - left;
    }
}
