int smallestRangeI(int* nums, int numsSize, int k){
    /* Local variables */
    int result = 0;
    int min = 10001;
    int max = 0;
    
    /* Cycle through the array and update the maximum and minimum values accordingly */
    for(int i = 0; i < numsSize; i++)
    {
        if(nums[i] < min)
        {
            min = nums[i];
        }
        
        if(nums[i] > max)
        {
            max = nums[i];
        }
    }
    
    /* Check if k can shorten the gap between max and min */
    result = max - min - 2*k;

    /* Return */
    return (result >= 0)? result : 0;
        
}
