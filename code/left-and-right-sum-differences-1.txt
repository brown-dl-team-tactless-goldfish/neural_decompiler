int* leftRigthDifference(int* nums, int numsSize, int* returnSize){
    int* ans = (int*)calloc(numsSize, sizeof(int));
    int sum = 0;
    for(int i = 1; i < numsSize; i++)
        ans[i] = ans[i - 1] + nums[i - 1];
    
    for(int i = numsSize - 1; i >= 0; i--){
        ans[i] = abs(ans[i] - sum);
        sum += nums[i];
    }
  
    *returnSize = numsSize;
    return ans;
}
