public:
    bool checkPowersOfThree(int n) {
        while(n!=1)
        {
            if(n%3==0)
                n=n/3;
            else if(n%3==1)
                n=n-1;
            else
                break;
        }
        
        if(n==1)
            return true;
        else
            return false;
    }
};
