int rearrangeCharacters(char *s, char *target) {
    int cnt1[26] = {0};
    int cnt2[26] = {0};
    int i, ans = INT_MAX;
    for (i = 0; s[i] != '\0'; i++) {
        cnt1[s[i] - 'a']++;
    }
    for (i = 0; target[i] != '\0'; i++) {
        cnt2[target[i] - 'a']++;
    }
    for (i = 0; i < 26; i++) {
        if (cnt2[i] > 0) {
            ans = (ans < cnt1[i] / cnt2[i]) ? ans : cnt1[i] / cnt2[i];
        }
    }
return ans;
}
