int* shuffle(int* nums, int numsSize, int n, int* returnSize){
    int index1 = 0;
    int index2;
    int iterator = 0;
    int* returned = (int*) malloc(sizeof(int) * numsSize);
    *returnSize = numsSize;
    
    for(index2 = n; index2 < numsSize; index2++) {
        returned[iterator++] = nums[index1++];
        returned[iterator++] = nums[index2];
    } 
    
    return returned;
}
