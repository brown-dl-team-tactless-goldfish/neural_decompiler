long long pickGifts(int* gifts, int giftsSize, int k){
    int i, j, l, max;
    long long sum = 0;
    for(i=0; i<k; i++){
        max = 0;
        for(j=0; j<giftsSize; j++){
            if(max<gifts[j]){
                max = gifts[j];
                l = j;
            }
        }
        gifts[l] = sqrt(gifts[l]);
    }
    for(i=0; i<giftsSize; i++){
        sum = sum + gifts[i];
    }
    return sum;
}
