class Solution {
public:
    bool kLengthApart(vector<int>& nums, int k) {
        
        int n = nums.size();
        unordered_map<int, int>hm;
        for (int i = 0; i < n; i++) {
            if (nums[i] == 1) {
                if (hm.find(nums[i]) != hm.end()) {
                    if (i-hm[nums[i]] < k+1)
                        return false;
                }
                hm[nums[i]] = i;
            }
        }
        
        return true;
    }
};
