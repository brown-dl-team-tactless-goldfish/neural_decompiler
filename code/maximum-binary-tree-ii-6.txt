class Solution {
public:
    void f(TreeNode* root, int val){
        if(root->right==nullptr){
            root->right = new TreeNode(val);
            return;
        }
        if(root->right->val < val){
            auto t = new TreeNode(val);
            t->left = root->right;
            root->right = t;
            return;
        }
        else{
            f(root->right,val);
        }
    }
    TreeNode* insertIntoMaxTree(TreeNode* root, int val) {
        if(root==nullptr){
            return new TreeNode(val);
        }
        if(val>root->val){
            auto t = new TreeNode(val);
            t->left = root;
            return t;
        }
        else {
            f(root,val);
            return root;
        }
    }
};
