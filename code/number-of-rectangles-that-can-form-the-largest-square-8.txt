int countGoodRectangles(int** rectangles, int rectanglesSize, int* rectanglesColSize)
{
    // storing the smaller side for each rectangle in its index 0
    int i;
    for(i=0; i<rectanglesSize; i++)
        rectangles[i][0] = fmin(rectangles[i][0], rectangles[i][1]);
    
    // finding the max. of these stored smaller sides
    int max = rectangles[0][0];
    for(i=0; i<rectanglesSize; i++)
        if(max < rectangles[i][0])
            max = rectangles[i][0];
    
    // counting the no. of max. smaller sides
    int ans = 0;
    for(i=0; i<rectanglesSize; i++)
        if(rectangles[i][0] == max)
            ans++;
    
    return ans;
}
