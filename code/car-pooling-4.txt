class Solution {
public:
    bool carPooling(vector<vector<int>>& trips, int capacity) {
        vector<pair<int, int>> trev;
        for(auto it: trips) {
            trev.push_back({it[1], it[0]});
            trev.push_back({it[2], -it[0]});
        }
        int curr = 0;
        sort(trev.begin(), trev.end());
        for(int i = 0; i < trev.size(); i++) {
            curr += trev[i].second;
            if(curr > capacity) return false;
        }
        return true;
    }
};
