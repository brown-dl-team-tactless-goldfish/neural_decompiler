int minFlipsMonoIncr(char * s){    
    int oneCount, zerosAfter1stOne, onesBeforeLastZero, ans;
    oneCount = zerosAfter1stOne = onesBeforeLastZero = ans = 0;
    
    while (*s) {
        if (*s == '0') {
            onesBeforeLastZero += oneCount;
            oneCount = 0;
            if (onesBeforeLastZero || oneCount) zerosAfter1stOne++;
        }
        else if (*s == '1') oneCount++;
        
        int min = INT_MAX;
        if (*s == '1') temp = ans;
        if (onesBeforeLastZero < temp) temp = onesBeforeLastZero;
        if (zerosAfter1stOne < temp) temp = zerosAfter1stOne;
        if (ans + 1 < temp) temp = ans + 1;
        ans = temp;
        s++;
    }
    return ans;
}
