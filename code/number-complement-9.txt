
int findComplement(int num){
    int highest_bit_position = 0;
    
    // loop through all 32 bits, every time we see a 1 update highest_bit_set
    for(int i = 0; i < 32; i++)
    {
        const bool bit_is_set = (num >> i) & 1;
        if(bit_is_set)
        {
            highest_bit_position = i;
        }
    }
    
    // loop up to and including the highest_bit_position XOR and updating num
    for(int i = 0; i <= highest_bit_position; i++)
    {
        num ^= 1 << i;
    }
    
    return num;
}
