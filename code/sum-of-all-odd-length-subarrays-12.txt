int sumOddLengthSubarrays(int* arr, int arrSize){
    int ret=0;
    int left,right;
    int sum=0;
    for(int len=1;len<=arrSize;len+=2){
        left=0;
        right=left+len-1;
        sum=0;
        
        for(int i=left;i<=right;i++) sum+=arr[i];
        ret+=sum;
        
        while(right<arrSize-1){
            sum+=arr[++right];
            sum-=arr[left++];
            ret+=sum;
        }
    }
    return ret;
}
