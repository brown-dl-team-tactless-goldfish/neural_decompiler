class Solution:
    def makeEqual(self, words: List[str]) -> bool:
        n = len(words)
        data = {}
        for word in words:
            for c in word:
                if (c in data):
                    data[c] += 1
                else:
                    data[c] = 1


            
        print ("data = ", data)
        
        for k, v in data.items():
            if (v % n != 0):
                return False
            
        return True
