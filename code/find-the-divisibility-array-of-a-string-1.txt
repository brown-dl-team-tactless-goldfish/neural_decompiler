/**
 * Note: The returned array must be malloced, assume caller calls free().
 */
int* divisibilityArray(char * w, int m, int* returnSize){
    int *a=(int *)malloc(sizeof(int )*strlen(w));
    *returnSize=strlen(w);
    long long int c=0,j=0;
    for(int i=0;i<strlen(w);i++){
        c=(c*10+(w[i]-'0'))%m;
        if(c==0){
            a[j++]=1;
        }else{
            a[j++]=0;
        }
        
    }
    return a;

}
