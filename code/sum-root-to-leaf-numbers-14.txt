void traverse(struct TreeNode* node, int sum, int* total) {
    if(!node) return;
    
    if(!node->left && !node->right) {
        (*total) += sum * 10 + node->val;
        return;
    }
    
    traverse(node->left, sum * 10 + node->val, total);
    traverse(node->right, sum * 10 + node->val, total);
}


int sumNumbers(struct TreeNode* root){
    int total = 0;
    traverse(root, 0, &total);
    return total;
}
