bool isCircularSentence(char * sentence){
    if (sentence[0] != sentence[(int)strlen(sentence) - 1])
        return false;
    for (int i = 0; sentence[i] != '\0'; i++) {
        if (isspace(sentence[i])) {
            if (sentence[i - 1] != sentence[i + 1])
                return false;
        }
    }
    return true;
}
