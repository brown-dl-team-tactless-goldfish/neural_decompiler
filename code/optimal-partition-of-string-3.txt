int partitionString(char* s) {

    int l = strlen(s), al[26], st = 0, ans = 1;

    memset(al, -1, sizeof(al));

    for (int i = 0; i < l; i++) 

    {

        int c = s[i] - 'a';

        if (al[c] >= st) 

        {

            st = i;

            ans++;

        }

        al[c] = i;

    }

    return ans;

}
