
#define u64 unsigned long long

static inline u64 count(unsigned zero_num)
{
    return (u64) (1 + zero_num) * zero_num / 2;
}

long long zeroFilledSubarray(int* nums, int numsSize)
{
    u64 ans = 0;
    int left = -1;
    for (int i = 0; i < numsSize; i++) {
        if (nums[i] != 0) {
            ans += count(i - left - 1);
            left = i;
        }
    }
    ans += count(numsSize - left - 1);
    return ans;
}
