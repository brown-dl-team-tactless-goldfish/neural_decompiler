int cmpfunc(const void* a, const void* b){
    return *(int*)a - *(int*)b;
}

int minimumOperations(int* nums, int numsSize){
    qsort(nums, numsSize, sizeof(int), cmpfunc);
    int type = 0;
    for(int i = 0; i < numsSize; i++){
        if(i == 0 && nums[i] != 0){
            type++;
        }
        else if(nums[i] == 0)
            continue;
        else if(nums[i] != nums[i-1])
            type++;
    }
    return type;
}
