char slowestKey(int* releaseTimes, int releaseTimesSize, char* keysPressed) {
    int* arr = (int*) calloc(releaseTimesSize, sizeof(int));
    arr[0] = releaseTimes[0];
    int max = 0;
    for(int i = 1; i < releaseTimesSize; i++) {
        arr[i] = releaseTimes[i] - releaseTimes[i-1];
        if(arr[i] > arr[max]) max = i;
        if(arr[i] == arr[max]) {
            if(keysPressed[i] > keysPressed[max]) max = i;
        }
    }
    free(arr);
    return keysPressed[max];
}
