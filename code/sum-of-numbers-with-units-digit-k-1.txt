int minimumNumbers(int num, int k){
    if (num == 0){
        return 0;
    }
    if (num == k){
        return 1;
    }
    int unitsdigit = num % 10;
    int ans = 1;
    int now = (k*ans) % 10;
    while (now != unitsdigit && ans <= 10){
        ans++;
        now = (k*ans) % 10;
    }
    if (k*ans > num || ans == 11){
        return -1;
    }
    return ans;
}
