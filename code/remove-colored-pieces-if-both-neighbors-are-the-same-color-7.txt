class Solution {
public:
  int helper(string &s, char ch){
    int k = 0, lim = s.size() - 1;
    for(int i = 1; i < lim; i++)
      if(s[i-1] == ch && s[i] && s[i] == ch && s[i+1] == ch) k++;
    
    return k;
  }
  
    bool winnerOfGame(string colors) {
      return helper(colors, 'A') > helper(colors, 'B');  
    }
};
