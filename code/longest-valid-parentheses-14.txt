	int num = 0, tmp = 0, count = 0;
	for (int i = 0; i < len; i++) {
		if (st.top->ch == '(' && s[i]==')') {
			count = st.top->indx;
			--st.top;
			st.top->indx += count + 1;
			int preIndx = st.top->indx;
			num = preIndx > num ? preIndx : num;
		} else {
			++st.top;
			st.top->ch = s[i];
			st.top->indx = 0;
		}
	}
	
	return num << 1;
}
