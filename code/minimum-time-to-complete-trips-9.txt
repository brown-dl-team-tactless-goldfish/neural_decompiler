class Solution {
public:
    long long count(vector<int>&time,long long mid){
        long long c = 0;
        for(auto &t:time){
            c += mid/t;
        }
        return c;
    }
    long long minimumTime(vector<int>& time, int totalTrips) {
        long long m = *max_element(time.begin(),time.end());
        long long low = 1;
        long long high = totalTrips*m;
        while(low<=high){
            long long mid = low + (high-low)/2;
            if(count(time,mid)>=totalTrips){
                high = mid - 1;
            }
            else{
                low = mid + 1;
            }
        }
        return low;
    }
};
