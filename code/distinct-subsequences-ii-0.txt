int mod = 1e9+7;
class Solution {
public:
    int distinctSubseqII(string s) {
        int n = s.length();
        vector<int> dp(26);
        for(auto c : s){
            int tmp = 0;
            for(int j=0; j<26; ++j)
                tmp = (tmp + dp[j])%mod;
            dp[c-'a'] = (tmp+1)%mod;
        }
        
        int ans = 0; for(int i=0; i<26; ++i) ans = (ans+dp[i])%mod;
        return ans;
        
    }
};
