// OJ: https://leetcode.com/problems/find-closest-number-to-zero/
// Author: github.com/lzl124631x
// Time: O(N)
// Space: O(1)
class Solution {
public:
    int findClosestNumber(vector<int>& A) {
        int diff = INT_MAX, ans = INT_MAX;
        for (int n : A) {
            if (abs(n) < diff || (abs(n) == diff && n > ans)) {
                diff = abs(n);
                ans = n;
            } 
        }
        return ans;
    }
};
