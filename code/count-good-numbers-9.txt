int mod = 1e9+7;

long long power(long x,long y)
{
   long res = 1; 

    
    while (y > 0)
    {
        if (y & 1)
        {
            res = (res*x) % mod;
        }
        
        y = y / 2;
        
        x = (x*x) % mod;
    }
    
    return res % mod;
}
int countGoodNumbers(long long n) {
    
    long temp = power(20,n/2);
    
    if((n&1) == 0)
    {
        return temp;
    }
    return (temp*5) % mod;
}
