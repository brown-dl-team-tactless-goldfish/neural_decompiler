int minMoves(int target, int maxDoubles)
{
    int moves=0;
    while(target!=1)
    {
        if(maxDoubles==0)
        {
            break;
        }

        if(maxDoubles!=0 && (target%2==0))
        {
            target=target/2;
            moves++;
            maxDoubles--;
        }
        else
        {
            target--;
            moves++;
        }
    }
    moves+=target-1;
    return moves;
}
