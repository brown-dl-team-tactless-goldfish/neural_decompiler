class Solution {
public:
    int partitionArray(vector<int>& nums, int k) {
        int n = nums.size();
        int cnt = 1;
        sort(nums.begin(), nums.end());
        int mn = INT_MAX, mx = INT_MIN;
        for(int i = 0; i<n; i++) {
            mn = min(mn, nums[i]);
            mx = max(mx, nums[i]);
            if(mx-mn > k)  {
                mn = mx;
                cnt++;
            }
        }
        return cnt;
    }
};
