    int n = a.size();
    int sum[20000] = {0};
    int maxn = 0;

    for (int i = 0; i < n; i++)
    {
        int j = i;
        int cnt = 1;
        while (sum[j] == 0)
        {
            sum[j] = cnt++;
            j = a[j];
        }
    }
    for (int i = 0; i < n; i++)
    {
        maxn = max(sum[i], maxn);
    }

    return maxn;
}
