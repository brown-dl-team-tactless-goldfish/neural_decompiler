select c.id, c.date, c.people
from stadium c 
left join stadium a
on a.id = c.id-2
left join stadium b
on b.id = c.id-1
left join stadium d
on d.id = c.id+1
left join stadium e
on e.id = c.id+2
where
(a.people>=100 and b.people>=100 and c.people>=100) or
(d.people>=100 and b.people>=100 and c.people>=100) or
(c.people>=100 and d.people>=100 and e.people>=100)
order by c.date;
