char * reversePrefix(char * word, char ch){
    int ind = 0;
    for(int i=0;word[i]!='\0';i++){
        if(word[i]==ch){ind = i;break;}
    }
    
    if(ind != 0){
        int low = 0,high = ind;
        while(low<=high){
            char temp = word[low];
            word[low] = word[high];
            word[high] = temp;
            low++;
            high--;
        }
    }
    return word;
}
