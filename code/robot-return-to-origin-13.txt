    switch(*moves){
        case 'U':
            a[0]++;
        break;
        case 'D':
            a[0]--;
        break;
        case 'R':
            a[1]++;                
        break;                
        case 'L':
            a[1]--;                
        break;                
    }
    moves++;
}

return true;
}
