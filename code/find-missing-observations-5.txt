class Solution {
public:
    vector<int> missingRolls(vector<int>& nums, int mean, int n) 
    {
        int total=nums.size()+n;
        int sum=accumulate(nums.begin(),nums.end(),0);
        int miss=mean*total-sum;
        if(miss<n||miss>6*n)
        {
            return {};
        }
        int complete=miss/n;
        int extra=miss%n;
        vector<int>ans(n,complete);
        for(int i=0;i<extra;i++)
        {
            ans[i]++;
        }
        return ans;
    }
};
