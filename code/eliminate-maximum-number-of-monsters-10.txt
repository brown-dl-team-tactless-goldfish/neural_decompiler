
class Solution {
public:
    int eliminateMaximum(vector<int>& dist, vector<int>& speed) {
		int n = dist.size();
        vector<int> t(n);
		for (int i = 0; i < n; i++) {
			t[i] = ceil((1.0 * dist[i]) / speed[i]) - 1; // max time to kill
		}
		sort(t.begin(), t.end());
		int tim = 0, ans = 0;
		for (const int &x: t) {
			if (x < tim) break;
			ans++;
			tim++;
		}
		return ans;
    }
};
