class Solution {
public:
    typedef long long ll;
    ll big = 1e9 + 7;
    
    int sumSubseqWidths(vector<int>& nums) {
        int n = nums.size();
        vector<ll> pow(n+1, 1);
        for(int i=1; i<n+1; ++i) pow[i] = (pow[i-1] * 2) % big;
        
        sort(nums.begin(), nums.end());
        ll ans = 0;
        for(int i=0; i<n; ++i) {
            ll pos = pow[i] - 1, neg = pow[n-1-i] - 1;
            ans = (ans + (pos - neg) * nums[i]) % big; 
        }
        return ans;
    }
};
