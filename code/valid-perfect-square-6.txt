 public bool IsPerfectSquare(int num) 
    {
        long high = num;
        long low = 0;
        
        while(high >= low)
        {
            long mid = (high+low)/2;
            long sqroot = mid * mid;
            if(sqroot > num)
                high = mid - 1;
            else if(sqroot < num)
                low = mid+1;
            else if(num % mid == 0)
                return true;
        }
        
        return false;
    }
