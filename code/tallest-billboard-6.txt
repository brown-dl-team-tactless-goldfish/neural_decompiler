public int TallestBillboard(int[] rods) 
{
	Dictionary<int, int> d = new Dictionary<int,int>() { [0] = 0 };

	foreach(var r in rods)
	{
		//Handle excluding a beam, at worst, we're back to where we started
		var next = d.ToDictionary(x => x.Key, x => x.Value);

		foreach(var kvp in d.ToList())
		{
			next[kvp.Key + r] = Math.Max(next.GetValueOrDefault(kvp.Key + r), kvp.Value + r);
			next[kvp.Key - r] = Math.Max(next.GetValueOrDefault(kvp.Key - r), kvp.Value + r);
		}

		d = next;
	}
	return d[0] / 2;
}
