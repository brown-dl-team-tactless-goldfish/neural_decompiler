int longestPalindrome(char * s){
    int slen = strlen(s);
    if (slen < 2) return 1;
    int freq[256] = {0};
    int i = 0;
    while (i < slen) {
        ++freq[s[i]];
        ++i;
    }
    int odds = 0;
    i = 0;
    while (i < 256) {
        odds += (freq[i] & 1);
        ++i;
    }
    return slen - odds + (odds > 0);
}
