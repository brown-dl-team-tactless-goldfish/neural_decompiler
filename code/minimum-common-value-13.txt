class Solution {
public:
    int getCommon(vector<int>& nums1, vector<int>& nums2) {
        unordered_set<int> st;
        for(int i = 0; i < nums1.size(); i++) st.insert(nums1[i]);
        for(int i = 0; i < nums2.size(); i++)
            if(st.count(nums2[i])) return nums2[i];
        return -1;
    }
};
