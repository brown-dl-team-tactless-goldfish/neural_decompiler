int i;

int max = 0;
int score = 0;
int size = 0;

 for(i=0;i<numsSize;i++) {
    
     nums[i] ? score -- : score ++;
             
    if (max < score) {
        max = score;
        nums[0] = i+1;
        size = 1;
    } 
    else if (max == score) {
        nums[size] = i+1;
        size ++;
    }
}

if (max == 0) {
    nums[size] = 0;
    size ++;
}

*returnSize = size;
return nums;
