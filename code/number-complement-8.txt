

int findComplement(int num){

    int i=0,new=0,bit;
    while(num)
    {
        bit=!(num&01);          // extracting each bit from LSB to last set bit
        new|=(bit<<i);          // complement that bit and add it in new number
        num>>=1;
        i++;
    }
    return new;                 // return result.
}
