int averageValue (int* nums, int numsSize) {
    int i, c, sum;
    for (i = c = sum = 0; i < numsSize; i++) {
        if (nums[i]%2 == 0) {
            if (nums[i]%3 == 0) {
                sum += nums[i];
                c++;
            }
        }
    }
    if (!c)
        return c;
    return (sum / c);
}
