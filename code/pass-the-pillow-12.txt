class Solution {
public:
    int passThePillow(int n, int time) {
        int x = 1,t=0;
        bool flg = true;
        while(t<time){
            if(flg){
                x+=1;
            }else if(!flg){
                x-=1;
            }
            if(x==n){
                flg = !flg;
            }else if(x==1){
                flg = !flg;
            }
            t+=1;
        }
        return x;
    }
};
