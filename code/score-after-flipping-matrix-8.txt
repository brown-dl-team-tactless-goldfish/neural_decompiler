	public int MatrixScore(int[][] A) {
        int res = A.Length;
        for(int i = 1; i < A[0].Length; i++) {
            int tmp = 0;
            for(int j = 0; j < A.Length; j++) {
                if(A[j][0] == 1) tmp += A[j][i];
                else tmp += 1 - A[j][i];
            }
            res = 2 * res + Math.Max(tmp, A.Length - tmp);
        }
        return res;
    }
