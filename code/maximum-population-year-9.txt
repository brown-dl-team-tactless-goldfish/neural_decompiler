class Solution {
public:
	int maximumPopulation(vector<vector<int>>& logs) {
		map<int, int> m;
		for(int i = 0; i < logs.size(); i++){
			for(int j = logs[i][0]; j < logs[i][1]; j++){
				m[j]++;
			}
		}
		int max = 0;
		int res = 1950;
		for(auto i: m){
			if(i.second > max){
				max = i.second;
				res = i.first;
			}
		}
		return res;
	}
};
