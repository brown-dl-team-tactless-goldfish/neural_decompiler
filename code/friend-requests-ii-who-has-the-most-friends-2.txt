SELECT id, 
       Sum(friends) AS num 
FROM   ((SELECT requester_id AS id, 
                Count(*)     AS friends 
         FROM   request_accepted 
         GROUP  BY requester_id) 
        UNION ALL 
        ((SELECT accepter_id AS id, 
                 Count(*)    AS friends 
          FROM   request_accepted 
          GROUP  BY accepter_id))) AS c 
GROUP  BY id 
ORDER  BY 2 DESC 
LIMIT  1
