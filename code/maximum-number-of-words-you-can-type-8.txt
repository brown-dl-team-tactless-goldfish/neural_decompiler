int canBeTypedWords(char * text, char * brokenLetters){
    char delims[] = " ";
    char *word = strtok(text, delims), *pch;
    int result = 0;
    do {
        for (pch = word; *pch; pch++) {
            if (strchr(brokenLetters, *pch) != NULL) {
                break;
            }
        }
        
        if (*pch == '\0')
            result++;
    } while ((word = strtok(NULL, delims)) != NULL);
    
    return result;
}
