    dummyNode = numEdges + 1
    uf = UnionFind(dummyNode)
    # print "1 uf.dic = ", uf.dic

    for edge in edges:
        # print "edge = ", edge
        p_0, p_1 = edge[0], edge[1]
        # print "p_0 = ", p_0, " p_1 = ", p_1
        if uf.isConnected(p_0, p_1) == True:
            return edge
        else:
            uf.union(p_0, p_1)
