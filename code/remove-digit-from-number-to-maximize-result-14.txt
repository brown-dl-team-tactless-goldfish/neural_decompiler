class Solution {
public:
    string removeDigit(string number, char digit) {
        string maxNumber;
        for (int i = 0; i < number.size(); ++i)
        {
            if (number[i] == digit)
            {
                maxNumber = max(maxNumber, number.substr(0, i) + number.substr(i + 1));
            }
        }
        return maxNumber;
    }
};
