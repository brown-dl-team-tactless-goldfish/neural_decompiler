int rearrangeCharacters(string s, string target) {
    int cnt = 0;
    unordered_map<char ,int > m , t;
    for(char c : s)
        m[c]++;
    for(char c : target)
        t[c]++;
    while(true){
        for(auto i : t){
            m[i.first] -= i.second;
            if(m[i.first] < 0)
                return cnt;
        }
        cnt++;
    }
    return cnt;
}
