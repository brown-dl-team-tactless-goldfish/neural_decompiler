

int longestPalindrome(char * s){
    int map[256]={0};                   //map array
    int i,count=0;
    
    for(i=0;s[i];i++)                   //count occurance of character
        map[s[i]]++;
    
    for(i=0;i<256;i++)
        count=count+(map[i]/2*2);       //now divide and multiply occurance by 2 so we get the count f character that occured in even times
    
    if(count<strlen(s))                 //if count is less that means there is still one character that we can include in our palindrome string so increment by one
        count++;
    
    return count;                       //return count
}
