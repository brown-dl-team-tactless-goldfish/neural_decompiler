class Solution {
public:
    int preimageSizeFZF(int K) {
        const int highestPower = 13;
        int tz = 0;
        vector<int> zeroInc;
        zeroInc.push_back(0);
        for(int i = 1; i <= highestPower; i++){
            zeroInc.push_back(zeroInc[i-1]*5+1);
        }
        for(int i = highestPower; i > 0; i--){
            tz = K/zeroInc[i];
            if(tz > 4) return 0;
            K -= tz*zeroInc[i];
            if(K == 0) return 5;
        }
        return 0;
    }
};
