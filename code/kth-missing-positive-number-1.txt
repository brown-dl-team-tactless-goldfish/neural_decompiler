int findKthPositive(int* arr, int arrSize, int k){
int i,flag=0,sayac=0;

 for(i=1;i<=arrSize+k;i++) {
     for(int j=0;j<arrSize;j++) {
         if(arr[j]==i) {
             flag=1;
break;
         }
        
     } 
     if(flag==0) {
sayac++;
if(sayac==k) {
    break;
}
     }
     else {
         flag=0;

     }
     }


return i;
}
