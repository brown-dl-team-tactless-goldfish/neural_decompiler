(SELECT A.product_id, ("store1") AS store, A.store1 AS price FROM Products A WHERE A.store1 is NOT NULL) UNION (SELECT B.product_id, ("store2") AS store, B.store2 AS price FROM Products B WHERE B.store2 is NOT NULL) UNION (SELECT C.product_id, ("store3") AS store, C.store3 AS price FROM Products C WHERE C.store3 is NOT NULL)  ORDER BY product_id, store
