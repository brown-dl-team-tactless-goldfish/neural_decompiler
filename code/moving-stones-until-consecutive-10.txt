public class Solution
{
    public int[] NumMovesStones(int a, int b, int c)
    {
        var v = new int[] { a, b, c };
        Array.Sort(v);
        int min = v[0];
        int mid = v[1];
        int max = v[2];

        int minMoves =
            mid > min + 2 && max > mid + 2 ? 2 :
            max > min + 2 ? 1 :
            0;
        return new int[] { minMoves, max - min - 2 };
    }
}
