typedef struct {int inner[3];} ParkingSystem;


ParkingSystem* parkingSystemCreate(int big, int medium, int small) {
    ParkingSystem* ps = malloc(sizeof(ParkingSystem));
    ps->inner[0] = big;
    ps->inner[1] = medium;
    ps->inner[2] = small;
    return ps;
}

bool parkingSystemAddCar(ParkingSystem* ps, int carType) {
    if (ps->inner[carType - 1] - 1 >= 0) {
        ps->inner[carType - 1] -= 1;
        return true;
    }

    return false;
}

void parkingSystemFree(ParkingSystem* ps) {
   free(ps); 
}
