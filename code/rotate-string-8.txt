bool rotateString(char* A, char* B) 
{
    if(NULL == A || NULL == B)
        return false;
    
    int lena = strlen(A);
    int lenb = strlen(B);
    /*length not equal*/
    if(lena != lenb)
        return false;
    
    char *pos = NULL;
    
    char *AA = (char *)malloc((2 * lena +1) *sizeof(char));
    snprintf(AA,2*lena + 1,"%s%s",A,A);
    
    /*whether B is substr of AA*/
    pos = strstr(AA,B);
    if(pos)
        return true;
    
    return false;
    
    
}
