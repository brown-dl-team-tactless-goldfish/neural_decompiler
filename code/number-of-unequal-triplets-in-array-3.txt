int unequalTriplets(int* nums, int numsSize){
int i,j,k,count=0;
    for(i=0;i<numsSize;i++){
        for(j=i+1;j<numsSize;j++){
            for(k=j+1;k<numsSize;k++){
                if(nums[i]!=nums[j] && nums[i]!=nums[k] && nums[j]!=nums[k]){
                    count++;
                }
            }
        }
    }
    return count;
}
