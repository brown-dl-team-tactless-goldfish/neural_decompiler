class Solution {
public:
    bool checkIfCanBreak(string A, string B) {
        sort(A.begin(), A.end());
        sort(B.begin(), B.end());
        return all_of(A.begin(), A.end(), [i = 0, &A, &B](auto _) mutable { return A[i] <= B[i++]; })
            || all_of(B.begin(), B.end(), [i = 0, &A, &B](auto _) mutable { return B[i] <= A[i++]; });
    }
};
