class Solution:
    def maximumSwap(self, num: int) -> int:
        A = list(str(num))
        # print("A = ", A)

        self.res = []

        def helper(remainA):
            # print("remainA = ", remainA)

            if (remainA == []):
                return

            tmpMax = max(remainA)
            tmpIdxes = []
            for idx in range(len(remainA)):
                if (remainA[idx] == tmpMax):
                    tmpIdxes.append(idx)

            # tmpIdx = remainA.rindex(tmpMax)
            print("tmpIdxes = ", tmpIdxes)

            if (0 in tmpIdxes):
                self.res.append(remainA[0])
                helper(remainA[1:])
            else:
                remainA[0], remainA[tmpIdxes[-1]
                                    ] = remainA[tmpIdxes[-1]], remainA[0]

                self.res += remainA
                return

        helper(A)
        print("self.res = ", self.res)

        return int(''.join(self.res))
