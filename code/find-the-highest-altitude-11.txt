int largestAltitude(int* gain, int gainSize){
    int support = 0, pos = 0, max = 0;
    int altitude[gainSize+1];
    
    altitude[pos++] = 0;
    for (int i = 0; i < gainSize; i++) {
        altitude[pos] = (support + gain[i]);
        support = altitude[pos];
        pos++;
    }
    for (int i = 0; i < pos; i++)
        if (altitude[i] > max)
            max = altitude[i];
    return max;
}
