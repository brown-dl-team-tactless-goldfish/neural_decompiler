
double average(int* salary, int salarySize){
    int sum = 0, i, min=INT_MAX, max=0, tmp;
    for(i=0;i<salarySize;i++){
        if(salary[i]<min){
            tmp = min;
            min = salary[i];
            salary[i] = (tmp != INT_MAX) ? tmp : 0;
        }
        if(salary[i]>max){
            tmp = max;
            max = salary[i];
            salary[i] = tmp;
        }
        sum+=salary[i];
    }
    return (double)sum/(salarySize-2);
}
