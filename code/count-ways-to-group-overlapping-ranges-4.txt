class Solution {
public:
    int countWays(vector<vector<int>>& ranges) {
        sort(ranges.begin(), ranges.end());
        int n = 0, r = -1;
        for(auto& v:ranges){
            if(v[0] > r)  n++;
            r = max(r, v[1]);
        }
        long long res = 1, mod = 1e9+7;
        while(n--) res = 2*res%mod;
        return res;
    }
};
