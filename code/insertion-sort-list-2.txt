struct ListNode* insertionSortList(struct ListNode* head){
    if(!head || !head->next) return head;
    
    struct ListNode* ans = head;
    struct ListNode* prev;
    struct ListNode* curr = head;
    prev = curr;
    curr = curr->next;
    while(curr){
        if(curr->val < prev->val){
            if(curr->val <= ans->val){
                /* always set to the head */
                struct ListNode* tmp = curr->next;
                prev->next = curr->next;
                curr->next = ans;
                ans = curr;
                curr = tmp;
            }
            else{
                /* insert the node */
                struct ListNode* tmp = ans;
                struct ListNode* tmpPrev;
                /* try to find the insert position from head */
                while(tmp->val < curr->val){
                    tmpPrev = tmp;
                    tmp = tmp->next;
                }
                tmpPrev->next = curr;
                tmpPrev = curr->next;
                curr->next = tmp;
                prev->next = tmpPrev;
                curr = tmpPrev;
            }
        }
        else{
            prev = curr;
            curr = curr->next;
        }
    }
    return ans;
}
