class Solution {
public:
  int minCost(vector<int>& sP, vector<int>& hP, vector<int>& rC, vector<int>& cC) {
    int def = 0;
    if     (sP[1] < hP[1]) for(int i = sP[1] + 1; i <= hP[1]  ;i++) def += cC[i];
    else if(sP[1] > hP[1]) for(int i = sP[1] - 1; i >= hP[1]  ;i--) def += cC[i];    
    
    if     (sP[0] < hP[0]) for(int i = sP[0] + 1; i <= hP[0]  ;i++) def += rC[i];
    else if(sP[0] > hP[0]) for(int i = sP[0] - 1; i >= hP[0]  ;i--) def += rC[i];    
    
    return def;
  }
};
