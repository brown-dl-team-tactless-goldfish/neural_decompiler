

int smallestRangeI(int* A, int ASize, int K){

    int range;
    range = 2*K;
    int max = INT_MIN;
    int min = INT_MAX;

    for(int count = 0;count<ASize;count++)
    {
        if(A[count]>max)
        {
            max = A[count];
        }
        
        if(A[count]<min)
        {
            min = A[count];
        }
    }

    if((max-min)<=range)
    {
        return 0;
    }else
    {
        return (max-min-range);
    }
    
    return 0;
}
