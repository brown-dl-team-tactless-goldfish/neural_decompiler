int minimumMoves(string s) {
        int ans=0;
        for(int i=0; i<s.size(); i++)
        {
            if(s[i] == 'X')
            {
                if(i+1<s.size())
                    s[i+1]='O';
                if(i+2<s.size())
                    s[i+2]='O';
                s[i]='O';
                ans++;
            }
        }
        return ans;
    }
