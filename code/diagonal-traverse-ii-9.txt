class Solution:
    def findDiagonalOrder(self, nums: List[List[int]]) -> List[int]:
        
        output=[]
        mydict = defaultdict(list)
        
        for r in range(0, len(nums)):
            for c in range(0, len(nums[r])):
                mydict[r + c].append(nums[r][c])
                
        for k, v in mydict.items():
            v.reverse()
            for n in v:
                output.append(n)
            
        return output
