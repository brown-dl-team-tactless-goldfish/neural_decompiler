int splitNum(int num) {
	vector<int> digits;
	while (num) {
		digits.push_back(num % 10);
		num /= 10;
	}
	sort(begin(digits), end(digits));
	int num1 = 0, num2 = 0;
	for (const int& n : digits) {
		if (num1 <= num2)
			num1 = (num1 * 10) + n;
		else
			num2 = (num2 * 10) + n;
	}
	return num1 + num2;
}
