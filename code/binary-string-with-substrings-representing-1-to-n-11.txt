bool match(int i, string s){
        string str;
        while(i){
            str+=to_string(i%2);
            i=i/2;
        }
        reverse(str.begin(), str.end());
        size_t found=s.find(str);
        if( found!=string::npos) return true;
        else return false;
    }
    bool queryString(string S, int N) {
        int i=1;
        while(i<=N){
            if(!match(i,S)) return false;
            i++;
        }
        return true;
    }
