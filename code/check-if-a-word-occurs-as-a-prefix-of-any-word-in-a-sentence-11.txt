int isPrefixOfWord(char * sentence, char * searchWord){
    bool word = true;
    char *p = sentence;
    int wordLength = strlen(searchWord);
    int i = 1;
    
    while (*p != '\0') {
        if (word && strncmp(p, searchWord, wordLength) == 0) {
            return i;
        }
        word = false;
        if (*p == ' ') {
            word = true;
            i++;
        }
        p++;
    }
    
    return -1;
}
