struct TreeNode* trimBST(struct TreeNode* root, int low, int high)
{
    if (!root)
        return NULL;
    
    root->left = trimBST(root->left, low, high);
    root->right = trimBST(root->right, low, high);
    
    // If the node is not in the range:
    // 1. If the left child is not Null, set it to be the new node.
    // 2. If the left chile is Null, and the right child isn't make it the to be the new node.
    // 3. Non of the above applies, make the node Null.
    if (!(root->val >= low && root->val <= high)) {
        if (root->left)
            root = root->left;
        else if (root->right)
            root = root->right;
        else
            root = NULL;
    }
    
    return root;
}
