void preorderaddval(struct TreeNode* root, int* ar, int* returnSize)
{
    if(!root) return;
    
    ar[(*returnSize)++] = root->val;
    preorderaddval(root->left, ar, returnSize);
    preorderaddval(root->right, ar, returnSize);
    
}

int* preorderTraversal(struct TreeNode* root, int* returnSize){
    
    int* arry = malloc(100 * sizeof(int));
    *returnSize = 0;
    
    preorderaddval(root, arry, returnSize);
    
    return arry;
}
