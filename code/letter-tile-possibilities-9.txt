    public class Solution
    {
        private void Helper(HashSet<string> res, string prefix, StringBuilder tiles)
        {
            for (int i = 0; i < tiles.Length; i++)
            {
                var c = tiles[i];

                if (c == '0')
                {
                    continue;
                }

                tiles[i] = '0';

                var letter = prefix + c;
                res.Add(letter);

                Helper(res, letter, tiles);

                tiles[i] = c;
            }
        }

        public int NumTilePossibilities(string tiles)
        {
            HashSet<string> res = new HashSet<string>();
            Helper(res, "", new StringBuilder(tiles));
            return res.Count;
        }
    }
