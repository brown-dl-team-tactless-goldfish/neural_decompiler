int comp(const void* a, const void* b){
    return *(int*)a-*(int*)b;
}

int findContentChildren(int* g, int gSize, int* s, int sSize){
    int index=0;
    qsort(s, sSize, sizeof(int), comp);
    qsort(g, gSize, sizeof(int), comp);
    for(int i=0; i < sSize; i++){
        if(index < gSize && g[index]<= s[i]){
            index++;
        }

    }
    return index;
}
