class Solution {
public:
    /// 4 ms, faster than 68.27% of C++
    /// 7.1 MB, less than 97.21% of C++
    vector<int> numOfBurgers(int tomatoSlices, int cheeseSlices)
    {
        if (tomatoSlices % 2)
            return vector<int>();
        int jb = (tomatoSlices - 2*cheeseSlices) / 2;
        int sb = cheeseSlices - jb;
        if (jb < 0 || sb < 0)
            return vector<int>();
        return {jb,sb};
    }
};
