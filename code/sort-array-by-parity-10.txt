int* sortArrayByParity(int* nums, int numsSize, int* returnSize){
    int *output=(int*)malloc(numsSize*sizeof(int*));
    *returnSize=numsSize;
    int index=0;
    for(int i=0;i<numsSize;i++){
        if(nums[i]%2==0)
            output[index++]=nums[i];
    }
    for(int j=0;j<numsSize;j++){
        if(nums[j]%2!=0)
            output[index++]=nums[j];
    }
return output;
}
