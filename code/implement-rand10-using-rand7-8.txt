/**
 * The Rand7() API is already defined in the parent class SolBase.
 * public int Rand7();
 * @return a random integer in the range 1 to 7
 */
public class Solution : SolBase {
    public int Rand10() {
                
        int r = Int32.MaxValue;
        
        while( r >= 40)
        {
            r = (Rand7() - 1) * 7 + Rand7() - 1;
        }
        
        return r % 10 + 1;    
    }
}
