class Solution {
 public:
  auto maxScoreIndices(vector<int> &nums) -> vector<int> {
    auto rightOnes = accumulate(begin(nums), end(nums), 0);
    auto leftZeros = 0;
    int n = nums.size();
    auto finalZeros = n - rightOnes;
    vector<int> ans{0};
    int maxScore = leftZeros + rightOnes, currScore{};
    for (int i = 0; i < n; ++i) {
      if (nums[i] == 0) {
        ++leftZeros;
      } else {
        --rightOnes;
      }
      currScore = leftZeros + rightOnes;
      if (currScore > maxScore) {
        maxScore = currScore;
        ans = {i + 1};
      } else if (currScore == maxScore) {
        ans.push_back(i + 1);
      }
    }
    return ans;
  }
};
