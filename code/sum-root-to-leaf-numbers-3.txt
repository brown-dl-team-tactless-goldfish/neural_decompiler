void travel(struct TreeNode* root,int *sum,int num)
{
    if(!root)
        return;
    num=num*10+root->val;
    if(root->left==NULL&&root->right==NULL)
    {
        *sum+=num;
    }
    else
    {
        travel(root->right,sum,num);
        travel(root->left,sum,num);
    }
}
int sumNumbers(struct TreeNode* root){
    int sum=0;
    travel(root,&sum,0);
    return sum;
}
