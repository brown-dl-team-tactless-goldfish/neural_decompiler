int max(int x, int y) {
    return x > y ? x : y;
}

int lengthOfLongestSubstring(char* s) {
    int index[256];
    memset(index, -1, sizeof(index));
    int current_max  = 0;
    int current_left = -1;
    for (int i = 0; s[i] != '\0'; ++i) {
        current_left = max(current_left, index[s[i]]);
        current_max  = max(current_max,  i - current_left);
        index[s[i]] = i;
    }
    return current_max;
}
