int findLengthOfLCIS(int* nums, int numssize) {
    if (numssize==0)
        return 0;
    int i,j,curr,maxlen;
    maxlen=j=i=0;
    while(j<numssize-1){
        if(nums[j]<nums[j+1])
            j++;
        else  {
            curr=j-i+1;
            if(maxlen<curr)
                maxlen=curr;  
            j++;
            i=j;
        }
    }
    curr=j-i+1;
    return curr>maxlen?curr:maxlen;
}
