int timeRequiredToBuy(int* tickets, int ticketsSize, int k){
    int     time = 0;
    int     i;
    
    while (tickets[k])
    {
        for (i = 0; i < ticketsSize; ++i)
        {
            time += (tickets[i] && tickets[k]);
            tickets[i] -= !!tickets[i];
        }
    }
    return (time);
}
