

bool digitCount(char * num){
    int i,map[10]={0};      //map that store the occurance of character in string
    for(i=0;num[i];i++)
    {
        map[num[i]-'0']++;  //store the occurance 
    }
    for(i=0;num[i];i++)
    {
        
        if(num[i]!=map[i]+48)   //now check if it matches with the num[]
            return false;
    }
    return true;
}
