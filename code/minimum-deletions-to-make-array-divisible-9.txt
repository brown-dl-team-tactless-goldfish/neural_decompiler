    public int MinOperations(int[] nums, int[] numsDivide)
    {
        int numsDivideGcd = FindArrayGcd(numsDivide);
        Array.Sort(nums);
        for (int i = 0; i < nums.Length && nums[i] <= numsDivideGcd; i++)
            if (numsDivideGcd % nums[i] == 0)
                return i;
        return -1;
    }

    static int FindArrayGcd(int[] arr)
    {
        int result = arr[0];
        for (int i = 1; i < arr.Length && result != 1; i++)
            result = Gcd(arr[i], result);
        return result;
    }
    static int Gcd(int a, int b)
    {
        if (a == 0)
            return b;
        return Gcd(b % a, a);
    }
