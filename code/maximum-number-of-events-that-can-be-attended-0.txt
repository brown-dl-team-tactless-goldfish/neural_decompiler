#include <stdlib.h>

int compare (const void* a, const void* b) {
    int sa = (*(int**)a)[0];
    int ea = (*(int**)a)[1];
    int sb = (*(int**)b)[0];
    int eb = (*(int**)b)[1];
    
    if (ea > eb) return 1;
    else if (ea < eb) return -1;
    else {
        if (sa < sb) return -1;
        else if (sa > sb) return 1;
        
        return 0;
    }
}

int maxEvents(int** events, int eventsSize, int* eventsColSize){
    int i = 0, j = 0, last = 0, update = 0, evnts_cnt = 0;
    char days_map[100000] = {0};
    
    qsort (events, eventsSize, sizeof(int)*2, compare);
        
    for (i = 0; i < eventsSize; i ++) {
//        printf ("i %u, start %u end %u\n", i, events[i][0], events[i][1]);
        if (i == 0) {
            days_map[events[i][0] - 1] = 1;
            if (events[i][0] > 1)
                last = 0;
            else 
                last = 1;
            evnts_cnt ++;
            continue;
        }
      
        if ((events[i][0] - 1) >= last)
            j = events[i][0] - 1;
        else {
            j = last;
            update = 1;
        }
        
        for (; j <= (events[i][1] - 1); j ++) {
            if (!days_map[j]) {
                days_map[j] = 1;
                evnts_cnt ++;
                if (!update)
                    break;
                
                for (;j < eventsSize; j ++) {
                    if (!days_map[j]) {
                        last = j;
                        update = 0;
                        break;
                    }
                }
                
                break;
            } 
        }
    }
        
        
    return evnts_cnt;
}
