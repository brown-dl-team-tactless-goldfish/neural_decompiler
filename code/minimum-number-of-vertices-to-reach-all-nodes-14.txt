class Solution {
public:  
    vector<int> findSmallestSetOfVertices(int n, vector<vector<int>>& edges) 
    {
        vector<int> in(n);
        for(int i = 0; i < edges.size(); i++)
            in[edges[i][1]]++;
        
        vector<int> res;
        for(int i = 0; i < in.size(); i++)
            if(in[i] == 0)
                res.push_back(i);
        
        return res;
    }
};
