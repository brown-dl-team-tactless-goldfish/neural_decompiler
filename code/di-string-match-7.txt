int* diStringMatch(char * S, int* returnSize)
{
    int n = strlen(S);
    int *A = malloc( sizeof(int) * (n+1) );
    *returnSize = n + 1;

    int l = 0, r = n;
    for (int i = 0; i < n; i++)
    {
        if (S[i] == 'I') { // 下一个数增大
            A[i] = l++;
        } else { // 下一个数减小
            A[i] = r--;
        }
    }

    A[n] = l;
    return A;
}
