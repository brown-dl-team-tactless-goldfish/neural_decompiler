class Solution {
public:
    int countTriplets(std::vector<int>& arr) {
        const auto n = arr.size();

        int result = 0;

        for ( int i = 0; i + 1 < n; ++i ) {
            auto xr = arr[i];

            for ( int j = i+1; j < n; ++j ) {
                xr ^= arr[j];
                if ( xr == 0 )
                    result += (j-i);
            }
        }

        return result;
    }
};
