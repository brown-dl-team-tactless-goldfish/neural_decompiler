    long long left = 0, right = 0, sum = 0, score, count = 0;
    
    for(right = 0; right<nums.size(); right++)
    {
        sum += nums[right];
        
        score = sum * (right-left+1);
        
        while(left <= right && score >= k)
        {
            sum -= nums[left++];
            
            score = sum*(right-left+1);
        }
        
        if(score < k)
        {
            count += right - left + 1;
        }
    }
    
    return count;
}
