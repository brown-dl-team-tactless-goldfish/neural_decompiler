bool digitCount(char * num){
    int hash[10] = {0};
    for (int i = 0; i < strlen(num); i++) {
        hash[num[i] - 48]++;
    }
    for (int i = 0; i < strlen(num); i++) {
        if (hash[i] < (num[i] - 48) || hash[i] > (num[i] - 48)) {
            return false;
        }
    }
    return true;
}
