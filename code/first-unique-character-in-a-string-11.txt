#include <string.h>


int firstUniqChar(char * s){
    
    int len_s = strlen(s);

    for (int i = 0; i < len_s; i++) {
        
        int j;
        for (j = 0; j < len_s; j++) {
            if (i == j)
                // skip comparing same char
                continue;
            
            if (s[i] == s[j]) {
                // match found, try next char
                break;
            }
        }
        if (j == len_s)
            // match not found
            return i;
    }
    return -1;
}
