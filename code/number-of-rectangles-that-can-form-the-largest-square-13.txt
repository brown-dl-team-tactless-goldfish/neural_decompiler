class Solution {
public:
    int countGoodRectangles(vector<vector<int>>& rectangles) {
        vector<int> res;
        int maxi = INT_MIN;
        int d = 0;
        for(vector<int> v : rectangles)
        {
            d = min(v[0],v[1]);
            res.push_back(d);
            maxi = max(maxi,d);
        }
        
        int c = 0;
        for(int i : res)
        {
            if(i==maxi) c++;
        }
        return c;
    }
};
