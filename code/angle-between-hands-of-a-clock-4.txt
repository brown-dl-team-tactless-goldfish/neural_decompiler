double angleClock(int hour, int minutes){
    double m=(double)minutes/60*360;
    double h=(double)(hour%12)/12*360;
    double m_in_h=(double)minutes/60/12*360;
    double hm=h+m_in_h;
      
    double ret=m-hm;
    if(ret<0)
        ret=-ret;
    if(ret>180)
        ret=360-ret;
        
    return ret;
}
