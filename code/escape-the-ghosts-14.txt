class Solution {
public:
    bool escapeGhosts(vector<vector<int>>& ghosts, vector<int>& target) {
        // If the distance b/w a ghost and the target is less than the one b/w us and the target,
        // then the ghost can always hit us. Otherwise the ghost can never hit us. Hence we will
        // fail the game as long as there exists a ghost whose distance to the target <= distance b/w us and the target.
        int dist = abs(target[0]) + abs(target[1]);
        for (const auto& g : ghosts)
        {
            if (abs(g[0] - target[0]) + abs(g[1] - target[1]) <= dist)
            {
                return false;
            }
        }
        return true;
    }
};
