int compare_func(int *a, int *b){
    return (*a - *b);
}

int findGCD(int* nums, int numsSize){
    qsort(nums, numsSize, sizeof(int), compare_func);
    int euc_remainder = nums[numsSize-1] % nums[0];
    int i, GCD = 1;
    if(euc_remainder == 0){
         return nums[0];
    }else{
        for (i=1; i<=nums[0] && i<=euc_remainder; i++)
        {
            if(nums[0]%i == 0 && euc_remainder%i == 0)
            {
                GCD = i;
            }
        }     
    }
    return GCD;
}
