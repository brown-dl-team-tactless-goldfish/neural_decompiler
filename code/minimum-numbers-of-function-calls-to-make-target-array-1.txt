int minOperations(int* nums, int numsSize){
//  option 0 is one idx in arr (+ 1)
//  option 1 is all arr (* 2)
    if(numsSize==1&&nums[0]==0) return 0;
    int high_bit=-1;
    int one_count=0;
    int res=0;
    for(int i=0;i<numsSize;i++){
        one_count=__builtin_popcount(nums[i]);
        res+=one_count;
        int temp=0;
        while(nums[i]!=0){
            temp++;
            nums[i]>>=1;
        }
        high_bit = fmax(high_bit,temp);
    } 
    return res+high_bit-1;
}
