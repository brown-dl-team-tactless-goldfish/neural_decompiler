bool isPalindrome(char * s){
    int len = strlen(s);
    int low = 0, high = len - 1;
    
    while(low < high){
        if(isalnum(s[low]) && isalnum(s[high])) {
            if(tolower(s[low]) != tolower(s[high]))
                return false;
            low++;
            high--;
        }
        if(!isalnum(s[low]))
            low++;
        if(!isalnum(s[high]))
            high--;        
    }
    return true;
}
