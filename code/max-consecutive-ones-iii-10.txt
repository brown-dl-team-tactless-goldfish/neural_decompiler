int longestOnes(vector<int>& nums, int k) {
        
        int zeros=0, i=0, j=0, ans=0; 
        int n = nums.size();
        
        while(j < n){
            
            if(nums[j] == 0) zeros++;
            
            while(zeros > k){
                if(nums[i] == 0) zeros--;
                i++;
            }
            
            ans = max(ans, j-i+1);
            j++;
        }
        
        return ans;
    }
