double findMaxAverage(int* nums, int numsSize, int k) {
    int sum,sumMax;   
    int i;
    
    sum = 0;
    for(i=0;i<k;i++){
        sum+= nums[i];
    }
    sumMax=sum;
    
    for (i=1;i<=(numsSize-k);i++){
        sum+= nums[i+k-1]-nums[i-1];
        if (sum>sumMax)
            sumMax =sum;
    }
    return (double)sumMax/(double)k;       
}
