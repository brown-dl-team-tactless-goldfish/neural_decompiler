class Solution {
public:
    int minPatches(vector<int>& nums, int n) 
    {
        int res=0,i=0;
        long long sum=1;
        while(sum<=n)
        {
            if(i<nums.size() && nums[i]<=sum)
                sum+=nums[i++];
            else
            {
                sum+=sum;
                res++;
            }
        }
        return res;
    }
};
