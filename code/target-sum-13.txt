class Solution {
public:
    void solve(int &c, int sum, int i, vector<int>& nums, int target){
        int n = nums.size();
        if(i >= n) {
            if(sum == target) c++;
            return;
        }    
        solve(c, sum-nums[i], i+1, nums, target);
        solve(c, sum+nums[i], i+1, nums, target);
    }
    
    int findTargetSumWays(vector<int>& nums, int target) {
        int c = 0;
        solve(c, 0, 0, nums, target);
        return c;
    }
};
