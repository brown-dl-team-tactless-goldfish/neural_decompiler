int maximumProduct(int* nums, int numsSize){
    int m[3] = { INT_MIN, INT_MIN, INT_MIN, }, n[2] = { INT_MAX, INT_MAX, }, p1, p2, *p = nums;
    for (p-- ; numsSize-- && p++ ; n[0] > *p ? n[1] = n[0], n[0] = *p : n[1] > *p ? n[1] = *p : 0)
        m[0] < *p ? m[2] = m[1], m[1] = m[0], m[0] = *p :
        m[1] < *p ? m[2] = m[1], m[1] = *p :
        m[2] < *p ? m[2] = *p : 0;
    return (p1 = m[0] * m[1] * m[2]) > (p2 = m[0] * n[0] * n[1]) ? p1 : p2;
}
