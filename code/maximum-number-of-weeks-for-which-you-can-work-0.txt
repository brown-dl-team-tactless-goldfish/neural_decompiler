long long numberOfWeeks(int* milestones, int milestonesSize){
    long long sum = 0 ;
    int max = -1 ;
    for(int i = 0; i < milestonesSize; i++){
        sum += milestones[i] ;
        max = fmax(max, milestones[i]) ;
    }
    if(max <= sum/2)
        return sum ;
    else
        return (sum - max)*2 + 1 ;
}
