#define MIN(a, b) (a) < (b) ? (a) : (b)

int minCostToMoveChips(int* chips, int chipsSize){
    int even = 0;
    int odd = 0;
    for (int i = 0; i < chipsSize; ++i) {
        if ((chips[i] & 0x1) == 0) {
            even++;
        }
        else {
            odd++;
        }
    }
    return MIN(even, odd);
}
