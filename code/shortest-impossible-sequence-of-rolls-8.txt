class Solution {
public:
    int shortestSequence(vector<int>& rolls, int k) 
    {
        int n=rolls.size(),res=1;
        unordered_set<int>s;
        for(auto &x:rolls)
        {
            s.insert(x);
            if(s.size()==k)
            {
                res++;
                s.clear();
            }
        }
        return res;
    }
};
