typedef struct {
    int *n;
} NumArray;

NumArray* numArrayCreate(int* nums, int numsSize) {
    NumArray *na;
    for (int *n = (na = malloc(sizeof *na))->n = nums, i = 0 ; ++i < numsSize ; n[i] += n[i - 1]);
    return na;
}

int numArraySumRange(NumArray* obj, int i, int j) {
    return obj->n[j] - (i ? obj->n[i - 1] : 0);
}

void numArrayFree(NumArray* obj) {
    free(obj);
}
