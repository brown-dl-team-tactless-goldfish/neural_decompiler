class Solution {
public:
    int minimumNumbers(int num, int k) {
        if(num == 0)    return 0;
        if(num < k) return -1;
        
        int tmp = num%10;
        int n = k;
		
        for(int i=1; i<=10; i++){
            if((n*i)%10 == tmp and n*i <= num){
                return i;
            }
        }
        return -1;
    }
};
