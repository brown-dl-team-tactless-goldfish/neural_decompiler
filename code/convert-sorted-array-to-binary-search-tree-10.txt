struct TreeNode *arrToTree(int *nums, int low, int high)
{
  if(/*!nums ||*/ low>high)
    return NULL;
  int mid = low+(high-low)/2;
  struct TreeNode *res = malloc(sizeof(*res));
  res->val=nums[mid];
  res->left=arrToTree(nums, low, mid-1);
  res->right=arrToTree(nums, mid+1, high);
  return res;
}

struct TreeNode *sortedArrayToBST(int *nums, int numsSize)
{
  return !nums ? NULL : arrToTree(nums, 0, numsSize-1);
}
