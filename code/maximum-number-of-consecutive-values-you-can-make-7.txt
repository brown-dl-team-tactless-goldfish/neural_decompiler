class Solution {
public:
    int getMaximumConsecutive(vector<int>& coins) {
        int c = 0;
        sort(coins.begin(), coins.end());
        for (int i = 0; i < coins.size(); i++) {
            int cc = coins[i];
            if (cc <= c + 1) {
               c += cc; 
            }
            else {
                return c + 1;
            }
        }
        return c + 1;
    }
};
