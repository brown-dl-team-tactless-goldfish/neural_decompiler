public:
    bool canConvertString(string s, string t, int k) {
        if (s.size() != t.size()) return false;
        
        int ans = 0;
        vector<int> bucket(26);
        for (int i = 0; i < s.size(); ++ i){
            if (s[i] == t[i]) continue;
            int temp = (-s[i] + t[i]) > 0 ? (-s[i] + t[i]) : (-s[i] + t[i]) + 26;
            ans = max(ans, (bucket[temp]) * 26 + temp);
            bucket[temp] ++;
        }
        
        return ans <= k;
    }
