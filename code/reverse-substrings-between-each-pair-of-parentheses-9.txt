class Solution {
public:
    string reverseParentheses(string s) {
        string ret ;
        stack<int>stk ;
        for(auto& c : s){
            if(isalpha(c))
                ret.push_back(c) ;
            else if(c == '(')
                stk.push( ret.size()) ;
            else{
                int start = stk.top() ;
                stk.pop() ;
                reverse(ret.begin() + start, ret.end()) ;
            }
        }
        return ret ;
    }
};
