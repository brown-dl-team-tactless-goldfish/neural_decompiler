inline int min(int a, int b) { return a < b ? a : b; }

int distributeCandies(int* candies, int candiesSize){
    char tbl[200001];
    memset(tbl, 0x01, 200001 * sizeof(char));
    
    int cnt = 0;
    for (int i = 0; i < candiesSize; ++i) {
        int key = candies[i] + 100000;
		
        cnt      = cnt + tbl[key];
        tbl[key] = 0x00;
    }
    
    return min(candiesSize/2, cnt);
}
