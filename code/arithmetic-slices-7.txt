int numberOfArithmeticSlices(int* nums, int numsSize){
int count = 0; // 用於計數的變數
for (int i = 0; i < numsSize - 2; i++) { // 遍歷陣列
int diff = nums[i + 1] - nums[i]; // 計算相鄰元素之差
int j = i + 2; // j 是用於檢查的指標
while (j < numsSize && nums[j] - nums[j - 1] == diff) { // 如果 j 指向的元素是算術子陣列的一部分，則進行迭代
count++; // 將計數器加 1
j++; // 將 j 向前移動一個位置
}
}
return count; // 返回計數器
}

//此函式使用兩個迴圈遍歷整數陣列。第一個迴圈以每個數字為起點，檢查其後面是否存在算術子陣列。第二個迴圈用於檢查算術子陣列的長度。

//在每次迭代中，它會計算相鄰元素的差，然後使用另一個指標檢查後續元素是否符合算術陣列的要求。如果是，則將計數器加 1。

//最後，函式返回計數器的值。
