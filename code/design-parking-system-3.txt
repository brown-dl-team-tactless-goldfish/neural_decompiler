typedef struct {
    int * type;
} ParkingSystem;


ParkingSystem* parkingSystemCreate(int big, int medium, int small) {
    ParkingSystem *result = (ParkingSystem*)malloc(sizeof(ParkingSystem));
    result->type = (int *)malloc(3 * sizeof(int));
    result->type[0] = big;
    result->type[1] = medium;
    result->type[2] = small;
    return result;
}

bool parkingSystemAddCar(ParkingSystem* obj, int carType) {
    if(obj->type[carType - 1] > 0)
    {
        obj->type[carType - 1]--;
        return true;
    }
    else return false;
}

void parkingSystemFree(ParkingSystem* obj) {
    free(obj->type);
    free(obj);
}
