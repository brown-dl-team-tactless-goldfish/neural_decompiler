bool threeConsecutiveOdds(int* arr, int arrSize){
    int i, count=0;
    for(i=0; i<arrSize;i++)
    {
        if(arr[i]%2!=0)
        {
            count++;
            if(count==3)
            {
                break;
            }
        }
        else
            count=0;
    }
    if(count==3)
        return 1;
    else
        return 0;
}
