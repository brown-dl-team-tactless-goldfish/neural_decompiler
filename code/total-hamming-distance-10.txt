class Solution {
public:
    int totalHammingDistance(vector<int>& nums) {
        int ans = 0, b = 1;
        while (b) {
            int c = 0;
            for (int num : nums) {
                if (num & b)
                    ++c;
            }
            ans += c * (nums.size() - c);
            b <<= 1;
        }
        return ans;
    }
};
