class Solution {
public:
    long long countFairPairs(vector<int>& v, int L, int U) {
        int n=v.size();
        long long ans=0;
        sort(v.begin(),v.end());
        for(int i=0;i<n;i++){
            int j=lower_bound(v.begin()+i+1,v.end(),L-v[i])-v.begin();
            int k=lower_bound(v.begin()+i+1,v.end(),U-v[i]+1)-v.begin();
            ans+=(k-j);
        }
        return ans;
    }
};
