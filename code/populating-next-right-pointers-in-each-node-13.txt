struct Node* connect(struct Node* root) {
	if(root==NULL || root->left==NULL || root->right==NULL) return root;
	// if we have a left and right node then connect them from the root (connect 2 to 3 from 1)
    root->left->next=root->right;
	// if there exists a next node (like in 2->3) connect the right of 2 to the left of 3
    if(root->next!=NULL){
        root->right->next=root->next->left;
    }
	//recursivly call
    connect(root->left);
    connect(root->right);
    return root;
}
