map<string,int>mapp;
int maxans=0;

void getMaxAns(string &str, int index, int words){
    if(index>=str.length()){
        maxans=max(maxans, words);
        return ;
    }
    string curr="";
    for(int j=index;j<str.length();j++){
        curr+=str[j];
		// store if this string is not occured previously
        if(mapp[curr]==0){
            mapp[curr]++;
            getMaxAns(str,j+1,words+1);
            mapp[curr]--; // during backtrack reduce it by 1
           
    }
}
int maxUniqueSplit(string str) {
    
    getMaxAns(str,0,0);
    
    return maxans;
}
