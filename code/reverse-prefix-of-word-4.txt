char * reversePrefix(char * s, char ch)
{
    
    char temp;
    int i=0,end=0,start=0;
    
    
    for(i=0;i<strlen(s)&&s[i]!=ch;i++)
    {
        
          end++;
    }
    if(i<=strlen(s)-1&&s[i]==ch)
    {
        while(end>=start)
        {
            temp=s[end];
            s[end--]=s[start];
            s[start++]=temp;
        }
    }
    
   return s; 
    
}
