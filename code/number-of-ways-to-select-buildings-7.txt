class Solution {
public:
    long long numberOfWays(string s) {
        return f(s,"101") + f(s,"010");
    }
    long long f(string s,string t) {
        long long c = 0,a = 0, b = 0;
        for(auto i : s) {
            if(i == t[2]) c += b;
            if(i == t[1]) b += a;
            if(i == t[0]) a += 1;
        }
        return c;
    }
};
