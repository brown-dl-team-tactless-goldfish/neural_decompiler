class Solution {
public:
	vector<int> findLonely(vector<int>& nums) {
		map<int, int> m;
		for(int i = 0; i < nums.size(); i++){
			m[nums[i]]++;
		}
		vector<int> res;
		for(auto i : m){
			if(i.second == 1 && m[i.first + 1] == 0 && m[i.first - 1] == 0){
				res.push_back(i.first);
			}
		}
		return res;
	}
};
