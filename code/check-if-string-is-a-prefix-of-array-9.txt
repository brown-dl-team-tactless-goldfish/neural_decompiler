class Solution {
public:
    bool isPrefixString(string s, vector<string>& words) {
        string c;
        int cnt = 0;
        for (int i=0; i<words.size(); i++) {
            c += words[i];
            cnt += words[i].size();
            if (c != s.substr(0, cnt)) return false;
            if (c == s) return true;
        }
        return false;
    }
};
