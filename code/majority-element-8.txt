int compare(const void *p1, const void *p2) {
    return *(int *)p1 - *(int *)p2;
}


int majorityElement(int* nums, int numsSize){
    
    qsort(nums, numsSize, sizeof(int), compare);

    int majority = numsSize / 2;
    int count = 1;
    
    for (int i = 1; i < numsSize; i++) {
        if (nums[i] == nums[i-1]) {
            count++;
            
            if (count > majority)
                return nums[i];
        } else {
            count = 1;
        }
    }
    return nums[0];
}
