class Solution {
public:
    vector<int> vowelStrings(vector<string>& w, vector<vector<int>>& q) {
        vector<int> preSum(w.size()+1), res;
        string v = "aoeiu";
        for(int i = 0; i < w.size(); i++) preSum[i+1] = preSum[i]+(v.find(w[i][0]) != string::npos && v.find(w[i][w[i].size()-1]) != string::npos);
        for(int i = 0; i < q.size(); i++) res.push_back(preSum[q[i][1]+1]-preSum[q[i][0]]);
        return res;
    }
};
