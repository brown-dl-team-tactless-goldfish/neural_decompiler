int cmp(const void *a, const void *b)
{
    return *(int *)a-*(int *)b;
}
int minimumOperations(int* nums, int numsSize){

int i=0,j=0,cnt=0;
qsort(nums,numsSize,sizeof(int),cmp);
j=nums[0];
cnt = (j==0) ? 0 : 1;
for(i=1;i<numsSize;)
{
	while((i<numsSize) && (nums[i] == j)) i++;
	if(i<numsSize)
	{
		j = nums[i];
		cnt++;
	}
}
return cnt;
}
