class Solution {
public:
    int repeatedStringMatch(string A, string B) {
        int i = 1;
        string s(A);
        while (s.size() < B.size()) {
            s += A;
            i++;
        }
        if (s.find(B) != string::npos) return i;
        if ((s + A).find(B) != string::npos) return i + 1;
        return -1;
    }
};
