class Solution {
public:
    int minimumBuckets(string s) {
        int n=s.size();
        if(n==1) return s[0]=='.'?0:-1;
        int ans=0;
        for(int i=0;i<n;i++){
            if(s[i]=='H'){
                if(i>0&&s[i-1]=='B') continue;
                else if(i+1<n&&s[i+1]=='.'){s[i+1]='B';ans++;}
                else if(i>0&&s[i-1]=='.'){s[i-1]='B';ans++;}
                else return -1;
            }
        }
        return ans;
    }
};
