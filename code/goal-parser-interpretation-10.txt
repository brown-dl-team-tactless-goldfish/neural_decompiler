while(*command != '\0'){
    if(*command == 'G'){
        result[cnt++] = 'G';
    }
    else if(*command == 'l' && *(command-1) == 'a'){
        result[cnt++] = 'a';                     
        result[cnt++] = 'l';                     
    }            
    else if(*command == ')' && *(command-1) == '('){
        result[cnt++] = 'o';                     
    }        
    command++;
}

result[cnt] = '\0';

return result;
