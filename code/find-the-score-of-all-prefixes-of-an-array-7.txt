class Solution {
public:
    typedef long long ll;
    vector<long long> findPrefixScore(vector<int>& nums) {
        vector<ll> ans;
        int n = nums.size();
        ll sum = 0,maxVal=0,val;
        for(int i=0;i<n;i++){
            maxVal = max(maxVal, (ll)nums[i]);
            val = nums[i] + maxVal;
            sum += val;
            ans.push_back(sum);
        }
        return ans;
    }
};
