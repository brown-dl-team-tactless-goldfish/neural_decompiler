char ** findOcurrences(char * text, char * first, char * second, int* returnSize){
    char **o = malloc(sizeof(char *[1000])), *w[1000], *f = first, *s = second, *t = text, *p, i, j, k;
    for (j = 0 ; *t ; w[j++] = t, p = strchr(t, ' '), t = p ? *p++ = 0, p : "");
    for (i = k = 0 ; k + 2 < j ; !strcmp(w[k], f) && !strcmp(w[k + 1], s) ? o[i++] = w[k += 2] : k++);
    return *returnSize = i, o;
}
