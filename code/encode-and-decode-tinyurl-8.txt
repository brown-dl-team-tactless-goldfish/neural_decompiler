class Solution {
public:

    unordered_map<string, string> mp;
    char c = 'a';
    string s = "http://tinyurl.com/";
    
    // Encodes a URL to a shortened URL.
    string encode(string longUrl) {
        for(int i=0; i<5; i++) {
            int randNo = rand()%26;
            s += (c+randNo);
        }
        mp[s] = longUrl;
        return s;
    }

    // Decodes a shortened URL to its original URL.
    string decode(string shortUrl) {
        return mp[shortUrl];
    }
};
