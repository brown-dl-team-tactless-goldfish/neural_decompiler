

int strStr(char * string1, char * string2)
{
int len1, len2 = 0;
len1 = strlen(string1);
len2 = strlen(string2);
int i =0,counter =0;
for(i=0;i<len1;i++)
{
if(counter==len2)
break;
if(string2[counter]==string1[i])
{
counter++;
}
else
{
if(counter > 0)
{
i -= counter;
}
counter = 0;
}
}
return counter < len2 ?-1:i-counter;  
}
