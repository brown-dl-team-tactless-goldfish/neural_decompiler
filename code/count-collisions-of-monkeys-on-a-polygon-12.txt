class Solution {
public:
    int monkeyMove(int n) {
        long long res = 1, mod = 1e9+7;
        vector<long long> pows(32, 2);
        for(int i=1;i<32;i++) pows[i] = pows[i-1] * pows[i-1] % mod;
        for(int i=0;i<31;i++) if(n & (1<<i)) res = res * pows[i] % mod;
        return (res + mod - 2) % mod;
    }
};
