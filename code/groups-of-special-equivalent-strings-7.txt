public class Solution {
    public int NumSpecialEquivGroups(string[] A) {
            var set = new HashSet<string>();

            foreach (var s in A)
            {
                var arr = new int[52];

                for (var i = 0; i < s.Length; i++)
                    arr[s[i] - 'a' + 26 * (i % 2)]++;

                set.Add($"{string.Join(",", arr)}");
            }

            return set.Count;
    }
}
