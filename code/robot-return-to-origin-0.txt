bool judgeCircle(char * moves){
    int temp[4] = {0}; //0=L, 1=R, 2=U, 3=D
    for(int i=0; i<strlen(moves); i++)
        if(moves[i] == 'L') temp[0]++;
        else if(moves[i] == 'R') temp[1]++;
        else if(moves[i] == 'U') temp[2]++;
        else if(moves[i] == 'D') temp[3]++;
    return (temp[0] == temp[1]) && (temp[2] == temp[3]);
}
