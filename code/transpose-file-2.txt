# Read from the file file.txt and print its transposed content to stdout.
declare -A transposed
nrows=0
ncols=0
while read -r -a row; do
    for ((i = 0; i < ${#row[@]}; i++)); do
        transposed[$i,$nrows]=${row[i]}
    done
    ((nrows++))
    ncols=${#row[@]}
done < file.txt

for ((r = 0; r < ncols; r++)); do
    sep=
    for ((c = 0; c < nrows; c++)); do
        printf '%s%s' "$sep" "${transposed[$r,$c]}"
        sep=" "
    done
    printf '\n'
done
