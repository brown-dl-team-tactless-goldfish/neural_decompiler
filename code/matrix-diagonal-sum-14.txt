int diagonalSum(int** mat, int matSize, int* matColSize){
    int primary = 0;
    int secondary = 0; 
    for (int k = 0; k < matSize; k++) {
        primary += mat[k][k];
        secondary += mat[matSize - k - 1][k];
    }
    
    int total = primary + secondary;
    
    if (matSize % 2 == 1) {
        int middle = matSize / 2;
        total -= mat[middle][middle];
    }
    
    return total;
}
