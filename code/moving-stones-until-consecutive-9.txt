class Solution {
public:
    vector<int> numMovesStones(int a, int b, int c) 
    {
        const int x = min(a, min(b,c));
        const int z = max(a, max(b,c));
        const int y = a ^ b ^ c ^ x ^ z;
        
        const int max_moves = z - x - 2;
        const int min_moves = min(max_moves, (y - x <= 2 || z - y <= 2) ? 1 : 2);
        return {min_moves, max_moves};
    }
};
