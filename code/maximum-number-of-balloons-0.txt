int maxNumberOfBalloons(char * text){
    int n = strlen(text);
    int* alpha = calloc(26, sizeof(int));
    for(int i = 0; i < n; i++){
        alpha[text[i] - 'a']++ ;
    }
    int ans =  n;
    ans = fmin(alpha['a' - 'a'], ans);
    ans = fmin(alpha['b' - 'a'], ans);
    ans = fmin(alpha['l' - 'a']/2, ans);
    ans = fmin(alpha['o' - 'a']/2, ans);
    ans = fmin(alpha['n' - 'a'], ans);
    return ans;
}
