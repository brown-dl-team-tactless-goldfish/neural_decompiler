class Solution:
    def maximumScore(self, nums: List[int], mults: List[int]) -> int:
        dp = [[0 for x in range(len(mults))] for y in range(len(mults))] 
        def dfs(l: int, i: int):
            if i >= len(mults):
                return 0;
            if not dp[l][i]:
                r = len(nums) - 1 - (i - l)
                dp[l][i] = max(nums[l] * mults[i] + dfs(l + 1, i + 1), nums[r] * mults[i] + dfs(l, i + 1))
            return dp[l][i]
        return dfs(0, 0)
