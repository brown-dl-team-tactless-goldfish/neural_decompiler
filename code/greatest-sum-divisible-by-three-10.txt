class Solution {
public:
    int maxSumDivThree(vector<int>& nums) {
        vector<vector<int>> dp(nums.size() + 1, vector<int>(3, INT_MIN));
        dp[0][0] = 0;
        for (int i = 1; i <= nums.size(); i++) {
            int mod = nums[i-1] % 3;
            for (int k = 0; k < 3; k++) {
                dp[i][k] = max(dp[i-1][k], dp[i-1][(k - mod + 3) % 3] + nums[i-1]);
            }
        }
        
        return dp[nums.size()][0];
    }
};
