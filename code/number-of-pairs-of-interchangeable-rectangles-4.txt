class Solution {
public:
    long long interchangeableRectangles(vector<vector<int>>& rect) {
        
        int n = rect.size();
        unordered_map<double,int>track;
        double t;
        long long int res = 0;
        for(int i=0;i<n;i++)
        {
            t = (double)rect[i][0]/(double)rect[i][1];
            res+= track[t];
            track[t]++;
        }
        
        return res;
    }
};
