bool rotateString(char * s, char * goal){
    int n1 = strlen(s);
    int n2 = strlen(goal);
    if(n1 != n2)
        return false;

    for(int i = 0; i < n1; i++){
        int j;
        for(j = 0; j < n1; j++){
            if(s[(j+i)%n1] != goal[j])
                break;
        }
        if(j == n1)
            return true;
    }
    return false;
}
