class Solution {
public:
    int wateringPlants(vector<int>& plants, int capacity) {
        int n = plants.size(), cap = capacity, steps = 0; 
        for (int i = 0; i < n; i++) { 
            if (capacity < plants[i]) return -1; 
            if (cap < plants[i]) {
                cap = capacity; 
                steps += 2*i; 
            }
            steps++; 
            cap -= plants[i]; 
        }
        return steps;
    }
};
