class RangeFreqQuery {
    
    vector< int > t[10000+1];
    
public:
    RangeFreqQuery(vector<int>& A ) {
        int N = A.size();
        for( int i = 0 ; i < N ; i++ )
            t[A[i]].push_back(i);
    }
    
    int query(int a , int b , int x ) {
        
        auto it = upper_bound( t[x].begin() , t[x].end() , b );
        auto it1 = lower_bound(t[x].begin() , t[x].end() , a );
        return it-it1;
        
    }
};
