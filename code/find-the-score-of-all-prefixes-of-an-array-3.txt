using LL = long long ;
class Solution {
public:
    vector<long long> findPrefixScore(vector<int>& nums) {
        int n = nums.size() ;
        vector<LL>ret(n) ;
        ret[0] = (LL)2*nums[0] ;
        int M = nums[0] ;
        for(int i = 1; i < n; i++){
            M = max(M, nums[i]) ;
            ret[i] = ret[i-1] + M + nums[i] ;
        }
        return ret ;
    }
};
