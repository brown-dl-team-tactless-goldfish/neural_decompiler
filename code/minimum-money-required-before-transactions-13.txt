long long minimumMoney(vector<vector<int>>& transactions) {
        
        long long int res = 0 ;
        
        for(auto t : transactions)
        {
            long long int minVal = min(t[0], t[1]) ;
            res = max(res, minVal) ; 
        }   
        for(auto t : transactions)   res += max(t[0] - t[1], 0) ;
            
        return res ;
    }
