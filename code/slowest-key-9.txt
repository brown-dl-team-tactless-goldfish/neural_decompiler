public class Solution {
    public char SlowestKey(int[] releaseTimes, string keysPressed) {
        var dict = new Dictionary<char, int>(); // key the keyPressed, value would be duration
        int n = releaseTimes.Length;
        
        for (int i=0; i<n; i++) {
            var c = keysPressed[i];
            var duration = i == 0 ? releaseTimes[0] : releaseTimes[i]-releaseTimes[i-1];
            if (dict.ContainsKey(c)) {
                dict[c] = Math.Max(dict[c], duration);
            } else {
                dict[c] = duration;
            }
        }
        
        char res = '\0';
        foreach (var item in dict.OrderByDescending(x => x.Value)) {
            if (res == '\0') {
                res = item.Key;
            }
            if (dict[res] > item.Value) {
                break;
            } else if (res < item.Key) {
                res = item.Key;
            }
        }
        return res;
    }
}
