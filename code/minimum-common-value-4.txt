int getCommon(int* nums1, int nums1Size, int* nums2, int nums2Size) {
	register unsigned short i = 0, j = 0;

    if(nums1[nums1Size-1] < nums2[0] || nums2[nums2Size-1] < nums1[0])
        return -1;
        
	while(i < nums1Size && j < nums2Size) {
		if(nums1[i] < nums2[j])
			i++;
		else if(nums1[i] > nums2[j])
			j++;
		else
			return nums1[i];
	}
	return -1;
}
