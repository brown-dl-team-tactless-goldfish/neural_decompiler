class Solution {
public:
    bool canConstruct(string s, int k) {
        unordered_map<char,int> m;
        for (auto& c : s) ++m[c];
        int oddCount = 0, evenSum = 0;
        for (auto& [f,s] : m){
            if (s % 2 == 1) ++oddCount;
            evenSum += s - (s % 2 == 1);
        }
        if (oddCount > k) return false;
        return (k -= evenSum + oddCount) <= 0;
    }
};
