char * mergeAlternately(char * word1, char * word2){
    uint8_t len1 = strlen(word1);
    uint8_t len2 = strlen(word2);
    uint8_t len_total = len1+len2+1;

    char* res = (char*)malloc(len_total*sizeof(char));
    uint8_t i = 0, j = 0, k = 0;

    while(k < len_total-1){
        if(i < len1)
            res[k++] = word1[i++];
        if(j < len2)
            res[k++] = word2[j++];
    }
    res[len_total-1] = '\0';
    return res;
}
