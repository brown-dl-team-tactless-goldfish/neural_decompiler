/**
 * Note: The returned array must be malloced, assume caller calls free().
 */

int* findArray(int* pref, int prefSize, int* returnSize){
    int *arr=(int *)malloc(prefSize*sizeof(int));
    int i;
    for(i=0;i<prefSize;i++)
    {
        if(i==0)
           arr[i] = pref[i];
        else
           arr[i]=pref[i]^pref[i-1];
    }
    *returnSize=prefSize;
    return arr;
}
