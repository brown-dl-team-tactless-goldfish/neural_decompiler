int* getSumAbsoluteDifferences(int* nums, int numsSize, int* returnSize) {
    *returnSize = numsSize;
    int sum = 0, tmpsum = 0;
    for(int i = 0; i < numsSize; i++) {
        sum += nums[i];
    }
    
    int* res = (int*) malloc(numsSize * sizeof(int));
    for(int i = 0; i < numsSize; i++) {
        tmpsum += nums[i];
        int tmp = sum + (2 * i - numsSize) * nums[i];
        res[i] = (i == 0) ? tmp : tmp - 2 * (tmpsum - nums[i]);
    }
    return res;
}
