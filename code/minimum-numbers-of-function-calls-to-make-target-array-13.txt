class Solution {
public:
    int minOperations(vector<int>& nums) {
        int a{0}, maxD{0};
        for(const auto &n: nums) {
            auto [aa, dd] = get(n);
            a += aa, maxD = std::max(maxD, dd);
        }
        return a+maxD;
    }
    
    std::pair<int, int> get(int n) {
        int a{0}, d{0};
        while(n) {
            if(n&1) ++a, n -= 1;
            else ++d, n>>=1;
        }
        return {a, d};
    }
};
