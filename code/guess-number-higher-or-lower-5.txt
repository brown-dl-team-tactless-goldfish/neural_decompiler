int guessNumber(int n){
    long long left = 1, right = n;
	while(left < right){
        long long mid = (left + right) / 2;
        int type = guess(mid);
        if(type == 0) return mid;
        else if(type == -1) right = mid - 1;
        else if(type == 1) left = mid + 1;
    }
    return left;
}
