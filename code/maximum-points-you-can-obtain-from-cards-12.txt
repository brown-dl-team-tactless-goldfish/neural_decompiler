
#define min(a,b) (a<b?a:b)

int maxScore(int* cardPoints, int cardPointsSize, int k){
 // Presfix-Sum array
    int PS[cardPointsSize+1];
    PS[0]=0;
    for( int i=0; i<cardPointsSize; ++i)
        PS[i+1]=PS[i]+cardPoints[i];
    int Min=INT_MAX;
// find the minimum sum subarray with length cardPointsSize-k
    for( int i=0; i<=k; ++i)
        Min=min( Min, PS[i+cardPointsSize-k]-PS[i]);
// result will be the sum of the whole array - the sum of minimum sum subarray
    return PS[cardPointsSize]-Min;
}
