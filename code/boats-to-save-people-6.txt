int comp(const void* v1, void* v2) {
  return *(int*) v1 - *(int*) v2;
}

int numRescueBoats(int* people, int peopleSize, int limit) {
  qsort((void*) people, peopleSize, sizeof(int), comp);
  int l = 0, r = peopleSize - 1;
  int count = 0;
  while(r >= 0 && l < r + 1) {
    if(people[r] + people[l] > limit) {
      count++;
      r--;
    }
    else {
      count++;
      r--;
      l++;
    }
  }
  return count;
}
