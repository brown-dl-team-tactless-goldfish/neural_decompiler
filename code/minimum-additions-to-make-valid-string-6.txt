class Solution {
public:
    int addMinimum(string str) {
        int a = 0, b = 0, c = 0, ans = 0;
        int i = 0, n = str.length();
        if(n==1)return 2;
        for(i = 0; i < n; i++){
            if(str[i]=='a'){
                if(i+1<n&&str[i+1]=='b'){
                    if(i+2<n&&str[i+2]=='c'){
                        i++;
                    }else{
                        ans++;
                    }
                    i++;
                }else if(i+1<n&&str[i+1]=='a'){
                    ans += 2;
                }else if(i+1<n&&str[i+1]=='c'){
                    ans++;
                    i++;
                }
                else{
                    ans += 2;
                }
            }else if(str[i]=='b'){
                if(i+1<n&&str[i+1]=='c'){
                    ans++;
                    i++;
                }else if(i+1<n&&str[i+1]!='c'){
                    ans += 2;
                }else{
                    ans += 2;
                }
            }else if(str[i]=='c'){
                ans += 2;
            }
            cout<<ans<<" ";
        }
        // if(str[n-1]=='a')ans += 2;
        // if(n-2>=0&&str[n-1]=='b'&&str[n-2]=='a')ans++;
        // if(n-2>=0&&str[n-1]=='b'&&str[n-2]!='a')ans += 2;
        // if(n-2>0&&str[n-1]=='c'&&str[n-2]!='c')ans += 2;
            return ans;
    }
};
