public class Solution {
    public int EliminateMaximum(int[] dist, int[] speed) {
        var agenda = dist
            .Zip(speed, (d, s) => d / s + Math.Clamp(d % s, 0, 1))
            .OrderBy(item => item);
        
        int t = 0;
               
        foreach (int time in agenda) 
            if (time <= t++)
                return --t;
     
        return t;
    }
}
