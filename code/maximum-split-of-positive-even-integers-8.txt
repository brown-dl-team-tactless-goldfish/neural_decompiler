class Solution {
public:
  vector<long long> maximumEvenSplit(long long fs) {
    if(fs % 2 == 1) return {};
    vector<long long>ans;
    long k = 0;
    while(fs > k){
      k += 2;
      fs -= k;
      ans.push_back(k);
    }
    
    int id = ans.size()-1;
    while(fs) {
      ans[id--] += 2;
      fs-=2;
    } 
    
    return ans;
  }
};
