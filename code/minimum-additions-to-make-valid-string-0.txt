int addMinimum(char* word) {
    int len = strlen(word);
    if (len == 1) return 2;
    int cnt = 0, idx = 0;
    for (idx = 0; idx < len-1; idx++) {
        if (word[idx] == 'a' && word[idx+1] == 'a') {
            cnt += 2;
        } else if (word[idx] == 'b' && word[idx+1] == 'b') {
            cnt += 2;
        } else if (word[idx] == 'c' && word[idx+1] == 'c') {
            cnt += 2;
        } else if (word[idx] == 'a' && word[idx+1] == 'b') {
            if (idx+2 < len && word[idx+2] == 'c') {  
                idx += 2;
                continue;
            }
            cnt++;
            idx++;
        } else if (word[idx] == 'b' && word[idx+1] == 'a') {
            cnt += 2;
        } else if (word[idx] == 'b' && word[idx+1] == 'c') {
            cnt++;
            idx++;
        } else if (word[idx] == 'c' && word[idx+1] == 'b') {
            cnt += 2;
        } else if (word[idx] == 'a' && word[idx+1] == 'c') {
            cnt++;
            idx++;
        } else if (word[idx] == 'c' && word[idx+1] == 'a') {
            cnt += 2;
        }
    }
    return (word[idx] != '\0') ? cnt+2 : cnt;
}
