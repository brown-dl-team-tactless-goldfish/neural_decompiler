#define MAX(a,b) (((a)>(b))?(a):(b))

long long maxAlternatingSum(int* nums, int numsSize){
    long long sum = 0, odd = 0, even = 0;
    // loop run over the array.
    for (int i = 0; i < numsSize; i++){
        // compare between take the even number or not ans save max.
        even = MAX(odd+nums[i], even);
        // compare between minus the add number or not ans save max.
        odd = MAX(even-nums[i], odd);   
        // save maximum.
        sum = MAX(even, MAX(odd,sum));
    }
    return sum;
}
