bool judgeCircle(char * moves){
    int nH=0,nV=0;
    
    while(*moves != '\0'){
        if(*moves == 'R') nH++;
        if(*moves == 'L') nH--;
        if(*moves == 'U') nV++;
        if(*moves == 'D') nV--;
        *moves++;
    }
    
    if(nH == 0 && nV == 0) {
        return true;
    }else{
        return false;
    }
    
}
