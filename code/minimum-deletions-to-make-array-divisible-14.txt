class Solution {
public:
    int minOperations(vector<int>& nums, vector<int>& numsDivide) {
        
        sort(nums.begin(), nums.end());
        for(int i = 0; i < numsDivide.size() - 1; i++)
            numsDivide[i+1] = __gcd(numsDivide[i], numsDivide[i+1]);
        for(int i = 0; i < nums.size(); i++) if(numsDivide[numsDivide.size() - 1] % nums[i] == 0) return i;
        return -1;
    }
};
