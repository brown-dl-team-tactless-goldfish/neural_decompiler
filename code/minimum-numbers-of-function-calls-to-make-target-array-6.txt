class Solution {
public:
    int minOperations(vector<int>& nums) 
    {
        int multiply = 0, plus_one = 0;
        for(int i = 0; i < nums.size(); i++)
        {
            int mul = 0, plus = 0;
            int n = nums[i];
            plus_one += n > 0;
            while(n > 1) {
                mul++;
                plus_one += n % 2;
                n /= 2;
            }
            multiply = max(multiply, mul);
        }

        return multiply + plus_one;
    }
};
