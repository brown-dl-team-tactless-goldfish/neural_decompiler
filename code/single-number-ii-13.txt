int singleNumber(int* nums, int numsSize){
    int i,j,sum,result=0;
    
    for(i=0;i<32;i++)
    {
        sum = 0;
        for(j=0;j<numsSize;j++)
        {
            sum += (nums[j]>>i)&1;
        }
        sum %= 3;
        if(sum)
        {
            result |= ((unsigned int)1<<i);
        }
    }
    
    return result;
}
