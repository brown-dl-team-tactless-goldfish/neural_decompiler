public int LargestCombination(int[] A) {
        int ans = 1;
        for(int i = 0 ; i < 24 ; i++) {
            int count = 0;
            foreach(int j in A) {
                if(((1 << i) & j) > 0) 
                    count++;
            }
            ans = Math.Max(ans,count);
        }
        return ans;
    }
