int firstMissingPositive(int* nums, int n){
    int i=0,t;
    while(i < n){
        long c=(long)nums[i]-1;
        if(c > -1 && c < n && nums[c]!=nums[i]){
            t=nums[c];
            nums[c]=nums[i];
            nums[i]=t;
        }
        else 
            i++;
    }
    for(i=0;i<n && nums[i]==i+1;i++);
    return i+1;
}
