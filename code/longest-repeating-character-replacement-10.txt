int characterReplacement(char* s, int k) {
                int max = 1;
		int n = strlen(s);
		for (int i = 0; i < n; i++) {
			if (i!=0&&s[i - 1] == s[i])continue;
			int j;
			int count = k;
			for (j = i + 1; j < n; j++) {
				if (s[i] != s[j])
				    count--;
				if (count < 0)
				    break;
			}
			int t;
			if (j == n&&count >= 0)
			    t = j - i + (count > i ? i : count);
			else 
			    t = j - i;
			max = max > t ? max : t;
		}
		return max;
}
