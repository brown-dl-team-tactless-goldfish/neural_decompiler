class Solution {
public:
  long long gridGame(vector<vector<int>>& grid) {
    long az = 0, za = 0, ans = LONG_MAX;
    
    for(int i = 1; i != grid[0].size(); i++) za += grid[0][i];
    for(int i = 1; i != grid[0].size(); i++){
      ans = min<long>(ans, max<long>(az, za));
      az += grid[1][i-1], za -= grid[0][i];
    }
    
    return min<long>(ans, az);
  }
};
