

bool checkString(char * s){
    
    int n= strlen(s);
    int flag=0;
    
    for(int i=0;i<n-1;i++){
        
        if(s[i]-s[i+1]==-1){
            flag++;
        }
        if(s[i]-s[i+1]==1){
            return false;
        }
        if(flag>1){
            return false;
        }
    }
    return true;
}
