	class Solution {
	public:
		long long numberOfWays(string s) {
			long long res = 0;
			long long a = 0, b = 0, c = 0;
			//search for 010
			for(auto i: s){
				if(i=='0')a += 1;
				if(i=='1')b += a;
				if(i=='0')c += b;
			}
			res += c;

			a = 0;
			b = 0;
			c = 0;
			//search for 101
			for(auto i: s){
				if(i=='1')a += 1;
				if(i=='0')b += a;
				if(i=='1')c += b;
			}
			return res+c;


		}
	};
