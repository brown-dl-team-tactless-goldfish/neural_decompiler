public class Solution {
    public bool EvaluateTree(TreeNode root) {
        if (root.left == null && root.right == null)
        {
            return root.val == 1;
        }
        
        bool l = EvaluateTree(root.left),
             r = EvaluateTree(root.right);
        
        return root.val == 2 ? l || r : l && r;
    }
}
