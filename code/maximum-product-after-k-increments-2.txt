class Solution {
public:
    int maximumProduct(vector<int>& nums, int k) {
        multiset<int> ms(nums.begin(),nums.end());
        while(k--){
            auto f = ms.begin();
            int val = *f;
            ms.erase(f);
            ms.insert(val+1);
        }
        long long mul = 1, mod = 1e9 + 7;
        for(auto it : ms) mul *= it, mul %= mod;
        return mul;
    }
};
