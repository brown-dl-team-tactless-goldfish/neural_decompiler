class Solution {
public:
    int numOfSubarrays(vector<int>& arr) {
        int dp = arr[0] % 2 == 0 ? 0 : 1, res = dp;
        for (int i = 1; i < arr.size(); ++i)
        {
            bool odd = arr[i] % 2 == 1;
            dp = (odd ? 1 : 0) + (!odd ? dp : i - dp);
            res = (res + dp) % 1000000007;
        }
        
        return res;
    }
};
