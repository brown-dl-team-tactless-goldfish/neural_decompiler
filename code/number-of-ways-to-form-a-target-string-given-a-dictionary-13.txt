class Solution:
    def numWays(self, words: List[str], target: str) -> int:
        res = [1] + [0] * len(target)
        for i in range(len(words[0])):
            cnt = collections.Counter(w[i] for w in words)
            for j in range(min(i + 1, len(target)), 0, -1):
                res[j] = (res[j] + res[j - 1] * cnt[target[j - 1]]) % 1000000007
        return res[len(target)]
