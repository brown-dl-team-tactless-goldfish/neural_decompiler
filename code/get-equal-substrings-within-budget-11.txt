class Solution {
public:
    int equalSubstring(string s, string t, int maxCost) {
        int i=0;
        int j=0;
        int cost=0;
        int result=0;
        
        // Variable size window problem
        while(j<s.size())
        {
            // Do pre-calculation
            cost+=abs(s[j]-t[j]);
            
            // Play with condition
            while(cost>maxCost)
            {
                // remove calculation for i
                cost-=abs(abs(s[i]-t[i]));
                i++;
            }
            // Store result
            result=max(result,j-i+1);
            j++;
        }
        return result;
    }
};

Time Complexity: O(n)
Space Complexity: O(1)
