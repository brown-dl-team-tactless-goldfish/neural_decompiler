int findSpecialInteger(int* arr, int arrSize){

    int count =1;
    int c = arr[0];

    for(int i=1; i<arrSize; i++){

        if(c == arr[i]){
            count ++;
        }
        else{
            if(count > arrSize/4){
                return arr[i-1];
            }
            count = 1;
            c= arr[i];
        }

    }
    return arr[arrSize-1];

}
