class Solution {
public:
    long long waysToBuyPensPencils(int total, int cost1, int cost2) {
        long long ans = 0;
        int x = 0 , y =0;
        while(1){
            if(total - (y*cost2) < 0){
                break;
            }
            double x = (total - (y*cost2))/double(cost1);
            ans+= (int(x)+1);
            y++;
        }
        return ans;
    }
};
