class Solution {
public:
    vector<int> mostCompetitive(vector<int>& nums, int k) {
        int n=nums.size();
        vector<int> ans(k);
        for(int i=0,j=-1;i<n;++i){
            while(j>=0 && nums[i]<ans[j] && i+k-j-1<n)
                j--;
            if(j+1<k)
                ans[++j]=nums[i];
        }
        return ans;
    }
};
