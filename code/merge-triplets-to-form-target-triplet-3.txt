class Solution {
public:
    bool mergeTriplets(vector<vector<int>>& triplets, vector<int>& t) {
        bool a=false,b=false,c=false;
        for (vector<int>tri : triplets)
            if (tri[0] <= t[0] && tri[1] <= t[1] && tri[2] <= t[2]){
                if(tri[0]==t[0]) a=true;
                if(tri[1]==t[1]) b=true;
                if(tri[2]==t[2]) c=true;
            }
        return a and b and c;
    }
};
