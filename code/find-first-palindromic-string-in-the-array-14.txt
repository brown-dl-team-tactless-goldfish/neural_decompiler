public class Solution {
    public string FirstPalindrome(string[] words) {
        
        if(words == null || words.Length == 0)
            return "";
        
        foreach(var word in words)
        {
            if(isPalindrome(word))
                return word;
        }
        
        return "";
    }
        
    private bool isPalindrome(string word)
    {
        int i = 0, j = word.Length - 1;
        while(i < j)
        {
            if(word[i] != word[j])
                return false;
            
            i++;
            j--;
        }
        
        return true;
    }
}
