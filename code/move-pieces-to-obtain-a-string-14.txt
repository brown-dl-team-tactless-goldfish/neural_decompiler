class Solution {
public:
	bool canChange(string s, string t) {
		queue<int> q;
		int n = s.size();

		for(int i = 0; i < n; i++) 
			if(t[i] != '_') q.push(i);

		for(int i = 0; i < n; i++) {
			if(s[i] == '_') 
				continue;

			if(s[i] == 'L') {
				if(q.empty() || t[q.front()] == 'R' || q.front() > i)
					return false;
				q.pop();
			}

			if(s[i] == 'R'){
				if(q.empty() || t[q.front()] == 'L' || q.front() < i)
					return false;
				q.pop();
			}
		}
		return q.empty();
	}
};
