bool isValidSudoku(char** board, int boardSize, int* boardColSize){
    // check each row
    for (int i = 0; i < boardSize; i++){
        int map[10] = {0};
        for (int j = 0; j < boardSize; j++){
            if (board[i][j] == '.') continue;
            else if (map[board[i][j]-'0'] > 0) return false;
            map[board[i][j]-'0']++;
        }
    }
    // check each column
    for (int i = 0; i < boardSize; i++){
        int map[10] = {0};
        for (int j = 0; j < boardSize; j++){
            if (board[j][i] == '.') continue;
            else if (map[board[j][i]-'0'] > 0) return false;
            map[board[j][i]-'0']++;
        }
    }
    // check sub-boxes
    int map[10] = {0};
    for (int cnt = 0; cnt < 3; cnt++){
        for (int i = 0; i < boardSize; i++){
            if (i % 3 == 0) memset(map, 0, sizeof(map));
            for (int j = 3*cnt; j < 3*(cnt+1); j++){
                if (board[i][j] == '.') continue;
                else if (map[board[i][j]-'0'] > 0) return false;
                map[board[i][j]-'0']++;
            }
        }
    }

    return true;
}
