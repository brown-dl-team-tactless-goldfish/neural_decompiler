   public:
     int videoStitching(vector<vector<int>>& clips, int Time) {
        sort(begin(clips), end(clips));
       
        int i = 0, a = 0, b = 0, count = 0;
        while(a < Time) {
            for (; i < clips.size() && clips[i][0] <= a; ++i){
                b = max(b, clips[i][1]);
            }
            
            if (a == b) return -1;
            a = b;
            count ++;
        }
        
        return count;
    }
