class FindElements {
public:
    TreeNode* tempRoot;	
    
    void correct(TreeNode* root){
        if(!root)
            return ;
        
        if(root->left)
            root->left->val=2*root->val+1;
        
         if(root->right)
            root->right->val=2*root->val+2;
        
        correct(root->left);
        correct(root->right);   
    }
    FindElements(TreeNode* root):tempRoot(root) {
        root->val=0;
        correct(root);
    }
    bool check(TreeNode* root,int target){
         if(!root)
          return false;
      
     if(root->val==target)
         return true;
        
     bool l=check(root->left,target);
     bool r=check(root->right,target);
     return (l||r);
      
    }
    bool find(int target) {
     return check(tempRoot,target);
    }
};
