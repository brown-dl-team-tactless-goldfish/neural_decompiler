bool isCovered(int** ranges, int rangesSize, int* rangesColSize, int left, int right){
    for(int i = left; i <= right; i++){
        for(int j = 0; j < rangesSize; j++){
            if(i >= ranges[j][0] && i <= ranges[j][1] )
                break;
            if(j == rangesSize-1)
                return false;
        }        
    }
    return true;
}
