int
compareVersion(const char *v1, const char* v2)
{
	int a, b;

	do {
		a = 0;
		for (v1 += *v1 == '.'; *v1 != '\0' && *v1 != '.'; ++v1)
			a = a * 10 + *v1 - '0';
		b = 0;
		for (v2 += *v2 == '.'; *v2 != '\0' && *v2 != '.'; ++v2)
			b = b * 10 + *v2 - '0';
	} while (a == b && (*v1 != '\0' || *v2 != '\0'));

	return a < b ? -1 : a != b;
}
