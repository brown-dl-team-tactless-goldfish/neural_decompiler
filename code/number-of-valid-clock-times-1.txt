    if (t[0] == '?' && t[1] == '?' && t[3] == '?' && t[4] == '?') return 1440;   
    if (t[0] == '?' && t[1] == '?' && t[3] == '?') return 144;
    if (t[0] == '?' && t[1] == '?' && t[4] == '?') return 240;
    if (t[0] == '?' && t[3] == '?' && t[4] == '?') return (t[1] - '0' > 3) ? 120 : 180;
    if (t[1] == '?' && t[3] == '?' && t[4] == '?') return (t[0] == '2') ? 240 : 600;  
    if (t[0] == '?' && t[1] == '?') return 24;
    if (t[0] == '?' && t[3] == '?') return (t[1] - '0' > 3) ? 12 : 18;
    if (t[0] == '?' && t[4] == '?') return (t[1] - '0' > 3) ? 20 : 30;
    if (t[1] == '?' && t[3] == '?') return (t[0] == '2') ? 24 : 60;
    if (t[1] == '?' && t[4] == '?') return (t[0] == '2') ? 40 : 100;
    if (t[3] == '?' && t[4] == '?') return 60;

    if (t[1] == '?') return (t[0] == '2') ? 4 : 10;
    if (t[3] == '?') return 6;
    if (t[4] == '?') return 10;

}
