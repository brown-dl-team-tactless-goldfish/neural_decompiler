int numberOfBeams(vector<string>& bank) {
        int count =0,m=0,n=0;
        int r = bank.size(), c = bank[0].size();
        for(int i=0;i<r;i++){
            m=0;
            for(int j=0;j<c;j++){
                m = bank[i][j]=='1' ? m+1 : m;
            }
            
            count = m && n ? count+m*n : count;
            n = m ? m : n;
            //cout<<m<<" "<<n<<" "<<count<<endl;
        }
        return count;
    }
