class Solution:
    def construct2DArray(self, original: List[int], m: int, n: int) -> List[List[int]]:
        data = [[0 for i in range(n)] for j in range(m)]
        # print ("m = ", m, " n = ", n)
        
        if (m*n != len(original)):
            return []
        for i in range(m):
            for j in range(n):
            #     print("i = ", i, " j = ", j)
                
                data[i][j] = original[i*n + j]
                
        return data
