class Solution {
public:
    vector<int> rearrangeArray(vector<int>& nums) {
        sort(nums.begin(), nums.end());
        vector<int> ans(nums.size());
        int s = 0, e = nums.size()-1, idx = 0, n = nums.size();
        while(idx < n)
        {
            ans[idx] = nums[s];
            idx++;
            if(idx >= n)
                break;
            ans[idx] = nums[e];
            s++;
            e--;
            idx++;
        }
        return ans;
    }
};
