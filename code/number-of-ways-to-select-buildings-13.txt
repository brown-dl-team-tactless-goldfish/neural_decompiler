class Solution {
public:
    long long numberOfWays(string s) {
        int n = s.size();
        vector<vector<long long>> dp(3, vector<long long> (2, 0));
        
        for(int i=0;i<n;i++) {
            int c = s[i] - '0';
            for(int len=2;len>0;len--) {
                dp[len][c] = dp[len][c] + dp[len-1][c^1];
            }
            dp[0][c]++;
        }
        
        return dp[2][0] + dp[2][1];
    }
};
