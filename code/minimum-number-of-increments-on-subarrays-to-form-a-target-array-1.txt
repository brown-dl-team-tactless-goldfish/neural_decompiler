#define max(x,y) (((x) > (y))?(x):(y))
int minNumberOperations(int* target, int targetSize){
    int count = target[0];                      //minimum operations shall always be 
                                                //the value of first element in the array
    for (int i = 0; i < targetSize-1; i++) {
        if(target[i] < target[i+1]) {            //if next elements is greater,
            count += target[i+1] - target[i];   //then increment count to its difference 
                                                //as we shall perform so many operations.
        }
    }
    return count;
}
