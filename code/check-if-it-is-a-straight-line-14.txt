class Solution:
	def checkStraightLine(self, coordinates):
		k, k_, c = 1, 0, coordinates[0][0]
		
		if coordinates[0][0] != coordinates[1][0]:
			x1, y1 = coordinates[0]
			x2, y2 = coordinates[1]

			k, k_ = (y2-y1)/(x1-x2), 1
			c = (y1*x2-y2*x1)/(x2-x1)

		for x, y in coordinates[2:]:
			if k * x + k_ * y - c:
				return False

		return True
