class Solution {
public:
    vector<int> maxDepthAfterSplit(string seq) {
        vector<int>s;
        int d = 0;
        for(auto c : seq){
            bool flag = c == '(';
            if(flag)d++; // open bracket
            s.push_back(d%2);
            if(!flag)d--; // closed bracket
        }
        return s;
    }
};
