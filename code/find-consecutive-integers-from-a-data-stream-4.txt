class DataStream {
public:
    int v, n, cnt;
    DataStream(int value, int k) {
        v = value;
        n = k;
        cnt = 0;
    }
    
    bool consec(int num) {
        return (cnt = (num == v)* (cnt+(num == v))) >= n;
    }
};
