int peakIndexInMountainArray(int* arr, int arrSize){
    int start = 0;
    int end = arrSize - 1 ;
    int midPoint ;
    do{
        midPoint = (start + end) / 2 ;
        if(midPoint == 0)
            return 1;
        else if(arr[midPoint] > arr[midPoint-1] && arr[midPoint]>arr[midPoint+1]){
            //here I ask if arr[midPoint] is the leargest between two values
            return midPoint ;
        }
        else if(arr[midPoint] < arr[midPoint+1]){
            /*if this condition is true so 
            we are in the left side of the wanted number*/
            start = midPoint + 1 ;
        }
        else{
            /*if this condition is true so 
            we are in the right side of the wanted number*/
            end = midPoint-1 ;
        }
    }while(start <= end);
    return 0;
}
