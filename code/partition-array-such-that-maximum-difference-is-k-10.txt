    public int PartitionArray(int[] nums, int k) 
    {
        int counter = 0;
        Array.Sort(nums);
        for (int i = 0; i < nums.Length; i++)
        {
            int j = i;
            while (j < nums.Length && nums[j] - nums[i] <= k)
            {
                j++;
            }
            counter++;
            i = j - 1;
        }
        return counter;
    }
