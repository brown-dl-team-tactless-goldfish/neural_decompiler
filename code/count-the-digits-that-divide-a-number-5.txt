int size(int num){
    int x = 0;
    while (num != 0){
        num /= 10;
        x++;
    }
    return x;
}

int countDigits(int num){
    int sizeNum = size(num);
    int* res = malloc(sizeof(int)*sizeNum);
    int temp = num;
    int count = 0;
    for (int i = 0; i<sizeNum; i++){
        res[i]=temp%10;
        temp = temp/10;
        if (num%res[i] == 0){
            count++;
        }
    }
    return count;

}
