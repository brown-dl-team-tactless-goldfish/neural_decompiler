class Solution {
public:
    int maxNumOfMarkedIndices(vector<int>& nums) {
        sort(nums.begin(),nums.end());
        int n = nums.size();
        int i = 0, j = (nums.size()+1)/2;
        while(j < n){
            if(nums[i] * 2 <= nums[j])
                i++;
            j++;
        }
        return i*2;
    }
};
