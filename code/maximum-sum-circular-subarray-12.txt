class Solution {
public:
    int maxSubarraySumCircular(vector<int>& nums) {
        int mxs=nums[0],mns=nums[0],pmx=nums[0],pmn=nums[0],s=nums[0];
        for(int i=1;i<size(nums);++i){
            s+=nums[i];
            pmx=max(pmx+nums[i],nums[i]);
            pmn=min(pmn+nums[i],nums[i]);
            mxs=max(mxs,pmx);
            mns=min(mns,pmn);
        }
        return s==mns?mxs:max(mxs,s-mns);
    }
};
