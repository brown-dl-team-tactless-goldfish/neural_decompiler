

int maxDepth(char * s){

    int maxdepth = 0;
    int depth = 0;
    while(*s!='\0')
    {
        if(*s == '(')
            depth++;
        else if(*s == ')')
            depth--;
        
        if(maxdepth < depth)
        {
            maxdepth = depth;
        }
        s++;
    }
    
    return maxdepth;
}
