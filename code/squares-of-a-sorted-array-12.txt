int cmpfunc (const void * a, const void * b)
{
   return ( *(int*)a - *(int*)b );
}
int* sortedSquares(int* A, int ASize, int* returnSize) {
    bool qsrt=false;
    if(A[0]<0) qsrt=true;
    for(int i=0;i<ASize;i++)
        A[i]*=A[i];    
    if(qsrt)
    qsort(A, ASize, sizeof(int), cmpfunc);
    *returnSize=ASize;
    return A;
}
