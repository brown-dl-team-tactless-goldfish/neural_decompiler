class Solution:
    def arrayRankTransform(self, arr: List[int]) -> List[int]:
        print("arr = ", arr)

        data = list(set(arr))
        data.sort()
        print("data = ", data)
        idx = 1
        data_map = {}
        for v in data:
            data_map[v] = idx
            idx += 1

        res = []
        for val in arr:
            res.append(data_map[val])

        print("end res = ", res)
        return res
