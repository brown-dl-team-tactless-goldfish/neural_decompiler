    public int[] AnswerQueries(int[] nums, int[] queries)
    {
        Array.Sort(nums);
        int n = nums.Length;
        int m = queries.Length;

        int[] result = new int[m];
        int sum = 0;
        for (int j = 0; j < n; j++)
        {
            sum += nums[j];
            for (int k = 0; k < m; k++)
            {
                if (sum <= queries[k])
                    result[k] = j + 1;
            }
        }
        return result;
    }
