/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     struct ListNode *next;
 * };
 */
 
struct ListNode* removeZeroSumSublists(struct ListNode* head){
    int sum = 0;
    struct ListNode *ptr = NULL;

    if (head == NULL) {
        return NULL;
    }
    head->next = removeZeroSumSublists(head->next);
    ptr = head;
    sum = ptr->val;
    ptr = ptr->next;
    while(ptr != NULL && sum != 0) {
        sum += ptr->val;
        ptr = ptr->next;
    }
    if (sum == 0) {
        return ptr;
    }
    return head;
}
