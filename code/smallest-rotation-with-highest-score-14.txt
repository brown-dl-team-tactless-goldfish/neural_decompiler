class Solution {
public:
    int bestRotation(vector<int>& nums) {
        int n = size(nums); 

        vector<int> diff(n, 1); 
        for (int i = 0; i < n; ++i) 
            diff[(i+n-nums[i]+1) % n]--; 
        
        int ans = 0, prefix = 0, mx = INT_MIN; 
        for (int i = 0; i < n; ++i) {
            prefix += diff[i]; 
            if (prefix > mx) {
                mx = prefix; 
                ans = i; 
            }
        }
        return ans; 
    }
};
