class Solution {
public:
    long long putMarbles(vector<int>& weights, int k) {
        vector<long long>v;
        int n=weights.size();
        
        for(int i=0;i<n-1;i++)
        {
            v.push_back(weights[i]+weights[i+1]);
        }
        cout<<1;
        sort(v.begin(),v.end());
        long long max=0;
        long long min=0;
        for(int i=0;i<k-1;i++)
        {
            min+=v[i];
            max+=v[n-2-i];
        }
        return abs(max-min);
    }
};
