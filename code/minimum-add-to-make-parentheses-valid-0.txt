int minAddToMakeValid(char * s){
    int n = strlen(s);
    int size = 0;
    int A = 0;
    for (int i = 0; i < n; i++) {
        if (s[i] == '(') {
            size++;
            A++;
        } 
        else {
            if (size > 0) {
                size--;
            }
        }
    }
    return n - (A-size)*2;
}
