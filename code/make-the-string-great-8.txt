char * makeGood(char * s){
    const char * read;
    char * write;
    
    do {
        read = write = s;
        
        for(; *read != '\0'; ++read) {
            if(*read != read[1] && tolower(*read) == tolower(read[1])) {
                ++read;
            }
            else {
                *write = *read;
                ++write;
            }
        }
        *write = '\0';
    } while(write != read);
        
    return s;
}
