bool canBeEqual(int* target, int targetSize, int* arr, int arrSize)
{
        int a[1001]={0},i;
        if(targetSize!=arrSize)
                return 0;
        for(i=0;i<targetSize;i++)
        {
                a[target[i]]++;
                a[arr[i]]--;
        }
        for(i=0;i<1001;i++)
                if(a[i]!=0)
                        return 0;
        return 1;

}
