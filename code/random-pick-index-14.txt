    Solution(vector<int> nums) {
        for (int i = 0; i < nums.size(); ++i)
        {
            map[nums[i]].push_back(i);
        }
    }
    
    int pick(int target) {
        std::vector<int> vec = map[target];
        int idx = rand() % vec.size();
        return vec[idx];
    }
    
private:
    std::unordered_map<int, std::vector<int>> map;
