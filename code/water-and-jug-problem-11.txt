public class Solution 
{
    public bool CanMeasureWater(int x, int y, int z) 
    {
        if (z == 0) return true;
        if (z > x + y) return false;

        var gcd = GCD(x, y);

        if (z % gcd == 0) return true;

        return false;
    }

    private int GCD(int x, int y)
    {
        if (y == 0) return x;

        return GCD(y, x % y);
    }
}
