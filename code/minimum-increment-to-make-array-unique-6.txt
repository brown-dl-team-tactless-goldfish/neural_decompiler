
int cmp (void * a , void* b)
{
    return *(int*)a - *(int*)b;
}

int minIncrementForUnique(int* nums, int numsSize){
    
    qsort(nums,numsSize,sizeof(int),cmp);
    int no_incs = 0;
  
    for(size_t i = 1 ; i < numsSize ; i++)
    {
        if (nums[i] <= nums[i-1])
        {
            no_incs += (nums[i-1] - nums[i] +1);
            nums[i] += (nums[i-1] - nums[i] +1); 
        }
    }
        
    return no_incs;
}
