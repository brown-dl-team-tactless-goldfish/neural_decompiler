class Solution {
public:
    vector<int> rowAndMaximumOnes(vector<vector<int>>& mat) {
        int maxi=0,temp=0;
       
        for(int i=0;i<mat.size();i++){
             int cnt=0;;
            for(int j=0;j<mat[i].size();j++){
                if(mat[i][j]==1)cnt++;
            }
            if(cnt>maxi){
                maxi = cnt;
                temp=i;
            }
        }
        return {temp,maxi};
    }
};
