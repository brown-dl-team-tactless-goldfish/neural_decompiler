int cmp(const int* a, const int* b) {
    return *(const int*) a - *(const int*) b;
}

long long minimumRemoval(int* beans, int beansSize) {
    qsort(beans, beansSize, sizeof(int), cmp);
    long long sum = 0;
    for(int i = 0; i < beansSize; i++) {
        sum += beans[i];
    }
    long long ans = sum, cnt, tmp;
    for(int i = 0; i < beansSize; i++) {
        cnt = beansSize - i;
        tmp = sum - cnt * beans[i];
        ans = tmp < ans ? tmp : ans;
    }
    return ans;
}
