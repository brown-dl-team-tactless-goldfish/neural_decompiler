public class Solution {
    public int ChalkReplacer(int[] chalk, int k) {
        
        long count = 0;
        foreach(var c in chalk)
        {
            count += c;
        }
        count = k%count;
        for(int i = 0; i < chalk.Length; i++)
        {
            if(chalk[i] <= count)
                count -= chalk[i];
            else
                return i;
        }
        return -1;
    }
}
