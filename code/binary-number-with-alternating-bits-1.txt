bool hasAlternatingBits(int n)
{
    int index=0;
for(int i=30;i>=0;i--) //recode where the n != 0
{
  
    if((n&(1<<i))!=0)
    {
        index=i;
        break;
    }
     
}
    int count=0;
    for(int i=index;i>=0;i--) //from index to 0 check the 0 or 1 didn't repeat
    {
        if((n&(1<<i))!=0)
            count++;
        else
            count--;
        if(count>=2||count<0)
            return false;
    }
       return true;
}
