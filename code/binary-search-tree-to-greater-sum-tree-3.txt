struct TreeNode *bt(struct TreeNode*root, int *sum)
{
    int right,left;
    if (root == NULL)
        return 0;
    
    bt(root->right,sum);
    root->val = root->val + *sum;
    *sum = root->val;
    bt(root->left,sum);
    return;
}

struct TreeNode* bstToGst(struct TreeNode* root){

    int currsum=0;
    bt(root,&currsum);
    return root;
}
