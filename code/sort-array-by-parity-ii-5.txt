int* sortArrayByParityII(int* nums, int numsSize, int* returnSize){
    *returnSize = numsSize;
    for (int i = 0; i < numsSize; i++){
        if ((i + nums[i]) % 2 == 1){
            for (int j = i + 1; j < numsSize; j++){
                if ((i + nums[j]) % 2 == 0){
                    int tmp = nums[j];
                    nums[j] = nums[i];
                    nums[i] = tmp;
                    break;
                }
            }
        }
    }
    return nums;
}
