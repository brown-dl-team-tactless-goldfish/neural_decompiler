class Solution {
public:
    int maxCount(vector<int>& banned, int n, int maxSum) {
        unordered_set<int> st;
        for(int i = 0; i < banned.size(); i++) st.insert(banned[i]);
        int s = 0;
        int i = 1;
        int count = 0;
        while(i <= n && s + i <= maxSum){
            if(!st.count(i)){
                s += i;
                count++;
            }
            i++;
        }
        return count;
    }
};
