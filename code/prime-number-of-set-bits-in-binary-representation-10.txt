int countBits(int n) {
    int count = 0;
    while(n) {
        count += n&1;
        n >>= 1;
        
    }
    return count;
}


int countPrimeSetBits(int left, int right){
    int count = 0, bits;
    bool prime;
    for(int i = left; i <= right; i++ ) {
        
        bits = countBits(i);
        
        prime = 0x54451456 & (1<<(bits-1));
        
        if(prime == true) count++;
    }
    
    return count;

}
