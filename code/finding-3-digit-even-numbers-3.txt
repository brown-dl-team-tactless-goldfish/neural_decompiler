vector<int> findEvenNumbers(const vector<int>& digits) {
	vector<int> result;
	array<int, 10> freq{};
	for(auto d: digits)
		++freq[d];
	find_even_numbers(freq, result);
	return result;
}

void find_even_numbers(array<int, 10>& freq, vector<int>& result, int curr=0, int i=0) {
	if(i == 3 && curr%2==0) 
		result.push_back(curr);
	else if(i < 3) {
		for(int d = 0; d < 10; ++freq[d++]) {
			if(--freq[d] >= 0 && (curr || d))
				find_even_numbers(freq, result, curr * 10 + d, i+1);
		}
	}
}
