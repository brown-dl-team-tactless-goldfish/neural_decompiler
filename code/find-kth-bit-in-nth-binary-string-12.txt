class Solution {
public:
    char invert(char a){
        if(a=='1')
            return '0';
        return '1';
    }
    
    char findKthBit(int n, int k) {
        if(k==1 or n==1)    return '0';
        if(k==pow(2,n-1))  return '1';
        if(k<pow(2,n-1))
            return findKthBit(n-1,k);
        return invert(findKthBit(n-1,pow(2,n)-k));
    }
};
