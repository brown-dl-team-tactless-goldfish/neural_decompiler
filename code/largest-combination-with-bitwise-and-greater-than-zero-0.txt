int largestCombination(int* candidates, int candidatesSize){
    int cnt = 0, maxCnt  = 0;
    for(int i = 0; i < 32; i++)
    {
        cnt = 0;
        for(int j = 0; j < candidatesSize; j++)
        {
            if(candidates[j] & 1)
                cnt++;
            candidates[j] >>= 1;
        }
        if(cnt > maxCnt) maxCnt = cnt;
    }
    return maxCnt;
}
