class Solution {
public:
    int largestCombination(vector<int>& cand) {
        
        // total cand
        int totalCandidate = cand.size();
        
        // max number of bit 1 count in same bit position
        int maxCount = 0;
        
        // loop from bit 0 to 24 as max cand limit 10^7
        // also we terminate checking if we already get totalCandidate number of bit 1
        for(int i=0;i<25 && maxCount<totalCandidate;i++)
        {
            // total bit 1 count in current bit position
            int count = 0;
            for(int & val : cand)
                count += ((val>>i)&1);
            
            // update max count
            maxCount = max(count,maxCount);
        }
        
        return maxCount;
    }
};
