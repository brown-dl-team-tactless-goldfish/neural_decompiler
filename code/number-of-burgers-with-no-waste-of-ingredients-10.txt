class Solution {
public:
    vector<int> numOfBurgers(int tomatoSlices, int cheeseSlices) {
        if(tomatoSlices==0 && cheeseSlices==0) return vector<int> {0,0};
        int y = tomatoSlices - (4*cheeseSlices);
        if(y<0 && y%2==0) {
            int total_small = abs(y/2);
            int total_jumbo = cheeseSlices - total_small;
            if(total_small<0 || total_jumbo<0) goto ret;
            vector<int> v{total_jumbo, total_small};
            return v;
        }
        
        ret: return vector<int>();
    }
};
