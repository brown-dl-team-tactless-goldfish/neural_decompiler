class Solution {
public:
    int longestIdealString(string s, int k) {
        int n = s.size();
        int dp[26] = {0};
        for(int i=0;i<n;i++)
        {
            int x = s[i]-'a';
            int y = dp[x];
            for(int c = 0;c<26;c++)
            {
                if(abs(c-x)<=k)
                {
                    y = max(y,dp[c]+1);
                }
            }
            dp[x] = y;
        }
        return *max_element(begin(dp),end(dp));
    }
};
