class Solution {
public:
    int addRungs(vector<int>& rungs, int dist) {
        int count{0}, i{0}, start{0};
        for(; i < rungs.size(); ++i) {
            count += getCount(dist, start, rungs[i]);
            start = rungs[i];
        }
        
        return count;
    }
    
    int getCount(int dist, int i, int j) {
        if(j-i <= dist) return 0;
        return (j-i-1)/dist;
    }
};
