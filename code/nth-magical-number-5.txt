int nthMagicalNumber(int N, int A, int B) {
    vector<int> nums;
    for (auto i = 1, j = 1; nums.empty() || nums.back() < A * B;) {
        if (A * i == B * j) nums.push_back(A * i++), ++j;
        else if (A * i < B * j) nums.push_back(A * i++);
        else nums.push_back(B * j++);          
    }      
    return ((long long)(N / nums.size() * A) * B + nums[N % nums.size() - 1]) % 1000000007;
}
