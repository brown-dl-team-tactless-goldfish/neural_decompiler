    public int MinimizeSum(int[] nums)
    {
        int n = nums.Length;
        Array.Sort(nums);

        //Case1: remove first 2
        int score1 = (nums[n - 1] - nums[2]);

        //Case2: remove last 2
        int score2 = (nums[n - 3] - nums[0]);

        //Case3: remove first and last
        int score3 = (nums[n - 2] - nums[1]);

        return Math.Min(Math.Min(score1, score2), score3);
    }
