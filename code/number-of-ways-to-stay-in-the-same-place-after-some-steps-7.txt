class Solution {
public:
  int dp[501][1100] = {};
  int mod = (int)1e9 + 7;
  int numWays(int steps, int arrLen, int c = 0) {
    if (steps == 0 && c == 0) return 1;
    if (c < 0 || steps < c || c >= arrLen) return 0;
    if (dp[steps][c]) return dp[steps][c];
    return dp[steps][c] = ((numWays(steps - 1, arrLen, c + 1) + numWays(steps - 1, arrLen, c - 1)) % mod + numWays(steps - 1, arrLen, c)) % mod;
  }
};
