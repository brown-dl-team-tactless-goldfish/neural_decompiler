struct TreeNode* convertBSTUtil(struct TreeNode* root, int *sum) {
    if(root == NULL) return NULL;
    
    root->right = convertBSTUtil(root->right, sum);
    *sum+= root->val;
    root->val = *sum;
    root->left = convertBSTUtil(root->left, sum);
    
    return root;
}

struct TreeNode* convertBST(struct TreeNode* root){
    int sum = 0;
    return convertBSTUtil(root, &sum);
}
