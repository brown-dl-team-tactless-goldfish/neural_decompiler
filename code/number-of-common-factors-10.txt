int HCF(int a, int b){
    int hcf = 0;
    if(a<b){
        int temp = a;
        a = b;
        b = temp;
    }
    while(b!=0){
        int temp = b;
        b = a%b; 
        a = temp;
    }
    return a;
}
int commonFactors(int a, int b){
    int result = 0;
    int max = HCF(a, b);
    for(int i = 1; i <= max; i++){
        if(max%i == 0) result++;
    }
    return result;
}
