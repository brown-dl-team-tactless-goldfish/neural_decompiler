struct TreeNode* constructMaximumBinaryTree(int* nums, int numsSize){

	struct TreeNode *root = NULL;
	int maxIndex = indexOfMaxValue(nums, numsSize);

	if (maxIndex >= 0) {     
		int suffix = numsSize - (maxIndex+1);
		int prefix = (numsSize - (suffix+1));
    
		root = malloc(sizeof(struct TreeNode));
		root->val = nums[maxIndex];
		root->left = constructMaximumBinaryTree(nums, prefix);
		root->right = constructMaximumBinaryTree(&nums[maxIndex+1], suffix);
	}
	return root;
}
