int countCharacters(char ** words, int wordsSize, char * chars){
/*Table used to find the count of characters */
int* hash_table_chars = (int*)calloc(26,sizeof(int));
int* hash_table_chars_temp = (int*)calloc(26,sizeof(int));
for(int idx =0 ;idx < strlen(chars); idx++){
       hash_table_chars[chars[idx] - 'a']++;
}

int return_total_lenght = 0;
int current_word_length_calc = 0;    
//now to find the string is good or not in words array
for(int word_idx=0; word_idx< wordsSize; word_idx++){
    current_word_length_calc = 0;
    memcpy(hash_table_chars_temp,hash_table_chars,sizeof(int)*26);    
    for(int ptr =0 ;ptr < strlen(words[word_idx]); ptr++){
        if(hash_table_chars_temp[words[word_idx][ptr]  - 'a'] > 0){
            hash_table_chars_temp[words[word_idx][ptr]  - 'a']--;
            current_word_length_calc++;
        }
    }
    if(current_word_length_calc == strlen(words[word_idx])){
        return_total_lenght += current_word_length_calc;
    }
}

free(hash_table_chars_temp);
free(hash_table_chars);
return return_total_lenght;       
}
