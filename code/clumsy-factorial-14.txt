    int clumsy(int N) {
        int res = 0, sign = 1;
        while(N>=4){
            res += N*(N-1)/(N-2)*sign + N-3;
            sign = -1;
            N-=4;
        }
        if(N == 1)      res += sign*N;
        else if(N == 2) res += sign*N*(N-1);
        else if(N == 3) res += sign*N*(N-1)/(N-2);
        return res;
    }
