public class Solution {
    public int MinOperations(string s) {

        return Math.Min(Alternate('0', s), Alternate('1', s));
    }
    
    private int Alternate(char baseChar, string s)
    {
        int cnt = 0;
        
        for(int i = 0; i < s.Length; i++)
        {
            if(i % 2 == 0 && s[i] != baseChar)
                cnt++;
               
            else if(i % 2 != 0 && s[i] == baseChar)
                cnt++;
        } 
    
        return cnt;
    }
}
