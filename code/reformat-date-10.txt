unordered_map<string, string> m_d = {{"Jan", "01"}, {"Feb", "02"}, {"Mar", "03"}, {"Apr", "04"}, 
    {"May", "05"}, {"Jun", "06"}, {"Jul", "07"}, {"Aug", "08"}, 
    {"Sep", "09"}, {"Oct", "10"}, {"Nov", "11"}, {"Dec", "12"}};    
string reformatDate(string dt) {
    return dt.substr(dt.size() - 4) + "-" 
        + m_d[dt.substr(dt.size() - 8, 3)] + "-" 
        + (isdigit(dt[1]) ? dt.substr(0, 2) : string("0") + dt[0]);
}
