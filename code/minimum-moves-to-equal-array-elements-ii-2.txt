int comp(const void* a, const void* b){
    return *(int*)a > *(int*)b ? 1 : 0;
}

int minMoves2(int* nums, int numsSize){
    qsort(nums, numsSize, sizeof(int), comp);
    int mid = numsSize/2;
    int k = nums[mid];
    int ans = 0;
    for(int i = 0 ; i < mid ; i++){
        ans += k - nums[i];
    } 
    for(int i = mid+1 ; i < numsSize ; i++){
        ans += nums[i] - k;
    }
    return ans;
}
