class Solution {
public:
    long long maximumImportance(int n, vector<vector<int>>& roads) {
        long long ans=0;
        vector<long long> mp(n+1, 0);
        
        for(auto i:roads){
            mp[i[0]]++;
            mp[i[1]]++;
        }
        
        sort(mp.begin(), mp.end(), greater<>());
        
        for(auto i:mp){
            ans += i*(n--);
        }
        
        return ans;
    }
};
