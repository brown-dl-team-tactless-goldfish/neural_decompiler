// Runtime: 0 ms, faster than 100.00% of C++ online submissions for Percentage of Letter in String.
// Memory Usage: 6.2 MB, less than 12.95% of C++ online submissions for Percentage of Letter in String.
int percentageLetter(string s, char letter) {
	return (1.0*count_if(begin(s), end(s), [&letter](const char& c) { return c == letter; } ) / s.size()) * 100;
}
