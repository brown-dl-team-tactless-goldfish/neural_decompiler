class Solution {
    static bool cmp(vector<int>& a, vector<int>& b) {
        return (a[1]-a[0])>(b[1]-b[0]);
    }
public:
    int minimumEffort(vector<vector<int>>& tasks) {
        sort(tasks.begin(), tasks.end(), cmp);
        int sum = 0, level = 0;
        for (vector<int> task : tasks) {
            int diff = max(task[1]-level, 0);
            level += diff;
            level -= task[0];
            sum += diff;
        }
        return sum;
    }
};
