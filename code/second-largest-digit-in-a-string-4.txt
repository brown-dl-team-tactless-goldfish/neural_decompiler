#include <ctype.h>

int secondHighest(char * s){
    int largest = -1, second_largest = -1;
    for (size_t i = 0; s[i] != '\0'; i++)
    {
        if (isdigit(s[i]))
        {
            int digit = s[i] - '0';
            if (digit > largest)
            {
                second_largest = largest;
                largest = digit;
            }
            else if (digit > second_largest && digit != largest)
            {
                second_largest = digit;
            }
        }
    }
    return second_largest;
}
