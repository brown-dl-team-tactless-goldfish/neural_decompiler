int checkChar(char c)
{
  if((c >= 65 && c <= 90) || (c >= 97 && c <= 122)) return 1;
  return 0;
}

char * reverseOnlyLetters(char * s){
    char *begin = s, *end = s, temp;
        
    while(*end != '\0') end++;
    end--;
    
    while(begin < end)
    {
      if(!checkChar(*begin)) begin++;
      if(!checkChar(*end))   end--;
      if(checkChar(*begin) && checkChar(*end))
      {
        temp = *begin;
        *begin++ = *end;
        *end-- = temp;
      }
    }
    return s;
}
