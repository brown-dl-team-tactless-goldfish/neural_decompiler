int chalkReplacer(int* chalk, int chalkSize, int k){
    long sum = 0, temp = 0;
    for(int i = 0; i < chalkSize; i++)
        sum += chalk[i];
    sum = k % sum;
    for(int i = 0; i < chalkSize; i++)
    {
        temp += chalk[i];
        if(temp > sum)
            return i;
    }
    return 0;
}
