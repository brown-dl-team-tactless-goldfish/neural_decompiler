int cmp(const void* a, const void* b){
    return *(int*)b - *(int*)a ;
}
int fillCups(int* amount, int amountSize){
    qsort(amount, amountSize, sizeof(int), cmp) ;
    if(amount[0] >= (amount[1] + amount[2]))
        return amount[0] ;
    else{
        return amount[0] + (amount[1] + amount[2] - amount[0] + 1) / 2 ;
    } 
}
