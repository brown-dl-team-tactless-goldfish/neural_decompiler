    long long n = nums.size(), breakk,res = 0;
    
    if(n == 1)
    {
        return 0;
    }
    
    int right = nums[n-1];
    
    for(int i=n-2; i>= 0; i--)
    {
        if(right >= nums[i])
        {
            right = nums[i];
        }
        else
        {
            breakk = ((nums[i]-1)/ right)+1;
            
            right = nums[i]/breakk;
            
            res += (breakk-1);
        }
    }
    
    return res;
    
}
