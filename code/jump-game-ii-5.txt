#define MAX(a, b) (((a) > (b)) ? (a) : (b))

int jump(int* nums, int numsSize) {
    int currEnd = 0, farthest = 0, jumps = 0;
    for (int i = 0; i < numsSize - 1; i++) {
        farthest = MAX(farthest, i + nums[i]);
        if (i == currEnd) {
            jumps++;
            currEnd = farthest;
            if (farthest >= numsSize - 1) break;
        }
    }
    return jumps;
}
