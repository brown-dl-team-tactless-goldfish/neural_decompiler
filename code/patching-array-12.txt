class Solution {
public:
    int minPatches(vector<int>& nums, int n) {
        long curr=0;
        nums.push_back(INT_MAX);
        int ans=0;
        int l=0;
        while(l<nums.size()){
            if(curr>=n) break;
            if(nums[l]>curr+1){
                ans++;
                curr+=curr+1;
            }
            else{
                curr+=nums[l];
                l++;
            }
        }
        return ans;
    }
};
