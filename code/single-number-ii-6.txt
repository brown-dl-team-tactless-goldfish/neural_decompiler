
int singleNumber(int* nums, int numsSize){
    
    long long int unique = 0;
    
    for(int i = 0; i < 32; i++){
        
        int total = 0;
        
        for(int j = 0; j < numsSize; j++){
            
            total += (nums[j] & 1);
            nums[j] >>= 1;
            
        }
        
        if(total % 3){
            unique |= (1LL << i);
        }
    }
    
    return (int)unique;

}
