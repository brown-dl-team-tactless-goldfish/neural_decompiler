public class Solution {
    public int SplitNum(int num) {
        var digits = string.Concat(num.ToString().OrderBy(c => c)).Select(c => c - '0');

        int s1 = 0;
        int s2 = 0; 

        foreach (int digit in digits) 
            if (s1 <= s2)
                s1 = s1 * 10 + digit;
            else
                s2 = s2 * 10 + digit;
        
        return s1 + s2; 
    }
}
