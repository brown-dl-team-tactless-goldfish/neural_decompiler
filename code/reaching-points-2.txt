    if(sx > tx || sy > ty)
    {
        return false;
    }
    
    if(sx == tx && sy == ty)
    {
        return true;
    }
    if(tx == sx)     
    {
       return (ty-sy)% tx == 0;
    }
    if(ty == sy)
    {
        return (tx-sx)% ty == 0;
    }
    
    if(tx > ty)
    {
        return reachingPoints(sx,sy,tx%ty,ty);
    }
else
{
     return reachingPoints(sx,sy,tx,ty%tx);
        
}
