bool isMonotonic(int* A, int ASize){
    bool inc = false;
    bool dec = false;
    int size = ASize - 1;
    for (int i = 0; i < size; i++)
    {
        if (A[i] < A[i + 1])
        {
            if (dec)
            {
                return false;
            }
            inc = true;
        }
        else if (A[i] > A[i + 1])
        {
            if (inc)
            {
                return false;
            }
            dec = true;
        }
    }
    
    return true;
}
