class Solution {
public:
    int minMoves(int target, int maxDoubles) {
        int moves=0;
        while(target!=1){
            if(maxDoubles==0){
                return moves+target-1;
            }
            else if(maxDoubles>0&&(target&1)==0){
                target/=2;
                maxDoubles--;
            }
            else{
                target--;
            }
            moves++;
        }
        return moves;
    }
};
