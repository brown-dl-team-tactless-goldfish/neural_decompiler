void duplicateZeros(int* arr, int arrSize){
    int temp = 0;
    int i = 0;
    while(i < arrSize) {
        if(arr[i] == 0) {
            for(int j = arrSize - 2; j > i; j--) {
                arr[j + 1] = arr[j];
            }
            if(i == arrSize - 1) {
                break;
            }
            arr[i + 1] = 0;
            i += 2;
            continue;
        }
        i++;
    }
}
