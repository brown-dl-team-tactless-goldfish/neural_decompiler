bool isBadVersion(int version);

int firstBadVersion(int n) {
    if(n==1) return 1;
    unsigned long long int mid, low=1, high=n;
    int ret;
    while(low<=high)
    {
        mid=(high+low)/2;
        if(isBadVersion(mid)){
            high=mid-1;
            ret=mid;
        }
        else{
            low=mid+1;
            ret=mid+1;
        }
    }
    return ret;
}
