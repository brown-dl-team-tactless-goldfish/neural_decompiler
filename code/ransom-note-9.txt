
bool canConstruct(char * ransomNote, char * magazine){
    int map[26]={0};
    int i;
    
	//store count of each alphabet in map
    for(i=0;magazine[i];i++)
        map[magazine[i]-'a']++;
    
	//decrement count
    for(i=0;ransomNote[i];i++)
        map[ransomNote[i]-'a']--;
    
	//now check if there is any -ve count in map 
    for(i=0;i<26;i++)
        if(map[i]<0)
            return false;
    
    return true;
}
