struct ListNode* swapPairs(struct ListNode* head){
    struct ListNode * start = malloc(sizeof(struct ListNode));
    struct ListNode * former, * later, * temp;
    start->next = head;
    temp = start->next;
    former = start;
    while(temp != NULL){
        later = temp->next;
        if (later == NULL){
            former->next = temp;
            return start->next;
        }
        former->next = later;
        former = temp;
        temp = later->next;
        later->next = former;
        former->next = NULL;
    }
    return start->next;
}
