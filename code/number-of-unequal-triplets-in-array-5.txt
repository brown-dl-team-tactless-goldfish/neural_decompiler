
int unequalTriplets(int *nums, int numsSize)
{
int ans = 0;
int n = numsSize;
for (int i = 0; i < n - 2; i++)
{
    // Fix the second element as A[j]
    for (int j = i + 1; j < n - 1; j++)
    {
        // Now look for the third number
        for (int k = j + 1; k < n; k++)
            if (nums[i] != nums[j] && nums[j] != nums[k] && nums[i] != nums[k])
                ans++;
    }
}
return ans;
}
