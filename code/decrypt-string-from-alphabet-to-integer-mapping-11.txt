char * freqAlphabets(char * s){
    int i = 0, j = 0, len = strlen(s);
    char * res = malloc(sizeof(char) * (len + 1));
    while (j < len) {
        if (j + 2 < len && s[j + 2] == '#') {
            res[i++] = (s[j] - '0')*10 + (s[j + 1] - '0') - 1 + 'a';
            j += 3;
        }
        else {
            res[i++] = (s[j] - '0') - 1 + 'a';
            j++;
        }
    }
    res[i] = '\0';
    return res;
}
