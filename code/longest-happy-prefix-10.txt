class Solution {
public:
    string longestPrefix(string s) {
        int c = 0;
        vector<int> next(1,0);
        for(int i = 1;i < s.size(); i++){
            while(s[i] != s[c] and c > 0) c = next[c-1];
            if(s[i] == s[c]) c++;
            next.push_back(c);
        }
        return s.substr(0,next.back());
    }
};
