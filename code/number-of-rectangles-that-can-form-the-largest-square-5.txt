int countGoodRectangles(int** rectangles, int rectanglesSize, int* rectanglesColSize){
    int map[rectanglesSize], i, count = 1, max = 0;
    for (i = 0; i < rectanglesSize; ++i) {
        map[i] = (rectangles[i][0] >= rectangles[i][1]) ? rectangles[i][1] : rectangles[i][0];
    }
    max = map[0];
    for (i = 1; i < rectanglesSize; ++i) {
        if (map[i] > max) {
            max = map[i];
            count = 1;
        } 
        else if (map[i] == max) ++count;
    }
    return count;
}
