#define h_size 101 /*0-100 */
#define range  100
int countKDifference(int* nums, int numsSize, int k){
    int* hash = (int*)calloc(h_size, sizeof(int));
    for(int i = 0; i < numsSize; i++)
        hash[nums[i]]++;
    int ans = 0;
    for(int i = 1; (i+k) <= range && (i <= range); i++)
    {
        if(hash[i] && hash[i+k])
        {
            ans += (hash[i]*hash[i+k]);
        }
    }
    return ans;
}
