public class Solution 
{
    public bool IsCircularSentence(string sentence) 
    {
        var s = sentence.Split(' ').ToList(); 
        for (int i = 0; i < s.Count-1; i++)
        {
            if (s[i][s[i].Length - 1] != s[i + 1][0]) return false; 
        } 
        return s[s.Count - 1][s[s.Count - 1].Length - 1] == s[0][0];
    }
}
