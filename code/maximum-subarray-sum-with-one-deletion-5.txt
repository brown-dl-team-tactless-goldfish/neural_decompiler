int max(int a, int b)
{
    return a>b?a:b;
}
int maximumSum(int* arr, int arrSize){
    int ans = arr[0];
    int excl=0,incl=arr[0];
    for(int i=1;i<arrSize;i++)
    {
        excl= max(excl+arr[i],incl);    //Taking which is maximum of previous including or the current + previous excluding
        incl = max(arr[i],incl+arr[i]);    // Taking maximum of current element or previous including + current element
        ans = max(ans,max(excl,incl)); //Keeping track of maximum so far
    }
    return ans;
}
