char * baseNeg2(int n){
    char* ret = malloc(41);
    memset(ret, '0', 40);
    ret[40] = '\0';
    int ind = 39;
    while (n > 0) {
        ret[ind] = (n & 1) + '0';
        if (ret[ind] == '1' && !(ind & 1))
           n++;
        n >>= 1;
        ind--;
    }
    return ind == 39 ? ret + ind : ret + ind + 1;
}
