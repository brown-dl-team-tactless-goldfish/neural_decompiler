public class Solution 
{
    public int MinimumSwap(string s1, string s2) 
    {
        int x = 0, y = 0;
        for(int i = 0; i < s1.Length; i++)
        {
            if(s1[i]!=s2[i])
            {
                if(s1[i]=='x') x++;
                else y++;
            }
        }

        return ((x + y) % 2 == 1) ? - 1 : (x + 1)/2 + (y + 1)/2;
    }
}
