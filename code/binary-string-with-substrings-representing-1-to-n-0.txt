char* intToString(int val){
    char* ans = malloc(32 * sizeof(char));
    ans[31] = '\0';
    int idx = 30;
    while(val){
        ans[idx] = val % 2 + '0';
        val /= 2;
        idx--;
    }
    return &ans[idx+1];
}
bool queryString(char * s, int n){
    bool ans = true;
    for(int i = n; i > 0; i--){
        char* tmp = intToString(i);
        if (strstr(s, tmp) == NULL) {
            ans = false;
            break;
        }
    }
    return ans;
}
