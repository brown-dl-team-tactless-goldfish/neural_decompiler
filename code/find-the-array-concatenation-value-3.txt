long long findTheArrayConcVal(int* nums, int numsSize) {

	register unsigned long long sum = 0;
	register unsigned short i = 0, j = numsSize - 1;
    register unsigned char j_size;
	while(i < j) {
		j_size = floor(log10(abs(nums[j]))) + 1;
		sum += (nums[i] * (int)pow(10, j_size)) + nums[j];
		i++;
		j--;
	}
	if(numsSize % 2 == 1)
		sum += nums[(numsSize / 2)];

	return sum;
}
