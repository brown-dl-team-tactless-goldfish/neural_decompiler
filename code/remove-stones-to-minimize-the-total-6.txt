class Solution:
    def minStoneSum(self, piles: List[int], k: int) -> int:
        prv_sum = sum(piles)
        print ("prv_sum = ", prv_sum)
        
        data = []
        for p in piles:
            data.append(0 - p)
            
        heapq.heapify(data)
        # print ("1 data = ", data)
        
        diff = 0
        for i in range(k):
            tmp1 = 0 - heapq.heappop(data)
            # print ("tmp1 = ", tmp1)
            tmpDiff = tmp1 // 2
            diff += tmpDiff
            heapq.heappush(data, tmpDiff-tmp1)
            
        print ("diff = ", diff)
        
        return prv_sum - diff
