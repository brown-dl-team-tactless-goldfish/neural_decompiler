class Solution {
public:
    bool makeEqual(vector<string>& words) {
        unordered_map<char,int>m;
        
        for(string s : words){
            for(char c : s)m[c]++;
        }
        
        for(auto i : m){
            if(i.second % words.size() != 0) return false;
        }
        return true;
    }
};
