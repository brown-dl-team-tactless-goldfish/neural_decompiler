int missingNumber(int* nums, int numsSize){
    int arr[numsSize];
    for(int i =0; i<numsSize;i++){
        arr[i] = i;
    }
    for(int i = 0; i<numsSize; i++){
        
        if(nums[i]>=numsSize){
            continue;
        }
        else{
            arr[nums[i]] = -1;
        }
    }
    for(int i=0;i<numsSize;i++){
        if(arr[i]!= -1){
            return arr[i];
        }
    }
    return numsSize;
}
