class Solution {
public:
    int stoneGameVIII(vector<int>& stones) {
        int s = accumulate(stones.begin(), stones.end(), 0);
        int best = numeric_limits<int>::min();
        int sol = 0;
        for (int i = stones.size() - 2; i >= 0; --i)
        {
            best = max(best, s - sol);
            s -= stones[i + 1];
            sol = best;
        }
        return best;
    }
};
