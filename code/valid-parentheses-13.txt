    if (s[i] == '(' || s[i] == '{' || s[i] == '[')
    {
        top++;
        stackstr[top] = s[i];
        
    }
    else if(s[i] == ')' || s[i] == '}' || s[i] == ']')
    {
        if(top == -1)
        {return false;}
        if(s[i] == ')' && stackstr[top] != '('|| s[i] == ']' && stackstr[top] != '[' || s[i] == '}' && stackstr[top] != '{' )
        {
            return false;
        }
        else{top--;}
    }
    
}
if(top >= 0 )
{return false;}
return true;
