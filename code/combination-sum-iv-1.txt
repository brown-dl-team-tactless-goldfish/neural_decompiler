int combinationSum4(int* nums, int numsSize, int target) {
    double* dp=(int*)calloc(target+1,sizeof(double));
    dp[0]=1;
    for(int i=0;i<=target;i++){
        for(int j=0;j<numsSize;j++){
            if(i>=nums[j]){
                dp[i]+=dp[i-nums[j]];
            }
        }
    }
    int temp = (int) dp[target];
    free(dp);
    return temp;
}
