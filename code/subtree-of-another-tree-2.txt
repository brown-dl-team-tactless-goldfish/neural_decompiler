/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     struct TreeNode *left;
 *     struct TreeNode *right;
 * };
 */

bool is_match(struct TreeNode* root, struct TreeNode* subRoot)
{
        if (!root && !subRoot)
                return true;
        if (!root || !subRoot)
                return false;
        
        if (root->val != subRoot->val)
                return false;
        
        return is_match(root->left, subRoot->left) && is_match(root->right, subRoot->right);
}

bool isSubtree(struct TreeNode* root, struct TreeNode* subRoot)
{
        if (!root)
                return false;
        
        if (root->val == subRoot->val && is_match(root, subRoot))
                return true;
        
        return isSubtree(root->left, subRoot) || isSubtree(root->right, subRoot);
}
