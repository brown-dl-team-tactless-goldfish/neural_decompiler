bool validateStackSequences(int* pushed, int pushedSize, int* popped, int poppedSize){
    
    int i = 0;
    int j = 0;
    
    for(int x = 0; x<pushedSize; x++)
    {
        pushed[i] = pushed[x];
        i++;
        
        while(i>0 && pushed[i-1] == popped[j])
        {
            i--;
            j++;
        }
    }
    
    return i==0;

}
