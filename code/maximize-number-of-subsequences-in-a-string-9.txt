class Solution {
public:
    long long maximumSubsequenceCount(string text, string pattern) {
        long long ans = 0, c1 = 0, c2 = 0;
        for (auto c : text) {   
            if (c == pattern[1])
                ans += c1, c2++;
            if (c == pattern[0])
                c1++;
        }
        return ans + max(c1, c2);
    }
};
