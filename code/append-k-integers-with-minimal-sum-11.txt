    long long minimalKSum(vector<int>& nums, int k) {
        nums.push_back(0);
        nums.push_back(2e9);
        sort(nums.begin(), nums.end());
        long long sum = 0;
        for(int i=0;i<nums.size()-1 && k>0;i++){
            long long a = nums[i]+1, b = nums[i+1]-1;
            if(a>b) continue;
            if(b-a+1 > k) b = a+k-1;
            k -= (b-a+1);
            sum += (a+b)*(b-a+1)/2;
        }
        return sum;
    }
