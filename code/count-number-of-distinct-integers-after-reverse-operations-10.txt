class Solution {
public:
    int countDistinctIntegers(vector<int>& nums) {
        set<int> st;
        int n = nums.size();
        for(int i= 0;i<n;i++){
            string temp = to_string(nums[i]);
            reverse(temp.begin(),temp.end());
            st.insert(stoi(temp));
            st.insert(nums[i]);
            nums.push_back(stoi(temp));
        };

         return st.size();
    }
};
