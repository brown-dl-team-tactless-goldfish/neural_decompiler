/**
 * Note: The returned array must be malloced, assume caller calls free().
 */
int* twoSum(int* numbers, int numbersSize, int target, int* returnSize){
    int *output = (int*)malloc(2*sizeof(int));
    *returnSize = 2;
    int startIndex = 0;
    int endIndex = numbersSize - 1;
    
    while(startIndex<endIndex) {
        int sum = numbers[startIndex] + numbers[endIndex];
        if (sum == target) {
            output[0] = startIndex + 1;
            output[1] = endIndex + 1;
            return output;
        } else if (sum < target) {
            startIndex++;
        } else {
            endIndex--;
        }
    }
    return output;
}
