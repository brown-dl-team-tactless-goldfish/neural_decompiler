class Solution {
public:
    int smallestRangeII(vector<int>& A, int K) {
        std::sort(A.begin(), A.end());
        int min_{A.back() - A[0]};
        for(int i{1}, size{(int)A.size()}; i < size; ++i)
            min_ = std::min(min_,  std::max(A[size-1]-K, A[i-1]+K) - std::min(A[0]+K, A[i]-K));
        
        return min_;
    }
};
