class Solution {
public:
    int preimageSizeFZF(int k) {      
        long long lo = 0, hi = 1LL << 62;
        while (lo <= hi) {
            long long mid = (lo + hi) >> 1;
            
            long long count = 0;
            for (long long i = 5; i <= mid; i *= 5) count += mid / i;
            
            if (count == k) return 5;
            else if (count > k) hi = mid - 1;
            else lo = mid + 1;
        }
        
        return 0;
    }
};
