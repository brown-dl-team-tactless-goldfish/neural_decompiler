char* addSpaces(char* s, int* spaces, int spacesSize) {
    int slen = strlen(s), len = slen + 1 + spacesSize;
    char* res = (char*) malloc(len * sizeof(char));
    res[len - 1] = '\0';
    for (int i = 0; i < spacesSize; i++) {
        res[spaces[i] + i] = ' ';
    }
    int idx = 0;
    for (int i = 0; i < len; i++) {
        if (res[i] != ' ') res[i] = s[idx++];
    }
    return res;
}
