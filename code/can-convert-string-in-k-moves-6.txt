class Solution {
public:
    bool canConvertString(string s, string t, int k) {
        int val=0;
        if(s.size()!=t.size())return false;
        unordered_map<int,int>m;//it contains the number of times diff appeared
        for(int i=0;i<s.size();i++){
            int diff=t[i]-s[i];
            if(diff==0)continue;
            if(diff<0)diff+=26;
            if(m[diff]*26+diff>k)return false;
            m[diff]++;
        }
        return true;
    }
};
