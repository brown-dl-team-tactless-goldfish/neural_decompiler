class Solution {
public:
    vector<int> findSmallestSetOfVertices(int n, vector<vector<int>>& nums)
    {
        vector<int>v;
        vector<int>mark(n,false);
        for(auto it:nums)
        {
            mark[it[1]]=true;
        }
        for(int i=0;i<n;i++)
        {
            if(mark[i]==false)
            {
                v.push_back(i);
            }
        }
        return v;
    }
};
