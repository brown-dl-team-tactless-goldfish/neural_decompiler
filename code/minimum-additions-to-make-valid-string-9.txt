class Solution {
public:
    int addMinimum(string word) {
        int n=word.length(),ans=0;
        for(int i=0;i<n;i++){
            if(word[i]=='a'){
                if(i<n-2 && word[i+1]=='b' && word[i+2]=='c') i+=2;
                else if(i<n-1 && (word[i+1]=='b' || word[i+1]=='c')) {
                    ans+=1;
                    i+=1;
                }
                else ans+=2;
            }
            else if(word[i]=='b'){
                if(i<n-1 && word[i+1]=='c'){
                    ans+=1;
                    i+=1;
                }
                else ans+=2;
            }
            else ans+=2;
        }
        return ans;
    }
};
