class Solution:
    def largestOddNumber(self, num: str) -> str:
        n = len(num)
        self.res = ""
        
        idx = -1
        for i in range(n-1, -1, -1):
            c = num[i]
            if (int(c) % 2 == 1):
                idx = i
                break
                
        print ("idx = ", idx)
        
        return num[:idx+1]
