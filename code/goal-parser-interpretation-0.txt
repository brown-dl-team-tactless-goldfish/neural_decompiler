char* interpret(char* command) {
    int size = strlen(command) + 1, residx = 0;
    char* res = (char*) malloc(size * sizeof(char));
    for (int i = 0; i < size; i++) {
        if (command[i] == 'G') res[residx++] = 'G';
        if (command[i] == ')' && command[i-1] == '(') res[residx++] = 'o';
        if (command[i] == 'a') res[residx++] = 'a';
        if (command[i] == 'l') res[residx++] = 'l';
    }
    res[residx] = '\0';
    return res;
}
