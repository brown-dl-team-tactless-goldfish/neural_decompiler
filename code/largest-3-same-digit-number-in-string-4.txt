char * largestGoodInteger(char * num){
    int cMax = -1, pos = -1;
    int length = strlen(num);
    for(int i = 2; i < length; i++) {
        if (num[i] == num[i - 1] && num[i] == num[i - 2]) {
            if (cMax < num[i] - '0') {
                cMax = num[i] - '0';
                pos = i - 2;
            }
        }
    }
    if (pos == -1) return "";
    num[pos + 3] = '\0';
    return &num[pos];
}
