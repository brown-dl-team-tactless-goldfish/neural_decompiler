    unordered_map<int,int> m;
    
    for(auto &x: nums)
    {
        sum += x;
        
        if(sum == goal)
        {
            count++;
        }
        
        if(m.find(sum - goal) != m.end())
        {
            count += m[sum-goal];
        }
        
        m[sum]++;
    }
    
    return count;
