int strstr_a(char *p1, char *p2) {
    while(*p1 && *p2) {
        if(*p1 != *p2) return 0;
        p1++;
        p2++;
    }
    return (*p2 == '\0');
}
int isPrefixOfWord(char * sentence, char * searchWord){
    int i = 1;
    char *p = sentence;
    char *token = strtok(p, " ");
    while(token) {
        if(strstr_a(token, searchWord)) return i;
        token = strtok(NULL, " ");
        i++;
    }
    return -1;
}
