class Solution {
public:
    int countTime(string time) {
        int a = 1, b = 1, c = 1, d = 1;
        if(time[0] == '?')
        {
            if(time[1] - '0' <= 3 || time[1] == '?')
              a = 3;
            else a = 2;
        }
        if(time[1] == '?')
        {
            if(time[0] - '0' == 2)
                b = 4;
            else if(time[0] == '?')
                b = 8; // for ?? : mm we can have 24 options, so in the end we will multiply a*b (ie, 3*8) to get 24 choices
            else b = 10;
        }
        if(time[3] == '?')
        {
            c = 6;
        }
        if(time[4] == '?')
        {
            d = 10;
        }
        return a*b*c*d;
    }
};
