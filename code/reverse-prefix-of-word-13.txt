

char * reversePrefix(char * word, char ch){
    int ind = -1;
    
    for(int i = 0; i < strlen(word); i++)
        if(word[i] == ch) {
            ind = i;
            break;
        }
    
    
    if(ind == -1)
        return word;
    else {
        char temp;
        for(int i = 0; i <= ind / 2; i++) {
            temp = word[i];
            word[i] = word[ind - i];
            word[ind - i] = temp;
        }            
    }
    return word;    
}
