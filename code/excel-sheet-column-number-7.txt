

int titleToNumber(char * columnTitle){
    int i,column=0,j=0;
    int len=strlen(columnTitle);                    //length of string
    for(i=len-1;i>=0;i--)                           //traverse in string
    {
        column+=pow(26,j)*(columnTitle[i]-'A'+1);   //base of each character is 26 so we multiply ith character with pow(26,i)
        j++;
    }
    return column;                                  //return column number
}
