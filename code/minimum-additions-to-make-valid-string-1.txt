int addMinimum(char * word){
    int len=strlen(word);
    if(len==1) return 2;
    int idx=0;
    int ans=0;
    while(idx<len){
        if(idx==len-1){
            ans+=2;
            idx++;
        }
        else if(idx+2 < len && word[idx]=='a' && word[idx+1]=='b' && word[idx+2]=='c'){
            idx+=3;
        }
        else if(idx+1<len){
            if(word[idx]>word[idx+1]){
                ans+=2;
                idx++;
            }
            else if(word[idx]=='a' && word[idx+1]=='b'){
                ans++;
                idx+=2;
            }
            else if(word[idx]=='a' && word[idx+1]=='c'){
                ans++;
                idx+=2;
            }
            else if(word[idx]=='b' && word[idx+1]=='c'){
                ans++;
                idx+=2;
            }
            else if(word[idx]==word[idx+1]){
                ans+=2;
                idx++;
            }
        }
        
        
    }
    return ans;
}
