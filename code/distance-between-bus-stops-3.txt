int distanceBetweenBusStops(int* distance, int distanceSize, int start, int destination){
    int s1=0,s2=0;
    if(start>destination)
    {
        int temp=start;
        start=destination;
        destination=temp;
    }
   for(int i=0;i<distanceSize;i++)
   {
      if(i>=start&&i<destination)
      {
        s1+=distance[i];
      }
      else
      s2+=distance[i];
   }
   return s1>s2?s2:s1; 
}
