int countSegments(char * s){

    int ret = 0;
    if (s == NULL) return 0;

    while (*s != '\0')
    {
        if(*s != ' ')
        {
            while(*s != '\0')
            {
                s++;
                if(*s == ' ' || *s == '\0')
                {
                    ret++;
                    break;
                }
            }
        }
        if (*s == '\0') break;
        s++;
    }
    return ret;
}
