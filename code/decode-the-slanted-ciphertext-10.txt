string decodeCiphertext(string encodedText, int rows) {
        
	int n = encodedText.size();
	int cols = n / rows;

	string decodedText = "";

	for(int i=0; i < cols; i++) {
		// Move by distance (col+1) to get next character
		for(int j=i; j<n; j += (cols+1)) {
			decodedText += encodedText[j];
		}
	}

	while(decodedText.back() == ' ') decodedText.pop_back();

	return decodedText;
}
