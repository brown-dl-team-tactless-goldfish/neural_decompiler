class Solution {
public:
    int maxDotProduct(vector<int>& nums1, vector<int>& nums2) 
    {
        int n=nums1.size();
        int m=nums2.size();
        vector<vector<int>>v(n+1,vector<int>(m+1,-1e6));
        for(int i=1;i<=n;i++)
        {
            for(int j=1;j<=m;j++)
            {
                v[i][j]=max({nums1[i-1]*nums2[j-1],nums1[i-1]*nums2[j-1]+v[i-1][j-1],v[i-1][j],v[i][j-1]});
            }
        }
        return v[n][m];
    }
};
