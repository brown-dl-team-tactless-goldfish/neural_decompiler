class Solution {
public:
    
    int shortestSequence(vector<int>& rolls, int k) {
        unordered_map<int, int> mp;
        int c = 0, seq = 0;
        for(auto i: rolls){
            mp[i]++;
            if(mp[i] == 1) c++;
            
            if(c == k){
                seq++;
                c = 0;
                mp.clear();
            }
        }
       return seq + 1;
        
       
    }
};
