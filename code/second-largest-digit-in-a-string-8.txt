int secondHighest(char * s){
int max = -1, smax = -1;

for(int i =0; i < strlen(s); i++){
    if(isdigit(s[i])){
        int num = s[i] - '0';
        if(num > max){
            smax = max;
            max = num;
        }
        else if(num > smax && num < max){ // Checks if it is between second largest and largest
            smax = num;
        }
    }
}
return smax;
}
