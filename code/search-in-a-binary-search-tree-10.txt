struct TreeNode* searchBST(struct TreeNode* root, int val){
    struct TreeNode* t=root;
    while(t!=NULL)
    {
        if(val==t->val) return t;
        else if(val>t->val)
        {
            if(t->right!=NULL) t=t->right;
            else return NULL;
        }
        else if(val<t->val)
        {
            if(t->left!=NULL) t=t->left;
            else return NULL;
        }
    }
    return NULL;
}
