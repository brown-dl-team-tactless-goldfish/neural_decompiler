  public int CountWords(string[] words1, string[] words2) {
        
        var a = RemoveDuplicate(words1);
        var b = RemoveDuplicate(words2);
        return a.Intersect(b).Count();         
    }
    
    private IEnumerable<string> RemoveDuplicate(string[] a)
    {
       return a.GroupBy(g => g)
            .Select(s => new {
                count = s.Count(),
                word = s.Key
            })
            .Where(w => w.count == 1)
           .Select(s => s.word);
    }
