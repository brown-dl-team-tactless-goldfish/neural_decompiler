class Solution {
public:
  vector<int> sortEvenOdd(vector<int>& nums) {
    vector<int>v0, v1, ans(nums.size());
    for(int i = 0; i < nums.size(); i+=2) v0.push_back(nums[i]);
    for(int i = 1; i < nums.size(); i+=2) v1.push_back(nums[i]);
    sort(v0.begin(),v0.end());
    sort(v1.begin(),v1.end(),greater());
    
    for(int id = 0, i = 0; i != v0.size(); id+=2, i++) ans[id] = v0[i];
    for(int id = 1, i = 0; i != v1.size(); id+=2, i++) ans[id] = v1[i];
    
    return ans;
  }
};
