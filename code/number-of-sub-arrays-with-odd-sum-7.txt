    long even = 0, odd = 0, ans = 0, mod = 1e9+7;
    
    for(auto &x: arr)
    {
        if(x%2 == 0)
        {
            even++;
        }
        
        else
        {
            swap(odd,even);
            odd++;   
        }
        ans = (ans + odd)% mod;
    }
    
    return ans;
    
}
