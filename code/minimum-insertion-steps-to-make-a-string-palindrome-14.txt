    int dp[n+1][n+1];
    
    memset(dp,0,sizeof(dp));
    
    for(int i=0; i<n; i++)
    {
        for(int j=0; j<n; j++)
        {
            if(s[i] == s[n-1-j])
            {
                dp[i+1][j+1] = 1+dp[i][j];
            }
            else
            {
                dp[i+1][j+1] = max(dp[i+1][j],dp[i][j+1]);
            }
        }
    }
    
    return n - dp[n][n];
}
