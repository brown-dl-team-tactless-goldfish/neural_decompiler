bool predicate(vector<int>& nums, int threshold, int val)
{
    int c = 0 ;
    for(auto x:nums)
        c += (x + val-1)/val ;
    
    return c > threshold ;
    
}

int smallestDivisor(vector<int>& nums, int threshold) {
    
    int hi = *max_element(nums.begin(),nums.end()) ;
    int lo = 1 ;
    
    while(hi >= lo)
    {
        int mid = lo + (hi - lo)/2 ;
        
		// We can also use ternary operator 
        if(predicate(nums,threshold,mid)) lo = mid + 1 ;
        else hi = mid - 1 ;
    }
    
    return lo;
    
}
