long mod = 1e9+7;

int nthMagicalNumber(int n, int a, int b) {
    
    long low = min(a,b), high = (long)n*min(a,b);
    
    long lcm = (a*b) / __gcd(a,b);
    
    while(low < high)
    {
        long mid = low + (high-low) / 2;

       long  temp = mid/a + mid/b - mid/lcm;
        
        if(temp < n)
        {
            low = mid+1;
        }
        
        else
        {
            high = mid;
        }
    }
    
    return low %  mod;
    
}
