class Solution {
public:
    int getMaximumConsecutive(vector<int>& arr) 
    {
        sort(arr.begin(),arr.end());
        int count=1;
        int n=arr.size();
        for(int i=0;i<n;i++)
        {
            if(count<arr[i])
            {
                break;
            }
            count+=arr[i];
        }
        return count;
    }
};
