int* findBall(int** grid, int gridSize, int* gridColSize, int* returnSize){
	// set up return values
    int ncol = *gridColSize;
    *returnSize = ncol;
    
	// malloc return array (user will free later) and initialize to 0:ncol
    int *retval = malloc(sizeof(int) * ncol);
    for (int i=0; i<ncol; i++)
        retval[i] = i;

	// iterate over each ball
    int height, v, temp;
    for (int i=0; i<ncol; i++){
        height = 0;
        v = i;
		// follow the path of the ball
        while (height < gridSize){
			temp = grid[height][v];              // direction we're moving
            v += temp;                           // change the x position of the ball
			
			// if the ball falls off the grid or the ball would have been caught in a v, 
			// set to -1 and break
            if (v == ncol || v == -1 || temp + grid[height][v] == 0){
                v = -1;
                break;
            }
			// else move down 1 in y position
            height++;
        }
		// save result
        retval[i] = v;
    }
    
    return retval;
}
