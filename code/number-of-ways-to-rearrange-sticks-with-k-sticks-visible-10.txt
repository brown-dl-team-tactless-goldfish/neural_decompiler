class Solution {
public:
    int rearrangeSticks(int n, int k) {
        vector<vector<int>>dp(n,vector<int>(k+1,0));
        dp[0][1]=1;
        int mod = pow(10,9)+7;
        for(long int i=1;i<n;i++){
            for(int j=1;j<=k;j++){
                dp[i][j]=(dp[i-1][j-1]+(i*dp[i-1][j])%mod)%mod;
            }
        }
        return dp[n-1][k];
    }
};
