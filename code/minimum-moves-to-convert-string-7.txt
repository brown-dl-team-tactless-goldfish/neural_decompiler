class Solution {
public:
    int minimumMoves(string s) 
    {
        int n=s.size();
        int count=0;
        for(int i=0;i<n;)
        {
            if(s[i]=='O')
            {
                i++;
            }
            else
            {
                count++;
                i+=3;
            }
        }
        return count;
    }
};
