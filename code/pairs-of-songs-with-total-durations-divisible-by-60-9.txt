#include <stdio.h>
#include <stdlib.h>

int numPairsDivisibleBy60(int* time, int timeSize){
    if(timeSize==1) return 0;
    int ans=0;
    int rem[60]={0};
    for(int i=0;i<timeSize;i++){
        rem[time[i]%60]++;
    }
    printf("rem[0]=%d\n",rem[0]);
    for(int i=1;i<30;i++){
        ans+=rem[i]*rem[60-i];
    }
    printf("ans_now=%d\n",ans);
    if(rem[30]!=0){
        int half=0;
        for(int i=rem[30]-1;i>0;i--)
            half+=i;
        ans+=half;
    }
    if(rem[0]!=0){
        int canDiv_0=0;
        for(int i=rem[0]-1;i>0;i--)
            canDiv_0+=i;
        printf("canDiv_0=%d\n",canDiv_0);
        ans+=canDiv_0;
    }
    return ans;
}
