char* defangIPaddr(char* address){
    int i=0;
    int j=0;
    int len=strlen(address);
    char* result = (char*) malloc(sizeof(char)*(len+7)); // 6 for [ and ] and 1 for null character.
    while(address[i]!='\0'){       
        if(address[i]=='.'){
            result[j]='[';
            result[++j]='.';
            result[++j]=']';
        }
        else{
            result[j]=address[i];
        }
        j++;
        i++;
    }
    result[j]='\0';
    return result;
}
