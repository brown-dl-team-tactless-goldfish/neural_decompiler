int wiggleMaxLength(int* nums, int numsSize){
    int dir = 0, ans = 1;
    
    for (int i = 1; i < numsSize; i++) {
        if (nums[i] == nums[i-1]) continue;
        if (dir == 0) {
            ans++;
            if (nums[i] > nums[i-1]) dir = 1;
            else dir = -1;
        }
        else if (dir * (nums[i] - nums[i - 1]) < 0) {
            ans++;
            dir *= -1;
        }
    }
    
    return ans;
}
