string kthSmallestPath(vector<int>& D, int k) {
    k--; // make 0-indexed
    int n = D[0] + D[1];
    string ans(n, 'H');
    for (int i = D[0]; i > 0; i--) {
        int a = i - 1, b = 0, c = 1;
        while (k >= c) {
            k -= c;
            c = c * ++a / ++b;
        }
        ans[n - 1 - a] = 'V';
    }
    return ans;
}
