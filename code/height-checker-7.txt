int heightChecker(int* heights, int heightsSize){
    int result = 0;
    int cmp(const void *x, const void *y){
        return *(int*)x > *(int*)y;
    }
    
    int* new_arr = (int*)malloc(sizeof(int)*heightsSize);
    memcpy(new_arr, heights, sizeof(int)*heightsSize);
    qsort(heights, heightsSize, sizeof(int), cmp);
    for (int i = 0;i<heightsSize;i++){
        if (heights[i] != new_arr[i]){
            result++;
        }
    }
    return result;    
}
