class Solution {
public:
    string greatestLetter(string s) {
        unordered_map<char,int> m;
        for (auto& c : s) ++m[c];
        string ans = "";
        for (char c = 'A'; c <= 'Z'; ++c){
            if (m[c] > 0 && m[tolower(c)] > 0)
            {
                ans.clear();
                ans += c;
            }
        }
        return ans;
    }
};
