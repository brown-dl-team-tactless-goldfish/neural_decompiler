class Solution {
public:
	int numOfSubarrays(vector<int>& arr, int k, int threshold) {
		int t = k * threshold;
		int s = 0;
		int c = 0;
		for(int i = 0; i < k; i++){
			s += arr[i];
		}
		if(s >= t){
			c++;
		}
		for(int i = k; i < arr.size(); i++){
			s += arr[i] - arr[i - k];
			if(s >= t){
				c++;
			}
		}        
		return c;
	}
};
