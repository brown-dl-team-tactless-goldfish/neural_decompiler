int minSteps(char * s, char * t){

    int rem[26] = {0};
    int len = strlen(s);
    int ans=0;
    
    for(int i=0; i < len; i++) {
        
        rem[s[i]-'a']++;
        rem[t[i]-'a']--;
        
    }
    
    for(int i=0; i<26; i++) {
        if(rem[i] > 0)
            ans = ans + rem[i];
        
    }
    
    return ans;
}
