int cmpfunc (int **a, int **b)
{
    int *A=*a;
    int *B=*b;
    if(A[1] == B[1]) return A[0] - B[0];
    return A[1] - B[1];
}

int findLongestChain(int** pairs, int pairsSize, int pairsColSize) {
    qsort(pairs, pairsSize, sizeof(int *), cmpfunc);
    int ret=1;
    int k=0;
    for(int i=1;i<pairsSize;i++){
        if(pairs[i][0]>pairs[k][1]){
            ret++;
            k=i;
        }
    }
    return ret;
}
