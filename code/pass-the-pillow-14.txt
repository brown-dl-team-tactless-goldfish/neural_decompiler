class Solution {
public:
    int passThePillow(int n, int time) {
        int c=1;
        bool flag=false;
        while(time>0) {
            if(!flag) {
                ++c;
                --time;
            }
            if(time == 0) continue;
            if(c==n) flag=true;
            if(flag) {
                --c;
                --time;
            }
            if(time == 0) continue;
            if(c==1) flag=false;
                
        }
        return c;
    }
};
