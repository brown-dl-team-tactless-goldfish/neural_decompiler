bool makeStringsEqual(char * s, char * target){
    int n = strlen(s) ;
    int sOne =  0 , tOne = 0;
    for(int i = 0; i < n; i++){
        if(s[i] == '1')
           sOne++ ;
        if(target[i] == '1')
            tOne++ ;
    }
    if(sOne == 0 && (tOne > 0))
        return false ;
    if(sOne > 0 && tOne == 0)
        return false ;

    return true ;

}
