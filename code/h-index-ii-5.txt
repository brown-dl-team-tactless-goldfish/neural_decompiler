public class Solution 
{
    public int HIndex(int[] citations) 
    {
        int start = 0, end = citations.Length;
        while(start < end)
        {
            int mid = start + (end - start + 1) / 2;
            if(IsValidIndex(citations, mid))
                start = mid;
            else
                end = mid - 1;
        }
        
        return start;
    }
    
    private bool IsValidIndex(int[] citations, int n)
    {
        int count = 0;
        foreach(var c in citations)
            if(c >= n)
                count++;
        return count >= n;
    }
}
