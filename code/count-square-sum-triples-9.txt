public class Solution {
    public int CountTriples(int n) {
        int[] sq = new int[n * n + 1];
        for (int i = 1; i <= n; ++i) sq[i * i] = 1;
        
        int r = 0;
        for (int i = 1; i <= n; ++i) {
            for (int j = i; i * i + j * j <= n * n; ++j) {
                r += sq[i * i + j * j] * 2;
            }
        }
        
        return r;
    }
}
