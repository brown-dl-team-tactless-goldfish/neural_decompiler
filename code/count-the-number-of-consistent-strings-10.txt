class Solution:
    def countConsistentStrings(self, allowed: str, words: List[str]) -> int:
        candidate = set()
        for c in allowed:
            candidate.add(c)
            
        res = 0
        for w in words:
            flag = True
            for c in w:
                if (c not in candidate):
                    flag = False
                    break
            if (flag == True):
                res += 1
                
        print ("res = ", res)
        
        return res
