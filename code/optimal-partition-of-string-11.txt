    public int PartitionString(string s) 
    {
        HashSet<char> hs = new HashSet<char>();
        int count = 1;
        foreach (char c in s)
        {
            if (hs.Contains(c))
            {
                hs.Clear();
                hs.Add(c);
                count++;
            }
            else
            {
                hs.Add(c);
            }
        }
        return count;
    }
