int compare(int * a, int * b) {
   return ( *a - *b ); // For quicksort
}
int heightChecker(int* heights, int heightsSize){
    int new [heightsSize]; // Create copy array, new
    for(int i = 0; i < heightsSize; i++)
    {
        new[i] = heights[i];
    }
    qsort(new,heightsSize,sizeof(int),compare); // Sort copy array
    int counter = 0; // Return value
    for(int i = 0; i < heightsSize; i++)
    {
        if(new[i] != heights[i]) // Check for mismatches
        {
            counter = counter + 1;
        }
    }
    return counter;
}
