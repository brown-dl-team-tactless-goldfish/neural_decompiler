class Solution {
public:
    string removeDigit(string number, char digit) {
        string ans = "0";
        for(int i=0;i<number.length();i++){
            if(number[i]==digit){
                string a = i>0?number.substr(0,i):"";
                string b =  i<number.length()-1?number.substr(i+1,number.end()-number.begin()):"";
                string s = a+b;
                ans=max(ans,s);
            }
        }
        return ans;
    }
};
