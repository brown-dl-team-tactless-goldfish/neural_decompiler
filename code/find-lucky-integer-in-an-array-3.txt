int compare(void *elem1, void *elem2)
{
	return (*((int *)elem1) - *((int *)elem2));
}

int findLucky(int* arr, int arrSize)
{
	int i, j;
	int new_idx = 0;
	int lucky = -1;
	int freq;
	qsort(arr, arrSize, sizeof(int), compare);
	for (i = 0, j = 1;  j <= arrSize; ++i, ++j) {
		if (j == arrSize || arr[i] != arr[j]) {
			freq = i - new_idx + 1;
			if (freq == arr[i])
				lucky = freq;
			new_idx = j;
		}
	}
	return lucky;
}
