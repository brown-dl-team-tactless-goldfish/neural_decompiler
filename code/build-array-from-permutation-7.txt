int* buildArray(int* nums, int numsSize, int* returnSize){
       
    *returnSize =numsSize ;                          //dereferencing the given pointer as this represent the size of array "ans" . 
     int *ans = (int*) malloc( numsSize * sizeof(int )) ;    //allocating the memory space  to the array "ans" 
    
     for ( int i =0; i<numsSize ; i++)
         ans[i] = nums[ nums[i]]; 
    return   ans ; 
}
