int isValidSerialization(char* preorder) {
    if(!preorder) return 1;
    int i=0, pLen = strlen(preorder);
    int nulls = 0, commas=1;
    for(i=pLen-1; i>=0; i--) {
        if(preorder[i] == ',') {commas =1; continue; }
        else {
            if(preorder[i] == '#') nulls +=1;
            else {
                 if(!commas) continue;
                 if(nulls <2) return 0;
                 else nulls -=1;
            }
           commas =0;
       }
   }
  return (nulls == 1);
}
