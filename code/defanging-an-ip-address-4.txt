char * defangIPaddr(char * address)
{
	int len = strlen(address);
	char *res = malloc(sizeof(char) * (len + 7)); // 6 for 3 sets of brackets + 1 NULL

	int i = 0, j = 0;
	while (i < len)
	{   
		res[j++] = address[i];

		if (address[i+1] == '.') 
		{
			res[j++] = '[';
			res[j++] = address[++i];
			res[j++] = ']';
		}
		i++;
	}
	res[j] = '\0';
	return res;
}
