class Solution {
public:
    int partitionDisjoint(vector<int>& nums) {
        int ans = 1, largest = 0, curLargest = nums[0];
        for (int i = 0; i < nums.size(); ++i)
            if (nums[i] < curLargest)
                curLargest = largest, ans = i + 1;
            else largest = max(nums[i], largest);
        return ans;
    }
};
