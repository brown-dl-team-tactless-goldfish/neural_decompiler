class Solution {
public:
    int minimumDeletions(string s) {
        int a = 0, b = 0;
        for (int i = 0; i < s.size(); i++) {
            if (s[i] == 'a') {
                b++;
            } else {
                b = min(a, b);
                a++;
            }
        }
        return min(a, b);
    }
};
