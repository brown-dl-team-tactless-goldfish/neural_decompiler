class Solution {
public:
    vector<long long> findPrefixScore(vector<int>& nums) {
        int x = nums.size();
        vector<long long>ans(x,0);
        int maxi=0;
        long long count=0;
        long long a = 0;
        for(int i=0;i<x;i++){
             maxi = max(maxi,nums[i]);
              a =(long long) (nums[i] + maxi);
             count+=a;
            ans[i]=count;
        }
        return ans;
    }
};
