//int n=strlen(s);
//int m=strlen(p);
char * ch=NULL,* ch1=NULL;
while (*s != '\0') 
{
    if(*p=='*')
    {
       ch=s;
       ch1=p++; 
    }
    else if(*p=='?'|| *p==*s)
    {
       s++;
       p++;
    }
    else if(ch1==NULL)
    {
        return 0;
    }
    else 
    {
        s=++ch;
        p=ch1+1;
    }
}

while (*p == '*') 
    p++;
