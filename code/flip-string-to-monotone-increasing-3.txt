int minFlipsMonoIncr(char * s) {
    int Len = strlen(s);
    int sum = 0, ans, tmp;

    for (int i = 0; i < Len; i ++) {
        if (s[i] == '1') sum ++;
    }
    ans = Len - sum;
    tmp = Len - sum;
    for (int i = 0; i < Len; i ++) {
        tmp --;
        if (s[i] == '1') tmp += 2;
        if (ans > tmp) ans = tmp;
    }
    return ans;
}
