class Solution {
public:
    int smaller_equal(int m,int n,int x)
    {
        int count=0;
        for(int i=1;i<=m;i++)
            count+=min(x/i,n);
        return count;
    }
    int findKthNumber(int m, int n, int k) 
    {
        int l=1,h=m*n;
        while(l<h)
        {
            int mid=(l+h)/2;
            int se=smaller_equal(m,n,mid);
            if(se<k)
                l=mid+1;
            else
                h=mid;
        }
        return l;
    }
};
