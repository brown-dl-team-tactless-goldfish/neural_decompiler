bool judgeCircle(char* moves) {
    int x = 0;
    int y = 0;
    int i = 0;
    int len = strlen(moves);
    for (i = 0; i <	len; ++i) {
    	if(moves[i] == 'U'){
    		y=y+1;
    	}else if(moves[i] == 'D'){
    		y = y-1;
    	}else if(moves[i] == 'L'){
    		x = x-1;
    	}else{
    		x = x +1;
    	}
	}
    if(x == 0 && y == 0){
    	return true;
    }else{
    	return false;
    }
}
