bool uniqueOccurrences(int* arr, int arrSize){
    int occur[2001] = {0};
    for(int i=0; i<arrSize; i++){
        occur[arr[i]+1000]++;
    }
    bool occupied[1000];
    memset(occupied, 0, 1000*sizeof(bool));
    for(int i=0; i<2001; i++){
        if(occur[i] == 0) continue;
        if(occupied[occur[i]]) return false;
        occupied[occur[i]] = true;
    }
    return true;
}
