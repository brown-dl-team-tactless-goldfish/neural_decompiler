int findCenter(int** edges, int edgesSize, int* edgesColSize){
    
    int count = 0;
    
    for(int i = 0; i < 2; i++){
        for(int j = 1; j < edgesSize; j++){
            for(int k = 0; k < 2; k++){
                if(edges[0][i] == edges[j][k]){
                    count++;
                    if(count == edgesSize - 1){
                        return edges[0][i];
                    }
                }
            }
        }
    
    count = 0;
    }
    
    return 0;
}
