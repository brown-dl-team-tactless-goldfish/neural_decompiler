int minimumCost(int* cost, int costSize)


{

   int size=0,i=0,sum=0,count=0;
   int * hist[101]={0};
    
   for(i=0;i<costSize;i++)
   {
      hist[cost[i]]++;
   }
    
    
   for(i=100;i>0;i--)
   {
      while(hist[i]!=0)
      {
          count++;
          hist[i]--;
          
         if(count!=3) 
          sum+=i;
          else
          {
              count=0;
          }
         
      }
       
   }
    return sum;
}
