class Solution {
public:
    int distinctPrimeFactors(vector<int>& nums) {
        unordered_set<int> sett;
        for (auto &i: nums) {
            int c = 2;
            while (i > 1) {
                if (i % c == 0) {
                    i /= c;
                    sett.insert(c);
                }else c++;
            }
        }
        return sett.size();
    }
};
