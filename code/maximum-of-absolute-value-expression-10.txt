    def maxAbsValExpr(self, x, y):
        res, n = 0, len(x)
        for p, q in [[1, 1], [1, -1], [-1, 1], [-1, -1]]:
            smallest = p * x[0] + q * y[0] + 0
            for i in xrange(n):
                cur = p * x[i] + q * y[i] + i
                res = max(res, cur - smallest)
                smallest = min(smallest, cur)
        return res
