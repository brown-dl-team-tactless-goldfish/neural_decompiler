class Solution {
public:
    int edgeScore(vector<int>& edges) {

        unordered_map<long long, long long> m;
        vector<long long> v(edges.size(), 0);
        for(long long i=edges.size() - 1; i>=0; i--) m[edges[i]] += i;
        
        long long ans = 0, fr = -1;
        for(auto i : m) v[i.first] = i.second;
        for(long long i=0; i<v.size(); i++) {
            if(v[i] > fr) ans = i, fr = v[i];
        }
        return ans;
    }
};
