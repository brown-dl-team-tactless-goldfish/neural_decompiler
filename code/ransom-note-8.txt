    bool canConstruct(char* ransomNote, char* magazine) {
        int mm[256] = {0};
        for (char *c = ransomNote; *c; c++)
            mm[*c] --;
        for (char *d = magazine; *d; d++)
            mm[*d] ++;
        bool ok = true;
        for (int i = 0; ok && i < 256; i++) 
        {
            if (mm[i] < 0)
                ok = false;
        }
        return ok;
    }
