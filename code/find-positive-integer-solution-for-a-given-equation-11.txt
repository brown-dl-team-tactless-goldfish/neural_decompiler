class Solution {
public:
    vector<vector<int>> findSolution(CustomFunction& customfunction, int z) {
        vector<vector<int>> result;
        for(int i=1;i<=z;i++){
            for(int j=z;j>=1;j--){
                if(customfunction.f(i,j) == z){
                    result.push_back({i,j});
                }
            }
        }
        return result;
    }
};
