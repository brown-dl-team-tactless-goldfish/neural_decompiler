char * restoreString(char * s, int* indices, int indicesSize) {
    char * solution = (char *) malloc((sizeof(char) * indicesSize)+1);
    
    for(int i = 0; i <= indicesSize; i++)
    {
        if(i == indicesSize)
        {
            solution[i] = '\0';
        } else
        {
            solution[indices[i]] = s[i];
        }
    }
    
    return solution;
}
