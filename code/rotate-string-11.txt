bool rotateString(char * A, char * B){
    if(*A==NULL && *B==NULL)return true;
    int alen=strlen(A);
    int blen=strlen(B);
    if(alen!=blen)return false;
    for(int i=0;i<alen;i++){
        if(A[i]==B[0]){
            bool tr=true;
            for(int j=i;j<alen;j++){
                if(A[j]!=B[j-i])tr=false;
            }
            for(int j=0;j<i;j++){
                if(A[j]!=B[j+alen-i])tr=false;
            }
            if(tr)return true;
        }
    }
    return false;
}
