class Solution {
public:
    
    int subarrayLCM(vector<int>& nums, int k) {
        int ans=0;
        int n=nums.size();
        for(int i=0;i<n;i++)
        {
            int lm=nums[i];
            for(int j=i;j<n;j++)
            {
                lm=lcm(lm,nums[j]);
                if(lm==k)
                    ans++;
                if(lm>k) break;
            }
        }
        return ans;
    }
};
