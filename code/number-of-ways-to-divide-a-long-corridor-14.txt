class Solution {
public:
    int numberOfWays(string corridor) {
        vector<int> spos;
        for (int i = 0; i < corridor.size(); i++)
            if (corridor[i] == 'S')
                spos.push_back(i);
        if (spos.size() < 2 || (spos.size() & 1))
            return 0;
        long res = 1, mod = 1e9 + 7;
        for (int i = 2; i < spos.size(); i += 2) {
            res = (res * (spos[i] - spos[i - 1])) % mod;
        }
        return res;
    }
