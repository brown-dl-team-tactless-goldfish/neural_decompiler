{
    if (root == NULL)
        return (0);
    return (1 + count_nodes(root->left) + count_nodes(root->right));
}


int check_complete(struct TreeNode* root, int index, int number_nodes)
{
    if (root == NULL)
        return (1);

    if (index >= number_nodes)
        return (0);

    return (check_complete(root->left, 2 * index + 1, number_nodes) &&
            check_complete(root->right, 2 * index + 2, number_nodes));
}

bool isCompleteTree(struct TreeNode* root)
{
	if (root == NULL)
		return (0);

	return (check_complete(root, 0, count_nodes(root)));
}
