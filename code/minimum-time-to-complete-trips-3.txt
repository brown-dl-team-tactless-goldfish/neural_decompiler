long long minimumTime(vector<int>& time, int totalTrips) {
  auto [lo, hi] = std::make_pair(1ll, static_cast<long long>(time[0]) * totalTrips);
  for (long long mid; hi > (mid = lo + (hi - lo) / 2); std::tie(lo, hi) = std::any_of(std::begin(time), std::end(time), [trip = 0ll, &mid, &totalTrips] (int t) mutable { return totalTrips <= (trip += (mid / t)); }) ? std::make_pair(lo, mid) : std::make_pair(mid + 1, hi))
    ;
  return lo;
}
