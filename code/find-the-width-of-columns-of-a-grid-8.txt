public class Solution 
{
    public int[] FindColumnWidth(int[][] grid) 
    {
        int m = grid.Length;
        int n = grid[0].Length;
        int[] result = new int[n];
        for (int i = 0; i < n; i++)
        {
            int lenght = 0;
            for (int j = 0; j < m; j++)
            {
                lenght = Math.Max(lenght, grid[j][i].ToString().Length);
            }
            result[i] = lenght;
        }
        return result;
    }
}
