class Solution {
public:
    int minOperations(vector<int>& nums) {
        int res = 0, max = 0;
        if (nums.size() == 1 && nums[0] == 0) return 0;
        for (int i : nums) {
            int c = 0;
            while (i) {
                res += i & 1;
                i >>= 1;
                c++;
            }
            max = std::max(max, c);
        }
        return res + max - 1;
    }
};
