class Solution {
public:
    string findDifferentBinaryString(vector<string>& nums) {
        unordered_set<string> s;
        for (auto &a : nums) s.emplace(a);
        int n = nums.size();
        string ans(n, '0');
        do {
            int i = rand() % n;
            ans[i] = ans[i] == '0' ? '1' : '0';
            if (!s.count(ans)) return ans;
        } while(1);
        return ans;
    }
};
