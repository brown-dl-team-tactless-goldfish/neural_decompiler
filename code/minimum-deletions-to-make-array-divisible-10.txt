class Solution {
public:
    int minOperations(vector<int>& nums, vector<int>& numsDivide) {
        int mynum=0;
        for(int &i:numsDivide)mynum=max(mynum,i);
        for(int &i:numsDivide)mynum=__gcd(mynum,i);
        sort(nums.begin(),nums.end());
        int count=0;
        for(int &i:nums){
            if(i==mynum || mynum%i==0)return count;
            count++;
        }
        return -1;
    }
};
