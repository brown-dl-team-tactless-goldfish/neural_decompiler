public class Solution {
    public int MinSwaps(string s) {
        
        if(string.IsNullOrEmpty(s))
            return 0;
              
        int openBracket = 0, invalid = 0;
        
        foreach(var c in s)
        {
            if(c == '[')
                openBracket++;
            else
            {
                if(openBracket == 0)
                    invalid++;
                else
                    openBracket--;
            }
        }
        
        return (invalid + 1) / 2;
    }
}
