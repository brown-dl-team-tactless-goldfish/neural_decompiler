class Solution {
public:
    vector<string> cellsInRange(string s) {
        vector<string> result;
        
        for (char r = s[0]; r <= s[3]; r++){
            for (char c = s[1]; c <= s[4]; c ++) {
                result.push_back({r, c});
            }
        }
        
        return result;
    }
};```
