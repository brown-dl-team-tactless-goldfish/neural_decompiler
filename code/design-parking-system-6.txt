typedef struct {
    int b;
    int m;
    int s;
    
} ParkingSystem;


ParkingSystem* parkingSystemCreate(int big, int medium, int small) {
    ParkingSystem *obj =(ParkingSystem*) malloc(sizeof(ParkingSystem));
    obj->b=big;
    obj->m=medium;
    obj->s=small;
    return obj;
}

bool parkingSystemAddCar(ParkingSystem* obj, int carType) {
    if(carType==1 && obj->b!=0)
    {
        obj->b--;
        return true;
    }
    else if(carType==2 && obj->m!=0)
    {
        obj->m--;
        return true;
    }
    else if(carType==3 && obj->s!=0)
    {
        obj->s--;
        return true;
    }


  return false;
}

void parkingSystemFree(ParkingSystem* obj) {
    free(obj);
}
