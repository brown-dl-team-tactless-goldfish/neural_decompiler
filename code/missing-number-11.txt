int missingNumber(int* nums, int numsSize){

    for(int i=0;i<numsSize;i++)
    {
        int z=0;
        for(int j=0;j<numsSize;j++)
        {
            if(nums[j]==i)
            {
                z=1;
                break;
            }
        }
        if(z==0)
        {
            return i;
        }
    }
    return numsSize;
}
