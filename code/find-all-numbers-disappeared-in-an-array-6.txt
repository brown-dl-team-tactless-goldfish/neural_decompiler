int* findDisappearedNumbers(int* nums, int numsSize, int* returnSize){
    int a[numsSize+1];
    int *ans = (int *)malloc(sizeof(int) * (numsSize+1));
    memset(a, 0, sizeof(a));
    memset(ans, 0, sizeof(int) * (numsSize+1));
    
    for (int i=0; i<numsSize; i++) {
        a[nums[i]]++;
    }
    *returnSize = 0;
    for (int i=1; i<=numsSize; i++) {
        if (a[i] == 0) {
            ans[*returnSize] = i;
            *returnSize += 1;
        }
    }
    return ans;
}
