    int maxsum=INT_MIN;
    int sum=0;
    
    for(int i=0;i<n;i++)
    {
        sum+=nums[i];
        
        maxsum=max(maxsum , sum);
        if(sum<0)
        {
            sum=0;
        }
    }
    return maxsum;
}
int maxAbsoluteSum(vector<int>& nums) {
    int ans1=kadane(nums);
    int n=nums.size();
    
    for(int i=0;i<n;i++)
    {
        nums[i]=nums[i]*-1;
    }
    int ans2=kadane(nums);
    
    return max(ans1,ans2);
}
