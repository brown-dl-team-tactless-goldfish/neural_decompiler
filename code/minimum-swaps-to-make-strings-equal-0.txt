int minimumSwap(char * s1, char * s2){
    int n = strlen(s1) ;
    int xy = 0, yx = 0 ;
    for(int i = 0; i < n; i++){
        if(s1[i] == s2[i])
            continue ;
        if(s1[i] == 'x')
            xy++ ;
        else
            yx++ ;
    }
    if(xy%2 != yx %2)
        return -1 ;
    
    return xy/2 + yx/2 + (xy%2)*2 ;
}
