int deleteGreatestValue(vector<vector<int>>& grid) {
	for (int i = 0; i < grid.size(); i++)
		sort(grid[i].begin(), grid[i].end());
	int res = 0;
	for (int j = 0; j < grid[0].size(); j++) {
		int cur = INT_MIN;
		for (int i = 0; i < grid.size(); i++) {
			cur = max(cur, grid[i][j]);
		}
		res += cur;
	}
	return res;
}
