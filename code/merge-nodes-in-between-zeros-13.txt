struct ListNode* mergeNodes(struct ListNode* head){
    
    struct ListNode * temp = head;
    struct ListNode * nextnode;
    
    while(temp->next != NULL){
        
        // The nodes with value 0 will remain in existence
        // and contain the sum. Ex: if value of next node is nonzero
        // then its value will be summed to value in starting node
        // and then node with nonzero value will be deleted
        if(temp->next->val == 0){       
            if(temp->next->next == NULL){
                temp->next = NULL;
                free(temp->next);
            }
            else {
                temp = temp->next;
            }     
        }
        else {
            nextnode = temp->next->next;        // get link for future node
            temp->val = temp->val + (temp->next->val);// compute sum for current node and next node
            free(temp->next);       // delete next node
            temp->next = nextnode;  // set current node link to nextnode
        }   
    }
    return head;
}
