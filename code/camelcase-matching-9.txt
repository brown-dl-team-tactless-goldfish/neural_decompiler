class Solution {
public:
    vector<bool> camelMatch(vector<string>& queries, string pattern) {
        vector<bool> res;
        int pattern_idx = 0, success = 0;
        for(string S : queries){
            pattern_idx = 0;
            success = 1;
            for(char C : S){
                if(C == pattern[pattern_idx] && pattern_idx < pattern.size())
                    pattern_idx++;
                else if(isupper(C)){
                    success = 0;
                    break;
                }
            }
            if(pattern_idx == pattern.size() && success)
                res.push_back(1);
            else
                res.push_back(0);
        }
        return res;
    }
};
