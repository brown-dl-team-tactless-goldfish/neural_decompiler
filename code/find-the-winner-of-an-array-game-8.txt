class Solution {
public:
    int getWinner(vector<int>& arr, int k) {
        int i{0};
        for(int size{(int)arr.size()}; i < size;) {
            int j{i+1};
            while(j < size && arr[j] < arr[i]) ++j;
            if(j-i-(i==0) >= k) return arr[i];
            if(j == size) break;
            i = j;
        }
        return arr[i];
    }
};
