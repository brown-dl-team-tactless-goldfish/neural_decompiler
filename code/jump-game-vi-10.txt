class Solution {
public:
    int maxResult(vector<int>& nums, int k) {
        deque<pair<int, int> > q;
        for(int i = 0; i < nums.size(); ++i) {
            int num = nums[i];
            while(q.size() && i - q.front().second > k) q.pop_front();
            if(q.size())
                num += q.front().first;
            while(q.size() && q.back().first <= num) q.pop_back();
            q.push_back({num, i});
        }
        return q.back().first;
    }
};
