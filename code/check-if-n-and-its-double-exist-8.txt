#define SIZE 2001

bool checkIfExist(int* arr, int arrSize){
    int map[SIZE] = {0};
    int i, tmp;
    
    for (i = 0; i < arrSize; ++i) {
        tmp = arr[i];
        if (tmp <= 500 && tmp >= -500) {
            ++map[tmp*2 + 1000];
        }
    }
    
    for (i = 0; i < arrSize; ++i) {
        tmp = arr[i] + 1000;
        if ((map[tmp] && tmp != 1000) ||  map[1000] == 2)
            return true;
    }
    
    return false;
}
