public class Solution {
    public int MinSteps(string s, string t) {
        
        if(string.IsNullOrEmpty(s) || string.IsNullOrEmpty(t))
            throw new ArgumentException("Invalid Input");
        
        if(s == t)
            return 0;
        
        int[] freq = new int[26];
        foreach(var c in s)
            freq[c - 'a']++;
        foreach(var c in t)
            freq[c - 'a']--;
        
        int res = 0;
        foreach(var f in freq)
            res = f >= 0? res + f : res - f;
        
        return res;
    }
}
