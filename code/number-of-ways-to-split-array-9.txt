class Solution {
public:
    int waysToSplitArray(vector<int>& nums) {
        long long sum = 0;
        for(int i=0; i<nums.size(); i++)
            sum += nums[i];
			
        long long b = 0, ans = 0;
        for(int i=0; i<nums.size() - 1; i++){
            b += nums[i];
            long long a = sum - b;
            if(b >= a)
                ++ans;
        }
        return ans;
    }
};
