class Solution {
public:
    int hIndex(vector<int>& citations) {
        int n = citations.size();
        int l = 0, h = n-1, m = 0;
        
        if (n == 0) return 0;
        while (l <= h) {
            m = (l + h)/2;
            (citations[m] >= n-m) ?
                h = m - 1 :
                l = m + 1;
        }
        return (n-l);      
    }
};
