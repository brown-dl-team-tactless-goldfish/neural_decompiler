int minCost(string s, vector<int>& nums) {
    int n = nums.size();
    pair<int,int>p = {s[0],nums[0]};
    int c = 0;
    for(int i=1;i<n;i++){
        if(p.first==s[i]){
            if(p.second<nums[i]){
                c+=(p.second);
                p = {s[i],nums[i]};
            }else{
                c+=nums[i];
            }
        }else{
            p = {s[i],nums[i]};
        }
    }
    return c;
}
