
 public int MaxConsecutiveAnswers(string answerKey, int k) {
        Dictionary<char,int> dict= new Dictionary<char,int>();
        int ws=0,maxLen=0,maxRepeatingItem=0;
        for(int we=0;we<answerKey.Length;we++){
            if(!dict.ContainsKey(answerKey[we])){
                dict.Add(answerKey[we],0);
            }
            dict[answerKey[we]]++;
            maxRepeatingItem=Math.Max(maxRepeatingItem,dict[answerKey[we]]);
            if(we-ws+1-maxRepeatingItem > k){
                dict[answerKey[ws]]--;
                ws++;
            }
            maxLen=Math.Max(maxLen,(we-ws)+1);
        }
        
        return maxLen;
    }
