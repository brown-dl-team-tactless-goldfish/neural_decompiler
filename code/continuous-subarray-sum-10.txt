public class Solution {
	public bool CheckSubarraySum(int[] nums, int k) {

		var dict = new Dictionary<int,int>();
		var sum = 0;
		dict.Add(0,-1);

		for(int i=0; i<nums.Length; ++i){
			sum += nums[i];

			if(dict.TryGetValue(sum % k, out int x)){
				if(i-x>=2)return true;
			}

			dict.TryAdd(sum % k, i);

		}

		return false;

	}
}
