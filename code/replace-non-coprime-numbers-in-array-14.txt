class Solution {
public:
    vector<int> replaceNonCoprimes(vector<int>& nums) {
        
        vector<int> ans;
        for(int i = 0 ;i < nums.size() ; i++)
        {
            while(!ans.empty() and __gcd(ans.back(), nums[i]) > 1)
            {
                nums[i] = lcm(ans.back(), nums[i]);
                ans.pop_back();
            }
            ans.push_back(nums[i]);
        }
        return ans;
    }
};
