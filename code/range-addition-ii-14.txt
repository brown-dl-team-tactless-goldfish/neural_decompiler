public class Solution {
    public int MaxCount(int m, int n, int[][] ops) {
        
        if(ops == null || ops.Length == 0)
            return m * n;
        
        int minX = m, minY = n;
        foreach(var op in ops)
        {
            minX = Math.Min(minX, op[0]);
            minY = Math.Min(minY, op[1]);
        }
        
        return minX * minY;
    }
}
