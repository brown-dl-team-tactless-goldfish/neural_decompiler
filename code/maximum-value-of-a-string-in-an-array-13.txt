class Solution {
public:
    int maximumValue(vector<string>& strs) {
        int ans = 0;
        for (auto& s : strs){
            bool valid = true;
            for (auto& c : s) if (!isdigit(c)) valid = false;
            if (valid) ans = max(ans, stoi(s));
            else ans = max(ans, (int) s.size());
        }
        return ans;
    }
};
