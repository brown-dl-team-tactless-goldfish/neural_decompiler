int minSteps(char * s, char * t){
    int chars[123] = {0}; // ascii value of z is 122
    int i;
    int sum = 0;
    for (i = 0; s[i] != '\0'; i++)
        chars[s[i]] += 1;
    for (i = 0; t[i] != '\0'; i++)
        chars[t[i]] -= 1;
    for (i = 0; i < 123; i++ )
            sum = (chars[i] >= 0) ? sum + chars[i] : sum - chars[i];
    return sum;
}
