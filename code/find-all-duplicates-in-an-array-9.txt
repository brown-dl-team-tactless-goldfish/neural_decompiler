int* findDuplicates(int* nums, int numsSize, int* returnSize){
    int *a = calloc(1, sizeof(int [numsSize + 1])); // As per the question needs
    for (int i = 0 ; i < numsSize ; i++)
        a[nums[i]] = a[nums[i]] == nums[i] ? -1 : nums[i];
    for (int j = 1, k = 1 ; *returnSize = k - 1, j < numsSize + 1 ; j++)
        a[j] == -1 ? a[k++] = j : 0;
    return a + 1;
}
