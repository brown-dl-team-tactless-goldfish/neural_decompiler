int* distributeCandies(int candies, int num_people, int* returnSize){
    *returnSize=num_people;
    int *ret=calloc(num_people,sizeof(int)),idx=1,i=0;
    while(candies>0)
    {
        
        if(candies-idx>=0)
        {
            ret[i++]+=idx;
            candies-=idx;
        }
        else 
        {
            ret[i]+=candies;
            break;
        }
        idx++;
        i=i%num_people;
        
    }
    return ret;
}
