bool canPlaceFlowers(int* flowerbed, int flowerbedSize, int n){
    int count = 0;
    int last_one = -1;
    for (int i = 0; i < flowerbedSize; i ++)
    {
        if (flowerbed[i] == 0)
        {
            count ++;
        }
        else
        {
            n -= (count - (last_one != -1)) / 2;
            count = 0;
            last_one = i;
        }
    }
    
    n -= (count + (last_one == -1)) / 2;
    
    return (n <= 0);
}
