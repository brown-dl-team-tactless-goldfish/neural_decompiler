int balancedStringSplit(char * s){
    int counter=0;
    int count=0;
    for(int i=0; i<strlen(s); i++)
    {
        if(s[i]=='R')
        {
            ++counter;
        }
        else
        {
            --counter;
        }
        if(counter==0)
        {
            ++count;
        }
    }
    return count;
    
}
