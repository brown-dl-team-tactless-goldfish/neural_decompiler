    public int EdgeScore(int[] edges)
    {
        int n = edges.Length;
        long[] sum = new long[n];
        for (int i = 0; i < n; i++)
            sum[edges[i]] += i;

        long max = 0;
        int maxIndex = 0;
        for (int i = 0; i < n; i++)
        {
            if (sum[i]> max)
            {
                max = sum[i];
                maxIndex = i;
            }
        }

        return maxIndex;
    }
