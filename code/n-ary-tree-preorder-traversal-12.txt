void pOrder(struct Node * root, int * out, int * idx)
{
    if(!root) return;
    out[(*idx)++] = root->val;
    for(int i=0; i<root->numChildren; i++)
        pOrder(root->children[i], out, idx);
}
int* preorder(struct Node* root, int* returnSize) {
    int idx = 0;
    int * result = (int *)malloc(10000*sizeof(int));
    pOrder(root, result, &idx);
    *returnSize = idx;
    return result;
}
