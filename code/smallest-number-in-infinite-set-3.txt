class SmallestInfiniteSet {
public:
    multiset<int> ms ;
    SmallestInfiniteSet() {
        for(int i  = 1 ; i <= 1000 ; ++i) ms.insert(i);
    }
    
    int popSmallest() {
        int res = *begin(ms) ;
        ms.erase(res) ;
        return res ;
    }
    
    void addBack(int num) {
        ms.insert(num) ;
    }
};
