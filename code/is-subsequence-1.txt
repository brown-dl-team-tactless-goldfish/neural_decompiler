bool isSubsequence(char * s, char * t){
    int s_idx=0;
    int t_idx=0;
    
    while (s[s_idx]!='\0') {
        while ((t[t_idx]!='\0') & (t[t_idx]!=s[s_idx])) 
            t_idx++;
        if (t[t_idx]=='\0')
            return false;
        t_idx++;
        s_idx++;
    }
    
    return true;
    
}
