int minimumAverageDifference(int* nums, int numsSize){
    long sum=0,left=0;
    for(int i=0;i<numsSize;i++)
        sum+=nums[i];
    int min=INT_MAX,idx=-1,a=0;
    for(int i=0;i<numsSize;i++){
        sum -= nums[i];
        left += nums[i];
        if(i<numsSize-1)
            a = abs(left/(i+1)-sum/(numsSize-i-1));
        else
            a = left/(i+1);
        if(a<min){
            min=a;
            idx=i;
        }
    }
    return idx;
}
