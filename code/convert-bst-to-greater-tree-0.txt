void process(struct TreeNode* root, int* sum){
    if(root == NULL)
        return;
    process(root->right, sum);
    *sum = root->val  + *sum;
    root->val = *sum;
    process(root->left, sum);
}

struct TreeNode* convertBST(struct TreeNode* root){
    int sum = 0;
    
    process(root, &sum);
    
    return root;

}
