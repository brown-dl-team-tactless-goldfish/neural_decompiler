class Solution {
    bool static comp(string a,string b){
    if(a==b) return false;
    if(a.size()>b.size()) return true;
    else if(a.size()<b.size()) return false;
    else{
        if(a<b) 
        return false;
        else 
        return true;
    }
}
public:
    string kthLargestNumber(vector<string>& nums, int k) {
    sort(nums.begin(),nums.end(),comp);
        return nums[k-1];
    }
};
