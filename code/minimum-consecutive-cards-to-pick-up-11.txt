class Solution {
public:
    int minimumCardPickup(vector<int>& v) {
        int n=v.size();
        if(n<2) return -1;
        else if(n==2) return v[0]==v[1]?2:-1;
        unordered_map<int,int> m;
        int ans=INT_MAX;
        for(int i=0; i<n; i++) {
            if(m.find(v[i])!=m.end()) ans=min(i-m[v[i]]+1,ans);
            m[v[i]]=i;
        }
        return ans==INT_MAX?-1:ans;
    }
};
