select c.name, 
CASE
    WHEN c.travelled_distance is NULL THEN 0
    ELSE c.travelled_distance
END as travelled_distance
from(
select a.name as name, b.travelled_distance
from Users a
left outer join 
(select user_id, sum(distance) as travelled_distance
from Rides group by user_id) b
on a.id = b.user_id
) c order by c.travelled_distance desc, c.name
