int getMaximumGenerated(int n){
    if(n==0)
        return 0;
    int *arr=(int *)calloc((n+1),sizeof(int));
    arr[0]=0;
    arr[1]=1;
    for(int i=2;i<=n;i++)
    {
       if(i%2==0)
       {
           arr[i]=arr[i/2];
       }
        else
        {
            arr[i]=arr[(i-1)/2]+arr[1+(i-1)/2];
        }
    }
    int max=0;
    for(int i=1;i<=n;i++)
    {
        if(arr[i]>max)
        {
            max=arr[i];
        }
    }
    free(arr);
    return max;
}
