int sumOfUnique(int* nums, int numsSize){
    int* hashCnt = (int*)calloc(101, sizeof(int));
    int ans = 0;
    for(int i = 0; i < numsSize; i++){
        hashCnt[nums[i]]++;

        if(hashCnt[nums[i]] == 1)
            ans += nums[i];
        else if(hashCnt[nums[i]] == 2)
            ans -= nums[i];  
    }
    return ans;
}
