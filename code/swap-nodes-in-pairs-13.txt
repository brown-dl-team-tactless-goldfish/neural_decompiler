struct ListNode* swapPairs(struct ListNode* head){
    
    
    if(head == NULL || head->next == NULL)
        return head;
    
    struct ListNode * curr;
    
    curr = head;
    head = curr->next;
    
    struct ListNode *ptr1, *next;
    
    while (curr != NULL && curr->next != NULL) {
        
        ptr1 = curr->next;
        next = ptr1->next;
        ptr1->next = curr;
        
        if(next)
            curr->next = next->next ? next->next : next;
        else
            curr->next = NULL;
        
        curr = next;
        
        
    }

    return head;
}
