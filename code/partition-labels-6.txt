int* partitionLabels(char * S, int* returnSize){
    int arr[26]={0},i=0;
    int *ret=malloc(sizeof(int)*100);
    int len=strlen(S),range=0,idx=0;
    for(int i=0;i<len;i++)
        arr[S[i]-'a']=i;
    while(i<len)
    {
        range=arr[S[i]-'a'];
        int start=i;
        while(i<=range)
        {
            if(arr[S[i]-'a']>range)
                range=arr[S[i]-'a'];
            i++;
        }
        ret[idx++]=i-start;
    }
    *returnSize=idx;
    return ret;
}
