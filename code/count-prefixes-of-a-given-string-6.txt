

int countPrefixes(char ** words, int wordsSize, char * s){
    int count =0 ;
    bool add ;
    for(int i=0 ;i<wordsSize;i++){
            
       add = true;
        char* x = words[i]; 
        for(int j =0;x[j];j++)
           if(!s[j] || s[j] != x[j] ){
                add = false;
                break;
            }
        
            if(add) count++;
        }
        return count;
        
}
