int* findErrorNums(int* nums, int numsSize, int* returnSize){
    int *result = (int*)malloc(sizeof(int)*3);
    *returnSize = 0;
    for(int i=0; i< numsSize; i++){
        int index = abs(nums[i]) - 1;   //Since we have nums[] array values from 1 to n, we can derive index from the array numbers.
										//and 'index' variable shall have an index of nums[] array such that it points to unique value in nums[]
										//we mark that visited index with nums[index]*(-1) in the else part below
        if(nums[index] < 0){            //If we visited that index twice, the value shall be less than 0, and hence a duplicate
            result[(*returnSize)++] = abs(nums[i]); 
        } else {
            nums[index] = nums[index] * (-1);
        }
    }
    for(int i=0; i<numsSize; i++){
        if(nums[i] > 0){                    //If we have not visited that index, it means that value is not multiplied by -1, 
		                                    //hence greater than 0 and is the missing number in the array nums[]
            result[(*returnSize)++] = i+1;
        }
    }
    return result;
}
