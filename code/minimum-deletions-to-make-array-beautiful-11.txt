int minDeletion(vector<int>& nums) {
        int c = 0;
        for(int i = 0 ; i < nums.size() - 1; i++){
		   //Current index after deleting
		    int index = i - c;
			//Check whether it is even
            if( index % 2 == 0){
			   //If adjacent is similar
                if(nums[i] == nums[i+1]){
				    c++;
				}
			}
		}
        
		int remaining_elements = nums.size() - c;
        if( remaining_elements % 2 == 0){
             return c;
        }
        return c+1;
    }
