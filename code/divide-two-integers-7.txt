
//Maximum of quotient such that
// dividend - (quotient * divisor) >= 0
int divide(long long int dividend,long long int divisor){
    int sign=1;
    if(dividend<0)
    {
        sign*=-1;
        dividend=-dividend;
    }
    if(divisor<0)
    {
        sign*=-1;
        divisor=-divisor;
    }
    long long int lo=0,hi=dividend,mid;
    while(lo<hi)
    {
        mid=hi-(hi-lo)/2;
        if(dividend-mid*divisor>=0)
            lo=mid;
        else
            hi=mid-1;
    }
    return sign*lo>INT_MAX?INT_MAX:sign*lo;
}
