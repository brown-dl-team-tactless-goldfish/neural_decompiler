class Solution:
    def minOperations(self, nums1: List[int], nums2: List[int], k: int) -> int:
        ans = total = 0 
        for x, y in zip(nums1, nums2): 
            diff = x-y
            if k == 0 and diff or k and abs(diff) % k: return -1 
            if k: ans += abs(diff)//k
            total += diff
        return ans//2 if total == 0 else -1
