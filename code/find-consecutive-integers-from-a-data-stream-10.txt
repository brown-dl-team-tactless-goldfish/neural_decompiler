class DataStream {
public:
    DataStream(int value, int k) {
        _counter = 0;
        _k = k;
        _value = value;
    }
    
    bool consec(int num) {
        if(num != _value)
            _counter = 0;
        else
            _counter++;
        return _counter >= _k;
    }
private:
    int _k;
    int _counter;
    int _value;
};

 * Your DataStream object will be instantiated and called as such:
 * DataStream* obj = new DataStream(value, k);
 * bool param_1 = obj->consec(num);
 */
