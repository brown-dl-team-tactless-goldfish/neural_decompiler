    long long interchangeableRectangles(vector<vector<int>>&nums) {
        map<double,int>mp;
        long long int count=0;
        for(int i=0;i<nums.size();i++)
        {
            double x=(double)nums[i][0]/(double)nums[i][1];
            if(mp.find(x)!=mp.end())
            {
                count+=mp[x];
            }
            mp[x]++;
        }
        return count;
    }
};
