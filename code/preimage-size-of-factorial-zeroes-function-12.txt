int preimageSizeFZF(int K) {
        
        int zeroesnow=0;   //for counting number of zeroes from 0
        int cnt=0;          //counting the number
        int I=0;            // value for 0 t0 10e9
        
        while(zeroesnow<=K)  //Exit when number of zeroes become greater than K
        {
            int a=0;
            int i=I;
            while(i>0)       //function for counting total number of zeroes for every "i"
            {
                a+=i/5;
                i/=5;
            }
            if(a==K)          
                ++cnt;
            zeroesnow=a;     // update value of zeroesnow
            I++;             //Increment I
        }
        return cnt;
    }
