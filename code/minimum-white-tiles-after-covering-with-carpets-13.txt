class Solution {
public:
    int dp[1001][1001];
    int get(int i,int k,string &s,int l)
    {
        if(i>=s.size())
            return 0;
        if(dp[i][k]!=-1)
            return dp[i][k];
        int res=s[i]-'0'+get(i+1,k,s,l);
        if(s[i]=='1' && k>0)
            res=min(res,get(i+l,k-1,s,l));
        return dp[i][k]=res;
    }
    int minimumWhiteTiles(string &s, int k, int l) 
    {
        memset(dp,-1,sizeof(dp));
        return get(0,k,s,l);
    }
};
