int leastBricks(vector<vector<int>>& wall) {
        unordered_map<int,int> crease;
     
        for(auto& row: wall){
            for(int c = 0, pos = 0; c < row.size()-1; ++c){
                pos += row[c];
                ++crease[pos];
            }
        }
        
        int res = wall.size(), h = wall.size();
        for(auto& [pos, gapCnt]: crease){
            res = min(res, h - gapCnt);
        }
        return res;
    }
