class Solution {
public:
    int addRungs(vector<int>& nums, int dist) 
    {
        nums.insert(nums.begin(),0);
        int n=nums.size();
        int count=0;
        for(int i=1;i<n;i++)
        {
            if(nums[i]-nums[i-1]>dist)
            {
                int sum=nums[i]-nums[i-1]-1;
                count+=sum/dist;
            }
        }
        return count;
    }
};
