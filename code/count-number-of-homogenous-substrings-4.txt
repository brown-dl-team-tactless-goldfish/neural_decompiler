int countHomogenous(char* s){
        int len=strlen(s);
        int suffix[len];
        suffix[0]=1;
        for(int i=1;i<len;i++){
                if(s[i]==s[i-1])suffix[i]=suffix[i-1]+1;
                else suffix[i]=1;
        }
        int count=0;
        for(int i=0;i<len;i++){
                count=(count%(1000000007))+suffix[i];
        }
        return count%(1000000007);
}
