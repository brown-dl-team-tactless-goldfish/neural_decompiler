class Solution {
public:
    int subsetXORSum(vector<int>& nums) {
        int Xor=0;
        return fun(nums,0,Xor);
    }
    
    int fun(vector<int>&nums,int index,int Xor)
    {
        if(index==nums.size())
            return Xor;
        
        int result=0;
    // Choose
        // Do pre-calculation
          Xor^=nums[index];
        result+=fun(nums,index+1,Xor);
        
        // Do post-calculation
         Xor^=nums[index];
        
    // Not Choose
         result+=fun(nums,index+1,Xor);
    return result;
    }
};

Time Complexity: O(2^n)
Space Complexity: O(1)
