    public bool CheckDistances(string s, int[] distance) 
    {
        int[] lastIndex = new int[26];
        Array.Fill(lastIndex, -1);
        for (int i = 0; i < s.Length; i++)
        {
            if (lastIndex[s[i] - 'a'] == -1)
                lastIndex[s[i] - 'a'] = i;
            else if (i - lastIndex[s[i] - 'a'] - 1 != distance[s[i] - 'a'])
                return false;
        }
        return true;
    }
