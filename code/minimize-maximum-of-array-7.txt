class Solution {
public:
    int minimizeArrayValue(vector<int>& nums) {
        long long sum = 0, res = INT_MIN;
        for (int i = 0; i < nums.size(); i++) {
            sum += nums[i];
            long long temp = ceil((sum * 1.0) / (i + 1));
            res = max(res, temp);
        }
        return res;
    }
};
