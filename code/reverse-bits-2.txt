uint32_t
reverseBits(uint32_t n)
{

	return n = n << 16               | n >> 16,
	       n = n << 8 & 037700177400 | n >> 8 & 000077600377,
	       n = n << 4 & 036074170360 | n >> 4 & 001703607417,
	       n = n << 2 & 031463146314 | n >> 2 & 006314631463,
	           n << 1 & 025252525252 | n >> 1 & 012525252525;
}
