int* findErrorNums(int* nums, int numsSize, int* returnSize){
    int * result = (int *)malloc(2*sizeof(int));
    int temp[10001] = {0};
    for(int i=0; i<numsSize; i++)
        temp[nums[i]]++;
    for(int i=1; i<=numsSize; i++)
    {
        if(temp[i] == 2) result[0] = i;
        if(temp[i] == 0) result[1] = i;
    }
    *returnSize = 2;
    return result;
}
