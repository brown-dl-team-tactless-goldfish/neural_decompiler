class Solution {
public:
    int minImpossibleOR(vector<int>& nums) {
        unordered_set<long long int>s;
        for(auto &i: nums){
            s.insert(i);
        }
        long long int x = 1;
        while(x<=1000000000L&&s.count(x)){
            x <<= 1;
        }
        return x;
    }
};
