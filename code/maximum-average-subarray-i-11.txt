double findMaxAverage(int* nums, int numsSize, int k){
    int i = 0; 
    int j = 0; 
    double avg = 0.0; 
    double max = -10000.1; 
    
    
    while(j< numsSize){
        avg+=nums[j];
        if(j-i+1 == k){
            max = (avg/k>max)?avg/k:max;
            avg-=nums[i];
            i++;
        }
        j++;
    }
    return max;
}
