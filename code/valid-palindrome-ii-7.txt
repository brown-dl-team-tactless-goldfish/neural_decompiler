#include <stdbool.h> 

bool isPalindrome(char *s, int start, int end) {
    while (start < end) {
        if (s[start] != s[end]) {
            return false;
        }
        start++;
        end--;
    }
    return true;
}

bool validPalindrome(char *s) {
    int start = 0;
    int end = strlen(s) - 1;
    while (start < end) {
        if (s[start] != s[end]) {
           
            bool deleteStart = isPalindrome(s, start + 1, end);
            bool deleteEnd = isPalindrome(s, start, end - 1);
            if (deleteStart || deleteEnd) {
                return true;
            } else {
                return false;
            }
        }
        start++;
        end--;
    }
    return true; 
}
