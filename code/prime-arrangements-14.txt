    if(n<=1) return 1;
    long long p=(long long )n*fact(n-1);
    p=p%1000000007;
    return p;
}

int numPrimeArrangements(int n) {
    prime.resize(n+1,1);
    SieveOfEratosthenes(n);
     
    int primenumber=0;
    for (int p = 2; p <= n; p++)
    if (prime[p])
        primenumber++;
    
    long long ans=(long long )fact(primenumber)*fact(n-primenumber);
    
    return ans%1000000007;
}
