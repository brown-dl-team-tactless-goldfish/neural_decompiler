int minDeletionSize(char ** strs, int strsSize){
    int result=0, asc;

    for(int i=0; strs[0][i]!='\0'; i++){
        asc = strs[0][i]-'a';
        for(int j=1; j<strsSize; j++){
            if((strs[j][i]-'a')<asc){
                result++; break;
            } else {
                asc = strs[j][i]-'a';
            }
        }
    }

    return result;
}
