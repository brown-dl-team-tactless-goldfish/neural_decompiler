class Solution {
public:
    int subsetXORSum(vector<int>& nums) {
        int sum = 0;
        const int size = nums.size();
        const int code = (1 << size) - 1;
        for (int i = 1; i <= code; ++i) {
            int sub_sum = 0;
            int c = i;
            for (int j = 0; j < size; ++j) {
                if (c & 1) sub_sum ^= nums[j];
                c >>= 1;
            }
            sum += sub_sum;
        }
        return sum;
    }
};
