int cmp(void *a,void *b){
    return *(int*)b - *(int*)a;
}
int hIndex(int* citations, int citationsSize){
    qsort(citations,citationsSize,sizeof(citations[0]),cmp);
    for(int i=citationsSize-1;i>=0;i--){
        if(citations[i]<citationsSize)
            citationsSize--;
        else
            return citationsSize;
    }
    return citationsSize;
}
