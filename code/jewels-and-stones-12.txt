int numJewelsInStones(char * jewels, char * stones){
    int list [256];
    int count = 0;
    for(int i = 0; i < 256; i++) // set list
    {
        list[i] = -1;
    }
    for(int i = 0; i < strlen(jewels); i++)
    {
        int val = jewels[i];
        list[val] = val;
    }    
    for(int i = 0; i < strlen(stones); i++)
    {
        if(list[stones[i]] == stones[i])
        {
            count = count + 1;
        }
    }
    return count;
}
