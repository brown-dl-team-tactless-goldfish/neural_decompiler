#include <string.h>
#define NUL '\0'
bool isSubsequence(char * s, char * t){
    
    if (!(*s && *t))
    {
        return *s == NUL;
    }
    
    unsigned int s_len = strlen(s); 
    
     // For each char of s
     for (int i = 0; i < s_len; i++)
        {
            while (*t && *t != s[i])
               t++;
               
            if (*t == NUL)
                return false;
         
            t++;
        }
        
    return true;
}
