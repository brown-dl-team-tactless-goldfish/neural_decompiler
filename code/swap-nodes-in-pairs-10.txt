/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     struct ListNode *next;
 * };
 */


struct ListNode* swapPairs(struct ListNode* head){
    struct ListNode* temp;
    if (head==NULL || head -> next == NULL){
        return head;
    } else {
        temp = head -> next;
        head -> next = temp -> next;
        temp -> next = head;
        head = temp;
        head -> next-> next = swapPairs(head->next->next);
        return head;
    }
}
