int longestOnes(int* s, int n, int k){
    int i,count,max=0;
    int *record=calloc(k+1,4);
    for(i=0,count=0;i<n&&count<k;i++){
        if(!s[i]){
            record[++count]=i+1;
        }
    }
    for(count=0,max=0;i<n;i++){
        if(!s[i]){
            max=i-record[count]>max?i-record[count]:max;
            record[count++]=i+1;
            if(count>k)count=0;
        }
    }
    max=i-record[count]>max?i-record[count]:max;
    return max;
}
