  def treeQueries(self, root, queries, ans = {}) -> List[int]:
     ht = lru_cache(maxsize = 20000)(lambda r: 1 + max(ht(r.left), ht(r.right)) if r else 0)
     def dfs(r, depth, mx):
         if not r: return
         ans[r.val] = mx
         dfs(r.left, depth + 1, max(mx, depth + ht(r.right)))
         dfs(r.right, depth + 1, max(mx, depth + ht(r.left)))
     dfs(root, 0, 0)
     return [ans[v] for v in queries]
