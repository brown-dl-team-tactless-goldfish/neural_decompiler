int kthSmallest(int** matrix, int matrixSize, int* matrixColSize, int k){
    
    int* loc= calloc(matrixSize, sizeof(int));
    int i;
    
    int smallest;
    int smallloc = 0;
    while(k){
        smallest = INT_MAX;
        smallloc = 0;
        for(i=0;i<matrixSize;i++){
            if(loc[i] >= matrixSize)
                continue;
            if(matrix[i][loc[i]]<smallest){
                smallest = matrix[i][loc[i]];
                smallloc = i;
            }
        }
        
        // update smallloc
        loc[smallloc]++;
        k--;
    }
    free(loc);
    return smallest;
}
