class Solution {
public:
    bool sumOfNumberAndReverse(int num) {
        
        for(int i = 0; i <= num; i++){
            int a = i;
            int b = i;
            string s = to_string(b);
            reverse(s.begin(), s.end());
            b = stoi(s);
            if(a + b == num) return true;
        }
        return false;
    }
};
