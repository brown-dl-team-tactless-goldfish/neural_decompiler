    int numWays(int steps, int arrLen) {
        const int n = min(steps/2+1, arrLen);
        vector<long long> pre(n, 0), cur=pre;
        pre[0] = 1, cur[0] = 1;
        const long long mod = 1e9+7;
        while(steps--){
            for(int i=0; i<n;i++){
                if(i>0) cur[i] += pre[i-1];
                if(i<n-1) cur[i] += pre[i+1];
                cur[i] %= mod;
            }
            pre = cur;
        }
        return pre[0];
    }
