int cmp(const void *a, const void *b)
{
    return *(int *)b - *(int *)a;
}

int numRescueBoats(int* people, int peopleSize, int limit){
    qsort(people, peopleSize, sizeof(int), cmp);
    int left = 0, right = peopleSize - 1, count = 0;
    while(left <= right)
    {
        if(people[left] + people[right] <= limit)
            right--;
        left++;
        count++;
    }
    return count;
}
