int cmpfunc (const void * a, const void * b) {
   return ( *(int*)a - *(int*)b );
}

int minMovesToSeat(int* seats, int seatsSize, int* students, int studentsSize){
    qsort(seats, seatsSize, sizeof(int), cmpfunc);
    qsort(students, studentsSize, sizeof(int), cmpfunc);
    
    int sum = 0;
    
    for(int i=0;i<seatsSize;i++){
        int diff = students[i] - seats[i];
        sum+= diff<0?diff*-1:diff;
    }
    return sum;
}

/*
1.Sort both students array and seats array.
2.Get the absolute diff between corrseponding seats and students array elements and add it to sum.
3.Return sum
