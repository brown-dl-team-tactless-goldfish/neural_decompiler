 public bool MakeEqual(string[] words) {
        int[] arr= new int[128];
        StringBuilder sb= new StringBuilder();
        foreach(var w in words){
            sb.Append(w);
        }
        for(int i=0;i<sb.Length;i++){
            arr[sb[i]]++;
        }
        int c= words.Length;
        foreach(var i in arr){
            if(i% c != 0) return false;
        }
        
        return true;
    }
