class Solution {
public:
    int maximizeGreatness(vector<int>& nums) {
        int n = nums.size(), ans = 0, last = 0;
        sort(nums.begin(), nums.end());

        for(int i=0; i<n; i++) {
            int idx = lower_bound(nums.begin() + last + 1, nums.end(), nums[i] + 1) - nums.begin();
            if(idx < n) {
                last = idx;
                ans++;
            }
        }
        return ans;
    }
};
