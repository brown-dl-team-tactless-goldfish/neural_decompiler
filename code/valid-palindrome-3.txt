

bool isPalindrome(char * s){
if (s == NULL || !strcmp(s,"")) {return true;}

    

    int i,  k = 0;

    int count = 0;

    int len = strlen(s);

    char str[len];

    

    for (i = 0; i < len; i++) {

        if(isalnum(s[i])) {

            str[k] = tolower(s[i]);

            k++;

        }

    }

    for (i = 0; i < k; i++) 
    {

       

        

        if ((str[i] != str[k-1-i]) || (i >= (k-1-i))) 
        {

            break;

        }

        count++;

    }

    

    if (count ==(k/2)) 
    { 
        return true;
    }

    
else
    return false;
}
