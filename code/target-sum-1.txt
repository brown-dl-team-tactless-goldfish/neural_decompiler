

int TargetSumWays(int* nums, int numsSize, int S, int ind, int curr_sum, int ways) {
    //Choose '+' during forward recursion and then '-'
    if (ind == numsSize) {
        if (curr_sum == S)
            return ways +1;
        else
            return ways;
    }
    else {
        int new_sum = curr_sum + nums[ind];
        //printf("Current sum is %d and current no. of ways is %d\n",new_sum,ways);
        int w = TargetSumWays(nums,numsSize,S,ind+1,new_sum,ways);
        new_sum = curr_sum - nums[ind];
        //printf("XXCurrent sum is %d and current no. of ways is %d\n",new_sum,w);
        w = TargetSumWays(nums,numsSize,S,ind+1,new_sum,w);
        return w;
    }
}

int findTargetSumWays(int* nums, int numsSize, int S){

    return TargetSumWays(nums,numsSize,S,0,0,0);
}
