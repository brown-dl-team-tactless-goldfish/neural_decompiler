class Solution {
public:
    long long countSubarrays(vector<int>& nums, long long k) {
        long long ans=0,sm=0,l=0,r=0,n=size(nums),len=0;
        for(;r<n;++r){
            sm+=nums[r];
            ++len;
            while(len*sm>=k){
                sm-=nums[l++];
                --len;
            }       
            ans+=len;
        }
        return ans;
    }
};
