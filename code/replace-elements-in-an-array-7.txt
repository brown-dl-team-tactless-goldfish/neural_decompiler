class Solution {
public:
    vector<int> arrayChange(vector<int>& nums, vector<vector<int>>& operations) {
        int n = nums.size();
        vector<int> idx(1e6 + 1, -1), ans(nums.begin(), nums.end());
        for (int i = 0; i < n; i++)
            idx[nums[i]] = i;
        for (auto& op : operations)
        {
            int i = idx[op[0]];
            idx[op[0]] = -1;
            idx[op[1]] = i;
            ans[i] = op[1];
        }
        return ans;
    }
};
