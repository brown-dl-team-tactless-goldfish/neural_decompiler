class Solution {
public:
    int halveArray(vector<int>& nums) {
        priority_queue<double>pq(nums.begin(),nums.end());
        long double sum=0,cs=0;
        for(auto&i:nums)sum+=i;
        cs=sum/2.0;
        int c=0;
        while(sum>cs && !pq.empty()){
            double x=pq.top()/2.0;
            sum-=x;
            pq.pop();
            pq.push(x);
            c++;
        }
        return c;
    }
};
