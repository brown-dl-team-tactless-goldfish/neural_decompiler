inline void pi(char *s, char *a, char *b){
    char *p = strchr(s, '+'), *i = strchr(p + 1, 'i');
    *a = atoi((*p = 0, s));    
    *b = atoi((*i = 0, p + 1));
}

char * complexNumberMultiply(char * a, char * b){
    char x, y, l, m, *p = calloc(1, 100);
    pi(a, &x, &y);
    pi(b, &l, &m);
    snprintf(p, 100, "%d+%di", x * l + -(y * m), x * m + y * l);
    return p;
}
