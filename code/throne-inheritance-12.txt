       public class ThroneInheritance 
        {
            Dictionary<string,IList<string>> personMap;
            HashSet<string> deadSet;
            string kingName;

            public ThroneInheritance(string kingName) {
                this.kingName = kingName;
                this.personMap = new Dictionary<string, IList<string>>();
                this.deadSet = new HashSet<string>();
                personMap.Add(kingName,new List<string>());
            }
            
            public void Birth(string parentName, string childName) {
                personMap[parentName].Add(childName);
                personMap.Add(childName, new List<string>());
            }
            
            public void Death(string name) {
                deadSet.Add(name);
            }
            
            private void dfs(IList<string> res, string name)
            {
                if(!deadSet.Contains(name))
                {
                    res.Add(name);
                }
                foreach(string child in personMap[name])
                {
                    dfs(res,child);
                }
            }
        }
