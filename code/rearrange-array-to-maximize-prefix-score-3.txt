public class Solution {
    public int MaxScore(int[] nums) {
        Array.Sort(nums, (a, b) => b.CompareTo(a));

        long current = nums[0];  
        int result = current > 0 ? 1 : 0;
            
        for (int i = 1; i < nums.Length; ++i)
            if ((current += nums[i]) > 0)
                result += 1;

        return result;
    }
}
