void calculate(int* nums, int numsSize, int i, int sum, int S, int* count){
    if (i == numsSize) {
        if (sum == S) {
            (*count)++;
        }
    } else {
        calculate(nums, numsSize, i+1, sum + nums[i], S, count);
        calculate(nums, numsSize, i+1, sum - nums[i], S, count);
    }
}
int findTargetSumWays(int* nums, int numsSize, int S){
    int count = 0;
    calculate(nums, numsSize, 0, 0, S, &count);
    return count;
}
