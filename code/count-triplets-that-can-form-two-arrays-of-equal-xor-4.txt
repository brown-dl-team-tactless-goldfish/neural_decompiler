int countTriplets(vector<int>& nums) {
    int n = nums.size();
    int c = 0;
    vector<int>left(n);
    left[0] = nums[0];
    for(int i=1;i<n;i++){
        left[i] = (left[i-1]^nums[i]);
    }
    for(int i=0;i<n;i++){
        int x = nums[i];
        for(int j=i+1;j<n;j++){
            if(x==(left[j]^left[i])){
                c+=(j-i);
            }
        }
    }
    return c;
}
