
int maxLengthBetweenEqualCharacters(char * s){
    int max=-1;
for(int i=0;i<strlen(s)-1;i++){
    int count=0;
    for(int j=i+1;j<strlen(s);j++){
        //if(s[i]!=s[j]){
            count++;
       // }
        if(count>max && s[i]==s[j]){
            max=count-1;
        }
    }
}
    return max;
}
