int numberOfBeams(char ** bank, int bankSize){
	int tmp[bankSize];
	memset(tmp, 0, sizeof(tmp));
	for(int i = 0 ; i < bankSize; i++)
	{
		while(*bank[i]){
			if(*bank[i] == '1'){
				tmp[i]++;
			}
			bank[i]++;
		}
	}
	int ans = 0;
	for(int i = 0; i < bankSize; i++)
	{
		if(tmp[i] == 0) continue;
		for(int j = i + 1; j < bankSize; j++){
			if(tmp[j] != 0){
				ans += tmp[i] * tmp[j];
				break;
			}
		}
	}
	return ans;
}
