class Solution {
public:
    int maxEqualRowsAfterFlips(vector<vector<int>>& matrix) {
        map<vector<int>, int> m;
        for(auto& v: matrix){
            ++m[flipToMakeAllSame(v, 0)];
            ++m[flipToMakeAllSame(v, 1)];
        }
        int res = 0;
        for(auto& [v, count] : m) res = max(res, count);
        return res;
    }
    
    vector<int> flipToMakeAllSame(const vector<int>& v, int val){
        vector<int> res;
        for(auto x:v)  res.push_back(x == val ? 0 : 1);
        return res;
    }
};
