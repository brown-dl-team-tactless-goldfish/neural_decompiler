int triangularSum(int* nums, int numsSize){
    if(numsSize==1)
        return nums[0]%10;
    if(numsSize==2)
        return (nums[0]+nums[1])%10;
    for(int i=1;i<numsSize;i++)
    {       
        for(int j=0;j<numsSize-i;j++)                   
            nums[j]=(nums[j]+nums[j+1])%10;        
    }
    
    return nums[0]; 
    
}
