int minTimeToVisitAllPoints(int** points, int pointsSize, int* pointsColSize){
    int time = 0,  index;
    for (int index = 0; index < pointsSize-1; index++)
        time += abs(points[index+1][0]-points[index][0]) > abs(points[index+1][1]-points[index][1]) ? abs(points[index+1][0]-points[index][0]) : abs(points[index+1][1]-points[index][1]);
                // absolute value in X-direction            // absolute value in Y-direction            // absolute value in X-direction            // absolute value in Y-direction
    return time;    
}
