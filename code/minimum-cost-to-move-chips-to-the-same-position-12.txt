int minCostToMoveChips(int* position, int positionSize){

    int i,j;
    
    int sum=1000000;
    
    for(i=0;i<positionSize;i++)
    {
        int sun=0;
        for(j=0;j<positionSize;j++)
        {
            if((abs(position[i]-position[j])%2)!=0)
            {
                sun=sun+1;
            }
        }
        if(sun<sum)
        {
            sum=sun;
        }
    }
    
    return sum;
    
}
