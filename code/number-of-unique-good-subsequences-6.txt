
int numberOfUniqueGoodSubsequences(string binary) {
        unsigned long long dp[2] = {0};
        reverse(binary.begin(), binary.end());
        bool zero = false;
        const long long mod = 1e9 + 7;
        for (char &c: binary) {
            dp[c - '0'] = dp[0] + dp[1] + 1LL;
            dp[c - '0'] %= mod;
            zero = zero || c == '0';
        }
        return (dp[1] + zero) % mod;
    }
