class Solution {
public:
   unordered_map<int, int> dp;
int minDays(int n) {
    if (n <= 1)
        return n;
    if (dp.count(n) != 0) return dp[n];
    
      int a=INT_MAX;
      int b=INT_MAX;
      int c=INT_MAX;
    
  
         if(n%2==0) b=1 +minDays(n/2);
         if(n%3==0) c=1 +minDays(n/3);
		 
		 /*This is tricky*/
    
    if((n%2!=0)||(n%3!=0)) a=1+minDays(n-1); 
         
    return dp[n]=  min(min(b,c),a)  ;
}
};
