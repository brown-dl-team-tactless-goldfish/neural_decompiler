class Solution {
public:
    int mostFrequent(vector<int>& nums, int key) {
        map<int,int> mp;
        for(int i = 0; i<nums.size();++i){
            if(nums[i]==key && i+1<nums.size()) mp[nums[i+1]]++; 
        }
        int re = 0,c=0;
        for(auto &q:mp){
            if(q.second>c){
                c = q.second;
                re = q.first;
            }
        }
        return re;
    }
};
