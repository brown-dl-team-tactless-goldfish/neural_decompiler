class Solution {
public:
    int numSub(string s) {
        
        int n = s.length();
        int ans  = 0, c = 0;
        int MOD = 1000000007;
        
        for (int i = 0; i < n; i++) {
            if (s[i] == '1')
                c++;
            else
                c = 0;
            ans = (ans + c)%MOD;
        }
        
        return ans%MOD;
    }
};
