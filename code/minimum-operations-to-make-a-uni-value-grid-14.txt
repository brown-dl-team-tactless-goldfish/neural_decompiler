int minOperations(vector<vector<int>>& grid, int x) {
    vector<int> v;
    for(int i=0; i<grid.size(); i++){
        for(int j=0; j<grid[0].size(); j++){
            v.push_back(grid[i][j]);
        }
    }
    
    sort(v.begin(),v.end());
    for(int i=1; i<v.size(); i++){
        if((v[i]-v[i-1])%x!=0) return -1;
    }
    int m=v[(v.size()-1)/2];
    int res=0;
    for(int i=0; i<v.size(); i++){
        res+=abs(m-v[i])/x;
    }
    return res;
}
