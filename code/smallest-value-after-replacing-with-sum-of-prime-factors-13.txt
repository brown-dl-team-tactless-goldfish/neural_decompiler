class Solution {
public:
    void calculate(int &num, int res) {
        int div = 2;
        while(num > 1) {
            if(num % div == 0) {
                res += div;
                num /= div;
            }
            else div += 1;
        }
        num = res;
    }
    int smallestValue(int &n) {
        int res = 0;
        int cntr = 1e3;
        while(cntr--) {
            calculate(n, res);
        }
        return n;
    }
};
