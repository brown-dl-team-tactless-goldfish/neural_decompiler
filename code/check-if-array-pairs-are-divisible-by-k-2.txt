bool canArrange(int* arr, int arrSize, int k){
    int* check;
    check = (int*)malloc(sizeof(int) * k);
    int i;
    for(i = 0; i < k; i++) check[i] = 0;
    for(i = 0; i < arrSize; i++){
        while(arr[i] < 0) arr[i] += k;
        check[(arr[i] % k)] += 1;
    }
    for(i = 1; i < k; i++){
        if(check[i] != check[k - i]) return false;
    }
    if(check[0] % 2 == 0) return true;
    else return false;
}
