int* smallerNumbersThanCurrent(int* nums, int numsSize, int* returnSize){
    int *r = malloc(sizeof(int) * numsSize);
    int i, j;
    for(i=0; i<numsSize; i++) {
        r[i] = 0;
        for(j=0, r[i]=0; j<numsSize; j++) {
            if (nums[i] > nums[j])
                r[i]++;
        }
    }
    *returnSize = numsSize;
    return r;
}
