class Solution {
public:
    int countBinarySubstrings(string s) {
        vector<int>v;
        int n = s.length() , c = 1 , ans = 0;
        bool flag = false;
        for(int i = 0 ; i < n ; i++)
        {
            if(s[i] == s[i+1])
                ++c;
            else
            {
                v.push_back(c);
                c = 1;
            }  
        }
        for(int i = 0 ; i < v.size()-1; i++)
            ans += min(v[i],v[i+1]);
        return ans;
    }
};
