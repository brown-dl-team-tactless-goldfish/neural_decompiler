int dfs(struct TreeNode* root,int prevSum){
	if (!root)
        return NULL;
    prevSum = prevSum * 10 + root->val;
    if (!root->left && !root->right)
        return prevSum;
    else
        return dfs(root->left,prevSum)+dfs(root->right,prevSum);
}

int sumNumbers(struct TreeNode* root){
    return dfs(root, 0);
}
