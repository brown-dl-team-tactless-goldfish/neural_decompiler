bool checkZeroOnes(char * s){
    int max1 = 0, max0 = 0, cnt = 0;
    
    for(int i = 0; s[i] != '\0'; i++){
        cnt ++;
        if(s[i] == '1'){
            if(max1 < cnt)
                max1 = cnt;
            if(s[i + 1] != '1')
                cnt = 0;
        }
        else{
            if(max0 < cnt)
                max0 = cnt;
            if(s[i + 1] != '0')
                cnt = 0;  
        }
    }
    if(max1 > max0)
        return true;
    return false;
}
