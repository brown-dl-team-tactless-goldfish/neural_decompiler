

int maximumUniqueSubarray(int* nums, int n){
    int s=0,max=-1,i=0,j=0,*ht;
    ht=(int*)calloc(10001,sizeof(int));
    while(i<n && j<n){
        if(ht[nums[j]]==0){
            s+=nums[j];
            ht[nums[j]]=1;
            j++;
            max=fmax(max,s);
        }else{
            s-=nums[i];
            ht[nums[i]]=0;
            i++;
        }
    }
    return max;
}
