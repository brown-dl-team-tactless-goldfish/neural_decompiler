#define Max(a,b) (a > b) ? a : b

int findLengthOfLCIS(int* nums, int numsSize) {
    if (numsSize == 0)
        return 0;
    
    int count = 1, maxLength = 1;
    for (int i = 0; i+1 < numsSize; i++) {
        if (nums[i] < nums[i+1]) {
            count++;
            if (i+1 == numsSize-1) {
                maxLength = Max(maxLength, count);
            }
        } else {
            maxLength = Max(maxLength, count);
            count = 1;
        }
    }
    
    return maxLength;
}
