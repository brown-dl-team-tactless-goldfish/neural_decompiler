vector<int> applyOperations(vector<int>& nums) {
	int sz = nums.size();
	for (int i = 0; i < sz - 1; i++)
		if (nums[i] == nums[i + 1]) {
			nums[i] *= 2;
			nums[i + 1] = 0;
		}
	int j = 0;
	for (int i = 0; i < sz; i++)
		if (nums[i] > 0)
			nums[j++] = nums[i];
	while (j < sz)
		nums[j++] = 0;
	return nums;
}
