int getWinner(int *arr, int arrSize, int k) {
    for (int i = 0; i < arrSize - 1; i++) {
        int start = i == 0 ? 0 : 1;
        int steps = start;
        if (steps == k) return arr[i];
        int j = i + 1;
        for (; j < arrSize; j++) {
            if (arr[i] > arr[j]) steps++; else break;
            if (steps == k) return arr[i];
        }
        if (j == arrSize) return arr[i];
        i += steps - start;
    }
    return arr[arrSize - 1];
}
