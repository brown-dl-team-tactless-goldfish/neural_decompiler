class Solution {
public:
    int leastBricks(vector<vector<int>>& wall) {
        unordered_map<int, int> umap;
        int cnt = 0;
        for (auto &row : wall) {
            for (int i = 0, psum = 0; i < row.size() - 1; i++) {
                psum += row[i];
                auto &tmp = ++umap[psum];
                cnt = max(cnt, tmp);
            }
        }
        return wall.size() - cnt;
    }
};
