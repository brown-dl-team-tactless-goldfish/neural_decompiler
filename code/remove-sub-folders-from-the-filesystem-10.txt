vector<string> removeSubfolders(vector<string>& folder) {
        sort(folder.begin(),folder.end());
        vector<string> ans;
        for(int i=0;i<folder.size();){
            string cmp="";
            ans.push_back(folder[i]);
            cmp=folder[i];
            i++;
            while(i<folder.size() && folder[i].find(cmp)==0 &&folder[i][cmp.length()]=='/'){
                i++;
                cout<<i<<endl;
            }
        }
        
        return ans;
