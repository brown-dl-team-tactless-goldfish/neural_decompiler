    public string FindDifferentBinaryString(string[] nums) {
        
        
        Array.Sort(nums);

        string maxBinary = new string('1', nums[nums.Length - 1].Length);

        int number = Convert.ToInt32(maxBinary, 2);

        for (int i=0; i<=number; i++)
        {
            string binary = Convert.ToString(i, 2).PadLeft(maxBinary.Length, '0');
            if (!nums.Contains(binary))
            {
                return binary;
            }
        }

        return "";
        
    }
