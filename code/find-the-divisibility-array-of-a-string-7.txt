class Solution {
public:
    vector<int> divisibilityArray(string word, int m) {
        long long c = 0;
        vector<int> ans;
        for(char i: word) {
            long long x = c*10 + (i-'0');
            c = x%m;
            x = (c == 0);
            ans.push_back(x);
        }
        return ans;
    }
};
