class Solution {
public:
    int minimumSwap(string s1, string s2) 
    {
        int xy=0;
        int yx=0;
        for(int i=0;i<s1.size();i++)
        {
            if(s1[i]=='x'&&s2[i]=='y')
            {
                xy++;
            }
            if(s1[i]=='y'&&s2[i]=='x')
            {
                yx++;
            }
        }
        if((xy+yx)%2==1)
        {
            return -1;
        }
        return ceil(double(xy/2.0))+ceil(double(yx/2.0));
    }
};
