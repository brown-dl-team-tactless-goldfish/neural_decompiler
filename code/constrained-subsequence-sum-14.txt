   int n=A.size();
   deque<int> dp;
   int res=A[0];
    for(int i=0;i<n;i++){
        
        A[i]=A[i]+ (dp.size() ? dp.front():0);
        
        res=max(res,A[i]);
        
        while(dp.size() && A[i]>dp.back()){
            dp.pop_back();
        }
        
        if(A[i]>0){
            dp.push_back(A[i]);
        }
        
        if(i>=k && dp.size() && A[i-k]==dp.front()){
            dp.pop_front();
        }
    
    }
    
    return res;
    
}
