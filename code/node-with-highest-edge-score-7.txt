
class Solution {
public:
    int edgeScore(vector<int>& edges) {
        vector<long long> ans(edges.size());
        for(long long i=0;i<edges.size();i++){
            ans[edges[i]] += i;
        }
        long long maxi = 0, a = -1;
        for(int i=0;i<ans.size();i++){
            if(ans[i] > maxi){
                maxi = ans[i]; a = i;
            }
        }
        return a;
    }
};
