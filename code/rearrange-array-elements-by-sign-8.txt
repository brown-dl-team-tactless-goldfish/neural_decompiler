int* rearrangeArray(int* nums, int numsSize, int* returnSize) {
	*returnSize = numsSize;

	int array_size = numsSize / 2;
	int pos[array_size], neg[array_size];
	register unsigned int i, j, x = 0, y = 0;
	for(i = 0; i < numsSize; i++) {
		if(nums[i] < 0) {
			neg[x] = nums[i];
			x++;
		} else {
			pos[y] = nums[i];
			y++;
		}
	}

	x = y = 0;
	for(j = 0; j <numsSize; j++) {
		if(j % 2 == 0) {
			nums[j] = pos[x];
			x++;
		} else {
			nums[j] = neg[y];
			y++;
		}
	}
	return nums;
}
