class Solution {
public:
    int findSmallestInteger(vector<int>& nums, int value) {
        unordered_map<int,int> umap;
        int n = nums.size();
        for(auto x : nums)
            umap[(x%value + value)%value]++;
        for(int i = 0; i < n; i++)
        {
            if(umap[i%value] == 0)
                return i;
            umap[i%value]--;
        }
        
        return n;
    }
};
