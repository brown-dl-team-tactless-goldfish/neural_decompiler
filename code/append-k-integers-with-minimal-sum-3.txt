long long solve(long long x,long long y){
    return (y+x)*(y-x+1)/2;
}
long long minimalKSum(vector<int>& nums, int k) {
    nums.push_back(0);
    nums.push_back(INT_MAX);
    int n = nums.size();
    sort(nums.begin(),nums.end());
    long long c = 0;
    for(int i=0;i<n-1 && k;i++){
        long long y = (nums[i+1]-nums[i]-1);
        if(y<=0) continue;
        if(y>k){
            y = k;
        }
        c+=solve(nums[i]+1,nums[i]+y);
        k-=y;
    }
    return c;
}
