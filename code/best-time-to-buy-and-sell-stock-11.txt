int maxProfit(int* prices, int pricesSize){
    
    int max;
    int ans=0;
    if(pricesSize == 0 || pricesSize==1)
        return 0;
    max = prices[pricesSize-1];
    
    for (int i=pricesSize-2; i >= 0; i--) {
        
        if(max - prices[i] > ans)
            ans = max - prices[i];
        
        if(prices[i] > max)
            max = prices[i];
        
        
    }

    return ans;

}
