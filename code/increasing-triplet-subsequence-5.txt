class Solution {
public:
    using VI = vector<int>;
    static constexpr auto INF = long(6e9);
    bool increasingTriplet(VI& A, long a = INF, long b = INF, long c = INF) {
        for (auto x: A)
            if      (x <= a) a = x;
            else if (x <= b) b = x;
            else if (x <= c) c = x;
        return c < INF;
    }
};
