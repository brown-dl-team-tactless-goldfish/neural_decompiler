class Solution {
public:
    bool canBeIncreasing(vector<int>& nums) {
        int cnt = 0, n=nums.size();
        int index = -1; //index conatin the index of the element to be removed
        for (int i = 1; i < n ; i++) 
            if (nums[i - 1] >= nums[i]) {
                cnt++;
                index = i;
            }
        if (cnt > 1) return false;
        
        // If no element is removed or only the last or the first element is removed
        if (index==-1 || index == n - 1 || index == 1)  return true;

        // If a[index] is removed
        if (nums[index - 1] < nums[index + 1])   return true;

        // If a[index - 1] is removed
        if (nums[index - 2] < nums[index])  return true;

        return false;
    }
};
