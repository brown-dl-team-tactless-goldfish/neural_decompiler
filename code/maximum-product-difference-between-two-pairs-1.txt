           // sort the array 
           for (int i = 0 ; i < numsSize-1 ; i++)
           {
          for ( int j = i+1 ; j < numsSize ; j++)
          {
              if ( nums[i] > nums[j])
              {
                  temp = nums[i];
                  nums[i] = nums[j];
                  nums[j] = temp;
                      
              }
          }
      }
      
      
      diff = ( nums[numsSize-1] * nums[numsSize-2] ) - ( nums[0] * nums[1] );
      
      return diff;
  }
