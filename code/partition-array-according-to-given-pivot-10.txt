vector<int> pivotArray(vector<int>& nums, int pivot) {
	vector<int> result;
	result.reserve(nums.size());
	for(const auto n:nums)
		if(n<pivot)
			result.push_back(n);
	for(const auto n:nums)
		if(n==pivot)
			result.push_back(n);
	for(const auto n:nums)
		if(n>pivot)
			result.push_back(n);
	return result;
}
