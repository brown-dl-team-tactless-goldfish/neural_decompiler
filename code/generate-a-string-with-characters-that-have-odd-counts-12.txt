

char * generateTheString(int n){
char* arr=(char*)malloc(sizeof(char)*(n+1));
    int i;
     if(n%2!=0)
     {
         for( i = 0;i<n;i++)
             arr[i]='a';
         arr[i]='\0';
     }
    else 
    {
        for( i = 0;i<n-1;i++)
        {
            arr[i]='a';
        }
        arr[i]='b';
        arr[++i]='\0';
    }
    return arr;
}
