class Solution {
public:
    string removeDigit(string number, char digit) {
        int offset = 0, pos = 0;
        
        string ans;
        while ((pos = number.find(digit, offset)) != std::string::npos) {
            string s = number.substr(0, pos) + number.substr(pos + 1);
            ans = max(ans, s);
            offset = pos + 1;
        }
        
        return ans;
    }
};
