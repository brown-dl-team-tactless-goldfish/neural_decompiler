class Solution {
public:

    int largestSumAfterKNegations( vector<int>& A, int K, int ans=0 )
    {
        priority_queue< int, vector< int >, greater< int >> q{ A.begin(), A.end() };
        for( auto top{ 0 }; K--; top = -1 * q.top(), q.pop(), q.push( top ) );
        for(; ! q.empty(); ans += q.top(), q.pop() );
        return ans;
    }
    
};
