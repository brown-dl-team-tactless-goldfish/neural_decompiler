class Solution {
public:
    int minOperations(vector<int>& nums, vector<int>& a) {
        sort(nums.begin(),nums.end());
        int g = a.front();
        for(int i=1;i<a.size();i++){
            g = __gcd(a[i],g);
            if(g == 1) break;
        }
        for(int i=0;i<nums.size() and nums[i] <= g;i++){
            if(g % nums[i] == 0) return i;
        }
        return -1;
    }
};
