class Solution {
public:
    bool isCovered(vector<vector<int>>& ranges, int left, int right) {
        sort(ranges.begin(),ranges.end());
        
        for(int i=0;i<ranges.size();i++)
        {
            if(ranges[i][0]<=left && left<=ranges[i][1])
            {
                //left is present in the range
                left=ranges[i][1]+1;
            }
        }
        if(left>right)
            return true;
        return false;
    }
};
