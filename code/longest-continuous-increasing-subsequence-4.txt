int findLengthOfLCIS(int* nums, int numsSize) {
    if (numsSize == 0)
        return 0;
    int longest = 0;
    int continuous = 1;
    for (int i = 1; i < numsSize; i++){
        if (nums[i] > nums[i-1]){
            ++continuous;
        } else {
            if (continuous > longest){
                longest = continuous;
            }
            continuous = 1;
        }
    }
    if (continuous > longest){
        longest = continuous;
    }
    return longest;
}
