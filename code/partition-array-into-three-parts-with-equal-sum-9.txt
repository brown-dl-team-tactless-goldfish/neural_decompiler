bool canThreePartsEqualSum(vector<int>& A) {
	auto totalSum = accumulate(begin(A), end(A), 0);
	if (totalSum % 3 != 0)
		return false;

	auto parts = 0;
	for (auto i = 0, tmpSum = 0; i < A.size() && parts < 3; i++) {
		tmpSum += A[i];
		if (tmpSum == totalSum / 3) {
			parts++;
			tmpSum = 0;
		}
	}
	return parts == 3;
}
