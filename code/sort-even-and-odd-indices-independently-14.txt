public class Solution
{
    public int[] SortEvenOdd(int[] nums) => nums.Select((num, i) => (IsEven: i % 2, Number: num))
        .GroupBy(tuple => tuple.IsEven)
        .Select(group => group.OrderBy(x => group.Key == 0 ? x.Number : -x.Number))
        .SelectMany(x => x.Select((x, i) => (Index: i * 10 + x.IsEven, Number: x.Number)))
        .OrderBy(x => x.Index)
        .Select(x => x.Number)
        .ToArray();
}
