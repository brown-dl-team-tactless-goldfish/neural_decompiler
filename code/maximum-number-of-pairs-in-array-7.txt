class Solution {
public:
    vector<int> numberOfPairs(vector<int>& nums) {
        unordered_map<int, int> count;
        int a = 0, b = 0;
        for (auto n : nums) {
            count[n]++;
        }
        for (auto c : count) {
            a += c.second / 2;
            b += c.second % 2;
        }
        return {a, b};
    }
};
