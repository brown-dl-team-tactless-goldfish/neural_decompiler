int minimumLength(char * s){
    int l = 0, r = strlen(s) - 1;
    if (r < 1) return strlen(s);
    while(l <= r) {
        int pl = l, pr = r;
        while(l < r && s[l] == s[r]) l++;
        while(l > 0 && r >= l && s[r] == s[l - 1]) r--;
        if (pl == l && pr == r) break;
    }
    return r - l + 1;
}
