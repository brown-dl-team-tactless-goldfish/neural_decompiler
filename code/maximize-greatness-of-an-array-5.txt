class Solution {
public:
    int maximizeGreatness(vector<int>& nums) {
        multiset<int> st(nums.begin(),nums.end());
        int res=0;
        for(int i=0;i<nums.size();i++)
        {
            auto it=st.upper_bound(nums[i]);
            if(it!=st.end())
            {
                res++;
                st.erase(it);
            }
        }
        return res;
    }
};
