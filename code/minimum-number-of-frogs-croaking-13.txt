    int minNumberOfFrogs(string cs) {
        int res = 0, memo[128] = {}, prev[128] = {};
        for (char cur = 'c'; const auto& c : "croak") {
            prev[c] = cur;
            cur = c;
        }
        for (const auto& c : cs) {
            if (++memo[c] > memo[prev[c]]) {
                return -1;
            }
            res = max(res, memo['c'] - memo['k']);
        }
        return memo['c'] == memo['k'] ? res : -1;
    }
