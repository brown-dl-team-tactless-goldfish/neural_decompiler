struct ListNode* swapPairs(struct ListNode* head){
    
    struct ListNode *next;
    struct ListNode *temp;
    
    if(head == NULL || head->next == NULL)
        return head;

    next = head->next->next;
    
    temp = head->next;
    head->next->next = head;
    head->next = swapPairs(next);
    
    return temp;
    
    
    
}
