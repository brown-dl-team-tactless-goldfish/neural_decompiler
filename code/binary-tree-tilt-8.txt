int calMaxTree(struct TreeNode* root, int * sum)
{
    if(root == NULL) return 0;
    int left = calMaxTree(root->left, sum);
    int right = calMaxTree(root->right, sum);
    *sum += abs(left - right);
    return root->val + left + right;
}
int findTilt(struct TreeNode* root){
    int sum = 0;
    calMaxTree(root, &sum);
    return sum;
}
