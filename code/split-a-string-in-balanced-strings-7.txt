int balancedStringSplit(char * s){
int res=0;
int i, countR=0, countL=0;
char c = NULL;
    
    for ( i = 0; i < strlen(s); i++){
        if ( s[i] == 'L'){
            countL++;
        } else if ( s[i] == 'R' ){
            countR++;
        }
        if ( countL == countR ){
            res++;
            countL=0;
            countR=0;
        }
    }

return res;
}
