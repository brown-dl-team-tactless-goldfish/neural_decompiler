int totalMoney(int n) {
    int money = 0, daily_increase = 1, monday_increase = 2;
    
    for (int i = 0; i < n; i++) {
        if ((i != 0) && (i % 7) == 0) {
            daily_increase = monday_increase;
            monday_increase++;
        }
        money+=daily_increase;
        daily_increase++;
    }
    
    return money;
}
