    long long maximumImportance(int n, vector<vector<int>>& roads) {
        vector<int> g(n); //store number of neighbor
        for(auto& r : roads) g[r[0]]++, g[r[1]]++;
        sort(g.begin(), g.end(), greater<int>());
        long long res = 0;
        for(long long x : g) res += x * n--;
        return res;
    }
