int* findClosestElements(int* arr, int n, int k, int x, int* returnSize){
    int* ans = (int*)malloc(k*sizeof(int));
    *returnSize = k;
    
    int l = 0;
    int r = n-1;
    while((r-l) >= k){
        (x - arr[l] <= arr[r] - x) ? r-- : l++;
    }
    return memcpy(ans,arr+l,((r-l)+1)*sizeof(int));
}
