int oddCells(int m, int n, int **indices, int indicesSize, int *indicesColSize)
{
    char *row = calloc(m, sizeof(char));
    char *column = calloc(n, sizeof(char));
    int r_odd = 0, c_odd = 0;
    for (int i = 0; i < indicesSize; i++)
    {
        row[indices[i][0]] ^= 1;
        column[indices[i][1]] ^= 1;
        r_odd += row[indices[i][0]] ? 1 : -1;
        c_odd += column[indices[i][1]] ? 1 : -1;
    }
    return r_odd * (n - c_odd) + c_odd * (m - r_odd);
}
