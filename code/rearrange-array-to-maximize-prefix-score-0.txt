public class Solution {
    public int MaxScore(int[] nums) {
        nums = nums.OrderByDescending(x => x).ToArray(); //sort in descending order
        long sum = 0;
        int ans = 0;
        
        foreach(var num in nums)
        {
            sum += num; //sum of numbers till this iteration
            if(sum > 0)
                ++ans;  //number of elements till iteration
            else
                break;
        }
        
        return ans;
    }
}
