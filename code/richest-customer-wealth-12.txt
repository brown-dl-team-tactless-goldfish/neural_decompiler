int maximumWealth(int** accounts, int accountsSize, int* accountsColSize){
    int max = 0;
    int i = 0;
    
    while(i < accountsSize) {
        int temp = 0;
        int j = 0;
        
        while(j < *accountsColSize) {
            temp += accounts[i][j];
            j++;
        }
        
        
        if(temp > max) {
            max = temp;
        }
        
        i++;
    }
    
    return max;
}
