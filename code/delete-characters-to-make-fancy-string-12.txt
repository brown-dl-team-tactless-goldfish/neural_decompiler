string makeFancyString(string s) {
        string ans;
        if (s.size() <= 2) return s;
        ans += s[0];
        for (int i=1;i<s.size()-1;++i) {
            char c = s[i];
            if (s[i-1] == c && s[i+1] == c) continue;
            ans += c;
        }
        ans += s[s.size()-1];
        return ans;
    }
