class Solution:
    def stoneGameVI(self, aliceValues: List[int], bobValues: List[int]) -> int:
        n = len(aliceValues)
        combinedValues = []
        for i in range(n):
            tmpV = aliceValues[i] + bobValues[i]
            combinedValues.append([tmpV, i])

        combinedValues.sort(reverse=True)
        print("combinedValues = ", combinedValues)

        score_a, score_b = 0, 0
        for i in range(n):
            curV = combinedValues[i]
            if (i % 2 == 0):
                score_a += aliceValues[curV[1]]
            else:
                score_b += bobValues[curV[1]]

        print("score_a = ", score_a)
        print("score_b = ", score_b)

        if (score_a > score_b):
            return 1
        elif (score_a == score_b):
            return 0
        else:
            return -1
