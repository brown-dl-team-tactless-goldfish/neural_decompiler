char * lastSubstring(char * s) {
    char *sl = s, *sr = s + 1;
	// compare 2 subsrings, left and right, starting from the beggining of the initial string
	// left subsring should become the greatest
    for (int k = 0; ; ++sr, k = 0) {    // increment pointer of right substring
        for (; sl[k] == sr[k]; ++k);    // increment k, until symbols match
        if (sr[k] == 0) break;          // end of the initial string, return
        if (sl[k] < sr[k]) sl = sr;     // right substing is greater, assign to the left
    }
    return sl;
}
