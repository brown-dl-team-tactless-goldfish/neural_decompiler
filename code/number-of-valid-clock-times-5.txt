class Solution {
public:
    int countTime(string time) {
        char a = time[0], b = time[1], c = time[3], d = time[4];
        int hour = 1, min1 = 1, min2 = 1;
        //there is two case possible in hour's substring.
        if(a == '?' and b == '?') hour = 24;
        else if(a == '?' and b != '?') hour = b<='3'?3:2;
        else if(a !='?' and b == '?') hour = a<'2'?10:4;
        if(c == '?') min1 = 6;
        if(d == '?') min2 = 10;
        return hour*min1*min2;
    }        
};
