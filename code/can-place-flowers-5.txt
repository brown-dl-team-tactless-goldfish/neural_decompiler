bool canPlaceFlowers(int* flowerbed, int flowerbedSize, int n){

    for ( int i =0 ; i < flowerbedSize ; i++){
        if ( flowerbed[i]==1)
            continue; 
        else
            if(  (   i-1 <0 || flowerbed[i-1]==0  )   && (  i+1>= flowerbedSize || flowerbed[i+1]==0) ) {
                flowerbed[i]=1 ; 
                n--;
            }
    }
    
    return     n<=0 ? true : false  ; 
}
