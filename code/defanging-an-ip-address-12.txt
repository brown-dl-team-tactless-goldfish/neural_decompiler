char * defangIPaddr(char * address){
    
    char* result = (char*)malloc(22* sizeof(*address) ); //22 covers max possible length

    int j=0;
    
    while(*address)
    {
        if(*address == '.')
        {
            result[j++] = '[';
            result[j++] = '.';
            result[j++] = ']';
            address++;
            continue;
        }
        result[j++] = *address++;
    }
    result[j] = '\0';
    return result;

}
