class Solution {
public:
    double averageWaitingTime(vector<vector<int>>& customers) {
        long long int chefAvail = 0;
        double waitTime = 0;
        for (auto c : customers) {
            if(chefAvail < c[0]) {
                chefAvail = c[0];
            }
            long long int end = chefAvail + c[1];
            waitTime += end - c[0];
            chefAvail = end;
        }

        return waitTime / customers.size();
    }
};
