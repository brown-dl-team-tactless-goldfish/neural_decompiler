

int minCost(char * s, int* cost, int costSize)
{
    int sum = 0;
   
    for(int i=0;i<strlen(s)-1;++i)
    {
        if(s[i]==s[i+1])
        {
            if(cost[i]<cost[i+1])
            {
                sum += cost[i];
                s[i] = s[i+1];
                cost[i] = cost[i+1];
            }
            else
            {
                sum += cost[i+1];
                s[i+1] = s[i];
                cost[i+1] = cost[i];
            }
        }
    }
    
    return sum;
}
