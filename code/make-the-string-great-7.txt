char * makeGood(char * s){
    int len = strlen(s), i, top=0;
    char diff = abs('A'-'a');
    char c;
    
    for(i=1; i<len; i++) {
        c=s[i];
        if(top>=0 && abs(c-s[top]) == diff){
            top--;
        } else {
            s[++top] = c;
        }
        
    }
    s[++top] = '\0';
    return s;

}
