class Solution {
public:
    string thousandSeparator(int n) {
        string s = to_string(n), ans;
        int l = s.length(), c = 0;
        for (int i=l-1; i>=0; --i) {
            if (c == 3) {
                ans = '.' + ans;
                c = 0;
            }
            ans = s[i] + ans;
            ++c;
        }
        return ans;
    }
};
