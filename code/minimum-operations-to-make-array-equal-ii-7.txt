class Solution {
public:
    long long minOperations(vector<int>& nums1, vector<int>& nums2, int k) {
        int n = nums1.size();
        if(k == 0)
            return nums1 == nums2 ? 0:-1;
        long long a = 0, b = 0;
        for(int i=0; i<n; ++i){
            long long x = nums1[i] - nums2[i];
            if(x < 0)
                a += -x;
            else
                b += x;
        }
        if(a != b or a%k != 0)
            return -1;
        return a/k;
    }
};
