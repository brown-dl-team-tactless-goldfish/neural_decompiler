public class Solution {
    public bool DigitCount(string num) {
        
        if(string.IsNullOrEmpty(num))
            throw new ArgumentException("Invalid Input.");
        
        int[] freq = new int[10];
        foreach(var c in num)
            freq[c - '0']++;
        
        for(int i = 0; i < num.Length; i++)
        {
            if(freq[i] != num[i] - '0')
                return false;
        }
        
        return true;
    }
}
