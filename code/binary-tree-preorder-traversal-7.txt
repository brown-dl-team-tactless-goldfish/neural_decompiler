void tra(struct TreeNode* root,int *arr,int *size) {
    if(root != NULL) {
        arr[(*size)++] = root->val;
        tra(root->left,arr,size);
        tra(root->right,arr,size);
    }
}
int* preorderTraversal(struct TreeNode* root,int *returnSize){
    *returnSize = 0;
    int * arr = malloc(100*sizeof(int));
    tra(root, arr, returnSize);
    arr = realloc(arr, (*returnSize)*sizeof(int));
    return arr;
}
