#define MAX(a, b) ((a) > (b) ? (a) : (b))

int getMaximumGenerated(int n) {
    if (n == 0) return n;
    int* arr = (int*) calloc(n + 1, sizeof(int));
    arr[1] = 1;
    for (int i = 2; i < n + 1; i++) {
        arr[i] = (i % 2 == 0) ? arr[i / 2] : arr[(i - 1) / 2] + arr[1 + (i - 1) / 2];
    }
    int max = 0;
    for (int i = 0; i < n + 1; i++) {
        max = MAX(max, arr[i]);
    }
    free(arr);
    return max;
}
