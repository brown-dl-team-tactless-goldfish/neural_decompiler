class Solution 
{
public:
    vector<int> minSubsequence(vector<int>& nums)
    {
        // sort the elements so that we can pick large elements first
        sort(nums.begin(),nums.end());
        int total=0,sum=0;
        
        // calculate the total sum
        for(int i=0;i<nums.size();i++)
        {
            total+=nums[i];
        }
        
        // ans will store the result 
        vector<int>ans;
        
        // Now iterate from the last so that we can pick large elements first and also keep track whether the sum of the currently formed subsequence is greater than sum of the rest. If it is then break the loop ans return the ans.
        for(int i=nums.size()-1;i>=0;i--)
        {
            sum+=nums[i];
            ans.push_back(nums[i]);
            if(sum>total-sum)
            {
                break;
            }
        }
        return ans;
    }
};
