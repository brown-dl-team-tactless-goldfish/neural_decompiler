    int reverse(int n){
        int reverse=0;
        int remainder;
        while(n){
            remainder = n % 10;
            reverse = reverse * 10 + remainder;
            n /= 10;
        }
        return reverse;
    }
    int countDistinctIntegers(vector<int>& nums) {
        // insert all values in set
        // insert reverse in set 
        // return set size
        unordered_set<int> st;
        for(int i=0;i<nums.size();i++){
            st.insert(nums[i]);
            int rev = reverse(nums[i]);
            st.insert(rev);
        }
        return st.size();
    }
