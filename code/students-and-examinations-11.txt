with cte as
(
select student_id,student_name,subject_name
from students s cross join subjects s1 
),

temp as
(select student_id, subject_name, count(1) as attended_exams
 from Examinations e
 group by student_id, subject_name

)

select c.student_id, c.student_name, c.subject_name, ifnull(t.attended_exams, 0) as attended_exams
from cte c left join temp t
on c.student_id = t.student_id and c.subject_name = t.subject_name
order by student_id, subject_name
