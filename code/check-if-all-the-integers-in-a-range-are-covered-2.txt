class Solution:
    def isCovered(self, ranges: List[List[int]], left: int, right: int) -> bool:
        data = []
        
        for ran in ranges:
            print ("ran = ", ran)
            
            for d in range(ran[0], ran[1] + 1):
                data.append(d)
                
        print ("data = ", data)
        
        for d in range(left, right+1):
            if (d not in data):
                return False
            
        return True
