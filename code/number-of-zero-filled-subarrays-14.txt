class Solution {
public:
    long long zeroFilledSubarray(vector<int>& arr) {
        int i = 0;
        int j = 0;
        long long c = 0;
        long long ans = 0;
        for(int i = 0;i < arr.size();i++){
            if(arr[i] == 0) c++;
            else{
                ans += (c*(c+1))/2;
                c = 0;
            }
        }
        ans += (c*(c+1))/2;
        return ans;
    }
};
