int cmp(int *x, int *y){
    return(*x-*y);
}
int numRescueBoats(int* people, int peopleSize, int limit){
    qsort(people, peopleSize, sizeof(int), cmp);
    int cnt=0, st=0;
    for(int i=peopleSize-1; i>=st; i--){
        cnt++;
        if(people[i]+people[st]<=limit)
            st++;
    }
    return cnt;
}
