class Solution {
public:
    long long numberOfWeeks(vector<int>& milestones) {
        
        long long sum = 0;
        int maxl = INT_MIN;
        for(auto &t : milestones)
        {
            sum+=t;
            maxl = max(maxl,t);
        }
        
        // if maxL > all other milestones then return 2 * (sum-maxl) + 1 
        return maxl > (sum-maxl) ? 2 * (sum-maxl) + 1 : sum; 
    }
};
