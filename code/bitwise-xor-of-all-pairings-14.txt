class Solution {
public:
    int xorAllNums(vector<int>& n1, vector<int>& n2) {
        int s1 = n1.size(), s2 = n2.size(), res = 0;
        if(n2.size()%2) res ^= accumulate(begin(n1), end(n1), 0, [](auto sum, auto e) {return sum ^e;});
        if(n1.size()%2) res ^= accumulate(begin(n2), end(n2), 0, [](auto sum, auto e) {return sum ^e;});
        return res;
    }
};
