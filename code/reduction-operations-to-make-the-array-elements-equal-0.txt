int cmp(const void* a, const void* b){
    return *(int*)b - *(int*)a;
}
int reductionOperations(int* nums, int numsSize){
    int n = numsSize;
    qsort(nums, n, sizeof(int), cmp);
    int ans = 0; 
    int cn = 1;
    int k = 0;
    for(int i = 1; i < n; i++){
        if(nums[i] == nums[i-1])
            cn++;
        else{
            k += cn;
            ans += k;
            cn = 1;
        }
    }
    return ans;
}
