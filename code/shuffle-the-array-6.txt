int* shuffle(int* nums, int numsSize, int n, int* returnSize){
    *returnSize = numsSize;
    
    int *result = malloc(numsSize * sizeof(int));
    
    int *ptr_1 = &nums[0];
    int *ptr_2 = &nums[n];
    int w = 0;
    
    for (int i = 0; i < n; i++){
        result[w++] = *(ptr_1++);
        result[w++] = *(ptr_2++);
    }
    
    return result;
}
