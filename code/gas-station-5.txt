int canCompleteCircuit(int* gas, int gasSize, int* cost, int costSize){
    int total=0,index=0,sum=0;
    for(int i=0;i<gasSize;i++)
    {
        total+=gas[i]-cost[i];
        if(total<0)
        {
            sum+=total;
            total=0;
            index=i+1;
        }
    }
    sum+=total;
    index=sum<0?-1:index;
    return index;
}
