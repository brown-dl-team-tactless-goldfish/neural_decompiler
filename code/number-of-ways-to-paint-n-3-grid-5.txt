class Solution {
public:
    int numOfWays(int n) {
        long long aba = 6, abc = 6, mod = 1e9 + 7;
        for(int i=1; i<n; i++) {
            int temp = aba;
            aba = (3 * aba + 2 * abc) % mod;
            abc = (2 * temp + 2 * abc) % mod;
        }
        return (aba + abc) % mod;
    }
};
