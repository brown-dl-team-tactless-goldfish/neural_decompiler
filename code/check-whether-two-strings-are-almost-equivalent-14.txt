    public bool CheckAlmostEquivalent(string word1, string word2) {
        
        for(char i = 'a'; i <= 'z'; i++){
            int count1 = word1.ToCharArray().Where(c => c == i).Count();
            int count2 = word2.ToCharArray().Where(c => c == i).Count();
            
            if(Math.Abs(count1 - count2) > 3){
                return false;
            }
        }
        
        return true;
    }
