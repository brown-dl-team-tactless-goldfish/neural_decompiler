    int countTriplets(vector<int>& nums) {
        unordered_map<int, int> m;
        int n = nums.size();
        for (int i = 0; i< n; i++)
            for (int j = 0; j < n; j++)
                m[(nums[i] & nums[j])]++;
        int c = 0;
        for (auto it: m)
            for (int i = 0; i < n; i++)
                if (!(it.first & nums[i]))
                    c += it.second;
        return c;
    }
