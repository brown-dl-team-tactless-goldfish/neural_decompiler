int* diStringMatch(char * S, int* returnSize)
{
        *returnSize=strlen(S)+1;
        int *a=malloc(sizeof(int)*(*returnSize));
        int i;
        int I_val=0,D_val=strlen(S),j=0;
        for(i=0;i<strlen(S);i++)
        {
                if(S[i]=='I')
                        a[j++]=I_val++;
                else
                        a[j++]=D_val--;
        }
        a[j]=I_val++;
        return a;

}
