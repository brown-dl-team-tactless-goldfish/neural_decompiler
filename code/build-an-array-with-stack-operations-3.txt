char **buildArray(int *target, int targetSize, int n, int *returnSize)
{
    *returnSize = 2*target[targetSize-1] - targetSize;
    char **ret = malloc(sizeof(char*) * *returnSize);
    for(int targ_index=0,ret_index=0,counter=1; targ_index < targetSize; ++targ_index)
    {
        while(counter++ < target[targ_index])
        {
            ret[ret_index++] = "Push";
            ret[ret_index++] = "Pop";
        }
        ret[ret_index++] = "Push";
    }
    return ret;
}
