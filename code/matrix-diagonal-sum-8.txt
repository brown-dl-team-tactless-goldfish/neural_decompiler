int diagonalSum(int** mat, int matSize, int* matColSize){
int sum=0,i,k,sum1=0,j;
    for(i=0;i<matSize;i++){
        sum = sum+mat[i][i];
    }
    for(j=matSize-1;j>=0;j--){
        sum1= sum1+mat[matSize-1-j][j];
    }
    if(matSize%2!=0){
        k = mat[matSize/2][matSize/2];
    }
    else{
        k=0;
    }
    return sum1+sum-k;
}
