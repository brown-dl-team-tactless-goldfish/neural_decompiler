class Solution {
public:
    int minMoves(int t, int k) {
        queue<vector<int>>q;
        int count = 0;
        while(t != 0 and k != 0)
        {
            if(t % 2 == 0)
            {
                t = t / 2;
                k--;
                count++;
            }
            else
            {
                t = t - 1;
                count++;
            }
        }
        count += (t - 1);
        return count;
        
        
    }
};
