class Solution {
public:
    int reductionOperations(vector<int>& nums) {
        unordered_map<int,int> um;
        for(int x : nums){
            um[x]++;
        }
        priority_queue<pair<int,int> > pq(um.begin(),um.end());
        int ans = 0;
        int freq = 0;
        while(pq.size()>1){
            pair<int,int> p = pq.top(); 
            pq.pop();
            freq += (p.second);
            ans+=freq;
        }
        return ans;
    }
};
