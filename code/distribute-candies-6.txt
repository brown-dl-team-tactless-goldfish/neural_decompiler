int count=1,k=0;
int kind[10000] = {-100001};

quickSort(candies,0,candiesSize-1);

for(int i=1;i<candiesSize;i++)
{
	if(candies[i] == candies[i-1])
	{
		count++;
	}
	else 
	{
		kind[k++] = count;
		count = 1;
	}
	if(i == candiesSize-1)
		kind[k++] = count;  
}

if(k<candiesSize/2)
	return k ;
else
	return candiesSize/2 ;
