int countStudents(int* students, int students_size, int* sandwiches, int sandwiches_size){
    int     student_idx;
    int     count = students_size;
    int     i, j;
    
    for (i = 0, j = 0; count != 0 && j != sandwiches_size; ++i)
    {
        student_idx = i % students_size;
        if (students[student_idx] == -1)
            continue;
        if (students[student_idx] != sandwiches[j])
        {
            --count;
            continue;
        }
        students[student_idx] = -1;
        ++j;
        count = students_size - j;
    }
    return (sandwiches_size - j);
}
