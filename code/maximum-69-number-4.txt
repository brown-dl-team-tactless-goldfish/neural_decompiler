int maximum69Number (int num){
    int dig = floor(log10(num));
    int div = pow(10,dig);
    int res = 0;
    while (div){
        int q = num/div;
        int rem = num%div;
        res *= 10;
        if (q == 6){
            
            res += 9;
            res *= div;
            res += rem;
            return res;
        }
        res += q;
        num = rem;
        div /= 10;
    }
    return res;

}
