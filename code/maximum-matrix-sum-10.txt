    long long maxMatrixSum(vector<vector<int>>& g) {
        
        long long res = 0; int neg = 0;
        priority_queue<int, vector<int>, greater<int>> q;
        
        for(int i = 0 ; i < g.size() ; i++)
            for(int j = 0 ; j < g[0].size() ; j++) {
                
                if(g[i][j] <= 0) neg++; 
                q.push(abs(g[i][j]));
                res += abs(g[i][j]);
            }
        
        if(neg % 2 && q.size()) res -= 2 * q.top();
        return res;
    }
