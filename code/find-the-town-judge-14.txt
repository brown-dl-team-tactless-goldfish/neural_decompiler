

int findJudge(int N, int** trust, int trustSize, int* trustColSize){
    
    int in[N+1], out[N+1];
    
    for(int i=0; i<N; i++) {
        
        in[i+1] = out[i+1] = 0 ;
    }
    
    for(int i=0; i<trustSize; i++) {
                   
            out[trust[i][0]]++;
            in[trust[i][1]]++;
        
        
    }
    
    for(int i=0; i < N; i++) {
        if(in[i+1] == N-1 && out[i+1] == 0)
            return i+1;
        
    }
    return -1;
}
