int numOfSubarrays(int* arr, int arrSize, int k, int threshold){

    int suma        = 0;
    int sumofsubarr = 0;

    for(int left=0, right=0;right<arrSize;right++){
        suma+=arr[right];

        if(right-left==k-1)
        {
            if(suma/k>=threshold) sumofsubarr++;
            suma-=arr[left];
            left++;
        }  
    }
    return sumofsubarr;
}
