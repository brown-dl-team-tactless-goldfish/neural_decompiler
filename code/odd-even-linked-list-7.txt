struct ListNode* oddEvenList(struct ListNode* head){
    if(head == NULL || head->next == NULL || head->next->next == NULL) return head;
    struct ListNode *odd = head, *even = head->next, *temp = head->next;
    while(even && even->next)
    {       
        odd->next = even->next;
        odd = odd->next;
        even->next = odd->next;
        even = even->next;
    }
    odd->next = temp;
    return head;
}
