#define max(a,b) (a>b?a:b)

int maximumValue(char ** strs, int strsSize)
{
    int ans = 0;
    bool var = false;

    for(int i = 0; i < strsSize; ++i)
    {
        var = false;
        int len = strlen(*(strs+i));
        for(int j = 0; j < len; ++j)
        {
            if(*(*(strs+i)+j)>='a')
            {
                var = true;
                break;
            }
        }
        
        ans = max(ans, (var ? len : atoi(*(strs+i))));
    }
    return ans;
}
