int dayOfYear(char * date){
    int days[]={31,28,31,30,31,30,31,31,30,31,30,31};
    int d,yr,mon;
    int sum=0;
    yr=atoi(date);
    mon=atoi(date+5);
    d=atoi(date+8);
    if(yr%4==0&&yr%100!=0 || yr%400==0)
    {
      days[1]++;  
    }
    for(int i=0;i<mon-1;i++)
    {
        sum=sum+days[i];
        
    }
return d+sum;
}
