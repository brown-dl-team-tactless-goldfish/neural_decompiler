int differenceOfSum(int* nums, int numsSize){

    int elementSum = 0, digitSum = 0;
    for(int i = 0; i < numsSize; i++){
        elementSum += nums[i];
    }
    for(int i = 0; i < numsSize; i++){
        do{
            digitSum += nums[i] % 10;
            nums[i] /= 10;
        }while(nums[i] > 0);
    }
    if(elementSum - digitSum <= 0){
        return digitSum - elementSum;
    }else{
        return elementSum - digitSum;
    }
}
