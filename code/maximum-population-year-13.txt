class Solution {
public:
    int maximumPopulation(vector<vector<int>>& logs) {
        int a[2051] = {};
        for(auto & l : logs) {
            a[l[0]]++;
            a[l[1]]--;
        }

        int y = -1;
        for(int i = 1950, cnt = 0, max_v = 0; i <= 2050; i++) {
            cnt += a[i];
            if(cnt > max_v) y = i;
            max_v = max(max_v, cnt);
        }
        return y;
    }
};
