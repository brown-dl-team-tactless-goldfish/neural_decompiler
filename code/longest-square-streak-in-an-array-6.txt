class Solution:
    def longestSquareStreak(self, nums: List[int]) -> int:
        dp = defaultdict(int)
        for x in sorted(nums): 
            dp[x] = max(dp[x], 1)
            v = isqrt(x)
            if v**2 == x: dp[x] = 1 + dp[v]
        ans = max(dp.values())
        return ans if ans > 1 else -1
