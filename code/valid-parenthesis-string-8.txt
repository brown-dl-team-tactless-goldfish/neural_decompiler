public class Solution {
    public bool CheckValidString(string s) {
        int min = 0; 
        int max = 0;
        
        foreach (char c in s) {
           min += c == '(' ? 1 : -1;
           max += c != ')' ? 1 : -1;
           if (max < 0) 
               break;
           min = Math.Max(min, 0);
        }
        
        return min == 0;
    }
}
