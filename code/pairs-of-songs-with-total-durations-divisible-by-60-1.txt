int numPairsDivisibleBy60(int* time, int timeSize) {
    int ans = 0, arr[60] = {0};
    for(int i = 0; i < timeSize; i++) {
        time[i] %= 60;
        arr[time[i]]++;
    }
    if(arr[0] > 0) ans += ((long long) arr[0] * (long long) (arr[0] - 1)) / (long long) 2;
    if(arr[30] > 0) ans += ((long long) arr[30] * (long long) (arr[30] - 1)) / (long long) 2;
    for(int i = 1; i < 30; i++) {
        if(arr[i] > 0) ans += (arr[i] * arr[60-i]);
    }
    return ans;
}
