bool checkIfExist(int* arr, int arrSize) {
    int temp;
    int doubles;
    for(int i = 0; i < arrSize; i++)
    {
        temp = arr[i];
        for(int j = 0; j < arrSize; j++)
        {
            doubles = arr[j] * 2;
            if(i != j && temp == doubles)
            {
                return true;
            }
        }
    }
    
    return false;
}
