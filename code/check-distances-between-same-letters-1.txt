bool checkDistances(char * s, int* distance, int distanceSize){
    int map[26];
    memset(map,-1,26*sizeof(int));
    
    int len=strlen(s);
    for(int i=0;i<len;i++){
        if(map[s[i]-'a']==-1) map[s[i]-'a']=i;
        else if(map[s[i]-'a']!=i && distance[s[i]-'a']!=i-map[s[i]-'a']-1){         
                return false;
        }
    }
    return true;
}
