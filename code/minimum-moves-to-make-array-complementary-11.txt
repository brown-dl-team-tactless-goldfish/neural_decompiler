    for(int i=0;i<n/2;i++){
        vec[min(nums[i],nums[n-i-1])+1]-=1;
        vec[nums[i]+nums[n-i-1]]-=1;
        vec[nums[i]+nums[n-i-1]+1]+=1;
        vec[max(nums[i],nums[n-i-1])+limit+1]+=1;
    }
    for(int i=1;i<2*limit+2;i++){  // precomputation
        vec[i]=vec[i]+vec[i-1];
    }
    return *min_element(vec.begin()+2,vec.end()-2);
