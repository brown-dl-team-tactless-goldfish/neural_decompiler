int subarraysDivByK(int* nums, int numsSize, int k){
    int n = numsSize ;
    int* hash = calloc(k , sizeof(int)) ;
    hash[0] = 1 ;
    int sum = 0 ;
    int ans = 0 ;
    for(int i = 0; i < n ; i++){
        sum += nums[i] ;
        int val = sum + 10000 * k;
        while(val < 0){
            val += k ;
        }
        ans += hash[val % k] ;
        hash[val % k]++;
    }
    free(hash) ;
    return ans ;
}
