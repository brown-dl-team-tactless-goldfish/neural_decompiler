int cmp(const void *a, const void *b){
    return *(int*)a - *(int*)b;
}
int distinctAverages(int* nums, int numsSize){
    int ans[201] = {0};
    int cnt = 0;
    qsort(nums,numsSize, sizeof(int), cmp);

    for (int i=0; i<numsSize/2; i++){
        int tmp = nums[i] + nums[numsSize -1 -i];
        ans[tmp] ++;
    }
    for (int i=0; i<201; i++){
        if(ans[i] != 0) cnt++;
    }

    return cnt;
}
