int* luckyNumbers (int** matrix, int matrixSize, int* matrixColSize, int* returnSize){
    int r = matrixSize, c = *matrixColSize, **m = matrix, *a = malloc(sizeof(int [50])), l = 0;
    for (int i = 0, n = INT_MAX, cl ; i < r ; n > 0 && (a[l++] = n), n = INT_MAX, i++) {
        for (int j = -1 ; ++j < c ; n > m[i][j] && (cl = j, n = m[i][j]));
        for (int k = -1 ; ++k < r && n > 0 ; k != i && m[k][cl] > n && (n = -1));
    }
    return *returnSize = l, a;
}
