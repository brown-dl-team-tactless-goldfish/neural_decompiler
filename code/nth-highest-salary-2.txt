CREATE FUNCTION getNthHighestSalary(N INT) RETURNS INT
BEGIN
  RETURN (
      # Write your MySQL query statement below.
      
      WITH c AS (
          SELECT salary, DENSE_RANK() OVER(ORDER BY salary DESC) as dr
          FROM Employee
      )
      SELECT salary 
      FROM c
      WHERE dr = N
      LIMIT 1
      
  );
END
