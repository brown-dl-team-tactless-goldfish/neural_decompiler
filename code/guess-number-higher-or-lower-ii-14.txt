class Solution {
public:
    int dp[201][201];
    int f(int i,int j)
    {
        if(i>=j) return 0;
        if(dp[i][j]!=-1) return dp[i][j];
        int ans=INT_MAX;
        for(int it=i;it<=j;it++)
        {
            ans=min(ans,it+max(f(i,it-1),f(it+1,j)));
        }
        return dp[i][j] = ans;
    }
    int getMoneyAmount(int n) {
        memset(dp,-1,sizeof(dp));
        return f(1,n);
    }
};
