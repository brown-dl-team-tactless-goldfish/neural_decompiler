int firstUniqChar(char * s){
    
    int *freq = (int*)calloc(26,sizeof(int));
    int i,res;
    int len = strlen(s);
    res = -1;
    
    for(i=0;i<len;i++){
        freq[s[i]-'a']++;
    }
	
    for(i=0;i<len;i++){
        if(freq[s[i]-'a'] == 1){
            res = i;
            break;
        } 
    }
    
    if(res!=-1)
        return res;
		
    return res;
}
