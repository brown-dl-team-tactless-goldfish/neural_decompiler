int mostWordsFound(char ** sentences, int sentencesSize){
    int i, j, max = 0, temp;
    
    for(i = 0; i < sentencesSize; i++){
        for(j = 0, temp = 0; j < strlen(sentences[i]); j++){
            if(sentences[i][j] == ' ')
                temp++;
        }
        max = temp > max ? temp : max;
    }
    
    return max + 1;
}
