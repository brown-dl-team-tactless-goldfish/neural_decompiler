class Solution {
public:
    vector<int> rearrangeArray(vector<int>& nums) {
        vector<int> arr = nums;
        sort(arr.begin(), arr.end());
        int j = 0, n = nums.size();
        for (int i = 0; i < n; i+=2) {
            nums[i] = arr[j++];
        }
        for (int i = 1; i < n; i+=2) {
            nums[i] = arr[j++];
        }
        return nums;
    }
};
