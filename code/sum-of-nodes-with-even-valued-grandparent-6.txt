// Function for summing all the grandchildrens
int sumGrandchilds(struct TreeNode* root)
{
    if (root->left && root->right) {
        return root->left->val + root->right->val;
    } else if (root->left) {
        return root->left->val;
    } else if (root->right) {
        return root->right->val;
    } else {
        return 0;
    }
}

void sumEven(struct TreeNode* root, int* sum)
{
    if (!root)
        return;
    // If the root value is even, send its left and right child for  summing its grandchildrens
    if (root->val % 2 == 0) {
        if (root->left)
            *sum += sumGrandchilds(root->left);
        if (root->right)
            *sum += sumGrandchilds(root->right);
    } 
    // Traverse the tree
    sumEven(root->left, sum);
    sumEven(root->right, sum);
}

int sumEvenGrandparent(struct TreeNode* root)
{
    int sum = 0;
    sumEven(root, &sum);
    return sum;
}
