	vector<int> prevPermOpt1(vector<int>& A) {
		int size = A.size(), i;
		for(i = size - 2; i >= 0; i--) {
			if(A[i] > A[i + 1])
				break;
		}

		// it is already a smallest permutation
		if(i < 0)
			return A;

		int swap_pos = size - 1;
		while(swap_pos > i and A[swap_pos] >= A[i])
			swap_pos--;

		swap(A[i], A[swap_pos]);

		return A;
	}
