const int Modulo = 1000000007;
	
public int CountHomogenous(string s) {
	long result = 0;
    long count = 1;
    for(int i=1; i<s.Length; i++)
    {
        if (s[i] != s[i-1])
        {
            result += count * (count+1)/2;
			count = 0;
        }
		count++;               
    }        
    result += count * (count+1)/2;  
    return (int)(result%Modulo);
}
