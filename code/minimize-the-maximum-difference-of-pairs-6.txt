class Solution {
public:
    int minimizeMax(vector<int>& nums, int p) 
    {
        if(p==0)return 0;
        int n=nums.size();
        sort(nums.begin(),nums.end());
        int l=0,r=nums[n-1]-nums[0]+1;
        while(l<r)
        {
            int mid=(l+r)>>1;
            int cnt=0;
            for(int i=0;i+1<n;i++)
                if(nums[i+1]-nums[i]<=mid)cnt++,i++;
            if(cnt>=p)r=mid;
            else l=mid+1;
        }
        return l;
    }
};
