class Solution {
public:
    int val;
    map<int,int>mp;
    int countMaxOrSubsets(vector<int>& nums) 
    {   
         val=INT_MIN;
         int idx=0;
         fun(nums,idx,0);
         return mp[val];
    }
    void fun(vector<int>&nums,int idx,int ele)
    {
        if(idx >= nums.size())
        {
            val=max(val,ele);
            mp[ele]++;
            return;
        }
        fun(nums,idx+1,ele|nums[idx]);
        fun(nums,idx+1,ele);
    }
};
