int scoreOfParentheses(char *s)
{
    int result = 0, lvl = 0;
    while (*s)
    {
        if (*s == '(')
        {
            lvl++;
        }
        else
        {
            lvl--;
            if (*(s - 1) == '(')
            {
                result += 1 << lvl;
            }
        }
        s++;
    }

    return result;
}
