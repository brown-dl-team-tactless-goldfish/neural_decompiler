class Solution {
public:
    bool findSubarrays(vector<int>& nums) {
        int n = nums.size();
        if (n == 2)
            return false;
        unordered_set<int> s;
        int k = nums[0];
        for (int i = 1; i < n; i++)
        {
            k += nums[i];
            if (s.count(k))
                return true;
            s.insert(k);
            k -= nums[i - 1];
        }
        return false;
    }
};
