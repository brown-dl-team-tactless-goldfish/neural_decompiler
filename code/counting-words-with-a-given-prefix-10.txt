int prefixCount(char ** words, int wordsSize, char * pref) {

	char buff[strlen(pref)+1];

	register unsigned char i, sum = 0;
	for(i = 0; i < wordsSize; i++) {
        if(strlen(&words[i][0]) >= strlen(pref)) {
            memcpy(buff, &words[i][0], strlen(pref));
            buff[strlen(pref)] = '\0';
            if(strcmp(pref, buff) == 0) {
                sum++;
            }
        }
	}
	return sum;
}
