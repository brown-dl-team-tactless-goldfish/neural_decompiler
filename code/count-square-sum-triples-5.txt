    int c =  0;
    
    for(int i=1; i<=n; i++)
    {
        for(int j=i+1; j<=n; j++)
        {
            int a = (i*i) + (j*j);
            
            int sq = sqrt(a);
            
            if(sq* sq == a && sq<= n)
            {
                c = c + 2;
            }
        }
    }
    
    return c;
