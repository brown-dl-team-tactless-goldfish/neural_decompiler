 vector<int> getSumAbsoluteDifferences(vector<int>& A ) {
        int T = accumulate( A.begin() , A.end() , 0 );
        int N = A.size();
        vector<int>res;
        int P[N];
        P[0] = A[0] ;
        
        for( int i = 1 ; i < N ; i++ )
            P[i] = P[i-1] + A[i];
        
        for( int i = 0 ; i < N ; i++ )
        {
            int a = (i+1)*A[i] - P[i];
            a += P[N-1]-P[i] - (N-i-1)*A[i];
            res.push_back(a);
        }
        return res;
    }
