public class Solution {
    public int MaximizeGreatness(int[] nums) {
        Array.Sort(nums);
        Array.Reverse(nums);
        int[] dup = nums;
        int n = nums.Length;
        int res = 0;
        
        for(int i = 0; i < n; i++)
        {
            if(nums[i] < dup[res])
                res++;
        }
        
        return res;
    }
}
