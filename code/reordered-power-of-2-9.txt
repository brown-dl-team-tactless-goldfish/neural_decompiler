class Solution {
public:
    long counter(int n) {
        long res = 0;
        while (n != 0) {
            res += pow(10, n % 10);
            n /= 10;
        }
        return res;
    }
    bool reorderedPowerOf2(int N) {
        long arr = counter(N);
        for (int i = 0; i < 32; i++) {
            if ( counter(1 << i) == arr )
                return true;
        }
        return false;
    }
};
