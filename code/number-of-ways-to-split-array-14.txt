class Solution {
public:
    int waysToSplitArray(vector<int>& nums) {
        long long r = accumulate(nums.begin(),nums.end(),0ll);
        long long l = 0; 
        int ret = 0; 
        for(int i = 0;i<nums.size()-1;i++) {
            l += nums[i];
            r -= nums[i];
            if(l >= r) {
                ret++;
            }
        }
        return ret; 
    }
};
