int singleNumber(int* nums, int n){
    int ans = nums[0];
    for(int i = 1; i < n; i++){
        //Because the XOR operator forms an abelian group, we can freely
        //swap the order of operations (commutative). In addition, 
        //since every member of the group is it's own inverse, a^a = 0, we
        //can show that: a^b^c^a^c => (a^a)^(c^c)^b => (0^0)^b => b. 
        ans ^= nums[i];
    }
    return ans;
}
