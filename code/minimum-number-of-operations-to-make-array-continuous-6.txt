class Solution {
public:
    int minOperations(vector<int>& A) {
        sort(begin(A), end(A));
        queue<int> q;
        int N = A.size(), res = N;
        for(int i = 0; i < N; i++) {
            if(!q.empty() && A[i] == A[q.back()]) continue;
            while(!q.empty() && A[i]-A[q.front()] >= N) q.pop();
            q.push(i);
            res = min(res, N-(int)q.size());
        }
        return res;
    }
};
