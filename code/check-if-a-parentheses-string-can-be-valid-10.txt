    bool canBeValid(string s, string l) {
        auto func = [&](char c, bool reverse) {
            for(int j = 0, i = 0, count = 0; j < s.size(); j++) {
                i = reverse ? s.size()-j-1 : j;
                if(l[i] == '0' || s[i] == c) count++;
                else if(--count < 0) return false;
            }
            return true;
        };
        return s.size()%2 == 0 && func('(', false) && func(')', true);
    }
