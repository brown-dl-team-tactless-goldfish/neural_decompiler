bool checkIfCanBreak(string s1, string s2) {

	vector<int> cnt1(26, 0), cnt2(26, 0);
	bool b1 = true, b2 = true;
	int sumVal1 = 0, sumVal2 = 0;
	for (char& c : s1) ++cnt1[c - 'a'];
	for (char& c : s2) ++cnt2[c - 'a'];

	for (int i = 25; i >= 0; --i) {
		sumVal1 += cnt1[i];
		sumVal2 += cnt2[i];
		if (sumVal1 < sumVal2) b1 = false;
		if (sumVal2 < sumVal1) b2 = false;
	}

	return (b1 || b2);
}
