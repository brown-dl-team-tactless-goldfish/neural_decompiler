class Solution {
public:
    int minimumDeletions(vector<int>& nums) {
          int p1 = min_element(a.begin(), a.end()) - a.begin();
        int p2 = max_element(a.begin(), a.end()) - a.begin();
        int n = a.size();
        if(p1 > p2)
            swap(p1, p2);
        if(n == 1) 
            return 1;
        return min({max(p1, p2) + 1, n - min(p1, p2), p1 + 1 + n - p2});
    }
};
