#include<math.h>

int findNthDigit(int n) {
    
    if (n < 10) return n;
    
    long int x = 9;
    int i = 1;
    long int j = 1;
    while(n>x) {
        n -= x;
        i++;
        j = j * 10;
        x = i*j*9;
    }
        
    if (n%i == 0) {
        n /= i;
        int ans;
        ans = n%10;
        if (ans==0) return 9;
        return ans-1;
    }
    
    long int num;
    
    num = j+(n/i);
    
    int d = n%i;
    
    int p = pow(10,(d-1));
    return (num/(j/p))%10;  
}
