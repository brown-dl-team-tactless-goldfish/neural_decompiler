int getMaximumGenerated(int n){
    if (n < 1 | n > 100)
        return 0;
    
    int *arr = (int*)malloc((n+1)*sizeof(int));
    arr[0] = 0;
    arr[1] = 1;
    int count = 1;
    int maxNum = 0;
    while (count <= n) {
        if (count%2)
            arr[count]=arr[(count/2)+(count%2)] + arr[count/2];
        else
            arr[count] = arr[count/2];
        if (maxNum < arr[count])
            maxNum = arr[count];
        count++;
    }
    return maxNum;
}
