class Solution {
public:
    int tupleSameProduct(vector<int>& nums) {
        int n = nums.size();
        unordered_map<int,int> m;
        int ans = 0;
        for(int i = 0 ; i < n ; ++i){
            for(int j = 0 ; j < i ; ++j){
                int p = nums[i] * nums[j];
                ans += 8 * m[p];
                m[p]+=1;
            }
        }
        return ans;
    }
};
