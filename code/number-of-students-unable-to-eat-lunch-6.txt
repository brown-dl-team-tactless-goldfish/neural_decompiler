int countStudents(int* students, int studentsSize, int* sandwiches, int sandwichesSize){
    int pref[2] = {0};
    for (int i = 0; i < studentsSize; i++)
    {
        pref[students[i]]++;
    }
    for (int i = 0; i < sandwichesSize; i++)
    {
        pref[sandwiches[i]]--;
        if (pref[sandwiches[i]] < 0)
            return sandwichesSize - i;
    }
    return 0;
}
