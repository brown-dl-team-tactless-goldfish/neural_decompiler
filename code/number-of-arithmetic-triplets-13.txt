int arithmeticTriplets(int* nums, int numsSize, int diff){
    int count = 0, i, map[300] = {0};
    for (i = 0; i < numsSize; ++i) ++map[nums[i]];
    for (i = 0; i < numsSize - 2; ++i) {
        int a = nums[i];
        int b = a + diff;
        int c = a + 2 * diff;
        if (map[b] && map[c]) ++count;
    }
    return count;
}
