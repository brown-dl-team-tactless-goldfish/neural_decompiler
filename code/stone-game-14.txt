class Solution {
public:
    using VI = vector<int>;
    using VVI = vector<VI>;
    bool stoneGame(VI& A) {
        int N = A.size();
        VVI dp(N, VI(N)); // memo ðŸ¤”
        for (auto i{ N - 1 }; 0 <= i; --i) // i-th first stone
            for (auto j = i + 1; j < N; ++j) // j-th last stone
                dp[i][j] = max(A[i] - dp[i + 1][j], A[j] - dp[i][j - 1]); // max(first, last) ðŸŽ¯
        return dp[0][N - 1];
    }
};
