class Solution {
public:
    long long ProductLessThanK(vector<int>& nums,long long k) {
        map<int,int> mp;
        int n=nums.size();
        long long prod=0;
        long long ans=0;
        int j=0;
        for(int i=0;i<n;i++)
        {
            prod+=nums[i];
            while((long long)prod*(long long)(i-j+1)>=k)
            {
                prod-=nums[j++];
            }
            ans+=i-j+1;
        }
        return ans;
    }
    long long countSubarrays(vector<int>& nums, long long k) {
        long long ans=ProductLessThanK(nums,k);
        return ans;
    }
};
