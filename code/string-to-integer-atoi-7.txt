int myAtoi(char * s){
    int i=0;
    long long number = 0;
    int pos_neg = 1;
    while(s[i] == ' ')
        i++;
    if(s[i] == '+' || s[i] == '-')
       pos_neg = 1 - 2 * (s[i++] == '-');
    
    while('0' <= s[i] && s[i] <= '9'){
        number = number * 10 + (s[i++] - '0');
        if(number > INT_MAX || number * -1 < INT_MIN){
            return pos_neg == -1 ? INT_MIN : INT_MAX;
        }
    }
    return number * pos_neg;
    
}
