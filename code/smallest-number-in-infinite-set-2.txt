class SmallestInfiniteSet {
public:
    int dp[1001];
    SmallestInfiniteSet() {
        memset(dp,1,sizeof(dp));
    }
    
    int popSmallest() {
        int t;
        for(int i=1;i<1001;i++)
            if(dp[i])
            {
                dp[i]=0;
                t=i;
                break;
            }
        return t;        
    }
    
    void addBack(int num) {
        dp[num]=1;
    }
};
