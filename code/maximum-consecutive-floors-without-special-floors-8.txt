    //Method 1
    public int MaxConsecutive(int bottom, int top, int[] special)
    {
        Array.Sort(special);
        int result = Math.Max(special.First() - bottom, top - special.Last());
        for (int i = 1; i < special.Length; i++)
        {
            result = Math.Max(result, special[i] - special[i - 1] - 1);
        }
        return result;
    }
	
	
	
   //Method 2
    public int MaxConsecutive(int bottom, int top, int[] special)
    {
        Array.Sort(special);
        int result = 0;
        foreach (int spe in special)
        {
            if (bottom < spe)
                result = Math.Max(result, spe - bottom);
            bottom = spe + 1;
        }
        result = Math.Max(result, top - special.Last());

        return result;
    }
