bool checkTree(struct TreeNode* root){
    //Access the values of the left and right leafs with the arrow operators 
    //Add the values together and save it in the sum variable
    int sum = root->left->val + root->right->val; 
    
    //Return true if sum is the same as root value, else return false
    if(sum == root->val){
        return true;
    }
    return false;
}
