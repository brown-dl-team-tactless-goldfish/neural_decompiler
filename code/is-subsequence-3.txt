bool isSubsequence(char * s, char * t){
    int i, trace, j;

    i = 0;
    trace = -1;
    while (1)
    {
        j = trace + 1;
        while (1)
        {
            if (t[j] == s[i])
            {
                trace = j;
                break;
            }
            if (t[j] == '\0')
                return 0;
            j++;
        }
        if (s[i] == '\0')
            return 1;
        i++;
    }
    return 0;
}
