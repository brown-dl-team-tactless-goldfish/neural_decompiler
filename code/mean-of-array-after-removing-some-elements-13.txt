class Solution 
{
public:
    double trimMean(vector<int>& arr)
    {
        sort(arr.begin(),arr.end());
        int amount=(arr.size()*5)/100,sum=0,count=0;
        for(int i=amount;i<arr.size()-amount;i++)
        {
            sum+=arr[i];
            count++;
        }
        return (double)sum/(double)count;
    }
};
