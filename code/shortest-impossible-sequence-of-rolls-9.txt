class Solution {
public:
    static int shortestSequence(const vector<int>& rolls, const int k) {
        vector<bool> arr(k, false);
        int ret = 1;
        int c = 0;
        for (auto i : rolls)
            if (!arr[i - 1]) {
                arr[i - 1] = true;
                if (++c >= k) {
                    c = 0;
                    ++ret;
                    arr = vector<bool>(k, false);
                }
            }
        return ret;
    }
};
