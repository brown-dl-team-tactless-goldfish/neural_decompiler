class Solution {
public:
    int longestSubsequence(string s, int k) {
        string best = "";
        for(int i = s.size()-1; i >= 0; i--){
            string c = "";
            c.push_back(s[i]);
            c = c + best;
            bitset<32> b(c);
            long long res = b.to_ullong();
            if(res <= k) best = c;
        }
        return best.size();
    }
};
