public class Solution {
    public int ConvertTime(string current, string correct) {
        
        int diff = int.Parse(correct.Substring(0,2)) * 60 - int.Parse(current.Substring(0,2)) * 60
            + int.Parse(correct.Substring(3,2)) - int.Parse(current.Substring(3,2));
        
        int cnt = 0;  
        int[] arr = new int[]{60, 15, 5, 1};
        foreach(var val in arr)
        {
            cnt  += diff / val;
            diff %= val;
        }

        return cnt;
    }
}
