int maxAbsValExpr(vector<int>& a1, vector<int>& a2, int res = 0) {
  for (auto s1 : { -1, 1 })
    for (auto s2 : { -1, 1 })
      for (auto i = 0, m = 0; i < a1.size(); ++i) {
        auto val = a1[i] * s1 + a2[i] * s2 + i;
        res = max(res, val - m);
        m = min(m, val);
      }
  return res;
}
