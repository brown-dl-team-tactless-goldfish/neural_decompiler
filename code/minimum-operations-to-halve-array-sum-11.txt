class Solution {
public:
    int halveArray(vector<int>& nums) {
        priority_queue<double> pq;
        double totSum = 0, reqSum = 0;
        
        for(int val: nums) {
            totSum += val;
            pq.push(val);
        }
        
        reqSum = totSum / 2;
        int oprs = 0;
        
        while(totSum > reqSum) {
            double currMax = pq.top();
            pq.pop();
            
            currMax /= 2;
            totSum -= currMax;
            pq.push(currMax);
            oprs++;
        }
        
        return oprs;
    }
};
