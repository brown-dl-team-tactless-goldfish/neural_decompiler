int minAddToMakeValid(char * S){
    if(!S || strlen(S)==0)return 0;
    char* stack=(char*)calloc(1001,sizeof(char));
    int top=-1;
    stack[++top]=S[0];
    int size=1;
    for(int i=1;i<strlen(S);i++){
        if(top>=0){
            if(stack[top]=='('){
                if(S[i]==')'){
                    top--;
                    size--;
                    continue;
                } 
            }    
        }
        stack[++top]=S[i];
        size++;
        
    }
    free(stack);
    return size;
}
