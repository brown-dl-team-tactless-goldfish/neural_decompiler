class Solution 
{
public:
    int reductionOperations(vector<int>& nums) 
    {
        map<int,int,greater<int>>m;
        for(int i=0;i<nums.size();i++)
        {
            m[nums[i]]++;
        }
        int sesh=m.size()-1,op=0;
        for(auto it:m)
        {
            int x=it.second;
            while(x)
            {
                op+=sesh;
                x--;
            }
            sesh--;
        }
        return op;
    }
};
