int cmp(const void *x, const void *y)
{
    int a = *(int*)x;
    int b = *(int*)y;
    return a-b;
}
bool asteroidsDestroyed(int mass, int* asteroids, int asteroidsSize)
{
    long long sum = mass;
    qsort(asteroids, asteroidsSize, sizeof(int), cmp);
    for(int i = 0; i < asteroidsSize; i++)
    {
        if(sum >= asteroids[i])
        {
            sum = sum + asteroids[i];
        }
        if(sum < asteroids[i])
        {
            return false;
        }
    }
    return true;
}
