bool canBeIncreasing(int* nums, int numsSize){
  bool removed = nums[0] >= nums[1];
  for (int *i = nums + (removed ? 2 : 1); i != nums + numsSize; ++i) {
    if (*(i - 1) < *i)
      continue;
      
    if (removed)
      return false;

    if (i + 1 == nums + numsSize)  // end of list
      return true;
    
    // test by removing i - 1 or i
    if (*(i - 2) < *i || *(i - 1) < *(i + 1))
      removed = true;
    else
      return false;
  }
  
  return true;
}
