class Solution {
public:
    int mostFrequent(vector<int>& nums, int key) {
        unordered_map<int,int>m;
        for(int i=0;i<nums.size();i++)
        {
            if(nums[i]==key && i<nums.size()-1)
            {
                m[nums[i+1]]++;
            }
        }
        int a=INT_MIN,ans;
        for(auto it:m)
        {
            if(it.second>a)
            {
                a=it.second;
                ans=it.first;
            }
        }
        return ans;
    }
};
