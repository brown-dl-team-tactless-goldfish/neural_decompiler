    int destroyTargets(vector<int>& nums, int space) {
        unordered_map<int,pair<int,int>> m; //mod, {cnt, min}
        int maxCnt = 0, xmin = 1e9;
        for(auto x:nums){
            auto& [cnt, xmin] = m[x%space];
            cnt++;
            maxCnt = max(maxCnt, cnt);
            if(xmin == 0) xmin = x;
            else xmin = min(xmin, x);
        }
        for(auto& [mod, p] : m)
            if(p.first == maxCnt) xmin = min(xmin, p.second);
        return xmin;
    }
