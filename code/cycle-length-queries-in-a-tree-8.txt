class Solution {
public:
    vector<int> cycleLengthQueries(int n, vector<vector<int>>& queries) {
        vector<int> ans;
        for(auto q : queries) {
            int a = q[0], b = q[1], l = 1;
            while(a != b) {
                a > b ? a /= 2 : b /= 2;
                l++;
            }
            ans.push_back(l);
        }
        return ans;
    }
};
