int* rowAndMaximumOnes(int** mat, int matSize, int* matColSize, int* returnSize) {
    *returnSize = 2;
    int* ans = (int*) calloc(2, sizeof(int));
    for (int i = 0; i < matSize; i++) {
        int cnt = 0;
        for (int j = 0; j < matColSize[i]; j++) {
            cnt += mat[i][j];
        }
        if (cnt > ans[1]) {
            ans[1] = cnt;
            ans[0] = i;
        }
    }
    return ans;
}
