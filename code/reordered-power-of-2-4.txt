bool reorderedPowerOf2(int n)
{
    int a[10]={0},k=0,map[30][10]={0};
    while(n)
    {
        a[n%10]++;
        n/=10;
    }
    int l=0;
    for(int i=1;i<=pow(10,9);i*=2)
    {
        int k=i;
        while(k)
        {
            map[l][k%10]++;
            k/=10;
        }
        l++;
    }
    l=0;
    for(int i=1;i<=pow(10,9);i*=2)
    {
        k=0;
        for(int j=0;j<10;j++)if(a[j]!=map[l][j])k=1;
        if(k==0)return true;
        l++;
    }
    return false;
}
