int min(int a, int b) {
    if(a==-1)return b;
    if(b==-1)return a;
    else return a<b?a:b;
}

int findSecondMinimumValue(struct TreeNode* root){
    if(root==NULL)return -1;
    if(root->left==NULL)return -1;
    int m=root->left->val, n=root->right->val;
    if(m==n)
    {
        m=findSecondMinimumValue(root->left);
        n=findSecondMinimumValue(root->right);
        if(m==n)return -1;
    }else{
        if(m>n) n=findSecondMinimumValue(root->right);
        else m=findSecondMinimumValue(root->left);
    }
    return min(m, n);
}
