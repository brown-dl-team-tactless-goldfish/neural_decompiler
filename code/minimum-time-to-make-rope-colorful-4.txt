#define MIN(a,b) ((a<b)?a:b)
#define MAX(a,b) ((a>b)?a:b)
/* keep the maximum time if more than two same color balloons */
int minCost(char * colors, int* neededTime, int neededTimeSize){
    int minTime = 0;
    char prevColor = colors[0];
    int prevTime = neededTime[0];
    for(int i = 1; i < neededTimeSize; i++){
        if(prevColor == colors[i]){
            minTime += MIN(prevTime, neededTime[i]);
            prevTime = MAX(prevTime, neededTime[i]);          
        }
        else {         
            prevColor = colors[i];
            prevTime = neededTime[i];
        }
    }
    return minTime;
}
