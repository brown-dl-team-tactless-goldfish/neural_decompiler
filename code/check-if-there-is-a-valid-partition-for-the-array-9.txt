class Solution {
public:
    int dp[100001];
    int n;
    bool find(int i,vector<int>&nums)
    {
        if(i>=n)
            return true;
        if(dp[i]!=-1)
            return dp[i];
        bool res=false;
        if(i+1<=n-1 && nums[i]==nums[i+1])
            res=res||find(i+2,nums);
        if(i+2<=n-1 && nums[i]==nums[i+1] && nums[i]==nums[i+2])
            res=res||find(i+3,nums);
        if(i+2<=n-1 && nums[i]+1==nums[i+1] && nums[i+1]+1==nums[i+2])
            res=res||find(i+3,nums);
        return dp[i]=res;
    }
    bool validPartition(vector<int>& nums) 
    {
        n=nums.size();
        memset(dp,-1,sizeof(dp));
        return find(0,nums);
    }
};
