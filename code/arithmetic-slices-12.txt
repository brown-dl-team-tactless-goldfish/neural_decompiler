int numberOfArithmeticSlices(int* A, int ASize)
{
    int i=0,count=0,z=0; 
    while(i<ASize-2)
    {
        if(A[i+2]-A[i+1]==A[i+1]-A[i])
        {
            count++;
            z++;
            if(z>1)
                count=count+z-1;           
        }
        else
            z=0;
        i++;
    }     
    return(count);    
}
