class Solution {
public:
    int miceAndCheese(vector<int>& r1, vector<int>& r2, int k) {
        int n=r1.size(),ans=0;
        vector<vector<int>> x(n,vector<int>(3,0));
        for(int i=0;i<n;i++) x[i]={r1[i]-r2[i],r1[i],r2[i]};
        sort(x.begin(),x.end());
        for(int i=0;i<n;i++){
            if(i>=n-k) ans+=x[i][1];
            else ans+=x[i][2];
        }
        return ans;
    }
};
