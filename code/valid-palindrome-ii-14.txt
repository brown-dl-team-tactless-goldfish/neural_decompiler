
bool check_palindrome (char* s, int* i, int* j) {
    
    while (*i <= *j) {
        if (s[*i] != s[*j])
            return false;
        (*i)++;
        (*j)--;
    }
    return true;
}

bool validPalindrome(char * s){

    int i = 0;
    int j = strlen (s) -1;
    int temp = 0;
    int index = 0;
    bool ret;
    
    
    ret = check_palindrome (s, &i, &j);
    if (ret)
        return true;
    
    temp = j;
    index = i+1;
    ret = check_palindrome(s, &index, &j);
    if (ret)
        return true;
    
    index = temp-1;
    ret = check_palindrome(s, &i, &index);
    if (ret)
        return true;
    
    return false;
    
}
