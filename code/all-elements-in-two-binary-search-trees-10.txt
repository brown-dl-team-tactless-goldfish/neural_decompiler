class Solution {
public:
    void inOrder(TreeNode* root)
    {
        if(!root) return;
        inOrder(root -> left);
        values.push_back(root -> val);
        inOrder(root -> right);
    }
    
    vector<int> getAllElements(TreeNode* root1, TreeNode* root2) {
        inOrder(root1);
        inOrder(root2);
        sort(values.begin(), values.end());
        return values;

    }
    
private:
    vector<int> values;
};
