class Solution {
public:
    long long countBadPairs(vector<int>& nums) {
        long long ans = 0;
        
        int n = nums.size();
        long long cnt = 0;
        
        for(int i = 0 ; i < n ; i++){
            nums[i] -= i;
        }
        map<int,long long> m;
        for(int i = 0 ; i < n ; i++){
            ans += cnt - m[nums[i]];
            m[nums[i]]++;
            cnt++;
        }
        return ans;
    }
};
