	// occurrence/frequency in the nums as value.
    unordered_map<int,int> m;

        m[nums[i]]++;
        
    }

    int n = nums.size()/2;

    for(auto pr : m){
        
        int freq = pr.second;

            pairsPossible += (freq/2);
            
        }else{
            
            int c = freq/2;

        }
    }
    
	// If possible pairs are equal to the required pairs then return true.
