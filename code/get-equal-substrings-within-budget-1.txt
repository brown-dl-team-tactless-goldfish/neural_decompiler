int equalSubstring(char *s1, char *s2, int cost)
{
    int max_len = 0;
    for (int left = 0, right = 0; s1[right]; right++) {
        cost -= abs(s1[right] - s2[right]);
        while (cost < 0) {
            cost += abs(s1[left] - s2[left]);
            left++;
        }
        int t_max_len = right - left + 1;
        max_len = t_max_len > max_len ? t_max_len : max_len;
    }

    return max_len;
}
