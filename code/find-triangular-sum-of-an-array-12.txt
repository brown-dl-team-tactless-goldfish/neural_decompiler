class Solution {
public:
    int triangularSum(vector<int> nums) {
        int count = 0 , i = 0 ;
        while(nums.size() > 1){
            nums[i] = (nums[i] + nums[i+1])%10 ;
            count ++ ; ++i ; 
            if(count == nums.size() - 1){
                nums.pop_back() ;
                count = 0 ;
                i = 0 ;
            }
        }
        return nums.back() ;
    }
};
