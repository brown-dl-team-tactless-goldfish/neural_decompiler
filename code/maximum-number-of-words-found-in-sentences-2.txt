int mostWordsFound(char ** sentences, int sentencesSize)
{    
    int nWords, max;
    //There exists at least one word, therefore init # words and max to 1
    nWords = max = 1;

    //first loop traverses each pointer to char
    for(int i = 0; i < sentencesSize; i++) {
        //Second loop traverses each char.
        for(int j = 0; sentences[i][j] != '\0'; j++) {
            //If c = space, then we have left a word, therefore # words ++
            if(sentences[i][j] == ' ')
                nWords++;
        }
        //if # words is largest, save it
        if(nWords > max)
            max = nWords;
        //After leaving innermost loop we have finished counting # words
        //in sentence i. the next sentence is i + 1, therefore we have
        //to set # words back to 1.
        nWords = 1;
    }
    return max;
}
