int min(int a, int b) { return a < b ? a : b; }
int max(int a, int b) { return a > b ? a : b; }

int rec(int *v, int begin, int end, int (*compare)(int, int)) {
    int m;
    if (begin == end) return v[begin];
    m = (begin + end) / 2;
    return compare(rec(v, begin, m, min), rec(v, m+1, end, max));
}

int minMaxGame(int* nums, int numsSize){
    return rec(nums, 0, numsSize-1, min);
}
