int maxDepth(struct TreeNode* root){
    if (root == NULL)
        return 0;
    else{
        int treeleft, treeright;
        treeleft = maxDepth(root->left);
        treeright = maxDepth(root->right);
        if (treeleft < treeright)
            return treeright + 1;
        else
            return treeleft + 1;
    }
}
# python solution : 60ms
class Solution:
     def maxDepth(self, root: TreeNode) -> int:
        if root == None:
             return 0
         return max(self.maxDepth(root.left), self.maxDepth(root.right)) + 1
