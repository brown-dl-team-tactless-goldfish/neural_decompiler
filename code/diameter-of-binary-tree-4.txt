int MAX(int a,int b)
{
    return (a<b) ? b : a;
}
int height(struct TreeNode* root)
{
	if(root==NULL)
	{
		return -1;
	}
	int lh=height(root->left);
	int rh=height(root->right);
	return MAX(lh,rh)+1;
}

int diameterOfBinaryTree(struct TreeNode* root){
if(root==NULL)
	{
		return 0;
	}
	
	int lh=height(root->left);
	int rh=height(root->right);
	//printf("%d and %d",lh,rh);
	return MAX(MAX(diameterOfBinaryTree(root->left),diameterOfBinaryTree(root->right)),lh+rh+2);
}
