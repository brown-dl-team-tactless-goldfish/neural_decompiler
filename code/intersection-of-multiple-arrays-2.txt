class Solution {
public:
    vector<int> intersection(vector<vector<int>>& nums) {
        int check[1001] = {0};
        for(auto num : nums) {
            for(auto i : num) {
                check[i]++;
            }
        }
        vector<int> ans;
        for(int i = 0; i != 1001; i++) {
            if(check[i] == nums.size())
                ans.push_back(i);
        }
        return ans;
    }
};
