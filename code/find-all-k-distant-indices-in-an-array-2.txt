
vector<int> findKDistantIndices(vector<int>& a, int key, int k) {

	int n = a.size();

	vector<int> ans;

	int r = 0;
	for(int i=0;i<n;i++){

		if(a[i] != key) continue;


		int left = max(r, i-k);
		int right = min(i+k,n-1);

		for(int j=left;j<=right; j++)
			ans.push_back(j);

		r = right+1;
	}

	return ans;

}
