int* countSmaller(int* nums, int numsSize, int* returnSize){
    
     int* arr=malloc(sizeof(int)*numsSize);
    *returnSize=numsSize;
    int count;
    int i,j;
       
    for(i=0;i<numsSize;++i){
        count=0;
        for( j=i+1;j<numsSize;++j){
            if(nums[j]<nums[i]){
                count++;
            }
            
        }
          arr[i]=count;
    
           
      }
    return arr;
}
