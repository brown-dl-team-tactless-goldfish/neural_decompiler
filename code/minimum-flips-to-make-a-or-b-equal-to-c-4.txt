int minFlips(int a, int b, int c) {
    // get bits, where a | b are different with c
    int dif = (a | b) ^ c;
	// get number of bits, where c has 1s and a | b 0s
    int set_ones = __builtin_popcount(dif & c);
	// get number of bits, where c has 0s and a has 1s, c has 0s and has b 1s,
    int set_zeros = __builtin_popcount(dif & a) + __builtin_popcount(dif & b);
    return set_ones + set_zeros;
}
