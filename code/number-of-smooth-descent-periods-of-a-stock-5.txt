public class Solution {
    public long GetDescentPeriods(int[] prices) {
        long r = prices.Length;
        long a = 0;
        for (long i = 0; i < prices.Length; ++i) {
            if (i > 0 && prices[i - 1] - prices[i] == 1) {
                r += ++a;
            } else {
                a = 0;
            }
        }
        
        return r;
    }
}
