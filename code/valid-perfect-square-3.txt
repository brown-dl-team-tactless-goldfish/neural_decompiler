bool isPerfectSquare(int num){
    int left, right, mid;
    
    left = 1;
    right = num;
    
    while (left <= right) {
        mid = left + (right-left) / 2;
        long midsqr = (long)mid * (long)mid;
            
        if (midsqr == num) {
            return true;
        }
        else if (midsqr < num) { // mid < num/mid   may work.. but integer '/' is different from math as we know... 
            left = mid+1;
        }
        else if (midsqr > num) {
            right = mid-1;
        }
    }
    
    return false;
}
