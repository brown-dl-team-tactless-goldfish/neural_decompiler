int maxVowels(string s, int k) {
    
    int a[26]={};
    int n=s.length();
    int ans=0;
    for(int i=0;i<n;i++)
    {
        if(i<k)
        {
         a[s[i]-'a']++;   
        }
        else
        {
            a[s[i-k]-'a']--;
             a[s[i]-'a']++;  
        }
        
        int sum=0;
        sum+=a['a'-'a'];
        sum+=a['e'-'a'];
        sum+=a['i'-'a'];
        sum+=a['o'-'a'];
        sum+=a['u'-'a'];
        ans=max(sum,ans);
    }
    
     return ans;
    }
