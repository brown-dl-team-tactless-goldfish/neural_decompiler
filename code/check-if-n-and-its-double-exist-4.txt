bool checkIfExist(int* arr, int arrSize){
    for (int i = 0; i < arrSize; i++) {
        if (!(arr[i] % 2)) {
            int j = (i + 1) % arrSize;
            while (i != j) {
                if (arr[j++] * 2 == arr[i])
                    return true;
                j %= arrSize;
            }
        }
    }
    return false;
}
