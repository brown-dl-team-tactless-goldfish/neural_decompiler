float find_rate(int** points){
    if(points[1][0]==points[0][0]){
        return (float)(INT_MAX);
    }
    return ((float)(points[1][1]-points[0][1]))/((float)(points[1][0]-points[0][0]));
}
bool equal(int** points){
    if(points[1][1]==points[0][1]&&points[1][0]==points[0][0]){
        return true;
    }
    return false;
}
bool isBoomerang(int** points, int pointsSize, int* pointsColSize){
    if(equal(points)||equal(&points[1])){
        return false;
    }
    float k1 =find_rate(points);
    float k2 =find_rate(&points[1]);
    return k1==k2?false:true;
}
