SELECT DISTINCT a.*  FROM Stadium a JOIN Stadium b JOIN Stadium c 
ON (a.id=b.id-1 AND a.id=c.id-2)  # condition for lowest number
OR (a.id=b.id-1 AND a.id=c.id+1)  # condition for middle number
OR (a.id=b.id+1 AND a.id=c.id+2)  # condition for largest number
WHERE a.people>=100 AND b.people>=100 AND c.people>=100 # condition 
ORDER BY visit_date
