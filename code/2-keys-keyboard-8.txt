int minSteps(int n) {
    if(n==1){return 0;}
    if(n<5){return n;}
    int ret=2;
    int copyLength=1;
    int length=2;
    while(length<n){
        if((n-length)%length==0){
            copyLength=length;
            ret+=2;
            length=length*2;
        }else{
            length+=copyLength;
            ret++;
        }
    }
    return ret;
}
