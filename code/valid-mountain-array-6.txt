bool validMountainArray(int* arr, int arrSize)
{
    
    int flag = 0;
    if ( arrSize < 3)
        return false;
    
    for ( int i = 1 ; i < arrSize-1 ; i++)
    {
        if ( arr[i]  > arr[i-1] && arr[i] < arr[i+1] && flag == 0);
         
         else if ( arr[i]  > arr[i-1] && arr[i] > arr[i+1] && flag == 0 )
             flag = 1;
        else if ( arr[i]  < arr[i-1] && arr[i] > arr[i+1] && flag == 1);
            else
                return false;
    }
    
    return flag;
    
}
