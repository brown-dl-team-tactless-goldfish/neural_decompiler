class Solution {
public:
    int countDistinct(vector<int>& nums, int k, int p) {
        set<vector<int>>s;
        for(int i=0;i<nums.size();i++){
            int cnt=0;
            for(int j=i;j<nums.size();j++){
                if(nums[j]%p==0)    cnt++;
                if(cnt==k+1){break;}
                s.insert(vector<int>(nums.begin()+i,nums.begin()+j+1));
            }
        }
        return s.size();
    }
};
