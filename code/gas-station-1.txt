int canCompleteCircuit(int* gas, int gasSize, int* cost, int costSize){

    int tsum = 0, sum = 0, index = 0;
    
    for(int i = 0; i < gasSize; ++i)
    {
        int val = *(gas + i) - *(cost + i);
        tsum += val;
        sum += val;
        
        if(sum < 0)
        {
            sum = 0;
            index = i+1;
        }
    }
    
    return ((tsum < 0) ? -1 : index);    
}
