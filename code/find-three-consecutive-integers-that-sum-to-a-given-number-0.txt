long long* sumOfThree(long long num, int* returnSize){
    if((num - 3) %3 != 0){
        *returnSize = 0;
        return NULL;
    }
        
    *returnSize = 3;
    long long* ans = malloc(3 * sizeof(long long));
    ans[0] = (num - 3)/3;
    ans[1] = ans[0] + 1;
    ans[2] = ans[1] + 1;
    return ans;
}
