class Solution:
    def countCharacters(self, words: List[str], chars: str) -> int:
        chars = ''.join(sorted(chars))
        # print ("chars = ", chars)
        
        def canBuild(remainS, remainTarget):
            if (remainS == ""):
                return True
            elif (remainTarget == ""):
                return False
            elif(remainS[0] == remainTarget[0]):
                return canBuild(remainS[1:], remainTarget[1:])
            else:
                return canBuild(remainS, remainTarget[1:])
            
        res = 0
        
        for word in words:
            word = ''.join(sorted(word))
            
            if (canBuild(word, chars) == True):
                res += len(word)
                
        return res
