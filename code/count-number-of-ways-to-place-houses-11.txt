class Solution {
public:
    int countHousePlacements(int n) {
        long long mod = 1e9 + 7;
        vector<long long> dp(n + 1, 0);
        dp[0] = 1;
        for (int i = 1; i <= n; i++)
            dp[i] = (dp[i - 1] + (i >= 2 ? dp[i - 2] : 1)) % mod;
        return (dp[n] * dp[n]) % mod;
    }
};
