class Solution:
    def findGCD(self, nums: List[int]) -> int:
        min_n = min(nums)
        max_n = max(nums)
        
        print ("min_n = ", min_n, " max_n = ", max_n)
        
        res = 1
        for n in range(1, min_n+1):
            print ("n = ", n)
            if (min_n % n == 0 and max_n % n == 0):
                res = max(res, n)
                
        return res
