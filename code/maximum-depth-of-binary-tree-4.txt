int maxDepth(struct TreeNode* root){
    //If there are no more nods left
    if(root == NULL)
        return 0;
    
    //use recursion to go to end of tree nodes
    int ldepth = maxDepth(root->left);
    int rdepth = maxDepth(root->right);
    
    //if the left branch is greater than the right branch
    if(ldepth > rdepth)
        return ldepth+1 ;
    else
        return rdepth+1 ;
}
