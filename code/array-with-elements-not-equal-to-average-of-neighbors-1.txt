int* rearrangeArray(int* nums, int numsSize, int* returnSize){
    int flag = 1; // to check if array is arranged has requirements.
    *returnSize = numsSize;
    while(flag) // to recheck the array.
    {
        flag = 0;
        for(int i = 1; i < numsSize - 1; i++) 
        {
            if((nums[i-1] + nums[i+1])/2 == nums[i]) // if condition is true then swap the array i and i+1.
            {
                int temp = *(nums + i);
                *(nums + i) = *(nums + (i+1));
                *(nums + (i+1)) = temp;
                flag = 1;  // its going to continue untill the array is arranged has per the requirements.
            }
        }
    }
    return nums;
}
