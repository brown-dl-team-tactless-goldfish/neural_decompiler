class Solution {
public:
    int twoEggDrop(int n) {
        int min_{0};
        std::vector<int> dp(n+1, 42069420);
        dp[0] = 0, dp[1] = 1, dp[2] = 2;
        for(int i{3}; i <= n; ++i)
            for(int j{1}; j <= i; ++j)
                dp[i] = std::min(dp[i], std::max(j,  1 + dp[i-j]));
        
        return dp[n];
    }
};
