class Solution {
public:
    long long childrenServed(vector<int>& candies, long long pilesSize){
        
        long long children=0;
        for(auto c : candies)
            if(c >= pilesSize)
                children+=(c/pilesSize);
        return children;
    }
    
    int maximumCandies(vector<int>& candies, long long k) {
        int maxVal=0;
        for(auto c : candies)
           maxVal = max(maxVal, c);
       
    
        
        long long ans=0;
        long long l = 1, r = maxVal;
        while(l <= r)
        {
            long long m = l+(r-l)/2;
          
            if(childrenServed(candies, m) >= k)
            {
                ans = m;
                l = m+1;
            }
            else
            {
                r = m-1;
            }
        }
        
        return ans;
    }
};
