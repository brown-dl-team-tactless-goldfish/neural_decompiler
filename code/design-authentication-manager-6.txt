class AuthenticationManager {
public:
    int time; map<string, int> mp;
    AuthenticationManager(int timeToLive) {
        time = timeToLive;
    }
    
    void generate(string tokenId, int currentTime) {
        mp[tokenId] = currentTime+time;  
    }
    
    void renew(string tokenId, int currentTime) {
        if(mp.find(tokenId) == mp.end()) return; 
        
        else{
            if(mp[tokenId] < currentTime) return;
            
            else if(mp[tokenId] > currentTime) mp[tokenId] = currentTime+time;
        }
    }
    
    int countUnexpiredTokens(int currentTime) {
        int count = 0;
        for(auto i: mp){
            if(i.second> currentTime) count++;
        }
        return count;
    }
};
