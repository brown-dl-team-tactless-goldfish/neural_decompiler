int countGoodTriplets(int* arr, int arrSize, int a, int b, int c){
    register unsigned int i, j, k, ct = 0;
	for(i = 0; i < arrSize; i++)
		for(j = 0; j < arrSize; j++)
			for(k = 0; k < arrSize; k++)
				if( (j < k && i < j) &&
                    (abs(arr[i] - arr[j]) <= a) &&
					(abs(arr[j] - arr[k]) <= b) &&
					(abs(arr[i] - arr[k]) <= c) )
                            ct++;
	return ct;
}
