    int max = 0, min = 0, steps = 0;;
    
    while(max < n)
    {
        for(int i=0; i<ranges.size(); i++)
        {
            if(i-ranges[i] <= min && i + ranges[i] > max)
            {
                max = i + ranges[i];
            }
        }
    
        
        if(min == max)
        {
            return -1;
        }
        
         steps++;
        
        min = max;
        
        
    }
    
    return steps;
    
}
