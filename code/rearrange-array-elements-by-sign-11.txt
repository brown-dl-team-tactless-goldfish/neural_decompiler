class Solution {
public:
  vector<int> rearrangeArray(vector<int>& nums) {
    vector<int>p,n;
    
    for(auto x: nums)
      if(x>0) p.push_back(x);
      else n.push_back(x);
    
    for(int i = 0, j = 0; i != n.size(); i++, j++ ){
      nums[j] = p[i];
      j++;
      nums[j] = n[i];
    }
    
    return nums;
  }
};
