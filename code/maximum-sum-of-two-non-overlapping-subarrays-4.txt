int maxSumTwoNoOverlap (int* A, int ASize, int L, int M) {
	int i, j, ie, je, sl, sm, sum = 0;

	for (i = 1; i < ASize; i++) A[i] += A[i-1];

	for (i = -1; (ie = i + L) < ASize; i++) {
		sl = i < 0 ? A[ie] : A[ie] - A[i];
		sm = i >= M ? A[M-1] : -1;
		for (j = 0; (je = j + M) <= i; j++) {
			if (sm < A[je] - A[j]) sm = A[je] - A[j];
		}
		for (j = ie; (je = j + M) < ASize; j++) {
			if (sm < A[je] - A[j]) sm = A[je] - A[j];
		}
		if (sm >= 0 && sum < sl + sm) sum = sl + sm;
	}

	return sum;
}
