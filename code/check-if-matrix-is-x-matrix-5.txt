class Solution {
public:
    bool checkXMatrix(vector<vector<int>>& grid) {
        const int N = grid.size();
        for(int r = 0; r < N; ++r) {
            if(grid[r][r] == 0 || grid[r][N - r - 1] == 0)
                return false;
            for(int c = 0; c < N; ++c) {
                if(c == r || c == N - r - 1) {
                    continue;
                }
                if(grid[r][c] != 0)
                    return false;
            }
        }
        return true;
    }
};
