int minimum(int a[], int n)
{
    int min = a[0];
    for(int i=1; i<n; i++)
    {
        if(a[i] < min)
        {
            min = a[i];
        }
    }
    return min;
}
int minMoves(int* nums, int numsSize)
{
    long int min, sum = 0;
    min = minimum(nums, numsSize);
    for(int i = 0; i < numsSize; i++)
    {
        sum = sum + nums[i] - min;
    }
    // min = minimum(nums, numsSize);
    return sum;
}
