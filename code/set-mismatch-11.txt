int* findErrorNums(int* nums, int numsSize, int* returnSize){
    int i;
    int *ans = (int*)malloc(sizeof(int)*2);
    *returnSize = 2;
    
    int hashtable[10001]={0};
    int sum = 0;
    
    for(i=0;i<numsSize;i++){
        sum+=(i+1); 
        if(hashtable[nums[i]]==0){
            hashtable[nums[i]] = 1;
            sum -=nums[i];
        }else{
            ans[0] = nums[i];
        }
    }
    
    ans[1] = sum;
    
    return ans;
}
