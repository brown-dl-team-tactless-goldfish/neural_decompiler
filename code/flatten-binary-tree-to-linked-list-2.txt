void check(struct TreeNode* root, struct TreeNode** copy){
    if (root == NULL){
        return;
    }
    check(root->right, copy);
    check(root->left, copy);
    root->right = *copy;
    root->left = NULL;
    *copy = root;
}

void flatten(struct TreeNode* root){
    struct TreeNode** copy = malloc(sizeof(struct TreeNode*));
    *copy = NULL;
    check(root, copy);
    free(copy);
}
