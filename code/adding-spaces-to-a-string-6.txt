class Solution {
public:
    string addSpaces(string s, vector<int>& spaces) {
        string ans;
        int cnt = 0, i = 0, n = spaces.size();
        if(spaces[0] == 0) ans += " ", i++;

        for(char c : s) {
            ans += c;
            cnt++;
            if(i < n && spaces[i] == cnt) ans += " ", i++;
        }
        return ans;
    }
};
