int removeElement(int* nums, int numsSize, int val){
int read_index = 0;
int write_index = 0;
// Just check each int, writing it to an index that only
// advances when a non-val int is seen.
    while (read_index < numsSize)
        if (nums[read_index++]!=val)
            nums[write_index++] = nums[read_index-1];
    return write_index;
}
