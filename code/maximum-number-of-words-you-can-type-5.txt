class Solution:
    def canBeTypedWords(self, text: str, brokenLetters: str) -> int:    
        
        res = 0
        for word in text.split(" "):
            flag = True
            for c in word:
                if (c in brokenLetters):
                    flag = False
                    break
                    
            if (flag == True):
                res += 1
                
                
        return res
