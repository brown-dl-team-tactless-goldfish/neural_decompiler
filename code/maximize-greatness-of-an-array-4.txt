class Solution {
public:
    int maximizeGreatness(vector<int>& nums) {
        sort(nums.begin(),nums.end());
        int cnt=0;
        int i=0;
        for (int select=1; select<nums.size(); select++){
            if (nums[i]<nums[select]){
                i++; cnt++;
            }
        }
        return cnt;
    }
};
