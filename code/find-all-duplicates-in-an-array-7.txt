int* findDuplicates(int* nums, int numsSize, int* returnSize){
    int* ans = malloc(sizeof(int)*numsSize);
    int* table = calloc(sizeof(int), numsSize+1);
    *returnSize = 0;
    for (int i = 0; i < numsSize; i++)
    {
        if (++table[nums[i]] == 2)
        {
            ans[*returnSize] = nums[i];
            (*returnSize)++;
        }
    }
    return ans;
}
