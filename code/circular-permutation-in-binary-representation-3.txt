    vector<int> circularPermutation(int n, int start) {
        vector<int> v = {0,1};
        for(int i=1;i<n;i++){
            auto v2 = v;
            reverse(v2.begin(), v2.end());
            for(auto& x:v2) x |= (1<<i);
            v.insert(v.end(), v2.begin(), v2.end());
        }
        auto it = find(v.begin(), v.end(), start);
        vector<int> res(it, v.end());
        res.insert(res.end(), v.begin(), it);
        return res;
    }
