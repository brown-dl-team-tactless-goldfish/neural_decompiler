int* diStringMatch(char* S, int* returnSize) {
	int i;
	int I_value = 0, D_value;
	D_value = strlen(S);

	*returnSize = strlen(S) + 1;

	int *retArray = (int *)malloc(sizeof(int) * (*returnSize));

	for(i = 0; S[i] != '\0'; i++)
	{
		if(S[i] == 'I')
			retArray[i] = I_value++;
		else if(S[i] == 'D')
			retArray[i] = D_value--;
		else
			return NULL;
			
	}

	if(S[i-1] == 'I')
		retArray[i] = I_value++;
	else if(S[i-1] == 'D')
		retArray[i] = D_value--;

	return retArray;
}
