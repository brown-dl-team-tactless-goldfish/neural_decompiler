public class Solution {
    public bool MakeEqual(string[] words) {
        var dict = new Dictionary<char, int>();
        foreach(var s in words){
            foreach(var c in s){
                if(dict.ContainsKey(c)){
                    dict[c] = dict[c] + 1;
                }else{
                    dict.Add(c, 1);
                }
            }
        }
        foreach(var p in dict){
            if(p.Value%words.Length!=0){
                return false;
            }
        }
        return true;
    }
}
