int GCD(int a, int b){
    while(b != 0){
        int tmp = b;
        b = a % b;
        a = tmp;
    }
    return a;
}

bool canMeasureWater(int jug1Capacity, int jug2Capacity, int targetCapacity){
    if (jug1Capacity + jug2Capacity < targetCapacity){
        return false;
    }
    return targetCapacity % GCD(jug1Capacity, jug2Capacity) == 0;
}
