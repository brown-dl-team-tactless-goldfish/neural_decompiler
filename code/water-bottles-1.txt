int numWaterBottles(int numBottles, int numExchange){
    int drink = numBottles;
    int empty_b = 0;
    int full = numBottles;
    int exchange = numBottles;
    
    while (exchange > 0){
        exchange = (empty_b + full)/numExchange;
        empty_b = (empty_b + full)%numExchange;
        full = exchange;
        drink += exchange;
    }
    return drink;
}
