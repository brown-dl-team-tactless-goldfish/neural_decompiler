int sortf (const void * a, const void * b) {
   return (*(int*)a - *(int*)b);
}

double trimMean(int* arr, int arrSize){
    int rm = 0.05 * arrSize, avg = 0;
    qsort(arr, arrSize, sizeof(int), sortf);
    arrSize -= rm;
    for (int i = rm; i < arrSize; i++) avg += arr[i];
    return ((double) avg) / (arrSize - rm);
}
