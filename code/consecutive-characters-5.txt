 int maxPower(char * s)
 {
     
     int count = 1;
     int max = 0;
     
     for (int  i = 0 ; i < strlen(s) ; i++)
     {
         if ( s[i] == s[i+1])
         {
             count++;
         }
         else
             
         {
             if ( count > max)
             {
                 max = count;
             }
             
             count = 1;
         }
     }
     
     
     
     return max;
     
 
 }
