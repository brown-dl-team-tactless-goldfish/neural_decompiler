int getLucky(char * s, int k){
    int n = strlen(s);
    int ans = 0;
    for(int i = 0; i < n; i++){
       int t = s[i] -'a' + 1;
        while(t){
            ans += t % 10 ;
            t /= 10;
        }
    }

    for(int i = 1; i < k; i++){
        int tmp = ans;
        ans = 0;
        while(tmp){
            ans += tmp % 10 ;
            tmp /= 10;            
        }
    }
    return ans;
}
