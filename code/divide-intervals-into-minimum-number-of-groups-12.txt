class Solution {
public:
    int minGroups(vector<vector<int>>& intervals) {
        map<int, int> m;
        for(auto i : intervals) {
            m[i[0]]++;
            m[i[1] + 1]--;
        }
        int ans = 0, s = 0;
        for(auto i : m) {
            s += i.second;
            ans = max(ans, s);
        }
        return ans;
    }
};
