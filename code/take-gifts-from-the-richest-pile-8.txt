class Solution {
public:
    long long pickGifts(vector<int>& gifts, int k) {
        priority_queue<int> pq(begin(gifts),end(gifts));
        long long sum = 0;
        while(k--){
            int e = pq.top();
            pq.pop();
            pq.push(floor(sqrt(e)));
        }
        while(!pq.empty()){
            sum += pq.top();
            pq.pop();
        }
        return sum;
    }
};
