class Solution {
public:
    vector<vector<int>> restoreMatrix(vector<int>& rowSum, vector<int>& colSum) {
        int r = rowSum.size(), c = colSum.size();
        vector<vector<int>> ans (r, vector<int>(c));
        
        for(int i=0; i<r; i++) {
            for(int j=0; j<c; j++) {
                int mn = min(rowSum[i], colSum[j]);
                rowSum[i] = rowSum[i]-mn;
                colSum[j] = colSum[j]-mn;
                
                ans[i][j] = mn;
            }
        }
        return ans;
    }
};
