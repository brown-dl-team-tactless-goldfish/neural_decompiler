bool carPooling(vector<vector<int>>& trips, int capacity) {
        int n = trips.size();
        map<int,int>mp;
        for(int i=0;i<n;i++)
        {
            mp[trips[i][1]] += trips[i][0];
            mp[trips[i][2]] -= trips[i][0];
        }
        int c=0;
        for(auto it=mp.begin();it!=mp.end();it++)
        {
            c += it->second;
            if(c>capacity)
                return 0;
        }
        return 1;
    }
