class Solution {
public:
    vector<int> replaceNonCoprimes(vector<int>& nums) {
        vector<int> ans;
        
        for(auto& x:nums){
            while(!ans.empty() && gcd(ans.back(), x)>1){
                x=lcm(ans.back(), x);
                ans.pop_back();
            }
            ans.push_back(x);
        }
        
        return ans;
    }
};
