void input(struct Node* root, struct Node** copy, int deep){
    if (root == NULL){
        return;
    }
    root->next = copy[deep];
    copy[deep] = root;
    input(root->right, copy, deep+1);
    input(root->left, copy, deep+1);
}

struct Node* connect(struct Node* root) {
    struct Node* copy[80];
    memset(copy, 0, 80 * sizeof(struct Node*));
    input(root, &copy, 0);
    return root;
}
