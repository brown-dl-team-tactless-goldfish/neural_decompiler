char* removeOccurrences(char* s, char* part) {
    int slen = strlen(s), plen = strlen(part), idx = 0;
    char* res = (char*) malloc((slen + 1) * sizeof(char));
    for (int i = 0; i < slen; i++) {
        res[idx++] = s[i];
        if (idx >= plen && (!strncmp(res + idx - plen, part, plen))) idx -= plen;
    }
    res[idx] = '\0';
    return res;
}
