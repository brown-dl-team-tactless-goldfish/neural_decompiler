class Solution {
public:
    int bestClosingTime(string c) {
        int N = c.size(), cnt[100001]{0}, res = 0, &min_v = cnt[0];
        for(int i = N-1; i >= 0; i--) cnt[i] += cnt[i+1] + (c[i] == 'Y');
        for(int i = 0, p = 0; i <= N; i++) {
            int cur = p + cnt[i];
            if(cur < min_v) res = i;
            min_v = min(cur, min_v);
            p += (i < N ? c[i] == 'N' : 0);
        }
        return res;
    }
};
