class RLEIterator {
public:
    vector <int> b;
    int i;
    RLEIterator(vector<int>& a) {
        b = a;
        i = 0;
    }
    int next(int n) {
        if (i == b.size()) return -1;
        if (n <= b[i]) {
            b[i] -= n;
            return b[i + 1];
        }
        n -= b[i];
        i += 2;
        return next(n);
    }
};
