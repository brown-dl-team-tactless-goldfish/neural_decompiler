class Solution {
public:
    int backtrack(vector<int>& nums, int index, int curXor)
    {
        if (index == nums.size()) {
            return curXor;
        }
        
        int withElement = backtrack(nums, index + 1, curXor ^ nums[index]);
        int withoutElement = backtrack(nums, index + 1, curXor);
        
        return withElement + withoutElement;    
    }
    
    int subsetXORSum(vector<int>& nums) 
    {
        int index = 0;
        int curXor = 0;
        int totalXor = backtrack(nums, index, curXor);
        return totalXor;
    }
};
