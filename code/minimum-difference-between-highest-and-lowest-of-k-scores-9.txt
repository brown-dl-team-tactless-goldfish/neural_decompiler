class Solution {
public:
    int minimumDifference(vector<int>& nums, int k) {
        if(k==1 || nums.size()==1)return 0;
        
        sort(nums.begin(),nums.end());
        
        int mind = INT_MAX;
        int p=0;
        for(int i=k-1;i<nums.size();i++,p++)
        {
            mind = min(mind,nums[i]-nums[p]);
        }
        
        return mind;
    }
};
