/**
 * Note: The returned array must be malloced, assume caller calls free().
 */
int* leftRigthDifference(int* n, int s, int* re){
    int *rs=(int *)malloc(sizeof(int)*s);
    *re=s;
    for(int i=0;i<s;i++){
        int l=0,r=0;
        for(int j=i-1;j>=0;j--){
            l+=n[j];
        }
        for(int k=i+1;k<s;k++){
            r+=n[k];
        }
        rs[i]=abs(l-r);
    }
    return rs;
}
