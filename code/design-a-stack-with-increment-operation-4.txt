class CustomStack {
public:
    vector<int> st;
    int endPointer;
    int size;
    CustomStack(int maxSize) {
        size = maxSize;
        st.resize(size);
        endPointer = -1;
    }
    
    void push(int x) {
        if (endPointer == size - 1) {
            return;
        }
        st[endPointer + 1] = x;
        endPointer ++;
        return;
    }
    
    int pop() {
        if (endPointer != -1) {
            endPointer -= 1;
            return st[endPointer + 1];
        }
        return -1;
    }
    
    void increment(int k, int val) {
        int endIndex = k >= endPointer + 1 ? endPointer : k - 1;
        for (int i = 0; i <= endIndex; i ++) {
            st[i] += val;
        }
        return;
    }
};

 * Your CustomStack object will be instantiated and called as such:
 * CustomStack* obj = new CustomStack(maxSize);
 * obj->push(x);
 * int param_2 = obj->pop();
 * obj->increment(k,val);
 */
