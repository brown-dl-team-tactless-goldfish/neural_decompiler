char findTheDifference(char * s, char * t){
    
    int s_a[26]={0};
    int t_a[26]={0};
    int i=0;
    
    while(s[i]!='\0')
    {
        s_a[s[i]-'a']++;
        i++;
    }
    
    i = 0;
    while(t[i]!='\0')
    {
        t_a[t[i]-'a']++;
        i++;
    }
    
    for(i=0; i<26; i++)
        if(s_a[i]!=t_a[i])
            break;
    
    return 'a'+i;
}
