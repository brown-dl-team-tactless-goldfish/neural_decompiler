#include <unordered_map>
class Solution {
public:
    int digArtifacts(int n, vector<vector<int>>& art, vector<vector<int>>& dig) {
        map<vector<int> , bool> di;
                for(vector<int> x : dig){
                    di[x] = true; 
                }
        int ans = 0;
        
        for(int i = 0 ; i < art.size() ; i++){
            bool found = true;
            for(int j = art[i][0] ; j <= art[i][2] ; j++){
                for(int k = art[i][1] ; k <= art[i][3] ; k++){
                    if(di[{j,k}] == false){
                        found = false;
                    }
                }
            }
            if(found) ans++;
        }
        return ans;
        
    }
};
