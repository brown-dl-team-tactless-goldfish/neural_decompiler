vector<int> mostVisited(int n, vector<int>& rounds) {
	vector<int> dict(n+1,0), ans;
	int maximum=0, sz = rounds.size();
	for(int i=0;i<sz-1;++i){
	  for(int j=rounds[i]; j!=rounds[i+1]; j=j%n+1)  dict[j]++;
	  if(i==sz-2) dict[rounds[sz-1]]++;
	}
	maximum = *max_element(dict.begin(),dict.end());
	for(int i=1;i<n+1;++i)
	  if(dict[i]==maximum)  ans.push_back(i);
	return ans;
}
