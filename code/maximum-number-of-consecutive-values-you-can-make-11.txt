class Solution {
public:
    int getMaximumConsecutive(vector<int>& coins) {
        std::sort(coins.begin(), coins.end());
        int count, i{0}, size{(int)coins.size()};
        while(i < size && coins[i] == 1) ++i;
        count = i;
        if(i == size || coins[i] > count+1) return count+1;
        int acc{0};
        for(; i < size; ++i) {
            if(coins[i] - acc > count+1) return acc+count+1;
            acc += coins[i];
        }

        return acc+count+1;
    }
};
