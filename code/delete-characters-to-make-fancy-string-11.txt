public class Solution {
    public string MakeFancyString(string s) {
        int count = 0;
        char prev = '0';
        StringBuilder res = new StringBuilder();
        foreach(var c in s){
            if(c == prev)
            {
                count++;
                if(count >= 3){
                    continue;
                }
            }
            else{
                prev = c;
                count = 1;
            }
            res.Append(c);
        }
        return res.ToString();
    }
}
