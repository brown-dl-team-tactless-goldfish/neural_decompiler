int maxRepeating(char * seq, char * word){
    int sLen=strlen(seq);
    int wLen=strlen(word);
    int x=0;
    int max=0;
    
    while(x<sLen){
        int count=0;
        int i=x;
        int j=0;
        while(i<=sLen&&seq[i]==word[j]){
            if(j+1==wLen){count++;x=i-1;j=-1;}
            i++;j++;
        }
        
        if(count>max)max=count;
        x++;
    }
    return max;
}
