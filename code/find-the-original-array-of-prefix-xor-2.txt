/**
 * Note: The returned array must be malloced, assume caller calls free().
 */
int* findArray(int* pref, int prefSize, int* returnSize){
    int *solution = calloc(prefSize, sizeof(int));
    solution[0] = pref[0];
    for (int index = 1; index < prefSize; index++){
        solution[index] = pref[index-1] ^ pref[index];
    }
    *returnSize = prefSize;
    return solution;
}
