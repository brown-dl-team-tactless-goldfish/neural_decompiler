#define mod 1000000007
vector<int> memo;
int dp(int i,string& s,int k)
{
	if(i>=s.length())
		return 1;
	else if(memo[i]!=-1)
		return memo[i];
	else
	{
		long q=0, val=0;
		for(int j=i;j<s.length();++j)
		{
			val*=10;
			val+=s[j]-'0';
			if(val>k)
				break;
			if(j+1==s.length()||s[j+1]!='0')
				q+=dp(j+1,s,k);
		}
		return memo[i]=q%mod;;   
	}
}
int numberOfArrays(string s, int k) 
{
	memo.resize(s.length(),-1);
	return dp(0,s,k);
}
