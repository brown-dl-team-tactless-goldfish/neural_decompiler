func minStartValue(nums []int) int {
    // initialize start value to lowest possible output value such
	// that, if the array is all zeros, the output will still be valid
    startVal := 1 
    runSum := startVal
    
    for _, n := range nums {
	    // get a running sum (runSum) of the array
        runSum += n
		
        // if running sum dips below one, increase the start
		// value (starVal) so that the sum is equal to one.    
        if runSum < 1 {
            startVal += -runSum + 1
            runSum = 1
        }
    }
    
    return startVal
}
