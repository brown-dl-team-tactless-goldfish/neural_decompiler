    unsigned int sum = 0;
    
    for(auto &x: target)
    {
        pq.push(x);
        
        sum += x;
    }
    
    while(pq.top() != 1)
    {
        int cur = pq.top();
        pq.pop();
        
        sum -= cur;
        
        if(sum <= 0 || sum >= cur)
        {
            return false;
        }
        
        cur = cur % sum;
        
        sum = sum + cur;
        
        pq.push(cur > 0? cur: sum);
    }
    
    return true;
}
