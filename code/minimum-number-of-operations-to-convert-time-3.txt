int convertTime(char * current, char * correct){
    int timeNow = ((current[0]-'0')*10+(current[1]-'0'))*60 + (current[3]-'0')*10+(current[4]-'0');
    int timeCorr = ((correct[0]-'0')*10+(correct[1]-'0'))*60 + (correct[3]-'0')*10+(correct[4]-'0');
    int d = timeCorr - timeNow;
    return d/60+(d%60)/15+((d%60)%15)/5+(((d%60)%15)%5)/1;
}
