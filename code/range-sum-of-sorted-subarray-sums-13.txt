class Solution 
{
public:
    int rangeSum(vector<int>& nums, int n, int left, int right) 
    {
        int mod=1e9+7;
        vector<int> v;
        for(int x=0; x<n; x++)
        {
            int sum=0;
            for(int y=x; y<n; y++)
            {
                sum=(sum+nums[y])%mod;
                v.push_back(sum);
            }
        }
        sort(v.begin(),v.end());
        int ans=0;
        for(int x=left-1; x<right; x++) ans=(ans+v[x])%mod;
        return ans;
    }
};
