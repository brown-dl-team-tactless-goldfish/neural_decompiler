class Solution {
public:
    int partitionArray(vector<int>& nums, int k) {
        if(size(nums) == 1) return 1;
        sort(begin(nums),end(nums));
        int i=1,cnt = 0;
        int minEle = nums[0];
        while(i < size(nums)){
            if(abs(minEle - nums[i]) > k){ // when the condition is not satisfied just update the min element and increase the count .
                minEle = nums[i];
                cnt++;
            }
            i++;
        }
        return cnt+1; // + 1 for the last subsequence .
    }
};
