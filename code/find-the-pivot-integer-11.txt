int pivotInteger(int n){
    int sum,sum1;
    
    for(int i=1;i<=n;i++)
    {
        int x=i;
        sum=0;
        sum1=0;
       for(int j=1;j<=x;j++)
       {
           sum=sum+j;
       }
        for(int k=x;k<=n;k++)
        {
            sum1=sum1+k;
        }
        if(sum==sum1)
            return x;
    }
    return -1;

}
