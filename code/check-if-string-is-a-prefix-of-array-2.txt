class Solution:
    def isPrefixString(self, s: str, words: List[str]) -> bool:
        
        def helper(remainS, remainWords):
            if (remainS == ""):
                return True
            
            n = len(remainWords)
            
            if (n == 0):
                return False
            
            if (remainWords[0] not in remainS):
                return False
            
            idx = remainS.index(remainWords[0])
            if (idx != 0):
                return False
            
            return helper(remainS[len(remainWords[0]):], remainWords[1:])
        
        return helper(s, words)
