char * capitalizeTitle(char * title)
{
    int i, j;

    for (i = 0, j= 0; title[i] != '\0'; ++i)
    {
        if (title[i] == ' ')
        {
            if (i - j > 2)
                title[j] = toupper(title[j]);

            j = i + 1;                
        } 
        else 
            title[i] = tolower(title[i]);
    }

    if (i - j > 2)
        title[j] = toupper(title[j]);
        
    return title;
}
