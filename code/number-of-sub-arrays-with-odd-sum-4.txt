class Solution {
public:
    int numOfSubarrays(vector<int>& arr) {
        
        int ans = 0, sum = 0, odd = 0, even = 1, MOD = 1000000007;
        for (int i = 0; i < arr.size(); i++) {
            sum += arr[i];
            (sum % 2 == 0) ? ans += odd : ans += even;            
            (sum % 2 == 0) ? even++ : odd++;
            ans %= MOD;
        } 

        return ans;
    }
};
