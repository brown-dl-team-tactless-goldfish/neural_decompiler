#include <stdio.h>

int totalFruit(int* fruits, int fruitsSize){
    int i; 
    int j;
    int tempMax = 0;
    int numOne = *fruits;
    int numTwo = *(fruits + 1);
    int count = 0;

    if(fruitsSize == 1) {
        return 1;
    }
    for ( i = 0; i < fruitsSize; i++ ) {
        if ( *(fruits + i) != numOne && *(fruits + i) != numTwo) {
            if (count > tempMax) {
                tempMax = count;
            }

            numOne = *(fruits + i - 1);
            numTwo = *(fruits + i);
            j = 1;

            while ((i - j) >= 0 && (*(fruits + i - j) == numOne)) {
                j++;
            }
            count = j;
        }
        else {
            count++;
        }
    }
    if(count > tempMax) {
        tempMax = count;
    }
    return tempMax;
}
