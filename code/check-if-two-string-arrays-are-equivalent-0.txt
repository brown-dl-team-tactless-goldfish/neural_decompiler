
bool arrayStringsAreEqual(char ** word1, int word1Size, char ** word2, int word2Size){
    int i=0;int j=0;
    int m=0;int n=0;
    char a=' ';
    char b=' ';
    while(i<word1Size&&m<word2Size&&a==b){ 
        a=word1[i][j++]; 
        if(word1[i][j]=='\0'){i++;j=0;}
        b=word2[m][n++]; 
        if(word2[m][n]=='\0'){m++;n=0;}
    }
    if(a==b&&i==word1Size&&m==word2Size)return true;
    return false;
}
