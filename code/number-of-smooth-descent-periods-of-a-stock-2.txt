class Solution {
public:
    long long getDescentPeriods(vector<int>& prices) {
        long long res=0;
        int t=1;
        for(int i=0;i<prices.size();i++)
        {
            while(i<prices.size()-1 && prices[i]==prices[i+1]+1)
            {
                t++;
                i++;
            }
            res+=fun(t);
            t=1;
        }
        return res;
    }
    long long fun(int t)
    {
        long long res=t;
        res=res*(res+1)/2;
        return res;
    }
};
