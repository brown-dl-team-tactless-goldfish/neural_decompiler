class Solution {
public:
    int maxScore(vector<int>& nums) {
        int score = 0;
        sort(nums.begin(), nums.end(), std::greater<int>());
        long long sum = 0;
        for(int i = 0; i < nums.size(); i++){
            sum += nums[i];
            if(sum > 0) score++;
        }
        return score;
    }
    
};
