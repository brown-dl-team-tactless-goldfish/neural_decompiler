class Solution {
public:
    bool isCovered(vector<vector<int>>& ranges, int left, int right) {
        for (int n = left; n <= right; n++) {
            if (isCovered(ranges, n) == false) {
                return false;
            }
        }
        return true;
    }
    
    bool isCovered(vector<vector<int>>& ranges, int n) {
        for (vector<int> &range : ranges) {
            if (n >= range[0] && n <= range[1]) {
                return true;
            }
        }
        return false;
    }
};
