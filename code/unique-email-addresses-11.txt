class Solution:
    def numUniqueEmails(self, emails: List[str]) -> int:
        email_list = []
        for email in emails:
            local, domain = email.split("@")

            
            local = local.replace(".", "")
            if ("+" in local):
                idx = local.index("+")
                local = local[:idx]
            print ("local = ", local, " domain = ", domain)
            
            if (local + "@" + domain not in email_list):
                email_list.append(local+"@" + domain)
                
        return len(email_list)
