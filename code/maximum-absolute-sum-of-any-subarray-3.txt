int maxAbsoluteSum(int* nums, int numsSize){
int i,max_endinghere=0,max_sofar=-9999,min_endinghere=0,min_sofar=9999;
    for(i=0;i<numsSize;i++){ 
    max_endinghere=max_endinghere+nums[i]; // using Kadane's algo for calculating the max sum of an subarray
    if(max_endinghere<nums[i]){
        max_endinghere=nums[i];
    }
    if(max_sofar<max_endinghere){
        max_sofar=max_endinghere;
    }
    min_endinghere=min_endinghere+nums[i];  // using Kadane's algo for calculating the min sum of an subarray
    if(min_endinghere>nums[i]){
        min_endinghere=nums[i];
    }
    if(min_sofar>min_endinghere){
        min_sofar=min_endinghere;
    }
    }
    if(abs(max_sofar)>abs(min_sofar)){return abs(max_sofar);} // returning max absolute value 
    else{return abs(min_sofar);}}
