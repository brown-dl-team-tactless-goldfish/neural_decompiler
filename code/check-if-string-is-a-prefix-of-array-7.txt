public class Solution {
    public bool IsPrefixString(string s, string[] words) 
    {
        for (int i= words.Length-1; i>=0; i--)
        {
			if (s == String.Join("", string.Join("", words.Take(i+1).ToArray())))
            {
                return true;
            }
        }

        return false;    
    }
}
