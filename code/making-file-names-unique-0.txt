vector<string> getFolderNames(vector<string>& s) {
    int n = s.size();
    unordered_map<string,int>m;
    int c;
    for(int i=0;i<n;i++){
        if(m.find(s[i])==m.end()){
            m[s[i]] = 1;
        }else{
            string t = s[i];
            c = m[s[i]]++;
            while(m.find(t)!=m.end()){
                t = (s[i]+"("+to_string(c)+")");
                c++;
            }
            s[i] = t;
            m[s[i]] = 1;
        }
    }
    return s;
}
