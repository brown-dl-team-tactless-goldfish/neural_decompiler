bool checkZeroOnes(char * s) {

	register unsigned char i = 0, zero = 0, one = 0, zero_sum = 0, one_sum = 0;
	
	while(s[i] != '\0') {
		if(s[i] == '1') {
			one++;
			zero = 0;
			if(one > one_sum) {
				one_sum = one;
			}
		} else {
			zero++;
			one = 0;
			if(zero > zero_sum) {
				zero_sum = zero;
			}
		}
		i++;
	}

	if(one_sum > zero_sum)
		return true;
	return false;
}
