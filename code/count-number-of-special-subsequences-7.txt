public: 
    int mod = (int)1e9 + 7;
    int countSpecialSubsequences(vector<int>& nums) {
        long long int n = nums.size();
        long long int a = 0, b = 0, c = 0;
        for(int i = 0; i < n;i++){
            if(nums[i] == 0)
                a = ((2*a)%mod + 1) % mod;
            else if(nums[i] == 1)
                b = ((2*b)% mod + a % mod) % mod;
            else
                c = ((2*c)%mod + b%mod)%mod;
        }
        return c;
    }
};
