class Solution {
public:
    int minOperations(int n) {
        vector<int> v;
        for(int i=0; i<=21; i++) {
            v.push_back(pow(2, i));
        }

        int cnt = 0;
        while(n) {
            int d = 1e9;
            for(int i : v) {
                if(abs(i - n) < d) d = abs(i - n);
            }
            n = d;
            cnt++;
        }
        return cnt;
    }
};
