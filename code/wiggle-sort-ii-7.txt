class Solution {
public:
    void wiggleSort(vector<int>& nums) {
        
        int n = nums.size();
        
        vector<int>temp(nums);
        sort(temp.begin(), temp.end());
        
        int i =  n-1;
        int j = (n-1)/2;
        int k = 0;
        
        while (k < n) {
            if (k & 1) {
                nums[k++] = temp[i--];
            } else {
                nums[k++] = temp[j--];
            }
        }
    }
};
