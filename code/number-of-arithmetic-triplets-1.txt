int arithmeticTriplets(int* nums, int numsSize, int diff) {
    int arr[251] = {0};
    for (int i = 0; i < numsSize; i++) {
        arr[nums[i]]++;
    }
    int cnt = 0;
    for (int i = 0; i < numsSize; i++) {
        if (arr[nums[i] + diff] != 0 && arr[nums[i] + diff + diff] != 0) cnt++;
    }
    return cnt;
}
