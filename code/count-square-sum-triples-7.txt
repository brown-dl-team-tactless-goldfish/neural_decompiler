public class Solution
{
    public int CountTriples(int n)
    {
        int count = 0;
        for (int left = 1; left <= n; left++)
        {
            for (int right = left + 1; right <= n; right++)
            {
                int c = (int) Math.Sqrt(left * left + right * right);
                if (c * c == left * left + right * right && c <= n) count += 2;
            }
        }

        return count;
    }
}
