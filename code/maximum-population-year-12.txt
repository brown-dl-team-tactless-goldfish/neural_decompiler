int maximumPopulation(vector<vector<int>>& logs) {
        vector<int>v(101,0);
        for(vector<int>cur :logs){
            for(int i=cur[0];i<cur[1];i++){
                v[i-1950]++;
            }
        }
        int maxi =0,ind;
        for(int i=0;i<101;i++){
            if(maxi < v[i]){
                maxi = v[i];
                ind = i+1950;
            }
        }
        return ind;
    }
