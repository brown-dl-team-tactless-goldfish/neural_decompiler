class ProductOfNumbers {
public:
    vector<int> arr;
    ProductOfNumbers() {
        arr.clear();
        arr.emplace_back(1);
    }
    
    void add(int num) {
        if(num)
            arr.emplace_back(num * arr.back());
        else{
            arr.clear();
            arr.emplace_back(1);
        }
    }
    
    int getProduct(int k) {
        int n = arr.size();
        return k < arr.size() ? arr[n-1] / arr[n-k-1] : 0;
    }
};
