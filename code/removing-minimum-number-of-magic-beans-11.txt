class Solution {
public:
    using ll = long long;
    long long minimumRemoval(vector<int>& B) {
        sort(begin(B), end(B));
        ll res = LONG_MAX, sum = accumulate(begin(B), end(B), 0LL);
        for(int i = 0; i < B.size(); i++) {
            if(i > 0 && B[i] == B[i-1]) continue;
            res = min(res, sum - B[i]*((ll)B.size()-i));
        }
        return res;
    }
};
