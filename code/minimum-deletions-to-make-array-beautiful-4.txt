		class Solution {
		public:
			int minDeletion(vector<int>& v) {
				int n = v.size();
				stack<int> s;
				int c = 0;
				for(int i=0; i<n; i++){
					if(s.empty()){
						s.push(v[i]);
					}else{
						if(s.top()==v[i]){
							c++;
						}else{
							s.pop();
						}
					}
				}
				return c+s.size(); 

			}
		};
