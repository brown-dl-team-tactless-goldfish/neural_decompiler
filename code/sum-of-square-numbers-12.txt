#include<math.h>
bool judgeSquareSum(int c){
    long long int i=0,pow1=pow(c,0.5),j=pow1+1;
    while(i<=j){
        long long int x=((i*i)+(j*j));
        if ((i*i)==c || (j*j)==c){
            return true;
        }
        if (x==c){
            return true;
        }
        if(x>c){
            j-=1;
        }
        else{
            i+=1;
        }
    }
    return false;
}
