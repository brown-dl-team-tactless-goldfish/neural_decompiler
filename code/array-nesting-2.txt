int arrayNesting(int* nums, int numsSize){
    int max = 0;
    for(int i=0; i<numsSize; i++)
    {
        int count = 0, idx = i;
        while(nums[idx] != -1 && i != nums[idx])
        {
            count++;
            int temp = nums[idx];
            nums[idx] = -1;
            idx = temp;
        }
        if(max < count) max = count;
    }
    return max + 1;
}
