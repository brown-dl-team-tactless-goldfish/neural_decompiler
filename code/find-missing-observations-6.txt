class Solution {
public:
	vector<int> missingRolls(vector<int>& rolls, int mean, int n) {
		int m = rolls.size();
		int mSum = 0;
		for(int i = 0; i < m; i++){
			mSum += rolls[i];
		}
		int nSum = (m + n) * mean - mSum;
		if(nSum <= 0 || nSum > 6 * n || nSum / n == 0){
			return {};
		}
		vector<int> res;
		int temp = 0;
		while(nSum){
			temp = nSum / n;
			res.push_back(temp);
			n--;
			nSum -= temp;
		}
		return res;
	}
};
