class Solution {
public:
    int mod=1000000007;
    int sumOfFlooredPairs(vector<int>& nums) 
    {
        vector<int>v(200002,0);
        for(auto x:nums)
            v[x+1]++;
        for(int i=1;i<=200001;i++)
            v[i]+=v[i-1];
        int res=0;
        for(auto x:nums)
        {
            for(int s=x;s<=100000;s+=x)
            {
                int k=s/x;
                int l=v[s+x]-v[s];
                res=(res+k*l)%mod;
            }
        }
        return res;
    }
};
