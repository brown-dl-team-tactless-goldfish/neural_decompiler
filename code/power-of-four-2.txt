bool isPowerOfFour(int n){
    
    if(n==0){return false;}
    if(n>0){
            if((n&(n-1))==0){if((n&0x55555555)==n){return true;}}        
            return false;   
           } 
    else{
        if((n&(n+1))==0){if((n&0x55555555)==n){return true;}}        
            return false;   
    }
}
