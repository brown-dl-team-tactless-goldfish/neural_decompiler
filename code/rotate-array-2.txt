void rotate_right_by_one(int* nums, int numsSize){
    int count = 0;
    // step 1, remember the last element
    int last = nums[numsSize-1];
    // step 2, move the second last to last and so on
    for(count=numsSize-1;count>0;count--){
        nums[count]= nums[count-1];
    }
    // step 3, fill the first one wih last element
    nums[0] = last;
}


void rotate(int* nums, int numsSize, int k){
    k = k%numsSize;
    int c = 0;
    for(c=1;c<=k;c++)
    {
        rotate_right_by_one(nums,numsSize);
    }

}
