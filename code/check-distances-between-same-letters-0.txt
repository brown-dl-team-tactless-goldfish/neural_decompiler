bool checkDistances(char * s, int* distance, int distanceSize){
    int n = strlen(s);
    bool* flag = calloc(n, sizeof(bool));
    for(int i = 0; i < n; i++){
        if(flag[i] == false){
            int d = distance[s[i] - 'a'] ;
            if((i + 1 + d) >= n || s[i] != s[i + 1 + d])
                return false;
            flag[i] = true;
            flag[i+1+d] = true;
        }        
    }
    return true;
}
