    vector<vector<double>>v(101,vector<double>(101, 0));
    
    v[0][0] = poured;
    
    for(int i=0; i < query_row; i++)
    {
        for(int j=0; j<=query_glass; j++)
        {
            double remaining = max(v[i][j]-1, 0.0);
            
            v[i+1][j] += remaining / 2.0;
            
            v[i+1][j+1] += remaining / 2.0;
        }
    }
    
    return min(1.0, v[query_row][query_glass]);
    
}
