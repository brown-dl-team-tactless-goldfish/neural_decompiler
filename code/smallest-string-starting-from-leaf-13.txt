string ans = "";

void helper(TreeNode* root, string temp)
{
    
    if(!root)
    {
        return;
    }
    
    temp =  char(root->val+'a')+temp;
    
    if(!root->left && !root->right)
    {
      ans = ans == ""? temp : min(ans,temp);
    }
    
    if(root->left)
    {
        helper(root->left, temp);
    }
    
    if(root->right)
    {
        helper(root->right,temp);
    }
}

string smallestFromLeaf(TreeNode* root) {
    
   if (!root)
    {
        return ans;
    }
    
    helper(root,"");
    
    return ans;
    
}
