public bool CanConstruct(string s, int k) 
{
	if(s.Length < k)
	{
		return false;
	}        

	int[] charCount = new int[26];
	foreach(char c in s)
	{
		charCount[c - 'a']++;
	}

	int count = 0;
	for(int i = 0; i < charCount.Length; i++)
	{
		if(charCount[i] % 2 == 1)
		{
			count++;
		}
	}

	return count <= k;
}
