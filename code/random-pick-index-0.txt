typedef struct {
    int* arr;
    int size;
} Solution;


Solution* solutionCreate(int* nums, int numsSize) {
    Solution* obj = (Solution*)malloc(sizeof(Solution));
    
    obj->arr = nums;
    obj->size = numsSize;
    
    return obj;
}

int solutionPick(Solution* obj, int target) {
    int i,count = 0;
    int result = 0;
    
    for(i=0;i<obj->size;i++)
    {
        if(obj->arr[i] == target)
        {
            count++;
            if(count == 1)
                result = i;
            else if((rand()%count) == 0)
                result = i;
        }
    }
    
    return result;
}
