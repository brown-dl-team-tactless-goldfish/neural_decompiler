
class Solution
{
public:
  
void check(string &s, string &match)
{
    if (s.size() < match.size())
        return;

    int sidx = s.size() - match.size();

    if (s.substr(sidx) == match)
    {
        s.resize(sidx);
    }
}

string removeOccurrences(string s, string part)
{

    string res;

    for (char c : s)
    {
        res.push_back(c);
        check(res, part);
        //cout<<res<<endl;
    }

    return res;
}
};
