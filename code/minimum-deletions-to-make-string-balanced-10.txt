class Solution {
public:
    int minimumDeletions(string s) {
        
        int a = 0;
        int b = 0;
        
        int n = s.size();
        
        for( char c: s)
        {
            if( c == 'a')
            {
                b = b+1;
            }
            else
            {
                b = min( a, b);
                a = a + 1;
            }
        }
        return min(b ,a);
    }
};
