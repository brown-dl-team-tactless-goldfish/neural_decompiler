
```public:
    static bool cmp(vector<int>&v1,vector<int>&v2)
    {
        return (v1[0]-v1[1])>(v2[0]-v2[1]);
    }
    int minimumEffort(vector<vector<int>>& tasks) {
        sort(tasks.begin(),tasks.end(),cmp);
        int i,sum=0;
        for(i=0;i<tasks.size();i++)
        {
            sum=max(tasks[i][1],tasks[i][0]+sum);
        }
        return sum;
    }
};
