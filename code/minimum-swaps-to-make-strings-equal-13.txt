int minimumSwap(string s1, string s2) 
{
	int x_y = 0, y_x = 0;

	for (int i = 0; i < s1.size(); i++)
	{
		if (s1[i] != s2[i]){
			if (s1[i] == 'x')
				x_y++;
			else
				y_x++;
		}
	}
	//cout << x_y << ' ' << y_x << endl;
	if ((x_y % 2) != (y_x % 2))
		return -1;
	else
		return (x_y - x_y % 2) / 2 + (y_x - y_x % 2) / 2 + x_y % 2 * 2;
}
