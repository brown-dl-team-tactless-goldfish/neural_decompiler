class Solution {
public:
    vector<vector<int>> findMatrix(vector<int>& nums) {
        unordered_map<int, int>Map ; //{val, count}
        for(auto n : nums)
            Map[n]++ ;
        int row = 0;
        for(auto& m : Map)
            row = max(row, m.second) ;
        vector<vector<int>> ret(row) ;
        for(auto& m : Map){
            for(int i = 0; i < m.second; i++){
                ret[i].push_back(m.first) ;
            }
        }
        return ret ;
    }
};
