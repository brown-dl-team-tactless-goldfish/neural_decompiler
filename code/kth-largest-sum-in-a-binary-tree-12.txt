class Solution {
public:
    vector<long long> v;
    void go(TreeNode* n, int i) {
        if (!n)
            return;
        if (v.size() <= i)
            v.push_back(0);
        v[i] += n->val;
        go(n->left, i + 1);
        go(n->right, i + 1);
    }
    long long kthLargestLevelSum(TreeNode* root, int k) {
        go(root, 0);
        sort(v.rbegin(), v.rend());
        return (k > v.size()) ? -1 : v[k - 1];
    }
};
