
bool judgeCircle(char * moves)
{
    int u = 0,l = 0;
    while(*moves)
    {
        switch(*moves++)
        {
            case 'U':
                u++;
                break;
            case 'D':
                u--;
                break;
            case 'L':
                l++;
                break;
            case 'R':
                l--;
                break;
        }
    }
    return !u && !l;
}
