int maxScore(char * s){
    int l = strlen(s), f[l], b[l], m = -1;
    for (int i = (b[l - 1] = s[l - 1] == '1', l - 1) ; --i > 0 ; b[i] = b[i + 1] + (s[i] == '1'));
    for (int i = (f[0] = s[0] == '0', 0) ; ++i < l ; m < f[i - 1] + b[i] ? m = f[i - 1] + b[i] : 0)
        f[i] = f[i - 1] + (s[i] == '0');    
    return m;
}
