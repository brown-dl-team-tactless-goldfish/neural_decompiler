bool digitCount(char * num){
    int n = strlen(num);
    int cn[10] = {0};
    for(int i = 0; i < n; i++){
        cn[num[i] - '0']++;
    }
    for(int i = 0; i < n; i++){
        if(cn[i] !=  (num[i] -'0') )
            return false;
    }
    for(int j = n; j < 10; j++){
        if(cn[j] != 0)
            return false;
    }
    return true;
}
