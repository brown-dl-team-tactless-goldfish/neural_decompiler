

int secondHighest(char * s){
    char x = -1, y = -1;
    while(*s)
    {
        if(*s >= '0' && *s <= '9' )
        {
            if(*s > y)
            {
                x = y;
                y = *s;
            }
            else if( y != x  && y != *s)
            {
                x = x > *s ? x : *s;
            }
        }
        s++;
    }
    return x < 0 ? x :  x - '0';
}
