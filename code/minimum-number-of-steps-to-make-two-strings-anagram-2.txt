class Solution {
public:
    int minSteps(string s, string t) 
    {
        int res=0;
        vector<int> map(26, 0);
        for(int i=0;i<s.size();i++)
        {
            map[s[i]-'a']++;
            map[t[i]-'a']--;
        }
        for(int i=0;i<26;i++)
            res+=abs(map[i]);
        return res/2; 
    }
};
