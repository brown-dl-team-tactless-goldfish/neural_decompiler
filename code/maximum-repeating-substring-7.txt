class Solution {
public:
    int maxRepeating(string sequence, string word) {
        string ex=word; //store the string word in ex
        size_t found = sequence.find(word); //check the presence of word in sequence
        while(found != string::npos){
            word+=ex; //increase the size of word by adding ex( given word)
            found = sequence.find(word); //again check the presence of word in sequence
        }
		//word.size()-ex.size() as this word is not present so subtract the last ex added 
        return (word.size()-ex.size())/ex.size(); 
    }
};
