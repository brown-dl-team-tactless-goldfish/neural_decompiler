int diagonalSum(int** mat, int matSize, int* matColSize){
    
    int index = 0, diagSum = 0;
    for(index = 0; index < matSize; index++)
        diagSum += (*(*(mat + index) + index) + *(*(mat + index) + matSize - index -1));
    return ((matSize %2) == 0) ? diagSum : (diagSum - *(*(mat + (matSize-1)/2) + (matSize-1)/2));
}
