class Solution {
public:
    #define ll long long 
    int chalkReplacer(vector<int>& chalk, ll K ) {
        //find the student when chalk[i] > K 
        ll sum = accumulate(begin(chalk),end(chalk),0LL) ;
        K = K % sum ;
        for(int i = 0 ; i < chalk.size() ; ++ i){
            if(chalk[i] > K) return i ;
            K -= chalk[i] ;
        }
        return -1 ;
        
    }
};
