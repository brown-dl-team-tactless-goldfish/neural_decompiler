class Solution
{
public:
    int minimumMoves(string s) 
    {
        int i=0,count=0;
        while(i<s.length())
        {
            // Whenever we encounter a 'X' we convert 3 characters starting from 'i'th character to 'O' and increment pointer by 3 as 3 characters have been changed
            if(s[i]=='X')
            {
                i+=3;
                count++;
            }
            
            // Otherwise just move by 1 position
            else
            {
                i++;
            }
        }
        return count;
    }
};
