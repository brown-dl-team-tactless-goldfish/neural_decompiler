int* decompressRLElist(int* nums, int numsSize, int* returnSize){
    int i,j=0, *dec;
    for(i=0,*returnSize=0;i<numsSize;i+=2)
        *returnSize+=nums[i];
    dec = (int*)malloc(sizeof(int)*(*returnSize));
    for(i=0;i<numsSize;i+=2)
        for(;nums[i]>0;nums[i]--)
            dec[j++]=nums[i+1];
    return dec;
}
