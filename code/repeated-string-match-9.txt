int repeatedStringMatch(char* A, char* B) {
    int count, i=0, j, k;

    while(A[i]!='\0')
    {
        if (A[i]==B[0])
        {
            j=i;
            k=0;
            count=1;
            while(B[k++]==A[j++])
            {
                if (B[k]=='\0')
                    return count;
                if (A[j]=='\0')
                {
                    j=0;
                    count++;
                }
            }
        }
        i++;
    }
    return -1;
}
