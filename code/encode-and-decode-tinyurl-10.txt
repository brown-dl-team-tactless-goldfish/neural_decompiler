#include <cstdlib>
#include <ctime>

class Solution {
    string encodings = "0123456789abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ";
    unordered_map<string, string> code_to_url;
    
public:

    Solution() {
        srand(time(nullptr));    
    }
    
    string getKey() {
        string key;
        for (int i = 0; i < 6; ++i) {
            char c = encodings[rand() % encodings.size()];
            key.push_back(c);
        }
        
        return key;
    }
    
    // Encodes a URL to a shortened URL.
    string encode(string long_url) {      
        string key = getKey();
        while (code_to_url.count(key)) {
            key = getKey();
        }
        
        code_to_url[key] = long_url;
        return "http://tinyurl.com/" + key;
    }

    // Decodes a shortened URL to its original URL.
    string decode(string short_url) {
        string key = short_url.substr(short_url.size() - 6);
        return code_to_url[key];
    }
};
