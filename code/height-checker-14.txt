int cmp(const void* a,const void * b)
{
	return *(int *)a-*(int*)b;
}
int heightChecker(int* heights, int heightsSize)
{
	int count = 0;
	int *buf = (int *)malloc(sizeof(int)*heightsSize);
	for(int i = 0;i<heightsSize;i++)
	{
		buf[i] = heights[i];
	}
	qsort(buf,heightsSize,sizeof(int),cmp);
	for(int i=0;i<heightsSize;i++)
	{
		if(buf[i]!=heights[i])
			count++;
	}
	if(buf!=NULL)
	{
		free(buf);
		buf = NULL;
	}
	return count;

}
