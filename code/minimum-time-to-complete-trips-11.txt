#define ll long long int
class Solution {
public:

    ll trips(vector<int>&time, ll currTime, int totalTrips) {
        ll c = 0;
        for(ll i=0;i<time.size();i++) {
            c += (currTime/time[i]);
            if(c>=totalTrips) {
                return c;
            }
        }
        return c;
    }

    ll minTime(vector<int>&time, int totalTrips) {
        ll minTime = 0;
        ll maxTime = 1e15;
        ll resp = 1e15;
        while(minTime<=maxTime) {
            ll mid = minTime + (maxTime - minTime)/2;
            if(trips(time, mid, totalTrips) < totalTrips) {
                minTime = mid+1;
            } else {
                resp = mid;
                maxTime = mid-1;
            }
        }
        return resp;
    }

    long long minimumTime(vector<int>& time, int totalTrips) {
        return minTime(time, totalTrips);
    }
};
