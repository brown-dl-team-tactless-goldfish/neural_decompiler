bool isPerfectSquare(int num){
if((num==0)||(num==1))
	return(true);	
int min=1;
int max=num>>1;
max = (max>46340)?46340:max;
int mid;

while(max-min>1)
{
    mid = (min+max)>>1;
    if(mid*mid<num)
        min=mid;
    else if(mid*mid>num)
        max=mid;
    else
        return(true);
}

if((min*min==num)||(max*max==num))
    return(true);
else
    return(false);
