class Solution {
public:
    int triangleNumber(vector<int>& nums) {
        
        if (nums.size() < 3) return 0;
        
        sort(nums.begin(), nums.end(), greater<int>());
        
        int ans = 0;
        int n = nums.size();
        
        for (int c = 0; c < n-2; c++)
        {
            int b = c + 1;
            int a = n - 1;
            
            while (a > b)
            {
                if (nums[a] + nums[b] > nums[c])
                {
                    ans += a - b;
                    b++;
                }
                else
                {
                    a--;
                }
            }
        }
        
        return ans;
    }
};
