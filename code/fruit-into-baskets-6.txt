#define MAX(a,b)(a>b?a:b)
int totalFruit(int* fruits, int fruitsSize){
    int res=0,left=0,right=-1;
    for(int i=1;i<fruitsSize;i++){
        if(fruits[i]==fruits[i-1])continue;
        if(right>=0 && fruits[right]!=fruits[i]){
            res=MAX(res,i-left);
            left=right+1;
        }
        right=i-1;
    }
    return MAX(fruitsSize-left,res);
}
