    start[0] = '\0';

    if(n==1 && k == 1)
           return "0";

    ans = (char*)malloc(sizeof(char)*100000);

    memset(ans, 0, sizeof(ans));

    for(i=0; i < n-1; i++) {
            sprintf(buf, "%d", 0);
            strcat(start, buf);
    }

    visited = push_visited(visited, start);
    dfs(start, n, k, &visited, &ans);
    strcat(ans, start);
    return ans;
