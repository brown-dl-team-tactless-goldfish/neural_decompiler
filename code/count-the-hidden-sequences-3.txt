int numberOfArrays(vector<int>& v, int l, int u) {
    
   long long  minv = INT_MAX , maxv = INT_MIN;
   long long  s= 0;
    for(auto i : v){
        s+=i;
        maxv = max(maxv , s);
        minv = min(minv , s);
    }
    if(minv==INT_MIN) return 0;
    long long  c =0;
    for(long long  i=l;i<=u;i++){
        if(i+minv>=l && i+maxv<=u) c++;
    }
    return c;
    
}
