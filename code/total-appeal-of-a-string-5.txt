class Solution {
public:
    #define ll long long
    long long appealSum(string s) {
        int n = s.length();
		// prev vector tracks the last occurence of char c
        vector<int> prev(26,-1);
        ll ans = 0;
        for(int i=0;i<n;i++){
            if(prev[s[i]-'a']==-1){
                ans+= (i+1)*(n-i);
            }else{
                ans+=(i-prev[s[i]-'a'])*(n-i);
            }
            prev[s[i]-'a']=i;
        }
        return ans;
    }
};
