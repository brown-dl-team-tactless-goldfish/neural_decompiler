class Solution {
public:
    int minimumRecolors(string blocks, int k) {
        unordered_map<char , int> um;
        for(int i=0;i<k;i++){
            um[blocks[i]]++;
        }
        int operations = um['W'];
        
        for(int i = k ; i < blocks.size() ; i++){
            um[blocks[i - k]]--;
            um[blocks[i]]++;
            operations = min(operations , um['W']);
        }      
        
        return operations;     
    }
};
