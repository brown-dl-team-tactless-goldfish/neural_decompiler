class Solution { 
public:
 int minimumSwap(string s, string t) {  
    int c = 0 ,d = 0;
    for(int i=0;i<s.size();i++){
        if(s[i] == 'x' && t[i] == 'y')c++;
        else if(s[i] == 'y' && t[i] == 'x')d++;
    }
    if(!(c&1) && (d&1) || !(d&1) && (c&1))return -1;
    return (c + 1)/2 + (d + 1)/2;
}
