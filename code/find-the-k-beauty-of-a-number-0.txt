int divisorSubstrings(int num, int k){
    int digitCnt = 0, n = num, div, ans = 0;
    while(n){ digitCnt++; n /= 10; }
    int mod = pow(10, k);
    digitCnt -= (k - 1);
    n = num;
    while(digitCnt){
        div = n % mod;
        if(div)
            ans += (num % div)? 0 : 1;
        n /= 10;
        digitCnt--;
    }
    return ans;
}
