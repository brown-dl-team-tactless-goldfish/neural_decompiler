class Solution {
public:
    int minPairSum(vector<int>& nums) {
        vector<int> myv(100001,0);
        for(auto it : nums) myv[it]++;
        int left = 0, right = 100000;
        int res = INT_MIN;
        int times = 0;
        while(times++ != nums.size()/2){
            while(myv[left] == 0) left++;
            while(myv[right] == 0) right--;
            res = max(res, left + right);
            myv[left]--,myv[right]--;
        }
        return res;
    }
};
