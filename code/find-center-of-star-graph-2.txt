int findCenter(int** edges, int edgesSize, int* edgesColSize){
    int map[100001] = {0}, i, j, center;
    for (i = 0; i < edgesSize; ++i) {
        for (j = 0; j < 2; ++j) {
            ++map[edges[i][j]];
        }
    }
    for (i = 0; i < 100001; ++i) {
        if (map[i] == edgesSize) {
            center = i;
            break;
        }
    }
    return center;
}
