int maximumDifference(int* arr, int n){
   int max=0;
     for(int i=0;i<n;i++)
     {
         for(int j=0;j<n;j++)
         {
             if(i==j)
             {
                 continue;
             }
             else if(i<j&&arr[i]<arr[j])
             {
                if(arr[j]-arr[i]>max)
                {
                    max=arr[j]-arr[i];
                }
             }
         }

     }
     if(max>0)
     return max;
     return -1;

}
