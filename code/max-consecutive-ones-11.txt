class Solution {
public:
    int findMaxConsecutiveOnes(vector<int>& nums) {
        int ans=0,maxx=0;
        for(auto x:nums){
            if(x==1){
                ans++;
            }
            else{
                maxx=max(maxx,ans);
                ans=0;
            }
        }
        maxx=max(maxx,ans);
        return maxx;
    }
};
