bool hasAlternatingBits(int n) {
    // checks for repeted bits
    int alt = -1;
    while(n > 0) {
        // gets LSB and cheks if it's zero or one
        // if alt is the same as before, return false
        if(n & 1 == 1) {
            if(alt == 1) return false;
            alt = 1;
        } else {
            if(alt == 0) return false;
            alt = 0;
        }
        // right shift and get next bit
        n >>= 1;
    }
    return true;
}
