    def minimumCardPickup(self, cards):
        locs = collections.defaultdict(list)
        for i, c in enumerate(cards):
            locs[c].append(i)
        res = float("inf")
        for x in locs:
            for i in range(len(locs[x]) - 1):
                res = min(res, locs[x][i + 1] - locs[x][i] + 1)
        return -1 if res == float("inf") else res
