bool checkStraightLine(int** c, int coordinatesSize, int* coordinatesColSize){
    int dy = c[1][1] - c[0][1];
    int dx = c[1][0] - c[0][0];
    
    
    for(int i = 0; i < coordinatesSize - 1; i++)
    {
        if(!(dx*(c[i+1][1] - c[i][1]) == dy*(c[i+1][0] - c[i][0])))
            return false;
    }
    
    return true;
}
