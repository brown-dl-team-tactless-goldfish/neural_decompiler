/*
    straight forward, just keep appending the string and 
    checking at the same time if it matches 's'
	We can break earlier when the length exceeds the length of 's'. Thanks to @MSR_Yang
*/
class Solution {
public:
    bool isPrefixString(string s, vector<string>& words) {
        string check = "";
        
        for(const string &word : words) {
            check += word;
            
			if(check == s)                   return true;
			if(check.length() >= s.length()) return false;
        }
        
        return false;
    }
};
