bool validMountainArray(int* A, int ASize){
    if(ASize<3||A[0]>=A[1]){
        return false;
    }
    int i=0;
    for(;i<ASize-1;i++){
        if(A[i]>=A[i+1]){
            break;
        }
    }
    if(i>=ASize-1){
        return false;
    }
    for(;i<ASize-1;i++){
        if(A[i]<=A[i+1]){
            return false;
        }
    }
    return true;
}
