class Solution
{
public:
    int minStoneSum(vector<int> &p, int k)
    {
        priority_queue<int> pq(p.begin(), p.end());
        int sum = 0;
        while (k--)
        {
            int c = floor(pq.top() - pq.top() / 2);
            pq.pop();
            pq.push(c);
        }

        while (!pq.empty())
        {
            sum += pq.top();
            pq.pop();
        }
        return sum;
    }
};
