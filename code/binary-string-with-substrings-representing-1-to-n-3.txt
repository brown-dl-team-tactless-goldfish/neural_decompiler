public bool QueryString(string S, int N) {
	for (var i = 1; i <= N; i++)
		if (!S.Contains(ToBinary(i)))
			return false;

	return true;

	string ToBinary(int n) {
		var str = new StringBuilder();

		while (n > 0) {
			str.Append(n % 2);
			n /= 2;
		}

		return new string(str.ToString().Reverse().ToArray());
	}
}
