bool validPalindrome(char * s){
    int left,right,preLeft,preRight;
    int len = strlen(s);
    int found = 0;
    
    left = 0;
    right = len-1;
    while(left<right)
    {
        if(s[left]==s[right])
        {
            left++;
            right--;
        }
        else
        {
            if(found == 2)
            {
                return false;
            }
            else
            {
                found++;
                if(found == 1)
                {
                    preLeft = left;
                    preRight = right;
                    left = preLeft + 1;
                    right = preRight;
                }
                else
                {
                    left = preLeft;
                    right = preRight - 1;
                }
            }
        }
    }
    
    return true;
}
