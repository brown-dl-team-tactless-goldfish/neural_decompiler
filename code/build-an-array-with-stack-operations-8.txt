class Solution {
public:
    vector<string> buildArray(vector<int>& target, int n) {
     
        stack<int> s;
        vector<string> v;
        
        int j=0; // iterating target array 
        
        
        for(int i=1;i<=n;i++){  // checking 1 to n with our target array 
            
            if(j>=target.size()){ // if j will greate than size of target array exit loop
                break;
            }
            
            if(target[j]==i){  // if our target element and our i both are same mena we need this element in out stac jo just push
                v.push_back("Push");
                j++;
            }
            else { // if target eleent and ourt j not same means we have to do push and pop after that cozz we can't skip any element
                v.push_back("Push");
                v.push_back("Pop");
            }
            
        }
        return v; // this is out ans
        
    }
};
