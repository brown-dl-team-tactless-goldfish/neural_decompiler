// OJ: https://leetcode.com/contest/biweekly-contest-33/problems/thousand-separator/
// Author: github.com/lzl124631x
// Time: O(N)
// Space: O(1)
class Solution {
public:
    string thousandSeparator(int n) {
        if (n == 0) return "0";
        int d = 0;
        string ans;
        while (n) {
            ans += '0' + n % 10;
            n /= 10;
            if (++d % 3 == 0 && n) ans += '.';
        }
        reverse(begin(ans), end(ans));
        return ans;
    }
};
