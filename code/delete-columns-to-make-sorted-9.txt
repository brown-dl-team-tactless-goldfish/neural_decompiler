int minDeletionSize(char ** strs, int strsSize){
    
    int counter = 0;

    
    int strLen = strlen(strs[0]);

    
    for (int i = 0; i < strLen; i++){
        
        bool sorted = true;

       
        for (int j = 0; j < strsSize - 1; j++){
           
            if (strs[j][i] > strs[j+1][i]){
                sorted = false;
                break;
            }
        }

       
        if (!sorted){
            counter++;
        }
    }

    return counter;
}
