struct Node* connect(struct Node* root) {
    struct Node *parent = root;
    struct Node *temp = NULL;

    if (root == NULL) {
        return NULL;
    }
    /* If there is a child, then go ahead else stop*/
    /* Since its a balanced binary tree */
    while(parent->left != NULL) {
        /* First connect the parent's left to its right child */
        parent->left->next = parent->right;
        temp = parent;
        /* If the parent has a next, then connect its right child to 
         the next's left. Also connect next's left to right child.
         Move to its next node if present */
        while(temp->next != NULL) {
            temp->right->next = temp->next->left;
            temp->next->left->next = temp->next->right;
            temp = temp->next;
        }
        /* Set the last node's to NULL */
        temp->next = NULL;
        parent = parent->left;
    }
    return root; 
}
