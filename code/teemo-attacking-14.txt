int max(int a,int b,int c){
    int max=a>b?a:b;
    max=max>c?max:c;
    return max;
}
int findPoisonedDuration(int* time, int size, int d){
    if(size==0)
        return 0;
    if(size==1)
        return d;
    int i,cnt=0,excess=0;
    for(i=0;i<size;i++){
        int gap=(i==size-1?0:time[i+1]-time[i]);
        excess=max(0,d-gap,excess-gap);
        if(excess==0)
            cnt+=d;
        else
            cnt+=gap;
    }
    return cnt+excess;
}
