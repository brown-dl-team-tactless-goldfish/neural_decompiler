long long numberOfWays(char * s){
    int box[2] = {0};
    long long zero_one = 0;
    long long one_zero = 0;
    int len = strlen(s);
    long long ans = 0;
    for(int i = len-1 ; i >= 0 ; i--){
        if(s[i] == '0'){
            box[0]++;
            zero_one += box[1];
            ans += one_zero;
        } else {
            box[1]++;
            one_zero += box[0];
            ans += zero_one;
        }
    }
    return ans;
}
