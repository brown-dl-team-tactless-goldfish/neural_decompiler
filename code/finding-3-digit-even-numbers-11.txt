class Solution {
public:
  vector<int> findEvenNumbers(vector<int>& digits) {
    vector<int> answer;
    int D[10] = {0};
    for(auto &d: digits) D[d]++;
    
    for(int i = 1; i != 10; D[i++]++)
      if(D[i]--)
        for(int j = 0, n = 100*i; j != 10; D[j++]++)
          if(D[j]--)
            for(int k = 0, N = n + 10*j; k != 10; k += 2)
              if(D[k]) answer.push_back(N + k);
            
    return answer;
  }
};
