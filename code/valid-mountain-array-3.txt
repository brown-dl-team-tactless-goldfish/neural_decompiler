
bool validMountainArray(int* arr, int arrSize){
    if(arrSize<3)return false;//if size is less than 3.
    int i=0;
    if(arr[0]>arr[1])return false;//if first element is greater than second it measn there is no up slope so false.
    for(i = 0;i<arrSize-1;i++){
        if(arr[i]==arr[i+1])return false;//if any two element is same then it's not strictly increasing.
        if(arr[i]>arr[i+1])goto downSlope;//when we found downslope
    }
    return false;//if we did not found down slope it means it is false.
    downSlope: 
	for(;i<arrSize-1;i++){
        if(arr[i]==arr[i+1])return false;
        if(arr[i]<arr[i+1])return false;
    }
    return true;
}
