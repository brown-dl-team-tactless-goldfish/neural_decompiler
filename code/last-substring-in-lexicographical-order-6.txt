string lastSubstring(string s, int i = 0, int sz = 0) {
  for (auto j = 1; j < s.size(); ++j) {
    for (sz = 0; j + sz < s.size(); ++sz) {
      if (s[i + sz] == s[j + sz]) continue;
      i = s[j + sz] > s[i + sz] ? j : i;
      break;
    }
    if (j + sz == s.size()) break;
  }
  return s.substr(i);
}
