class Solution {
public:
        
   void uniquestr(string s,unordered_set<string> &se,int &ans){
    
        if(s.empty())return;   
     
        for(int j=0;j<s.length();j++){
                
             string res=s.substr(0,j+1);       // checking every possible string 
        
             if(se.find(res)==se.end()){
                se.insert(res);
                     
                uniquestr(s.substr(j+1),se,ans);
                                                          
                if(se.size()>ans){             // when string is empty we are checking our ans
                    ans=se.size(); 
                }
                se.erase(res);                 // erasing string from set 
             }                                 // to check other possible  partitions
        }
   }     
        
    int maxUniqueSplit(string s){
        int ans=0;
        unordered_set<string> se;
        uniquestr(s,se,ans);    
        
    return ans;        
    }
};
