class Solution {
public:
    bool canBeIncreasing(vector<int>& nums) {
        int n = nums.size();
        int c = 0;
        if(n == 1 || n == 2)
            return true;
        for(int i = 1; i < n; i++)
        {
            if(nums[i] <= nums[i - 1])
                c++;
            if( i >= 2 && nums[i-2] >= nums[i])
            {   
                nums[i] = nums[i-1];
            }
            if(c > 1)
                return false;
        }
        return true;
    }
};
