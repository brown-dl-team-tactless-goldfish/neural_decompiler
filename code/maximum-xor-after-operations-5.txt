class Solution {
public:
    int maximumXOR(vector<int>& nums) {
        vector<int> bits(32,0);
        for(auto it:nums){
            for(int i=0; i<32; i++){
                if((it&(1<<i))!=0)
                    bits[i]++;
            }
        }
        int ans = 0;
        for(int i=0; i<32; i++)
            if(bits[i] > 0)
            ans = (ans|(1<<i));
        return ans;
    }
};
