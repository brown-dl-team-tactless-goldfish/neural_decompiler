int integerReplacement(int n) {
    int step = 0;
    if (n == INT_MAX)
        return 32;

    if (n == 1 || n == 0)
        return 0;
        
    else if (n % 2 == 0 )
    {
        step = integerReplacement(n/2) + 1;
    } else {
        int step1 = integerReplacement(n - 1) + 1;
        int step2 = integerReplacement(n + 1) + 1;
        step = step1 < step2?step1:step2;
    }
    return step;
}
