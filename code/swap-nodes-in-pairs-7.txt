struct ListNode* swapPairs(struct ListNode* head)
{
int clk=0;
    struct ListNode*l1=head;
    if(head==NULL || head->next==NULL)
        return head;
    while(l1->next!=NULL)
    {
        
        if(clk%2==0)
        {
           int temp=l1->val;
            l1->val=l1->next->val;
            l1->next->val=temp;
         }
           l1=l1->next;
           clk++;
        
        
    }
    return head;
}
