#include <stdlib.h>    // For memset

unsigned int val[34];
int* getRow(int rowIndex, int* retSize){
    memset(val, 0, sizeof val);
    val[0] = 1;
    *retSize = rowIndex + 1;
    
    for (int i=0; i<rowIndex; ++i)  {
        int prev = 1;
        const int totalElems = rowIndex+1;
        const int half = totalElems/2;
        int j = 1;
        for (; j<half; ++j)  {
            val[j] = prev + val[j];
            prev = val[j] - prev;
            val[totalElems-j-1] = val[j];
        }
        if (totalElems & 1)  val[j] = prev+val[j];
        val[totalElems-1] = 1;
    }
    
    return val;
}
