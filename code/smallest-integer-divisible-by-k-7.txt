int smallestRepunitDivByK(int k){
    if(k % 2 == 0 || k % 5 == 0) return -1;
    int temp = 0, answer = 0, i;
    do{
        for(i = 0; i <= 9; i++){
            if((k * i + temp) % 10 == 1) break;
        }
        answer++;
        temp = (k * i + temp) / 10;
    }while(temp != 0);
    return answer;
}
