bool reachingPoints(int sx, int sy, int tx, int ty) {
        if(sx==tx && sy==ty) return true;
        
        while(tx>sx && ty>sy){
            if(tx>ty){
                tx%=ty;
            }else{
                ty%=tx;
            }
        }
        
        if(sx==tx){
            return ty>=sy && (ty-sy)%tx==0;
        }
            
        if(sy==ty){
            return tx>=sx && (tx-sx)%ty==0;
        }
        
        return false;
    }
