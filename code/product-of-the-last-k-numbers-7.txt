vector<int> v;

ProductOfNumbers() {
    
}

void add(int num)
{
    if(num == 0)
    {
        v.clear();
        
        return;
    }
    
    if(v.size() == 0)
    {
        v.push_back(num);
    }
    else
    {
        v.push_back(v.back() * num);
    }
    
}

int getProduct(int k) 
{
    if( k> v.size())
    {
        return 0;
    }
    
    if( v.size() == k)
    {
        return v[k-1];
    }
    
    else
    {
        return v.back() / v[v.size() - 1 - k];
    }
}
