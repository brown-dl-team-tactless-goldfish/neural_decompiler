void searchDelete(struct TreeNode** root,int target){
    if(*root){
        
        searchDelete(&((*root)->left),target);
        searchDelete(&((*root)->right),target);
         if(!(*root)->left&&!(*root)->right&&(*root)->val==target){
            (*root) = NULL;
            return;
        }
    }
}
struct TreeNode* removeLeafNodes(struct TreeNode* root, int target){
    searchDelete(&root,target);
    return root;
}
