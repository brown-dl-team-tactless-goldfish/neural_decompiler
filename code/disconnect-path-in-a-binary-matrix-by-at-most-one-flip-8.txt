class Solution {
public:
    bool isPossibleToCutPath(vector<vector<int>>& grid) 
    {
        int m=grid.size(),n=grid[0].size();
        vector<int> v(m+n-1,0);
        for(int i=0;i<m;i++)
        {
            for(int j=0;j<n;j++)
            {
                v[i+j]+=grid[i][j];
            }
        }
        for(int i=1;i<m+n-2;i++)
        {
            if(v[i]<=1)
                return true;
        }
        
        return false;
    }
};
