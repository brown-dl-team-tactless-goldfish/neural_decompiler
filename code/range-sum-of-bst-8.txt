/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     struct TreeNode *left;
 *     struct TreeNode *right;
 * };
 */
int rangeSumBST(struct TreeNode* root, int low, int high){
    //stopping the itertion 
    if(!root){return 0;}

    //condtion where we will add all the numbers which are greater than Low and lesser than High
    else if(root->val >= low && root->val <= high){
        //current value and traverse left and traverse right
        return root->val + rangeSumBST(root->left,low,high) + rangeSumBST(root->right,low,high);
    }
    else{
        //traverse without adding to return value 
        return rangeSumBST(root->left,low,high) + rangeSumBST(root->right,low,high);
    }

}
