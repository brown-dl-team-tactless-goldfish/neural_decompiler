    int lastStoneWeightII(vector<int>& stones) {
        int sum = accumulate(stones.begin(), stones.end(),0);
        int minDiff = sum;
        vector<bool> dp(sum+1, 0);
        dp[0] = true;
        for(auto x:stones)
            for (int i = dp.size()-1; i >=0; i--) if(dp[i]) dp[i+x] = true;  
        for(int i=0;i<dp.size();i++)
            if(dp[i]) minDiff = min(minDiff, abs(i*2-sum));
        return minDiff;
    }
