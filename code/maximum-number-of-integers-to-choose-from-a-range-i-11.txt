class Solution {
public:
    int maxCount(vector<int>& banned, int n, int maxSum) {
        vector<int> nums;
        long long sum=0;
        int count=0;

        for(int i=1;i<=n;i++){
            nums.push_back(i);//storing the numbers from [1,n]
        }
        unordered_set<int> ban(banned.begin(),banned.end());//storing banned numbers in set
        for(int i=0;i<nums.size();i++){
            if(ban.find(nums[i])!=ban.end())continue;//if the number is banned

            if(sum+nums[i]<=maxSum){//If we can add the number to sum and still it will be less than or equal to maxSum
                sum+=nums[i];
                count++;//Increment count as the number is used
            }
            else break;//If the sum exceeds max sum at any stage
        }
        return count;
    }
};
