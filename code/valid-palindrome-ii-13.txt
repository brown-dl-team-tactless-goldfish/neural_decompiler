bool isPalindrome(char* s,int left,int right){
    while(right>left){
        if(s[left]!=s[right]){
            return false;
        }
        right--;
        left++;
    }
    return true;
}

bool validPalindrome(char * s){
    int len=strlen(s);
    int right=len-1;
    int left=0;
    if(!s)
        return true;
    while(right>left){
        if(s[left]!=s[right]){
            return isPalindrome(s,left+1,right)||isPalindrome(s,left,right-1);
        }
        right--;
        left++;  
    }
    return true;
}
