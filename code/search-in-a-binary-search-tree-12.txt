struct TreeNode* searchBST(struct TreeNode* root, int val){
    struct TreeNode* temp = root;
    while(temp != NULL){
        if(temp->val == val){
            break;
        }
        else if(val > temp->val){
            temp = temp->right;
        }
        else{
            temp = temp->left;
        }
    }
    return temp;
}
