int binaryGap(int n) {
    int map[32] = {0}, idx = 0, ans = 0;
    while (n > 0) {
        map[idx++] = (n % 2 == 1) ? 1 : 0;
        n /= 2;
    }
    for (int i = 0; i < idx; i++) {
        if (map[i] == 1) {
            for (int j = i + 1; j < idx; j++) {
                if (map[j] == 1) {
                    ans = j - i > ans ? j - i : ans;
                    break;
                }
            }
        }
    }
    return ans;
}
