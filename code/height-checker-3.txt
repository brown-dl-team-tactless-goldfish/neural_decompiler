int cmpfunc(const void * a, const void * b){
    return (*(int*)a - *(int*)b);
}

int heightChecker(int* heights, int heightsSize){
    int i;
    int* temp = malloc(sizeof(int)*heightsSize);
    //sort it
    for(i=0;i<heightsSize;i++){
        temp[i] = heights[i];
    }
    
    qsort(temp, heightsSize, sizeof(int), cmpfunc);

    //find out how many are in wrong spot
    int ret = 0;
    for(i=0;i<heightsSize;i++){
        if(heights[i] != temp[i])
            ret++;
    }
    
    //return it
    return ret;    
}
