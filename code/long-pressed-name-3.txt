bool isLongPressedName(char * name, char * typed){
    int len1 = strlen(name);
    int len2 = strlen(typed);
    if(len1<1 || len1>len2 || name[0]!=typed[0]) return false;
    int i=1;
    int j=1;
    while(i<len1){
        if(j>=len2) return false;
        if(name[i]==typed[j]){
            i++;
            j++;
            continue;
        }
        if(name[i-1]==typed[j]){
            j++;
            continue;
        }
        return false;
    }
    return true;
}
