class Solution {
public:
    
    int reverse(int n)
    {
        int sum=0;
        while(n!=0)
        {
            int temp=n%10;
            sum=sum*10+temp;
            n=n/10;
        }
        return sum;
    }
    
    bool isSameAfterReversals(int num) {
        int x=reverse(num);
        int y=reverse(x);
        if(num==y)
        {
            return true;
        }
        return false;
    }
   };
