long long findTheArrayConcVal(int* nums, int numsSize){
    int n = numsSize ;
    long long ans = 0 ;
    for(int i = 0; i < n/2 ; i++){
        int d = 0 ;
        int tmp = nums[n-1-i] ;
        while(tmp){
            d++ ;
            tmp/= 10 ;
        }
        ans += (long long)nums[i]* pow(10, d) + nums[n-1-i] ;
    }

    if(n % 2)
        ans += nums[n/2] ;
    return ans ;
}
