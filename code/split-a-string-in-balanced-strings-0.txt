int balancedStringSplit(char * s){
    int ans = 0;
    int Ln = 0;
    int Rn = 0;
    for(int i = 0; i < strlen(s); i++){
        if(s[i] == 'L'){
            Ln++;
            if(Ln == Rn){
                ans++;
                Ln = 0;
                Rn = 0;
            }
        }
        if(s[i] == 'R'){
            Rn++;
            if(Ln == Rn){
                ans++;
                Ln = 0;
                Rn = 0;
            }
        }
    }
    return ans;
}
