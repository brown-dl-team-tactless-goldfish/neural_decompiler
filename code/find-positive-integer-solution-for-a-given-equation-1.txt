int** findSolution(int (*customFunction)(int, int), int z, int* returnSize, int** returnColumnSizes) {
    int **a = malloc(sizeof(int *[z])), k = 0;
	int *c = *returnColumnSizes = malloc(sizeof(int [z]));
    for (int i = 1 ; i <= z ; i++)
        for (int j = z ; j >= 1 ; j--)
            if (customFunction(i, j) == z)
                (a[k] = malloc(sizeof(int [c[k] = 2])))[0] = i, a[k++][1] = j;
    return *returnSize = k, a;
}
