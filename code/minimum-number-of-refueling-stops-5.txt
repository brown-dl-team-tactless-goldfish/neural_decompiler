class Solution {
public:
    int minRefuelStops(int t, int f, vector<vector<int>>& S) {
        sort(begin(S), end(S));
        priority_queue<int> pq;
        int res = 0, i = 0;
        while(true) {
            if(f >= t) return res;
            while(i < S.size() && f >= S[i][0]) pq.push(S[i++][1]);
            if(pq.empty()) break;
            f += pq.top(); pq.pop();
            res++;
        }
        return -1;
    }
};
