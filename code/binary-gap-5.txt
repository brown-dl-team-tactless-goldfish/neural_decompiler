
int binaryGap(int n){

    int max = 0;
    int c = 0;
    int flag = 0;

    while(n)
    {
        if(n & 1)
        { 
            if(flag == 0)
            { 
                flag = 1;
                c++;
            }
            else
            {       
                if(c > max)
                    max = c;
                c = 1;
            }
        }
        else
        {
            if(flag == 1)
                c++;
        }
        n = n >> 1;
    }
    return max;
}
