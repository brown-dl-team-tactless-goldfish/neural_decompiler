
int maxDistance(int* colors, int colorsSize){
    int max = 0;
    for (int i = 0; i < colorsSize; i++) {
        for (int j = colorsSize - 1; j > i + max; j--) {
            if (colors[j] != colors[i]) {
                int distance = j - i;
                max = (max > distance) ? max : distance;
                break;
            }
        }
    }
    return max;
}
