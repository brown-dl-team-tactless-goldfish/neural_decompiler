int kItemsWithMaximumSum(int numOnes, int numZeros, int numNegOnes, int k){
    int ret = 0 ;
    if(k <= numOnes){
        ret += k ;
        return ret ;
    }
    else{
        ret += numOnes ;
        k -= numOnes ;
    }
    
    if(k <= numZeros){  
        return ret ;
    }
    else
        k -= numZeros ;
    
    if(k <= numNegOnes)
        ret -= k ;
    else
        ret -= numNegOnes ;
    return ret ;
}
