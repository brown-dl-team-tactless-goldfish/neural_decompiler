
char* removeDuplicates(char* s) {
    int reslen = strlen(s), idx = 0;
    char* res = (char*) malloc(reslen * sizeof(char));
    res[idx++] = s[0];
    for (int i = 1; i < reslen; i++) {
        res[idx] = s[i];
        if (idx >= 1 && res[idx] == res[idx-1]) idx -= 2;
        idx++;
    }
    res[idx] = '\0';
    return res;
}
