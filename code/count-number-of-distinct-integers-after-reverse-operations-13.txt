public class Solution {
    public int CountDistinctIntegers(int[] nums) {
        HashSet<int> numDict = new();
        
        foreach(int num in nums)
        {
            int number = num;
            if(!numDict.Contains(number))
            {
                numDict.Add(number);
            }
            
            int rNum =0;
            while(number !=0)
            {
                int digit = number %10;
                rNum =rNum* 10+ digit;
                number= number/10;
                
            }
             if(!numDict.Contains(rNum))
            {
                numDict.Add(rNum);
            }        
        }
        return numDict.Count;
    }
}
