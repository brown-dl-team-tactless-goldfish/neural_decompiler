public class Solution {
    public long MinimumReplacement(int[] nums) {
        long result = 0;
        
        for (int i = nums.Length - 2; i >= 0; ) 
            if (nums[i] <= nums[i + 1]) 
                i -= 1;
            else {
                int breakCount = ((nums[i] - 1) / nums[i + 1]) + 1;
            
                nums[i] /= breakCount;
            
                result += breakCount - 1;
            }
           
        return result;
    }
 }
