class Solution {
public:
    vector<int> maxScoreIndices(vector<int>& nums) {
int one=0;
for(auto i : nums)
{
    if(i==1) one++;
}
int ans=one;
int zero=0;
int i=0;
vector<int>ok;
ok.push_back(0);
while(i<nums.size())
{
    if(nums[i]==0) zero++;
    else one--;
    if(ans==zero+one) ok.push_back(i+1);
    else if(ans<zero+one)
    {
        ok.clear();
        ok.push_back(i+1);
        ans=zero+one;
    }
i++;
}
        return ok;
    }
};
