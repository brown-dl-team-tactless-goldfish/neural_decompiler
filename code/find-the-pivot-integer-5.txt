int right_sum( int pivot, int limit )
{
    int sum = 0;
    for(int i = limit; i > pivot; i--){ sum += i; }
    return sum;
}
int pivotInteger(int n)
{
   for( int i = n ; i >= 1; i--)
   {
       int left_sum = (i * (i - 1))/2 ;
       if( left_sum == right_sum(i,n) ){ return i; }
   }
   return -1;
}
