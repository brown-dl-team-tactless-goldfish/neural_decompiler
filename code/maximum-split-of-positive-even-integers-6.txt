class Solution {
public:
    vector<long long> maximumEvenSplit(long long finalSum) {
        
        long long sum=2;
        vector<long long> ans;
        if(finalSum%2==1) return ans;
        while(finalSum!=0){
            if(finalSum-sum<0){
                ans[ans.size()-1]=ans[ans.size()-1]+finalSum;
                break;
            }
            finalSum-=sum;
            ans.push_back(sum);
            sum=sum+2;
        }
        return ans;
    }
};
