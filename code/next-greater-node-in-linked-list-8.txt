// brute force method 
int* nextLargerNodes(struct ListNode* head, int* returnSize)
{
    //count the number of nodes 
    struct ListNode* ptr = head;
    int size = 0;
    while(ptr != NULL)
    {
        size++;
        ptr=ptr->next;
    }
    
    *returnSize = size;
    int* arra = (int*)malloc(size * sizeof(int));
    
    struct ListNode* current = head;
    struct ListNode* traverse = head;
    int index = 0;
    
    while(current != NULL)
    {
        traverse = current->next;
        int set = 0;
        while (traverse != NULL)
        {
            if (traverse->val > current->val)
            {
                arra[index] = traverse->val;
                set = 1;
                index++;
                break;
                
            }
            traverse = traverse->next;
        }
        if (set==0)
        {
            arra[index] = 0;
            index++;
        }
        current = current->next;
    }
    
    return arra;

}
