int compare(void *a, void *b) {
    return (*(int*)a - *(int*)b);
}

int lastStoneWeight(int* stones, int stonesSize){
    
    int i;
    
    for (i = stonesSize - 1; i > 0; i--) {
        qsort(stones, i+1, sizeof(int), compare);
        stones[i-1] = stones[i] - stones[i-1];
    }

    return stones[0];  
}
