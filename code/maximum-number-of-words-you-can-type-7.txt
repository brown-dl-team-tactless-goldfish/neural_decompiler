int canBeTypedWords(char * text, char * brokenLetters){
    int textLength = strlen(text), wordCount = 1, typableWord = 1;
    for (int i = 0; i < textLength; i++) {
        if (text[i] == ' ') {
            wordCount++;
            typableWord = 1;
        } else if (strchr(brokenLetters, text[i]) != NULL && typableWord != 0) {
            wordCount--;
            typableWord = 0;
        }
    }
    return wordCount;
}
