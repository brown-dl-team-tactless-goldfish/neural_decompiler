typedef struct {
    volatile int turn;
} Foo;

Foo* fooCreate() {
    Foo* obj = (Foo*) malloc(sizeof(Foo));
    obj->turn = 1;
    return obj;
}

void first(Foo* obj) {
    printFirst();
    obj->turn = 2;
}

void second(Foo* obj) {
    while (obj->turn != 2)
        ;
    printSecond();
    obj->turn = 3;
}

void third(Foo* obj) {
    while (obj->turn != 3)
        ;
    printThird();
}

void fooFree(Foo* obj) {
    free(obj);
}
