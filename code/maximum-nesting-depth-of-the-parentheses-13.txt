

int maxDepth(char * s){
    int counter=0,max=0;// counter: present depth; max: present max depth
    int l = strlen(s);
    
    for(int x=0;x<l;x++){ // x: index to scan s
        if(s[x]=='('){ //if '(', present depth ++
            counter++;
            if(max<counter){ //update max
                max=counter;
            }
        }else if(s[x]==')'){//if ')', present depth --
            counter--;
        }
    }
    return max;
}
