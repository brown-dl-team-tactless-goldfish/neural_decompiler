int comp(int *a,int *b){
    return *b-*a;
}
int splitNum(int num){
    int n = num;
    int res[10]={0},i=0;
    while(n){
        res[i]=n%10;
        n=n/10;
        i++;
    }
    qsort(res,i,sizeof(int),comp);
    int s=1,k=1,n1=0,n2=0;
    for(int j=0;j<i;j++){
        if(j%2==0){
            n1+=res[j]*s;
            s=s*10;
        }
        else{
            n2+=res[j]*k;
            k=k*10;
        }
    }
    return n1+n2;
}
