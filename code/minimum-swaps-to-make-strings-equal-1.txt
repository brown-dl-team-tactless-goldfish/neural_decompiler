int minimumSwap(char * s1, char * s2){
    int l = strlen(s1);
    int xy = 0;
    int yx = 0;
    for(int i = 0; i < l; i++)
    {
        if (s1[i] != s2[i])
        {
            if (s1[i] == 'x')
                xy += 1;
            else
                yx += 1;
        }
    }
    if ((xy + yx) % 2 != 0)
    {
        return -1;
    }
    else
    {
        if (xy % 2 == 0)
            return xy / 2 + yx / 2;
        else
            return xy / 2 + yx / 2 + 2;
    }

}
