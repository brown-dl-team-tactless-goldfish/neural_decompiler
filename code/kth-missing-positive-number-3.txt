#define MAX 10000

int findKthPositive(int* arr, int arrSize, int k){
     int V[MAX] = {0},max=arr[0],count=0,ret,flag=0;
    for(int i=0;i<arrSize;i++){
        V[arr[i]]++;
        if(arr[i]>max)
            max = arr[i];
    }

    for(int i=1;i<max && flag==0;i++){
        if(V[i]==0)
            count++;
        if(count==k){
            ret = i;
            flag=1;
        }

    }
    if(count<k)
        ret = max+k-count;
    return ret;

}
