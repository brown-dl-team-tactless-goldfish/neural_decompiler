class Solution {
public:
    int sumFac(int n) {
        int sum = 0, c = 2;
        while(n > 1) {
            if(n % c == 0) {
                sum += c;
                n /= c;
            } else c++;
        }
        return sum;
    }
    
    int smallestValue(int n) {
        while(n != sumFac(n)) 
            n = sumFac(n);
        return n;
    }
};
