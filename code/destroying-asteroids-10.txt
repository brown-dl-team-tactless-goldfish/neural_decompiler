class Solution {
public:
    bool asteroidsDestroyed(int mass, vector<int>& asteroids) {
        sort(asteroids.begin(),asteroids.end());

        long long res = mass;
        for(auto i : asteroids){
            if(res < i){
                return false;
            }else{
                res += i;
            }
        }
        return true;
    }
};
