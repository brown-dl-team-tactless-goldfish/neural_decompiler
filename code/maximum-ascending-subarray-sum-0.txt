int maxAscendingSum(int* nums, int numsSize){

    int maxSum = nums[0];
    int temp = nums[0];
    
    for(int i = 1; i < numsSize; i++){
        if(nums[i] > nums[i-1]){
            temp += nums[i];
        }
        else{
            if(temp > maxSum)
                maxSum = temp;
            temp = nums[i];
        }
    }
    if(temp > maxSum)
        maxSum = temp;
    
    return maxSum;

}
