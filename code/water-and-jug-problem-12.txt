class Solution {
public:
    bool canMeasureWater(int x, int y, int z) {
        
        if (x == z || y == z || x + y == z) return true;
        
        if (x + y < z) return false;
        
        if (z == 0) return true;
        
        int a = gcd(x, y);
        int b = gcd(y, z);
        
        if (a == b)
        {
            x /= a;
            y /= a;
            z /= a;
        }
        
        return (gcd(x, y) == 1) ? true : false;
    }
};
