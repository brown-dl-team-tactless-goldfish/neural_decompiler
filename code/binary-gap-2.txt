int binaryGap(int N) {
    int index,max,gap,i;
    max = gap = 0;
    for(i = 0; N != 0 && N % 2 == 0; i++)
        N /= 2;
    index = i;
    for(; N != 0; i++)
    {
        if(N % 2 == 1)
        {
            gap = i - index;
            index = i;
            if(max < gap)
                max = gap;
        }
        N /= 2;
    }
    return max;
}
