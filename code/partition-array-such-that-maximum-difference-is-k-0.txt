int cmp(const void* a, const void* b) {
    return *(const int*) b - *(const int*) a;
}

int partitionArray(int* nums, int numsSize, int k) {
    qsort(nums, numsSize, sizeof(int), cmp);
    int start = 0, cnt = 0;
    for (int i = 1; i < numsSize; i++) {
        if (nums[start] - nums[i] > k) {
            start = i;
            cnt++;
        }
    }
    return cnt + 1;
}
