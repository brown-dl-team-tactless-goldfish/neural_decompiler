class Solution {
public:
	int bestClosingTime(string customers) {
		int ans=0, maxi=INT_MAX, pos;
		for(int i=0; i<customers.size(); i++){
			if(customers[i]=='Y'){ans++;}
		}
		if(ans<maxi){maxi=ans; pos=0;}
		for(int i=0; i<customers.size(); i++){
			if(customers[i]=='Y'){
				ans--;
			}
			else{ans++;}
			//cout<<ans<<endl;
			if(ans<maxi){maxi=ans; pos=i+1;}
		}
		return pos;
	}
};
