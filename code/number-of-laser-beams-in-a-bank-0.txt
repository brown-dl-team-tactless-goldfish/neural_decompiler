

int numberOfBeams(char ** bank, int bankSize){
    int ans = 0;
    int start = 0;
    for (int i = 0; i < bankSize; i++){
        int n = 0; 
        for (int j = 0; ;j++){
            if (!bank[i][j])
                break;
            else if (bank[i][j] == '1')
              n++;  
        }
        if (n > 0){
            if (start)
                ans += start * n;
            start = n;
        }
    }
    return ans;
}
