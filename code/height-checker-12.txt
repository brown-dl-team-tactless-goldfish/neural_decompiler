int cmp(const void *a,const void *b){
  return (*(int *)a -*(int *)b);
}
int heightChecker(int* a, int n){
    int count=0;
   int b[n];
   for(int i=0;i<n;i++){
       b[i]=a[i];
   }
   qsort(b,n,sizeof(int),cmp);
   for(int i=0;i<n;i++){
       if(a[i]!=b[i])
           count++;
   }
   return count;
}
