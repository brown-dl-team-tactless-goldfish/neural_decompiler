class Solution:
    def minOperations(self, s: str) -> int:
        n = len(s)
        
        self.res = float('inf')
        def helper(curIdx, target, curChanges):
            if (curIdx == n):
                self.res = min(self.res, curChanges)
                return
            
            newTarget = 1
            if (target == 1):
                newTarget = 0
                
            if (int(s[curIdx]) != target):
                helper(curIdx+1, newTarget, curChanges+1)
            else:
                helper(curIdx+1, newTarget, curChanges)
                
        #start with "0"
        helper(0, 0, 0)
        #start with "1"
        helper(0, 1, 0)
        
        return self.res
