long long subArrayRanges(int* nums, int numsSize){
    long long ans = 0;
    int n = numsSize;
    for(int i = 0; i < n; i++){
        int min = nums[i], max= nums[i];
        for(int j = i; j < n; j++){
            min = fmin(min, nums[j]);
            max = fmax(max, nums[j]);
            ans += (max - min);
        }
    }
    return ans;
}
