class Solution {
public:
    int minGroups(vector<vector<int>>& nums) {
        map<int,int> mp;
        for(int i = 0; i<nums.size();++i){
            mp[nums[i][0]]++; mp[nums[i][1]+1]--;
        }
        int re = 0, sum = 0;
        for(auto&[i,val]:mp){
            sum+=val;
            re = max(re,sum);
        }
        return re;
    }
};
