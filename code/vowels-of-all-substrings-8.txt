typedef long long ll;
class Solution {
public:
    long long countVowels(string word) {
        ll ans = 0, n = word.size();
        unordered_set<char> c = {'a','e','i','o','u'};
        for(ll i=0; i<n; i++){
            if(c.find(word[i]) != c.end()){
                ans += (i+1)*(n-i);
            }
        }
        return ans;
    }
};
