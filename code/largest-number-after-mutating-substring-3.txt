class Solution {
public:
    string maximumNumber(string num, vector<int>& change) {
        bool changed = false;
        
        for(auto &c : num){
            
            int curr = (c - '0');
            int from_change = change[c-'0'];
            
            if(curr < from_change) {
                c = '0' + from_change;
                changed = true;
            } else if(curr == from_change) {
                continue;
            } else {
                if(changed) break;
            }
        }
        
        return num;
    }
};
