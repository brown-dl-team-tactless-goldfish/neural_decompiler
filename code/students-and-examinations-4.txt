SELECT
	sd.student_id,
	sd.student_name,
	su.subject_name,
	ifnull( c.attended_exams, 0 ) attended_exams
FROM
	(Students sd,
	Subjects su)
	LEFT JOIN ( SELECT student_id, subject_name, count( * ) attended_exams FROM Examinations GROUP BY student_id, subject_name ) c ON sd.student_id = c.student_id 
	AND su.subject_name = c.subject_name 
ORDER BY
	sd.student_id,
	su.subject_name
