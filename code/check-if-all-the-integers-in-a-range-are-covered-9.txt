    for(int i=left; i<=right; i++)
    {
        bool seen = false;
        
        for(auto x : ranges)
        {
            if(i>= x[0] && i<= x[1])
            {
                seen = true;
            }
        }
        
        if(seen == false)
        {
            return false;
        }
    }
    
    return true;
    
}
