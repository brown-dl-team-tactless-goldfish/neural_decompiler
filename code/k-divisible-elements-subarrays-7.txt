class Solution {
public:
    set<vector<int>> s ;
    int countDistinct(vector<int>& nums, int k, int p) {
        
        for(int i = 0 ; i < nums.size() ; ++i ){
            int count = 0 ;
            vector<int> makeAns ;
            for(int j = i; j < nums.size() ; ++j ){
                count += (nums[j]%p == 0 ) ; if(count > k) break ;
                makeAns.push_back(nums[j]) ;
                s.insert(makeAns);
            }
        }
        return (int)s.size() ;
        
    }
};
