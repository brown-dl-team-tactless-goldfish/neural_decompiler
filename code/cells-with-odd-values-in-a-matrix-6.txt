int oddCells(int m, int n, int** indices, int indicesSize, int* indicesColSize){
    int matrix[m][n], i, j, count = 0;
    for (i = 0; i < m; ++i) for (j = 0; j < n; ++j) matrix[i][j] = 0;
    for (i = 0; i < indicesSize; ++i) {
        int r = indices[i][0], c = indices[i][1];
        for (j = 0; j < n; ++j) ++matrix[r][j];
        for (j = 0; j < m; ++j) ++matrix[j][c];
    }
    for (i = 0; i < m; ++i) for (j = 0; j < n; ++j) if (matrix[i][j] % 2 == 1) ++count;
    return count;
}
