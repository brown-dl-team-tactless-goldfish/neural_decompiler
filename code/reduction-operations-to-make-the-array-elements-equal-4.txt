class Solution {
public:
    int reductionOperations(vector<int>& nums) {
        map<int, int> mp;
        for(int &num : nums)
        {
            mp[num]++;
        }
        int ans = 0, val = 0;
        for(auto &[fst, snd] : mp)
        {
            ans += snd*val;
            val ++;
        }
        return ans;
    }
};
