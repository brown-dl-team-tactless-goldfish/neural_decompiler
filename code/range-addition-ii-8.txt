class Solution:
    def maxCount(self, m: int, n: int, ops: List[List[int]]) -> int:
        min_x, min_y = float('inf'), float('inf')

        for op in ops:
            min_x = min(min_x, op[0])
            min_y = min(min_y, op[1])

        # print("min_x = ", min_x, " min_y = ", min_y)

        if (min_x == float('inf') or min_y == float('inf')):
            return m*n
        return min_x*min_y
