class Solution {
public:
    int minCost(vector<int>& S, vector<int>& H, vector<int>& rowCosts, vector<int>& colCosts) {
        int res = 0, rinc = H[0] > S[0] ? 1 : -1, cinc = H[1] > S[1] ? 1 : -1;
        for(int i = S[0]; i != H[0]; i+= rinc) res += rowCosts[i+rinc];
        for(int i = S[1]; i != H[1]; i+= cinc) res += colCosts[i+cinc];
        return res;
    }
};
