public class Solution {
    public IList<IList<int>> FindDifference(int[] nums1, int[] nums2) {
        var res = new List<IList<int>>();
        var set = nums1.Concat(nums2).ToHashSet();
        res.Add(set.Except(nums2).ToList());
        res.Add(set.Except(nums1).ToList());

        return res;
    }
}
