class Solution {
public:
    int maxSumAfterPartitioning(vector<int>& arr, int k) {
        int n = arr.size();
        vector<int> dp(n + 1);

        for(int i=n-1; i>=0; i--) {
            int ans = 0, mx = 0;
            for(int j=i; j<min(i+k, n); j++) {
                mx = max(mx, arr[j]);
                ans = max(ans, (j - i + 1) * mx + dp[j + 1]);
            }
            dp[i] = ans;
        }
        return dp[0];
    }
};
