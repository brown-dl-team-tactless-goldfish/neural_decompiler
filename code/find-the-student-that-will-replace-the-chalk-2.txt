int chalkReplacer(int* chalk, int chalkSize, int k){
    if(k<chalk[0]) return 0;
    for(int i=1;i<chalkSize;i++){
        chalk[i]+=chalk[i-1];
        if(chalk[i] > k) return i;
    }
    k%=chalk[chalkSize-1];
    
    for(int i=0;i<chalkSize;i++){
        if(chalk[i]>k) return i;
    }
    return 0;
}
