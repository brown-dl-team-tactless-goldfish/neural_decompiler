    int sum = 0, ans = 0;
    
    for(auto &x: nums)
    {
        sum += x;
       
        
        if(sum == target)
        {
            ans++;
            sum = 0;
            mp.clear();
        }
         
       else if(mp.count(sum-target))
        {
            ans++;
            sum = 0;
            mp.clear();
        }
        
        mp[sum]++;
        
    }
    
    return ans;
}
