class Solution {
public:
	int minDeletionSize(vector<string>& s) {
		unordered_set<int> st;
		int n = s.size(), m = s[0].size();
		for (int i = 1; i < n; ++i) {
			for (int j = 0; j < m; ++j) {
				if (st.count(j) || s[i - 1][j] == s[i][j]) continue;
				if (s[i - 1][j] > s[i][j]) {
					st.insert(j);
					i = 0;
				}
				break;
			}
		}
        for(auto &e : st) cout<<e<<" ";
		return st.size();
	}
};
