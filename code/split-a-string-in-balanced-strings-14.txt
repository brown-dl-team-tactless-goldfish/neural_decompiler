

int balancedStringSplit(char * s){
    int count=0,count1=0;
    for(int i=0;s[i]!='\0';i++)
    {
        if(s[i]=='R')
        {
            count++;
            if(count==0)
            {
                count1++;
            }
        }
        else 
        {
            count--;
            if(count==0)
            {
                count1++;
            }
        }
    }
    return count1;
}
