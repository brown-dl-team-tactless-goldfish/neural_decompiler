int minDeletionSize(char ** strs, int strsSize){
    int n=strsSize;
    int m= strlen(strs[0]);
    int ans=0;
    for(int i=0;i<m;i++){
        for(int j=1;j<strsSize;j++){
            if(strs[j][i]<strs[j-1][i]){
                ans++;
                break;
            }
        }
    }
   return ans;
}
