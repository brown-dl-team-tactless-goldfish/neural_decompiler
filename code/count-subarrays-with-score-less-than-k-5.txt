class Solution {
public:
    inline int l(int i, int j) { return j - i + 1;}
    long long countSubarrays(vector<int>& nums, long long k) {
        int n = nums.size(), i = 0;
        long long sum = 0, ans = 0;
        for (int j = 0; j < n; j++) {
            sum += nums[j];
            while (sum * l(i, j) >= k) {
                sum -= nums[i];
                i++;
            }
            ans += l(i, j);
        }
        return ans;
    }
};
