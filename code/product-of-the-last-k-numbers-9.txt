class ProductOfNumbers {
    vector<int> a;
public:
    ProductOfNumbers() {
        
    }
    
    void add(int num) {
        a.push_back(num);
    }
    
    int getProduct(int k) {
        int ans=1;
        int res = accumulate(a.begin()+a.size()-k,a.end(),1,multiplies<int>()); 
        return res;
    }
};
