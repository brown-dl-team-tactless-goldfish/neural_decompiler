int ans;

int dfs(struct TreeNode *nd){
    if(nd==NULL)
        return 1;
    int l=dfs(nd->left),r=dfs(nd->right);
    if(l==1 && r==1)
        return -1;
    else if(l==-1 || r==-1){
        ans++;
        return 0;
    }
    else if(l==0 || r==0)
        return 1;
    return 1;
}


int minCameraCover(struct TreeNode* root){
    ans=0;
    if(dfs(root)==-1)
        ans++;
    return ans;
}
