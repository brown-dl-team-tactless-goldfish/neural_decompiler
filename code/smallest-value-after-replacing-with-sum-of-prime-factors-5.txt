class Solution {
public:
    int prime(int n)
    {
        int ans = 0;
        int c = 2;
        while(n > 1)
        {
            if(n%c==0)
            {
                ans+=c;
                n/=c;
            }
            else c++;
        }
        return ans;
    }
    int smallestValue(int n) {
        
        while(n != prime(n))
            n = prime(n);
        
        return n;
    }
};
