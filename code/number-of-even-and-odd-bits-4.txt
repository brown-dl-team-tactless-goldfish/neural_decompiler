vector<int> evenOddBit(int n) {
	int even = 0, odd = 0;
	bool bEven = true;
	while (n) {
		if (n & 1)
		   bEven == true ? even++ : odd++; 
		bEven = !bEven;
		n >>= 1;
	}
	return { even, odd };
}
