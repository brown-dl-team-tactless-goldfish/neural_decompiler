    public class FindElements
    {
        private ISet<int> _set;

        private void Helper(TreeNode node, int val, ISet<int> set)
        {
            if (node == null)
            {
                return;
            }

            _set.Add(val);
            Helper(node.left, 2 * val + 1, _set);
            Helper(node.right, 2 * val + 2, _set);
        }

        public FindElements(TreeNode root)
        {
            _set = new HashSet<int>();
            Helper(root, 0, _set);
        }

        public bool Find(int target) => _set.Contains(target);
    }
