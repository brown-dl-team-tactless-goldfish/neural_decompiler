class Solution {
public:
    int kItemsWithMaximumSum(int numOnes, int numZeros, int numNegOnes, int k) {
        priority_queue<int> pq;
        while(numOnes--) pq.push(1);
        while(numZeros--) pq.push(0);
        while(numNegOnes--) pq.push(-1);
        
        int ans = 0;
        while(k--) ans += pq.top(), pq.pop();
        return ans;
    }
};
