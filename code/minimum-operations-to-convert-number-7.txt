class Solution:
    def minimumOperations(self, nums: List[int], start: int, goal: int) -> int:
        operations_counts = []
        seen = set()
        self.recursive(nums, start, goal, operations_counts, 0, seen)
        print(operations_counts)
        return min(operations_counts) if len(operations_counts) > 0 else -1
    
    def recursive(self, nums, x, goal, operations_counts, currOp, seen):
        if x == goal:
            operations_counts.append(currOp)
            return 
        if x in seen: return
        if x >= 0 and x <= 1000:
            seen.add(x)
            for num in nums:
                self.recursive(nums, x - num, goal, operations_counts, currOp+1, seen)
                self.recursive(nums, x + num, goal, operations_counts, currOp+1, seen)   
                self.recursive(nums, x ^ num, goal, operations_counts, currOp+1, seen)
