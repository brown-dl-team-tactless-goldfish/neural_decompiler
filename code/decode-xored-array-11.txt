/**
 * Note: The returned array must be malloced, assume caller calls free().
 */
int* decode(int* encoded, int encodedSize, int first, int* returnSize){
    int *decodeArr = (int *)malloc(sizeof(int)*(encodedSize+1));
    int index = 0;
    decodeArr[index++] = first;
    for (int i = 0; i < encodedSize; i++)
    {
        /* code */
        decodeArr[index] = encoded[i] ^ first;
        first = decodeArr[index++];
    }
    *returnSize = index;
    return decodeArr;    
}
