/**
 * Note: The returned array must be malloced, assume caller calls free().
 */
int* shuffle(int* nums, int numsSize, int n, int* returnSize){
    
    assert(2*n == numsSize);
    
    *returnSize = numsSize;
    int *ans = malloc(*returnSize * sizeof(int));
    
    for (int i = 0; i < n; i++) {
        
        ans[i*2] = nums[i];
        ans[i*2+1] = nums[n+i];
    }
    return ans;
}
