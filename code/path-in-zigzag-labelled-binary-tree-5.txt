int* pathInZigZagTree(int label, int* returnSize){
    int row=0,power=1;
    while(power<=label){
        row++;
        power*=2;
    }
    int col = (row%2==1)? label-power/2 : power-label-1;
    int*result = (int*)calloc(row,sizeof(int));
    *returnSize = row;
    while(label!=1){
        result[row-1] = label;
        row--;
        col/=2;
        power/=2;
        label = (row%2==1)? power/2+col : power-col-1;
    }
    result[0]=1;
    return result;
}
