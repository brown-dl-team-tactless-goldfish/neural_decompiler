int passThePillow(int n, int time){
    int output = 1;
    for(int i=1;i<=time;i++)
    {
        if(i<n)
            output++;
        if(i>=n)
        {
            if(output<=1)
            {
                time=time-i+1;
                i=1;
                output++;
            }
            else
                output--;
        }
    }
    return output;
}
