/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    int distributeCoinsUtil(TreeNode* root, int &res) {
        if (!root) return 0;
         
        if (root->left)
            root->val += distributeCoinsUtil(root->left, res);
        if (root->right)
            root->val += distributeCoinsUtil(root->right, res);
    
        int temp = root->val-1;
        res += abs(temp);
        return temp;
    }
public:
    int distributeCoins(TreeNode* root) {
        
        if (!root) return 0;
        int res = 0;
        distributeCoinsUtil(root, res);
        return res;
    }
};
