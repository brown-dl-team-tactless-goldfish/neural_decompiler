int mod = 1e9 + 7;
class Solution {
public:
    int Size(int n){
        int c = 0;
        while(n > 0){
            n /= 2;
            c++;
        }
        return c;
    }
    int concatenatedBinary(int n) {
        unsigned long long int ans = 0;
        for(int i = 1; i <= n; i++){
            int j = Size(i);
            ans = (ans*(1 << j))%mod;
            ans = (ans+i)%mod;
        }
        return ans;
    }
};
