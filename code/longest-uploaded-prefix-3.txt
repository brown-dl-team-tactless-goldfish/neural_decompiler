class LUPrefix {
public:
    vector<int> data; 
    int s = 0; 
    LUPrefix(int n) {
        data.resize(n);
    }
    
    void upload(int i) {
        data[i-1] = 1; 
    }
    
    int longest() {
        while(s<data.size()&&data[s]) {
            s++;
        }
        return s; 
    }
};

 * Your LUPrefix object will be instantiated and called as such:
 * LUPrefix* obj = new LUPrefix(n);
 * obj->upload(video);
 * int param_2 = obj->longest();
 */
