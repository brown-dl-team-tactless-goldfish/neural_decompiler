    vector<string> getFolderNames(vector<string>& names) {
        // files keeps the number should be appended to the next file with the same name
        // if a new file after appending the number still has the same name of another file, then it means that
        // the number is outdated it needs to be updated, i.e. plus one
        // then let the new file append the new number and try again until no existing file with the same name as it
        unordered_map<string, int> files;
        vector<string> res;
        for (auto& s : names) {
            if (files.find(s) == files.end()) {
                files[s] = 1;
                res.push_back(s);
                continue;
            }
            string newStr = s + '(' + to_string(files[s]++) + ')';
            while (files.find(newStr) != files.end()) {
                newStr = s + '(' + to_string(files[s]++) + ')';
            }
            files[newStr] = 1;
            res.push_back(newStr);
        }
        return res;
    }
