#define MOD 1000000007

int numOfSubarrays(int* arr, int arrSize){
    int result = 0;
    int preSum = 0;
    int countOddSum = 0;
    int countEvenSum = 1;
    for (int i = 0; i < arrSize; i++){
        preSum += arr[i];
        if (preSum % 2 == 0){
            result = (result + countOddSum) % MOD;
            countEvenSum++;
        } else {
            result = (result + countEvenSum) % MOD;
            countOddSum++;
        }
    }
    return result;
}
