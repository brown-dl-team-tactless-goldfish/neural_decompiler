

/**
 * Note: The returned array must be malloced, assume caller calls free().
 */
void findWidthAndLength(int* ans, int area, int length, int width){
    if(area%length==0 && length>=(area/length)){
        ans[0]=length;
        ans[1]=area/length;
        return;
    }

    findWidthAndLength(ans, area, length+1, width);
}

int* constructRectangle(int area, int* returnSize){
    *returnSize=2;
    int* ans=malloc(sizeof(int)*(*returnSize));
    int sqrtArea=sqrt(area);
    findWidthAndLength(ans, area, sqrtArea, 0);
    return ans;
}
