public class Solution 
{
    public long FindTheArrayConcVal(int[] nums) 
    {
        long result = 0;
        int n = nums.Length;
        for (int i = 0; i < n / 2; i++)
        {
            result += int.Parse(nums[i].ToString() + nums[n - 1 - i].ToString());
        }
        if (n % 2 == 1)
            result += nums[n / 2];
        return result;
    }
}
