int partitionString(char* s) {
    int n = strlen(s), last[26], start = 0, count = 0;
    memset(last, -1, sizeof(last));
    for (int i = 0; i < n; i++) 
    {
        int c = s[i] - 'a';
        if (last[c] >= start) 
        {
            start = i;
            count++;
        }
        last[c] = i;
    }
    return count + 1;
}
