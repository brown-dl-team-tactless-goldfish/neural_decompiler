int countPrefixes(char ** words, int wordsSize, char * s) {
	
    register unsigned short i = 0, j = 0, s_len = strlen(s), ct = 0, sum = 0;
	while(i < wordsSize) {
		while(words[i][j] != '\0' && j <= s_len) {
			if(words[i][j] == s[j]) {
				ct++;
			}
			j++;
		}
		if(ct == strlen(words[i]))
			sum++;
		ct = 0;
		j = 0;
		i++;
	}
	return sum;
}
