int minSteps(char * s, char * t){
    int fs[26]={0},ft[26]={0};
    int i=0,num,cnt=0;
   
    while(s[i]!='\0'){
        num=s[i]-'a';
        
        fs[num]++;
        
        i++;
    }
   
    i=0;
    while(t[i]!='\0'){
        num=t[i]-'a';
        
        ft[num]++;
        i++;
    }
   
    
    for(int i=0;i<26;i++){
        
        if(fs[i]!=ft[i]){
            if(ft[i]>fs[i])
            {
                cnt+=ft[i]-fs[i];
            }
        }
    }
return cnt;
}
