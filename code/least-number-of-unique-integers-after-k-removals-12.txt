class Solution {
public:
    int findLeastNumOfUniqueInts(vector<int>& arr, int k) {
        unordered_map<int, int>Map ;
        for(auto& x : arr){
            Map[x]++ ;
        }
        vector<int>itemCount ;
        
        for(auto& m : Map)
            itemCount.push_back(m.second) ;
            
        sort(itemCount.begin(), itemCount.end()) ;
        int remove = 0 ;
        for(int i = 0; i < itemCount.size(); i++){
            if(k < itemCount[i])
                break ;
            k -= itemCount[i] ;
            remove++ ;
        }
        return itemCount.size() - remove ;
    }
};
