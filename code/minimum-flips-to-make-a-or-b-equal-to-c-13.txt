int minFlips(int a, int b, int c){
   int andRet = a&b;
   int orRet = a|b;
   if(orRet == c)
       return 0; 
   int ret = 0; 
   int orRetXorC = orRet ^ c;
   // count for steps to flip for a|b
   while(orRetXorC)
   {
       orRetXorC = orRetXorC&(orRetXorC-1);
       ret++;
   }
    // a&b returns duplicated bits & with ~c to remove bits set in c
    andRet &= ~c;
    // count for steps to flip duplicated bits. 
   while(andRet)
   {
       andRet = andRet&(andRet-1);
       ret++;
   }
   return ret;
}
