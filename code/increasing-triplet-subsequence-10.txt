class Solution {
public:
    bool increasingTriplet(vector<int>& nums) {
        double a = INFINITY;
        double b = INFINITY;
        double c = INFINITY;
        for(int i = 0;i<nums.size();i++){
            if(a>=nums[i]) a = nums[i];
            else if(b>=nums[i]) b = nums[i];
            else c = nums[i];
            if(a != INFINITY && b != INFINITY && c != INFINITY) return true;
        }
        return false;
    }
};
