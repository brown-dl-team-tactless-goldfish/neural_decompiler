char * reverseOnlyLetters(char * S){
    char * ptr1 = S;
    char * ptr2 = S + (strlen(S) - 1);
    char tempChar;
    
    while(ptr1 < ptr2) {
       if(!isalpha(*ptr1)) {
            ++ptr1;
           continue;
        }
        if(!isalpha(*ptr2)) {
            --ptr2;
            continue;
        }
        
        tempChar = *ptr1;
        *ptr1 = *ptr2;
        *ptr2 = tempChar;
        ++ptr1;
        --ptr2;
    }
    return S;
}
