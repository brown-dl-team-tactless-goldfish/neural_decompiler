int shortestSequence(vector<int>& rolls, int k) {
    unordered_set<int> curLevelNumbers; // roll values of current level
    int fullLevelsCount = 0;
    for (int roll : rolls) {
        curLevelNumbers.insert(roll);
        if (curLevelNumbers.size() == k) {
            // If current level is fully filled with `k` numbers {1, ..., k} - start a new level, increment the `fullLevelsCount`:
            ++fullLevelsCount;
            curLevelNumbers.clear();
        }
    }
    return fullLevelsCount + 1; // the shortest sequence is longer by `1` than the `full levels count`.
}
