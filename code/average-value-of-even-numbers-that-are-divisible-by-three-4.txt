int averageValue(int* nums, int numsSize){
    int sum = 0, cnt = 0;
    for (int i = 0; i < numsSize; i++)
    {
        /* code */
        if (nums[i] % 6 == 0)
        {
            /* code */
            cnt++;
            sum += nums[i];
        }
        
    }

}
