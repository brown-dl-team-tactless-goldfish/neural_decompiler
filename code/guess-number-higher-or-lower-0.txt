int guessNumber(int n) {
    int start = 1, end = n, num = start + (end - start) / 2;
    int ans = guess(num);
	while(ans != 0) {
        if(ans == -1) {
            end = num;
        } else if(ans == 1) {
            start = num + 1;
        }
        num = start + (end - start) / 2;
        ans = guess(num);
    }
    return num;
}
