bool* prefixesDivBy5(int* nums, int numsSize, int* returnSize){
    bool* ans = malloc(numsSize * sizeof(int));
    int  sum = 0;
    for(int i = 0; i < numsSize; i++){
        sum = sum*2 + nums[i];
        if(sum % 5 == 0)
            ans[i] = true;
        else
            ans[i] = false;
        sum %= 100;
    }
    *returnSize = numsSize;
    return ans;
}
