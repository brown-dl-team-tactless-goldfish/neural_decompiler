class Solution {
public:
    int chalkReplacer(vector<int>& chalk, int k) {
        long int sum=accumulate(chalk.begin() , chalk.end() , 0l);
		//we subtract the multiple of sum of chalks from the total chalks.
        k%=sum;
		//Now we distribute the remaining chalks from student 0 untill chalks are sufficient.
        for(int i=0;i<chalk.size();i++){
            k-=chalk[i];
            if(k<0) return i;
        }
        return -1;
    }
};
