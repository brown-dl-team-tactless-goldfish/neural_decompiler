  bool checkSubarraySum(vector<int>& nums, int k) {
      map<int, bool> modMap;
      
      int currSum = 0;
      for (int i = 0; i < nums.size(); i++) {
          currSum += nums[i];
          auto iter = modMap.find(currSum % k);
          if (iter != modMap.end()) {
              return true;
          }
      
          int rightSideOfEquation = -1 * ((nums[i] - currSum) % k);
          modMap[rightSideOfEquation] = true;
      }
      
      return false;
  }
