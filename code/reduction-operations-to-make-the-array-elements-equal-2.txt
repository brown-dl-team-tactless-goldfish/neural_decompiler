class Solution {
public:
    int reductionOperations(vector<int>& a) {
        int n = a.size();
        map<int,int>mp;
        
        for(int i = 0 ; i < n ;i++)
        {
            mp[a[i]]++;
        }
           //map<int,int>:: reverse_iterator it ;
        int c = 0,ans=0;
        for(auto it = mp.rbegin();it!=mp.rend();it++)
        {
            ans += it->second ;
            ans+=c;
            c+=it->second;
            
            
        }
        return ans-c;
    }
};
