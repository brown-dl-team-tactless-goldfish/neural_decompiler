char * greatestLetter(char * s){

    int* p = malloc(26*sizeof(int)); memset(p,0,26*4);
    int* q = malloc(26*sizeof(int)); memset(q,0,26*4);
	
    for(char* t = s; *t != '\0'; t++)
    {
        if( 'a' <= *t && *t <= 'z')
            p[*t - 'a']++;
        else
            q[*t-'A']++;
    }
    
    for(int i = 25; i >= 0; i--)
    {
        if(p[i] > 0 && q[i] >0)
        { char* c = malloc(2*sizeof(char)); *c = (char)('A'+i); *(c+1) = '\0'; return c; }
    }
    return "";
    
}
