int wateringPlants(vector<int>& plants, int capacity) {
        int res=0;
        int c = capacity;  //maximum capacity of can
        for(int i=0; i<plants.size(); i++)
        {
            if(plants[i]<=c)
            {
                c -= plants[i];
                res++;
            }    
            else
            {
                res += (2*i+1);
                c=capacity-plants[i];
            }
        }
        
        return res;
        
    }
