int numPairsDivisibleBy60(int* time, int timeSize){
    int i = 0, j = 0, count = 0;
    int bucket[60] = {0};
    
    for (i = 0; i < timeSize; i++) {
        bucket[time[i] % 60]++;
    }
    
    count += (bucket[0] * (bucket[0]-1))/2;
    
    for (i = 1; i < 30; i++) {
        count += bucket[i] * bucket[60 - i];
    }
    
    count += (bucket[30] * (bucket[30]-1))/2;
    
    return count;
}
