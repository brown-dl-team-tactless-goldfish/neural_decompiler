class Solution:
    def getMaximumGenerated(self, n: int) -> int:
        #Base Case 
        if n == 0:
            return 0
        #initialize nums and a counter i
        nums, i = [0, 1], 1
        #generate the array
        while 2 * i <= n:
            nums.insert(2 * i, nums[i])
            if 2 * i + 1 <= n:
                nums.insert(2 * i + 1, nums[i] + nums[i + 1])
            i += 1
        #return the max of nums
        return max(nums)
