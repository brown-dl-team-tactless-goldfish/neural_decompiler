class Solution {
public:
    int countBalls(int lowLimit, int highLimit) {
        unordered_map<int, int> cnt(46);
        
        for (int i = lowLimit; i <= highLimit; ++i) {
            int add = 0;
            for (auto c : to_string(i)) add += c - '0';
            if (!cnt.count(add)) cnt[add] = 1;
            else cnt[add] ++;
        }
        
        // find max
        int mx = INT_MIN;
        for (int i = 1; i < 46; ++ i) mx = max(mx, cnt[i]);
        return mx;
        
    }
};
