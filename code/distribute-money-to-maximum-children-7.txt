class Solution {
public:
    int distMoney(int money, int children) {
        if ((money -= children) < 0) return -1;
        int ans = 0;
        while (money >= 7 && children > 0)
            ++ans, money -= 7, --children;
        if (money == 3 && children == 1 || children == 0 && money > 0) --ans;
        return ans;
    }
};
