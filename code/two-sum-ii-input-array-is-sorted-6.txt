int* twoSum(int* numbers, int numbersSize, int target, int* returnSize){
    int i = 0, j = numbersSize - 1;
    for (*returnSize = 2 ; i < j && numbers[i] + numbers[j] != target ;)
        j -= numbers[i] + numbers[j] > target, i += numbers[i] + numbers[j] < target;    
    return memcpy(malloc(sizeof(int [2])), (int []) { i + 1, j + 1 }, sizeof(int [2]));
}
