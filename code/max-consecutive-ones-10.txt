int findMaxConsecutiveOnes(int* nums, int numsSize) {
    int ans = 0;
    int cnt = 0;
    for (int i = 0; i < numsSize; ++i) {
        if (nums[i] == 1) {
            cnt = cnt + 1;
            ans = ans > cnt ? ans : cnt;
        }
        else if (i > numsSize - ans - 2) {  // Early termination
            return ans;
        }
        else {
            cnt = 0;
        }
    }

    return ans;    
}```
