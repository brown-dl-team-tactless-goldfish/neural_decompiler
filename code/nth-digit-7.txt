    inline int getDigit(long num, int cnt_digits, int index)
    {
        int loop=cnt_digits-index-1;
        while(loop--) num/=10;
        return num%10;
    }
    int findNthDigit(long n) {
        if(n<10) return n;
        n-=10;
        for(long i=2, step=10; i<10; i++, step*=10)
        {
            if(n<9*step*i)
            {
                int val=step+n/i;
                return getDigit(val, i, n%i);
            }
            n-=(9*step*i);
        }
        return -1;
    }
