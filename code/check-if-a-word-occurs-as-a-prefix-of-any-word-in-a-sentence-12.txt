int isPrefixOfWord(char * sentence, char * searchWord)
{
    int cnt = 0;
    char s[2] = " ";
    char *token;
    
    token = strtok(sentence, s);
    
    while (token != NULL) {
        cnt++;
        if (strstr(token, searchWord) && (token[0] == searchWord[0]))
            return cnt;
        token = strtok(NULL, s);
    }
    
    return -1;
}
