char v[52]={'a','a','a','a','a','a','a','a','a','a','a','a','a','a','a','a','a','a','a','a','a','a','a','a','a','a','a','a','a','a','a','a','a','a','a','a','a','a','a','a','a','a','a','a','a','a','a','a','a','a'};
char arr[5] = {'a','e','i','o','u'};
int countVowelStrings(int n){
    if(n==0)
        return 1;
    int count = 0;
    for(int i=0;i<5;i++){
        v[n] = arr[i];
        if(v[n]>=v[n+1])
            count += countVowelStrings(n-1);
    }

    return count;

}
