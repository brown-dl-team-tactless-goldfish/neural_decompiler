class Solution {
public:
    int minimumRefill(vector<int>& plants, int capacityA, int capacityB) {
        auto i = plants.begin(), j = plants.end() - 1;
        int res = 0, ac = capacityA, bc = capacityB;
        while(i < j){
            if(ac < *i) ac = capacityA, res++;
            if(bc < *j) bc = capacityB, res++;
            ac -= *i, bc -= *j;
            i++, j--;
        }
        if(i == j && *i > ac && *j > bc) res++;
        return res;
    }
};
