	public bool IsGoodArray(int[] nums) {
        int d = nums[0];
        for(int i = 1; i < nums.Length; i++) {
            if(d == 1) break;
            int r = nums[i];
            while((r %= d) > 0) {
                int t = d;
                d = r;
                r = t;
            }
        }
        return d == 1;
    }
