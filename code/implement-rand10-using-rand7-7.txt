class Solution {
public:
    int rand10() {
        
        while (true)
        {
            int rand49 = (rand7() - 1) * 7 + rand7(); // uniform 1 ~ 49
            
            // random rejecion -> uniform 1 ~ 40 -> uniform 1 ~ 10
            if (rand49 <= 40) return rand49 % 10 /*uniform 0 ~ 9*/ + 1;
        }
        
    }
};
