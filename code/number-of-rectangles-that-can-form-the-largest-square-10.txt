class Solution:
    def countGoodRectangles(self, rectangles: List[List[int]]) -> int:
        n = len(rectangles)
        data = []
        for i in range(n):
            data.append(min(rectangles[i]))
            
        print ("data = ", data)
        
        max_len = max(data)
        
        self.res = data.count(max_len)
        
        return self.res
