class Solution {
public:
	int minimumPartition(string s, int k) {
		for( auto i : s ){ 
			if( i - '0' > k ) return -1;    
		}
		int i = 0, c = 0;
		long long tmp = 0;
		while( i < s.size() ){
			tmp = tmp * 10 + s[i]-'0';
			if( tmp > k ){
				tmp = tmp % 10; 
				c++;
			}
			i++;
		}
		c++;
		return c;
	}
};
