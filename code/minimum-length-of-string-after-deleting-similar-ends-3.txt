    int minimumLength(string s) {
        deque<char> v(s.begin(), s.end());
        while(v.size()>1 && v.front() == v.back()) {
            char c = v.front();
            while(!v.empty() && v.front() == c) v.pop_front();
            while(!v.empty() && v.back() == c) v.pop_back();
        }
        return v.size();
    }
