vector<vector<int>> mergeSimilarItems(vector<vector<int>>& items1, vector<vector<int>>& items2) 
{
	vector<vector<int>> result;
	map<int,int> total;
	int m=items1.size(),n=items2.size();
	for(int i=0;i<m;i++) total[items1[i][0]]+=items1[i][1];
	for(int i=0;i<n;i++) total[items2[i][0]]+=items2[i][1];
	for(map<int,int>::iterator it=total.begin();it!=total.end();it++) result.push_back({it->first,it->second});

	return result;
}
