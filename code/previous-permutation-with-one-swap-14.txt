 vector<int> prevPermOpt1(vector<int>& a) {
        int n=a.size();
        int i=n-2;
        while(i>=0 && a[i]<=a[i+1]) i--;
        if(i==-1) return a;
        int idx=-1;
        for(int j=n-1;j>i;j--){
            if(a[j]<a[i] && (idx==-1 || a[j]>=a[idx])) idx=j;
        }
        swap(a[i],a[idx]);
   //     sort(a.begin()+i+1,a.end(),greater<int>());
        
        return a;
    }
