    public class Solution
    {
        public string LastSubstring(string s)
        {
            if (s.Length == 1)
            {
                return s;
            }

            int resIdx = 0;
            int idx = 1;

            while (true)
            {
                if (idx >= s.Length)
                {
                    break;
                }

                char currChar = s[idx];

                if (currChar < s[resIdx])
                {
                    idx++;
                    continue;
                }

                if (currChar > s[resIdx])
                {
                    resIdx = idx;
                    idx++;
                    continue;
                }


                int ptrToRes = resIdx;
                int ptrToNewRes = idx;

                while (true)
                {
                    if (ptrToNewRes >= s.Length || ptrToRes >= s.Length)
                    {
                        idx = ptrToNewRes;
                        break;
                    }

                    if (ptrToRes >= idx)
                    {
                        idx = ptrToNewRes;
                        break;
                    }

                    if (s[ptrToNewRes] == s[ptrToRes])
                    {
                        ptrToNewRes++;
                        ptrToRes++;
                        continue;
                    }

                    if (s[ptrToNewRes] > s[ptrToRes])
                    {
                        resIdx = idx;
                    }

                    idx = ptrToNewRes;
                    break;
                }
            }

            return s.Substring(resIdx);
        }
    }
