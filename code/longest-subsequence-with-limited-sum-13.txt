    vector<int> answerQueries(vector<int>& nums, vector<int>& queries) {
        vector<int> v = {0}, res;
        sort(begin(nums), end(nums));
        for(auto x:nums) v.push_back(v.back() + x);
        for(auto q:queries) res.push_back(upper_bound(begin(v), end(v), q) - begin(v) - 1);
        return res;
    }
