bool isvowel(char c){
        return (c=='a' or c =='e' or c=='i' or c=='o' or c=='u');
    }
    long long countVowels(string word) {
        long long int ans=0;
        for(int i=0;i<word.size();i++){
            if(isvowel(word[i]))
                ans += (i+1)*(word.size()-i);
            else continue;
        }
        return ans;
    }
