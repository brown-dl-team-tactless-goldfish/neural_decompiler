int totalMoney(int n) {
    int base_mon = 0;
    int temp_sum = 0;
    int sum = 0;
    
    for(int i=0;i<n;i++) {
        if(i%7 == 0) {
            // printf("============\n");
            base_mon++;
            temp_sum = base_mon;
        }
        // printf("%d\n", temp_sum);
        sum += temp_sum;
        temp_sum ++;
    }

    return sum;
}
