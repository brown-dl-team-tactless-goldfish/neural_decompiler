

int largestAltitude(int* gain, int gainSize){
    int max = 0, n = gainSize + 1;
    int netgain[n];
    netgain[0] = 0;
    for(int i = 0; i < gainSize; i++){
        netgain[i + 1] = netgain[i] + gain[i]; 
    }
    for(int j = 0; j < n; j++){
        if(netgain[j] >= max) max = netgain[j];
    }
        return max;
}
