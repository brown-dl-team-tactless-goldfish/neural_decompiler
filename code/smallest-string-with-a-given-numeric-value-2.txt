char * getSmallestString(int n, int k){
    char* ans = malloc( (n + 1) * sizeof(char));
    ans[n] = '\0';
    k = k - n;
    int zCn = k/25;
    int left = k%25;
    int aCn = n - zCn - 1;

    for(int i = 0; i < aCn; i++){
        ans[i] = 'a';
    }
    //special case: all character are 'z'
    if(aCn >=0)
        ans[aCn] = 'a' + left;
    
    for(int i = aCn+1; i < n; i++){
        ans[i] = 'z';
    }
    
    return ans;

}
