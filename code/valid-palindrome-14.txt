char ignorecase(char a){

	if(a >= 'a' && a <= 'z')
		a=(char)(a-32);

	return a;
}

bool isPalindrome(char * s){
	int s_len=strlen(s)-1;
	int i=0;

	if(strlen(s)==0)
		return true;

	while(i<s_len){

		if(s[i]<'0' || (s[i]>'9' && s[i]<'A') || \
		   (s[i]>'Z' && s[i]<'a') || s[i]>'z'){
			   i++;
			   continue;
		   }

		if(s[s_len]<'0' || (s[s_len]>'9' && s[s_len]<'A') || \
		   (s[s_len]>'Z' && s[s_len]<'a') || s[s_len]>'z'){
			   s_len--;
			   continue;
		   }

		if(ignorecase(s[i])!=ignorecase(s[s_len]))
			return false;

		 i++;
		 s_len--;
	}

	return true;
}
