char * removeStars(char * s){
    int n = strlen(s);
    char* stack = malloc((n+1) * sizeof(char));
    int idx = 0;
    for(int i = 0; i < n; i++){        
        stack[idx] = s[i];
        if(stack[idx] == '*'){
            if(idx == 0)
                continue;
            else
                idx -=2;
        }
        idx++;
    }
    stack[idx] = '\0';
    return stack;
}
