int maximumCount(int* nums, int numsSize){
    int pos_count =0,neg_count = 0;
    for(int i=0;i<numsSize;i++)
    {
        if (nums[i]>0)
        {
            pos_count+=1;
        }
        else if (nums[i]<0)
        {
            neg_count +=1;
        }
    }
    if (pos_count>neg_count)
    {
        return pos_count;
    }
    else
    {
        return neg_count;
    }



}
