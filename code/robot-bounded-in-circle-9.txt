#pragma GCC optimize ("O3")


inline __attribute__((always_inline)) bool isRobotBounded(char *ins) {
    bool c_g = false;
    int l = 0;
    int r = 0;
    
    for(; *ins; ins++) {
        if(*ins == 'G')
            c_g = true;
        else if(*ins == 'L')
            l++;
        else
            r++;
    }
    
    if(!c_g)
        return true;
    
    if(l == 0 && r == 0)
        return false;
    
    if(l == 0 || r == 0)
        return true;
    
    return l % 4 != r % 4;        
}
