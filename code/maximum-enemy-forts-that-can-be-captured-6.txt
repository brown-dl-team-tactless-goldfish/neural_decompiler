class Solution {
public:
    int captureForts(vector<int>& f) {
        int ans=0;
        for(int i=0; i<f.size(); i++){
            int cnt=0;
            for(int j=i+1; j<f.size(); j++){
                if(f[i]==1 && f[j]==-1){
                    ans=max(ans, cnt);
                }
                if(f[i]==-1 && f[j]==1){
                    ans=max(ans, cnt);
                }
                if(f[j]!=0){break;}
                cnt++;
            }
        }
        return ans;
    }
};
