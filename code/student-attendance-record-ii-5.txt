int mod=1000000007;
class Solution {
public:
    int dp[100001][2][3]={0};
    int get(int n,int a,int l)
    {
        if(n==0)
            return 1;
        if(dp[n][a][l]!=0)
            return dp[n][a][l];
        int res=0;
        if(a==0)
            res=(res+get(n-1,a+1,0))%mod;
        if(l<2)
            res=(res+get(n-1,a,l+1))%mod;
        res=(res+get(n-1,a,0))%mod;
        return dp[n][a][l]=res;
    }
    int checkRecord(int n) 
    {
        return get(n,0,0);
    }
};
