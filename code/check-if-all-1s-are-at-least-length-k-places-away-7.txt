
bool kLengthApart(int* nums, int numsSize, int k){
    
    int i,j;
    
    for(i=0;i<numsSize;i++)
    {
        if(nums[i]==1)
        {
            for(j=1;i+j<=numsSize-1&&j<=k;j++)
            {
                if(nums[i+j]==1)
                    return 0;
            }
            i=i+j-1;
        }
    }
    
    
    return 1;

}
