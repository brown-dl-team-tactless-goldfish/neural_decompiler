class Solution {
public:
    long long countSubarrays(vector<int>& nums, long long k) {
        long long ans = 0, sum = 0;
        for(int i=0,j=0;j<nums.size();j++){
            sum += nums[j];
            int len = j-i+1;
            if(sum * len < k) ans += len;
            else{
                while(sum * (j-i+1) >= k and i <= j){
                    sum -= nums[i++];
                }
                ans += j - i + 1;
            }
        }
        return ans;
    }
};
