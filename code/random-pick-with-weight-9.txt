vector<int> v;

Solution(vector<int>& w) {
    
    int s = 0;
    
    for(auto x : w)
        
        
    {
        v.push_back(s+=x);
    }
    
}

int pickIndex() {
    
   int random = rand() % v.back();
    
    int x = upper_bound(v.begin(),v.end(),random) - v.begin();
    
    return x;
    
}
