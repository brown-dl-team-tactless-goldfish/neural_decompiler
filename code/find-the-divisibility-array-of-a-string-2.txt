class Solution {
public:
    // each step can be calculated by (previous * 10 + next) % m
    vector<int> divisibilityArray(string word, int m) {
        vector<int> ans;
        long long cur = 0;
        for (auto& c : word){
            cur *= 10;
            cur += c - '0';
            cur %= m;
            ans.push_back(cur == 0);
        }
        return ans;
    }
};
