class Solution {
public:
    string greatestLetter(string s) {
        vector<int> lower(26, 0), upper(26, 0);
        for(auto c : s){
            if(islower(c)) lower[c - 'a']++;
            if(isupper(c)) upper[c - 'A']++;
        }
        for(int i = 25; i >= 0; i--) if(lower[i] && upper[i]){
            string ret;
            ret += ('A' + i);
            return ret;
        }
        return "";
    }
};
