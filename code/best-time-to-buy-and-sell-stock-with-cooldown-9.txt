int maxProfit(int* prices, int pricesSize){
    int buy = INT_MIN, sell = 0, preSell = 0;
    for(int i=0; i<pricesSize; i++)
    {
        buy = buy > preSell - prices[i] ? buy : preSell - prices[i];
        preSell = sell;
        sell = sell > buy + prices[i] ? sell : buy + prices[i];
    }
    return sell;
}
