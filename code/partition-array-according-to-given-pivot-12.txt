class Solution {
public:
    vector<int> pivotArray(vector<int>& nums, int pivot) {
        vector<int>l, r;
        int p = 0;
        
        for(int i : nums){
            if(i < pivot) l.push_back(i);
            else if(i == pivot) p++;
            else r.push_back(i);
        }
        
        while(p--) l.push_back(pivot);
        
        for(int i : r) l.push_back(i);
        
        return l;
    }
};
