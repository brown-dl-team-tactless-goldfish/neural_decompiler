int pivotInteger(int n) {
    if (n == 1)
        return 1;
    
    int suml, sumr;
    int i, j;
    for (suml = 0, i = 1; i <= n; i++) {
        suml += i;
    }
    for (sumr = 0, j = n; j >= 1; j--) {
        suml -= j;
        if (suml == sumr)
            return j;
        sumr += j;
    }
    return -1;
}
