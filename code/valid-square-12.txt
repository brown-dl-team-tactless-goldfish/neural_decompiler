class Solution:
    def validSquare(self, p1: List[int], p2: List[int], p3: List[int], p4: List[int]) -> bool:
        ps, le = [p1, p2, p3, p4], []
        for i in range(3):
            for j in range(i+1,4):
                le.append((ps[i][0]-ps[j][0])**2 + (ps[i][1]-ps[j][1])**2)
        c = list(sorted([(a,b ) for a,b in Counter(le).items()]))
        return len(c)==2 and c[0][1] == 4 and c[1][1] == 2 and c[1][0] == 2 * c[0][0]
