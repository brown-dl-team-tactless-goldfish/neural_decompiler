class Solution {
public:
    vector<int> missingRolls(vector<int>& rolls, int mean, int n) {
        int sum = 0;
        for(int i : rolls) sum += i;
        
        int l = (rolls.size() + n) * mean - sum;
        if(l < n || l > n * 6) return {};
        
        int avg = l / n, rem = l % n;
        vector<int> ans(n, avg);
        
        for(int i=0; i<rem; i++) ans[i]++;
        return ans;
    }
};
