class Solution {
private: 
    int ret = 0 ;
    int helper(TreeNode* root){
        if(root == nullptr)
            return 0 ;
        int left = helper(root->left) ;
        ret += abs(left) ;
        int right = helper(root->right) ;
        ret += abs(right) ;
        
        return left + right + root->val - 1 ;
    }
public:
    int distributeCoins(TreeNode* root) {
        helper(root) ;
        return ret ;
    }
};
