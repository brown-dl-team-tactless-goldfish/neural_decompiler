class Solution {
public:
    bool containOnlyDigit(string& s) {
        for (auto c : s) {
            if (!isdigit(c)) {
                return false;
            }
        }
        return true;
    }
    int maximumValue(vector<string>& strs) {
        int ans = 0;
        for (auto str : strs) {
            if (containOnlyDigit(str)) {
                ans = max(ans, stoi(str));
            } else {
                ans = max(ans, (int)str.size());
            }
        }
        return ans;
    }
};
