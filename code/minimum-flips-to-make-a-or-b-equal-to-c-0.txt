

int minFlips(int a, int b, int c){
    int* digit = calloc(31, sizeof(int)) ;
    int idx = 0 ;
    while(a){
        digit[idx]+= (a&1) ;
        a >>= 1 ;
        idx++ ;
    }
    idx = 0 ; 
    while(b){
        digit[idx]+= (b&1) ;
        b >>= 1 ;
        idx++ ;
    }
        
    int ret = 0; 
    for(int i = 0; i < 31; i++){
        if((c>> i) & 1)
            ret += (digit[i] == 0) ;       
        else
            ret += digit[i] ;
    }     
    return ret ;    
}
