int res = 0;

int rec(struct TreeNode* root) {
    if (!root)
        return 0;
    int left = rec(root->left);
    int right = rec(root->right);
    int tilt = abs(left - right);
    res += tilt;
    return left + right + root->val;
}

int findTilt(struct TreeNode* root){
    res = 0;
    rec(root);
    return res;
}
