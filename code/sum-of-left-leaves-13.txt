int sumleft(struct TreeNode* root,int *lef,int f)
{  
    if(!root)
        return 0;
    if(!root->left && !root->right && f==1)
    *lef+=root->val;
        // if(root->left==NULL)
   return sumleft(root->left,lef,1)+sumleft(root->right,lef,0);
}


int sumOfLeftLeaves(struct TreeNode* root){
  
    int lef=0;
    sumleft(root,&lef,0);
    return lef;
}
