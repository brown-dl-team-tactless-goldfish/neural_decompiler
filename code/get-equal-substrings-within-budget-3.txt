class Solution(object):
    def equalSubstring(self, s, t, maxCost):
        ans, i, c = 0, 0, maxCost

        for j in range(len(s)):
            c -= abs(ord(s[j]) - ord(t[j]))
            if c >= 0: ans = max(ans, j - i + 1)
            else:
                c +=  abs(ord(s[i]) - ord(t[i]))
                i += 1
        
        return ans
