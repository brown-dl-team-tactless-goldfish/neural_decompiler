bool checkRecord(char* s) {
    int i = 0, anum = 0, lnum = 0, size = strlen(s);
    for( ; i < size; i++)
    {
        switch(*(s + i))
        {
            case 'A': anum++; lnum = 0; break;
            case 'L': lnum++; break;
            case 'P': lnum = 0; break;
        }
        if(anum == 2 || lnum == 3)
        {
            return false;
        }
    }
    return true;
}
