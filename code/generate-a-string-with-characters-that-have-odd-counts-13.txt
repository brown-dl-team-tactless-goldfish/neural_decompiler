class Solution:
    def generateTheString(self, n: int) -> str:
        if (n == 0):
            return ""
        
        res = ""
        count = 0
        if (n % 2 == 0):
            count = n - 1
        else:
            count = n
            
        for i in range(count):
            res += "a"
            
        if (count < n):
            res += "b"
            
        return res
