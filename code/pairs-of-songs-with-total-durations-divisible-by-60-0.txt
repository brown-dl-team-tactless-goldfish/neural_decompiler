int numPairsDivisibleBy60(int* time, int timeSize){
    int n = timeSize;
    int* tmp = malloc(n * sizeof(int));
    for(int i = 0; i < n; i++){
        tmp[i] = time[i] % 60 ;
    }
    int* hash = calloc(60, sizeof(int));
    int ans = 0 ;
    for(int i = 0; i < n; i++){
        int val = (60 - tmp[i]) % 60;
        ans += hash[val];
        hash[tmp[i]]++;
    }
    free(tmp);
    free(hash);
    return ans;
}
