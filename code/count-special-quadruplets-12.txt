    def countQuadruplets(self, nums: List[int]) -> int:
        res=0
        a_plus_b = collections.defaultdict(int)
        for c in range(len(nums)-1):
            for d in range(c+1,len(nums)):
                res+=a_plus_b[nums[d]-nums[c]]
            b = c
            for a in range(b):
                a_plus_b[nums[a]+nums[b]]+=1
        return res
