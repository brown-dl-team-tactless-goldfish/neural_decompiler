public int CountTime(string time)
{
	if (time[4] != '?' && time[3] != '?' && time[1] != '?' && time[0] != '?') return 1; 
	var arr = new int[4]
	{
	1,1,1,1
	};

	if (time[4] == '?') arr[3] = 10; 

	if (time[3] == '?') arr[2] = 6; 
	if (time[1] == '?')
	{
		if (time[0] == '2') arr[1] = 5; 
		else  arr[1] = 10; 
	}
	if (time[0] == '?')
	{
		if (time[1] == '?')
		{
			arr[1] = 4;
			arr[0] = 6;
		}
		else if (time[1] < '4') arr[0] = 3; 
		else if (time[1] == '4' && time[3] == '0' && time[4] == '0') arr[0] = 3; 
		else  arr[0] = 2 ;
	}
	if (time[0] == '2')
	{
		if (time[1] == '?')
		{
			arr[1] = 1;
			arr[0] = 4;
		}
		else arr[0] = 1; 
	}
	return arr[0] * arr[1] * arr[2] * arr[3];
}
