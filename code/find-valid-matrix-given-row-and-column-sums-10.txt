class Solution {
public:
    vector<vector<int>> restoreMatrix(vector<int>& rsum, vector<int>& csum) 
    {
        int n = rsum.size();
        int m = csum.size();
        
        vector<vector<int>> ans(n,vector<int>(m, - 1));
        
       for(int i = n - 1; i >= 0; i--)
       {
           int r = i, c = 0;
           while(r < n && c < m)
           {
               int mn = min(rsum[r],csum[c]);
               ans[r][c] = mn;
               rsum[r] -= mn;
               csum[c] -= mn;
               r++; c++;
           }
       }
        
        for(int j = 1; j < m; j++)
        {
            int r = 0,c = j;
            while(r < n && c < m)
            {
                int mn = min(rsum[r],csum[c]);
                ans[r][c] = mn;
                rsum[r] -= mn;
                csum[c] -= mn;
                r++; c++;
            }
            
        }
        
     return ans;    
        
    }
};
