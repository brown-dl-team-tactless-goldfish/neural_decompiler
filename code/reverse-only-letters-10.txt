char * reverseOnlyLetters(char * s){
    int i = 0;
    int j = strlen(s);
    char temp;
    while (i < j)
    {
        if(!((s[i] >= 65 && s[i] <= 90) || (s[i] >= 97 && s[i] <= 122)))
        {
            i++;
            continue;
        }
        if(!((s[j] >= 65 && s[j] <= 90) || (s[j] >= 97 && s[j] <= 122)))
        {
            j--;
            continue;
        }
        temp = s[j];
        s[j] = s[i];
        s[i] = temp;
        i++, j--;
    }
    return s;
}
