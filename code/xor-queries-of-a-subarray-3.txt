int* xorQueries(int* arr, int arrSize, int** queries, int queriesSize, int* queriesColSize, int* returnSize){
    int *a = malloc(sizeof(int *[queriesSize])), **q = queries, *x, i;
    for (i = 0 ; ++i < arrSize ; arr[i] ^= arr[i - 1]);
    for (i = -1 ; ++i < queriesSize ; a[i] = arr[(x = q[i])[1]] ^ (x[0] > 0 ? arr[x[0] - 1] : 0));
    return *returnSize = i, a;
}
