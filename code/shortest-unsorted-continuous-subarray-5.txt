int findUnsortedSubarray(int* s, int n){
    if(n<2)return 0;
    int i,l,max,r=0,signal=1;
    for(i=1,l=0,max=s[0];i<n;i++){
        if(s[i]>=max){
            if(signal)l++;
            max=s[i];
        }
        else{
            signal=0;
            for(;l>=0&&s[i]<s[l];l--);
            r=i;
        }
    }
    return r-l>0?r-l:0;
}
