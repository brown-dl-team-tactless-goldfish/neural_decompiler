class Solution {
public:
    int longestContinuousSubstring(string s) {
        int res = 0;
        int cnt = 1;
        int lastChar = 0;
        
		// Since a = 97, b = 98, ... (ASCII)
        for(int c: s) {
            cnt = lastChar + 1 == c ? cnt + 1 : 1;
            lastChar = c;
            res = max(res, cnt);
        }
        
        return res;
    }
};
