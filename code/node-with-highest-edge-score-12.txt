    int edgeScore(vector<int>& graph) {
        int n=graph.size();
        vector<long long> sum(n, 0);
        long long ans=n+1, score=-1; 
        for(int i=0; i<n; i++){
            sum[graph[i]]+=i;
            if(sum[graph[i]]>score)
                ans=graph[i], score=sum[graph[i]];
            if(sum[graph[i]]==score) 
                ans=min(ans, (long long)graph[i]);
        }
        return ans;
    }
