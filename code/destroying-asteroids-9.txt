class Solution {
public:
    bool asteroidsDestroyed(int mass, vector<int>& asteroids) {
        long long int massCopy = mass;
        sort(asteroids.begin(), asteroids.end());
        
        for(int itr = 0; itr < asteroids.size(); itr++){
            if(massCopy >= asteroids[itr]){
                massCopy += asteroids[itr];
            }
            else return false;
        }
        return true;
    }
};
