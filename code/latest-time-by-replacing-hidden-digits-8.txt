string maximumTime(string time) {
	// Minutes
	time[4] = time[4] == '?' ? '9' : time[4];
	time[3] = time[3] == '?' ? '5' : time[3];

	// Hours
	time[0] = (time[0] == '?') ? (time[1] == '?' || time[1] <= '3') ? '2' : '1' : time[0];
	time[1] = time[1] == '?' ? (time[0] == '2' ? '3' : '9') : time[1];
	return time;
}
