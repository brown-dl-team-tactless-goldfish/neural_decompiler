class Solution {
public:
    vector<int> rearrangeArray(vector<int>& nums) {
        
        size_t size(nums.size());
        
        std::vector<int> res(size);
        
        std::sort(nums.begin(), nums.end());
        
        int i{0}, j{0};
        
        for(; j < size; j += 2) {
            res[j] = nums[i++];
        }
        
        for(int k{1}; k < size; k += 2) {
            res[k] = nums[i++];
        }
        
        return res;
    }
};
