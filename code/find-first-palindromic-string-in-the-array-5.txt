bool ispalindrome (char* s) {
    char* t = "";
    int len = strlen(s)-1;
    
    for (int i = 0; i < len; i++, len --) {
        if (s[i] != s[len])
            return false;
    }
    return true;
}

char * firstPalindrome(char ** words, int wordsSize){

    char* ret = "";
    
    for (int i =0; i < wordsSize; i++) {
        bool s = ispalindrome(words[i]);
        if (s)
            return words[i];
    }
    return ret;
}
