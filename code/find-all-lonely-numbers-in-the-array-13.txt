class Solution {
public:
    vector<int> findLonely(vector<int>& nums) {
        unordered_map<int, vector<int>> mp;
        vector<int> result;
        for(int i = 0; i < nums.size(); i++) {
            mp[nums[i]].push_back(i);
        }
        for(auto& m : mp) {
            if(m.second.size() == 1) {
                if(mp.find(m.first - 1) == mp.end() && mp.find(m.first + 1) == mp.end()) result.push_back(m.first);
            }
        }
        return result;
    }
};
