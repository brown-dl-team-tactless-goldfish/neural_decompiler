/**
 * Note: The returned array must be malloced, assume caller calls free().
 */
int* sortArrayByParity(int* nums, int numsSize, int* returnSize){
    
    int j = -1;
    int i = 0;
    int temp = 0;
    for (; i < numsSize; i++) {
        if (!(nums[i]%2)) {
            j++;
            temp = nums[j];
            nums[j] = nums[i];
            nums[i] = temp;
        }
    }

    *returnSize = numsSize;
    return nums;
}
