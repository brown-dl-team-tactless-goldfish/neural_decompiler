#include<string.h>

bool isPalindrome(char* s,int* i,int* j) {
    if (*i>=*j)
        return true;
    
    while(1) {
        if (s[*i]!=s[*j])
            return false;
        (*i)++;
        (*j)--;
        if (*i>=*j)
            return true;
    }
}

bool validPalindrome(char* s) {
    
    unsigned int slen = strlen(s);
    
    int i,j,k,i2,j2;
    
    i = 0;
    j = slen-1;
    if ( isPalindrome(s,&i,&j) == true )
        return true;
    
    i2 = i+1;
    j2 = j;
    
    if ( isPalindrome(s,&i2,&j2) == true )
        return true;
    
    i2 = i;
    j2 = j-1;
    
    if ( isPalindrome(s,&i2,&j2) == true )
        return true;
    
    return false;  
}
