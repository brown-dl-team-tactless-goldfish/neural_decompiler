vector<int> leftRigthDifference(vector<int>& nums) {
    int sz = nums.size();
    vector<int> preSum(sz), sufSum(sz);
    partial_sum(nums.begin(), nums.end(), preSum.begin());
    partial_sum(nums.rbegin(), nums.rend(), sufSum.rbegin());
    vector<int> res(sz);
    for (int i = 0; i < sz; i++) {
        int leftSum = i > 0 ? preSum[i - 1] : 0;
        int rightSum = i + 1 < sz ? sufSum[i + 1] : 0;
        res[i] = abs(rightSum - leftSum);
    }
    return res;
}
