class Solution {
public:

    int clumsy( int N )
    {
        vector<int> A{ N };
        for( auto i{ N-1 }, j{ 0 }; i > 0; --i, ++j )
            if(      j % 4 == 0 ) A.back() *= i;
            else if( j % 4 == 1 ) A.back() /= i;
            else if( j % 4 == 2 ) A.push_back(  i );
            else if( j % 4 == 3 ) A.push_back( -i );
        return static_cast< int >( accumulate( A.begin(), A.end(), 0 ) );
    }

};
