int cmp(const void *x, const void *y){
    return (*(int*)x-*(int*)y);
}

int minimumDifference(int* a, int n, int k)
{
    if(k==1)return 0;
    qsort(a,n,sizeof(int),cmp);
    int l=a[k-1]-a[0];
    for(int i=1;i+k-1<n;i++)
    {
        if(a[i+k-1]-a[i]<l)
            l=a[i+k-1]-a[i];
    }
    return l;
}
