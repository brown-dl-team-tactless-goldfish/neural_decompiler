int numberOfBeams(char ** bank, int bankSize){
    int sum=0, precount=0;
    for(int i=0; i<bankSize; i++){
        int count=0;
        for(int j=0; bank[i][j]!='\0'; j++){
            if(bank[i][j]=='1')
                count++;
        }
        sum+=count*precount;
        if(count!=0)
            precount=count;; 
    }
    return sum;
}
