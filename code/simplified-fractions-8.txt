public class Solution
{
    public IList<string> SimplifiedFractions(int n) 
    {
        var result = new List<string>();
        for(int i = 1; i <= n; i++)
            for(int j = 1; j < i; j++)
                if(GCD(j, i) == 1)
                    result.Add($"{j}/{i}");
        return result;
    }
    
    private int GCD(int x, int y)
    {
        return x == 0 ? y : GCD(y % x, x);
    }
}
