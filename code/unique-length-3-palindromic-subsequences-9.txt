public int CountPalindromicSubsequence(string s) {
        var cnt = 0;
        var arr = s.ToCharArray();
        for (char c = 'a'; c <= 'z'; c++)
        {
            var st = Array.IndexOf(arr, c);
            var end = Array.LastIndexOf(arr, c);
            var hash = new HashSet<char>();
            for (var j = st + 1; j < end; j++)
            {
                hash.Add(arr[j]);
            }
            
            cnt += hash.Count;
        }
        
        return cnt;
    }
