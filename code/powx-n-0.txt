double myPow(double x, int n){
    if(x == 1.0) return x;
    if(x == -1.0) return (n%2 ? -1.0 : 1.0);
    if(n == INT_MIN) return 0;
    double result = 1;
    if(n<0)
    {
        x = 1/x;
        n = -n;
    }
    for(int i = 0; i < n && result*x != result ; i++)
        result *= x;
    return result;
}
