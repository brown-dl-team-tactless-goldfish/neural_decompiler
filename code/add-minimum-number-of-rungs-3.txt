
int addRungs(int* rungs, int rungsSize, int dist){

    int i=0;
    int f=0;
    int count=0;
    int k;
    while(i < rungsSize)
    {
        k = rungs[i] - f;
        if (k > dist)
        {
            k = k - 1;
            count = count + (k/dist);
        }
        f = rungs[i];
        i++;
    }
    return count;
}
