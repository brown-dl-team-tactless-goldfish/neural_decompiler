    public class Solution
    {
        public bool EscapeGhosts(int[][] ghosts, int[] target)
        {
            if (ghosts.Length == 0)
            {
                return true;
            }

            if(ghosts.Min(g => (Math.Abs(g[0] - target[0]) + Math.Abs(g[1] - target[1]))) < (Math.Abs(target[0]) + Math.Abs(target[1])))
            {
                return false;
            }

            return true;
        }
    }
