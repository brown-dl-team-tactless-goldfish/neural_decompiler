long long findTheArrayConcVal(vector<int>& nums) {
	long long res = 0;
	deque<int> dq(nums.begin(), nums.end());
	while (!dq.empty()) {
		string cur;
		if (!dq.empty()) {
			cur += to_string(dq.front());
			dq.pop_front();
		}
		if (!dq.empty()) {
			cur += to_string(dq.back());
			dq.pop_back();
		}
		res += stoi(cur);
	}
	return res;
}
