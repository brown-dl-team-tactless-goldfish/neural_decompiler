int min = 0;
int second_min = 0;
bool changed = false;

void check_num(int val)
{
    if(val < min){
        second_min = min;
        min = val;
    }
    else if((val <= second_min) && (val!=min)){
        changed = true;
        second_min = val;
    }
}




void traverse(struct TreeNode *root)
{
    if(root!=NULL){
        check_num(root->val);
        traverse(root->left);
        traverse(root->right);
    }
}


int findSecondMinimumValue(struct TreeNode* root){
    if(root==NULL){return root;}
    
    min = root->val;
    second_min = INT_MAX;
    changed = false;
    
    traverse(root);
    
    return (changed==false) ? -1 : second_min;
}
