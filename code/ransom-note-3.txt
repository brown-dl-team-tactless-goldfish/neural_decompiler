

bool canConstruct(char * ransomNote, char * magazine){
    int flag=0;
    int i,j;
  
    for(i=0;i<strlen(ransomNote);i++)
    {
     for(j=0;j<strlen(magazine);j++)
    {
        if(ransomNote[i]==magazine [j])
        {
          magazine[j]='0';
          flag=0;
          break;
        }
       else{
         
         flag=1;
         
          }
     }
      if(flag==1)
      break;
    }
  if(flag==0)
    return true;
  else 
    return false;
}
