class Solution {
public:
    int mod=1000000007;
    #define ll long long int
    int sumSubseqWidths(vector<int>& nums) 
    {
        sort(nums.begin(),nums.end());
        int n=nums.size();
        ll pow2[n];
        pow2[0]=1;
        for(int i=1;i<n;i++)
            pow2[i]=(2*pow2[i-1])%mod;
        ll res=0;
        for(int i=0;i<n;i++)
        {
            res=(res+(pow2[i]-pow2[n-i-1])*nums[i])%mod;
        }
        return res;
    }
};
