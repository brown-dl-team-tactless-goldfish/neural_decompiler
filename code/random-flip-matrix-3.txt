class Solution {
private:
    int row ;
    int col ;
    int total ;
    unordered_set<int>used ;
public:
    Solution(int m, int n) {
        row = m ;
        col = n ;
        total = m * n ;
    }
    
    vector<int> flip() {
        if(used.size() == total)
            return {} ;
        while(1){
            int index = rand() % total ;
            if( used.count(index) )
                continue ;
            used.insert(index) ;
            return {index/col, index%col} ;
        }
        return {} ;
    }
    
    void reset() {
        used.clear() ;
    }
};

 * Solution* obj = new Solution(m, n);
 * vector<int> param_1 = obj->flip();
 * obj->reset();
