vector<string> divideString(string s, int k, char fill) {
    string str;
    vector<string>res;
    for(char c : s){
        str += c;
        if(str.size() == k){
            res.push_back(str);
            str = "";   
        }
    }
    if(str.size()){
        for(int i = str.size(); i < k; i++)
            str += fill;
        res.push_back(str);
    }
    return res;
}
