class Solution {
public:
    #define ll long long
    int edgeScore(vector<int>& edges) {
        int n=edges.size();
        vector<ll> ans(n,0);
        for(int i=0;i<n;i++){
            ans[edges[i]]+=i;
        }
        ll maxi=0;//idx
        for(int i=1;i<n;i++){
            if(ans[i]>ans[maxi]){
                maxi=i;
            }
        }
        return maxi;
    }
};
