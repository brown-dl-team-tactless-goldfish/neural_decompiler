class Solution {
public:
    int findMin(vector<int>& nums) {
        int n = (int)nums.size();
        int lo = 0, hi = n-1;
        int mid;
        // see this we are only searching until thse become equal.
        while(lo<hi){
            mid = lo + (hi-lo)/2;
            if(nums[mid]>nums[hi]){
                lo = mid+1; // okay we can search on the right
                // this is because we are guaranteed to find a smaller element on the right search space
                // that is from mid+1 to hi.
            }else{
                hi = mid; // why only till mid, why not hi = mid-1?
                // this is because even mid itself can be the lowest.
            }
        }
        return nums[lo];
        
    }
};
