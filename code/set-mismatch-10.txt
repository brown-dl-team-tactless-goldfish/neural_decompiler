
/**
 * Note: The returned array must be malloced, assume caller calls free().
 */
int* findErrorNums(int* nums, int numsSize, int* returnSize){

    
int i,a[100000];

int* result=(int*)malloc(2*sizeof(int));
*returnSize=2;
    
for(i=0;i<=numsSize;i++)
{
a[i]=0;
}    

for(i=0;i<numsSize;i++)
{
if(a[nums[i]]==1)
result[0]=nums[i];
else if(a[nums[i]]==0)
a[nums[i]]=1;
}
    
for(i=1;i<=numsSize;i++)
{
if(a[i]==0)
{
result[1]=i;
return result;
}  
}         

return;
}
