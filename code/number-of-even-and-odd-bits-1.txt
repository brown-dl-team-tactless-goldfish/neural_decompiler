/**
 * Note: The returned array must be malloced, assume caller calls free().
 */
int* evenOddBit(int n, int* returnSize){
    *returnSize=2;
    int *a=(int *)malloc(2*sizeof(int));
    a[0]=0;
    a[1]=0;
    int i=0;
    while(n!=0)
    {
        if(i%2==0 && n%2==1)
        {
            a[0]+=1;
        }
        else if(i%2!=0 && n%2==1)
        {
            a[1]+=1;
        }
        i++;
        n/=2;        
    }
    return a;
}
