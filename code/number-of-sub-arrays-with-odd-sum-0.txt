int numOfSubarrays(int* arr, int arrSize){
    int even = 1, odd = 0;
    int ans = 0;
    int sum = 0;
    int mod = 1e9 + 7;
    for(int i = 0; i < arrSize; i++){
        sum += arr[i];
        if(sum%2){
            ans = (ans + even)%mod;
            odd++;
        }
        else{
            ans = (ans + odd)%mod;
            even++;
        }       
    }
    return ans;
}
