char * convertToBase7(int num){
	if(num == 0)return "0";
	bool negative = 0;
	if(num < 0){
		negative = 1;
		num *= -1; 
	}
	char* ans = malloc(sizeof(char) * 11);
	int index = 10;
	ans[index] = '\0';
	while(num){
		ans[--index] = num%7 + '0';
		num /= 7;
	}
	if(negative){
		ans[--index] = '-';
	}
	return ans + index;
}
