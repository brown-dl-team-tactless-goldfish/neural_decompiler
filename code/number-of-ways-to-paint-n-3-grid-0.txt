unsigned long long MODULO_CONSTANT = 1000000000 + 7;

int numOfWays(int n) {
	if (n == 1) return (6 + 6);
	if (n == 2) return (6*5 + 6*4);
	int i = 0;
	unsigned long long A = 5;
	unsigned long long B = 4;
	for (i = 3; i < n; i++) {
		unsigned long long tempA = A;
		unsigned long long tempB = B;
		A = (tempA * 3 + tempB * 2) % MODULO_CONSTANT;
		B = (tempA * 2 + tempB * 2) % MODULO_CONSTANT;
	}
	return 6 * (A * 5 + B * 4) % MODULO_CONSTANT;
}
