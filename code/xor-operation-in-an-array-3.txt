int xorOperation(int n, int start){
int* nums=(int *)malloc(sizeof(int)*n);
    int ret;
    int i;
    for(i=0;i<n;i++){
        nums[i] = start + 2*i;
    }
    if(n==1){
        ret=start;
    }
    else{
    int j;
        ret = nums[0]^nums[1];
    for(j=2;j<n;j++){
        ret = ret^nums[j];
    }
}
    return ret;
}
