class Solution {
public:
    int minDifference(vector<int>& nums) {
         if(nums.size()<=3){
             return 0;
         }
         
         sort(nums.begin(), nums.end());
         int min_val = abs(nums[3] - nums[nums.size()-1]);
         min_val = min(min_val, abs(nums[0] - nums[nums.size()-4]));
         min_val = min(min_val, abs(nums[1] - nums[nums.size()-3]));
         min_val = min(min_val, abs(nums[2]-nums[nums.size()-2]));
         return min_val;
         
    }
};
