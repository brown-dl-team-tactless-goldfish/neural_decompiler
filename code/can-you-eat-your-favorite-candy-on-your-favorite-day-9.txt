    vector<bool> canEat(vector<int>& candiesCount, vector<vector<int>>& queries) {
        vector<long> sum = {0};
        vector<bool> res;
        for(auto x:candiesCount) sum.push_back(sum.back() + x);
        for(auto& q:queries){
            long i = q[0], d = 1+q[1], cap = q[2];
            res.push_back(sum[i+1] >= d && cap * d > sum[i]);
        }
        return res;
    }
