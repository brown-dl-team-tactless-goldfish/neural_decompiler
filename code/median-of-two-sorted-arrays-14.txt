  int mid = (nums1Size + nums2Size) / 2;
  
  for(int i =0, j = 0; ;){
      
      if(j == nums2Size) median = nums1[i++];
      else if(i == nums1Size) median = nums2[j++];
      else if(nums2[j] > nums1[i]) median = nums1[i++];
      else median = nums2[j++];
  
      
      if(i+j>mid){
          if((nums1Size+nums2Size)%2 == 0) {
              median = (median + previous) / 2;
          }
          return median;
      }
      previous = median;
