class Solution {
public:
    int minNumberOperations(vector<int>& target) {
        int count=0;
        int operation=0;
        for(int i=0;i<target.size();i++)
        {
            if(target[i]>operation)
            {
                count+=(target[i]-operation);
                operation=target[i];
            }
            else
            {
                operation=target[i];
            }
        }
        return count;
    }
};
