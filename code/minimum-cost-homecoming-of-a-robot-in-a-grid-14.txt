class Solution {
public:
    int minCost(vector<int>& startPos, vector<int>& homePos, vector<int>& row, vector<int>& col) {
        
      /*
      To move from start to end , Cost will be same for all the paths becouse
      Robot will cross the required rows and columns always
      */
      
     int cost=0;
      
  for(int i=min(startPos[0],homePos[0]);i<=max(startPos[0],homePos[0]);i++)
    cost+=row[i];
  
   for(int i=min(startPos[1],homePos[1]);i<=max(startPos[1],homePos[1]);i++)
    cost+=col[i];
       
    cost=cost-row[startPos[0]]-col[startPos[1]]; //subtracting the cost of start pos
  
      return cost;
      
    }
};
