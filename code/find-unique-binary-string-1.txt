

char * findDifferentBinaryString(char ** nums, int n){
    char *ans = malloc(n+1);
    int j = 0;
    
    for (int i = 0; i < n; i++) {
        if (nums[i][i] == '0')
            ans[j++] = '1';
        else
            ans[j++] = '0';   
    }
    
    ans[j] = '\0';
    
    if (j)
        return ans;
    else 
        return "";
}
