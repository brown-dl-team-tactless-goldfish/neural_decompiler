int findTargetSumWays(int* nums, int numsSize, int S) {
    if(numsSize==1){
        if(nums[0]==S&&nums[0]==-S){
            return 2;
        }
        if(nums[0]==S||nums[0]==-S){
            return 1;
        }
        return 0;
    }
    return findTargetSumWays(&nums[1],numsSize-1,S+nums[0])+findTargetSumWays(&nums[1],numsSize-1,S-nums[0]);
}
