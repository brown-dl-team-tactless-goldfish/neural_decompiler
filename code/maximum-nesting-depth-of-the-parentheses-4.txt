int maxDepth(char * s){

    //since s is guaranteed to be VPS, qty of "(" is equal to qty of ")"
    unsigned char i, x, max;
    
    for(i=0,x=0,max=0; i<strlen(s); i++) {
        if (s[i] == ')') { x--; }
        else if (s[i] == '(') { x++; }
        
        if (x > max) max = x;
    }
    
    return max;
}
