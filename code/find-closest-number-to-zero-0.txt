int findClosestNumber(int* nums, int numsSize){
    int ans = INT_MAX;
    int absAns = INT_MAX;
    
    for(int i = 0; i < numsSize; i++){
        if(abs(nums[i]) == absAns && nums[i] > 0)
                ans = nums[i];
        
        else if(abs(nums[i]) < absAns){
            absAns = abs(nums[i]);
            ans = nums[i];
        }
    }
                         
    return ans;
    
}
