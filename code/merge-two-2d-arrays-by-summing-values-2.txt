    vector<vector<int>> mergeArrays(vector<vector<int>>& nums1, vector<vector<int>>& nums2) {
        map<int,int> m;
        for(auto& vv: {nums1, nums2}) for(auto& v:vv) m[v[0]]+=v[1];
        vector<vector<int>> res;
        for(auto& [x,n] : m) res.emplace_back(vector{x,n});
        return res;
    }
