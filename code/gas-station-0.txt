int canCompleteCircuit(int* gas, int gasSize, int* cost, int costSize){
    int cur =0;
    int total = 0;
    int start =0;
    for(int i = 0; i < gasSize; i++){
        cur += gas[i] - cost[i];
        if(cur < 0){
            start = i+1;
            cur = 0;
        }
        total += gas[i] - cost[i];
    }
    
    return (total >= 0) ? start : -1;

}
