string getSmallestString(int n, int k) {
        int count;
        string answer = "";	
		// In the loop below, we are scanning each alpha from 'a' to 'y' 
		// i is the numeric value of alpha
        for(int i = 1; i < 26; i += 1){
            count = floor(n * 26 - k)/(26 - i); // find the maximum count possible
			answer += string(count, 'a'+(i-1)); // add count no. of alpha's to the answer
			// update k and n
            k -= i * count;
            n -= count;    
        }  
        answer += string(n, 'z');  // the remaning letters must be all 'z's
        return answer;
    }
