class Solution {
public:
    int countWays(vector<vector<int>>& ranges)
    {
        sort(ranges.begin(),ranges.end());
        int ans=1,mod=1e9+7;
        int mx=-1;
        for(int i=0;i<ranges.size();i++)
        {
            if(mx<ranges[i][0])
            {
                ans*=2;
                ans%=mod;
                mx=ranges[i][1];
            }
            mx=max(mx,ranges[i][1]);
            
        }
        return ans;
        
    }
};
