class Solution {
public:
    vector<int> mostVisited(int n, vector<int>& rounds) {
        
        ++n;
        vector<int> output;
        int c = rounds.front();
        while(c != rounds.back()) {

            output.push_back(c);
            
            c++;    
            c %= n;
            c = c == 0 ? 1 : c;
        }
        output.push_back(c);
        sort(output.begin(), output.end(), std::less<int>());
        return output;
    }
};
