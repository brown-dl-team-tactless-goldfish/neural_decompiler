char * removeOuterParentheses(char * S){
    char* t=S, *start=S, R[10000];
    int count=0, countP=0, countR=0;;
    for(;*t!='\0';t++){
        count++;
        (*t=='(') ? countP++ : countP--;
        if(!countP){  
            start++;
            for(count-=2;count>0;count--)
                R[countR++] = *start++;
            start=t+1;
        }
    }
    sprintf(S,"%.*s",countR,R);
    return S;
}
