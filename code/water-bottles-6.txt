

int numWaterBottles(int numBottles, int numExchange){
    int cnt = 0;
    int drink = numBottles;
    int empty = 0;
    int total = numBottles;
    if (numBottles < numExchange)
        cnt = numBottles;
    while(total >= numExchange){
       cnt += drink;
       total = drink + empty;
       drink = total / numExchange;
       empty = total % numExchange;
    }
    return cnt;
}
