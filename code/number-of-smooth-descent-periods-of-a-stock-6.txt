class Solution {
public:
    long long getDescentPeriods(vector<int>& prices) {
        long long ans = 1;
        int n = prices.size();
        int i=0, j = 1;
        while(j<n)
        {
            if(prices[j]-prices[j-1]==-1)
            {
                ans += j-i+1;
                j++;
            }
            else
            {
                ans++;
                i=j;
                j++;
            }
        }
        return ans;
    }
};
