int countPrefixes(char ** words, int wordsSize, char * s){


int count = wordsSize;

for(int i=0; i<wordsSize; i++)
{
    int j=0;
    while(words[i][j] != '\0')
    {
      if(words[i][j] != s[j])
      {
          printf("%d ",i);
          count--;
          break;
      }
      j++;
    }
}
return count;
}
