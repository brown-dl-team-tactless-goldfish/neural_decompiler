int b_coefficient(int n, int k)
{
    long res = 1;
    for (int i = 0; i < k; ++i)
    {
        res *= (n - i);
        res /= (i + 1);
    }
    return res;
}
int *getRow(int rowIndex, int *returnSize)
{
    int *arr = (int *)malloc((rowIndex + 1) * sizeof(int));
    *returnSize = rowIndex + 1;
    arr[0] = 1;
    arr[rowIndex] = 1;
    for (int i = 1; i < rowIndex / 2 + 1; ++i)
    {
        arr[i] = b_coefficient(rowIndex, i);
        arr[rowIndex - i] = arr[i];
    }
    return arr;
}
