class Solution {
public:
static bool cmp(string &s1,string &s2){
    if(s1.size()!=s2.size()) return s1.size()<s2.size();
    return s1<s2;
}

string kthLargestNumber(vector<string>& nums, int k) {
    sort(nums.begin(),nums.end(),cmp);
    if(k>nums.size()) return "";
    return nums[nums.size()-k];
}
};
