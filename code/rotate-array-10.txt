#define SWAP(x,y) do {int tmp; tmp = (x); (x) = (y); (y) = tmp; } while(0)

void reverse(int* nums, int from, int to)
{
    while (from < to) {
        SWAP(nums[from], nums[to]);
        from++;
        to--;
    }
}

void rotate(int* nums, int numsSize, int k){
    if (k >= numsSize) {
        k %= numsSize;
    }
    
    if (k == 0) {
        return;
    }
    
    reverse(nums, 0, numsSize-1);
    
    reverse(nums, 0, k-1);
    reverse(nums, k, numsSize-1);
}
