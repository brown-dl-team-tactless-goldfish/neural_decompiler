 class Solution {
public:
    vector<int> numOfBurgers(int t, int c) {
        if(t < 2*c || t % 2 || t > 4*c) return {};
        return {(t - 2*c) / 2, c - (t - 2*c)/2};
    }
};

// if tomato slices are less than the 2*cheese slices then we can't use all cheese slice.
// if tomato slices are odd then we can't use all tomato slices
// if tomato slices are more than the cheese slices then we can't use all tomato slices
// initially we assume that all burgers are small one(small burger = number of cheese slice). Now the half of the remaning tomato slice (remaining tomato slice = t - 2*c) can make big burger so remove it from small burger.
