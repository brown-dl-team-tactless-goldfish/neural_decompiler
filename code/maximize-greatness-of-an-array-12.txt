class Solution {
public:
    int maximizeGreatness(vector<int>& nums) {
        sort(nums.begin(), nums.end());
        int res = 0;
        auto i = nums.begin(), j = nums.begin();
        while (j < nums.end()) {
            int count = upper_bound(i, nums.end(), *i) - i;
            j = upper_bound(j, nums.end(), *i);
            res += std::min(count, (int)(nums.end() - j));
            i += count;
            j += count;
        }
        return res;
    }
};
