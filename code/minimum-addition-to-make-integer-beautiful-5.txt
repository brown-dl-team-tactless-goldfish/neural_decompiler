    long long makeIntegerBeautiful(long long n, int target) {
        long long res = 0;
        for(long long pow=1; sumdig(n)>target; pow*=10){
            int last = (n/pow) %10;
            if(last!=0) {
                res += pow*(10-last);
                n += pow*(10-last);
            }
        }
        return res;
    }
    
    int sumdig(long long n){
        int sum = 0;
        while(n){
            sum += n% 10;
            n/=10;
        }
        return sum;
    }
