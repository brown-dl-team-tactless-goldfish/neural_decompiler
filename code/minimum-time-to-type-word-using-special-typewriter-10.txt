class Solution {
public:
    int minTimeToType(string word) {
        int ans = word.size();
        
        int prev = 0;
        
        for (auto c : word) {
            int i = c - 'a';
            
            if (i < prev) {
                i += 26;
            }
            
            ans += min(i - prev, 26 - (i - prev));
            
            prev = c - 'a';
        }
        
        return ans;
    }
};
