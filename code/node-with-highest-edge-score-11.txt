class Solution {
public:
    int edgeScore(vector<int>& edges) {
        map<long,long>m;
        for(int i=0; i<edges.size();i++)
        {
            m[edges[i]] += i;
        }

        long long maxe = -1, maxval = -1;

        for(auto it = m.begin(); it!= m.end(); it++)
        {
            if(it->second > maxval)
            {
                maxe = it->first;
                maxval = it->second;
            }
        }

        return maxe;
    }
};
