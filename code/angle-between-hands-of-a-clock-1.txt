#define MIN(a,b) ((a)<(b)?(a):(b))

double angleClock(int hour, int minutes)
{
    int cir = 360*2;
    int deg_hour = cir/12*hour;
    deg_hour += cir/60*minutes/12.0;
    int deg_mins = cir/60*minutes;
    return MIN(MIN(abs(deg_hour-deg_mins), abs(deg_mins-deg_hour)),
      MIN(abs(deg_hour+cir-deg_mins),abs(deg_mins+cir-deg_hour)))/2.0;
}
