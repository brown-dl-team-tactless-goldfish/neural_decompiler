int lengthOfLongestSubstring(char * s){
char *address[128]={NULL};
    int len=0;
    char *tmp,*start=s;
    while(*start)
    {
        tmp=address[*start];
        address[*start]=start;
        if(tmp>=s)
        {
            len=len>start-s?len:start-s;
            s=tmp+1;
        }
        start++;
    }
    len=len>start-s?len:start-s;
    return len;
}
