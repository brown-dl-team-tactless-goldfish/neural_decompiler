char ** printVertically(char * s, int* returnSize){
    char **v = malloc(sizeof(char *[100])), *w[100], *p, *r, sp, i = 0, k = 0;
    for (p = s-- ; *++s || (w[i++] = p, 0) ; !isalpha(*s) ? *s = 0, w[i++] = p, p = s + 1 : 0);
    for (r = v[k] = malloc(100), sp = 0 ; !sp++ ; !sp ? r = v[k] = malloc(100) : free(v[k])) {
        for (int j = -1 ; ++j < i ; (*r++ = *w[j] ? *w[j]++ : ' ') != ' ' ? sp = 0 : 0);
        while (!sp && (*--r == ' ' || (k++, r[1] = 0)));
    }
    return *returnSize = k, v;
}
