bool canMakeArithmeticProgression(int* arr, int arrSize){
    int temp, n = arrSize;
    for(int i = 0; i < n; i++){
        for(int j = i+1; j < n; j++){
            if(arr[i] > arr[j]){
                temp = arr[i];
                arr[i] = arr[j];
                arr[j] = temp;
            }
        }
    }
    for(int i = 1; i < n-1; i++)
        if((arr[i] - arr[i-1]) != (arr[i+1] - arr[i]))
            return false;
    return true;
}
