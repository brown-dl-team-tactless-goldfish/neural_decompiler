class OrderedStream {
public:
    unordered_map<int, string> records;
    int ptr = 1;
    
    OrderedStream(int n) {}
    
    vector<string> insert(int id, string value) 
    {
        records[id] = value;
        vector<string> res;
        
        while(records.count(ptr)) {
            res.push_back(records[ptr]);
            ptr++;
        }
        return res;
    }
};
