public class Solution {
    public int CountOrders(int n) {
        
        int mod = (int)(1e9 + 7);
        long res = 1;
        for(int i = 1; i <= n; i++)
        {
            res *= (2 * i - 1);
            res %= mod;
            res *= i;
            res %= mod;
        }
        
        return (int)res % mod;
    }
}
