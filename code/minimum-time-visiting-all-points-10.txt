int minTimeToVisitAllPoints(int** points, int pointsSize, int* pointsColSize){
    // If there are zero or 1 points, then the total time it would
    if(pointsSize < 2)
        return 0; 
    
    // Total distance away. 
    int m = 0; 
    
    // Run through each point. 
    for(pointsSize = pointsSize-1; pointsSize > 0; pointsSize--){
        // Since we are only looking at distance away
        int x = abs(points[pointsSize-1][0] - points[pointsSize][0]); 
        int y = abs(points[pointsSize-1][1] - points[pointsSize][1]); 
        // Whichever one is the greatest distance away, it will take that long to get there
        if(x >= y)
            m += x; 
        else{
            m += y; 
        }
    }   
    return m; 
}
