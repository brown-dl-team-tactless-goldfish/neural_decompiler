char * minRemoveToMakeValid(char * s){
    int oParen = 0;     // Open parenthesis count
	int cParen = 0;     // Close parenthesis count
    int rIdx = 0;       // Resultant index
    
	int n = strlen(s);
	for (int i = 0; i < n; i++)  
        if (s[i] == ')') 
            cParen++;
	        
	for (int i = 0; i < n; i++)  {
		if (s[i] == '(') {
			if (oParen == cParen)    	// Once balanced, do not allow this ( to add
                continue;           
			
            oParen++;
		} 
        else if (s[i] == ')') {
			cParen--;
			if (oParen == 0) 			// once balanced, do not allow this ) to add
                continue;
				
            oParen--;
		} 
		
        s[rIdx++] = s[i];
	}
	
    s[rIdx] = '\0';
	return s;
}
