class Solution {
public:
    int splitNum(int num) {
        vector<int> digits;
        
        //save all digits
        for (char c : to_string(num))
            digits.push_back(c - '0');
        
        //sort it, so that we can get lower number first
        sort(begin(digits), end(digits));
        
        int n1 = 0, n2 = 0;
        bool flag = true;
        
        //keep add the lowest number to n1 and n2 
        //we give the lowest digit to n1, the second lowest digit to n2, the third lowest digit to n1 and so on...
        for (int n : digits){
            if (flag)
                n1 = n1 * 10 + n;
            else
                n2 = n2 * 10 + n;
            
            flag = !flag;
        }
        
        //that the minimum number we can get, so return the sum of them
        return n1 + n2;
    }
};
