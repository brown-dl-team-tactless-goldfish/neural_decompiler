int cmp(const void *x, const void *y)
{
    int a = *(int*)x;
    int b = *(int*)y;
    return a-b;
}
bool divideArray(int* nums, int numsSize)
{
    bool x = false;
    qsort(nums, numsSize, sizeof(int), cmp);
    for(int i = 0; i < numsSize; i = i + 2)
    {
        if(nums[i] == nums[i+1])
        {
            x = true; 
        }
        else
        {
            x = false;
            break;
        }
    }
    return x;
}
