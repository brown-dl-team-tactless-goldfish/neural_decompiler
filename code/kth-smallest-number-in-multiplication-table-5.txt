class Solution(object):
    def findKthNumber(self, m, n, k):
        l,r=1,n*m
        while l <= r:
            mid = l + (r-l)//2
            if self.f(mid,m,n,k):
                r = mid - 1
            else:
                l = mid +1
        return l
    
    def f(self,mid,m,n,k):
        c = 0
        for i in range(1,m+1):
            c += min(mid//i,n)
        return c >= k
