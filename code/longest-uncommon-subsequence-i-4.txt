int
findLUSlength(char *a, char *b)
{
	bool doa, dob, eq;
	char cha, chb;
	int lena, lenb;

	doa = dob = eq = true;
	lena = lenb = 0;
	do {
		if (doa = doa && (cha = *a++) != '\0')
			++lena;
		if (dob = dob && (chb = *b++) != '\0')
			++lenb;
		eq = eq && doa == dob && cha == chb;
	} while (doa || dob);

	return eq ? -1 : lena > lenb ? lena : lenb;
}
