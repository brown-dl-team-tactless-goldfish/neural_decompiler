
#define TYPE 0
#define COLOR 1
#define NAME 2

int countMatches(char *** items, int itemsSize, int* itemsColSize, char * ruleKey, char * ruleValue){
    
    int NumMatches_i = 0;
    int Index_i = 0;
    
    /*
  -> TYPE COLOR NAME
   0 phone blue pixel
   1 compu silv lenov
   2 phone gold iphon    
    */
    
    switch(ruleKey[0])
    {
        case 't':
                Index_i = TYPE;
                break;
        case 'c':
                Index_i = COLOR;
                break;
        default:
                Index_i = NAME;
                break;
    }
    
    for(int i=0; i<itemsSize; i++)
    {
        if(strcmp(items[i][Index_i], ruleValue) == 0)
            NumMatches_i++;
    }    
    
    return NumMatches_i;    
}
