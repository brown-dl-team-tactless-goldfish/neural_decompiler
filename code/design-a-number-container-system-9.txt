class NumberContainers {
public:
    map<int, int> keyToVal;
    map<int, set<int>> index;
    NumberContainers() {}
    
    void change(int indx, int number) {
        if(keyToVal.find(indx) != keyToVal.end())
        {
            int oldVal = keyToVal[indx];
            index[oldVal].erase(indx);
        }
        keyToVal[indx] = number;
        index[number].insert(indx);
    }
    
    int find(int number) {
        if(index[number].size() == 0) return -1;
        return *(index[number].begin());
    }
};
