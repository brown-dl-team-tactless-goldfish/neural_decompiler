int cmpfunc (const void * a, const void * b) {
   return ( *(int*)a - *(int*)b );
}

int minPairSum(int* nums, int numsSize){
    
    int maxsum = (int) malloc(sizeof(int));
    maxsum = 0;
    int temp_maxsum = 0;
    qsort(nums,numsSize,sizeof(int),cmpfunc);
    for(int i=0;i<numsSize/2;i++)
    {
        temp_maxsum = (nums[i] + nums[numsSize - i - 1]);
        if(maxsum < temp_maxsum)
            maxsum = temp_maxsum;
        
    }
    
    return maxsum;
}
