int countAsterisks(char * s){
    int counter, ans;
    for (counter = ans = 0; *s != '\0'; s++) {
        if (*s == '|') {
            if (counter == 0)
                counter++;
            else
                counter--;
        } else if (*s == '*') {
            if (counter == 0)
                ans++;
        }
    }
    return ans;
}
