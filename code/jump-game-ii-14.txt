int jump(int* nums, int numsSize){
    int arr[numsSize+1], j = 0;
    for (int i = 0; i < numsSize; i++) {
        arr[i] = nums[i] + i;
        if (arr[j] < i) {
            if (arr[++j] >= numsSize - 1) break;
            i--;
        }
        else if (arr[i] > arr[j+1]) arr[j+1] = arr[i];
    }
    return (numsSize == 1) ? 0 : j+1;
}
