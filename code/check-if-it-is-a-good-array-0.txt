int gcd(int a, int b)
{
    if (a == 0) return (b);
    return (gcd(b % a , a));
}

bool isGoodArray(int* nums, int numsSize)
{
    if (numsSize == 1) return (nums[0] == 1);
    int flag = gcd(nums[0],nums[1]);
    for(int i = 2 ; i < numsSize ; i++) flag = gcd(flag, nums[i]);
    return (flag == 1);
}
