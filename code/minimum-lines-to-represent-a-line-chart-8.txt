    int minimumLines(vector<vector<int>>& v) {
        if(v.size() == 1) return 0;
        sort(v.begin(), v.end());
        int res = 1, dx = v[1][0] - v[0][0], dy = v[1][1] - v[0][1];
        for(int i=2;i<v.size();i++){
            long long dx1 = v[i][0] - v[i-1][0], dy1 = v[i][1] - v[i-1][1];
            if(dx1*dy != dx*dy1){
                res ++;
                dx = dx1, dy = dy1;
            }
        }
        return res;
    }
