int findPoisonedDuration(int* timeSeries, int timeSeriesSize, int duration) {
    if(timeSeriesSize==0){return 0;}
    int i=0;
    int sum=0;
    while(i<timeSeriesSize-1){
        sum=sum+(timeSeries[i]+duration>=timeSeries[i+1]?timeSeries[i+1]-timeSeries[i]:duration);      
        i++;
    }
    sum=sum+duration;
    return sum;
}
