char * capitalizeTitle (char * title)
{	
	int i=-1, wlen=0;
	while( NULL != title[++i] ) { 
		++wlen;
		title[i] = tolower(title[i]);

		if(title[i]==' ') {
			--wlen;
			if(wlen>2)
				title[i-wlen] = toupper(title[i-wlen]);
			wlen =0;
		}
	}
	/*since the last word is terminated with a NULL and not a space*/
	if(title[i]==NULL && wlen>2)
		title[i-wlen] = toupper(title[i-wlen]);

	return title;
}
