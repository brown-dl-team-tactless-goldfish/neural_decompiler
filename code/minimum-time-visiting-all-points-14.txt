int max(int a, int b){
    if(a<0) a=-a;
    if(b<0) b=-b;
    return a>b?a:b;
}

int minTimeToVisitAllPoints(int** points, int pointsSize, int* pointsColSize){
    int sum=0;
    for(int i=0;i<pointsSize-1;i++){
        int opt=0;
        opt=max(points[i][0]-points[i+1][0], points[i][1]-points[i+1][1]);
        sum+=opt;
    }
    
    return sum;
}
