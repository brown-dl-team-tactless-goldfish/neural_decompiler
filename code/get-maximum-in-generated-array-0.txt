int getMaximumGenerated(int n){
    int* data = malloc( (n + 1) * sizeof(int) );
    data[0] = 0;
    int max = 0;
    if(n > 0){
        data[1] = 1;
        max = 1;    
    }
    if(n > 1){
        data[2] = 1;
        max = 1;
    }
    for(int i = 3; i <=n ;i++){
        if(i % 2)
            data[i] = data[i/2] + data[i/2 + 1];
        else
            data[i] = data[i/2];
        if(data[i] > max)
            max = data[i];
    }
    return max;

}
