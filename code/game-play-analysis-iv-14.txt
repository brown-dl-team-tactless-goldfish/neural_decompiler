with cte as
(select player_id, 
    min(event_date) as first_date
    from Activity
group by player_id
),
b as 
(select count(*) as cnt from cte as a where exists 
(select 1 from Activity where player_id = a.player_id
 and datediff(event_date, a.first_date) = 1)),
 c as
 (select count(distinct player_id) as total from Activity)
 
 select round(cnt/total,2) as fraction from b inner join c;
