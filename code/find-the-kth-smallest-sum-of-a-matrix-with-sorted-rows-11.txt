    int kthSmallest(vector<vector<int>>& mat, int k) {
        vector<int> pre = {0}, cur;
        for(const auto& v:mat){
            for(auto x:pre) for(auto y:v) cur.push_back(x+y);
            sort(cur.begin(),cur.end());
            while(cur.size()>k) cur.pop_back();
            pre = move(cur);
        }
        return pre[k-1];
    }
