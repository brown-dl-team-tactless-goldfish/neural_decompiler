SELECT TOP 1 c.SecondHighestSalary FROM (
    (SELECT MIN(A.salary) AS SecondHighestSalary FROM 
        (SELECT TOP 2 salary FROM Employee GROUP BY salary ORDER BY salary DESC
        ) AS A HAVING COUNT(salary)>1
    )
    UNION 
    (SELECT NULL AS SecondHighestSalary
    )
) as c ORDER BY SecondHighestSalary DESC
