

char slowestKey(int* releaseTimes, int releaseTimesSize, char * keysPressed){
    
    int i ;
    int sum=0;
    
    int max=releaseTimes[0]; 
    
    char letter=keysPressed[0];

    
    for(i=1; i<releaseTimesSize;i++){
        
        
        sum=releaseTimes[i]-releaseTimes[i-1];
        
        if(max==sum){
            if(keysPressed[i]>letter){
                letter=keysPressed[i];
            }
        }
        
        if(max< sum){
            max=sum;
            letter = keysPressed[i];
        }
        
        
        
    }

    return letter;
    
        
}
