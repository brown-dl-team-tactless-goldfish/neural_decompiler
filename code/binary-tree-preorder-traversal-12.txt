void Howmanyroot(struct TreeNode* root, int* count)
{

	if(root==NULL)
	{
		return;
	}
	(*count) ++;
	Howmanyroot(root->left,count);
	Howmanyroot(root->right,count);

}


void Toarray(struct TreeNode* root, int* array,int* nth)
{
	if(root==NULL)
	{
    return;
	}

	array[*nth]=root->val;
	(*nth)++;
	Toarray(root->left,array,nth);
	Toarray(root->right,array,nth);

}




int* preorderTraversal(struct TreeNode* root, int* returnSize){

	int count=0;
	Howmanyroot(root,&count);

	*returnSize =count;

	int* ans=calloc(sizeof(int*),(*returnSize));
	int nth=0;
	Toarray(root,ans,&nth);

	return ans;
}
