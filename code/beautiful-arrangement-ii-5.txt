for(int i = 0,right = n ; i< k/2    ; i++, right-- )
    resultArr[2*i] = right;

for(int i = 0,left = 1 ; i< (k-1)/2 ; i++, left++ )
    resultArr[1 + 2*i] = left;
    
 for(int i= 0 ; i< n-k+1 ; i++){
     if(k % 2 == 0)//偶数正序
        resultArr[i+k-1] = i + (k+1)/2 ;
     else//奇数逆序
        resultArr[i+k-1] = n-i -k/2 ;
 }

return resultArr;
