void duplicateZeros(int* arr, int arrSize){
    /*i need to tested from 0 till arrSize -2 since last element cannot anyways be duplicated.*/
	for(int i = 0; i < arrSize - 1; i++) {
		if(arr[i] == 0) {            
		    /* Copy the elements backward, so that it can be done inplace. */
			for(int j = arrSize - 2; j > i; j--) 
				arr[j + 1] = arr[j];
			/* Duplicate next element */	
			arr[i + 1] = 0;
			i++;
		}
	}
	return;
}
