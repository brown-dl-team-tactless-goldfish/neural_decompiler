class Solution {
public:
    int destroyTargets(vector<int>& nums, int space) {
        int ans = INT_MAX;
        unordered_map<int, int> mp;
        int mx = INT_MIN;
        for(auto n: nums){
            int r = n % space;                  
            mp[r]++;                               
            if(mx < mp[r])  mx = mp[r];  
        }
        for(auto n: nums){ 
            if(mx == mp[n%space])  ans = min(ans, n);
        }
        return ans;
    }
};
