class Solution {
public:
    vector<int> cycleLengthQueries(int n, vector<vector<int>>& Q) {
        vector<int> res(Q.size(), 0);
        for(int i = 0; i < Q.size(); i++) {
            res[i]++;
            int a = Q[i][0], b = Q[i][1];
            while(a != b) {
                if(a > b) a >>= 1;
                else b >>= 1;
                res[i]++;
            }
        }
        return res;
    }
};
