

int nearestValidPoint(int x, int y, int** points, int pointsSize, int* pointsColSize){
    int i,j,min=987654321,idx=-1,temp;
    for(i=0;i<pointsSize;i++){
        if(points[i][0]==x||points[i][1]==y){
            temp=abs(x-points[i][0])+abs(y-points[i][1]);
            if(temp<min){
                min=temp;
                idx=i;
            }
        }        
    }

    return idx;
}
