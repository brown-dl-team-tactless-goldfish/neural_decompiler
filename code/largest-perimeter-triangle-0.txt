int cmp(const void *a, const void *b) {
    int *a1 = (int *)a;
    int *b1 = (int *)b;
    
    return *b1 - *a1;
}

int largestPerimeter(int* nums, int numsSize){
    int ans = 0;
    qsort(nums, numsSize, sizeof(nums[0]), cmp);
    for (int i=0; i<numsSize-2; i++) {
        int a = nums[i], b = nums[i+1], c = nums[i+2];
        if (a + b > c && a + c > b && b + c > a) {
            ans = fmax(ans, a+b+c);
        }
    }
    return ans;
}
