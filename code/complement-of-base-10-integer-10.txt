

int bitwiseComplement(int n){
    if (n == 0) {
        return 1;
    }
	// 0000 0101 n
	// 0000 0111 mask
	// 1111 1010 ~n
	// 0000 0010 (~n) & mask
    int finalOne = 0;
    int tmp = n;
    while (tmp) {
        finalOne++;
        tmp = tmp >> 1;
    }
    int mask = 0;
    for (int i = 0; i < finalOne; i++) {
        mask += (1 << i);
    }
    
    return (~n) & mask;
}
