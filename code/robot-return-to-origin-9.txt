bool judgeCircle(char * moves){
    int v = 0;
    int h = 0;
    while (*moves) {
        if (*moves == 'U'){
            v++;
        }
        else if (*moves == 'D'){
            v--;
        }
        else if (*moves == 'R'){
            h++;
        }
        else if (*moves == 'L'){
            h--;
        }
        moves++;
    }
    return v == 0 && h == 0;
}
