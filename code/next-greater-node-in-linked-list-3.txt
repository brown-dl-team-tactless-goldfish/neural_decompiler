int* nextLargerNodes(struct ListNode* head, int* t)
{
    int *res=(int *)malloc(sizeof(int)*10000);
    int k=0,m;
    struct ListNode *p=head;
    while(p!=NULL)
    {
        m=0;
        struct ListNode *q=p->next;
        while(q!=NULL)
        {
            if(q->val>p->val&&q->val>m)
            {
                m=q->val;
                break;
            }
            q=q->next;
        }
        p=p->next;
        res[k++]=m;
    }
    *t=k;
    return res;
}
