class Solution {
public:
    int minNumberOfHours(int initialEnergy, int initialExperience, vector<int>& energy, vector<int>& experience) {
        int n = energy.size(), needenergy = 0, curexp = initialExperience, needexp = 0;
        for (int i = 0; i < n; i++)
        {
            needenergy += energy[i];
            needexp += max(0, experience[i] + 1 - curexp - needexp);
            curexp += experience[i];
        }
        needenergy = max(0, needenergy + 1 - initialEnergy);
        return (needenergy + needexp);
    }
};
