class Solution {
public:
    int xorAllNums(vector<int>& a, vector<int>& b) {
        int n = a.size(); 
        int m = b.size(); 
        int ans = 0; 
        for (int i = 0; i < n; ++i) {
            if (m % 2) ans ^= a[i]; 
        } 
        
        for (int i = 0; i < m; ++i) {
            if (n % 2) ans ^= b[i]; 
        }
        
        return ans; 
    }
};
