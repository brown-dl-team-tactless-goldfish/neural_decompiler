class Solution {
public:
     string bestHand(vector<int>& ranks, vector<char>& suits) {
        unordered_map<int, int> rmp;
        for(auto s: suits) rmp[s]++;
        for(auto m: rmp) if(m.second == 5) return "Flush";
        rmp.clear();
        for(auto r: ranks) rmp[r]++;
        for(auto m: rmp) if(m.second >= 3) return "Three of a Kind";
        for(auto m: rmp) if(m.second >= 2) return "Pair";
        return "High Card";
    }

};
