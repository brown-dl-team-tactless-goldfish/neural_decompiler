class Solution {
public:
    string bestHand(vector<int>& ranks, vector<char>& suits) {
        bool valid = true;
        for (auto& c : suits) if (c != suits[0]) valid = false;
        if (valid) return "Flush";
        unordered_map<int,int> m;
        for (auto& n : ranks) ++m[n];
        for (auto& [f,s] : m) if (s >= 3) return "Three of a Kind";
        for (auto& [f,s] : m) if (s == 2) return "Pair";
        return "High Card";
    }
};
