public class Solution {
    public int NumberOfBoomerangs(int[][] points) {
        var result = 0;

        var n = points.Length;

        if (n == 0) return 0;

        for (int i = 0; i < n; i++) {
            var distanceAndCount = new Dictionary<int, int>();

            for (int j = 0; j < n; j++) {
                if (i == j) continue;

                var a = points[i][0] - points[j][0];
                var b = points[i][1] - points[j][1];

                var distance = a * a + b * b;

                if (!distanceAndCount.ContainsKey(distance)) {
                    distanceAndCount[distance] = 0;
                }

                distanceAndCount[distance]++;
            }

            foreach (var value in distanceAndCount.Values) {
                result += value * (value - 1);
            }
        }

        return result;
    }
}
