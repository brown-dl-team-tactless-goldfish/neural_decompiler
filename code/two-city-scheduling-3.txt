int cmp(const int*a,const int*b)
{
    int *x=*(int **)a;
    int *y=*(int**)b;
    return (x[0]-x[1])-(y[0]-y[1]);  
}
int twoCitySchedCost(int** costs, int costsSize, int* costsColSize)
{
    int total=costsSize;
    qsort(costs,costsSize,sizeof(int*),cmp);
    int result=0;
    for(int i=0;i<costsSize;i++)
    {
        if(i<total/2)
        result+=costs[i][0];
        else
            result+=costs[i][1];
    }
    return result;
}
