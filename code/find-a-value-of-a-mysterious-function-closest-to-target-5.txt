int closestToTarget(vector<int>& arr, int target) {
	int res = INT_MAX;

	for (int i = 0; i < arr.size(); ++i) {
		res = min(res, abs(target - arr[i]));

		for (int j = i - 1; j >= 0; --j) {
			arr[j] &= arr[i];
			res = min(res, abs(target - arr[j]));
			if (arr[j] <= target || arr[j] == arr[j + 1]) break;
		}

	}

	return res;
}
