public class Solution {
    public int GetLucky(string s, int k) {
        StringBuilder sb = new StringBuilder();

        foreach(var c in s){
            sb.Append(c-'a'+1);
        }

        for(int i=0; i<k; i++){
            int temp = 0;

            for(int j=0; j<sb.Length; j++){
                temp += sb[j] - '0';
            }

            sb.Clear();
            sb.Append(temp);
        }

        return Convert.ToInt32(sb.ToString());
    }
}
