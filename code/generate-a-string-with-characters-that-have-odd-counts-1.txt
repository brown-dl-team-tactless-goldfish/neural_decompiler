char* generateTheString(int n) {
    int part1 = 0, part2 = 0, idx = 0;
    char* ans = (char*) malloc((n + 1) * sizeof(char));
    if (n % 2 == 0) {
        part1 = (n / 2);
        part2 = part1;
        if ((n / 2) % 2 == 0) {
            part1--;
            part2++;
        }
    } else {
        part1 = n;
    }
    while (idx < part1) {
        ans[idx++] = 'a';
    }
    while (idx < n) {
        ans[idx++] = 'b';
    }
    ans[idx] = '\0';
    return ans;
}
