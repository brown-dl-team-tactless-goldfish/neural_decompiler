bool checkIfCanBreak (char * s1, char * s2) {
    int i, sum1, sum2, res1, res2;
    int cnt1[26] = {0};
    int cnt2[26] = {0};
    
    for (i = 0; s1[i]; i++) {
        cnt1[s1[i]-'a']++;
        cnt2[s2[i]-'a']++;
    }
    
    sum1 = sum2 = 0;
    res1 = res2 = 1;
    for (i = 0; i < 26; i++) {
        sum1 += cnt1[i];
        sum2 += cnt2[i];
        if (sum1 > sum2) res1 = 0;
        if (sum2 > sum1) res2 = 0;
    }
    
    return res1 || res2;
}
