public class Solution {
    public IList<string> PrintVertically(string s) {
        if (s == null || s == string.Empty)
            return new List<string>();
        
        IList<string> res = new List<string>();
        string[] strs = s.Split(' ');
        StringBuilder[] builders = new StringBuilder[strs.OrderByDescending(x => x.Length).ToList()[0].Length];
        
        for (int i = 0; i < builders.Length; i++)
            builders[i] = new StringBuilder();
        
        foreach (var str in strs)
        {
            int i = 0;
            
            for (; i < str.Length; i++)
                builders[i].Append(str[i]);
            
            while (i < builders.Length)
                builders[i++].Append(" ");
        }
        
        for (int i = 0; i < builders.Length; i++)
            res.Add(builders[i].ToString().TrimEnd());
        
        return res;
    }
}
