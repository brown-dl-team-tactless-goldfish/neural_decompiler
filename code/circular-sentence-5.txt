class Solution {
public:
    bool isCircularSentence(string sentence) {
        int n = sentence.size();
        int l=0, r = n-1;
        if(sentence[l]!=sentence[n-1]) return false;
        r = l;
        while(r<n && sentence[r]!=' ') r++;
        while(r<n){
            l = r-1;
            r++;
            if(sentence[l]!=sentence[r]) return false;
            while(r<n && sentence[r]!=' ') r++;
        }
        return true;
        
    }
};
