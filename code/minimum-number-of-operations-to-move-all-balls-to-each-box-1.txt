/**
 * Note: The returned array must be malloced, assume caller calls free().
 */
int* minOperations(char * boxes, int* returnSize){
	int len = strlen(boxes);
	int* ans = malloc(sizeof(int) * len);
	memset(ans, 0, sizeof(int) * len);
	*returnSize = len;
	for(int i = 0; i < len; i++){
		if(boxes[i] == '0') continue;
		for(int j = 0; j < len; j++){
			ans[j] += abs(i - j);
		}
	}
	return ans;
}
