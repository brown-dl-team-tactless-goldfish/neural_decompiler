int MaxDepth, sum=0;
void getSum(struct TreeNode* root, int depth){
    if(!root) 
        return 0;
    if (MaxDepth == depth)
        sum += root->val;
    if(depth > MaxDepth){
        MaxDepth = depth;
        sum = root->val;
    }
    getSum(root->left, depth+1);
    getSum(root->right, depth+1);
}
int deepestLeavesSum(struct TreeNode* root){
    MaxDepth = 0;
    getSum(root, 0);
    return sum;
}
