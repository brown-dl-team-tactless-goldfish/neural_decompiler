  vector<TreeNode*> allPossibleFBT(int N) {
    if (N % 2 == 0) return {};
	
    vector<TreeNode*> dp[N+1];
    dp[1] = {new TreeNode()};
    for (int n = 3; n <= N; n += 2) 
      for (int i = 1; i < n-1; i += 2) 
        for (TreeNode* left : dp[i]) 
          for (TreeNode* right : dp[n-i-1]) 
            dp[n].emplace_back(new TreeNode(0, left, right));
    return dp[N];
  }
