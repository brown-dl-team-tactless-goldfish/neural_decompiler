class Solution {
public:
    vector<int> successfulPairs(vector<int>& spells, vector<int>& potions, long long success) {
        sort(potions.begin(), potions.end());
        int n = potions.size();
        vector<int> ans;
        for(auto &e : spells){
            long long c = success/(long long)e;
            if(success%e > 0) ++c;
            
            auto lb = lower_bound(potions.begin(), potions.end(), c) - potions.begin();
            ans.push_back(n-lb);
        }
        return ans;
    }
};
