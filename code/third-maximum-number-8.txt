    int mx[3] = { INT_MIN, INT_MIN, INT_MIN };
    int ct[3] = {};
	
    for (int i = 0; i < 3; ++i) {
        for (int j = 0; j < numsSize; ++j) {
            if ((i == 0 || nums[j] < mx[i - 1]) && (nums[j] > mx[i] - 1)) {
                mx[i] = nums[j];
                ++ct[i];
            }
        }
    }
    
    return ct[2] ? mx[2] : mx[0];
