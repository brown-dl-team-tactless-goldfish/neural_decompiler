class Solution {
public:
    int maxWidthOfVerticalArea(vector<vector<int>>& points) {
        vector<int> v;
        for(vector<int> t : points) v.push_back(t[0]);
        sort(v.begin(),v.end());
        int c = 0;
        for(int i=1;i<v.size();i++) c = max(c, v[i]-v[i-1]);
        return c;
    }
};
