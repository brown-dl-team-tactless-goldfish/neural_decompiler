int countSegments(char * s){
    
    if (s == NULL) {
        return 0;
    }
    
    int segments = 0;
    int nonSpaceState = 0;
    int i = 0;
    
    while (s[i] != '\0') {
        if (s[i] != ' ') {
            segments += 1 - nonSpaceState;
            nonSpaceState = 1;
        } else {
            nonSpaceState = 0;
        }
        i++;
    }
    return segments;
}
