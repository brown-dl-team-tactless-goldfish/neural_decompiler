class Solution {
public:
    int find(vector<int>&nums,int k)
    {
        int j=0;
        int ans=0;
        for(int i=0;i<nums.size();i++)
        {
            if(nums[i]>k)
            {
                j=i+1;
            }
            ans+=i-j+1;
        }
        return ans;
    }
    int numSubarrayBoundedMax(vector<int>& A, int L, int R) {
        return find(A,R)-find(A,L-1);
    }
};
