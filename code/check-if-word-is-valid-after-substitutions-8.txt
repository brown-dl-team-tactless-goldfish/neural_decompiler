class Solution {
public:
    bool isValid(string s) {
        stack<char> stk;
        for (auto& c : s){
            if (c == 'a'){
                stk.push('a');
            }
            else if (c == 'b'){
              if (stk.empty() || stk.top() != 'a') return false;
                stk.pop();
                stk.push('b');
            }
            else{
                if (stk.empty() || stk.top() != 'b') return false;
                stk.pop();
            }
        }
        return stk.empty();
    }
};
