char* optimalDivision(int* nums, int numsSize) {
    char *ret=(char*)malloc(sizeof(char)*1000);
    int j=0;
    if(numsSize==1){
        j+=sprintf(ret+j,"%d",nums[0]);
        return ret;
    }
    if(numsSize==2){
        j+=sprintf(ret+j,"%d/%d",nums[0],nums[1]);
        return ret;
    }
    j=sprintf(ret+j,"%d/(",nums[0]);
    for(int i=1;i<numsSize;i++){
        j+=sprintf(ret+j,"%d/",nums[i]);
    }
    ret[j-1]=')';
    ret[j]='\0';
    return ret;
}
