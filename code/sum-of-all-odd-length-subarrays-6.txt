int sumOddLengthSubarrays(int* arr, int arrSize)
{
    int i,j,k,ans=0;
    for(i=0;i<arrSize;i++)
    {
        for(j=i;j<arrSize;j++)
        {
            for(k=i;k<=j;k++)
            {
                if((i+j)%2==0)
                    ans=ans+arr[k];
            }
        }
    }
    return ans;
}
