bool detectCapitalUse(char * word){
    int len=strlen(word);
    int count_big=0,count_small=0;
    for(int i=0;i<len;i++){
        if(word[i]>='A'&&word[i]<='Z')
            count_big++;
        else if(word[i]>='a'&&word[i]<='z')
            count_small++;
    }
    if(len==count_big||len==count_small) return true;
    else if(count_big==1&&word[0]>='A'&&word[0]<='Z') return true;
    else return false;
}
