int countVowelStrings(int n){
    int array[5]={1,1,1,1,1}, arr[5]={0}, i=0, sum;
    while(i!=n){
        sum=0;
        for(int j=0; j<5; j++)
            sum+=array[j];
        arr[0]=sum;
        for(int j=0; j<4; j++)
            arr[j+1]=arr[j]-array[j];
        for(int j=0; j<5; j++)
            array[j]=arr[j];
        i++;
    }
    return sum;
}
