class Solution {
public:
    bool canReach(string s, int minJump, int maxJump) {
        int n = s.size();
        vector<int> a(n);
        a[0] = 1;
        for(int i = 0; i < n; i++) {
            if(s[i] == '0' && a[i] > 0) {
                if(i == n - 1)
                    return true;
                for(int j = min(n - 1, i + maxJump); j >= i + minJump && a[j] == 0; j--) {
                    a[j]++;
                }
            }
        }
        return false;
    }
};
