bool checkPowersOfThree(int n) {
           int sum=0;
           int temp=n;
        vector<int>v(20,0);
         while(temp)
         {
             double power=log2(temp)/log2(3);
             cout<<power<<endl;
             power=(int)power;
             if(v[power]==1)
                 return false;
             v[power]=1;
             temp-=pow(3,power);
         }
        return true;
    }

Time Complexity: O(klogN)
Space Complexity: O(1)
