struct ListNode* middleNode(struct ListNode* head){
    if(head==NULL){
        return NULL;
    }
    
    int Count(struct ListNode *p)
{
    int l=0;
    while(p!=NULL)
    {
        l++;
        p = p->next;
    }

    return l;
}
    int x=Count(head);
    int i;
    for(i=0;i<x/2;i++){
        head=head->next;
    }
    
    return head;
   
    
}
