void helper(struct TreeNode* root, struct TreeNode** head, struct TreeNode** tail){
    struct TreeNode *hl;    // head-left
    struct TreeNode *tl;    // tail-left
    struct TreeNode *hr;    // head-right
    struct TreeNode *tr;    // tail-right
    
    if(root->left==NULL && root->right==NULL){
        *head = root;
        *tail = root;
    }else if(root->left==NULL){
        helper(root->right, &hr, &tr);
        
        root->right = hr;
        
        *head = root;
        *tail = tr;
    }else if(root->right==NULL){
        helper(root->left, &hl, &tl);
        
        tl->right = root;
        root->left = NULL;
        
        *head = hl;
        *tail = root;
    }else{
        helper(root->left, &hl, &tl);
        helper(root->right, &hr, &tr);
        
        tl->right = root;
        root->left = NULL;
        root->right = hr;
        
        *head = hl;
        *tail = tr;
    }
}

struct TreeNode* increasingBST(struct TreeNode* root){
    struct TreeNode *head;
    struct TreeNode *tail;
    helper(root, &head, &tail);
    return head;
}
