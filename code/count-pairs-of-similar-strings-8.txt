class Solution {
public:
    int similarPairs(vector<string>& words) {
        vector<vector<bool>> p(words.size(), vector<bool>(26, false));
        for (int i = 0; i < words.size(); ++i) {
            auto w = words[i];
            for (const auto& c : w) p[i][c-'a'] = true;
        }
        int cnt = 0;
        for (int i = 0; i < p.size(); ++i) 
            for (int j = i+1; j < p.size(); ++j)
                cnt += p[i] == p[j];
        return cnt;      
    }
};
