int* partitionLabels(char * S, int* returnSize){
    int endPostion[26] = {0};
    *returnSize = 0;
    int *result = malloc(sizeof(int)*26);
    for (int i=0; S[i] != 0; i++) {
        endPostion[S[i]-'a'] = i;
    }
    int end = 0, prev=0;
    for (int i=0;S[i] != 0;i++) {
        end = end>endPostion[S[i]-'a']?end:endPostion[S[i]-'a'];
        if (end == i) {
            result[(*returnSize)++] = end+1-prev;
            prev = end+1;
        }
    }
    return result;
}
