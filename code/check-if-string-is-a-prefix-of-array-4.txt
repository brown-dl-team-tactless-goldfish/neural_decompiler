public class Solution {
    public bool IsPrefixString(string s, string[] words) {
        StringBuilder builder = new StringBuilder();
        
        foreach (var item in words)
        {
            builder.Append(item);
            
            if (s.Length < builder.Length)
                return false;
            else if (s == builder.ToString())
                return true;
        }
        
        return false;
    }
}
