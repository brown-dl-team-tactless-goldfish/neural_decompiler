int countVowelStrings(int n){
    int arr[5] = {1, 1, 1, 1, 1};
    for (int i = 1; i < n; ++i) {
        arr[0] = arr[0];
        arr[1] = arr[0] + arr[1];
        arr[2] = arr[1] + arr[2];
        arr[3] = arr[2] + arr[3];
        arr[4] = arr[3] + arr[4];
    }
    int sum = 0;
    for (int i = 0; i < 5; ++i) {
        sum += arr[i];
    }
    return sum;
}
