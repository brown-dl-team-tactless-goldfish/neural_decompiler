char * reverseParentheses(char * s){
    char *st[100], *p = s--, tp = -1, *r = p, *u = p - 1, *b = p, *e = p;
    while (*++s && (*s == '(' ? st[++tp] = s : *s == ')' ? b = st[tp--], e = s - 1 : 1))
        while (b < e && (*b ^= *e, *e ^= *b, *b++ ^= *e--));
    while ((*++u && (isalpha(*u) ? *r++ = *u : 1)) || (*r = 0));
    return p;
}
