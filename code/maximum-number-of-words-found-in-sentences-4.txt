int sentence_length(char* sentence) {
    int k = 0;
    for (int i = 0; sentence[i] != '\0'; i++) {
        if (sentence[i] == ' ') k += 1;
    }
    return k > 0 ? k + 1 : 1;
}

int mostWordsFound(char ** sentences, int sentencesSize){
    int j = 0;
    for (int i = 0; i < sentencesSize; i++) {
        int k = sentence_length(sentences[i]);
        if (k > j) j = k;
    }
    return j; 
}
