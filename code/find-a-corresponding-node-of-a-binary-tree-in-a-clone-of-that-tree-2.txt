/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    TreeNode* getTargetCopy(TreeNode* original, TreeNode* cloned, TreeNode* target) {
        trav(original, cloned, target);
        return c;
    }

private:
    TreeNode* c = nullptr;
    
    void trav(TreeNode* og, TreeNode* cl, TreeNode* target){
        if(og == target){
            c = cl;
            return;
        }
        if(og->left != nullptr)
            trav(og->left, cl->left, target);
        if(og->right != nullptr)
            trav(og->right, cl->right, target);
        return;
    }
};
