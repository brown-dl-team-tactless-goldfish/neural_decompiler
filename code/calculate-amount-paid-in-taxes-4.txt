class Solution:
    def calculateTax(self, brackets: List[List[int]], income: int) -> float:
        n = len(brackets)
        print ("n = ", n)
        
        res = 0
        remainIncome = income
        
        for curIdx in range(n):
            
            print ("curIdx = ", curIdx, " remainIncome = ", remainIncome)
            moneyInThisBacket = 0
            if (curIdx == 0):
                moneyInThisBacket = min (brackets[curIdx][0], remainIncome)
            else:
                moneyInThisBacket = min (brackets[curIdx][0] - brackets[curIdx-1][0], remainIncome)
            
            print ("moneyInThisBacket = ", moneyInThisBacket)
            
            if (remainIncome <= 0):
                break
            
            res += moneyInThisBacket * brackets[curIdx][1] / 100
            
            remainIncome -= moneyInThisBacket
            
            
        return res***
