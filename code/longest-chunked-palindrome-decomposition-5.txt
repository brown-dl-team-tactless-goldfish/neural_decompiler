public class Solution 
{
    public int LongestDecomposition(string text) 
    {
        if(string.IsNullOrEmpty(text)) return 0;
        int result = 1;
        for(int i = 1; i <= text.Length / 2; i++)
        {
            string prefix = text.Substring(0, i), suffix = text.Substring(text.Length - i);
            if(prefix == suffix)
            {
                string mid =  text.Substring(i, text.Length - (2 * i));
                result = Math.Max(result, 2 + LongestDecomposition(mid));
            }
        }
        
        return  result;
    }
}
