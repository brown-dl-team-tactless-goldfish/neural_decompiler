void duplicateZeros(int* arr, int arrSize){
	int * res = (int*)malloc(arrSize * sizeof(int));
	register unsigned int i, j, x;
	for(i = j = 0; i < arrSize; i++) {
		if(j < arrSize)
            res[j] = arr[i];
		if(arr[i] == 0 && j+1 < arrSize) {
			j++;
			res[j] = arr[i];
		}
		j++;
	}
	for(x = 0; x < arrSize; x++)
		arr[x] = res[x];
}
