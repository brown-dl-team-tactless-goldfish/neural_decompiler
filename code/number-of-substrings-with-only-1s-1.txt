#define capacity 1000000007;
int numSub(char * s){
	int* count = (int*)calloc(strlen(s)+1,sizeof(int));
    int continuous = 0;
    long long sum = 0;
    for(int i=0;i<strlen(s);i++){
        if(s[i]=='1'){
            continuous++;
            count[continuous]++;
        }else if(s[i]=='0'){
            continuous=0;
        }
    }
    for(int i=strlen(s);i>=1;i--){
       sum += i*count[i];
    }
    
    return sum%capacity;
    
}
