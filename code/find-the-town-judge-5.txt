int findJudge(int n, int** trust, int trustSize, int* trustColSize){
    if (n == 1) {
        return 1;
    }
    int a[n+1];
    int b[n+1];
    memset(a, 0, sizeof(a));
    memset(b, 0, sizeof(b));
    
    for (int i=0; i<trustSize; i++) {
        a[trust[i][1]]++;
        b[trust[i][0]]++;
    }
    for (int i=0; i<=n; i++) {
        if ((a[i] == n-1) && (b[i] == 0)) {
            return i;
        }
    }  
    
    return -1;
}
