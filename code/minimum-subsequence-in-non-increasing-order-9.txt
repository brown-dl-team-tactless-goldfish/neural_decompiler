class Solution {
public:
    using VI = vector<int>;
    VI minSubsequence(VI& A, VI ans = {}) {
        sort(A.rbegin(), A.rend());
        auto total = accumulate(A.begin(), A.end(), 0),
		     half = total / 2;
        for (auto i{ 0 }, sum{ 0 }; i < A.size() && sum <= half; ++i)
            sum += A[i], ans.push_back(A[i]);
        return ans;
    }
};
