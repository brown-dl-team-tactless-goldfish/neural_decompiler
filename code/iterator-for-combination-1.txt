class CombinationIterator:

    def __init__(self, characters: str, combinationLength: int):
        self.data = []
        
        n = len(characters)
        
        def helper(curIdx, curSet):
            if (len(curSet) == combinationLength):
                self.data.append(''.join(curSet))
                
            for idx in range(curIdx, n):
                helper(idx+1, curSet+[characters[idx]])
                
        helper(0, [])
        
        print ("self.data = ", self.data)
        
        self.idx = 0

    def next(self) -> str:
        if (self.idx < len(self.data)):
            res = self.data[self.idx]
            self.idx += 1
            return res
        
    def hasNext(self) -> bool:
        return (self.idx < len(self.data))
