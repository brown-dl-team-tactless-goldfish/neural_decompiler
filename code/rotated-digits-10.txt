class Solution:
    def rotatedDigits(self, N):
        res = 0

        for i in range(1,N+1):
            t, c = i, 0

            while t > 0:
                if t%10 in [2,5,6,9]:
                    c += 1
                elif t%10 in [3,4,7]:
                    c = 0
                    break
                    
                t //= 10

            if c:
                res += 1

        return res
