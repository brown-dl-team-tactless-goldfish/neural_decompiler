bool checkRecord(char * s){
    int n = strlen(s);
    int consecuLate = 0;
    int absent = 0;
    for(int i = 0; i < n; i++){ 
        if(s[i] == 'P'){
            consecuLate = 0;
        }
        else if(s[i] == 'A'){
            absent++;
            consecuLate = 0;
            if(absent > 1)
                return false ;
        }
        else{// 'L'
            consecuLate++;
            if(consecuLate == 3)
                return false;
        }
    }
    return true;
}
