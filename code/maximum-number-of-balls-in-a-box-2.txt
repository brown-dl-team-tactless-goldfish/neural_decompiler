int box[50];

int ball_to_box(int ball)
{
	int box = 0;
	while(ball)
	{
		box += (ball % 10);
		ball /= 10;
	}
	return box;
}

int countBalls(int lowLimit, int highLimit){
	int i, max_box = 0;
	memset(box, 0, sizeof(box));
	for(i = lowLimit; i <= highLimit; i++)
	{
		box[ball_to_box(i)]++;
		if(box[ball_to_box(i)] > max_box)
			max_box = box[ball_to_box(i)];
	}
	
	
	return max_box;
}
