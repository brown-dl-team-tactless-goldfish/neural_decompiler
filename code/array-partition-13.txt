

int arrayPairSum(int* nums, int numsSize){
    int list [20001] = {0};
    
    for (int i = 0; i < numsSize; i++){
        list[nums[i]+10000]+=1;
    }
    
    int n = 0;
    for (int i = 0, k = 0; i < 20001; i++){
        while (list[i] > 0){
            if (k % 2 == 0){
                n = n + i - 10000;
            }
            k++;
            list[i]--;    
        }
    }
    return n;
}
