int findPoisonedDuration(int* timeSeries, int timeSeriesSize, int duration){
    if(timeSeriesSize==0)
        return 0;
    int count=0;
    for(int i=1;i<timeSeriesSize;i++)
    {
        if(timeSeries[i]-timeSeries[i-1]>=duration)
            count+=duration;
        else 
            count+=timeSeries[i]-timeSeries[i-1];
    }
    return count+duration;
}
