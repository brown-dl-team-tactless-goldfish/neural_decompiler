def calcEquation(self, equations, values, queries):
    dict1 = defaultdict(dict)
    
    for (a, b), c in zip(equations, values):
        dict1[a][a] = dict1[b][b] = 1
        dict1[a][b] = c
        dict1[b][a] = 1/c
        
        
    for i in dict1:
        for j in dict1[i]:
            for k in dict1[i]:
                dict1[j][k] = dict1[j][i]*dict1[i][k]
                
    return [dict1[a].get(b,-1) for a, b in queries]
