    vector<int> dp(366,0);
    
    for(int i=1; i<366; i++)
    {
        if(st.find(i) == st.end())
        {
            dp[i] = dp[i-1];
        }
        
      else 
      {
         int a = dp[i-1] + costs[0];
          
         int b = dp[max(0,i-7)]+costs[1];
          
         int c = dp[max(0,i-30)] + costs[2];
          
         dp[i] = min({a,b,c});
       
    }
}
    
    return dp[365];
}
