class Solution {
public:
    bool isPossibleDivide(vector<int>& nums, int k) {
        int n=nums.size();
        if(n%k)return false;
        map<int,int> m;
        for(auto it:nums)m[it]++;
        for(auto it:m){
            int freq=it.second;
            if(freq!=0){
                for(int i=0;i<k;i++)
                    if(m[it.first+i]<freq)return false;
                    else m[it.first+i]-=freq;
            }
        }
        return true;
    }
};
