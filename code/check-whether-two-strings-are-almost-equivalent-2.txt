bool checkAlmostEquivalent(char * word1, char * word2){
    int count1[26] = {0}, count2[26] = {0};
    int i;
    
    for(i = 0; i < strlen(word1); i++)
        count1[word1[i] - 'a']++;
    for(i = 0; i < strlen(word2); i++)
        count2[word2[i] - 'a']++;
    
    for(i = 0; i < 26; i++)
        if(abs(count1[i] - count2[i]) > 3)
            return 0;
    
    return 1;
}
