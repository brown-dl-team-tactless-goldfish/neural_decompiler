class Solution {
public:
    vector<int> minSubsequence(vector<int>& nums) {
        int n = nums.size();
        int sum = 0;
        for (int i = 0; i < n; i++)
            sum += nums[i];
        sort(nums.begin(), nums.end(), greater<int>());
        vector<int> ans;
        int partial = 0;
        for (int x: nums) {
            partial += x;
            ans.push_back(x);
            if (partial > sum - partial)
                break;
        }
        return ans;
    }
};
