int* smallerNumbersThanCurrent(int* nums, int numsSize, int* returnSize){
    *returnSize = numsSize;
    
    int i,j, *smallers, k;
    smallers = (int*)malloc(sizeof(int)*numsSize);
    for(i=0;i<numsSize;i++)
    {
        for(k=0,j=0;j<numsSize;j++)
           if(nums[i]>nums[j]&&i!=j)
               k++;
        smallers[i]=k;
    }            
    return smallers;
}
