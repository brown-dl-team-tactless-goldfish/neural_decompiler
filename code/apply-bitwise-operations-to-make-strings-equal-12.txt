class Solution {
public:
    bool makeStringsEqual(string s, string target) {
        int cnt1=0;
        int cnt2=0;
        for(int i=0;i<s.size();i++){
            if(s[i]=='1')cnt1++;
            if(target[i]=='1')cnt2++;
        }
        if(cnt1>=1 && cnt2==0)return false;
        if(cnt1==0 && cnt2>=1)return false;
        return true;
    }
};
