char * thousandSeparator(int n){
    char* ans = (char*)calloc(20, sizeof(char));
    int ansIdx = 0, digitCnt = 0;
    do
    {
        ans[ansIdx++] = (n % 10) + '0';
        n /= 10;
        digitCnt++;
        if(n && !(digitCnt % 3)) 
			ans[ansIdx++] = '.';
    } while(n);
	/* reverse string */
    int left = 0, right = ansIdx-1;
    while(left<right){
        char c=ans[left];
        ans[left++]=ans[right];
        ans[right--]=c;
    }
    ans[ansIdx] = '\0';
    return ans;
}
