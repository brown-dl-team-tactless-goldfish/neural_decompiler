class Solution {
public:
    long long beautifulSubarrays(vector<int>& nums) {
        long long ans=0;
        unordered_map<int,int> mp;
        long long xr = 0;
        for(int i=0;i<nums.size();i++){
            xr ^= nums[i];
            if(mp.find(xr)!=mp.end()){
                ans += mp[xr];
                mp[xr] += 1;
            }else{
                mp[xr] = 1;
            }
        }
        if(mp.find(0) != mp.end()) ans += mp[0];
        return ans;
    }
};
