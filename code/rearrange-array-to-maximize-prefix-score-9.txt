class Solution {
public:
    int maxScore(vector<int>& nums) {
        sort(nums.begin(),nums.end());
        reverse(nums.begin(),nums.end());
        vector <long long> psum(nums.size());
        psum[0] = nums[0];
        for(int i=1;i<nums.size();i++)
            psum[i] = nums[i] + psum[i-1];
        int ans = 0;
        for(auto &it:psum){
            if(it > 0)
                ans++;
            else
                break;
        }
        return ans;
    }
};
