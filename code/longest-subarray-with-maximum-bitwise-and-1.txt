class Solution {
public:
    int longestSubarray(vector<int>& nums) {
        int m = *max_element(nums.begin(),nums.end());
        int ans = 0,c = 0;
        for(int i=0;i<nums.size();i++){
            if(nums[i]==m) c++;
            else{
                ans = max(ans,c);
                c = 0;
            }  
        }
        ans = max(ans,c);
        return ans;
    }
};
