class Solution {
public:
    
    
    bool containsPattern(vector<int>& arr, int m, int k) {
        int c=0;
		int ans=k*m-m;
        for(int i=0;i+m<arr.size();i++){
            if(arr[i]!=arr[i+m])c=0;
            if(arr[i]==arr[i+m]){
			   c++;
		   }
            if(c==ans){
                return true;
            }
        }
        return false;
    }
};
