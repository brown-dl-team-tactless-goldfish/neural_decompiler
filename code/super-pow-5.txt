int superPow(int a, vector<int>& b) {
    int res = 1, mul = a % 1337;
    for (int i = b.size() - 1; i >= 0; --i) {
        int new_mul = 1;
        for (auto j = 0; j <= 9; ++j) {
            if (j < b[i])
                res = (res * mul) % 1337;
            new_mul = (new_mul * mul) % 1337;
        }
        mul = new_mul;
    }
    return res;
}
