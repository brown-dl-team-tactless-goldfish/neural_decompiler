class Solution {
public:
    unordered_map<int,int>m;
    int days(int n)
    {
        if(n<=1)
            return 1;
        if(m.find(n)!=m.end())
            return m[n];
        int res=min((n%3)+1+days(n/3),(n%2)+1+days(n/2));
        return m[n]=res;
    }
    int minDays(int n) 
    {
        return days(n);
    }
};
