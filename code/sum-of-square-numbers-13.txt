class Solution:
    def judgeSquareSum(self, N: int) -> bool:
        print("N = ", N)
        candidates = []
        dp = {}

        for i in range(0, N//2 + 10):
            if (i*i > N):
                break
            dp[i*i] = i

        # print("end dp = ", dp)

        for k, v in dp.items():
            target = N - k
            # print("k = ", k, " v = ", v, " target = ", target)
            if (target in dp):
                return True

        return False
