int countHomogenous(char * s){
	long int answer = 0;
	int tmp = 1;
	int i;
	
	for(i = 0; i < strlen(s); i++)
	{
		if(i == (strlen(s) - 1))
		{
			answer = answer + tmp;
			break;
		}
		if(s[i] == s[i+1])
		{
			answer = answer + tmp;
			tmp = tmp + 1;
		}
		else
		{
			answer = answer + tmp;
			tmp = 1;
		}
	}
	return answer%1000000007;
}
