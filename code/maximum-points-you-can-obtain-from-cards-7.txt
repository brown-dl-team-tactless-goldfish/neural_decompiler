int maxScore(int* cardPoints, int cardPointsSize, int k){
    int min;
    int totalSum=0,curSum;
    int i,j;
    
    for(i=0;i<cardPointsSize-k;i++) //Find the sum of the first k elements
        totalSum+=cardPoints[i];
    min=curSum=totalSum;
    
    for(i=1,j=cardPointsSize-k;j<cardPointsSize;i++,j++){ 
        totalSum+=cardPoints[j];
        curSum=curSum-cardPoints[i-1]+cardPoints[j];
        if(curSum<min)
            min=curSum;
    }
    return totalSum-min;
}
