int minFlipsMonoIncr(char* S) {
    int count=0;
    int len=strlen(S);
    int* array=(int*)malloc(len*sizeof(int));
    for(int i=0;i<len;i++){
        if(S[i]=='1'){
            count++;
        }
        array[i]=count;
    }
    int min=len-array[len-1]<array[len-1]?len-array[len-1]:array[len-1];
    for(int i=0;i<len-1;i++){
        int temp=2*array[i]+ (len-i-1)-array[len-1];
        if(min>temp)
            min=temp;
    }
    return min;
}
