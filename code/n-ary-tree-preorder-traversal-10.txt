/**
 * Definition for a Node.
 * struct Node {
 *     int val;
 *     int numChildren;
 *     struct Node** children;
 * };
 */

/**
 * Note: The returned array must be malloced, assume caller calls free().
 */

static void toarray(struct Node* root, int* retVar, int* returnSize) {

    if (root == NULL) return;
    
    retVar[(*returnSize)++] = root->val;
    for (int i = 0; i < root->numChildren; i++)
    {
         toarray(root->children[i], retVar, returnSize);   
    }
}

int* preorder(struct Node* root, int* returnSize) {
    
    int * retVar = (int *) malloc(sizeof(int*) * 100000);
    
    *returnSize = 0;
    
    toarray(root, retVar, returnSize);
    
    return retVar;
}
