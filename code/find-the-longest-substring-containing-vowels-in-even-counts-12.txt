    int findTheLongestSubstring(string s) {
        unordered_map<int, int> map{{0, -1}};
        int res = 0;
        for(int i = 0, v = 0; i < s.length(); i++) {
            v ^= 1 << string("aoeiu").find(s[i]) + 1 >> 1;
            if(!map.count(v)) map[v] = i;
            else res = max(res, i - map[v]);
        }
        return res;
    }
