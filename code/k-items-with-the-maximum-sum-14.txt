class Solution 
{
public:
    int kItemsWithMaximumSum(int numOnes, int numZeros, int numNegOnes, int k) 
    {
        int res = 0;
        int count = 0;

        count = std::min(numOnes, k);
        res += count;
        k -= count;
        if (k <= 0)
            return res;

        count = std::min(numZeros, k);
        k -= count;
        if (k <= 0)
            return res;
        
        res -= k;
        return res;
    }
};
