public class Solution {
    public void WiggleSort(int[] nums) {
        int n = nums.Length;
        var sorted = nums.OrderBy(x=>x).ToArray();
        for(int i = 0;i<n/2;i++)
        {
            int index = (n%2==0? n/2-1-i:n/2-i);
            nums[i*2] = sorted[index];
            nums[i*2+1] = sorted[n-i-1];
        }
        if(n%2==1)
        {
            nums[n-1] = sorted[0];
        }
    }
}
