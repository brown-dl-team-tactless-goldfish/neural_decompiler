int maxTurbulenceSize(int* arr, int arrSize){
    int res = 1, mask = 0, curr = 1, cmp;
    while(--arrSize){
        cmp = (*arr^mask)-(*(arr+1)^mask);
        mask = ~mask;
        if(cmp>0){
            res = res>++curr?res:curr;    
            arr++;
        }else{
            curr = 1;
            !cmp?arr++:arrSize++;
        }
    }
    return res;
}
