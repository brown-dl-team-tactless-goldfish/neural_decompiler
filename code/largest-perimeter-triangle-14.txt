class Solution:
    def largestPerimeter(self, nums: List[int]) -> int:
        def isTriangle(a,b,c):
            return (b+c>a)
        n = len(nums)
        nums.sort(reverse =True)
        i = 0
        while i < n-2:
            if isTriangle(nums[i],nums[i+1],nums[i+2]):
                return nums[i]+nums[i+1]+nums[i+2]
            else:
                i += 1
        return 0
