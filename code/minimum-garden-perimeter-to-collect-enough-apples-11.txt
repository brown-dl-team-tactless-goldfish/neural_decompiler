class Solution {
public:
    long long get (long long i)
    {
        return i*(i+1)/2;    
    }
    
    long long minimumPerimeter(long long neededApples) {
        long long sum=0;
        long long i=1;
        while(sum<neededApples)
        {
            if(i==1)
                sum+=12;
            else
            {
                sum+=(i*12);
                sum+=(get((i*2) -1) -get(i))*8;
            }
            i++;
        }
        return (i-1)*8;
        
    }
};
