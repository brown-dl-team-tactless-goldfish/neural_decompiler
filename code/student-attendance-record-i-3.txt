// assume everything is in uppercase
// if not UP, convert to uppercase
bool checkRecord(char * s)
{    
    int len=strlen(s);
    if(len==0)
        return true;
    
    int absent = 0;
    int consecutiveLate = 0;
    
    for(int i=0;i<len;i++)
    {
        if(toupper(s[i]=='A'))
        {
            absent++;
            consecutiveLate = 0;
            if(absent>1)
            {
                return false;
            }
        }
        else if(toupper(s[i]=='L'))
        {
            consecutiveLate++;
            if(consecutiveLate>2)
            {
                return false;
            }
        }
        else
        {
            consecutiveLate = 0;
        }
    }
    
    return true;
}
