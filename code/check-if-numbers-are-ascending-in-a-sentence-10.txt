public class Solution {
    public bool AreNumbersAscending(string s) {
        int previous =0, curr =0;
        var sArray = s.Trim().Split(' ');
       for(int i = 0 ;i < sArray.Length ; i++)
        {
           var input = sArray[i];
           StringBuilder sb = new StringBuilder();
           bool isDigit = false;
           foreach(char c in input)
           {
              if(Char.IsDigit(c))
              {
                  isDigit= true;
                  sb.Append(c);
              }
               else
               {
                   isDigit= false;
               }
           }
            if(isDigit)
            {
                
                curr = Convert.ToInt32(sb.ToString());
                if(previous < curr)
                {
                    previous = curr;
                }
                else
                {
                    return false;
                }
            }
            }
        return true;
        }
         
    }
