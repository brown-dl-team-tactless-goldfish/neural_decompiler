public class Solution
{
    public int MinimumRecolors(string blocks, int k)
    {
        var min = Int32.MaxValue;
        for (int i = 0; i <= blocks.Length - k; i++)
        {
            var word = blocks.Substring(i, k);
            var length = word.Count(x => x == 'W');
            min = Math.Min(min, length);
        }
        return min;
    }
}
