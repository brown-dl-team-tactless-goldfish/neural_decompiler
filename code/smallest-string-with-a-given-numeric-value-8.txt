char * getSmallestString(int n, int k){
    char* ans = malloc(sizeof(char)*(n+1));
    ans[n]='\0';
    while(n>0){
        if(k-n+1>26){
            ans[n-1] = 26+96;
            k-=26;
            n--;
        }
        else if(k-n+1>1){
            ans[n-1] = k-n+1+96;
            k-= k-n+1;
            n--;
        }
        else{
            ans[n-1]= 97;
            k--;
            n--;
        }
    }
    return ans;

}
