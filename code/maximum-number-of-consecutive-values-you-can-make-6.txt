class Solution {
public:
    int getMaximumConsecutive(vector<int>& coins) {
        int n = coins.size();
        sort(coins.begin(), coins.end());
        int prev = 0;
        for (int coin : coins) {
            if (coin<=prev+1) {
                prev += coin;
            }
            //cout << "prev = " << prev << endl;
        }
        return prev+1;
    }
};
