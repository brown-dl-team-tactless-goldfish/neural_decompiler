bool containsPattern(int* arr, int arrSize, int m, int k){
    int n = arrSize;
    for(int i = 0; i < n ; i++){
        for(int j = 1; j < k; j++){
            int p = i + m*j;
            for(int l = 0; l  < m; l++){
                if((p+l) >= n)
                    return false;
                if(arr[i+l] != arr[p+l])
                    goto next;
            }
        }
        return true;
        next:
            continue;
    }
    return false;
}
