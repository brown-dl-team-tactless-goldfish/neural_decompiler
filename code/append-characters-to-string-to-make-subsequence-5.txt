class Solution {
public:
    int appendCharacters(string s, string t) {
        
        int j=0,cur=0;
        
        for(int i=0;i<t.length() && j<s.length();i++){
            while(j<s.length() && t[i] != s[j]){
                j++;
            }
            if(j<s.length()){
                cur++; 
                j++;
            }
        }
        return t.length()-cur;
    }
};
