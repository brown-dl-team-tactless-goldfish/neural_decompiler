

char * freqAlphabets(char * s){
    int i,p=0;
    int len=strlen(s);
    char *newstr = (char*)malloc(sizeof(char)*len+1);
    for(i=0;i<len;i++)
    {
        if(i<(len-2)&&s[i+2]=='#') {newstr[p++]=(s[i]-48)*10 + (s[i+1]-48) +96;
                         i+=2;}
        else
            newstr[p++]=s[i]+48;
    }
    newstr[p]='\0';
    return newstr;
}
