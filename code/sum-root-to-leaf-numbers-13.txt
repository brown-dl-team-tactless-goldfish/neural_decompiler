int preOrder(struct TreeNode* node, int value)
{
    if(!node) return 0;
    value = value * 10 + node->val;
    if(!node->left && !node->right)
        return value;
    else
        return preOrder(node->left, value) + preOrder(node->right, value);
}
int sumNumbers(struct TreeNode* root)
{
    return preOrder(root, 0);
}
