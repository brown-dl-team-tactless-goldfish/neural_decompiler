class Solution {
public:
    int maximumGain(string S, int X, int Y) {
        char a = 'a', b = 'b';
        int ans = 0, aStore = 0, bStore = 0;
        if (Y > X) swap(a,b), swap(X,Y);
        for (char c: S)
            if (c == a) aStore++;
            else if (c == b)
                if (aStore > 0) ans += X, aStore--;
                else bStore++;
            else ans += Y * min(aStore, bStore), aStore = bStore = 0;
        return ans + Y * min(aStore, bStore);
    }
};
