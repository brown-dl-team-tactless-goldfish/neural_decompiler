int* countBits(int n, int* returnSize) {
    *returnSize = n+1;
    int* arr = (int*) calloc((*returnSize), sizeof(int));
    int idx = 1;
    for(int i = 1; i <= n; i++) {
        if(i % 2 == 1) {
            arr[idx] = arr[idx-1] + 1;
        } else {
            int cnt = i / 2;
            arr[idx] = arr[cnt];
        }
        idx++;
    }
    return arr;
}
