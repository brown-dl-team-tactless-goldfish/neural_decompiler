int cmp(const void* a, const void* b){
    return *(int*)b - *(int*)a ;
}
int findLucky(int* arr, int arrSize){

    qsort(arr, arrSize, sizeof(int), cmp);
    int cn = 1;
    for(int i = 1; i < arrSize; i++){
        if(arr[i] == arr[i-1])
            cn++;
        else{
            if(cn == arr[i-1])
                return cn;
            cn = 1;
        }
    }
    if(cn == arr[arrSize-1])
        return cn;
    return -1;
}
