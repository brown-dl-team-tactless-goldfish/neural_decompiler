class Solution {
public:
    int minOperations(string s) {
        return min(count(s, /* startWith= */ 0), count(s, /* startWith= */ 1));
    }
    
private:
    static int count(string s, int startWith) {
        int count = 0;
        for (int i = 0; i < s.length(); i++) {
            char value = s[i];
            char expect = (startWith + i) % 2 == 0 ? '0' : '1';
            if (value != expect) {
                count++;
            }
        }
        return count;
    }
};
