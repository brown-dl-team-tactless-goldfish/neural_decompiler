int minimumSum(int num) {
    int *v = (int *) malloc(4 * sizeof(int));
    int i, j;
    for (i = 0; i < 4; i++) {
        v[i] = num % 10;
        num /= 10;
    }
    for (i = 1; i < 4; i++) {                         
        int x = v[i];                                    
        for (j = i - 1; j >= 0 && x < v[j]; j--)
            v[j + 1] = v[j];                      
        v[j + 1] = x;
    }
    int ans = 10 * (v[0] + v[1]) + (v[2] + v[3]);
    free(v);
    return ans;
}
