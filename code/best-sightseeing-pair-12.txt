class Solution {
public:
    int maxScoreSightseeingPair(vector<int>& A) {
        int max_{0};
        for(int i{1}, maxPrev{A[0]}, size{(int)A.size()}; i < size; ++i) {
            max_ = std::max(max_, A[i]-i+maxPrev);
            maxPrev = std::max(maxPrev, A[i]+i);
        }
        
        return max_;
    }
};
