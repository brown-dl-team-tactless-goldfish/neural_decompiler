class Solution:
    def simplifiedFractions(self, n: int) -> List[str]:
        self.res = []
        vals = []
        
        for i in range(1, n+1):
            for j in range(1, i):
                
                
                tmpVal = j / i
                if (tmpVal not in vals):
                    vals.append(tmpVal)
                    self.res.append(str(j) + "/" + str(i))
                
        # print ("self.res = ", self.res)
        
        return self.res
