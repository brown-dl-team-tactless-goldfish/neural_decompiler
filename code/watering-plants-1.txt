int wateringPlants(int* plants, int plantsSize, int capacity){
    int can = capacity;
    int steps = 0;
    for(int i = 0; i < plantsSize; i++){
        if(can < plants[i]){
            steps += 2 * i;
            can = capacity;
        }
        can -= plants[i];
        steps += 1;
    }
    return steps;
}
