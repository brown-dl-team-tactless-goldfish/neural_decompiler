

int sumOddLengthSubarrays(int* arr, int arrSize){
    int retV=0;
    for(int x=1;x<=arrSize;x+=2){ //x:loop內一次取幾個值 
        for(int y=0;(y+x)<=arrSize;y++){ //y:取的第一個值
            for(int z=x,o=y;z>0;z--){ //z:取x次 o:loop內取的index
                retV += arr[o];
                o++;
            }
        }                
    }
    return retV;
}
