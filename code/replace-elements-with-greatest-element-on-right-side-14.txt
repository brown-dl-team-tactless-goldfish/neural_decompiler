int* replaceElements(int* arr, int arrSize, int* returnSize){
    int* dst = (int*)malloc(arrSize*sizeof(int));
    int i,max=arr[arrSize-1];
    
    dst[arrSize-1] = -1;
    
    for(i = arrSize-2 ; i >= 0 ; i--){
        dst[i] = max;
        if(arr[i]>max) max = arr[i];
    }
    
    *returnSize = arrSize;
    return dst;
}
