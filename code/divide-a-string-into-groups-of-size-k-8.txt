class Solution {
public:
  vector<string> divideString(string s, int k, char fill) {
    vector<string>ans;
    
    int i = 0;
    while(i != s.size()){
      string t;
      for(int j = 0; j != k && i != s.size(); j++, i++ ) t += s[i];
      ans.push_back(t);
    }
    
    while(ans.back().size() != k) ans.back().push_back( fill);
    return ans;  
  }
};
