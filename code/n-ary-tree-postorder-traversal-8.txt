"""
# Definition for a Node.
class Node:
    def __init__(self, val=None, children=None):
        self.val = val
        self.children = children
"""
class Solution:
    def postorder(self, root: 'Node') -> List[int]:
        self.res = []
        
        def helper(curNode):
            if (curNode == None):
                return
            
            nbs = curNode.children
            
            for nb in nbs:
                helper(nb)
                
            self.res.append(curNode.val)
            
        helper(root)
        
        return self.res
