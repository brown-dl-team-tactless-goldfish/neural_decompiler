class Solution 
{
public:
    bool escapeGhosts(vector<vector<int>>& ghosts, vector<int>& target) 
    {
        int mn=INT_MAX;
        for(auto it: ghosts)
        {
            mn=min(mn,abs(target[0]-it[0])+abs(target[1]-it[1]));
        }
        int dist=abs(target[0])+abs(target[1]);
        return dist<mn;
    }
};
