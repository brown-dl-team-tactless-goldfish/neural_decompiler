#include<string.h>
int dfs(int *alp);
int numTilePossibilities(char * tiles)
{
  int len=strlen(tiles),key[26]={0},combo=0;
  for(int i=0;i<len;i++)
  {
    key[tiles[i] - 'A']++; // assigns count to key for each alphabet
  }
  combo=dfs(key);
  return combo;
}
int dfs(int* alp)
{
  int i=0,sum=0;
  for(i=0;i<26;i++)
  {
    if(alp[i]==0)
      continue;
    alp[i]--; //Decrements because we are using this alphabet
    sum++;  // Since we are using the single alphabet
    sum += dfs(alp); //Moves deeper in recursion will stop when all aplha bets are used
    alp[i]++;
  }
  return sum;
}
