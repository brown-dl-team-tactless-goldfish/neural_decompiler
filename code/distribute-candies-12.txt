class Solution:
    def distributeCandies(self, candies: List[int]) -> int:
        candies_num = len(candies)
        # print("candies_num = ", candies_num)



        candies_kinds = len(set(candies))
        # print("candies_kinds = ", candies_kinds)

        # if (candies_kinds <= candies_num/2):
        #     return candies_kinds

        res = candies_kinds
        while (res > 0):
            if (res <= candies_num/2):
                break
            res -= 1

        return res
