int maxProfit(int* prices, int pricesSize){
    if (pricesSize == 0) return 0;
    int min = prices[0];
    int max = 0;
    
    int i = 1;
    int price;
    int profit;
    for (; i < pricesSize; i++)
    {
        price = prices[i];
        profit = price - min;
        if (profit < 0)
        {
            min = price;
        }
        else
        {
            if (max < profit)
            {
                max = profit;
            }
        }
    }
    
    return max;
}
