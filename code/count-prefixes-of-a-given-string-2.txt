int countPrefixes(char ** words, int wordsSize, char * s){
    int flag,count=0;
    for(int i=0;i<wordsSize;i++){
        if(strlen(s)>= strlen(words[i])) flag=1;
        for(int j=0;j< strlen(words[i]) && flag==1;j++){
            if(words[i][j]!=s[j]) flag=0;
        }
        if(flag==0) count++;
    }
    return wordsSize-count;
}
