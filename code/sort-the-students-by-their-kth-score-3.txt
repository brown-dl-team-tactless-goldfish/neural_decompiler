/**
 * Return an array of arrays of size *returnSize.
 * The sizes of the arrays are returned as *returnColumnSizes array.
 * Note: Both returned array and *columnSizes array must be malloced, assume caller calls free().
 */
int** sortTheStudents(int** score, int scoreSize, int* scoreColSize, int k, int* returnSize, int** returnColumnSizes){

    unsigned char row=0,col=0,i=0;
    int* tempArr=malloc(sizeof(int)*(*scoreColSize));

    for(row=0;row<scoreSize-1;row++)    //Loop1 Sorting
    {
        for(i=row+1;i<scoreSize;i++)    //Loop2 Sorting
        {
            if(score[row][k] < score[i][k])     //Check Sort Condition
            {
                tempArr = score[row];
                score[row] = score[i];
                score[i] = tempArr;

                // for(col=0;col<(*scoreColSize);col++)
                // {
                //     tempArr[col] = score[row][col];
                // }
                // for(col=0;col<(*scoreColSize);col++)
                // {
                //     score[row][col] = score[i][col];
                // }
                // for(col=0;col<(*scoreColSize);col++)
                // {
                //     score[i][col] = tempArr[col];
                // }
            }
        }
    }

    *returnSize = scoreSize;
    *returnColumnSizes = scoreColSize;
    return score;
}
