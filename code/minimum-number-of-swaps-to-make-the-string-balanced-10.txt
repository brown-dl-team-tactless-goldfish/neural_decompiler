int minSwaps(string s) {
//cnt is used to track orphan '[' and bal is used to keep track of balanced parenthesis.
    int bal = 0, cnt = 0;
    
    for(auto c: s){
        
        if(bal == 0 && c == ']') ++cnt;
        else bal += c == '[' ? 1 : -1;
    }
    
	//result will always be (cnt + 1) / 2 because single swap is balancing 4 brackets.
    return (cnt+1)/2;
}
