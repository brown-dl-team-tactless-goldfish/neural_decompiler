/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     struct ListNode *next;
 * };
 */
bool hasCycle(struct ListNode *head) {
struct ListNode *ptr1;
struct ListNode *ptr2;
if ((head==NULL) || (head->next==NULL)) return false;
ptr1 = head->next;
ptr2 = head;
while (ptr1 != ptr2) {
    ptr1 = ptr1->next;
    if (ptr1==NULL) return false;
    ptr1 = ptr1->next;
    if (ptr1==NULL) return false;
    ptr2 = ptr2->next;
}
return true;
}
