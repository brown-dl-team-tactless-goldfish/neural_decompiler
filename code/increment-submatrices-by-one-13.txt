public class Solution 
{
    public int[][] RangeAddQueries(int n, int[][] queries)
    {
        int[][] result = new int[n][];
        for (int i = 0; i < n; i++)
            result[i] = new int[n];

        foreach (int[] query in queries)
            for (int i = query[0]; i <= query[2]; i++)
                for (int j = query[1]; j <= query[3]; j++)
                    result[i][j]++;

        return result;
    }
}
