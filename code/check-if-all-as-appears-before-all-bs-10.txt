checkString(char * s)

{

    int counta=0;
    int i=0;
    
    while(i<strlen(s))
    {
      if(s[i]=='a')
          counta++;
        i++;
    }
    i=0;
    while(i<strlen(s))
    {
      if(s[i]=='b'&&counta)
      {
          return false;
      }
      if(s[i]=='a')
      {
          counta--;
      }
        i++;
    }
    
  return true;  
    
}
