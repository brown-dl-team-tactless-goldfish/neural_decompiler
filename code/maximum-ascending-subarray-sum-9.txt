class Solution {
    public int maxAscendingSum(int[] nums) {
        int i=1,n = nums.length,s1 = nums[0],s2 = 0;
        for(;i<n;i++)
        {
            if(nums[i-1] < nums[i])
            {
                s1 += nums[i];
            }
            else
            {
                if(s1 > s2)
                    s2 = s1;
                s1 = nums[i];
            }
        }
        if(s1 >= s2)
            return s1;
        return s2;
    }
}
