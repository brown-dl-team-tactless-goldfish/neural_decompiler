int process(char* s, int len) {
    int ans = 0;
    for (int i = 0; i < len; i++) {
        double tmp = (s[i] - 'a') * pow(10.0, (double)(len - i - 1));
        ans += tmp;
    }
    return ans;
}

bool isSumEqual(char* firstWord, char* secondWord, char* targetWord) {
    return process(firstWord, strlen(firstWord)) + process(secondWord, strlen(secondWord)) == process(targetWord, strlen(targetWord));
}
