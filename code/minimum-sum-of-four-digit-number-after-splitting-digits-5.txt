int minimumSum(int num){
    int arr[4];
        for(int i=0;i<4;i++){
            arr[i]=num%10;
            num=num/10;
        }
    for(int i=0;i<4;i++){
        for(int j=0;j<4;j++){
            if(arr[i]<arr[j]){
                int temp=arr[i];
                arr[i]=arr[j];
                arr[j]=temp;
            }
        }
    }
    int sum1 =arr[0]*10+arr[2]+arr[1]*10+arr[3];
    int sum2 =arr[0]*10+arr[3]+arr[1]*10+arr[2];
    return sum1>sum2?sum2:sum1;
}
