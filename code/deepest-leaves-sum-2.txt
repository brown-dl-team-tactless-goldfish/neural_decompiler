/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     struct TreeNode *left;
 *     struct TreeNode *right;
 * };
 */
//calculate depth of tree
int depth(struct TreeNode *root) {
    
    int lh,rh;
    if(root == NULL)
        return 0;
    lh = depth(root->left);
    rh = depth(root->right);
    
    if(lh > rh)
        return lh+1;
    else
        return rh+1;
    
}

//calculate maximum depth
int sum(struct TreeNode * root, int index, int depth) {
    
    if(root == NULL)
        return 0;
    
    if(index == depth)
        return root->val;
    
    return sum(root->left,index+1,depth) + sum(root->right,index+1,depth);
}

int deepestLeavesSum(struct TreeNode* root){
    
    int height;

    if(root == NULL)
        return 0;
    
    height = depth(root);
    
    return sum(root,1,height);
    
    
}
