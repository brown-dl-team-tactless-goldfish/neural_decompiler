class Solution {
public:
  int numberOfArrays(vector<int>& d, int lower, int upper) {
    int mini = lower, maxi = lower, n = lower, dif = upper - lower;
    
    for(auto x: d){
      n += x;
      mini = min(mini, n);
      maxi = max(maxi, n);
      if(maxi - mini > dif) return 0;
    }
    
    return dif - (maxi-mini) + 1;
  }
};
