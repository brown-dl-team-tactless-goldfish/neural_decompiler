int n_word_in_sentence(char *sentence) {

	int n_spaces = 0;

	while (*sentence != '\0') { // avoid storing i

		if (*sentence == ' ') { n_spaces++; }
		sentence += 1;
	}
	
	return n_spaces + 1;
}

int mostWordsFound(char ** sentences, int sentencesSize){

	int max = 0;
	while (sentencesSize--) {  // avoid storing i
	
		int cur = n_word_in_sentence(sentences[sentencesSize]);
		if (cur > max) { max = cur; }
	}

	return max;
}
