class Solution {
public:
    long long subArrayRanges(vector<int>& nums) {
        
        long long ans = 0;
        int len = nums.size();
        for(int i=0; i<len; i++)
        {
            int mini = nums[i], maxi = nums[i];
            for(int j=i; j<len; j++)
            {
                mini = min(mini, nums[j]);
                maxi = max(maxi, nums[j]);
                ans+=(maxi-mini);
                
            }
        }
        return ans;
    }
};
