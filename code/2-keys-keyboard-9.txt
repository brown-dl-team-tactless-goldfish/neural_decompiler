#include<math.h>
int is_prime(int n)
{
    double sq = sqrt(n);
    for(int i = 2; i < sq + 1; ++i)
        if(n % i == 0)
            return 0;
    return 1;
}

int minSteps(int n) {
    int acc = 0;
    while(true) {
        if(n == 1) {
            acc += 0;
            break;
        } else if(n==2) {
            acc += 2;
            break;
        } else {
            if(is_prime(n)) {
                acc += n;
                break;
            } else {
                int divisor = 1;
                int to_test = n;
                // add prime factors
                for (int i = 2; i <= (to_test / 2) + 1; i++){
                    if(to_test % i == 0){
                        divisor = i;
                        while((to_test/=i) % i == 0);
                    }
                }
                acc += divisor;
                n /= divisor;
            }
        }
    }
    return acc;
}
