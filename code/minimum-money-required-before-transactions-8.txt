    public long MinimumMoney(int[][] transactions)
    {
        long result = 0; int last = 0;
        foreach (int[] t in transactions)
        {
            if (t[0] < t[1])
            {
                last = Math.Max(last, t[0]);
            }
            else
            {
                last = Math.Max(last, t[1]);
                result += Math.Max(t[0] - t[1], 0);
            }
        }
        return result + last;
    }
