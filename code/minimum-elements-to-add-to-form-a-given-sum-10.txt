class Solution {
public:
    #define ll long long 
    int minElements(vector<int>& nums, int limit, int goal) {
        ll currSum = accumulate(begin(nums),end(nums),0LL) ;
        ll targetToCover = abs(goal - currSum) ;
        
        
        ll ans = targetToCover / limit ;
        if(targetToCover % limit == 0) return ans ;
        return ++ans ;
        
    }
};
