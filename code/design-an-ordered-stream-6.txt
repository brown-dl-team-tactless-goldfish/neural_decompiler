public class OrderedStream {
    
    private int pointer = 0;
    private string[] arr;

    public OrderedStream(int n) {
        
        arr = new string[n];
    }
    
    public IList<string> Insert(int idKey, string value) {
        
        List<string> res = new List<string>();
        
        arr[idKey - 1] = value;
        
        while(pointer < arr.Length && arr[pointer] != null)
        {
            res.Add(arr[pointer]);
            pointer++;
        }
        
        return res;
    }
}
