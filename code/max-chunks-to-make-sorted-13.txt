class Solution {
public:
    using VI = vector<int>;
    using Set = unordered_set<int>;
    int maxChunksToSorted(VI& A, Set seen = {}, int cnt = 0) {
        int i = 0,
            j = 0,
            N = A.size();
        while (i < N && j < N) {
            while (seen.find(i) == seen.end())
                seen.insert(A[j++]);
            while (seen.find(i) != seen.end())
                ++i;
            cnt += i == j;
        }
        return cnt;
    }
};
