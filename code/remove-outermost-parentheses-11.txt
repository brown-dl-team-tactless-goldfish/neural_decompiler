char* removeOuterParentheses(char* S) {
	char	*dest;
	int	counter;
	int	j;

	j = 0;
	counter = 0;
	dest = (char *)malloc(sizeof(char) * strlen(S));
	for (int i = 0; S[i]; i++)
	{
		if ((S[i] == '(' && ++counter != 1) || (S[i] == ')' && --counter != 0))
			dest[j++] = S[i];
	}
	dest[j] = '\0';
	return (dest);
}
