void swap (int* a, int* b) {
    int temp = *a;
    *a = *b;
    *b = temp;
}

int removeDuplicates(int* nums, int numsSize){
    
    int i = 0;
    int j = 1;
    
    while (j < numsSize) {
        if (nums[i] == nums[j]) {
            nums[j] = INT_MAX;
            j++;
            continue;
        }
        swap (&nums[i+1], &nums[j]);
        i++;
        j++;
    }
    
    return i+1;

}
