#define ll long long 
class Solution {
public:
	inline ll sqsum(ll n){
        if(n==1) return 1;
		ll ans = (n*(n+1));
		ll sq = (2*n+1);
		return ans*sq/6;
	}
   	ll minimumPerimeter(ll neededApples) {
        ll ans =0;
        ll s =0;
        ll old=0;
        while(ans<neededApples){
            s++;
			ans = 12*sqsum(s);
        }
        return 8*s;
    }
};
