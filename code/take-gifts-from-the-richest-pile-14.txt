class Solution {
public:
    long long pickGifts(vector<int>& gifts, int k) {
        long long int cnt=0;
        priority_queue<int>q;
        for(int i=0; i<gifts.size(); i++){
            q.push(gifts[i]);
        }
        while(k--){
            int tp=q.top();
            tp=sqrt(tp);
            q.pop();
            q.push(tp);
        }
        while(!q.empty()){cnt+=q.top(); q.pop();}
        return cnt;
    }
};
