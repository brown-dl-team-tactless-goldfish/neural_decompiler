class NumArray {
public:
    vector<int> ans;
    NumArray(vector<int>& nums) {
        int prev = 0;
        ans.assign(nums.size() + 1, 0);
        int i = 0;
        for(; i < nums.size(); i++) {
            ans[i] = prev;
            prev += nums[i];
        }
        ans[i] = prev;
    }
    
    int sumRange(int i, int j) {
        return ans[j + 1] - ans[i]; 
    }
};
