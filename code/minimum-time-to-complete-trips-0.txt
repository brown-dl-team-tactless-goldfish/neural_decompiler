long long divide(int* nums, int numsSize, double k){
    long long ret =0 ;
    for(int i = 0; i < numsSize; i++){
        ret += k/nums[i];
    }
    return ret;
}

long long minimumTime(int* time, int timeSize, int totalTrips){

    long long left = 1, right = 1e14;
    long long mid;
    int ans = 0;
    while(left < right){
        mid = (left + right) / 2;
        if(divide(time, timeSize, mid) >= totalTrips){
            right = mid;
        }
        else
            left = mid + 1;
    }

    if(divide(time, timeSize, left) == totalTrips)
        return left;
    else 
        return right;
}
