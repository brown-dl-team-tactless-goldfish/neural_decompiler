int* nextGreaterElements(int* nums, int numsSize, int* returnSize){
    int stack[numsSize],top=-1,*toReturn=malloc(sizeof(int)*numsSize);
    *returnSize=numsSize;
	
	//firstRun to Find the Next Greater Element.
    for(int i=0;i<numsSize;i++){
        while(top!=-1 && nums[i]>nums[stack[top]]){
            toReturn[stack[top--]]=nums[i];
        }
        stack[++top]=i;
    }
	
	//SecondRun to go with circular fashion for the leftout elements
	
    for(int i=0;i<numsSize;i++){
       while(top!=-1 && nums[i]>nums[stack[top]]){
            toReturn[stack[top--]]=nums[i];
        }
    }
	
	//making all the rest elements (i.e) maximum element as -1
	
    while(top!=-1) toReturn[stack[top--]]=-1;
	
	//returning the array
	
    return toReturn;
}
