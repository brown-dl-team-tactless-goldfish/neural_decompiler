
int distanceBetweenBusStops(int* distance, int distanceSize, int start, int destination){
int c_distance=0;
int cc_distance=0;
int ptr;
if (start==destination) return 0;
ptr=start;
while ((ptr % (distanceSize))!=destination) c_distance += distance[(ptr++ % distanceSize)];
ptr=start+distanceSize;
while ((ptr % (distanceSize))!=destination) cc_distance += distance[(--ptr % distanceSize)];
if (c_distance < cc_distance) return c_distance;
  else return cc_distance;
}
