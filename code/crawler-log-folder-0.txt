int minOperations(char** logs, int logsSize) {
    int top = 0;
    for (int i = 0; i < logsSize; i++) {
        if (logs[i][0] == '.' && logs[i][1] == '.') {
            top = top <= 0 ? 0 : top - 1;
        } else if (logs[i][0] != '.') {
            top++;
        }
    }
    return top <= 0 ? 0 : top;
}
