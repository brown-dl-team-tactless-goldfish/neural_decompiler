int alternateDigitSum(int n)
{
    int count =log10(n)+1;
    int new[count];
    int temp=0;
    int i;
    for(i=count-1;i>=0;i--)
    {
        new[i]=n%10;
        n=n/10;
    }
    for(i=0;i<count;i++)
    {
        if(i%2==1)
        temp-=new[i];
        else
        temp+=new[i];
    }
    return temp;
}
