class Solution {
public:
    bool isBoomerang(vector<vector<int>>& p) 
    {
	    constexpr int X = 0, Y = 1;
		
        // all 3 points must be distinct
        if (p[0] == p[1] || p[0] == p[2])
            return false;
        
        const int A = p[0][X] - p[1][X];
        const int B = p[0][Y] - p[1][Y];
		
        auto C = [=](vector<int>& pt) { return A * pt[Y] - B * pt[X]; };
		
        return C(p[0]) != C(p[2]);
    }
};
