class Solution {
public:
    int numOfPairs(vector<string>& a, string target) {
        int cnt = 0 ;
        int n = a.size();
        for(int i=0;i<n;i++)
        {
            for(int j=0;j<n;j++)
            {
                if(i!=j and (a[i]+a[j])==target)
                    cnt++;
            }
        }
        return cnt;
    }
};
