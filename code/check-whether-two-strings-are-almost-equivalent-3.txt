
bool checkAlmostEquivalent(char * word1, char * word2){
    char table[26] = {0};
    for(;*word1;table[*word1++ - 'a']++)
        ;
    for(;*word2;table[*word2++ - 'a']--)
        ;
    for(int i = 0; i < 26;i++){
        if(table[i] > 3 || table[i] < - 3){
            return false;
        }
    }
    return true;
}
