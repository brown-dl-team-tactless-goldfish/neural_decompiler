int numberOfArithmeticSlices(int* nums, int n){
    if(n < 3) return 0;
    int cnt = 0;
    int ans = 0;
    int a,b;
    for(int j = 0; j < n-2; j++){
        a = nums[j+2] - nums[j+1];
        b = nums[j+1] - nums[j];
        if(a == b) {
            cnt++;
        }else{
            ans += cnt*(cnt + 1)/2;
            cnt = 0;
        }
    }
    ans += cnt*(cnt + 1)/2;
    return ans;
}
