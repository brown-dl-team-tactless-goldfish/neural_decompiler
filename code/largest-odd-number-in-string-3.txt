char * largestOddNumber(char * num){    
    int len = strlen(num);
    int num1 = num[len-1]-'0';
    if(num1%2==1)
        return num;
    int i;
    for(i=len-1;i>=0;i--)
    {
        int n=num[i]-'0';
        if(n%2==1)
            break;
    }
    if(i==-1)
        return "";
    
    char *arr = malloc((i+2)*sizeof(char));
    printf("%d",i);
    for(int j=i;j>=0;j--)
        arr[j]=num[j];
    arr[i+1]='\0';

    return arr;
}
