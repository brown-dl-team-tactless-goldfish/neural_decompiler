int mostFrequent(int* nums, int numsSize, int key){
    char* count = calloc( 1001 , sizeof(char) );
    int s_idx = 0;
    int max = 0;
    int ans;
    for(int i = 0; i < numsSize-1; i++){
        if(nums[i] == key){
            count[nums[i+1]]++;
            if(count[nums[i+1]] > max){
                ans = nums[i+1];
                max = count[nums[i+1]];
            }
        }
    }

    return ans;
}
