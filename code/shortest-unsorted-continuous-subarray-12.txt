class Solution {
public:
    int findUnsortedSubarray(vector<int>& nums) 
    {
        vector<int> arr(nums.begin(), nums.end());
        
        sort(arr.begin(), arr.end());
        
        int n = nums.size(), s= 0, e = n-1;
        
        //compare these sorted and unsorted array to get sorted left and right subaray
        while(s<n && nums[s]==arr[s]) s++;
        while(e>s && nums[e]==arr[e]) e--;
        
        return e-s+1;
    }
};
