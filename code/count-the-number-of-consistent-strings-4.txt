public int CountConsistentStrings(string allowed, string[] words) {
        
        HashSet<char> hs = new HashSet<char>(allowed.ToCharArray());

        int count = 0;

        for (int i=0; i<words.Length; i++)
        {
            bool found = true;

            for (int j=0; j<words[i].Length; j++)
            {
                if (!hs.Contains(words[i][j]))
                {
                    found = false;
                }
            }

            if (found)
            {
                count++;
            }
        }

        return count;
    }
