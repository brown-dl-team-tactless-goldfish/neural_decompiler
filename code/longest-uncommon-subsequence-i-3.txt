#define MAX(a,b)  (a>b?a:b)

int findLUSlength(char * a, char * b)
{   
    int len_a = strlen(a);
    int len_b = strlen(b);
    
    if(len_a==0 && len_b==0)
        return -1;
    else if(len_a==0)
        return len_b;
    else if(len_b==0)
        return len_a;
        
    
    return ( strcmp(a,b)==0 ? -1 : MAX(len_a, len_b) );
}
