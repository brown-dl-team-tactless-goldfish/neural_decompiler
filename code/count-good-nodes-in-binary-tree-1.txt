int checkNode(struct TreeNode * root, int maxNode)
{
    if(!root) return 0;
    if(root->val >= maxNode)
        return 1 + checkNode(root->left,root->val) + checkNode(root->right,root->val);
    else 
        return checkNode(root->left, maxNode) + checkNode(root->right, maxNode);
}

int goodNodes(struct TreeNode* root){
    return checkNode(root,root->val);
}
