class Solution {
public:
    vector<int> arrayChange(vector<int>& nums, vector<vector<int>>& operations) {
        int n = nums.size();
        unordered_map<int,int> hm;
		//Mapping the element with its index
        for(int i=0;i<n;i++){
                hm[nums[i]] = i;
        }
        
        int toreplace,replacer;
        for(auto it:operations){
		
		//While Replacing, we just store the previous elements index at the current one and delete the previous element
            toreplace = it[0];
            replacer = it[1];
            hm[replacer] = hm[toreplace];
            hm.erase(toreplace);
        }
        vector<int> ans(n,0);
		
		//Iterate through our hashmap and store element at the required index
        for(auto it:hm){
            int index = it.second;
            ans[index] = it.first;
        }
        return ans;
    }
};
