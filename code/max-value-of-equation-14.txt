   //1- yi + yj + |xi - xj|
   //2- |xi - xj|= -(-xi -xj) && xj>xi
   //3- yi-xi + yj+xj
    
    for(int i=0;i<v.size();i++){
        
        while(!p.empty()&& (v[i][0]-p.top().second)>k)
            p.pop();
        
        if(!p.empty())
            ans=max(ans,p.top().first+v[i][0]+v[i][1]);

        p.push({v[i][1]-v[i][0],v[i][0]});
    }
    return ans;
}
