int smallestEqual(int* nums, int numsSize)
{
    int min = 100;
    for ( int i = 0 ; i < numsSize ; i++)
    {
        if ( (i % 10) == nums[i] )
        {
            if ( i < min )
            {
                min = i;
            }
        }
    }
    
    
  
    return ( min == 100 ? -1 : min);

}
