class Solution {
public:
    vector<int> longestObstacleCourseAtEachPosition(vector<int>& obstacles) {
        int n = obstacles.size();
        vector<int> lis(n, INT_MAX);
        vector<int> ans(n, 1);
        int j;
        for (int i = 0; i < n; ++i){
            j = upper_bound(lis.begin(), lis.end(), obstacles[i]) - lis.begin();
            ans[i] = j + 1;
            lis[j] = obstacles[i];
        }
        return ans;
    }
};
