class Solution {
public: 
    int dp[2000006]= {};
    int minGroups(vector<vector<int>>& intervals) {
        int mx = -1
        for(int i=0;i<intervals.size();i++) {
            dp[intervals[i][0]]++;
            dp[intervals[i][1]+1]--;
            mx = max(mx,intervals[i][1]);
        }
        int ans= INT_MIN;
        for(int i=1;i<=mx+1;i++) {
            dp[i]+=dp[i-1];
            ans = max(ans,dp[i]);
        }
        return ans;
    }
};
