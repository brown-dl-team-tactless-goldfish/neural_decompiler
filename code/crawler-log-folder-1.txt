int minOperations(char ** logs, int logsSize){
    int operations = 0;
    
    for(int i = 0; i < logsSize; i++) {
        // ../ operation - Ascend to parent directory. Reset operations to zero if negative
        if(strcmp(logs[i], "../") == 0) 
            operations = fmax(0, --operations);     
        // ./ operation - Stays in same directory
        else if(strcmp(logs[i], "./") == 0) 
            ;                                       
        // x/ operation - Descend to given logs[i] directory
        else             
            operations++;                           
    }
    return operations < 0 ? 0 : operations;
}
