// 1. Count number  of 0 operations.
// 2. Count max number of div (op 1) that can be used for all numbers
// 3. Result is the sum of (1) and (2)
int minOperations(vector<int>& nums) {
	int op0 = 0, op1 = 0;
	for (int i = 0; i < nums.size(); i++) {
		int curOp1 = 0;
		while (nums[i] > 0) {
			if (nums[i] & 1) {
				nums[i] -= 1;
				op0++;
			}
			else {
				nums[i] >>= 1;
				curOp1++;
			}
		}
		op1 = max(op1, curOp1);
	}
	return op0 + op1;
}
