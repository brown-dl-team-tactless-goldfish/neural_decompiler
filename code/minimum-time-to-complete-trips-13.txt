typedef long long ll;
bool isPossible(ll m,vector<int>& arr,int t){
    ll c = 0;
    for(int& x: arr){
        c += m/x;
    }
    return c < t?0:1;
}
long long minimumTime(vector<int>& time, int t) {

    ll l = 1,h = 1e14,ans=0;
    while(l <= h){
        ll m = l+(h-l)/2;
		
        if(isPossible(m,time,t)){
            ans = m;
            h = m-1;
        }
        else l = m+1;
    }
	
    return ans;
}
