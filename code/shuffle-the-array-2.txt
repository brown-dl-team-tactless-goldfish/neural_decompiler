int* shuffle(int* nums, int numsSize, int n, int* returnSize){
    *returnSize = numsSize;
    int i, j=0;
    int *shufflenums;
    shufflenums = (int*)malloc(sizeof(int)*numsSize);
    for(i=0;i<n;i++){
        shufflenums[j++] = nums[i];
        shufflenums[j++] = nums[i+n];
    }
    return shufflenums;
}
