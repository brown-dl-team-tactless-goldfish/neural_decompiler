class Solution {
public:
    using ll = long long;
    int minimumPartition(string s, int k) {
        int cnt = 0;
        // string t;
        ll p = 0;

        for(auto c : s) {
            ll t = p * 10 + c - '0';
            if(t <= k) {
                p = t;
            }
            else{
                cnt++;
                p = c - '0';
                t = 0;

                if(p > k) return -1;

            }
        }

        return cnt + 1;
    }
};
