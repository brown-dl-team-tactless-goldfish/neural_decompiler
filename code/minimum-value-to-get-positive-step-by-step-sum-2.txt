int minStartValue(int* nums, int numsSize){
    int dp[numsSize];
    memset(dp, 0, sizeof(dp));
    dp[0] = nums[0];
    int min_val = dp[0];
    
    for (int i=1; i<numsSize; i++) {
        dp[i] = dp[i-1] + nums[i];
        min_val = fmin(min_val, dp[i]);
    }
    if (min_val >= 1) {
        return 1;
    }
    return abs(min_val) + 1;
}
