class Solution {
public:
    bool checkPowersOfThree(int n) {
        set<int> ss ;
        while( n > 1 ){
            if(n == 2) return false ;
            int power = 0 , number = 1 ; 
            while(number <= n) ++power , number *= 3 ;
            number /= 3 ; --power ;
            if(ss.find(power) != end(ss)) return false ;
            ss.insert(power); n -= number ;
        }
        return true ;
    }
};
