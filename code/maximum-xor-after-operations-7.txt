class Solution {
public:
    int maximumXOR(vector<int>& v) {
        vector<bool> cnt(31);
        for(int x:v){
            int i=0;
            for(int i=0;x;i++){
                if(x&1)cnt[i]=true;
                x>>=1;
            }
        }
        int ans=0;
        for(int i=0;i<31;i++){
            if(cnt[i])
                ans+=(1<<i);
        }
        return ans;
    }
};
